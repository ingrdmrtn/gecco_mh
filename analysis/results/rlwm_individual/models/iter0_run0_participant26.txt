def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited Working Memory (WM) mixture.
    - Policy: convex combination of RL softmax policy and WM softmax policy.
    - WM is a one-shot cache that stores rewarded action per state, with decay.
    - WM contribution is down-weighted as set size increases (capacity effect).

    Parameters (tuple):
    - lr: learning rate for RL and WM storage after reward (0..1).
    - wm_weight: baseline mixture weight of WM vs RL (0..1).
    - softmax_beta: inverse temperature for RL softmax (scaled up internally).
    - wm_decay: WM decay/leak toward uniform on each state visit (0..1).
    - capacity_slope: scales the decrease of WM weight with larger set sizes (>=0).
      Effective WM weight per block: wm_weight / (1 + capacity_slope*(nS-3)).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, capacity_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values and WM "cache" values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-adjusted WM weight for this block
        wm_weight_eff = wm_weight / (1.0 + capacity_slope * max(0, nS - 3))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy probability of chosen action (softmax with high beta)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for visited state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward: move toward one-hot of chosen action
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - lr) * w[s, :] + lr * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with dynamic arbitration by WM confidence and set size.
    - Policy: mixture weight is state- and trial-specific, depending on WM confidence.
    - WM confidence = max activation in WM for current state (higher => more WM reliance).
    - WM contribution also down-weighted by set size (capacity effect).

    Parameters (tuple):
    - lr: learning rate for RL and WM storage (0..1).
    - wm_weight: baseline WM weight (0..1), scaled by arbitration and capacity.
    - softmax_beta: inverse temperature for RL softmax (scaled up internally).
    - wm_decay: WM decay/leak toward uniform on each visit (0..1).
    - arbitration_slope: sensitivity of WM weight to WM confidence (>=0).
      Effective per-trial weight: wm_weight_eff * sigmoid(arbitration_slope*(conf - 1/nA)),
      where conf = max(W_s), and wm_weight_eff = wm_weight / (1 + c*(nS-3)) with c=arbitration_slope.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, arbitration_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM
    blocks_log_p = 0.0

    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-adjusted baseline WM weight uses arbitration_slope as capacity factor
        wm_weight_base = wm_weight / (1.0 + arbitration_slope * max(0, nS - 3))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration by WM confidence (max of WM state vector)
            conf = float(np.max(W_s))  # in [~0,1]
            wm_gate = sigmoid(arbitration_slope * (conf - (1.0 / nA)))
            wm_weight_eff = wm_weight_base * wm_gate

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward toward one-hot
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - lr) * w[s, :] + lr * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with forgetfulness + WM win-stay cache + lapse/noise.
    - Policy: mixture of RL and WM as in template, then blended with a uniform lapse.
    - RL includes forgetting toward uniform that increases with set size.
    - WM behaves like a win-stay cache: rewarded action set strongly, otherwise decay.

    Parameters (tuple):
    - lr: RL learning rate (also used as WM storage strength on reward) (0..1).
    - wm_weight: baseline WM weight (0..1), capacity-scaled by set size.
    - softmax_beta: inverse temperature for RL softmax (scaled up internally).
    - wm_decay: WM decay/leak toward uniform (0..1).
    - lapse: probability of set-sizeâ€“scaled undirected noise (0..1). Effective lapse:
      lapse_eff = min(0.99, lapse * (nS / 3)), increasing with set size.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size effects:
        # - WM weight decreases with set size
        wm_weight_eff = wm_weight / (1.0 + 0.5 * max(0, nS - 3))
        # - Lapse increases with set size
        lapse_eff = min(0.99, lapse * (nS / 3.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (win-stay-like via one-hot after reward, plus decay)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture of WM and RL
            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            # Add lapse/uniform noise
            p_total = (1.0 - lapse_eff) * p_mix + lapse_eff * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with standard delta
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            # RL forgetting toward uniform for the visited state, scaled by set size
            rl_forget = min(0.5, 0.05 * max(1, nS - 2))  # increases slightly with nS
            q[s, :] = (1.0 - rl_forget) * q[s, :] + rl_forget * (1.0 / nA)

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - lr) * w[s, :] + lr * onehot

        blocks_log_p += log_p

    return -blocks_log_p
