def cognitive_model1(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL+WM mixture with graded WM strength that declines with set size and a decaying WM store.

    Mechanism:
    - RL: tabular Q-learning with softmax action selection.
    - WM: probabilistic cache of state→action associations that decays toward uniform and is strengthened by rewarded outcomes.
    - Policy: mixture of WM and RL policies.
    - Set size effect: WM weight is attenuated as a power-law of set size (nS^{-ss_gamma}).

    Parameters
    - lr: RL learning rate (0..1).
    - wm_weight: Base mixture weight for WM (0..1) before set-size modulation.
    - softmax_beta: Inverse temperature for RL policy (scaled by 10 inside; higher = more deterministic).
    - wm_decay: Per-trial decay of WM toward uniform for the current state (0..1).
    - wm_encode: Strength of WM encoding on rewarded trials (0..1).
    - ss_gamma: Exponent controlling how WM weight declines with set size (>=0). Effective WM weight = wm_weight * nS^{-ss_gamma}.

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_encode, ss_gamma = parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic

    blocks_log_p = 0
    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size–modulated WM weight
        wm_weight_eff = wm_weight * (nS ** (-ss_gamma))
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action a
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy: softmax over WM weights W_s with high inverse temperature
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy
            p_total = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # First, passive decay toward uniform for current state
            w[s, :] = (1 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, encode the chosen action into WM by moving toward one-hot
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1 - wm_encode) * w[s, :] + wm_encode * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL + one-shot WM with set-size–dependent forgetting and simple WM weight scaling by set size.

    Mechanism:
    - RL: tabular Q-learning with softmax.
    - WM: one-shot cache of the last rewarded action for each state (if any).
           The cache decays toward uniform with a forgetting rate that increases with set size.
    - Policy: mixture of WM and RL policies.
    - Set size effects:
        • WM mixture weight scaled by 3/nS (fewer items -> stronger WM contribution).
        • Forgetting rate increases with set size via an exponential scaling.

    Parameters
    - lr: RL learning rate (0..1).
    - wm_weight: Base mixture weight for WM (0..1).
    - softmax_beta: Inverse temperature for RL policy (scaled by 10).
    - forget_rate: Base forgetting rate per trial (0..1).
    - ss_scale: Scales how strongly set size magnifies forgetting (>0). Higher -> stronger forgetting in larger sets.

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, forget_rate, ss_scale = parameters
    softmax_beta *= 10
    softmax_beta_wm = 50

    blocks_log_p = 0
    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # WM mixture weight diminishes with set size
        wm_weight_eff = wm_weight * (3.0 / nS)
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        # Set-size–dependent forgetting rate (higher nS => larger effective forgetting)
        # f_eff in [0,1): monotone increasing with nS
        f_eff = 1.0 - (1.0 - forget_rate) ** (max(nS, 1) / max(ss_scale, 1e-6))
        f_eff = np.clip(f_eff, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay WM toward uniform for current state
            w[s, :] = (1.0 - f_eff) * w[s, :] + f_eff * w_0[s, :]

            # One-shot encoding if rewarded: set WM to one-hot on chosen action
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL with asymmetric learning rates + capacity-limited WM with lapse.

    Mechanism:
    - RL: tabular Q-learning with separate learning rates for positive and negative prediction errors.
    - WM: stores the rewarded action for up to K states within a block (capacity limit).
           If a state is stored, WM policy is near-deterministic; otherwise WM is uniform.
    - Policy: mixture of WM and RL, with an additional lapse that chooses uniformly at random.
    - Set size effect: WM's effective use depends on capacity K relative to nS; when nS > K, many states lack WM entries.

    Parameters
    - lr_pos: RL learning rate for positive prediction errors (0..1).
    - wm_weight: WM mixture weight (0..1).
    - softmax_beta: Inverse temperature for RL policy (scaled by 10).
    - lr_neg: RL learning rate for negative prediction errors (0..1).
    - capacity_K: WM capacity within a block (0..nS); number of distinct states that can be held in WM.
    - epsilon: Lapse probability mixing in a uniform random choice (0..1).

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr_pos, wm_weight, softmax_beta, lr_neg, capacity_K, epsilon = parameters
    softmax_beta *= 10
    softmax_beta_wm = 50

    blocks_log_p = 0
    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Track which states are currently stored in WM (capacity-limited)
        stored_mask = np.zeros(nS, dtype=bool)
        current_capacity = 0
        K_eff = float(np.clip(capacity_K, 0.0, nS))

        log_p = 0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # If state is in WM, use the stored W_s; otherwise WM is uniform on this trial
            if stored_mask[s]:
                p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            else:
                # Uniform WM when not stored
                uniform = w_0[s, :]
                p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (uniform - uniform[a])))

            # Mixture with lapse
            p_mix = wm_weight * p_wm + (1 - wm_weight) * p_rl
            p_total = (1 - epsilon) * p_mix + epsilon * (1.0 / nA)
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            if pe >= 0:
                q[s][a] += lr_pos * pe
            else:
                q[s][a] += lr_neg * pe

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # If rewarded, attempt to store this state-action in WM subject to capacity
            if r > 0:
                if (not stored_mask[s]) and (current_capacity + 1 <= K_eff + 1e-9):
                    stored_mask[s] = True
                    current_capacity += 1
                if stored_mask[s]:
                    one_hot = np.zeros(nA)
                    one_hot[a] = 1.0
                    w[s, :] = one_hot
            else:
                # If not rewarded, do not encode; keep previous WM contents
                pass

        blocks_log_p += log_p

    return -blocks_log_p
