def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM capacity model with age- and set-size-dependent WM weighting and perseveration.
    
    Model summary:
    - Choice is a mixture of a slow RL system and a fast, capacity-limited WM system.
    - WM weight is reduced at larger set sizes and further reduced for older adults.
    - WM is a fast cache that stores the last rewarded action for each state, with decay.
    - RL learns via a single learning rate.
    - Perseveration (stickiness) biases choices toward the previous action in a block.
    
    Parameters (model_parameters):
    - lr: scalar in [0,1], learning rate for RL.
    - wm_weight_base: base mixture weight of WM channel (0..1), before adjustments.
    - softmax_beta: inverse temperature for RL softmax, scaled by 10 inside (higher = more deterministic).
    - wm_decay: decay rate of WM toward uniform (0..1).
    - capacity_sensitivity: slope controlling how WM weight drops as set size increases.
    - stickiness: perseveration bias added to action values for last chosen action in the block.
    
    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, capacity_sensitivity, stickiness = model_parameters
    # Constraints/transform hints (no imports): clip where needed
    lr = np.clip(lr, 0.0, 1.0)
    wm_weight_base = np.clip(wm_weight_base, 0.0, 1.0)
    wm_decay = np.clip(wm_decay, 0.0, 1.0)
    capacity_sensitivity = float(capacity_sensitivity)
    stickiness = float(stickiness)

    # Base inverse temperature scaled; small age effect on RL stochasticity (older -> slightly lower beta)
    softmax_beta *= 10.0
    age_group = 0 if age[0] <= 45 else 1
    softmax_beta *= (1.0 - 0.2 * age_group)  # older adults a bit more stochastic
    
    softmax_beta_wm = 50.0  # WM nearly deterministic when information is in cache

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute effective WM mixture weight for this block based on set size and age
        # WM weight drops with set size; older group gets an additional drop
        # f_set = 1/(1+exp(k*(nS-4))) gives high weight at 3 and low at 6, controlled by capacity_sensitivity
        f_set = 1.0 / (1.0 + np.exp(capacity_sensitivity * (nS - 4.0)))
        f_age = (1.0 - 0.4 * age_group)  # older group has reduced WM reliance
        wm_weight_eff = np.clip(wm_weight_base * f_set * f_age, 0.0, 1.0)

        log_p = 0.0
        prev_a = None
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Add perseveration bias to both channels if previous action exists
            if prev_a is not None:
                bias_vec = np.zeros(nA)
                bias_vec[prev_a] = stickiness
                Q_s = Q_s + bias_vec
                W_s = W_s + bias_vec

            # RL policy: probability of chosen action under softmax
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1e-12

            # WM policy: softmax with high beta over WM cache values
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1e-12

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update:
            # 1) decay current state's WM toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # 2) if rewarded, store chosen action deterministically in WM cache for this state
            if r > 0.5:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0  # overwrite as a one-hot memory

            prev_a = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with asymmetric RL learning rates and set-size-dependent WM interference.
    
    Model summary:
    - Two RL learning rates: lr_pos for rewarded updates, lr_neg for non-rewarded updates.
    - WM is a fast store of rewarded state-action pairs, with decay increased under higher set size.
    - WM mixture weight decreases with set size and more strongly for older adults.
    - Small, fixed age effect reduces RL learning rates and WM weight for older adults.
    
    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive prediction errors (0..1).
    - lr_neg: RL learning rate for negative prediction errors (0..1).
    - wm_weight_base: base WM mixture weight (0..1).
    - softmax_beta: RL inverse temperature, scaled by 10 inside.
    - wm_decay_base: base WM decay toward uniform (0..1).
    - interference_gain: how much WM decay increases from set size 3 to 6.
    
    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr_pos, lr_neg, wm_weight_base, softmax_beta, wm_decay_base, interference_gain = model_parameters
    lr_pos = np.clip(lr_pos, 0.0, 1.0)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)
    wm_weight_base = np.clip(wm_weight_base, 0.0, 1.0)
    wm_decay_base = np.clip(wm_decay_base, 0.0, 1.0)
    interference_gain = np.clip(interference_gain, 0.0, 1.0)  # positive increase in decay

    softmax_beta *= 10.0
    age_group = 0 if age[0] <= 45 else 1
    # Older adults: slightly lower beta and reduced learning rates
    softmax_beta *= (1.0 - 0.15 * age_group)
    lr_pos *= (1.0 - 0.2 * age_group)
    lr_neg *= (1.0 - 0.2 * age_group)

    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent WM weight and interference (decay)
        # Weight downscales with set size; older adults get an extra downscale
        f_set = 1.0 if nS <= 3 else 0.5
        wm_weight_eff = np.clip(wm_weight_base * f_set * (1.0 - 0.4 * age_group), 0.0, 1.0)
        # Interference increases decay from set size 3 to 6
        size_factor = 0.0 if nS <= 3 else 1.0
        wm_decay = np.clip(wm_decay_base + interference_gain * size_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1e-12

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1e-12

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - q[s, a]
            alpha = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += alpha * pe

            # WM update: decay toward uniform; on reward, overwrite with chosen action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLâ€“WM arbitration by confidence with lapse, modulated by age and set size.
    
    Model summary:
    - RL with a single learning rate.
    - WM cache with decay (stores rewarded actions), as in RLWM.
    - Arbitration: WM weight increases when WM policy is more confident (peaky) than RL,
      and decreases with set size and old age. Confidence difference is scaled by an arbitration slope.
    - Lapse (epsilon) mixes uniform random choice; lapse increases for larger set sizes and older adults.
    
    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_base: base WM mixture weight (0..1).
    - softmax_beta: RL inverse temperature scaled by 10.
    - wm_decay: WM decay toward uniform (0..1).
    - arb_slope: slope controlling sensitivity of mixture to confidence difference.
    - lapse: base lapse rate (0..1).
    
    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_base, softmax_beta, wm_decay, arb_slope, lapse = model_parameters
    lr = np.clip(lr, 0.0, 1.0)
    wm_base = np.clip(wm_base, 0.0, 1.0)
    wm_decay = np.clip(wm_decay, 0.0, 1.0)
    arb_slope = float(arb_slope)
    lapse = np.clip(lapse, 0.0, 1.0)

    softmax_beta *= 10.0
    age_group = 0 if age[0] <= 45 else 1
    softmax_beta *= (1.0 - 0.2 * age_group)  # older adults: more stochastic RL

    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size and age effects on base WM weight and lapse
        # WM base weight drops with larger set size; older adults rely less on WM
        f_set = 1.0 / (1.0 + np.exp(1.0 * (nS - 4.0)))  # mild drop from 3->6
        wm_base_eff = np.clip(wm_base * f_set * (1.0 - 0.4 * age_group), 0.0, 1.0)
        # Lapse increases with set size and age
        lapse_eff = np.clip(lapse + 0.1 * (nS - 3) + 0.1 * age_group, 0.0, 0.5)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax policy vector
            logits_rl = softmax_beta * Q_s
            logits_rl -= np.max(logits_rl)
            pvec_rl = np.exp(logits_rl)
            pvec_rl /= np.sum(pvec_rl)

            # WM softmax policy vector
            logits_wm = softmax_beta_wm * W_s
            logits_wm -= np.max(logits_wm)
            pvec_wm = np.exp(logits_wm)
            pvec_wm /= np.sum(pvec_wm)

            # Confidence as max probability (lower entropy => higher confidence)
            conf_rl = np.max(pvec_rl)
            conf_wm = np.max(pvec_wm)

            # Arbitration weight based on confidence difference
            # wm_weight_dyn in [0,1], centered around wm_base_eff
            conf_diff = conf_wm - conf_rl
            wm_weight_dyn = 1.0 / (1.0 + np.exp(-arb_slope * conf_diff))
            wm_weight_dyn = wm_weight_dyn * wm_base_eff  # cap by base weight after age/set-size effects
            wm_weight_dyn = np.clip(wm_weight_dyn, 0.0, 1.0)

            # Mixed policy with lapse
            pvec_mix = wm_weight_dyn * pvec_wm + (1.0 - wm_weight_dyn) * pvec_rl
            pvec_final = (1.0 - lapse_eff) * pvec_mix + lapse_eff * (1.0 / nA) * np.ones(nA)

            p_chosen = max(pvec_final[a], 1e-12)
            log_p += np.log(p_chosen)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM update: decay toward uniform; reward overwrites
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p
