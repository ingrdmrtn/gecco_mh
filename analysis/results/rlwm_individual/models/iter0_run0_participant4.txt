def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with capacity-limited WM, decay, and reward-gated WM writes.

    Policy:
    - RL system: softmax over Q-values with inverse temperature softmax_beta.
    - WM system: softmax over W-values with very high inverse temperature (deterministic).
    - Mixture: p = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl.

    Learning:
    - RL: Q-learning with learning rate lr.
    - WM: per-visit passive decay toward uniform (wm_decay), and one-shot reward-gated replacement:
           if reward=1, W[s] <- (1-wm_replace)*W[s] + wm_replace*one_hot(a).

    Set-size effect:
    - Effective WM weight decreases with set size via capacity_k:
         wm_weight_eff = wm_weight * min(1, capacity_k / nS).

    Parameters (6):
    - lr: RL learning rate in [0,1].
    - wm_weight: Base WM mixture weight in [0,1].
    - softmax_beta: RL softmax inverse temperature; internally scaled up by 10.
    - wm_decay: WM decay toward uniform (0=no decay, 1=full reset each visit).
    - wm_replace: Reward-gated WM write strength when r=1 (0-1).
    - capacity_k: WM capacity (approximate) controlling how WM weight scales with set size.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_replace, capacity_k = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # set-size dependent WM weight
        wm_capacity_factor = min(1.0, max(0.0, capacity_k / max(1.0, nS)))
        wm_weight_eff = np.clip(wm_weight * wm_capacity_factor, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action a
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Working-memory policy: near-deterministic softmax over W
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM reward-gated replacement (store correct action when rewarded)
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_replace) * w[s, :] + wm_replace * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with set-size-specific RL learning rates and WM reliance,
    and recency-based (reward-agnostic) WM writes with decay.

    Policy:
    - RL system: softmax over Q with softmax_beta.
    - WM system: softmax over W with high inverse temperature.
    - Mixture: set-size-specific WM weights (wm_weight_small for nS<=3, wm_weight_large for nS>3).

    Learning:
    - RL: state-size-specific learning rate: lr_small (nS<=3) vs lr_large (nS>3).
    - WM: per-visit decay toward uniform (wm_decay), and recency-based write each trial
          (stores the chosen action irrespective of reward).

    Set-size effects:
    - Directly parameterized through (lr_small, lr_large) and (wm_weight_small, wm_weight_large).

    Parameters (6):
    - lr_small: RL learning rate used when set size <= 3.
    - lr_large: RL learning rate used when set size > 3.
    - wm_weight_small: WM mixture weight when set size <= 3.
    - wm_weight_large: WM mixture weight when set size > 3.
    - softmax_beta: RL softmax inverse temperature; internally scaled by 10.
    - wm_decay: WM decay toward uniform (0=no decay, 1=full reset each visit).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_small, lr_large, wm_weight_small, wm_weight_large, softmax_beta, wm_decay = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # set-size dependent params
        small_set = nS <= 3
        lr_eff = lr_small if small_set else lr_large
        wm_weight_eff = np.clip(wm_weight_small if small_set else wm_weight_large, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr_eff * delta

            # WM update: decay + recency write of chosen action (reward-agnostic)
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            # write softly toward last chosen action
            write_rate = 1.0 - (1.0 - wm_decay)  # ensures use of wm_decay param; could be decoupled but we keep <= 6 params
            w[s, :] = (1.0 - write_rate) * w[s, :] + write_rate * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with uncertainty-based arbitration and set-size sensitivity.
    WM decays; WM writes are reward-gated.

    Policy:
    - RL policy: softmax over Q (softmax_beta).
    - WM policy: softmax over W (high beta).
    - Arbitration: dynamic WM weight from a logistic of (RL confidence - WM uncertainty)
      plus a set-size bias. Final mixture weight is base_wm_weight scaled by this gate.

      wm_gate = sigmoid(arbitration_gain * (conf_rl - uncert_wm) + setsize_bias)
      wm_weight_eff = clip(base_wm_weight * wm_gate, 0, 1)

      where:
        conf_rl = max(Q_s) - mean(Q_s)  (higher => more certain)
        uncert_wm = entropy(W_s)        (higher => more uncertain)
        setsize_bias = setsize_sensitivity * (3 - nS)/3  (favor WM in small set sizes)

    Learning:
    - RL: Q-learning with lr.
    - WM: decay toward uniform (wm_decay). Reward-gated write toward one_hot(a) when r=1.

    Parameters (6):
    - lr: RL learning rate in [0,1].
    - base_wm_weight: Base WM mixture weight multiplier in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_decay: WM decay toward uniform each visit (0-1).
    - arbitration_gain: Gain on confidence-vs-uncertainty arbitration signal.
    - setsize_sensitivity: Scales set-size bias term (positive -> more WM in small sets).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, base_wm_weight, softmax_beta, wm_decay, arbitration_gain, setsize_sensitivity = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    eps = 1e-12

    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # static set-size bias for the whole block (small sets bias toward WM)
        setsize_bias = setsize_sensitivity * (3 - nS) / 3.0

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM policies
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration signals
            conf_rl = np.max(Q_s) - np.mean(Q_s)  # higher = more certain RL
            # entropy of WM policy (natural log)
            W_safe = np.clip(W_s, eps, 1.0)
            uncert_wm = -np.sum(W_safe * np.log(W_safe))

            wm_gate = sigmoid(arbitration_gain * (conf_rl - uncert_wm) + setsize_bias)
            wm_weight_eff = np.clip(base_wm_weight * wm_gate, 0.0, 1.0)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM reward-gated write
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # write strength tied to arbitration gate (more WM reliance -> stronger write)
                write_rate = wm_gate
                w[s, :] = (1.0 - write_rate) * w[s, :] + write_rate * one_hot

        blocks_log_p += log_p

    return -blocks_log_p
