Here are three alternative RL+WM cognitive models tailored to your rlwm task. Each model is a standalone function that returns the negative log-likelihood of the observed actions, and each uses up to 6 parameters. All functions follow (and fill in) the provided template structure, implementing a working-memory policy and updates alongside a standard RL learner. I also specify how set size (3 vs 6) impacts parameters within each model.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with load-scaled WM weight and decay.
    - The RL system learns with a Rescorla-Wagner update.
    - The WM system stores the last rewarded action for each state (one-shot encoding), with decay toward uniform.
    - WM contribution to choice is scaled down by set size via an exponential load factor.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: Base mixture weight for WM (0..1), modulated by set size
    - softmax_beta: Inverse temperature for RL policy (scaled by 10 internally)
    - wm_decay: Working-memory decay toward uniform per trial (0..1)
    - load_sensitivity: Scales how strongly set size reduces effective WM weight; higher values reduce WM more at set size 6

    Set size effect:
    - Effective WM weight = wm_weight * exp(-load_sensitivity * (nS - 3)), so it is smaller for 6 than for 3.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight, softmax_beta, wm_decay, load_sensitivity = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Load-scaled mixture weight
            wm_weight_eff = wm_weight * np.exp(-load_sensitivity * (nS - 3))
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward baseline
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM one-shot encoding only on rewarded trials: move W_s toward one-hot on action a
            if r > 0.5:
                # move probability mass toward the rewarded action a
                # make W_s closer to one-hot without new parameter
                phi = 1.0  # maximal push on rewarded trials
                w[s, :] = (1.0 - phi) * w[s, :] + (phi) * w_0[s, :]
                # Then overwrite to be strongly peaked at a (approx one-hot but normalized)
                w[s, :] = (1.0 / nA) * np.ones(nA)
                w[s, a] = 1.0  # temporarily set to 1, then renormalize
                w[s, :] = np.clip(w[s, :], 0.0, None)
                total = np.sum(w[s, :])
                if total > 0:
                    w[s, :] /= total
                else:
                    w[s, :] = (1.0 / nA) * np.ones(nA)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with entropy-gated WM and load-dependent down-weighting.
    - RL: standard Rescorla-Wagner.
    - WM: last rewarded action with decay toward uniform.
    - The WM contribution is adaptively reduced when WM is uncertain (high entropy).
    - Additionally, WM weight is explicitly reduced as set size increases.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: Base WM mixture weight (0..1)
    - softmax_beta: Inverse temperature for RL policy (scaled by 10 internally)
    - wm_decay: WM decay toward uniform per trial (0..1)
    - entropy_slope: Controls how steeply entropy reduces WM weight (higher -> stronger gating)
    - load_sensitivity: Scales how strongly set size reduces WM weight (higher -> more reduction at set size 6)

    Set size effect:
    - Effective WM weight is multiplied by exp(-load_sensitivity * (nS - 3)), making it smaller in set size 6.
    - Entropy gating further reduces WM influence when WM distribution is diffuse.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight, softmax_beta, wm_decay, entropy_slope, load_sensitivity = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Entropy of current WM distribution for the state
            P_wm_full = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            P_wm_full = P_wm_full / np.sum(P_wm_full)
            H = -np.sum(P_wm_full * (np.log(P_wm_full + eps)))
            H_max = np.log(nA)
            # entropy gate: small when entropy is high, large when low
            # map H in [0,H_max] to gate in (0,1): gate = sigmoid(-entropy_slope * (H - H_max/2))
            gate = 1.0 / (1.0 + np.exp(entropy_slope * (H - 0.5 * H_max)))

            # Load-scaled WM weight with entropy gating
            wm_weight_eff = wm_weight * np.exp(-load_sensitivity * (nS - 3)) * gate
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward baseline
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM encoding on rewarded trials
            if r > 0.5:
                # Move distribution toward one-hot on action a
                w[s, :] = (1.0 / nA) * np.ones(nA)
                w[s, a] = 1.0
                w[s, :] = np.clip(w[s, :], 0.0, None)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with misbinding (swap) noise that increases with set size.
    - RL: standard Rescorla-Wagner.
    - WM: last rewarded action with decay toward uniform.
    - The WM retrieval can be corrupted by a 'swap' process: with some probability, the policy is influenced by other states' WM associations.
      This swap probability increases with set size, capturing interference/misbinding in larger sets.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: Base WM mixture weight (0..1)
    - softmax_beta: Inverse temperature for RL policy (scaled by 10 internally)
    - wm_decay: WM decay toward uniform per trial (0..1)
    - swap_k: Controls growth of swap probability with set size (higher -> more swaps at larger set size)
    - swap_bias: Baseline swap tendency even at small set size

    Set size effect:
    - Swap probability = sigmoid(swap_bias + swap_k * (nS - 3)), thus larger for set size 6 than 3.
      This reduces effective WM precision more at higher set size.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight, softmax_beta, wm_decay, swap_k, swap_bias = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute index list for "other states" usage
        all_states = np.arange(nS)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM base softmax for the current state
            W_s = w[s, :]
            P_wm_s = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            P_wm_s = P_wm_s / np.sum(P_wm_s)

            # Swap component: average WM distribution over other states (if any)
            other_idx = all_states[all_states != s]
            if other_idx.size > 0:
                W_other = w[other_idx, :]
                # Softmax each other's WM row, then average
                P_wm_others = np.exp(softmax_beta_wm * (W_other - np.max(W_other, axis=1, keepdims=True)))
                P_wm_others = P_wm_others / np.sum(P_wm_others, axis=1, keepdims=True)
                P_swap = np.mean(P_wm_others, axis=0)
            else:
                P_swap = (1.0 / nA) * np.ones(nA)

            # Swap probability increases with set size
            swap_prob = 1.0 / (1.0 + np.exp(-(swap_bias + swap_k * (nS - 3))))
            swap_prob = np.clip(swap_prob, 0.0, 1.0)

            # Effective WM choice distribution as mixture of correct-state WM and swapped WM
            P_wm_eff = (1.0 - swap_prob) * P_wm_s + swap_prob * P_swap
            p_wm = np.clip(P_wm_eff[a], eps, 1.0)

            # Mixture with RL
            wm_weight_eff = np.clip(wm_weight, 0.0, 1.0)
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward baseline
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM encoding on rewarded trials
            if r > 0.5:
                w[s, :] = (1.0 / nA) * np.ones(nA)
                w[s, a] = 1.0
                w[s, :] = np.clip(w[s, :], 0.0, None)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
