Below are three alternative cognitive models that follow the requested RL+WM template, use all parameters meaningfully, and incorporate set-size effects in distinct ways. Each function returns the negative log-likelihood of the observed choices.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with power-law set-size gating and time-based WM retention.
    - RL uses a single learning rate.
    - WM contribution is gated down by set size via a power-law factor nS^(-phi).
    - WM retention depends on time since last visit to the state via exponential retention lam^ISI.
    - WM encodes toward a one-hot on rewarded trials with strength eta_wm.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0-1)
    - wm_weight: Base weight on WM policy before set-size gating (0-1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - phi: Exponent for set-size gating; effective WM weight scales as nS^(-phi)
    - lam: WM retention factor per intervening trial for a given state (0-1); higher = slower decay
    - eta_wm: WM encoding strength toward one-hot when rewarded (0-1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, phi, lam, eta_wm = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Time-since-last-visit for retention
        last_visit = -np.ones(nS, dtype=int)

        # Power-law set-size gate
        gate = (nS ** (-phi)) if nS > 0 else 1.0
        wm_weight_eff = wm_weight * gate

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Apply WM decay based on time since last visit (if any)
            if last_visit[s] >= 0:
                isi = t - last_visit[s]
                decay = lam ** isi
                w[s, :] = decay * w[s, :] + (1 - decay) * w_0[s, :]
            last_visit[s] = t

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (as in template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over WM weights (deterministic with beta=50)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: encode rewarded action toward one-hot with strength eta_wm
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1 - eta_wm) * w[s, :] + eta_wm * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with capacity-limited inclusion, uniform lapses, and WM decay.
    - RL uses a single learning rate and softmax.
    - WM policy is included with probability proportional to capacity C relative to set size.
      Effective WM weight: wm_weight * min(1, C/nS).
    - Uniform lapse epsilon mixes in random responding across all actions.
    - WM decays toward uniform each trial and stores one-shot on reward.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - wm_weight: Base mixture weight for WM (0-1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - C: WM capacity in items; inclusion scales as min(1, C / nS)
    - wm_decay: Per-trial decay rate of WM toward uniform (0-1), higher = faster forgetting
    - epsilon: Lapse probability mixing in uniform random choice (0-1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, C, wm_decay, epsilon = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Capacity-limited inclusion
        inclusion = min(1.0, C / max(1.0, float(nS)))
        wm_weight_eff = wm_weight * inclusion

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with lapse
            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - epsilon) * p_mix + epsilon * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Rewarded one-shot storing
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with WM-confidence-gated RL learning and set-size sigmoid for WM.
    - RL learning rate is attenuated when WM is confident for the current state:
      alpha_eff = alpha_base * (1 - conf_wm), where conf_wm = max(W_s) - 1/nA.
    - WM mixture weight is reduced by a sigmoid of set size: gate = 1/(1+exp(gamma*(nS - theta))).
    - WM decays slowly and encodes both reward presence and action choice (reward-scaled).

    Parameters (model_parameters):
    - alpha_base: Base RL learning rate (0-1)
    - wm_weight: Max WM mixture weight (0-1) before set-size gating
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - theta: Set-size point where WM weight is half-max (items)
    - gamma: Slope of set-size sigmoid; higher gamma => sharper drop for nS > theta
    - wm_decay: Per-trial WM decay toward uniform (0-1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    alpha_base, wm_weight, softmax_beta, theta, gamma, wm_decay = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size sigmoid gate for WM weight
        gate = 1.0 / (1.0 + np.exp(gamma * (nS - theta)))
        wm_weight_eff = wm_weight * gate

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # Confidence of WM for state s
            conf_wm = np.max(W_s) - (1.0 / nA)
            conf_wm = np.clip(conf_wm, 0.0, 1.0)

            # RL update with WM-confidence-gated learning rate
            delta = r - Q_s[a]
            alpha_eff = alpha_base * (1.0 - conf_wm)
            q[s, a] += alpha_eff * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Reward-scaled WM encoding: move W_s toward one-hot of chosen action by amount r
            if r > 0.0:
                target = np.zeros(nA)
                target[a] = 1.0
                # Stronger consolidation when rewarded
                eta = r  # here r in {0,1}
                w[s, :] = (1 - eta) * w[s, :] + eta * target

        blocks_log_p += log_p

    return -blocks_log_p

Notes on set-size dependence across models:
- Model 1: WM weight scales as nS^(-phi); larger set size reduces WM influence smoothly. Additional time-based retention causes more forgetting when revisits are spaced.
- Model 2: WM influence is capacity-limited via min(1, C/nS); larger set size dilutes WM coverage. Lapse parameter captures additional noise under load.
- Model 3: WM mixture weight is reduced by a sigmoid of set size with parameters theta and gamma, providing a flexible boundary between low and high load. RL learning is down-weighted when WM is confident, allowing division of labor between systems.
