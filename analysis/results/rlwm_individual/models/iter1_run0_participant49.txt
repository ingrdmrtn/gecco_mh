def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with load-dependent WM weight (logit-linear in set size) and leaky, rewarded WM encoding.

    Idea:
    - Choices are a mixture of an RL policy and a WM policy.
    - The WM mixture weight decreases with higher load (set size) via a logit-linear transform.
    - WM is leaky toward a uniform baseline and encodes rewarded actions with strength eta.
    - RL updates with a standard delta rule.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in (0,1), base WM mixture weight at set size 3 (nS=3).
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - wm_load_slope: float, signed slope governing how WM weight changes with load.
        Effective logit(w) = logit(wm_weight) + wm_load_slope * ((nS - 3) / 3).
    - wm_decay: float in [0,1], leak of WM toward uniform each time a state is encountered.
    - wm_eta: float in [0,1], strength with which WM stores the rewarded action (one-hot push).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_load_slope, wm_decay, wm_eta = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    def logit(p):
        p = np.clip(p, 1e-6, 1 - 1e-6)
        return np.log(p / (1 - p))

    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        # Load-dependent WM weight
        load_term = (nS - 3) / 3.0
        wm_weight_eff_block = sigmoid(logit(wm_weight) + wm_load_slope * load_term)

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            p_total = wm_weight_eff_block * p_wm + (1.0 - wm_weight_eff_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward (one-hot push with strength wm_eta)
            if r > 0.0:
                w[s, :] = (1.0 - wm_eta) * w[s, :] + wm_eta * w_0[s, :]
                w[s, a] += wm_eta  # emphasize chosen action
                # renormalize to keep a probability-like vector
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with asymmetric RL learning rates and load-dependent perseveration bias.

    Idea:
    - Choices mix RL and WM policies.
    - RL uses separate learning rates for positive and negative outcomes.
    - Perseveration bias (stickiness) biases repeating the last chosen action for each state,
      and this bias increases with set size (higher load => more stickiness).
    - WM stores rewarded actions deterministically (no decay here), acting as a fast cache.

    Parameters (model_parameters):
    - lr_pos: float in [0,1], RL learning rate for positive outcomes (r=1).
    - lr_neg: float in [0,1], RL learning rate for negative outcomes (r=0).
    - wm_weight: float in [0,1], mixture weight for WM policy.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - stickiness: float, base perseveration bias strength (added to last-chosen action).
    - stickiness_load: float, scales how stickiness grows with load:
         phi_eff = stickiness * (1 + stickiness_load * (nS - 3)/3).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, stickiness, stickiness_load = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        phi_eff = stickiness * (1.0 + stickiness_load * ((nS - 3) / 3.0))

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Add perseveration bias to last chosen action for this state
            la = int(last_action[s])
            if la >= 0:
                Q_s[la] += phi_eff
                W_s[la] += phi_eff

            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            delta = r - q[s, a]
            lr_eff = lr_pos if delta >= 0.0 else lr_neg
            q[s, a] += lr_eff * delta

            # WM update: store rewarded action as a one-hot attractor (no decay here)
            if r > 0.0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0  # strong cache of correct action

            # Update last action
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with uncertainty-gated WM mixture and load penalty on WM, plus WM leak.

    Idea:
    - Choices mix RL and WM policies; the mixture weight is not fixed.
    - The effective WM weight increases when RL is uncertain (high entropy)
      and decreases under higher set size (load penalty).
    - WM decays toward uniform and stores rewarded actions (one-hot push).

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - base_wm_weight: float in (0,1), baseline WM mixture weight.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - gate_sensitivity: float, scales the contribution of RL entropy to the WM weight.
        Effective logit(w) = logit(base_wm_weight) + gate_sensitivity * H_norm - load_effect * ((nS-3)/3).
    - load_effect: float >= 0, direct penalty on WM weight with load.
    - wm_decay: float in [0,1], leak of WM toward uniform on each state visit.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, base_wm_weight, softmax_beta, gate_sensitivity, load_effect, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    def logit(p):
        p = np.clip(p, 1e-6, 1 - 1e-6)
        return np.log(p / (1 - p))

    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability for chosen action (for likelihood)
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # RL full policy to compute entropy (uncertainty)
            logits = softmax_beta * (Q_s - np.max(Q_s))
            exp_logits = np.exp(logits)
            p_rl_full = exp_logits / np.sum(exp_logits)
            H = -np.sum(p_rl_full * np.log(np.clip(p_rl_full, 1e-12, 1.0)))
            H_max = np.log(nA)
            H_norm = H / max(H_max, 1e-12)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Uncertainty-gated, load-penalized WM mixture weight
            load_term = (nS - 3) / 3.0
            w_logit = logit(base_wm_weight) + gate_sensitivity * H_norm - load_effect * load_term
            wm_weight_eff = sigmoid(w_logit)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward (one-hot push)
            if r > 0.0:
                w[s, :] = 0.0 * w[s, :] + w_0[s, :]
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p
