def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 1: RL + WM mixture with set-size-sensitive WM availability and leak
    - RL: tabular Q-learning with softmax policy (as provided in the template).
    - WM: associative store that becomes more confident after rewards, with load-dependent leak.
    - Arbitration: convex mixture, with WM weight reduced under larger set sizes and upweighted when the WM representation for the current state is sharp/confident.

    Parameters (model_parameters):
    - lr: float in [0,1], reinforcement learning rate for Q updates.
    - wm_weight_base: float in [0,1], baseline WM mixture weight (at set size 3, with flat WM).
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - leak_base: float in [0,1], baseline WM leak per trial toward uniform (at set size 3).
    - phi: float >= 0, exponent controlling how both WM weight and leak scale with set size: factor ~ (3/nS)^phi.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight_base, softmax_beta, leak_base, phi = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Confidence-weighted WM readout: sharper W_s yields higher effective WM weight.
            # Compute WM action probability via a high-temperature softmax of W_s.
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Set-size scaling for WM availability and leak
            scale = (3.0 / float(nS)) ** max(0.0, phi)
            wm_weight_eff = wm_weight_base * scale

            # Additional up-weighting by current WM confidence for this state (max probability mass)
            wm_conf = float(np.max(W_s))  # in [1/nA, 1]
            wm_weight_eff = np.clip(wm_weight_eff * wm_conf, 0.0, 1.0)

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Load-dependent leak toward uniform
            leak = np.clip(leak_base * (1.0/scale), 0.0, 1.0)  # larger nS -> larger leak
            w = (1.0 - leak) * w + leak * w_0

            # Reward-driven sharpening: if reward, push W_s to one-hot at chosen action.
            # If no reward, mildly suppress the chosen action's WM weight and renormalize.
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Penalize the chosen action in WM slightly, then renormalize
                decrement = 0.1 * scale  # less change when set size is large
                w[s, a] = max(0.0, w[s, a] - decrement)
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 2: RL + WM mixture with time-constant WM decay and interference scaling
    - RL: tabular Q-learning with softmax policy (template).
    - WM: slot-like store that decays toward uniform with a time constant (tau), accelerated by set size.
           Errors (r=0) actively reduce the weight on the chosen action.
    - Arbitration: mixture with WM weight decreasing with set size via a rational form.

    Parameters (model_parameters):
    - lr: float in [0,1], reinforcement learning rate for Q updates.
    - wm_weight_base: float in [0,1], baseline WM mixture weight.
    - softmax_beta: float >= 0, inverse temperature for RL (internally scaled by 10).
    - tau: float > 0, WM time constant; larger tau = slower decay.
    - rho: float >= 0, load/interference exponent; both WM decay and mixture weight depend on it.
    - wm_err: float in [0,1], amount by which an error reduces WM weight on the chosen action.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, tau, rho, wm_err = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Precompute load-scaled decay step from tau and rho
        # Effective step size per trial: d = 1 - exp(-1 / (tau * (nS/3)^rho))
        load_scale = (float(nS) / 3.0) ** max(0.0, rho)
        decay_step = 1.0 - np.exp(-1.0 / max(1e-6, (tau * load_scale)))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Set-size dependent mixture weight: decreases with nS via rho
            wm_weight_eff = wm_weight_base / (1.0 + (max(0, nS - 3)) * max(0.0, rho))

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Global decay toward uniform (time-constant, load-scaled)
            w = (1.0 - decay_step) * w + decay_step * w_0

            if r > 0.5:
                # Reward: confidently store the chosen action
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Error: actively reduce WM mass on chosen action and renormalize
                w[s, a] = max(0.0, w[s, a] - wm_err)
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 3: RL + WM mixture with delta-rule WM learning and set-size-linear arbitration
    - RL: tabular Q-learning with softmax policy (template).
    - WM: learned via delta rules that move the row w[s,:] toward a one-hot when rewarded and toward uniform when not.
           This allows graded WM strengths rather than hard overwrites.
    - Arbitration: WM weight declines approximately linearly with set size above 3.

    Parameters (model_parameters):
    - lr: float in [0,1], reinforcement learning rate for Q updates.
    - wm_weight_base: float in [0,1], baseline WM mixture weight at set size 3.
    - softmax_beta: float >= 0, inverse temperature for RL (internally scaled by 10).
    - eta_wm_pos: float in [0,1], WM learning rate on rewarded trials toward the chosen one-hot.
    - eta_wm_neg: float in [0,1], WM learning rate on unrewarded trials toward the uniform prior.
    - ss_slope: float >= 0, slope controlling decline of WM weight with set size: wm_weight/(1+ss_slope*(nS-3)).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, eta_wm_pos, eta_wm_neg, ss_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Linear-like decline of WM weight with set size above 3
            wm_weight_eff = wm_weight_base / (1.0 + max(0, nS - 3) * max(0.0, ss_slope))

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Delta-rule WM learning:
            # If rewarded: move w[s,:] toward one-hot at action a
            # If not: move w[s,:] toward uniform prior
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - eta_wm_pos) * w[s, :] + eta_wm_pos * target
            else:
                w[s, :] = (1.0 - eta_wm_neg) * w[s, :] + eta_wm_neg * w_0[s, :]

            # Ensure the row remains a valid probability distribution
            row_sum = np.sum(w[s, :])
            if row_sum <= 0:
                w[s, :] = w_0[s, :].copy()
            else:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p
