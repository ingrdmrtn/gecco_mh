def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with set-size–dependent WM decay (interference)

    Mechanism
    - RL: standard delta rule with a single learning rate.
    - WM: a fast cache that stores action preferences per state (w). WM traces decay
      toward uniform with a rate that increases with set size (interference). Upon a
      rewarded outcome, WM writes a one-hot trace for the chosen action.
    - Policy: a mixture of WM and RL softmax policies.
    - Set-size effect: the WM decay rate increases with set size, degrading WM utility
      in the 6-item condition.

    Parameters
    - lr: float in [0,1], RL learning rate (for both rewards and non-rewards).
    - wm_weight: float in [0,1], mixture weight of WM vs RL in the choice policy.
    - softmax_beta: float >= 0, base inverse temperature for RL (internally scaled by 10).
    - wm_decay_base: float >= 0, base WM decay rate per trial toward uniform.
    - wm_interf_size: float >= 0, multiplicative factor linking set size to extra WM decay.
      Effective decay per trial d = 1 - exp(-wm_decay_base * (1 + wm_interf_size*(set_size-3))).

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay_base, wm_interf_size = model_parameters

    softmax_beta *= 10.0  # keep as in template
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size–dependent WM decay parameter (toward uniform)
        decay_factor = 1.0 + wm_interf_size * max(0.0, float(nS) - 3.0)
        d = 1.0 - np.exp(-wm_decay_base * decay_factor)
        d = np.clip(d, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy for chosen action (per template form)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (softmax over WM preferences)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform, then write if rewarded
            # Decay current state's WM trace
            w[s, :] = (1.0 - d) * w[s, :] + d * w_0[s, :]
            # Reward-gated overwriting to one-hot for chosen action
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with set-size–scaled learning rate + WM gated by surprise with mild decay

    Mechanism
    - RL: delta rule with learning rate reduced by set size:
        alpha_rl = lr_base / (1 + phi_size_on_rl * (set_size - 3)).
      This captures slower RL in higher load.
    - WM: decays mildly toward uniform each trial (wm_decay). WM writing is gated by
      surprise magnitude |r - Q(s,a)| via a sigmoid; higher surprise leads to stronger
      WM encoding on rewarded trials.
    - Policy: mixture of WM and RL softmax policies.

    Parameters
    - lr_base: float in [0,1], base RL learning rate.
    - wm_weight: float in [0,1], mixture weight of WM vs RL in the choice policy.
    - softmax_beta: float >= 0, base inverse temperature for RL (scaled by 10).
    - phi_size_on_rl: float >= 0, controls how much set size reduces RL learning rate.
    - gate_surprise: float >= 0, slope of the surprise-to-WM gate sigmoid.
    - wm_decay: float in [0,1], per-trial WM decay toward uniform for the current state.

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr_base, wm_weight, softmax_beta, phi_size_on_rl, gate_surprise, wm_decay = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size–scaled RL learning rate
        alpha_rl = lr_base / (1.0 + phi_size_on_rl * max(0.0, float(nS) - 3.0))
        alpha_rl = np.clip(alpha_rl, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy for chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with set-size scaled learning rate
            delta = r - Q_s[a]
            q[s, a] += alpha_rl * delta

            # WM update:
            # 1) mild decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # 2) surprise-gated write on rewarded trials
            # Surprise proxy: |delta|; gate via sigmoid centered at 0.5
            surpr = abs(delta)
            gate = 1.0 / (1.0 + np.exp(-gate_surprise * (surpr - 0.5)))
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - gate) * w[s, :] + gate * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with dynamic arbitration by WM confidence and set size

    Mechanism
    - RL: standard delta rule (single learning rate).
    - WM: stores rewarded actions as smoothed one-hot traces. Also applies a small
      penalty to the chosen action on non-reward to reduce false confidence.
    - Arbitration: the WM mixture weight is dynamic per trial based on:
        w_t = sigmoid(logit(wm_weight_base) - k_size*(set_size-3) - wm_noise*H(W_s)),
      where H(W_s) is the entropy of the WM policy proxy for the current state.
      Thus WM is down-weighted when set size is larger and when WM is uncertain.
    - Policy: p_total = w_t * p_wm + (1 - w_t) * p_rl.

    Parameters
    - lr: float in [0,1], RL learning rate.
    - wm_weight_base: float in [0,1], baseline WM mixture weight (before modulation).
    - softmax_beta: float >= 0, base inverse temperature for RL (scaled by 10).
    - k_size: float >= 0, strength of set size penalty on WM arbitration.
    - wm_noise: float >= 0, scales the entropy penalty on WM arbitration.
    - wm_smooth: float in [0,1], smoothing toward one-hot when WM writes on reward
                 and smoothing of demotion on non-reward.

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, k_size, wm_noise, wm_smooth = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Helper to compute logit safely
    def safe_logit(p):
        eps = 1e-6
        p = np.clip(p, eps, 1.0 - eps)
        return np.log(p) - np.log(1.0 - p)

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute constant size term
        size_term = k_size * max(0.0, float(nS) - 3.0)
        base_logit = safe_logit(wm_weight_base)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic-ish)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # WM confidence proxy: entropy of a moderate-temperature softmax over W_s
            # Use beta=5 to avoid trivial 0 entropy from very peaky distributions
            beta_conf = 5.0
            pw = np.exp(beta_conf * (W_s - np.max(W_s)))
            pw = pw / np.sum(pw)
            H = -np.sum(pw * np.log(np.clip(pw, 1e-12, 1.0)))

            # Dynamic arbitration weight
            w_logit = base_logit - size_term - wm_noise * H
            w_t = 1.0 / (1.0 + np.exp(-w_logit))
            w_t = np.clip(w_t, 0.0, 1.0)

            # Mixture
            p_total = w_t * p_wm + (1.0 - w_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # Reward: move toward one-hot for chosen action (smoothed by wm_smooth)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_smooth) * w[s, :] + wm_smooth * one_hot
            else:
                # Non-reward: gently demote the chosen action, redistribute mass
                demote = wm_smooth * 0.5
                w[s, a] = (1.0 - demote) * w[s, a]
                # Renormalize and mix a bit toward uniform to avoid collapse
                w[s, :] = w[s, :] / np.sum(w[s, :])
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p

Notes on set-size impacts:
- Model 1: WM decay d increases with set size via wm_interf_size, weakening WM in larger sets.
- Model 2: RL learning rate decreases with set size via phi_size_on_rl, slowing RL in larger sets.
- Model 3: Arbitration weight penalizes WM in larger sets via k_size, shifting control toward RL under higher load.
