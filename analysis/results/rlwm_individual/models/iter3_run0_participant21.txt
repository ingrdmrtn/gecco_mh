def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with load-dependent WM precision and decay.

    Overview:
    - RL: tabular Q-learning with a single learning rate and softmax policy (beta scaled by 10).
    - WM: fast delta-like update toward the rewarded action with decay toward uniform.
    - Policy: static mixture between RL and WM controlled by wm_weight (as in the template).
    - Set-size impact:
        * WM precision (effective inverse temperature) decreases with larger set sizes (noisier WM).
        * WM decay increases with larger set sizes (faster forgetting).

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], mixture weight of WM in the action policy.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - alpha_wm: float in [0,1], WM learning rate toward a one-hot code when rewarded.
    - wm_decay_base: float in [0,1], base WM decay toward uniform per trial (amplified by set size).
    - wm_noise_base: float >= 0, base WM noise level; effective WM beta scales as 1/(1 + wm_noise_base * load),
        where load = (nS-3)/3 (0 at set size 3; 1 at set size 6).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, alpha_wm, wm_decay_base, wm_noise_base = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load indices: 0 at set size 3, 1 at set size 6
        load = max(0.0, (float(nS) - 3.0) / 3.0)
        # Load-dependent WM decay (stronger decay at higher set size)
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) ** (1.0 + load)
        # Load-dependent WM precision scaling (lower precision at higher set size)
        wm_beta_scale = 1.0 / (1.0 + wm_noise_base * load)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy with effective inverse temperature reduced by load (noisier under higher load)
            W_s = w[s, :]
            # Scale logits by wm_beta_scale (equivalent to dividing temperature)
            W_centered = W_s - np.max(W_s)
            exp_W = np.exp((softmax_beta_wm * wm_beta_scale) * W_centered)
            p_wm = exp_W[a] / np.sum(exp_W)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(p_total, 1e-12))
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform, then reward-gated encoding toward the chosen action
            # Decay
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]
            # Reward-gated update: move probability mass toward chosen action when rewarded
            if r > 0.0:
                w[s, :] *= (1.0 - alpha_wm)
                w[s, a] += alpha_wm
                # Normalize to keep it a proper distribution (numerical safety)
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with asymmetric WM learning from positive vs negative feedback and load-dependent decay.

    Overview:
    - RL: tabular Q-learning with a single learning rate and softmax policy (beta scaled by 10).
    - WM: two learning rates:
        * Positive feedback (r=1): strengthen chosen action representation.
        * Negative feedback (r=0): actively suppress chosen action and redistribute probability to other actions.
      WM also decays toward uniform; decay increases with set size.
    - Policy: static mixture between RL and WM controlled by wm_weight.

    Set-size impact:
    - WM decay increases with set size via an exponential mapping of wm_decay_base.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], mixture weight of WM in the action policy.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - alpha_wm_pos: float in [0,1], WM learning rate when r=1 (toward chosen action).
    - alpha_wm_neg: float in [0,1], WM negative-feedback suppression applied to the chosen action when r=0.
    - wm_decay_base: float in [0,1], base WM decay rate toward uniform per trial (amplified by set size).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, alpha_wm_pos, alpha_wm_neg, wm_decay_base = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent WM decay: higher at larger set size
        load = max(0.0, (float(nS) - 3.0) / 3.0)
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) ** (1.0 + load)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Deterministic WM softmax from current WM weights
            W_s = w[s, :]
            W_centered = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * W_centered)
            p_wm = exp_W[a] / np.sum(exp_W)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(p_total, 1e-12))
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            if r > 0.0:
                # Positive feedback: pull mass toward chosen action
                w[s, :] *= (1.0 - alpha_wm_pos)
                w[s, a] += alpha_wm_pos
            else:
                # Negative feedback: push mass away from chosen action
                # Remove probability from chosen action and redistribute uniformly to others
                take = alpha_wm_neg * w[s, a]
                w[s, a] -= take
                w[s, :] += take / (nA - 1)  # equally to all actions, including chosen; then fix:
                w[s, a] -= take / (nA - 1)  # subtract back chosen portion to net-redistribute to others

            # Normalize for numerical stability
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with probabilistic recall policy and load-dependent recall plus WM decay.

    Overview:
    - RL: tabular Q-learning with a single learning rate and softmax policy (beta scaled by 10).
    - WM policy: mixture of a recall-driven delta (choosing the WM argmax) and a uniform guess.
        Recall probability depends on:
          * A base recall rate modulated by set size (worse recall at larger set size).
          * The confidence in WM (difference between top two WM action weights), scaled by rec_sensitivity.
      When recall occurs, action probabilities are a near-delta on the WM argmax via deterministic softmax.
    - WM update: decay toward uniform and reward-gated consolidation toward chosen action.

    Set-size impact:
    - Recall base decreases with set size.
    - WM decay increases with set size.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], mixture weight of WM in the action policy.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - rec_base: float in [0,1], base recall probability at set size 3.
    - rec_sensitivity: float >= 0, scales the effect of WM confidence (top-vs-next difference) on recall probability.
    - wm_decay_base: float in [0,1], base WM decay toward uniform per trial (amplified by set size).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, rec_base, rec_sensitivity, wm_decay_base = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        load = max(0.0, (float(nS) - 3.0) / 3.0)
        # Recall base declines with load
        rec_base_load = np.clip(rec_base * (1.0 - load), 0.0, 1.0)
        # WM decay increases with load
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) ** (1.0 + load)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Compute WM argmax and a confidence index (top vs second-best difference)
            W_s = w[s, :]
            # Confidence in WM representation
            sorted_W = np.sort(W_s)[::-1]
            top = sorted_W[0]
            second = sorted_W[1] if nA > 1 else 0.0
            conf = max(0.0, top - second)
            # Recall probability combines base (load-dependent) and confidence
            p_recall = np.clip(rec_base_load + rec_sensitivity * conf, 0.0, 1.0)

            # If recall: near-delta on WM argmax via deterministic softmax; else: uniform guess
            argmax_w = int(np.argmax(W_s))
            if argmax_w == a:
                p_wm_recall = 1.0  # under the deterministic WM softmax, argmax action gets prob ~1
            else:
                p_wm_recall = 0.0
            p_wm_guess = 1.0 / nA

            p_wm = p_recall * p_wm_recall + (1.0 - p_recall) * p_wm_guess

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(p_total, 1e-12))
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # Reward-gated consolidation toward chosen action
            if r > 0.0:
                # Move mass toward chosen action proportionally to its complement
                alpha = min(1.0, 0.5 + 0.5 * conf)  # stronger consolidation when confident
                w[s, :] *= (1.0 - alpha)
                w[s, a] += alpha
                # Normalize
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p
