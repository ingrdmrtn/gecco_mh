def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with load-dependent lapse.

    Idea:
    - RL learns state-action values with a single learning rate.
    - WM stores rewarded actions for each state and decays toward uniform.
    - A load-dependent lapse (epsilon) blends some uniform random choice into the final policy,
      increasing with set size (higher cognitive load => more lapses).

    Parameters (model_parameters):
    - lr:            RL learning rate (0..1)
    - wm_weight:     Weight of WM in the mixture (0..1), before lapse
    - softmax_beta:  Base RL inverse temperature (scaled internally by *10)
    - wm_decay:      WM decay toward uniform on each visit (0..1)
    - lapse_base:    Base lapse probability at set size 3 (0..1)
    - lapse_load:    Additional lapse per unit increase in set size from 3 to 6 (>=0). 
                     Effective epsilon: lapse_base + lapse_load * ((nS - 3) / 3).

    Set-size impact:
    - Lapse increases with set size: more random choices in larger set (6).
    """
    lr, wm_weight, softmax_beta, wm_decay, lapse_base, lapse_load = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent lapse
        epsilon = lapse_base + lapse_load * (max(0, nS - 3) / 3.0)
        epsilon = max(0.0, min(0.5, float(epsilon)))  # cap to keep meaningful mix

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax probability of chosen action (very deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture of WM and RL, then add load-dependent lapse to uniform
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - epsilon) * p_mix + epsilon * (1.0 / nA)
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM write on positive feedback (rewarded action stored)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with surprise-gated WM encoding and leak.

    Idea:
    - RL learns with a single learning rate.
    - WM maintains a decaying memory for each state and encodes a rewarded action only if
      the unsigned prediction error is large enough (surprise gate).
    - The gate uses a logistic function of |delta| with slope gate_k and threshold gate_theta.
      Higher set size raises the effective threshold (harder to commit to WM under load).

    Parameters (model_parameters):
    - lr:            RL learning rate (0..1)
    - wm_weight:     Weight of WM in the mixture (0..1)
    - softmax_beta:  Base RL inverse temperature (scaled internally by *10)
    - gate_k:        Slope of logistic gating on |delta| (positive)
    - gate_theta:    Threshold of logistic gating on |delta|
    - wm_leak:       WM leak toward uniform each visit (0..1)

    Set-size impact:
    - Effective gate threshold increases with set size: theta_eff = gate_theta + 0.5*(nS-3),
      making WM updates less likely in larger sets (higher cognitive load).
    """
    lr, wm_weight, softmax_beta, gate_k, gate_theta, wm_leak = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        theta_eff = gate_theta + 0.5 * max(0, nS - 3)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice prob
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM choice prob
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # Surprise-gated WM write on reward
            # Gate probability as logistic function of |delta| relative to theta_eff
            gate_input = gate_k * (abs(delta) - theta_eff)
            gate_prob = 1.0 / (1.0 + np.exp(-gate_input))

            if (r > 0.5) and (np.random.rand() < gate_prob):
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM + WSLS tri-policy mixture with load effects on RL temperature and WM interference.

    Idea:
    - RL learns with a single learning rate; its inverse temperature is reduced under higher load.
    - WM stores rewarded actions but suffers load-dependent interference (more decay under load).
    - A WSLS (win-stay/lose-shift) heuristic contributes to choices with its own weight.
      WSLS determinism is controlled by wsls_gain.

    Parameters (model_parameters):
    - lr:            RL learning rate (0..1)
    - wm_weight:     Weight of WM in the mixture before adding WSLS (0..1)
    - softmax_beta:  Base RL inverse temperature (scaled internally by *10, then reduced by load)
    - wsls_gain:     Controls WSLS determinism; higher -> more deterministic WSLS (epsilon = 1/(1+exp(wsls_gain)))
    - wm_interf:     WM interference strength; decay toward uniform increases with set size
    - rl_load:       Load sensitivity for RL temperature; beta_eff = beta / (1 + rl_load*(nS-3))

    Set-size impact:
    - RL inverse temperature decreases with set size (noisier RL) via rl_load.
    - WM decay toward uniform increases with set size via wm_interf.
    """
    lr, wm_weight, softmax_beta, wsls_gain, wm_interf, rl_load = model_parameters

    base_beta = softmax_beta * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Load-adjusted RL temperature
        beta_eff = base_beta / (1.0 + max(0.0, float(rl_load) * float(nS - 3)))

        # Load-dependent WM decay (bounded in 0..1 via exponential map)
        wm_decay = 1.0 - np.exp(-max(0.0, float(wm_interf)) * (float(nS) / 3.0))
        wm_decay = max(0.0, min(1.0, wm_decay))

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # For WSLS we need previous action and reward per state
        last_action = -1 * np.ones(nS, dtype=int)
        last_reward = np.zeros(nS)

        # Convert wsls_gain to an epsilon in (0, 0.5]; larger gain => smaller epsilon
        eps_wsls = 1.0 / (1.0 + np.exp(float(wsls_gain)))
        eps_wsls = max(0.0, min(0.5, eps_wsls))

        # WSLS mixture weight derived from wm_weight to keep all weights in [0,1]
        # Use wsls_weight = min(0.5, 0.5*(1 - wm_weight)) so RL retains some weight
        wsls_weight = 0.5 * (1.0 - wm_weight)
        wsls_weight = max(0.0, min(0.5, wsls_weight))
        rl_weight = 1.0 - wm_weight - wsls_weight
        rl_weight = max(0.0, rl_weight)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(beta_eff * (Q_s - Q_s[a])))

            # WM probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WSLS policy probability of chosen action
            if last_action[s] >= 0:
                la = int(last_action[s])
                lrwd = last_reward[s]
                p_wsls_vec = np.ones(nA) * (eps_wsls / 2.0)
                if lrwd > 0.5:
                    # Win-stay: put mass on last action
                    p_wsls_vec = np.ones(nA) * (eps_wsls / 2.0)
                    p_wsls_vec[la] = 1.0 - eps_wsls
                else:
                    # Lose-shift: reduce prob of last action, boost the other two
                    p_wsls_vec = np.ones(nA) * ((1.0 - eps_wsls) / 2.0)
                    p_wsls_vec[la] = eps_wsls / 2.0
                p_wsls = p_wsls_vec[a]
            else:
                p_wsls = 1.0 / nA

            # Tri-policy mixture
            p_total = wm_weight * p_wm + rl_weight * p_rl + wsls_weight * p_wsls
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay/interference toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM write on reward
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

            # Update WSLS memory
            last_action[s] = a
            last_reward[s] = r

        blocks_log_p += log_p

    return -blocks_log_p
