def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited, decaying Working Memory (WM).

    Idea:
    - Choices are a mixture of an incremental RL policy and a WM policy.
    - WM has a limited capacity (wm_capacity): its influence scales as wm_weight_base * min(1, wm_capacity / nS).
      Hence, WM is stronger when set size (nS) is small and weaker when nS is large.
    - WM representations decay toward a uniform baseline at rate wm_decay whenever a state is encountered.
    - WM encodes the rewarded response in a one-shot manner.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate (applies to signed prediction error).
    - wm_weight_base: float in [0,1], baseline mixture weight of WM.
    - softmax_beta: float >= 0, base inverse temperature for RL (internally scaled by 10).
    - wm_capacity: float >= 0, effective WM capacity in number of items; controls how WM weight drops with set size.
    - wm_decay: float in [0,1], decay rate of the WM map per state encounter toward a uniform baseline.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_capacity, wm_decay = model_parameters
    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent WM weight via capacity limit
        wm_weight_eff_block = wm_weight_base * min(1.0, float(wm_capacity) / max(1.0, float(nS)))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # Working Memory policy: softmax over WM weights (nearly deterministic)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm

            # Mixture of WM and RL with load-dependent WM weight
            p_total = p_wm*wm_weight_eff_block + (1-wm_weight_eff_block)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decay toward uniform baseline upon each encounter
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM encoding: if rewarded, form a one-shot memory for chosen action
            if r > 0.0:
                # Encode chosen action strongly; distribute the rest uniformly
                enc = 1.0  # maximal one-shot binding upon reward
                w[s, :] = (1.0 - enc) * w_0[s, :]
                w[s, a] += enc

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with uncertainty-weighted mixture and load-scaled WM precision.

    Idea:
    - Choices are a mixture of RL and WM policies.
    - The WM inverse temperature (precision) decreases with set size (resource-sharing):
      beta_wm_eff = softmax_beta_wm * wm_precision / (wm_precision + nS - 1).
    - The WM mixture weight increases with RL uncertainty (entropy of RL policy) and decreases with load:
      wm_weight_eff = sigmoid(wm_weight_base + unc_temp*(H_norm - 0.5)) * (3/nS),
      where H_norm is the entropy of the RL softmax policy normalized to [0,1].
    - WM decays slightly toward uniform each state encounter; rewarded trials produce one-shot encoding.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight_base: float, baseline bias toward WM in the mixture before uncertainty/load modulation.
    - softmax_beta: float >= 0, base inverse temperature for RL (internally scaled by 10).
    - wm_precision: float >= 0, controls WM precision and encoding strength; higher = more precise and more stable.
    - unc_temp: float, sensitivity of the WM mixture weight to RL uncertainty (entropy).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_precision, unc_temp = model_parameters
    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # WM resource-sharing: precision scales with set size
        beta_wm_eff = softmax_beta_wm * (wm_precision / (wm_precision + max(1.0, nS - 1.0)))

        # Small WM leak derived from wm_precision (higher precision -> smaller leak)
        wm_leak = 1.0 / (1.0 + max(1e-6, wm_precision))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]

            # RL policy for chosen action probability
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # RL full-policy to estimate uncertainty (entropy)
            z = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            pi = z / np.sum(z)
            H = -np.sum(pi * (np.log(pi + 1e-12)))  # entropy
            H_norm = H / np.log(nA)  # normalize to [0,1]

            # WM policy with load-scaled precision
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm

            # Uncertainty-weighted, load-scaled WM mixture weight
            x = wm_weight_base + unc_temp * (H_norm - 0.5)
            wm_weight_eff = 1.0 / (1.0 + np.exp(-x))  # sigmoid
            wm_weight_eff *= (3.0 / max(1.0, float(nS)))  # reduce with load

            p_total = wm_weight_eff*p_wm + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # WM encoding upon reward; encoding strength grows with wm_precision and is damped by load
            if r > 0.0:
                enc = wm_precision / (wm_precision + float(nS))
                enc = np.clip(enc, 0.0, 1.0)
                w[s, :] = (1.0 - enc) * w_0[s, :]
                w[s, a] += enc

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning + WM with signed one-shot updates (reinforce vs. avoid) and load-scaled mixture.

    Idea:
    - RL uses separate learning rates for positive and negative prediction errors.
    - WM contributes via a mixture whose weight decreases with load: wm_weight_eff = wm_weight * (3/nS).
    - WM updates are signed:
        - On reward: push probability mass toward the chosen action by eta_pos (one-shot strengthening).
        - On no-reward: push probability mass away from the chosen action by eta_neg (one-shot avoidance),
          redistributing mass to the unchosen actions.
    - This yields a fast-acting WM that can both reinforce correct mappings and suppress recently incorrect ones.

    Parameters (model_parameters):
    - lr_pos: float in [0,1], RL learning rate for positive prediction errors.
    - lr_neg: float in [0,1], RL learning rate for negative prediction errors.
    - wm_weight: float in [0,1], baseline WM mixture weight (scaled down by load).
    - softmax_beta: float >= 0, base inverse temperature for RL (internally scaled by 10).
    - eta_pos: float in [0,1], WM one-shot strengthening magnitude on rewarded trials.
    - eta_neg: float in [0,1], WM one-shot avoidance magnitude on unrewarded trials.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, eta_pos, eta_neg = model_parameters
    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: softmax over WM with high precision
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm

            # Load-scaled WM weight
            wm_weight_eff = wm_weight * (3.0 / max(1.0, float(nS)))

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update with asymmetric learning rates
            delta = r-Q_s[a]
            alpha = lr_pos if delta >= 0.0 else lr_neg
            q[s][a] += alpha*delta

            # WM signed one-shot updates
            if r > 0.0:
                # Move mass toward chosen action by eta_pos
                gain = eta_pos
                # Pull current distribution toward uniform, then add gain to chosen action
                w[s, :] = (1.0 - gain) * w[s, :] + gain * w_0[s, :]
                # Concentrate additional mass on chosen action
                extra = gain
                w[s, a] += extra
                # Renormalize to sum to 1 (for numerical stability)
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])
            else:
                # Push mass away from chosen action by eta_neg, redistribute to others
                loss_mass = eta_neg * w[s, a]
                w[s, a] -= loss_mass
                redistribute = loss_mass / (nA - 1)
                for a_other in range(nA):
                    if a_other != a:
                        w[s, a_other] += redistribute
                # Small pull toward uniform to prevent degenerate distributions
                w[s, :] = 0.95 * w[s, :] + 0.05 * w_0[s, :]
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
