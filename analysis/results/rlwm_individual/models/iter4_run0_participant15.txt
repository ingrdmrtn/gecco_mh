Note: Assumes numpy as np is already imported.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with set-size-specific arbitration weights and WM decay + global lapse.

    Mechanism
    - RL: tabular Q-learning with softmax policy (inverse temperature scaled by 10).
    - WM: probability table over actions per state; when rewarded, WM stores a peaked distribution; when not rewarded,
      WM decays toward uniform at rate wm_decay.
    - Arbitration: mixture of WM and RL with different mixture weights for small (3) vs large (6) set sizes.
    - Lapse: with small probability epsilon, choice is random uniform over actions.

    Set-size dependence
    - The arbitration weight depends directly on the block set size: wm_weight_small for set size 3, wm_weight_large for set size 6.

    Parameters
    ----------
    model_parameters : tuple
        lr : float
            RL learning rate (0..1).
        wm_weight_small : float
            WM mixture weight for set size 3 blocks (0..1).
        wm_weight_large : float
            WM mixture weight for set size 6 blocks (0..1).
        softmax_beta : float
            RL inverse temperature; internally scaled by 10.
        wm_decay : float
            WM decay toward uniform on non-rewarded trials (0..1).
        epsilon : float
            Lapse probability mixing uniform choice into the final policy (0..0.2 typical).

    Returns
    -------
    float
        Negative log-likelihood of observed actions.
    """
    lr, wm_weight_small, wm_weight_large, softmax_beta, wm_decay, epsilon = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    tiny = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # set-size-specific arbitration weight
        wm_weight = wm_weight_small if nS == 3 else wm_weight_large

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy: softmax over WM weights (very sharp)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration + lapse
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - epsilon) * p_total + epsilon * (1.0 / nA)

            p_total = max(p_total, tiny)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Rewarded: write a peaked distribution to WM for this state.
            # Not rewarded: decay WM toward uniform.
            if r > 0.5:
                w[s, :] = tiny
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Normalize and stabilize WM row
            w[s, :] = np.maximum(w[s, :], tiny)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM + Win-Stay/Lose-Shift (WSLS) three-way arbitration with set-size-specific WSLS weights.

    Mechanism
    - RL: tabular Q-learning with softmax policy.
    - WM: stores rewarded actions as peaked distributions; on non-rewarded trials, WM decays toward uniform at wm_decay.
    - WSLS: policy that repeats last rewarded action in a state (stay) and otherwise shifts uniformly to other actions.
    - Arbitration: convex combination of RL, WM, and WSLS; WSLS weight depends on set size.

    Set-size dependence
    - WSLS arbitration weight is wsls_weight_small for set size 3 and wsls_weight_large for set size 6.
      This captures the idea that simple WSLS heuristics may be more/less used under different cognitive loads.

    Parameters
    ----------
    model_parameters : tuple
        lr : float
            RL learning rate (0..1).
        wm_weight : float
            WM mixture weight in arbitration (0..1).
        softmax_beta : float
            RL inverse temperature; internally scaled by 10.
        wsls_weight_small : float
            WSLS mixture weight for set size 3 (>=0).
        wsls_weight_large : float
            WSLS mixture weight for set size 6 (>=0).
        wm_decay : float
            WM decay toward uniform when not rewarded (0..1).

    Returns
    -------
    float
        Negative log-likelihood of observed actions.
    """
    lr, wm_weight, softmax_beta, wsls_weight_small, wsls_weight_large, wm_decay = model_parameters

    softmax_beta *= 10
    softmax_beta_wm = 50
    tiny = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # maintain last action and last reward per state for WSLS
        last_action = -np.ones(nS, dtype=int)
        last_reward = np.zeros(nS)

        wsls_weight = wsls_weight_small if nS == 3 else wsls_weight_large

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Build WSLS policy for current state s as a categorical distribution
            # If last trial in this state was rewarded, stay with that action deterministically; else shift uniformly.
            if last_action[s] >= 0:
                if last_reward[s] > 0.5:
                    # Probability 1 on last_action
                    wsls_probs = np.zeros(nA)
                    wsls_probs[last_action[s]] = 1.0
                else:
                    # Uniform over actions excluding the last one
                    wsls_probs = np.ones(nA) / (nA - 1)
                    wsls_probs[last_action[s]] = 0.0
            else:
                # No history for this state yet: uniform
                wsls_probs = np.ones(nA) / nA

            # The WSLS probability of chosen action a
            p_wsls = max(wsls_probs[a], tiny)

            # Arbitration mixture; ensure weights form a valid convex combination
            # If wm_weight + wsls_weight > 1, renormalize so total <=1 and remainder goes to RL
            total_aux = wm_weight + wsls_weight
            if total_aux >= 1.0:
                wm_w = wm_weight / (total_aux + tiny)
                wsls_w = wsls_weight / (total_aux + tiny)
                rl_w = 0.0
            else:
                wm_w = wm_weight
                wsls_w = wsls_weight
                rl_w = 1.0 - wm_w - wsls_w

            p_total = rl_w * p_rl + wm_w * p_wm + wsls_w * p_wsls
            p_total = max(p_total, tiny)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            if r > 0.5:
                w[s, :] = tiny
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            w[s, :] = np.maximum(w[s, :], tiny)
            w[s, :] /= np.sum(w[s, :])

            # Update WSLS memory
            last_action[s] = a
            last_reward[s] = r

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with dynamic arbitration driven by prediction error magnitude (PE-gated WM reliance).

    Mechanism
    - RL: tabular Q-learning with softmax policy.
    - WM: rewarded actions are stored as peaked distributions; non-reward leads to decay toward uniform at wm_decay.
    - Arbitration: WM weight is dynamically reduced on trials with large absolute prediction error (|PE|),
      controlled by kappa_pe. This captures reduced confidence in WM/RL mapping when outcomes are surprising.

    Set-size dependence
    - The base arbitration weight wm_weight0 is scaled by a set-size factor: wm_weight_base = wm_weight0 * (3 / nS),
      reflecting that WM is less reliable in larger sets (6) than smaller sets (3).

    Parameters
    ----------
    model_parameters : tuple
        lr : float
            RL learning rate (0..1).
        wm_weight0 : float
            Base WM arbitration weight before set-size scaling (0..1).
        softmax_beta : float
            RL inverse temperature; internally scaled by 10.
        kappa_pe : float
            Sensitivity of WM weight to absolute prediction error (>=0). Larger values reduce WM reliance more with surprise.
        wm_decay : float
            WM decay toward uniform on non-rewarded trials (0..1).
        epsilon : float
            Lapse probability mixing uniform choice into final policy (0..0.2 typical).

    Returns
    -------
    float
        Negative log-likelihood of observed actions.
    """
    lr, wm_weight0, softmax_beta, kappa_pe, wm_decay, epsilon = model_parameters

    softmax_beta *= 10
    softmax_beta_wm = 50
    tiny = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size scaled base WM weight
        wm_weight_base = wm_weight0 * (3.0 / float(nS))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute dynamic WM weight based on previous Q (before update): |PE| = |r - Q(s,a)|
            pe = abs(r - Q_s[a])
            wm_weight_dyn = wm_weight_base * np.exp(-kappa_pe * pe)
            wm_weight_dyn = np.clip(wm_weight_dyn, 0.0, 1.0)

            p_total = wm_weight_dyn * p_wm + (1.0 - wm_weight_dyn) * p_rl
            p_total = (1.0 - epsilon) * p_total + epsilon * (1.0 / nA)

            p_total = max(p_total, tiny)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            if r > 0.5:
                w[s, :] = tiny
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            w[s, :] = np.maximum(w[s, :], tiny)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p

Model summaries and set-size impacts
- cognitive_model1: Different arbitration weights for set size 3 vs 6 directly model reduced WM reliance under higher load; WM decays on errors; global lapse accounts for occasional random responses.
- cognitive_model2: Adds a WSLS heuristic with set-size-specific weight, allowing simple strategies to contribute differently under low vs high load; WM decays on errors; convex three-way arbitration.
- cognitive_model3: Arbitration is dynamically gated by prediction error and scaled by set size (3/nS), reducing WM reliance when outcomes are surprising and when set size is larger. WM decays on errors; includes lapse.
