def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with Pearce-Hall associability and recency-based WM that is load-sensitive.

    Idea:
    - RL: learning rate is modulated per-state by associability (Pearce-Hall). Associability increases with
      recent absolute prediction error in that state and decays otherwise.
    - WM: retains a leaky trace of the last rewarded action per state. WM decay accelerates with larger set sizes,
      capturing load-induced interference.
    - Policy: mixture of RL and WM softmax policies.

    Parameters:
    - lr_base: float in [0,1]. Base RL learning rate, scaled by per-state associability.
    - beta_rl: float >= 0. RL inverse temperature; internally scaled by 10.
    - wm_weight: float in [0,1]. Mixture weight on WM vs RL policy.
    - wm_decay_base: float in [0,1]. Baseline per-trial decay of WM traces toward uniform when not refreshed.
    - xi_load_sensitivity: float >= 0. Load sensitivity scaling; larger values increase WM decay as set size grows.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_base, beta_rl, wm_weight, wm_decay_base, xi_load_sensitivity = model_parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Pearce-Hall associability per state-action (initialized moderate)
        alpha = 0.5 * np.ones((nS, nA))
        alpha_decay = 0.9  # passive decay of associability toward 0.5

        # Load-sensitive WM decay (higher nS => larger decay)
        load_factor = 1.0 + xi_load_sensitivity * max(0, nS - 3)
        wm_decay = 1.0 - (1.0 - wm_decay_base) / load_factor  # increases toward 1 with load

        log_p = 0.0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy (deterministic softmax on WM)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with associability
            delta = r - Q_s[a]
            q[s, a] += lr_base * alpha[s, a] * delta
            # small value leakage toward uniform to avoid saturation
            q[s, :] = 0.995 * q[s, :] + 0.005 * (1.0 / nA)

            # Update associability: increase with |PE| at chosen pair, decay elsewhere toward 0.5
            alpha = alpha_decay * alpha + (1.0 - alpha_decay) * 0.5
            alpha[s, a] = 0.9 * alpha[s, a] + 0.1 * min(1.0, abs(delta))

            # WM: leaky decay toward uniform every trial
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # If rewarded, refresh WM trace for the state toward the chosen action
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.8 * w[s, :] + 0.2 * one_hot  # partial overwrite

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with load-sensitive Q-value forgetting and precision-limited WM.

    Idea:
    - RL: standard delta rule but with per-trial forgetting (decay toward uniform) that accelerates with set size.
      This captures increased interference under higher load.
    - WM: stores rewarded actions but with limited precision (traces are not fully one-hot), and leaky decay.
    - Policy: fixed mixture of WM and RL.

    Parameters:
    - lr: float in [0,1]. RL learning rate.
    - beta_rl: float >= 0. RL inverse temperature; internally scaled by 10.
    - wm_weight: float in [0,1]. Mixture weight on WM vs RL policy.
    - q_forget_base: float in [0,1]. Baseline per-trial Q forgetting rate toward uniform.
    - load_sensitivity: float >= 0. Scales how much forgetting accelerates with larger set size.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, beta_rl, wm_weight, q_forget_base, load_sensitivity = model_parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-sensitive forgetting for RL
        q_forget = 1.0 - (1.0 - q_forget_base) / (1.0 + load_sensitivity * max(0, nS - 3))
        q_forget = min(max(q_forget, 0.0), 1.0)

        # WM decay fixed moderate
        wm_decay = 0.05
        # WM precision parameter: how strongly we reinforce the rewarded action
        wm_precision = 0.6  # keeps WM traces peaked but not deterministic

        log_p = 0.0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with forgetting
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            # Forget toward uniform across all state-action values each trial (load-sensitive)
            q = (1.0 - q_forget) * q + q_forget * (1.0 / nA)

            # WM update: leaky decay + precision-limited reinforcement on reward
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_precision) * w[s, :] + wm_precision * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with set-size-specific learning rates and PE-gated WM writing.

    Idea:
    - RL: uses different learning rates for small vs large set sizes (to capture altered plasticity under load).
    - WM: writing to WM is triggered only when the unsigned prediction error exceeds a threshold (surprising outcomes).
      WM traces then decay slowly. This captures selective WM encoding when outcomes are informative.
    - Policy: mixture of RL and WM.

    Parameters:
    - lr_small: float in [0,1]. RL learning rate used in small set size blocks (e.g., 3).
    - lr_large: float in [0,1]. RL learning rate used in large set size blocks (e.g., 6).
    - beta_rl: float >= 0. RL inverse temperature; internally scaled by 10.
    - wm_weight: float in [0,1]. Mixture weight on WM vs RL policy.
    - pe_threshold: float in [0,1]. Unsigned PE threshold to trigger WM write.
    - wm_decay: float in [0,1]. Per-trial decay of WM traces toward uniform.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_small, lr_large, beta_rl, wm_weight, pe_threshold, wm_decay = model_parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Choose RL learning rate by set size at the block level
        lr = lr_small if nS <= 3 else lr_large

        log_p = 0.0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay every trial
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # PE-gated WM write: only if surprising enough
            if abs(delta) >= pe_threshold and r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # Strong write on surprising reward
                w[s, :] = 0.85 * one_hot + 0.15 * (1.0 / nA)

        blocks_log_p += log_p

    return -blocks_log_p

Notes on set-size effects across models:
- Model1: xi_load_sensitivity scales WM decay as set size increases, reducing WM reliability under higher load.
- Model2: load_sensitivity accelerates RL forgetting with larger set size; WM remains moderate but precise only after reward.
- Model3: learning rate differs between sizes (lr_small vs lr_large), capturing plasticity changes with load; WM decay is size-independent, but the gating via PE favors writes primarily on informative outcomes.
