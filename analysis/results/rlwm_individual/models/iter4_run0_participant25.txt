def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with reward-gated encoding and set-size–dependent WM precision.
    - RL updates standard Q-values with a single learning rate.
    - WM stores action-probabilities that decay each revisit and are more noisy for larger set sizes.
    - WM precision is reduced as a function of set size via an inverse-temperature compression.
    - WM encoding strength depends on reward size.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0-1)
    - wm_weight: Base weight on WM policy (0-1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - tau_wm: Per-visit decay of WM towards uniform for the visited state (0-1), larger = faster decay
    - omega: Set-size interference on WM precision; WM beta scales as 50/(1 + omega*(nS-1)) (>=0)
    - kappa: Reward-gated WM encoding strength; WM moves toward chosen one-hot by kappa*r (>=0)

    Set-size impact:
    - Larger set sizes (nS=6) reduce WM precision through omega and can reduce WM influence when precision drops.
    """
    lr, wm_weight, softmax_beta, tau_wm, omega, kappa = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm_base = 50  # very deterministic baseline
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size dependent WM precision
        beta_wm_eff = softmax_beta_wm_base / (1.0 + omega * max(0, nS - 1))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Decay WM for the visited state toward uniform
            w[s, :] = (1.0 - tau_wm) * w[s, :] + tau_wm * w_0[s, :]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with set-size–reduced precision
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward-gated encoding toward chosen one-hot
            target = np.zeros(nA)
            target[a] = 1.0
            w[s, :] = (1.0 - kappa * r) * w[s, :] + (kappa * r) * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM slots mixture with choice-kernel bias in RL policy.
    - WM implements a limited-capacity "slot" memory: a state is retrievable with probability C_slot/nS.
    - If retrievable and previously rewarded, WM proposes the last rewarded action with noise xi.
      Otherwise WM is uniform.
    - RL uses standard Q-learning but the action probabilities are biased by a state-wise choice kernel.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0-1)
    - wm_weight: Base weight on WM policy (0-1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - C_slot: WM capacity in number of items; storage probability is min(1, C_slot/nS) (>=0)
    - xi: WM lapse/noise when an item is stored; probability mass xi distributed over non-remembered actions (0-1)
    - psi: Scale of the choice-kernel bias added to Q in the RL policy (>=0)

    Set-size impact:
    - Larger set sizes reduce the chance the current state is in WM via p_store = min(1, C_slot/nS).
    """
    lr, wm_weight, softmax_beta, C_slot, xi, psi = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # used if we fallback to softmax representation of w (not necessary here)
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))   # will mirror remembered distribution for transparency
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Remembered last rewarded action per state; -1 = none yet
        remembered = -np.ones(nS, dtype=int)

        # Simple exponentially-decaying choice kernel per state
        c = np.zeros((nS, nA))
        lambda_c = 0.9  # fixed decay within block

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Storage probability based on set size
            p_store = min(1.0, max(0.0, C_slot / max(1, nS)))

            # RL policy with choice kernel bias
            Q_s = q[s, :]
            Q_eff = Q_s + psi * c[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_eff - Q_eff[a])))

            # WM policy: if stored and remembered, propose last rewarded action with noise xi
            if remembered[s] >= 0:
                a_mem = int(remembered[s])
                # Stored case distribution
                if a == a_mem:
                    p_wm_stored = 1.0 - xi
                else:
                    p_wm_stored = xi / (nA - 1)
                # Mixture with probability of being stored
                p_wm = p_store * p_wm_stored + (1.0 - p_store) * (1.0 / nA)
            else:
                # Nothing remembered: effectively uniform, regardless of storage
                p_wm = 1.0 / nA

            # Combine RL and WM
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: on reward, remember the chosen action; also sync w to remembered distribution
            if r > 0.5:
                remembered[s] = a
                w[s, :] = (xi / (nA - 1)) * np.ones(nA)
                w[s, a] = 1.0 - xi
            # If no memory yet, keep w uniform; else keep prior w

            # Update choice kernel: decay then set chosen action to 1
            c[s, :] *= lambda_c
            c[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with uncertainty-based arbitration and set-size–dependent WM decay.
    - RL: standard Q-learning; arbitration weight depends on RL uncertainty (via visit count) and set size.
    - WM: cache of recent correct actions; encodes on reward and decays faster for larger set sizes.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0-1)
    - wm_weight0: Baseline arbitration bias toward WM before modulation (real-valued)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - sigma0: Prior uncertainty scale; fewer visits -> larger uncertainty -> more WM use (>=0)
    - k_ent: Scale translating RL uncertainty into WM weighting (>=0)
    - k_ns: Set-size penalty and WM decay scale; larger increases penalize WM at larger nS (>=0)

    Set-size impact:
    - Larger set sizes reduce WM weight via the -k_ns*(nS-3) term in the arbitration and increase WM decay per visit.
    """
    lr, wm_weight0, softmax_beta, sigma0, k_ent, k_ns = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # WM policy inverse temperature
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        visits = np.zeros(nS)  # per-state visit count

        # Set-size–dependent WM decay rate per visit (higher for larger nS)
        wm_decay = 1.0 - np.exp(-k_ns * max(0, nS - 1))  # in [0,1)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL uncertainty from visit count
            u = sigma0 / np.sqrt(visits[s] + 1.0)  # larger when fewer visits

            # Arbitration: sigmoid of baseline + uncertainty gain - set-size penalty
            z = wm_weight0 + k_ent * u - k_ns * max(0, nS - 3)
            wm_weight_eff = 1.0 / (1.0 + np.exp(-z))

            # Apply WM decay for visited state before computing policy
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: cache the correct action on reward (move to one-hot)
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = target

            # Update visit count
            visits[s] += 1.0

        blocks_log_p += log_p

    return -blocks_log_p
