def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with eligibility traces and set-size interference.

    Mechanism:
    - RL: tabular Q-learning augmented with an eligibility trace over state-action pairs.
      This allows credit to persist beyond the current (s,a) and accelerates learning with repetition.
    - WM: fast one-shot encoding toward the chosen action when rewarded, acting like a cached mapping.
      Its contribution to choice is reduced by set-size-dependent interference.
    - Arbitration: convex mixture between RL and WM policies. The WM weight is downscaled as:
          wm_weight_eff = wm_weight / (1 + interference * (nS - 1))
      Larger nS produces more interference and less WM influence.

    Parameters (in order):
    - lr: RL learning rate in [0,1].
    - wm_weight: Baseline mixture weight on WM in [0,1].
    - softmax_beta: RL inverse temperature (scaled by 10 internally).
    - trace: Eligibility trace decay (lambda) in [0,1]; higher sustains broader credit assignment.
    - interference: Set-size interference coefficient >=0 controlling WM weight downscaling with nS.
    - enc_gain: WM encoding gain in [0,1]; strength of shifting WM toward chosen action on reward.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, trace, interference, enc_gain = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Clamp parameters to sensible ranges
    wm_weight = np.clip(wm_weight, 0.0, 1.0)
    lr = np.clip(lr, 0.0, 1.0)
    trace = np.clip(trace, 0.0, 1.0)
    interference = max(0.0, float(interference))
    enc_gain = np.clip(enc_gain, 0.0, 1.0)

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        mask = (blocks == b)
        block_actions = actions[mask].astype(int)
        block_rewards = rewards[mask].astype(float)
        block_states = states[mask].astype(int)
        block_set_sizes = set_sizes[mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL, WM, and eligibility traces
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        e = np.zeros((nS, nA))  # eligibility

        # Set-sizeâ€“dependent WM weight
        wm_weight_eff = wm_weight / (1.0 + interference * max(0, nS - 1))
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            # RL policy likelihood of observed action
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy likelihood of observed action
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration and likelihood accumulation
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with eligibility trace
            delta = r - Q_s[a]
            # Decay eligibility and add for current (s,a)
            e *= trace
            e[s, a] += 1.0
            # Update all Qs proportional to eligibility
            q += lr * delta * e

            # WM update: reward-gated one-shot encoding toward chosen action
            if r > 0:
                # Move WM row toward a one-hot at 'a' with gain enc_gain
                w[s, :] = (1.0 - enc_gain) * w[s, :]
                w[s, a] += enc_gain

            # Keep WM rows normalized and bounded
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] = np.clip(w[s, :], 1e-9, 1.0)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with entropy-based arbitration and RL decay.

    Mechanism:
    - RL: tabular Q-learning with per-trial decay toward uniform (forgetting), plus softmax choice.
    - WM: rapid encoding of chosen action (reward-gated), with adjustable determinism.
    - Arbitration: weight on WM is computed online from WM certainty (1 - entropy).
        wm_weight_eff = wm_weight * sigmoid(arb_slope * (H_ref - H(W_s)))
      where H(W_s) is the entropy of the WM policy for state s, and H_ref = log(nA).
      When WM is sharp (low entropy), its weight increases; for diffuse WM, arbitration favors RL.

    Parameters (in order):
    - lr: RL learning rate in [0,1].
    - wm_weight: Maximum mixture weight for WM in [0,1].
    - softmax_beta: RL inverse temperature (scaled by 10 internally).
    - rl_decay: Per-trial RL forgetting toward uniform in [0,1]; 0=no decay.
    - wm_gain: Scales WM determinism by multiplying the baseline WM beta (>=0).
    - arb_slope: Slope of the arbitration sigmoid (>0) controlling sensitivity to entropy.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, rl_decay, wm_gain, arb_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    wm_weight = np.clip(wm_weight, 0.0, 1.0)
    lr = np.clip(lr, 0.0, 1.0)
    rl_decay = np.clip(rl_decay, 0.0, 1.0)
    wm_gain = max(0.0, float(wm_gain))
    arb_slope = max(1e-6, float(arb_slope))  # positive

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        mask = (blocks == b)
        block_actions = actions[mask].astype(int)
        block_rewards = rewards[mask].astype(float)
        block_states = states[mask].astype(int)
        block_set_sizes = set_sizes[mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        H_ref = np.log(nA)

        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with adjustable determinism
            W_s = w[s, :]
            beta_wm_eff = softmax_beta_wm * (1.0 + wm_gain)
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Entropy-based arbitration (state-specific)
            # Entropy of W_s (with safety for zeros)
            W_safe = np.clip(W_s, 1e-12, 1.0)
            H_w = -np.sum(W_safe * np.log(W_safe))
            # Higher when uncertain. Convert to weight via sigmoid scaled by arb_slope
            # Target is to increase weight when entropy is lower than H_ref
            wm_cert_sig = 1.0 / (1.0 + np.exp(-arb_slope * (H_ref - H_w)))
            wm_weight_eff = np.clip(wm_weight * wm_cert_sig, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with decay toward uniform
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            # Global decay toward uniform baseline
            q = (1.0 - rl_decay) * q + rl_decay * (1.0 / nA)

            # WM update: reward-gated encoding toward chosen action
            if r > 0:
                # Simple one-shot push toward chosen action, proportional to (1 - current prob)
                gain = 1.0 - w[s, a]
                w[s, a] += 0.5 * gain  # partial collapse toward one-hot when rewarded
                # Renormalize the row
                w[s, :] = np.clip(w[s, :], 1e-9, 1.0)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with size-dependent WM gating and lapse.

    Mechanism:
    - RL: standard Q-learning with softmax policy.
    - WM: rapid mapping that is both more confident (deterministic) and more reinforced with higher 'wm_conf'.
    - Arbitration: baseline WM weight downscales with set size via a power-law:
          wm_weight_eff = wm_weight0 / (1 + (nS - 1) ** size_power)
      Captures strong WM use at low set size and reduced WM at high load.
    - Lapse: a small probability of uniform random responding applied to the final choice.

    Parameters (in order):
    - lr: RL learning rate in [0,1].
    - wm_weight0: Baseline WM mixture weight in [0,1] (before size downscaling).
    - softmax_beta: RL inverse temperature (scaled by 10 internally).
    - size_power: Exponent (>0) controlling how sharply WM is down-weighted with set size.
    - wm_conf: Controls both WM encoding strength and determinism (>=0).
    - lapse: Probability of a random choice on each trial in [0,0.5).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight0, softmax_beta, size_power, wm_conf, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    lr = np.clip(lr, 0.0, 1.0)
    wm_weight0 = np.clip(wm_weight0, 0.0, 1.0)
    size_power = max(1e-6, float(size_power))
    wm_conf = max(0.0, float(wm_conf))
    lapse = np.clip(lapse, 0.0, 0.499)

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        mask = (blocks == b)
        block_actions = actions[mask].astype(int)
        block_rewards = rewards[mask].astype(float)
        block_states = states[mask].astype(int)
        block_set_sizes = set_sizes[mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size gated WM weight
        wm_weight_eff = wm_weight0 / (1.0 + max(0, nS - 1) ** size_power)
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with confidence-scaled determinism
            W_s = w[s, :]
            beta_wm_eff = softmax_beta_wm * (1.0 + wm_conf)
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture policy then lapse
            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: confidence-scaled one-shot encoding (reward-gated, stronger with wm_conf)
            if r > 0:
                enc = 1.0 - np.exp(-wm_conf)  # in [0,1) and increasing with wm_conf
                w[s, :] = (1.0 - enc) * w[s, :]
                w[s, a] += enc
                # Normalize and bound
                w[s, :] = np.clip(w[s, :], 1e-9, 1.0)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
