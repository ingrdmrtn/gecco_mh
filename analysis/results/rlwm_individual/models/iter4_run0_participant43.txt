def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with load-dependent WM decay and arbitration.

    Idea:
    - RL learns via delta rule and softmax choice.
    - WM stores recently rewarded action policies per state, but decays toward uniform.
    - Both WM decay and arbitration weight depend on cognitive load (set size).

    Parameters:
    - lr: RL learning rate (0..1).
    - wm_weight_base: Base mixture weight for WM contribution (0..1).
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - wm_decay: Baseline WM decay/storage rate (0..1). Higher -> faster updates and faster decay.
    - load_gain: Load modulation factor (>=0). Scales both WM decay and reduces WM arbitration under higher load.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, load_gain = model_parameters
    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Effective parameters under load
        wm_weight_eff = np.clip(wm_weight_base * (1.0 - load_gain * load), 0.0, 1.0)
        wm_decay_eff = np.clip(wm_decay * (1.0 + load_gain * load), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # Policy for the working memory: softmax over WM action distribution
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy with load-adjusted arbitration
            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM updating: reward strengthens the chosen action; no reward leaks toward uniform
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                # Move WM distribution toward the one-hot target at a rate tied to wm_decay_eff
                w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * target
            else:
                # Leak toward uniform baseline under no reward; faster leak under higher load via wm_decay_eff
                w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with novelty-weighted arbitration and load penalty.

    Idea:
    - RL learns via delta rule and softmax.
    - WM stores rewarded actions deterministically and leaks slightly when not rewarded.
    - Arbitration varies by state novelty: earlier visits rely more on WM; over time, RL dominates.
    - Higher load (larger set size) reduces WM arbitration via a logistic mapping.

    Parameters:
    - lr: RL learning rate (0..1).
    - wm_anchor: Base WM arbitration anchor in [0..1] (transformed via logit/logistic).
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - novelty_weight: Weight scaling of novelty contribution to WM arbitration (can be +/-).
    - load_arbi_slope: Strength with which load (0 for 3, 1 for 6) shifts arbitration toward RL (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_anchor, softmax_beta, novelty_weight, load_arbi_slope = model_parameters
    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        visits = np.zeros(nS)

        # Helper: logistic transform for arbitration
        def logistic(x):
            return 1.0 / (1.0 + np.exp(-x))

        # Start from anchor in probability scale; convert to logit
        eps = 1e-6
        anchor = np.clip(wm_anchor, eps, 1.0 - eps)
        base_logit = np.log(anchor) - np.log(1.0 - anchor)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Compute novelty for this state
            novelty = 1.0 / (1.0 + visits[s])  # 1 on first visit, decays with visits
            visits[s] += 1.0

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # Working memory policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration logit combines anchor, novelty, and load penalty
            logit_w = base_logit + novelty_weight * novelty - load_arbi_slope * load
            wm_weight_eff = logistic(logit_w)

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM updating:
            if r >= 0.5:
                # Deterministically store the rewarded action (strong WM encoding)
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = target
            else:
                # Mild leak toward uniform when not rewarded; leak increases slightly under load
                leak = 0.05 + 0.20 * load
                leak = np.clip(leak, 0.0, 1.0)
                w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with load-induced binding noise and RL recency forgetting.

    Idea:
    - RL learns via delta rule but also forgets (toward uniform Q) with a recency-dependent rate.
    - WM stores rewarded actions but suffers binding noise that increases with load:
      stored target is a mixture of the correct action and an incorrect action.
    - Under no reward, WM leaks toward uniform at a rate tied to recency_forget and load.

    Parameters:
    - lr: RL learning rate (0..1).
    - wm_weight: WM mixture weight (0..1).
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - bind_noise: Base binding noise level for WM storage (0..1). Scales with load.
    - recency_forget: Baseline forgetting rate (0..1) applied to both WM leak and RL forgetting.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, bind_noise, recency_forget = model_parameters
    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Effective load-dependent parameters
        noise_eff = np.clip(bind_noise * load, 0.0, 1.0)
        rl_forget_rate = np.clip(0.5 * recency_forget * load, 0.0, 1.0)
        wm_leak_rate = np.clip(recency_forget * load, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy with deterministic readout
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # RL recency forgetting toward uniform values
            q[s, :] = (1.0 - rl_forget_rate) * q[s, :] + rl_forget_rate * (1.0 / nA)

            # WM updating with binding noise on reward; leak on no reward
            if r >= 0.5:
                # Construct a noisy target distribution: mostly the chosen action,
                # but with probability mass noise_eff spread over the other actions.
                target = np.zeros(nA)
                target[a] = 1.0
                other_mask = np.ones(nA, dtype=bool)
                other_mask[a] = False
                uniform_others = np.zeros(nA)
                if np.sum(other_mask) > 0:
                    uniform_others[other_mask] = 1.0 / np.sum(other_mask)
                noisy_target = (1.0 - noise_eff) * target + noise_eff * uniform_others
                w[s, :] = noisy_target
            else:
                # Leak toward uniform under no reward, scaled by load and recency_forget
                w[s, :] = (1.0 - wm_leak_rate) * w[s, :] + wm_leak_rate * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
