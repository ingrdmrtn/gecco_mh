def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with prediction-error adaptive learning + WM gated by load and reward contingency.

    Idea:
    - RL learning rate adapts with recent surprise: lr_t = clip(lr0 * (1 + alpha_att * |PE_{t-1}|), 0..1).
    - WM provides a sharp mapping when recently rewarded; otherwise it decays toward uniform and mildly penalizes
      the last failed action.
    - WM influence is reduced under higher load via an exponential factor in set size.

    Parameters (model_parameters):
    - lr0: Base RL learning rate (0..1)
    - alpha_att: Prediction-error gain on learning rate (>=0); larger -> more adaptive learning
    - wm_weight0: Base WM mixture weight (0..1)
    - softmax_beta: Inverse temperature for RL choice (scaled internally by 10)
    - wm_decay: WM decay/leak to uniform per trial (0..1)
    - load_k: Load sensitivity reducing WM weight with larger set size (>=0); wm_weight_eff ~ exp(-load_k*(nS-3))

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr0, alpha_att, wm_weight0, softmax_beta, wm_decay, load_k = model_parameters

    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM mixture weight reduced under higher load
        wm_weight_eff = wm_weight0 * np.exp(-load_k * max(0, nS - 3))

        log_p = 0.0
        prev_abs_pe = 0.0  # to adapt learning rate
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probability of chosen action (as per template form)
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(1e-12, denom_rl)

            # WM policy: softmax over WM weights (sharp, but will be degraded through decay)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(1e-12, denom_wm)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(1e-12, min(1.0, p_total))
            log_p += np.log(p_total)

            # RL update with PE-adaptive learning rate
            pe = r - Q_s[a]
            lr_t = lr0 * (1.0 + alpha_att * prev_abs_pe)
            lr_t = max(0.0, min(1.0, lr_t))
            q[s, a] += lr_t * pe
            prev_abs_pe = abs(pe)

            # WM decay toward uniform each visit (load-independent here; load impact is in mixture)
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM reinforcement/suppression
            if r > 0.5:
                # One-shot reinforcement: store the rewarded mapping
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Mild suppression of failed action to reflect "avoid that" memory
                w[s, a] = max(1e-12, (1.0 - wm_decay) * w[s, a])
                # renormalize
                total = np.sum(w[s, :])
                if total > 0:
                    w[s, :] /= total
                else:
                    w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with UCB-style exploration bonus + capacity-limited WM (interference via set size).

    Idea:
    - RL uses a softmax over Q plus an uncertainty bonus U(s,a) = c / sqrt(1 + N(s,a)),
      encouraging exploration early on.
    - WM acts like a categorical association table that stores rewarded mappings,
      but suffers interference when set size exceeds a capacity parameter.
      Interference both increases WM noise and speeds up decay toward uniform.
    - Mixture weight is fixed per block but WM policy becomes noisier with higher interference.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - softmax_beta: Inverse temperature for RL choice (scaled internally by 10)
    - wm_weight0: Base WM mixture weight (0..1)
    - capacity: WM capacity (>=1, continuous proxy); interference grows when nS > capacity
    - ucb_c: Strength of the UCB exploration bonus (>=0)
    - wm_beta: Scales WM determinism (>=0); higher -> more deterministic WM

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta, wm_weight0, capacity, ucb_c, wm_beta = model_parameters

    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # base determinism
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        N = np.zeros((nS, nA))  # visit counts for UCB bonus

        # Interference factor: 0 when nS <= capacity, increases as nS exceeds capacity
        if capacity <= 0:
            phi = 1.0
        else:
            phi = max(0.0, min(1.0, (float(nS) - float(capacity)) / max(1.0, float(nS))))
        # Effective WM parameters under interference
        wm_leak_eff = 0.05 + 0.45 * phi  # more interference -> faster leak to uniform
        beta_wm_eff = max(1.0, wm_beta * softmax_beta_wm * (1.0 - 0.8 * phi))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL with UCB bonus
            bonus = ucb_c / np.sqrt(1.0 + N[s, :])
            Q_aug = q[s, :] + bonus
            denom_rl = np.sum(np.exp(softmax_beta * (Q_aug - Q_aug[a])))
            p_rl = 1.0 / max(1e-12, denom_rl)

            # WM policy with interference-modulated temperature
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / max(1e-12, denom_wm)

            p_total = wm_weight0 * p_wm + (1.0 - wm_weight0) * p_rl
            p_total = max(1e-12, min(1.0, p_total))
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe
            N[s, a] += 1.0

            # WM update: leak toward uniform + one-shot on reward
            w[s, :] = (1.0 - wm_leak_eff) * w[s, :] + wm_leak_eff * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Under interference, failed actions are less confidently suppressed
                suppress = max(0.0, 1.0 - 0.5 * (1.0 - phi))
                w[s, a] = max(1e-12, suppress * w[s, a])
                total = np.sum(w[s, :])
                if total > 0:
                    w[s, :] /= total
                else:
                    w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Entropy-based arbitration between RL and WM.

    Idea:
    - Compute RL and WM action distributions each trial.
    - Arbitration weight for WM depends on relative uncertainty: higher WM certainty (lower entropy) than RL
      increases WM influence, and vice versa. A sigmoid with slope arbit_k and bias arbit_b sets the mixture.
    - WM has its own temperature and leaks to uniform; it is also indirectly load-sensitive because when more
      states compete, WM distributions are flatter, increasing WM entropy and thus reducing its arbitration weight.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - softmax_beta: Inverse temperature for RL choice (scaled internally by 10)
    - arbit_k: Slope of the arbitration sigmoid; higher -> more decisive arbitration
    - arbit_b: Bias of the arbitration sigmoid (can be negative/positive)
    - wm_leak: Leak of WM toward uniform per visit (0..1)
    - wm_beta: Scales WM determinism (>=0); effective WM beta = max(1, wm_beta * 50)

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta, arbit_k, arbit_b, wm_leak, wm_beta = model_parameters

    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic (base)
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        beta_wm_eff = max(1.0, wm_beta * softmax_beta_wm)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy vector
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(logits_rl)
            pi_rl = pi_rl / max(1e-12, np.sum(pi_rl))
            p_rl = max(1e-12, pi_rl[a])

            # WM policy vector
            logits_wm = beta_wm_eff * (W_s - np.max(W_s))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / max(1e-12, np.sum(pi_wm))
            p_wm = max(1e-12, pi_wm[a])

            # Entropies
            H_rl = -np.sum(pi_rl * np.log(np.maximum(1e-12, pi_rl)))
            H_wm = -np.sum(pi_wm * np.log(np.maximum(1e-12, pi_wm)))

            # Arbitration: WM gets more weight when it is more certain than RL (H_wm < H_rl)
            # wm_weight_t = sigmoid(arbit_k * (H_rl - H_wm) + arbit_b)
            x = arbit_k * (H_rl - H_wm) + arbit_b
            wm_weight_t = 1.0 / (1.0 + np.exp(-x))

            p_total = wm_weight_t * p_wm + (1.0 - wm_weight_t) * p_rl
            p_total = max(1e-12, min(1.0, p_total))
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM update: leak to uniform plus one-shot on reward
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # With no reward, softly reduce the chosen action probability to reflect avoidance
                w[s, a] = max(1e-12, (1.0 - 0.5 * wm_leak) * w[s, a])
                tot = np.sum(w[s, :])
                if tot > 0:
                    w[s, :] /= tot
                else:
                    w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
