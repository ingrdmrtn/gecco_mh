def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Uncertainty-gated RL+WM mixture.

    Idea
    - Choices arise from a mixture of RL and WM policies.
    - The WM weight is computed dynamically each trial based on relative uncertainty:
      higher WM weight when RL is uncertain (high entropy) and WM is confident (low entropy).
    - WM weight also shifts with set size (3 vs 6) via a size-bias term.
    - WM has its own learning and decay dynamics.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback for each trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Number of unique states in the current block for each trial (3 or 6 here).
    model_parameters : iterable of length 6
        - lr: RL learning rate (0..1).
        - softmax_beta: Base RL inverse temperature (scaled internally by 10).
        - gate_bias: Intercept for WM gating; larger favors WM.
        - gate_size: Set-size bias for WM gating; positive values favor WM more in small sets.
        - wm_decay: Per-trial WM decay toward uniform (0..1).
        - wm_learn: WM imprint/learning strength on the current state (0..1).

    Set-size impact
    - WM weight per trial includes an additive bias gate_size * (3.5 - nS), increasing WM reliance at nS=3
      if gate_size > 0 and decreasing it at nS=6.

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta_param, gate_bias, gate_size, wm_decay, wm_learn = model_parameters

    softmax_beta = softmax_beta_param * 10.0
    softmax_beta_wm = 50.0  # deterministic WM readout
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability for chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy probability for chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Entropy-based gating
            # Convert Q and W to probabilities to measure entropy
            rl_probs = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            rl_probs = rl_probs / np.sum(rl_probs)
            wm_probs = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            wm_probs = wm_probs / np.sum(wm_probs)

            H_rl = -np.sum(rl_probs * np.log(np.clip(rl_probs, eps, 1.0)))
            H_wm = -np.sum(wm_probs * np.log(np.clip(wm_probs, eps, 1.0)))

            # Trial-wise WM weight via sigmoid
            size_term = gate_size * (3.5 - nS)
            wm_weight_t = 1.0 / (1.0 + np.exp(-(gate_bias + size_term + (H_rl - H_wm))))

            # Mixture likelihood
            p_total = wm_weight_t * p_wm + (1.0 - wm_weight_t) * p_rl
            log_p += np.log(np.clip(p_total, eps, 1.0))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: global decay + local imprint
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # imprint toward the chosen action, scaled by reward
            w[s, :] = (1.0 - wm_learn) * w[s, :]
            w[s, a] += wm_learn * r

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Reward-rate-adaptive temperature with size-biased WM mixture.

    Idea
    - RL temperature adapts online from the running reward rate: higher reward -> higher beta (more exploit).
    - WM weight is a logistic transform of a base weight with a size-bias (more WM for small sets if positive).
    - WM decays toward uniform and imprints more strongly on rewarded trials.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback for each trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Number of unique states in the current block for each trial (3 or 6 here).
    model_parameters : iterable of length 6
        - lr: RL learning rate (0..1).
        - beta0: Baseline (log) inverse temperature for RL.
        - beta_gain: Sensitivity of log-temperature to running reward rate (positive -> exploit when doing well).
        - wm_weight_base: Base WM mixture weight (0..1), transformed via logit internally.
        - wm_decay: Per-trial WM decay toward uniform (0..1).
        - wm_boost_size: Size-bias for WM mixture weight; positive -> more WM for smaller sets.

    Set-size impact
    - WM mixture weight includes wm_boost_size * (3.5 - nS), increasing WM reliance at nS=3 and reducing at nS=6.

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, beta0, beta_gain, wm_weight_base, wm_decay, wm_boost_size = model_parameters

    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Running reward rate (exponentially weighted) to adapt beta
        rr = 0.5
        rr_alpha = 0.2  # fixed smoothing within the model

        # Precompute size-biased WM weight (static per block)
        base_logit = np.log(np.clip(wm_weight_base, eps, 1 - eps)) - np.log(1 - np.clip(wm_weight_base, eps, 1 - eps))
        wm_weight_eff = 1.0 / (1.0 + np.exp(-(base_logit + wm_boost_size * (3.5 - nS))))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL temperature from reward rate
            softmax_beta = 10.0 * np.exp(beta0 + beta_gain * (rr - 0.5))

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(np.clip(p_total, eps, 1.0))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: decay + reward-contingent imprint
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0.0:
                # Stronger imprint when rewarded
                imprint = min(1.0, 2.0 * wm_decay)  # bounded by 1, uses wm_decay magnitude
                w[s, :] = (1.0 - imprint) * w[s, :]
                w[s, a] += imprint
            # else: only decay already applied

            # Update running reward rate
            rr = (1.0 - rr_alpha) * rr + rr_alpha * r

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Recall-probability WM with size effect + RL.

    Idea
    - WM contributes with a recall probability that declines with set size.
    - When recalled, WM is sharp (deterministic); otherwise RL governs.
    - WM updates with decay each trial; correct feedback strengthens the chosen item,
      while errors drive the memory toward uniform (erasing wrong associations).

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback for each trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Number of unique states in the current block for each trial (3 or 6 here).
    model_parameters : iterable of length 6
        - lr: RL learning rate (0..1).
        - softmax_beta: RL inverse temperature (scaled internally by 10).
        - recall_bias: Baseline recall log-odds; larger -> more WM.
        - recall_size: Size slope on recall log-odds; positive -> more recall for small sets.
        - wm_decay: Per-trial WM decay toward uniform (0..1).
        - wm_learn: WM learning strength applied on the current state (0..1).

    Set-size impact
    - Recall probability per trial is sigmoid(recall_bias + recall_size * (3.5 - nS)),
      increasing WM influence at nS=3 and reducing it at nS=6 if recall_size > 0.

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta_param, recall_bias, recall_size, wm_decay, wm_learn = model_parameters

    softmax_beta = softmax_beta_param * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute block-level recall probability (constant within block given its set size)
        recall_p = 1.0 / (1.0 + np.exp(-(recall_bias + recall_size * (3.5 - nS))))
        recall_p = np.clip(recall_p, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = recall_p * p_wm + (1.0 - recall_p) * p_rl
            log_p += np.log(np.clip(p_total, eps, 1.0))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: decay every trial
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Correct: imprint toward chosen action
            if r > 0.0:
                w[s, :] = (1.0 - wm_learn) * w[s, :]
                w[s, a] += wm_learn
            else:
                # Error: push memory back toward uniform for this state (erase wrong trace)
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
