Below are three alternative cognitive models that adhere to your template and constraints. Each model mixes an RL policy with a WM policy but uses distinct mechanisms and parameterizations to capture load (set size) effects and trial-by-trial control.

Note: All three functions assume numpy is available as np and follow the requested signature, returning the negative log-likelihood.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with PE-gated WM reliance and set-size–dependent lapses.

    Idea:
    - Choices are a mixture of RL and WM policies.
    - WM weight is gated on the fly by the signed reward prediction error (RPE), so rewarded,
      surprising events favor WM usage; non-rewarded or expected outcomes reduce it.
    - There is a set-size–dependent lapse that reduces effective WM contribution in larger sets.
    - WM stores the rewarded action for a state (supervised), and decays toward uniform otherwise.

    Parameters (model_parameters):
    - lr: RL learning rate in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_weight_base: base WM mixture weight in [0,1].
    - pe_gate_slope: positive scalar controlling how strongly RPE modulates WM reliance (sigmoidal gate).
    - lapse_base: base lapse component reducing WM effective weight (>=0).
    - lapse_setsize_slope: additional lapse increase per extra item over 3 (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_base, pe_gate_slope, lapse_base, lapse_setsize_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size–dependent lapse that downweights WM
        extra_load = max(0, nS - 3)
        lapse = 1.0 - np.exp(- (lapse_base + lapse_setsize_slope * extra_load))
        lapse = np.clip(lapse, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # RPE-gated WM weight (sigmoid in RPE = r - Q(s,a))
            rpe = r - Q_s[a]
            gate = 1.0 / (1.0 + np.exp(-pe_gate_slope * rpe))
            wm_w_eff = wm_weight_base * (1.0 - lapse) * gate
            wm_w_eff = np.clip(wm_w_eff, 0.0, 1.0)

            p_total = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM update: move toward uniform each trial; overwrite on reward
            # Decay a bit toward uniform if not rewarded
            if r > 0.5:
                # One-shot supervised overwriting toward the chosen action
                w[s, :] = 0.0 * w[s, :] + w_0[s, :]
                w[s, a] = 1.0
            else:
                # Gentle decay toward uniform on non-reward
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with set-size–dependent optimistic initialization + WM as leaky recency memory.

    Idea:
    - RL starts with optimistic initial Q-values that depend on set size (separate v0 for small vs large sets).
    - WM is a leaky integrator of recent rewarded actions per state (recency memory); the leak is a parameter.
    - Mixture of RL and WM policies with a set-size-independent mixture weight.

    Parameters (model_parameters):
    - lr: RL learning rate in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_weight: mixture weight for WM in [0,1].
    - v0_small: optimistic Q initialization for set size 3 blocks.
    - v0_large: optimistic Q initialization for set size 6 blocks.
    - wm_leak: leak parameter in [0,1]; higher -> faster forgetting toward uniform for WM.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight, v0_small, v0_large, wm_leak = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        v0 = v0_small if nS <= 3 else v0_large
        q = v0 * np.ones((nS, nA))

        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        wm_w = np.clip(wm_weight, 0.0, 1.0)
        leak = np.clip(wm_leak, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_w * p_wm + (1.0 - wm_w) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM leaky recency update:
            # 1) Leak toward uniform across all states
            w = (1.0 - leak) * w + leak * w_0
            # 2) If rewarded, push this state's WM toward the chosen action (recency memory)
            if r > 0.5:
                w[s, :] = 0.5 * w[s, :] + 0.5 * w_0[s, :]  # gentle reset before imprint
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with entropy-based arbitration and WM learning/decay.

    Idea:
    - Arbitration between RL and WM is dynamic: when a policy is more certain (lower entropy),
      it gets more weight. Arbitration is controlled by an inverse-temperature-like parameter.
    - WM learns via a supervised learning rate on rewarded trials and decays toward uniform each trial.
    - This allows load to impact arbitration indirectly (larger sets typically make WM more uncertain).

    Parameters (model_parameters):
    - lr: RL learning rate in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_prior: baseline prior weight in favor of WM (0-1) before arbitration.
    - arbitration_beta: non-negative; higher makes arbitration more sensitive to entropy differences.
    - wm_lr: WM supervised learning rate on rewarded trials (0-1).
    - wm_decay: per-trial decay of WM toward uniform (0-1).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_prior, arbitration_beta, wm_lr, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    def softmax_probs(x, beta):
        # stable softmax for a single state's preferences
        x = x - np.max(x)
        z = np.exp(beta * x)
        return z / np.sum(z)

    def entropy(p):
        p_safe = np.clip(p, 1e-12, 1.0)
        return -np.sum(p_safe * np.log(p_safe))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL and WM choice probabilities for state s
            Q_s = q[s, :]
            P_rl = softmax_probs(Q_s, softmax_beta)
            p_rl = P_rl[a]

            W_s = w[s, :]
            P_wm = softmax_probs(W_s, softmax_beta_wm)
            p_wm = P_wm[a]

            # Entropy-based arbitration: lower entropy -> higher reliability
            H_rl = entropy(P_rl)
            H_wm = entropy(P_wm)
            # Reliability scores (higher when entropy is lower)
            rel_rl = np.exp(-arbitration_beta * H_rl)
            rel_wm = np.exp(-arbitration_beta * H_wm)

            # Combine with wm_prior; normalize to form a mixture weight
            base_wm = np.clip(wm_prior, 0.0, 1.0)
            # Convert priors to pseudo-counts and multiply by reliability
            mix_wm = base_wm * rel_wm
            mix_rl = (1.0 - base_wm) * rel_rl
            denom = mix_wm + mix_rl
            wm_w = mix_wm / denom if denom > 0 else base_wm
            wm_w = np.clip(wm_w, 0.0, 1.0)

            p_total = wm_w * p_wm + (1.0 - wm_w) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # WM supervised learning on reward
            if r > 0.5:
                w[s, :] = (1.0 - wm_lr) * w[s, :] + wm_lr * w_0[s, :]
                w[s, a] = min(1.0, w[s, a] + wm_lr)

        blocks_log_p += log_p

    return -blocks_log_p

Notes on load (set size) effects across models:
- Model 1: WM contribution is explicitly reduced by a set-size–dependent lapse; larger sets downweight WM.
- Model 2: RL initial optimism differs between small vs large sets, shaping early exploration; WM is not directly load-scaled, but larger sets naturally dilute recency storage utility.
- Model 3: Arbitration uses entropy; WM tends to be more uncertain (higher entropy) in larger sets, so the arbitration automatically shifts weight toward RL under higher loads.
