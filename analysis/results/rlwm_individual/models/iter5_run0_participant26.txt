def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with RPE-gated arbitration and set-size cost.
    - Policy: convex mixture of RL softmax and WM softmax.
    - Arbitration: WM weight increases when unsigned RPE is small (RL confident),
      and decreases with larger set sizes (capacity cost).
    - WM dynamics: leaky cache toward uniform; reward strengthens chosen action.

    Parameters (tuple):
    - lr: learning rate for RL value updates and WM storage after reward (0..1).
    - wm_base: baseline WM contribution before gating (0..1 in practice).
    - softmax_beta: inverse temperature for RL softmax; internally scaled x10.
    - wm_decay: per-visit leak of WM toward uniform (0..1).
    - arbi_rpe_gain: strength of RPE-based arbitration (>=0). Weight term uses (1 - |RPE|).
    - ss_cost_gain: penalty on WM weight with larger set sizes (>=0). Weight reduced by ss_cost_gain*(nS-3).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_base, softmax_beta, wm_decay, arbi_rpe_gain, ss_cost_gain = model_parameters
    softmax_beta *= 10.0  # higher upper bound as specified
    softmax_beta_wm = 50.0  # very deterministic WM
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size penalty applied per-block to the arbitration baseline
        wm_weight_base = wm_base - ss_cost_gain * max(0, nS - 3)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probability of chosen action (implicit via "trick")
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute unsigned RPE from RL as a gating signal
            rpe = r - Q_s[a]
            # Arbitration weight: increase WM when RL is confident (small |RPE|)
            wm_weight_dyn = wm_weight_base + arbi_rpe_gain * (1.0 - np.abs(rpe))
            wm_weight_dyn = np.clip(wm_weight_dyn, 0.0, 1.0)

            p_total = wm_weight_dyn * p_wm + (1.0 - wm_weight_dyn) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # Reward-gated WM storage: move W_s toward chosen action if rewarded
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - lr) * w[s, :] + lr * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Asymmetric RL + Capacity-limited WM recall.
    - Policy: convex mixture of RL and WM softmax policies.
    - RL: separate learning rates for positive and negative prediction errors.
    - WM: acts like a cache updated by outcome; recall probability depends on K/nS.

    Parameters (tuple):
    - alpha_pos: RL learning rate for positive RPEs (0..1).
    - alpha_neg: RL learning rate for negative RPEs (0..1).
    - softmax_beta: inverse temperature for RL softmax; internally scaled x10.
    - wm_recall_base: baseline WM mixture weight (0..1).
    - K: WM capacity parameter (~ number of storable items; >0).
    - phi: capacity scaling exponent; effective WM weight scales as (K/nS)^phi.

    Set-size effect:
    - Effective WM weight per block: wm_recall_base * min(1, (K/nS)**phi). Larger nS reduces WM contribution.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    alpha_pos, alpha_neg, softmax_beta, wm_recall_base, K, phi = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-limited WM weight
        cap_ratio = (K / max(1.0, nS)) ** max(0.0, phi)
        wm_weight_eff = wm_recall_base * min(1.0, cap_ratio)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            alpha = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += alpha * pe

            # WM dynamics:
            # - Move toward uniform after negative feedback (unlearning)
            # - Move toward chosen action after positive feedback (storage)
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - alpha_pos) * w[s, :] + alpha_pos * onehot
            else:
                w[s, :] = (1.0 - alpha_neg) * w[s, :] + alpha_neg * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WSLS-like WM heuristic with set-size scaling and leak.
    - Policy: mixture of RL softmax and a WM softmax that implements win-stay / lose-shift.
    - WM trace: after reward, concentrate mass on the chosen action (stay);
                 after no reward, suppress chosen action and shift mass to the others.
      WM trace leaks toward uniform each visit.
    - WM mixture weight decreases with set size.

    Parameters (tuple):
    - lr: RL learning rate (0..1).
    - softmax_beta: RL inverse temperature; internally scaled x10.
    - wm_weight: baseline WM mixture weight at set size 3 (0..1).
    - ws_bias: strength of win-stay vs lose-shift shaping in WM (0..1).
               Higher means more deterministic WSLS shaping in WM values.
    - wm_leak: leak of WM toward uniform on each state visit (0..1).
    - ss_slope: slope reducing WM weight with larger set sizes (>=0).
                Effective weight = wm_weight / (1 + ss_slope*(nS-3)).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight, ws_bias, wm_leak, ss_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        wm_weight_eff = wm_weight / (1.0 + ss_slope * max(0, nS - 3))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # WSLS shaping:
            # - Win: push W_s toward one-hot of chosen action
            # - Loss: suppress chosen action, redistribute mass to other actions
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - ws_bias) * w[s, :] + ws_bias * onehot
            else:
                # Move probability mass away from chosen action toward others
                shift_vec = np.ones(nA) / (nA - 1)
                shift_vec[a] = 0.0
                # Normalize to sum to 1
                shift_vec /= shift_vec.sum()
                w[s, :] = (1.0 - ws_bias) * w[s, :] + ws_bias * shift_vec

        blocks_log_p += log_p

    return -blocks_log_p
