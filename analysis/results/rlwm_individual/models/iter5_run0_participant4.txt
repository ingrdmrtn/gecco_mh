def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with reward-gated writes, set-size-driven interference, and lapses.

    Policy:
    - RL: softmax over Q with softmax_beta (scaled internally by 10).
    - WM: softmax over W with high inverse temperature (softmax_beta_wm=50).
    - Mixture: wm_weight is reduced as set size increases via interference_k:
        wm_weight_eff = wm_weight_base / (1 + interference_k * (nS - 1))
    - Final policy has a lapse component (uniform random with prob=lapse).

    Learning:
    - RL: delta rule with learning rate lr.
    - WM: two steps each trial at current state:
        1) Set-size-driven interference/decay toward uniform:
           w[s,:] = (1 - decay)*w[s,:] + decay*w0[s,:], where
           decay = 1 - 1/(1 + interference_k*(nS-1))
        2) Reward-gated write: writes the chosen action with strength depending on reward:
           write_strength = wm_write_prob if reward==1 else (1 - wm_write_prob)

    Set-size effects:
    - WM reliance down-weighted via interference_k.
    - WM decay magnitude increases with set size via the same interference_k.

    Parameters (6):
    - lr: RL learning rate (0..1).
    - wm_weight_base: baseline WM mixture weight (0..1 before set-size scaling).
    - softmax_beta: RL inverse temperature; scaled by 10 internally.
    - wm_write_prob: probability/strength of WM write given reward=1
                     (for reward=0, write strength = 1 - wm_write_prob).
    - interference_k: controls how strongly set size reduces WM fidelity and weight.
    - lapse: lapse probability mixing in a uniform policy.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_write_prob, interference_k, lapse = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size-dependent weights/decay
        wm_weight_eff = np.clip(wm_weight_base / (1.0 + interference_k * max(0, nS - 1)), 0.0, 1.0)
        decay = 1.0 - 1.0 / (1.0 + interference_k * max(0, nS - 1))
        decay = np.clip(decay, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture + lapse
            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay/interference toward uniform
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # WM reward-gated write
            write_strength = wm_write_prob if r == 1 else (1.0 - wm_write_prob)
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            w[s, :] = (1.0 - write_strength) * w[s, :] + write_strength * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with PE-gated reliance and set-size-amplified WM decay.

    Policy:
    - RL: softmax over Q with softmax_beta (scaled internally).
    - WM: softmax over W with inverse temperature softmax_beta_wm = 50 * beta_wm_scale.
    - Mixture weight is trial-wise and depends on prediction error (PE):
        PE = |r - Q(s,a)|
        wm_weight_eff = wm_weight_base * exp(-pe_sensitivity * PE)
      (larger PE reduces reliance on WM).
    - No explicit lapse here; PE gate already provides flexibility.

    Learning:
    - RL: delta rule with learning rate lr.
    - WM: recency-based write (independent of reward), with set-size-amplified decay:
        dec_eff = 1 - (1 - wm_decay)^(nS/3), so larger set sizes increase forgetting.
        w[s,:] <- (1 - dec_eff) * w[s,:] + dec_eff * uniform
        then strong write to the chosen action:
        w[s,:] <- 0.0 * w[s,:] + 1.0 * one_hot(a)  (implemented as convex blend for stability)

    Set-size effects:
    - WM decay increases with set size via dec_eff.
    - The WM inverse temperature can vary via beta_wm_scale, modulating sharpness.

    Parameters (6):
    - lr: RL learning rate (0..1).
    - wm_weight_base: baseline WM mixture weight (0..1 before PE gating).
    - softmax_beta: RL inverse temperature; scaled by 10 internally.
    - wm_decay: base WM decay (0..1), amplified by set size as dec_eff.
    - pe_sensitivity: controls how strongly PE suppresses WM reliance (>=0).
    - beta_wm_scale: scales WM temperature: softmax_beta_wm = 50 * beta_wm_scale.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, pe_sensitivity, beta_wm_scale = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = max(1e-3, 50.0 * beta_wm_scale)
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size amplified decay
        dec_eff = 1.0 - (1.0 - np.clip(wm_decay, 0.0, 1.0)) ** (max(1.0, nS / 3.0))
        dec_eff = np.clip(dec_eff, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy probability
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # PE-gated WM mixture weight
            pe = abs(r - Q_s[a])
            wm_weight_eff = np.clip(wm_weight_base * np.exp(-np.clip(pe_sensitivity, 0.0, None) * pe), 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform then recency write
            w[s, :] = (1.0 - dec_eff) * w[s, :] + dec_eff * w_0[s, :]
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            # Strong recency write (but still a convex blend for numerical stability)
            write_strength = 1.0
            w[s, :] = (1.0 - write_strength) * w[s, :] + write_strength * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with state-swap errors in WM retrieval and a state-specific stickiness policy.

    Policy:
    - RL: softmax over Q with softmax_beta (scaled internally).
    - WM retrieval has swap errors that increase with set size:
        swap_eff = 1 - exp(-swap_gamma * (nS - 1))
        W_eff(s) = (1 - swap_eff)*w[s,:] + swap_eff*mean_{s'!=s} w[s',:]
      WM policy: softmax over W_eff(s) with softmax_beta_wm=50.
    - Mixture: p_mix = wm_weight * p_wm + (1 - wm_weight) * p_rl.
    - Stickiness: additional mixture with a state-specific "repeat last action" policy:
        p_sticky = 1 if a == last_action_in_state else 1/nA if no previous action else 0
        Final: p_total = (1 - stickiness)*p_mix + stickiness*p_sticky

    Learning:
    - RL: delta rule with learning rate lr.
    - WM: decay toward uniform with wm_decay, then reward-gated write (only on rewarded trials):
        if r==1: write chosen action with strength=1
        if r==0: only decay is applied (no write)

    Set-size effects:
    - Swap errors scale with set size through swap_gamma.

    Parameters (6):
    - lr: RL learning rate (0..1).
    - wm_weight: WM mixture weight (0..1).
    - softmax_beta: RL inverse temperature; scaled by 10 internally.
    - wm_decay: WM decay toward uniform each visit (0..1).
    - swap_gamma: controls how WM swap/interference increases with set size (>=0).
    - stickiness: mixture weight for the repeat-last-action policy (0..1).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, swap_gamma, stickiness = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        # Set-size-dependent swap error
        swap_eff = 1.0 - np.exp(-max(0.0, swap_gamma) * max(0, nS - 1))
        swap_eff = np.clip(swap_eff, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]

            # Effective WM with swap across other states
            if nS > 1:
                others = [i for i in range(nS) if i != s]
                mean_others = np.mean(w[others, :], axis=0)
            else:
                mean_others = w[s, :]
            W_eff = (1.0 - swap_eff) * w[s, :] + swap_eff * mean_others

            # RL and WM choice probabilities
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            p_mix = np.clip(wm_weight, 0.0, 1.0) * p_wm + (1.0 - np.clip(wm_weight, 0.0, 1.0)) * p_rl

            # Stickiness policy
            if last_action[s] == -1:
                p_sticky = 1.0 / nA
            else:
                p_sticky = 1.0 if a == last_action[s] else 0.0

            p_total = (1.0 - np.clip(stickiness, 0.0, 1.0)) * p_mix + np.clip(stickiness, 0.0, 1.0) * p_sticky
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - np.clip(wm_decay, 0.0, 1.0)) * w[s, :] + np.clip(wm_decay, 0.0, 1.0) * w_0[s, :]

            # WM reward-gated write (only on rewarded trials)
            if r == 1:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

            # Update last action for stickiness
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
