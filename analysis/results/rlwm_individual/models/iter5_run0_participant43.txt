def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with surprise-gated WM encoding and load-modulated forgetting.

    Idea:
    - RL learns action values via delta rule and softmax.
    - WM stores a one-shot policy per state when outcomes are surprising and rewarding.
    - Surprise is approximated from the current state's RL certainty: surprise â‰ˆ 1 - max(Q_s).
    - WM decay increases with load (larger set size => faster forgetting).
    - Arbitration between WM and RL is gated by surprise: more surprise => rely more on WM.

    Parameters:
    - lr: RL learning rate in [0,1].
    - wm_weight_base: Base weight for WM contribution in [0,1].
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - wm_forget: Baseline WM forgetting rate in [0,1]. Larger => faster decay toward uniform.
    - pe_sensitivity: Surprise sensitivity (>=0). Scales the sigmoid that maps surprise to WM arbitration and storage strength.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_forget, pe_sensitivity = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Effective forgetting grows with load
        wm_forget_eff = np.clip(wm_forget * (1.0 + load), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Surprise estimate from RL certainty; higher when RL is uncertain
            surprise = 1.0 - np.max(Q_s)
            # Map surprise to [0,1] with sigmoid
            attention = 1.0 / (1.0 + np.exp(-pe_sensitivity * surprise))

            # WM policy (deterministic over W_s)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: more surprising => rely more on WM
            wm_weight_eff = np.clip(wm_weight_base * attention, 0.0, 1.0)

            p_total = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update:
            # - On reward: encode chosen action with strength scaled by attention (surprise)
            # - On no reward: decay toward uniform
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                # Combine attention-gated encoding with load-modulated forgetting
                encode_strength = np.clip(attention * wm_forget_eff, 0.0, 1.0)
                w[s, :] = (1.0 - encode_strength) * w[s, :] + encode_strength * target
            else:
                w[s, :] = (1.0 - wm_forget_eff) * w[s, :] + wm_forget_eff * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + probabilistic WM slots with decision-level marginalization.

    Idea:
    - RL learns via delta rule and softmax.
    - WM capacity is limited to a fixed number of 'slots' shared across states in a block.
    - For any state, the probability it is in WM is p_in = min(1, slots / set_size).
    - Decision policy marginalizes over whether the current state is in WM:
        p_wm_eff = p_in * softmax(W_s) + (1 - p_in) * uniform.
    - WM updates: reward strengthens one-hot memory, no-reward decays to uniform.
    - WM reliability (both update strength and policy determinism) is controlled by noise_wm.

    Parameters:
    - lr: RL learning rate in [0,1].
    - wm_weight: Mixture weight for WM contribution in [0,1].
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - slots: Number of WM slots (>=0). Effective p_in = min(1, slots / set_size).
    - noise_wm: WM noise in [0,1]. Higher => weaker updates and more stochastic WM policy.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, slots, noise_wm = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    base_wm_beta = 50   # maximum determinism; will be scaled by noise
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Probability that current state is represented in WM
        p_in = min(1.0, max(0.0, float(slots) / max(1.0, float(nS))))
        # Effective WM temperature and update strengths modulated by noise
        softmax_beta_wm = base_wm_beta * max(0.0, 1.0 - noise_wm)
        update_strength = max(0.0, 1.0 - noise_wm)  # reward encoding strength
        decay_rate = min(1.0, max(0.0, noise_wm))   # no-reward decay toward uniform

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy if state is in WM
            p_wm_mem = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a]))) if softmax_beta_wm > 0 else 1.0 / nA
            # Marginalize over WM-inclusion
            p_wm_eff = p_in * p_wm_mem + (1.0 - p_in) * (1.0 / nA)

            # Mixture with RL
            p_total = p_wm_eff * wm_weight + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - update_strength) * w[s, :] + update_strength * target
            else:
                w[s, :] = (1.0 - decay_rate) * w[s, :] + decay_rate * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning + WM with load-driven interference (cross-talk).

    Idea:
    - RL has separate learning rates for positive vs negative outcomes.
    - WM encodes rewarded associations but suffers interference that increases with set size:
        after each update, WM rows are pulled toward the block-wise mean policy.
    - Arbitration is a fixed mixture between WM and RL.

    Parameters:
    - lr_pos: RL learning rate for positive prediction errors in [0,1].
    - lr_neg: RL learning rate for negative prediction errors in [0,1].
    - wm_weight: Mixture weight for WM contribution in [0,1].
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - interference_gain: Scales WM cross-talk with load (>=0). Effective interference = gain * load, where load=0 for set size 3 and 1 for 6.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, interference_gain = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6
        interference = np.clip(interference_gain * load, 0.0, 1.0)

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = p_wm * wm_weight + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetry
            delta = r - Q_s[a]
            alpha = lr_pos if delta >= 0 else lr_neg
            q[s][a] += alpha * delta

            # WM update: reward encodes, no-reward decays
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * target  # moderate consolidation
            else:
                w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]  # mild decay

            # Cross-talk interference toward mean policy across states (load-dependent)
            mean_policy = np.mean(w, axis=0, keepdims=True)  # shape (1, nA)
            w = (1.0 - interference) * w + interference * np.repeat(mean_policy, nS, axis=0)

        blocks_log_p += log_p

    return -blocks_log_p
