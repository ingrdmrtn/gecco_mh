def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Hybrid RL + WM with set-size-specific arbitration weights and WM decay.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate.
    - WM: maintains a distribution over actions per state; rewarded associations are stored as one-hot patterns.
           Memory traces passively decay toward a uniform prior each trial.
    - Arbitration: WM mixture weight depends on set size (different weights for set size 3 vs 6).

    Parameters:
    - lr: RL learning rate (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - wm_weight_ss3: WM weight in small set size blocks (0..1)
    - wm_weight_ss6: WM weight in large set size blocks (0..1)
    - wm_decay: per-trial WM decay toward uniform (0..1); higher = faster forgetting

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_ss3, wm_weight_ss6, wm_decay = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Pick set-size-specific WM arbitration weight
        wm_weight_block = wm_weight_ss3 if nS <= 3 else wm_weight_ss6

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy: softmax over WM weights (high precision)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = p_wm * wm_weight_block + (1.0 - wm_weight_block) * p_rl
            log_p += np.log(p_total + 1e-15)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Global decay toward uniform prior
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Reward-gated storage: on reward, store a one-hot binding for the chosen action
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Hybrid RL + WM with set-size-dependent RL learning rate and WM recency buffer (win-stay) with exponential decay.

    Mechanism:
    - RL: tabular Q-learning; learning rate differs between small and large set sizes.
    - WM: stores the most recently rewarded action per state; memory decays exponentially each trial.
          Decay is faster for larger set sizes (scaled by nS/3).
    - Arbitration: fixed WM mixture weight across blocks.

    Parameters:
    - lr_small: RL learning rate for small set size blocks (0..1)
    - lr_large: RL learning rate for large set size blocks (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - wm_weight: mixture weight for WM in choice (0..1)
    - tau: WM time constant; effective per-trial decay = 1 - exp(-1/tau) scaled by nS/3

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_small, lr_large, softmax_beta, wm_weight, tau = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    # Convert tau to a base per-trial decay (0..1); protect against zero/negative tau
    tau = max(1e-6, tau)

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent RL learning rate
        lr = lr_small if nS <= 3 else lr_large

        # Set-size dependent WM decay: faster with larger nS
        base_decay = 1.0 - np.exp(-1.0 / tau)  # in (0,1)
        wm_decay = np.clip(base_decay * (nS / 3.0), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy reflects the current decayed "recent-reward" trace
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            log_p += np.log(p_total + 1e-15)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Exponential decay toward uniform prior
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Win-stay storage: only store when rewarded
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Hybrid RL + WM with entropy-based arbitration, set-size penalized WM reliability, and choice perseveration.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate.
    - WM: maintains action-probability traces, updated by reward; traces decay each trial.
    - Arbitration: dynamic WM weight proportional to the certainty (low entropy) of WM, and penalized by set size.
    - Policy extras: lapse probability mixing in uniform random choice; perseveration bias in RL toward the last chosen action per state.

    Parameters:
    - lr: RL learning rate (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - wm_strength: overall scaling of WM contribution to arbitration (0..1)
    - gamma: load penalty on WM reliability with set size; higher gamma reduces WM influence as nS increases (>=0)
    - epsilon: lapse probability mixing uniform choice into the final policy (0..1)
    - kappa: perseveration strength added to the last chosen action in each state (can be >=0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_strength, gamma, epsilon, kappa = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last chosen action per state for perseveration bias; -1 indicates none yet
        last_choice = -1 * np.ones(nS, dtype=int)

        # WM decay: modest baseline, scaled with set size to emulate interference
        base_decay = 0.1  # implicit baseline
        wm_decay = np.clip(base_decay * (1.0 + gamma * max(0, nS - 3)), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()

            # Add perseveration bias to the last chosen action for this state
            if last_choice[s] >= 0:
                Q_s[last_choice[s]] += kappa

            W_s = w[s, :]

            # RL softmax probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy via softmax of WM weights
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Entropy-based dynamic WM weight (0..1), penalized by set size
            # Compute normalized entropy of W_s
            H = -np.sum(W_s * np.log(W_s + 1e-15))
            H_norm = H / np.log(nA)  # in [0,1]
            wm_reliability = 1.0 - H_norm  # higher when WM is peaked
            load_penalty = 1.0 / (1.0 + gamma * max(0, nS - 3))
            wm_weight = np.clip(wm_strength * wm_reliability * load_penalty, 0.0, 1.0)

            # Mix WM and RL
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl

            # Apply lapse to final choice probability
            p_total = (1.0 - epsilon) * p_mix + epsilon * (1.0 / nA)
            log_p += np.log(p_total + 1e-15)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Reward-gated sharpening: increase focus on rewarded action; if no reward, leave as decayed
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Update perseveration memory
            last_choice[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
