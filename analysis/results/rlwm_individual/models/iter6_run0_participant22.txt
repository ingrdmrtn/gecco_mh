def cognitive_model1(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL + WM with uncertainty-based arbitration and within-state action stickiness.

    Idea
    - Choices are a mixture of RL and WM.
    - Arbitration dynamically increases WM contribution when RL is uncertain (flat policy).
    - WM retrieval is precision-limited by set size; also includes a within-state action stickiness bias.

    Parameters
    ----------
    parameters : tuple/list of length 5
        lr : float in [0,1]
            RL learning rate.
        wm_base : float in [0,1]
            Baseline WM mixture weight.
        softmax_beta : float >= 0
            RL inverse temperature (internally scaled by 10).
        arb_slope : float (can be negative or positive)
            Slope mapping RL uncertainty (1 - max RL choice prob) into additional WM weight.
            Larger values increase WM reliance when RL is uncertain.
        stickiness : float >= 0
            Adds a bias in WM toward repeating the last action taken in the same state.

    Set-size impact
    ---------------
    - WM precision decreases with set size: beta_wm_eff = 50 * (3 / nS).
    - Arbitration does not directly depend on set size, but WM precision does, which affects p_wm.
    """
    import numpy as np

    lr, wm_base, softmax_beta, arb_slope, stickiness = parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        last_choice = -1 * np.ones(nS, dtype=int)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (as given)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Working-memory precision scales with set size
            beta_wm_eff = softmax_beta_wm * (3.0 / max(1.0, float(nS)))

            # Add within-state action stickiness as an additive utility bias in WM
            wm_util = W_s.copy()
            if last_choice[s] >= 0:
                wm_util[last_choice[s]] += stickiness

            # Softmax probability of chosen action under WM with set-size–scaled precision
            p_wm_soft = 1.0 / np.sum(np.exp(beta_wm_eff * (wm_util - wm_util[a])))

            # Arbitration: increase WM weight when RL is uncertain (1 - max RL prob)
            # Compute RL softmax over all actions (for uncertainty)
            prl_all = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl_all /= np.sum(prl_all)
            rl_uncertainty = 1.0 - np.max(prl_all)

            wm_weight_eff = wm_base + arb_slope * rl_uncertainty
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)

            p_wm = p_wm_soft
            p_total = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: decay toward uniform then store the chosen action with reward-scaled bump
            decay = lr * (3.0 / max(1.0, float(nS)))  # slightly slower decay for larger sets
            decay = min(max(decay, 0.0), 1.0)
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]
            w[s, a] += wm_base * (0.5 + 0.5 * r)

            # Update last choice for stickiness
            last_choice[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL with set-size–dependent forgetting + WM with capacity-sensitivity scaling.

    Idea
    - RL values decay toward uniform on each visit (forgetting), with a user-controlled rate.
      This makes RL less stable under high load if decay is strong relative to learning.
    - WM mixture weight is scaled by set size via a power-law: eff_wm = wm_weight * (3/nS)^size_sensitivity.
    - WM precision decreases with set size; WM values decay toward uniform at a rate tied to RL decay.

    Parameters
    ----------
    parameters : tuple/list of length 5
        lr : float in [0,1]
            RL learning rate.
        wm_weight : float in [0,1]
            Baseline WM mixture weight before set-size scaling.
        softmax_beta : float >= 0
            RL inverse temperature (internally scaled by 10).
        rl_decay : float in [0,1]
            Forgetting rate for RL towards uniform on each state visit.
        size_sensitivity : float >= 0
            Scales WM mixture with set size as (3/nS)^size_sensitivity. Larger => more drop in WM for nS=6.

    Set-size impact
    ---------------
    - WM mixture: eff_wm = wm_weight * (3/nS)^size_sensitivity.
    - WM precision: beta_wm_eff = 50 * (3 / nS).
    - WM decay increases with set size via coupling to rl_decay: wm_decay = rl_decay * (nS/3).
    """
    import numpy as np

    lr, wm_weight, softmax_beta, rl_decay, size_sensitivity = parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size–scaled WM mixture and precision
        size_scale = (3.0 / max(1.0, float(nS))) ** max(0.0, float(size_sensitivity))
        wm_weight_eff_block = wm_weight * size_scale
        wm_weight_eff_block = min(max(wm_weight_eff_block, 0.0), 1.0)
        beta_wm_eff = softmax_beta_wm * (3.0 / max(1.0, float(nS)))

        # WM decay coupled to RL decay and set size
        wm_decay_base = rl_decay * (float(nS) / 3.0)
        wm_decay_base = min(max(wm_decay_base, 0.0), 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm_soft = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = p_wm_soft

            # Mixture
            p_total = p_wm * wm_weight_eff_block + (1 - wm_weight_eff_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL forgetting toward uniform then TD update
            q[s, :] = (1.0 - rl_decay) * q[s, :] + rl_decay * (1.0 / nA)
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform then store chosen action with reward-scaled boost
            w[s, :] = (1.0 - wm_decay_base) * w[s, :] + wm_decay_base * w_0[s, :]
            w[s, a] += wm_weight * (0.5 + 0.5 * r)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL + WM with item-interference and set-size–dependent lapse.

    Idea
    - WM retrieval suffers from interference between memoranda; at retrieval, the current state's WM
      vector is blended with the average WM vector of other states. Interference increases with set size.
    - WM also exhibits set-size–dependent lapses (epsilon-greedy to uniform).
    - RL is standard delta-rule.

    Parameters
    ----------
    parameters : tuple/list of length 6
        lr : float in [0,1]
            RL learning rate.
        wm_weight : float in [0,1]
            Mixture weight for WM.
        softmax_beta : float >= 0
            RL inverse temperature (internally scaled by 10).
        rho_interf : float in [0,1]
            Base WM interference strength; scales with set size for retrieval blending.
        eta_lapse : float >= 0
            Base WM lapse rate multiplier; effective lapse scales with set size.
        wm_boost : float >= 0
            Magnitude of WM storage boost applied to the chosen action (scaled by reward).

    Set-size impact
    ---------------
    - WM precision decreases with set size: beta_wm_eff = 50 * (3 / nS).
    - Interference increases with set size: rho_eff = rho_interf * max(0, (nS-3)/3), capped at 1.
    - Lapse increases with set size: epsilon = eta_lapse * max(0, (nS-3)/nS), capped at 1.
    """
    import numpy as np

    lr, wm_weight, softmax_beta, rho_interf, eta_lapse, wm_boost = parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        beta_wm_eff = softmax_beta_wm * (3.0 / max(1.0, float(nS)))
        rho_eff = min(max(rho_interf * max(0.0, (float(nS) - 3.0) / 3.0), 0.0), 1.0)
        epsilon = min(max(eta_lapse * max(0.0, (float(nS) - 3.0) / max(1.0, float(nS))), 0.0), 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM retrieval with interference: blend current state's WM with average of others
            if nS > 1:
                mean_others = (np.sum(w, axis=0) - w[s, :]) / max(1, (nS - 1))
                W_s_eff = (1.0 - rho_eff) * w[s, :] + rho_eff * mean_others
            else:
                W_s_eff = w[s, :].copy()

            # Softmax WM, then add set-size–dependent lapse to uniform
            p_wm_soft = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s_eff - W_s_eff[a])))
            p_wm = (1.0 - epsilon) * p_wm_soft + epsilon * (1.0 / nA)

            # Mixture
            p_total = p_wm * wm_weight + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: decay then reward-scaled boost to the chosen action
            decay = lr * (1.0 + rho_eff) * 0.5  # mild coupling of decay to interference
            decay = min(max(decay, 0.0), 1.0)
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]
            w[s, a] += wm_boost * (0.5 + 0.5 * r)

        blocks_log_p += log_p

    return -blocks_log_p
