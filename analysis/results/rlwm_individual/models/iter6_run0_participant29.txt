def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Surprise-gated dual-rate RL with load-suppressed WM contribution.

    Idea:
    - RL updates use a dynamic learning rate that interpolates between slow and fast
      based on momentary surprise (|RPE|). Larger absolute RPE -> faster learning.
    - WM stores quickly learned action weights and decays to uniform otherwise.
    - The policy is a mixture of a WM softmax and an RL softmax.
    - Under higher set size, the WM contribution is suppressed (reduced mixture weight).

    Parameters (model_parameters):
    - lr_slow:       Slow RL learning rate (0..1) used under low surprise
    - lr_fast:       Fast RL learning rate (0..1) used under high surprise
    - wm_weight_base:Base WM mixture weight at set size 3 (0..1)
    - softmax_beta:  Base inverse temperature for RL (scaled internally by *10)
    - wm_eta:        WM write strength toward one-hot when rewarded (0..1)
    - wm_load_beta:  Load penalty scaling for WM; wm_weight_eff = wm_weight_base / (1 + wm_load_beta*(nS-3))

    Set-size impact:
    - WM mixture weight decreases with set size via wm_load_beta.
    """
    lr_slow, lr_fast, wm_weight_base, softmax_beta, wm_eta, wm_load_beta = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Load-suppressed WM contribution
        wm_weight = wm_weight_base / (1.0 + max(0.0, float(wm_load_beta) * float(nS - 3)))
        wm_weight = min(max(wm_weight, 0.0), 1.0)

        # Initialize RL and WM value maps
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with surprise-gated learning rate
            delta = r - q[s, a]
            # map |delta| in [0,1] to gate between lr_slow and lr_fast
            gate = min(max(abs(delta), 0.0), 1.0)
            lr_eff = lr_slow + gate * (lr_fast - lr_slow)
            q[s, a] += lr_eff * delta

            # WM update: decay to uniform; one-hot write only on reward
            # decays mildly each visit toward uniform
            decay = 0.1  # small fixed decay to avoid unused parameter inflation
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_eta) * w[s, :] + wm_eta * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with perseveration bias and load-dependent exploration; WM implements win-stay/lose-shift.

    Idea:
    - RL softmax includes a state-specific perseveration bonus for the last chosen action.
    - Inverse temperature (exploration) decreases with set size (more load -> more exploration).
    - WM realizes a win-stay/lose-shift heuristic: after reward, push weight to chosen action;
      after no reward, push weight away from chosen toward the unchosen actions.
    - Policy is a mixture of WM and (biased) RL.

    Parameters (model_parameters):
    - lr:              RL learning rate (0..1)
    - wm_weight:       WM mixture weight (0..1)
    - softmax_beta:    Base RL inverse temperature (scaled internally by *10)
    - beta_load_shrink:Exploration increase with load; beta_eff = softmax_beta / (1 + beta_load_shrink*(nS-3))
    - perseveration:   Bonus added to the last action in this state in the RL softmax (>=0)
    - wm_decay:        WM decay toward uniform on each visit (0..1)

    Set-size impact:
    - RL inverse temperature shrinks with set size via beta_load_shrink.
    """
    lr, wm_weight, softmax_beta, beta_load_shrink, perseveration, wm_decay = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Load-dependent RL temperature
        beta_eff = softmax_beta / (1.0 + max(0.0, float(beta_load_shrink) * float(nS - 3)))

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last action per state for perseveration
        last_action = -np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with perseveration bias
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                bias_vec = np.zeros_like(Q_s)
                bias_vec[last_action[s]] = perseveration
                Q_s = Q_s + bias_vec

            p_rl = 1.0 / np.sum(np.exp(beta_eff * (Q_s - Q_s[a])))

            # WM policy: win-stay/lose-shift softmax from w
            W_s = w[s, :].copy()
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM win-stay/lose-shift update
            if r > 0.5:
                # increase chosen
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta_win = 0.6
                w[s, :] = (1.0 - eta_win) * w[s, :] + eta_win * one_hot
            else:
                # shift away from chosen toward others
                eta_lose = 0.3
                anti = np.ones(nA) / (nA - 1.0)
                anti[a] = 0.0
                anti = anti / anti.sum()
                w[s, :] = (1.0 - eta_lose) * w[s, :] + eta_lose * anti

            # Update perseveration memory
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning rates and capacity-limited WM (K slots).

    Idea:
    - RL uses separate learning rates for positive and negative outcomes.
    - WM stores mappings only for up to K states (capacity limit). When a rewarded
      mapping is encountered, that state is inserted/updated; if capacity exceeded,
      the least-recently-rewarded state is evicted. WM entries decay toward uniform.
    - Policy is a mixture of WM and RL. When a state is not in WM, its WM distribution
      is near-uniform and thus contributes little to choice.

    Parameters (model_parameters):
    - lr_pos:     RL learning rate for rewards (0..1)
    - lr_neg:     RL learning rate for no-reward (0..1)
    - wm_weight:  WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by *10)
    - K_cap:      WM capacity (number of states maintainable; real-valued, internally rounded)
    - wm_forget:  WM decay/forgetting rate toward uniform per visit (0..1)

    Set-size impact:
    - When set size nS > K_cap, some states cannot be maintained in WM, effectively reducing
      WM guidance under higher load.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, K_cap, wm_forget = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Round capacity to a feasible integer between 0 and nS
        K = int(np.clip(np.round(K_cap), 0, nS))

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Maintain an ordered list of states currently in WM (most recent at the end)
        wm_list = []

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: if s in capacity, use w[s]; else near-uniform
            if s in wm_list and K > 0:
                W_s = w[s, :]
            else:
                # Uniform prior if not actively maintained
                W_s = w_0[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            delta = r - q[s, a]
            lr_eff = lr_pos if delta > 0.0 else lr_neg
            q[s, a] += lr_eff * delta

            # WM decay for maintained states only
            if s in wm_list and K > 0:
                w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]

            # WM write and capacity management only on reward
            if r > 0.5 and K > 0:
                # Insert/update state s into WM list (move to end as most recent)
                if s in wm_list:
                    wm_list.remove(s)
                wm_list.append(s)
                # Enforce capacity
                while len(wm_list) > K:
                    evict = wm_list.pop(0)
                    # Optionally reset evicted slot closer to uniform
                    w[evict, :] = (1.0 - wm_forget) * w[evict, :] + wm_forget * w_0[evict, :]

                # Strong one-hot write for the rewarded mapping
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta_write = 0.8
                w[s, :] = (1.0 - eta_write) * w[s, :] + eta_write * one_hot

        blocks_log_p += log_p

    return -blocks_log_p
