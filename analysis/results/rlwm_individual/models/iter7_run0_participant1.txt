def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with overload lapses and forgetting.

    Idea:
    - Choices are generated by a mixture of a slow RL policy and a capacity-limited WM policy.
    - WM is near-deterministic when set size nS is within capacity K_capacity, but becomes noisy
      (lapses to uniform) when nS exceeds capacity. WM also forgets unrewarded pairs toward uniform.
    - Arbitration weight of WM is reduced by overload (nS > K_capacity).

    Parameters:
    - model_parameters[0]: lr (float in [0,1]) RL learning rate.
    - model_parameters[1]: wm_weight_base (float in [0,1]) Baseline weight on WM policy.
    - model_parameters[2]: softmax_beta (float >= 0) RL inverse temperature (internally scaled by 10).
    - model_parameters[3]: K_capacity (float >= 1) WM capacity (in number of items).
    - model_parameters[4]: wm_lapse (float >= 0) Lapse growth rate under overload; larger -> more uniform WM.
    - model_parameters[5]: wm_forget (float in [0,1]) Forgetting rate toward uniform after no reward.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight_base, softmax_beta, K_capacity, wm_lapse, wm_forget = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute overload-driven terms once per block
        overload = max(0.0, float(nS - K_capacity))
        # Lapse increases with overload (0 when within capacity)
        lapse = 1.0 - np.exp(-wm_lapse * overload)
        lapse = float(np.clip(lapse, 0.0, 1.0))
        # WM arbitration weight is reduced when overloaded
        capacity_scale = float(np.clip(K_capacity / max(K_capacity, nS), 0.0, 1.0))
        wm_weight_block = float(np.clip(wm_weight_base * capacity_scale, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (given by template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WORKING MEMORY POLICY: near-deterministic, but with overload-induced lapse to uniform
            p_wm_clean = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = (1.0 - lapse) * p_wm_clean + lapse * (1.0 / nA)

            # Arbitration
            wm_weight = wm_weight_block
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = float(np.clip(p_total, 1e-12, 1.0))
            log_p += np.log(p_total)

            # RL update (template)
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM UPDATE:
            # - Reward: store strong association (one-hot attractor)
            # - No reward: forget toward uniform at rate wm_forget
            if r > 0.5:
                # Move strongly toward one-hot pattern
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps
            else:
                w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + PE-gated WM encoding with load-dependent arbitration and WM decay.

    Idea:
    - WM encoding is gated by the magnitude of the RL prediction error (PE).
      Larger surprising rewards lead to stronger WM storage of the chosen action.
    - WM forgets toward uniform when no reward is observed.
    - Arbitration (WM weight) decreases with load (set size): wm_weight = wm_weight_base / (1 + load_slope*(nS-1)).
    - RL policy is from the template (softmax over Q); WM policy is a near-deterministic softmax over W.

    Parameters:
    - model_parameters[0]: lr (float in [0,1]) RL learning rate.
    - model_parameters[1]: wm_weight_base (float in [0,1]) Baseline WM arbitration weight.
    - model_parameters[2]: softmax_beta (float >= 0) RL inverse temperature (internally scaled by 10).
    - model_parameters[3]: enc_sensitivity (float >= 0) Sensitivity of WM encoding strength to |PE|.
    - model_parameters[4]: decay_wm (float in [0,1]) WM decay rate toward uniform after no reward.
    - model_parameters[5]: load_slope (float >= 0) Load penalty slope reducing WM arbitration with set size.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight_base, softmax_beta, enc_sensitivity, decay_wm, load_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent arbitration penalty
        wm_weight_block = wm_weight_base / (1.0 + load_slope * max(0.0, nS - 1.0))
        wm_weight_block = float(np.clip(wm_weight_block, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (near-deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration
            wm_weight = wm_weight_block
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = float(np.clip(p_total, 1e-12, 1.0))
            log_p += np.log(p_total)

            # Compute PE before RL update for WM gating
            pe = r - Q_s[a]

            # RL update (template)
            q[s][a] += lr * pe

            # WM UPDATE:
            # - Reward: encode toward one-hot with strength gated by |PE|
            # - No reward: decay toward uniform
            if r > 0.5:
                # Gate strength via sigmoid of |PE|
                g = 1.0 / (1.0 + np.exp(-enc_sensitivity * abs(pe)))
                # Move row toward one-hot on chosen action
                target = np.full(nA, 1e-6)
                target[a] = 1.0 - (nA - 1) * 1e-6
                w[s, :] = (1.0 - g) * w[s, :] + g * target
            else:
                w[s, :] = (1.0 - decay_wm) * w[s, :] + decay_wm * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with cross-state interference and volatility-weighted arbitration.

    Idea:
    - WM suffers cross-state interference: when a rewarded association is stored for state s and action a,
      some probability mass "leaks" to the same action a in other states (confusions), mimicking load-induced interference.
    - WM policy is corrupted by load-dependent noise (lapse to uniform with increasing set size).
    - Arbitration weight for WM is reduced when local volatility (variance of recent PE for the current state) is high.

    Parameters:
    - model_parameters[0]: lr (float in [0,1]) RL learning rate.
    - model_parameters[1]: wm_weight_base (float in [0,1]) Baseline WM arbitration weight.
    - model_parameters[2]: softmax_beta (float >= 0) RL inverse temperature (internally scaled by 10).
    - model_parameters[3]: wm_cross (float in [0,1]) Cross-state interference strength on reward.
    - model_parameters[4]: wm_load_noise (float >= 0) Load noise rate for WM policy (larger -> more lapse with nS).
    - model_parameters[5]: arb_volatility (float in (0,1]) EWMA rate for volatility tracking used to down-weight WM.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight_base, softmax_beta, wm_cross, wm_load_noise, arb_volatility = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Volatility trackers per state: mean and variance proxy of PE (EWMA)
        m_pe = np.zeros(nS)
        v_pe = np.zeros(nS)

        # Load-driven WM lapse
        noise = 1.0 - np.exp(-wm_load_noise * max(1, nS))
        noise = float(np.clip(noise, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with load lapse
            p_wm_clean = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = (1.0 - noise) * p_wm_clean + noise * (1.0 / nA)

            # Arbitration weight reduced by local volatility for this state
            # Volatility proxy: v_pe[s] (EWMA of squared deviation from mean)
            vol_factor = np.exp(-v_pe[s])  # higher volatility -> smaller vol_factor
            wm_weight = float(np.clip(wm_weight_base * vol_factor, 0.0, 1.0))

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = float(np.clip(p_total, 1e-12, 1.0))
            log_p += np.log(p_total)

            # Compute PE before RL update (for volatility update)
            pe = r - Q_s[a]

            # RL update (template)
            q[s][a] += lr * pe

            # Update volatility trackers for the current state
            beta_v = float(np.clip(arb_volatility, 1e-6, 1.0))
            m_pe[s] = (1.0 - beta_v) * m_pe[s] + beta_v * pe
            v_pe[s] = (1.0 - beta_v) * v_pe[s] + beta_v * (pe - m_pe[s]) ** 2

            # WM UPDATE with cross-state interference on reward; otherwise small decay
            if r > 0.5:
                # Set current state's row strongly to one-hot on a
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps

                # Cross-state leakage of chosen action probability to other states
                if nS > 1 and wm_cross > 0.0:
                    leak = float(np.clip(wm_cross, 0.0, 1.0))
                    per_other = leak / max(1, nS - 1)
                    for s_other in range(nS):
                        if s_other == s:
                            continue
                        # Move w[s_other, a] slightly up and renormalize row
                        w[s_other, a] = (1.0 - per_other) * w[s_other, a] + per_other
                        # Renormalize row to sum to 1 by proportionally shrinking others
                        row_sum = np.sum(w[s_other, :])
                        if row_sum > 0:
                            w[s_other, :] = w[s_other, :] / row_sum
            else:
                relax = 0.05
                w[s, :] = (1.0 - relax) * w[s, :] + relax * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
