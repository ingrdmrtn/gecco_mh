def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with entropy-based arbitration and load-dependent WM decay.

    Overview:
    - RL: tabular Q-learning with learning rate lr and softmax policy (beta scaled by 10).
    - WM policy: softmax over WM weights W with high inverse temperature (near-deterministic).
    - Arbitration: the effective WM mixture weight increases as RL policy becomes less certain
      (higher entropy). This arbitration also respects a base WM weight.
    - WM update: decay toward uniform that increases with set size; supervised update toward a
      reward-dependent target using wm_alpha.

    Set-size impact:
    - WM decay increases with set size.
    - Arbitration favors WM more when RL is uncertain; set size affects decay which indirectly
      reduces WM certainty at larger set size.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight_base: float in [0,1], base mixture weight of WM in the action policy.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - wm_alpha: float in [0,1], strength of WM supervised update toward target.
    - wm_decay_base: float in [0,1], base WM decay toward uniform per trial (amplified by set size).
    - arb_entropy_slope: float >= 0, slope for increasing WM reliance as RL entropy increases.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_alpha, wm_decay_base, arb_entropy_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load factor (0 at set size 3, 1 at set size 6)
        load = max(0.0, (float(nS) - 3.0) / 3.0)
        # Load-amplified decay
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) ** (1.0 + load)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            # RL choice probability of chosen action a (softmax trick with baseline Q[a])
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # Compute RL policy distribution to get entropy (with numeric stability)
            logits = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(logits)
            pi_rl /= np.sum(pi_rl)
            # Entropy of RL policy
            H = -np.sum(pi_rl * np.log(np.clip(pi_rl, 1e-12, 1.0)))
            H_max = np.log(nA)
            # Normalize entropy [0,1]
            H_norm = H / max(H_max, 1e-12)
            # Dynamic WM weight via entropy arbitration
            wm_weight_dyn = 1.0 / (1.0 + np.exp(-(np.log(np.clip(wm_weight_base, 1e-6, 1-1e-6)) - np.log(1-np.clip(wm_weight_base,1e-6,1-1e-6)) + arb_entropy_slope*(H_norm - 0.5))))
            wm_weight_dyn = np.clip(wm_weight_dyn, 0.0, 1.0)

            # WM policy: softmax over WM weights
            W_s = w[s,:]
            p_wm = np.exp(softmax_beta_wm*(W_s[a]-np.max(W_s))) / np.sum(np.exp(softmax_beta_wm*(W_s-np.max(W_s))))

            # Mixture
            p_total = p_wm*wm_weight_dyn + (1-wm_weight_dyn)*p_rl
            log_p += np.log(max(p_total, 1e-12))
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # WM supervised update toward target: if rewarded, target is one-hot on a;
            # if not rewarded, target is uniform (encodes uncertainty)
            if r > 0.0:
                target = np.zeros(nA)
                target[a] = 1.0
            else:
                target = w_0[s, :].copy()
            w[s, :] = (1.0 - wm_alpha) * w[s, :] + wm_alpha * target

            # Ensure normalization
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with interference-driven WM degradation and load-modulated recall.

    Overview:
    - RL: tabular Q-learning with learning rate lr and softmax policy (beta scaled by 10).
    - WM policy: softmax over WM with high inverse temperature; overall recall mixes with uniform
      based on set-size-dependent recall probability.
    - WM update: decay toward uniform plus rewarded consolidation; additionally, cross-item
      interference pushes other items (other states) toward uniform after each rewarded update.

    Set-size impact:
    - Recall probability decreases with load.
    - Interference strength onto other states increases with load.
    - WM decay increases with set size.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], mixture weight of WM in the action policy.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - wm_interference: float in [0,1], strength of cross-item interference after a rewarded update.
    - wm_decay_base: float in [0,1], base WM decay toward uniform per trial (amplified by set size).
    - load_penalty: float in [0,1], linear penalty on WM recall probability with load.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_interference, wm_decay_base, load_penalty = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        load = max(0.0, (float(nS) - 3.0) / 3.0)
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) ** (1.0 + load)
        # Load-modulated recall
        p_recall_base = np.clip(1.0 - load_penalty * load, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            W_s = w[s,:]
            # WM softmax for chosen action
            p_wm_soft = np.exp(softmax_beta_wm*(W_s[a]-np.max(W_s))) / np.sum(np.exp(softmax_beta_wm*(W_s-np.max(W_s))))
            # Mix with uniform by recall probability
            p_wm = p_recall_base * p_wm_soft + (1.0 - p_recall_base) * (1.0 / nA)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(p_total, 1e-12))
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            if r > 0.0:
                # Rewarded consolidation on current state
                w[s, :] *= (1.0 - wm_interference)
                w[s, a] += wm_interference
                # Normalize
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :].copy()

                # Cross-item interference: push other states toward uniform
                gamma_int = wm_interference * load
                if gamma_int > 0:
                    for s2 in range(nS):
                        if s2 == s:
                            continue
                        w[s2, :] = (1.0 - gamma_int) * w[s2, :] + gamma_int * w_0[s2, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with perseveration bias modulated by surprise and load.

    Overview:
    - RL: tabular Q-learning with learning rate lr. The RL policy includes a perseveration bias
      toward the last chosen action in each state. Bias magnitude increases with load and decreases
      with surprise (|r - Q[a]|).
    - WM policy: softmax over WM with high inverse temperature.
    - Arbitration: WM weight is reduced under higher load.

    Set-size impact:
    - WM mixture weight decreases with load.
    - Perseveration increases with load.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], base mixture weight of WM in the action policy (at small load).
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - persev_base: float >= 0, base perseveration strength.
    - surprise_gain: float >= 0, how much surprise reduces perseveration.
    - load_slope: float >= 0, increases perseveration and decreases WM reliance with load.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, persev_base, surprise_gain, load_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Track last chosen action per state for perseveration
        last_choice = -1 * np.ones(nS, dtype=int)

        load = max(0.0, (float(nS) - 3.0) / 3.0)
        # Effective WM weight decreases with load
        wm_weight_eff = np.clip(wm_weight * (1.0 - load_slope * load), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]

            # Surprise computed from current Q
            surprise = abs(r - Q_s[a])

            # Perseveration magnitude: base increased by load, reduced by surprise
            kappa = max(0.0, persev_base * (1.0 + load_slope * load) * (1.0 - surprise_gain * surprise))
            # Build bias vector for RL policy
            bias = np.zeros(nA)
            if last_choice[s] >= 0:
                bias[last_choice[s]] = kappa

            # RL policy with perseveration bias
            Q_bias = Q_s + bias
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_bias-Q_bias[a])))

            # WM policy: softmax over WM weights
            W_s = w[s,:]
            p_wm = np.exp(softmax_beta_wm*(W_s[a]-np.max(W_s))) / np.sum(np.exp(softmax_beta_wm*(W_s-np.max(W_s))))

            # Mixture
            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            log_p += np.log(max(p_total, 1e-12))
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decay toward uniform
            wm_decay_eff = 1.0 - (1.0 - 0.2) ** (1.0 + load)  # implicit moderate base decay; tied to load via load_slope indirectly through behavior
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # WM reward-gated update: strengthen chosen if rewarded, slight avoidance if not
            if r > 0.0:
                alpha_pos = 0.6
                w[s, :] *= (1.0 - alpha_pos)
                w[s, a] += alpha_pos
            else:
                alpha_neg = 0.2
                decrement = min(alpha_neg, w[s, a])
                w[s, a] -= decrement
                w[s, :] /= np.sum(w[s, :])

            # Normalize
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

            # Update last choice
            last_choice[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
