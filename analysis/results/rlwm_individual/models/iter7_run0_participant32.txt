def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with rehearsal and state-wise choice stickiness; WM suffers set-size–dependent decay.
    - RL: standard delta rule with softmax policy (beta scaled by 10).
    - WM: associative weights per state updated by rehearsal and decay; rewarded actions are reinforced.
    - Stickiness: WM policy is biased toward the last action taken in that state (perseveration).
    - Set size effect: WM decay is stronger in larger set sizes.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1).
    - wm_weight: Mixture weight of WM in the action policy (0-1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_rehearsal: Strength by which the chosen action is rehearsed in WM on each visit (>=0).
    - wm_decay: Base decay of WM toward uniform on each trial (0-1); scaled up by set size.
    - stickiness: Bias added to the WM logits for the previously chosen action in the same state (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_rehearsal, wm_decay, stickiness = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Track last action taken in each state for stickiness
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action under softmax)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy:
            # - Add stickiness bias toward last action in this state
            wm_logits = W_s.copy()
            if last_action[s] >= 0:
                bias_vec = np.zeros(nA)
                bias_vec[last_action[s]] = stickiness
                wm_logits = wm_logits + bias_vec

            # Convert logits to probabilities via softmax-like choice-prob computation for chosen action
            p_wm_soft = 1.0 / np.sum(np.exp(softmax_beta_wm * (wm_logits - wm_logits[a])))

            # Mixture
            p_total = p_wm_soft * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # - Set-size–dependent decay toward uniform for all states each trial
            #   Effective decay increases with set size.
            decay_eff = min(1.0, wm_decay * (float(nS) / 3.0))
            w = (1.0 - decay_eff) * w + decay_eff * w_0

            # - Rehearsal on the visited state: push toward chosen action
            onehot = np.zeros(nA)
            onehot[a] = 1.0
            w[s, :] = (1.0 - wm_rehearsal) * w[s, :] + wm_rehearsal * onehot

            # - Reward bonus: if rewarded, sharpen the memory toward chosen action
            if r > 0:
                w[s, :] = 0.5 * w[s, :] + 0.5 * onehot

            # Normalize to avoid drift
            w[s, :] = np.clip(w[s, :], eps, None)
            w[s, :] /= np.sum(w[s, :])

            # Update last action
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + reward-modulated WM with set-size–dependent WM lapse.
    - RL: standard delta rule with softmax policy (beta scaled by 10).
    - WM: Hebbian update toward the chosen action only when rewarded; otherwise slight relaxation.
    - WM policy lapse: with probability lapse(nS), WM outputs a uniform choice; otherwise uses WM softmax.
      Lapse increases with set size via a logistic transform.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1).
    - wm_weight: Mixture weight of WM in the action policy (0-1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_alpha: Hebbian WM learning rate toward the chosen action when rewarded (0-1).
    - lapse0: Baseline WM lapse log-odds (can be negative/positive).
    - ss_slope: How strongly set size inflates lapse; lapse = sigmoid(lapse0 + ss_slope*(nS-3)).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_alpha, lapse0, ss_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Compute WM lapse given set size for this block
        logit = lapse0 + ss_slope * (float(nS) - 3.0)
        lapse = 1.0 / (1.0 + np.exp(-logit))  # in [0,1]

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with lapse: mixture of uniform and softmax over WM weights
            p_wm_soft = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = (1.0 - lapse) * p_wm_soft + lapse * (1.0 / nA)

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # - Reward-modulated Hebbian update toward chosen action if rewarded
            onehot = np.zeros(nA)
            onehot[a] = 1.0
            if r > 0.0:
                w[s, :] = (1.0 - wm_alpha) * w[s, :] + wm_alpha * onehot
            else:
                # mild relaxation toward uniform when not rewarded
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

            # normalize to probability
            w[s, :] = np.clip(w[s, :], eps, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with cross-state interference and reward tagging; interference increases with set size.
    - RL: standard delta rule with softmax policy (beta scaled by 10).
    - WM: stores action distributions per state; rewarded actions are tagged and emphasized in policy.
    - Cross-state interference: WM for the current state is blended with the average WM of other states,
      with a blending coefficient that increases with set size.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1).
    - wm_weight: Mixture weight of WM in the action policy (0-1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_confusion: Base cross-state blending proportion (0-1), scaled by set size.
    - reward_tag: Degree to emphasize actions that were previously rewarded in this state (0-1).
    - ss_scale2: Scales how much larger set sizes increase confusion; phi = wm_confusion * ss_scale2 * (nS-3)/3.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_confusion, reward_tag, ss_scale2 = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Reward tags: track which actions in each state have been rewarded recently
        tags = eps * np.ones((nS, nA))

        # Set-size–scaled confusion parameter
        phi = wm_confusion * ss_scale2 * max(0.0, (float(nS) - 3.0) / 3.0)
        phi = max(0.0, min(1.0, phi))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # Build WM effective representation with cross-state blending
            if nS > 1:
                mean_others = (np.sum(w, axis=0) - w[s, :]) / max(1, (nS - 1))
            else:
                mean_others = w[s, :].copy()
            W_blend = (1.0 - phi) * w[s, :] + phi * mean_others

            # Apply reward tagging emphasis
            if np.sum(tags[s, :]) > 0:
                tag_norm = tags[s, :] / max(eps, np.sum(tags[s, :]))
            else:
                tag_norm = np.ones(nA) / nA
            W_aug = (1.0 - reward_tag) * W_blend + reward_tag * tag_norm

            # RL policy (probability of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy from augmented WM vector
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_aug - W_aug[a])))

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # - Light decay toward uniform in all states to prevent saturation
            w = 0.98 * w + 0.02 * w_0

            # - If rewarded, set memory strongly to chosen action
            onehot = np.zeros(nA)
            onehot[a] = 1.0
            if r > 0:
                w[s, :] = 0.2 * w[s, :] + 0.8 * onehot
                tags[s, :] = 0.5 * tags[s, :]  # decay old tags
                tags[s, a] += 1.0
            else:
                # If not rewarded, mild move away from the chosen action
                w[s, a] = 0.8 * w[s, a] + 0.2 * (1.0 / nA)
                tags[s, :] = 0.9 * tags[s, :]

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], eps, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
