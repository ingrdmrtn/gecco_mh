def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with surprise-gated arbitration and set-size-modulated gate.

    Idea:
    - RL learns Q-values with a softmax policy.
    - WM stores per-state action distributions that drift toward uniform (decay) but are sharpened
      by rewarded outcomes.
    - Arbitration weight (how much WM influences choice) is dynamically gated by current surprise
      (absolute reward prediction error) and down-regulated at larger set sizes.

    Parameters (6 total):
    - lr: RL learning rate (0..1)
    - softmax_beta: inverse temperature for RL policy (rescaled x10 internally)
    - wm_learn: WM learning rate toward the reinforced action when r==1 (0..1)
    - wm_decay: WM decay rate toward uniform on each visit (0..1)
    - gate_slope: sensitivity of the WM-vs-RL gate to surprise magnitude (>=0)
    - setsize_gate_shift: set-size effect on the gate; negative values reduce WM influence when nS>3

    Set-size impact:
    - Gate includes a set-size term: gate_raw += setsize_gate_shift * ((nS - 3) / 3).
      Thus, larger set sizes can reduce the arbitration weight if setsize_gate_shift < 0.
    """
    lr, softmax_beta, wm_learn, wm_decay, gate_slope, setsize_gate_shift = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic WM policy
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # Surprise-gated WM policy with set-size modulation
            # Compute unsigned PE (surprise) using current Q before update
            pe = r - Q_s[a]
            gate_raw = gate_slope * np.abs(pe) + setsize_gate_shift * ((nS - 3) / 3.0)
            wm_weight = 1.0 / (1.0 + np.exp(-gate_raw))  # sigmoid

            # WM policy
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm*(W_s-W_s[a])))

            p_total = wm_weight* p_wm + (1-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM update: decay toward uniform, then learn from positive outcome
            # Decay current state's WM toward uniform
            w[s, :] = (1 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # If reward, sharpen toward the chosen action
            if r == 1:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1 - wm_learn) * w[s, :] + wm_learn * target

            # Renormalize and clip
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with priority-based storage and set-size-scaled arbitration.

    Idea:
    - RL learns Q-values as usual.
    - WM has limited "slots" across states within a block; it prioritizes states that recently yielded reward.
    - If the number of stored states exceeds wm_slots, the lowest-priority states are evicted (reset to uniform).
    - Arbitration gives weight to WM only when the current state is stored, scaled down for larger set sizes.
    - A small lapse (epsilon) blends in uniform choice noise.

    Parameters (6 total):
    - lr: RL learning rate (0..1)
    - softmax_beta: inverse temperature for RL policy (rescaled x10 internally)
    - wm_weight_scale: baseline WM arbitration weight when the state is stored at set size 3 (0..1)
    - wm_slots: number of WM slots available per block (>=0; can be fractional but is treated as a cap)
    - wm_priority_decay: per-visit decay of WM state priority (0..1); larger decays empty slots faster
    - lapse_eps: probability of a uniform random choice (0..1)

    Set-size impact:
    - WM arbitration weight is scaled by (3 / nS), reducing WM influence at larger set sizes.
    - Limited wm_slots makes storage harder to maintain when nS is large; more states compete for fewer slots.
    """
    lr, softmax_beta, wm_weight_scale, wm_slots, wm_priority_decay, lapse_eps = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic for stored WM associations
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        # Per-state priorities for storage (0..1); >0 indicates stored
        priority = np.zeros(nS)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # Determine whether current state is stored in WM
            is_stored = 1.0 if priority[s] > 0 else 0.0

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy (if stored it is sharp, else near-uninformative uniform via W_s)
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm*(W_s-W_s[a])))

            # Arbitration: only meaningful if stored; and scaled by set size
            wm_weight = wm_weight_scale * (3.0 / max(1, nS)) * is_stored
            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            # Mix with a lapse
            p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
            p_total = (1 - lapse_eps) * p_total + lapse_eps * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM priority update and eviction policy
            # Decay all priorities slightly on each visit to encourage turnover
            priority *= (1.0 - wm_priority_decay)

            if r == 1:
                # Store/update this state with high priority on reward
                priority[s] = 1.0
                # Set memory toward a one-hot for chosen action
                target = np.zeros(nA); target[a] = 1.0
                w[s, :] = target.copy()
            else:
                # If not rewarded and not stored, keep uniform; if stored, leave as is but will decay out
                pass

            # Enforce capacity: if number of stored states > wm_slots, evict the lowest-priority ones
            stored_indices = np.where(priority > 0)[0]
            if wm_slots < len(stored_indices):
                # Sort stored by priority ascending and evict extras
                order = np.argsort(priority[stored_indices])  # ascending
                num_to_evict = int(len(stored_indices) - np.floor(wm_slots))
                if num_to_evict > 0:
                    evict_idx = stored_indices[order[:num_to_evict]]
                    priority[evict_idx] = 0.0
                    w[evict_idx, :] = w_0[evict_idx, :]

            # Ensure W_s row is a valid distribution
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with confidence-based arbitration and set-size penalty.

    Idea:
    - RL learns Q-values with softmax.
    - WM stores action distributions and updates toward one-hot on rewards, with visit-wise decay to uniform.
    - Arbitration is driven by relative confidence: WM confidence (low entropy) vs RL confidence (max policy prob).
      A sigmoid of their difference determines WM weight; larger set sizes penalize WM reliance.

    Parameters (6 total):
    - lr: RL learning rate (0..1)
    - softmax_beta: inverse temperature for RL policy (rescaled x10 internally)
    - wm_learn: WM learning rate toward the chosen action when rewarded (0..1)
    - wm_decay: WM decay toward uniform on each visit (0..1)
    - arb_sensitivity: gain on (conf_wm - conf_rl) for arbitration (>=0)
    - setsize_penalty: penalty multiplier for larger set sizes on the WM weight (>=0)

    Set-size impact:
    - WM arbitration weight is reduced by setsize_penalty * max(0, nS - 3), shifting reliance toward RL when load is high.
    """
    lr, softmax_beta, wm_learn, wm_decay, arb_sensitivity, setsize_penalty = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic WM policy
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s,:]
            W_s = w[s,:]

            # RL policy probability of chosen action
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy probability of chosen action
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm*(W_s-W_s[a])))

            # Confidence estimates
            # RL confidence as max softmax probability under RL policy
            rl_probs = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            rl_probs = rl_probs / np.sum(rl_probs)
            conf_rl = np.max(rl_probs)

            # WM confidence as 1 - normalized entropy of W_s
            ws = np.clip(W_s, 1e-12, None)
            ws = ws / np.sum(ws)
            H = -np.sum(ws * np.log(ws))
            H_max = np.log(len(W_s))
            conf_wm = 1.0 - (H / H_max)  # 0 when uniform, 1 when one-hot

            # Arbitration weight from relative confidence with set-size penalty
            set_pen = setsize_penalty * max(0, nS - 3)
            wm_weight = 1.0 / (1.0 + np.exp(-(arb_sensitivity * (conf_wm - conf_rl) - set_pen)))
            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            p_total = wm_weight* p_wm + (1-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM update: decay toward uniform, then learn on reward
            w[s, :] = (1 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r == 1:
                target = np.zeros(nA); target[a] = 1.0
                w[s, :] = (1 - wm_learn) * w[s, :] + wm_learn * target

            # Normalize and clip
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
