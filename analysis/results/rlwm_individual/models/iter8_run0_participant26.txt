def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited slot WM with confusion noise and set-size pressure.

    Mechanism:
    - Decision policy is a convex mixture of RL softmax and WM softmax.
    - WM access probability reflects limited capacity slots relative to set size,
      reduced further by a set-size pressure factor.
    - WM content is updated by "refresh" after rewards and otherwise remains close
      to uniform; confusion noise makes WM less precise even when recalled.

    Parameters (tuple):
    - lr: RL learning rate for Q updates (0..1).
    - beta_rl: inverse temperature for RL softmax; internally scaled x10.
    - K_slots: effective number of WM slots available to hold item-action mappings (>=0).
    - wm_confusion: probability mass mixed with uniform during WM recall (0..1).
    - refresh_rate: degree to overwrite WM with the rewarded action (0..1).
    - ss_pressure: additional capacity pressure when set size exceeds 3 (>=0).
      Effective slots become K_slots / (1 + ss_pressure*(nS-3)_+).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, beta_rl, K_slots, wm_confusion, refresh_rate, ss_pressure = model_parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute WM recall probability from slots and set size
        extra = max(0, nS - 3)
        eff_slots = K_slots / (1.0 + ss_pressure * extra)
        p_recall = min(1.0, eff_slots / max(1.0, nS))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM confusion: mix WM with uniform before policy
            W_eff = (1.0 - wm_confusion) * W_s + wm_confusion * (1.0 / nA)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            # Arbitration: probability to rely on WM equals recall probability
            wm_weight = np.clip(p_recall, 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: mild stabilization plus reward-based refresh
            # First keep WM near current content (no automatic leak here)
            # Then if rewarded, refresh toward chosen action one-hot
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - refresh_rate) * w[s, :] + refresh_rate * onehot
            else:
                # On no reward, slightly move back toward uniform baseline
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Win-Stay/Lose-Switch working memory with recency-driven arbitration and set-size cost.

    Mechanism:
    - WM stores the last rewarded action per state; it tends to repeat winning actions
      (win-stay) and de-emphasize losing actions (lose-switch), with leak toward uniform.
    - Arbitration weight increases when the state was seen very recently (recency gain),
      but decreases with larger set sizes (set-size cost).
    - Choice policy is a mixture of RL softmax and WM softmax.

    Parameters (tuple):
    - lr: RL learning rate for Q updates (0..1).
    - beta_rl: inverse temperature for RL softmax; internally scaled x10.
    - wm_arbi_base: baseline WM arbitration weight before recency and set-size adjustments (0..1).
    - wm_decay: per-visit leak of WM toward uniform (0..1).
    - setsize_cost: linear penalty on WM arbitration with larger set sizes (>=0).
    - recency_gain: strength of recency-based boost in WM arbitration (>=0).
      Recency boost is exp(-ISI) scaled by recency_gain, where ISI is trials since state was last observed.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, beta_rl, wm_arbi_base, wm_decay, setsize_cost, recency_gain = model_parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_seen = -1 * np.ones(nS, dtype=int)

        # Base weight penalized by set size
        wm_weight_base = wm_arbi_base - setsize_cost * max(0, nS - 3)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: strong if a clear "win-stay" trace exists
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Recency-based arbitration boost: recent re-occurrence yields higher WM reliance
            if last_seen[s] >= 0:
                isi = t - last_seen[s]
                rec_boost = recency_gain * np.exp(-float(isi))
            else:
                rec_boost = 0.0

            wm_weight = wm_weight_base + rec_boost
            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform every time the state is encountered
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Win-Stay: if rewarded, overwrite toward chosen action; Lose-Switch: if not, suppress chosen action
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - (1.0 - wm_decay)) * w[s, :] + (1.0 - wm_decay) * onehot
            else:
                # Penalize the chosen action, redistribute mass to others
                penal = 0.5 * wm_decay
                w[s, a] = (1.0 - penal) * w[s, a]
                redistribute = penal * w[s, a] / (nA - 1 + 1e-12)
                for a2 in range(nA):
                    if a2 != a:
                        w[s, a2] = (1.0 - wm_decay) * w[s, a2] + wm_decay * (1.0 / nA) + redistribute

            last_seen[s] = t

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with uncertainty-weighted arbitration and set-size dependent WM precision.

    Mechanism:
    - RL policy: softmax over Q-values.
    - WM policy: softmax over WM weights; WM precision degrades with set size via
      an interference gain and exponent. This is implemented by shrinking WM logits
      toward uniform before the softmax (lower effective temperature).
    - Arbitration: WM weight increases when RL's policy is certain (low entropy)
      and decreases when RL is uncertain (high entropy). Overall weight bounded [0,1].
    - WM dynamics: leaky toward uniform plus reward-based strengthening.

    Parameters (tuple):
    - lr: RL learning rate for Q updates (0..1).
    - beta_rl: inverse temperature for RL softmax; internally scaled x10.
    - wm_leak: per-visit leak of WM toward uniform (0..1).
    - interf_gain: strength of set-size driven WM interference (>=0).
    - ss_exponent: exponent controlling how interference scales with set size (>=0).
    - arbi_gain: gain of uncertainty-weighted arbitration (>=0); higher values put
      more weight on WM as RL entropy decreases.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, beta_rl, wm_leak, interf_gain, ss_exponent, arbi_gain = model_parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute WM precision shrink based on set size
        shrink = 1.0 / (1.0 + interf_gain * (float(nS) ** ss_exponent))
        shrink = np.clip(shrink, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # RL full softmax for entropy
            q_center = Q_s - np.max(Q_s)
            exp_q = np.exp(softmax_beta * q_center)
            pi_rl = exp_q / np.sum(exp_q)
            p_rl = pi_rl[a]

            # RL policy entropy (normalized to [0,1] by dividing by log(nA))
            ent_rl = -np.sum(pi_rl * (np.log(pi_rl + 1e-12))) / np.log(nA + 1e-12)

            # WM effective logits shrink toward uniform to emulate lower precision
            W_s = w[s, :]
            W_eff = (1.0 - shrink) * (1.0 / nA) + shrink * W_s
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            # Arbitration: weight inversely proportional to RL entropy
            wm_weight = np.clip(arbi_gain * (1.0 - ent_rl), 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # Reward-based WM strengthening
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * onehot

        blocks_log_p += log_p

    return -blocks_log_p
