Below are three standalone cognitive models that follow the provided template, implement RL + WM mechanisms, and return the negative log-likelihood of the observed choices. Each model uses up to 6 parameters, uses all parameters meaningfully, and includes a clear docstring.

Note: Assumes numpy as np is already imported by the environment.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with load-calibrated arbitration, reward-contingent encoding, and baseline leak.

    Idea:
    - RL learns per-state action values via delta rule and softmax.
    - WM is a leaky store that encodes the (state, chosen action) more strongly on reward.
    - Arbitration between WM and RL is modulated by load: smaller set sizes => stronger WM.
    - WM decays to a uniform baseline; encoding strength and decay are parameterized.

    Parameters:
    - lr: RL learning rate in [0,1].
    - wm_weight_base: Base WM arbitration weight in [0,1].
    - softmax_beta: RL inverse temperature (rescaled internally x10).
    - wm_encode_strength: Strength of WM update toward one-hot after reward in [0,1].
    - wm_decay_base: Baseline WM decay toward uniform per trial in [0,1].
    - load_sensitivity: Scales how arbitration weight changes with load (>=0).
        Effective WM weight = clip(wm_weight_base * (1 + load_sensitivity*(3/nS - 0.5)), 0, 1).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_encode_strength, wm_decay_base, load_sensitivity = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-calibrated arbitration weight; smaller sets => higher weight
        wm_weight_eff_block = wm_weight_base * (1.0 + load_sensitivity * (3.0 / float(nS) - 0.5))
        wm_weight_eff_block = float(np.clip(wm_weight_eff_block, 0.0, 1.0))

        # Decay grows mildly with load because more items compete
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6
        wm_decay_eff = np.clip(wm_decay_base * (1.0 + 0.5 * load), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic softmax over WM map)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = p_wm * wm_weight_eff_block + (1 - wm_weight_eff_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: reward-contingent encoding + baseline leak
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_encode_strength) * w[s, :] + wm_encode_strength * target
            # Leak toward uniform each trial
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL (asymmetric via negative-PE scaling) + WM with surprise-gated readout and load-sensitive decay.

    Idea:
    - RL uses a single learning-rate 'lr' but scales negative PE by 'neg_pe_scale' (<,=,>1).
      Effective lr_neg = lr * neg_pe_scale.
    - WM stores a distribution over actions per state and decays faster with higher set size.
    - Arbitration weight is wm_weight, but the use of WM is further gated by momentary surprise:
      surprise ≈ |PE|, transformed via a logistic with slope 'wm_surprise_gate' to up-/down-weight WM.
    - Thus, when outcomes are surprising, WM contributes more to choice; otherwise RL dominates.

    Parameters:
    - lr: Base RL learning rate in [0,1].
    - neg_pe_scale: Multiplier on learning rate for negative prediction errors (>=0).
    - wm_weight: Base WM mixture weight in [0,1].
    - softmax_beta: RL inverse temperature (rescaled internally x10).
    - wm_decay: Baseline WM decay rate per trial in [0,1].
    - wm_surprise_gate: Slope for logistic gating by |PE| (>=0). Larger => stronger gating.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, neg_pe_scale, wm_weight, softmax_beta, wm_decay, wm_surprise_gate = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # WM decay increases with load (more items, more interference)
        wm_decay_eff = np.clip(wm_decay * (1.0 + load), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Surprise from RL PE magnitude
            pe = r - Q_s[a]
            surprise = abs(pe)
            gate = 1.0 / (1.0 + np.exp(-wm_surprise_gate * (surprise - 0.5)))  # ~0 when unsurprising, ~1 when surprising

            wm_weight_eff = np.clip(wm_weight * gate, 0.0, 1.0)

            p_total = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric negative PE scaling
            eff_lr = lr * (neg_pe_scale if pe < 0 else 1.0)
            q[s][a] += eff_lr * pe

            # WM update: reward strengthens chosen action; else decay
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                # Use a conservative encode strength tied to surprise
                encode_strength = np.clip(min(1.0, surprise + 0.1), 0.0, 1.0)
                w[s, :] = (1.0 - encode_strength) * w[s, :] + encode_strength * target
            # global decay toward uniform (stronger in high load)
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with confidence-amplified WM and load-driven random lapses.

    Idea:
    - RL learns via delta rule and softmax.
    - WM stores a one-hot-like distribution when rewarded; otherwise it softly decays.
    - Arbitration weight is wm_weight0, but WM influence is amplified by its confidence:
      confidence ≈ max(W_s) - mean(W_s), scaled by 'wm_conf_boost'.
    - Additionally, a load-driven random lapse process mixes in uniform choice:
      lapse = clip(rand_choice_base + load_rand_gain * load, 0, 1).
    - Final policy = (1-lapse) * [mix of WM and RL] + lapse * uniform.

    Parameters:
    - lr: RL learning rate in [0,1].
    - wm_weight0: Base WM mixture weight in [0,1].
    - softmax_beta: RL inverse temperature (rescaled internally x10).
    - rand_choice_base: Base lapse rate (>=0, <=1).
    - load_rand_gain: Increase of lapse with load (>=0).
    - wm_conf_boost: Scales how strongly WM confidence amplifies WM weight (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight0, softmax_beta, rand_choice_base, load_rand_gain, wm_conf_boost = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-driven lapse (uniform random)
        lapse = float(np.clip(rand_choice_base + load_rand_gain * load, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WM confidence amplification
            conf = float(np.max(W_s) - np.mean(W_s))  # in [0,1-1/nA]
            wm_weight_eff = np.clip(wm_weight0 * (1.0 + wm_conf_boost * conf), 0.0, 1.0)

            # Mixture with lapse
            p_mix = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: reward-anchored encoding with soft decay
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                # Encode proportional to confidence gap complement (more when currently weak)
                encode_strength = np.clip(0.5 + 0.5 * (1.0 - conf), 0.0, 1.0)
                w[s, :] = (1.0 - encode_strength) * w[s, :] + encode_strength * target
            # Soft decay toward uniform each trial
            decay_rate = np.clip(0.2 + 0.6 * load, 0.0, 1.0)  # faster decay under higher load
            w[s, :] = (1.0 - decay_rate) * w[s, :] + decay_rate * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
