def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with rehearsal and leak.

    Idea:
    - RL: standard delta-rule with softmax policy.
    - WM: state-specific cached action distribution that becomes one-hot after reward (perfect encoding),
      but decays (leaks) toward uniform otherwise. Stronger, nearly deterministic softmax for WM.
    - Capacity limit: effective WM mixture weight scales down when set size exceeds available "slots".
      wm_weight_eff = wm_weight0 * min(1, slot_count / nS).
    - WM also uses its own temperature (wm_beta) to allow sharp policies independent of RL beta.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight0: base mixture weight for WM (0..1), before set-size scaling.
    - softmax_beta: inverse temperature for RL (scaled internally by 10).
    - wm_beta: inverse temperature for WM (overrides default 50 to allow fitting WM noise).
    - slot_count: effective number of WM "slots" (>=1). Capacity shaping term for load manipulation.
    - wm_leak: WM leak toward uniform per trial (0..1). Larger -> faster forgetting.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight0, softmax_beta, wm_beta, slot_count, wm_leak = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = max(1.0, wm_beta)
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-limited WM weight per block
        cap_scale = min(1.0, float(slot_count) / float(nS))
        wm_weight_eff = np.clip(wm_weight0 * cap_scale, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(1e-12, denom_rl)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(1e-12, denom_wm)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # WM encoding on reward: set to one-hot for chosen action
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Normalize for safety
            w[s, :] = np.maximum(w[s, :], 1e-12)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with surprise-based arbitration + WM with partial encoding on errors.

    Idea:
    - RL: standard delta-rule with softmax.
    - WM: stores a sharp distribution after reward (one-hot), but also partially encodes after non-reward:
      nudges the chosen action up by epsilon_neg (weak, noisy encoding).
      WM decays toward uniform between trials (small leak implicit via partial renormalization).
    - Arbitration: WM mixture weight increases with RL surprise |PE| and is modulated by set size
      only indirectly through surprise dynamics. A logistic mapping governs surprise-to-WM gain.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - softmax_beta: inverse temperature for RL (scaled internally by 10).
    - wm_beta: inverse temperature for WM policy (>=1).
    - wm_weight0: base WM mixture weight (0..1).
    - k_surprise: sensitivity of WM arbitration to RL surprise (>=0). Larger -> more WM reliance when PE is high.
    - epsilon_neg: WM partial encoding mass added to chosen action after negative feedback (0..1).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta, wm_beta, wm_weight0, k_surprise, epsilon_neg = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = max(1.0, wm_beta)
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(1e-12, denom_rl)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(1e-12, denom_wm)

            # Surprise-based arbitration (per trial, state-specific)
            pe = r - Q_s[a]
            # Map surprise to [0,1] via logistic around 0 with slope k_surprise, centered at 0
            w_surprise = 1.0 / (1.0 + np.exp(-k_surprise * np.abs(pe)))
            wm_weight_eff = np.clip(wm_weight0 * w_surprise, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = pe
            q[s, a] += lr * delta

            # WM update:
            if r > 0.5:
                # Strong encoding on reward
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Partial encoding on error: add epsilon_neg mass to chosen action, then renormalize
                w[s, :] = (1.0 - epsilon_neg) * w[s, :] + (epsilon_neg / (nA - 1)) * np.ones(nA)
                # Move extra mass to chosen action
                spill = epsilon_neg
                w[s, a] += spill
                # Normalize
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])

            # Small implicit leak toward uniform to prevent overconcentration without reward
            # (achieved through convex combination above)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with uncertainty-adaptive temperature + WM with load-dependent binding failures.

    Idea:
    - RL: delta-rule with softmax; inverse temperature is fixed externally (scaled by 10).
    - WM: when reward occurs, it attempts to bind the chosen action to the state.
      Binding succeeds with probability p_bind = exp(-bind_k * (nS - 1)); if binding fails,
      WM resets toward uniform (simulating mis-binding/interference).
      Between updates, no additional leak (WM is stable except binding failures).
    - Arbitration: fixed mixture weight (wm_weight0) but WM reliability also depends on its own
      temperature (wm_beta0). Set-size affects WM via binding failures only.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - softmax_beta0: base RL inverse temperature (scaled internally by 10).
    - wm_beta0: base WM inverse temperature (>=1).
    - wm_weight0: mixture weight for WM (0..1).
    - bind_k: load-sensitivity of WM binding failures (>=0). Higher -> more failures as set size increases.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta0, wm_beta0, wm_weight0, bind_k = model_parameters

    softmax_beta = softmax_beta0 * 10.0
    softmax_beta_wm = max(1.0, wm_beta0)
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Binding success probability as a function of set size
        p_bind = np.exp(-bind_k * max(0.0, float(nS) - 1.0))
        p_bind = np.clip(p_bind, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(1e-12, denom_rl)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(1e-12, denom_wm)

            # Fixed mixture weight
            wm_weight_eff = np.clip(wm_weight0, 0.0, 1.0)
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update with binding failures
            if r > 0.5:
                # With probability p_bind, write the correct action deterministically; otherwise misbind (uniform fallback)
                # We implement the expectation update (mean-field): a convex combination
                w_target = np.zeros(nA)
                w_target[a] = 1.0
                w[s, :] = p_bind * w_target + (1.0 - p_bind) * w_0[s, :]
                # Normalize for safety
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])
            else:
                # No update on negative outcome (WM holds its current contents)
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
