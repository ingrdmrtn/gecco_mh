def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with asymmetric WM learning and load-dependent WM noise

    Description:
    - RL: tabular Q-learning (fixed in template).
    - WM policy: softmax over WM weights, but its effective temperature decreases with set size
      (more noise under higher load).
    - WM learning: asymmetric delta-rule toward targets:
        * On reward: move toward one-hot on chosen action (fast rate wm_eta_pos).
        * On no reward: move toward uniform (rate wm_eta_neg), i.e., forget erroneous association.
    - Mixture: fixed mixture between WM and RL policies (wm_weight).

    Parameters (tuple):
    - lr: RL learning rate (0..1).
    - wm_weight: Mixture weight for WM policy (0..1).
    - softmax_beta: RL inverse temperature; internally scaled x10 (see template).
    - wm_eta_pos: WM learning rate on rewarded trials (0..1).
    - wm_eta_neg: WM learning rate on unrewarded trials (0..1).
    - noise_load: Degree to which WM becomes noisier with set size (0..1). 0=no extra noise; 1=full reduction at high load.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_eta_pos, wm_eta_neg, noise_load = model_parameters

    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load factor in [0,1]: 0 for set size 3, 1 for set size 6
        load_factor = 0.0 if nS <= 3 else 1.0
        beta_wm_eff = max(1.0, softmax_beta_wm * (1.0 - noise_load * load_factor))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / max(eps, np.sum(np.exp(softmax_beta*(Q_s-Q_s[a]))))

            # WM policy: load-dependent temperature (noisier when set size is larger)
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / max(eps, denom_wm)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(eps, p_total))
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM updating: asymmetric toward target
            if r > 0.5:
                # Reward: move toward one-hot on chosen action
                target = np.zeros(nA)
                target[a] = 1.0
                eta = wm_eta_pos
            else:
                # No reward: forget toward uniform
                target = w_0[s,:].copy()
                eta = wm_eta_neg
            w[s,:] = (1.0 - eta) * w[s,:] + eta * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with load-dependent swap (misbinding) and reward-gated WM strengthening, plus decay

    Description:
    - RL: tabular Q-learning (fixed in template).
    - WM: On rewarded trials, WM stores the chosen action but with a probability of swap/misbinding
      that increases with set size. The target distribution for WM after reward assigns:
        * (1 - p_swap) to the chosen action
        * p_swap uniformly to the non-chosen actions
      The update strength is scaled by wm_boost on reward.
      On unrewarded trials, WM decays toward uniform at rate wm_decay.
    - Mixture: fixed mixture of WM and RL policies (wm_weight).

    Parameters (tuple):
    - lr: RL learning rate (0..1).
    - wm_weight: Mixture weight for WM policy (0..1).
    - softmax_beta: RL inverse temperature; internally scaled x10 (see template).
    - swap_base: Base misbinding rate per item (0..1). Effective swap increases with set size as: 1 - (1 - swap_base)^nS.
    - wm_decay: WM decay rate toward uniform on no-reward (0..1).
    - wm_boost: Additional strengthening factor for WM on reward (0..1); higher values make rewarded updates stronger.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, swap_base, wm_decay, wm_boost = model_parameters

    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent swap/misbinding probability
        p_swap = 1.0 - (1.0 - np.clip(swap_base, 0.0, 1.0))**max(1, nS)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / max(eps, np.sum(np.exp(softmax_beta*(Q_s-Q_s[a]))))

            # WM policy with standard sharp softmax
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(eps, denom_wm)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(eps, p_total))
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM updating with swap and decay
            if r > 0.5:
                target = np.ones(nA) * (p_swap / (nA - 1.0))
                target[a] = 1.0 - p_swap
                eta = np.clip(wm_boost, 0.0, 1.0)
                w[s,:] = (1.0 - eta) * w[s,:] + eta * target
            else:
                # Decay toward uniform when feedback is negative
                eta = np.clip(wm_decay, 0.0, 1.0)
                w[s,:] = (1.0 - eta) * w[s,:] + eta * w_0[s,:]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Capacity-limited WM with load-dependent lapse and refresh

    Description:
    - RL: tabular Q-learning (fixed in template).
    - WM capacity: assumes a limited number of slots (cap_slots). When set size exceeds capacity,
      the effective WM reliability drops. This is modeled as a lapse in the WM policy that mixes
      some uniform choice probability into WM, proportional to the overload.
    - WM refresh: on each trial, WM updates the current state toward a reward-contingent target:
        * Reward: toward one-hot on chosen action.
        * No reward: toward uniform (forget incorrect mapping).
      The refresh strength scales with occupancy = min(1, cap_slots / nS), reflecting fewer resources under high load.
    - Mixture: fixed mixture between WM and RL policies (wm_weight).

    Parameters (tuple):
    - lr: RL learning rate (0..1).
    - wm_weight: Mixture weight for WM policy (0..1).
    - softmax_beta: RL inverse temperature; internally scaled x10 (see template).
    - cap_slots: Effective WM capacity in number of items (>=0). Larger means more robust WM under load.
    - wm_refresh: Base WM refresh/learning rate toward target (0..1).
    - lapse_base: Base WM lapse rate when fully overloaded (0..1). Actual lapse scales with overload.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, cap_slots, wm_refresh, lapse_base = model_parameters

    softmax_beta *= 10 # beta has a higher upper bound
    softmax_beta_wm = 50 # very deterministic
    blocks_log_p = 0
    eps = 1e-12

    for b in np.unique(blocks):

        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Occupancy (0..1): fraction of set that fits into capacity
        occ = 0.0 if nS <= 0 else min(1.0, max(0.0, float(cap_slots) / float(nS)))
        # Lapse grows with overload (1 - occ)
        lapse_eff = np.clip(lapse_base * (1.0 - occ), 0.0, 1.0)
        # WM refresh scales with occupancy
        wm_eta = np.clip(wm_refresh * occ, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / max(eps, np.sum(np.exp(softmax_beta*(Q_s-Q_s[a]))))

            # WM policy with lapse toward uniform reflecting overload
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm_sharp = 1.0 / max(eps, denom_wm)
            p_wm = (1.0 - lapse_eff) * p_wm_sharp + lapse_eff * (1.0 / nA)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            log_p += np.log(max(eps, p_total))
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM refresh/update scaled by occupancy
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
            else:
                target = w_0[s,:].copy()
            w[s,:] = (1.0 - wm_eta) * w[s,:] + wm_eta * target

        blocks_log_p += log_p

    return -blocks_log_p
