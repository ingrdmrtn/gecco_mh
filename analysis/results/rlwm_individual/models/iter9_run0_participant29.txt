def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Bayesian-Dirichlet Working Memory with entropy-based arbitration (load-adjusted).

    Summary:
    - RL learns Q-values with a fixed learning rate and softmax policy.
    - WM maintains Dirichlet-like counts per state-action with decay toward a symmetric prior.
      WM policy is near-deterministic on higher counts.
    - Arbitration weights WM more when the WM policy is low-entropy. The impact of entropy on
      arbitration is reduced under higher set size (load).

    Parameters (model_parameters):
    - lr:            RL learning rate (0..1)
    - wm_bias:       Base arbitration logit; higher favors WM policy (can be negative or positive)
    - softmax_beta:  RL inverse temperature; scaled by x10 internally
    - wm_decay:      Decay of WM counts toward the symmetric prior alpha0 each trial (0..1)
    - wm_alpha0:     Symmetric Dirichlet prior concentration per action (>0); larger means flatter prior
    - arb_kappa:     Gain on entropy-to-weight mapping; effective gain is arb_kappa/(1+(nS-3))

    Set-size impact:
    - Arbitration sensitivity to WM entropy is divided by (1 + (nS-3)), making WM less influential
      in larger set size blocks even when WM is confident.
    """
    lr, wm_bias, softmax_beta, wm_decay, wm_alpha0, arb_kappa = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))

        # WM Dirichlet counts; initialize to symmetric prior
        counts = wm_alpha0 * np.ones((nS, nA))
        prior = wm_alpha0 * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            # WM "values" are normalized counts (posterior mean); use as logits for near-deterministic WM policy
            cnt_s = counts[s, :]
            cnt_sum = np.sum(cnt_s)
            if cnt_sum <= 0:
                cnt_s = prior[s, :]
                cnt_sum = np.sum(cnt_s)
            W_s = cnt_s / cnt_sum

            # RL policy probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            # WM policy probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Entropy of WM policy on this state
            H = -np.sum(np.clip(W_s, eps, 1.0) * np.log(np.clip(W_s, eps, 1.0)))
            Hmax = np.log(nA)
            # Arbitration weight as a sigmoid of (Hmax - H): higher when WM is more peaked.
            k_eff = arb_kappa / (1.0 + max(0, nS - 3))
            w_eff = 1.0 / (1.0 + np.exp(-(wm_bias + k_eff * (Hmax - H))))

            p_total = w_eff * p_wm + (1.0 - w_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM counts decay toward prior, then increment chosen action by outcome-weighted evidence
            counts[s, :] = (1.0 - wm_decay) * counts[s, :] + wm_decay * prior[s, :]
            # Add outcome as evidence; ensure some signal even on no-reward
            evidence = 0.1 + 0.9 * r  # 1.0 on reward, 0.1 on no-reward
            counts[s, a] += evidence

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + One-shot WM mapping with load-dependent retrieval precision.

    Summary:
    - RL learns Q-values with a fixed learning rate and softmax policy.
    - WM stores a near one-hot mapping from state to last inferred correct action,
      updated more strongly on reward and decaying toward uniform each trial.
    - WM retrieval precision (inverse temperature) is reduced under higher set size.

    Parameters (model_parameters):
    - lr:             RL learning rate (0..1)
    - wm_weight:      Fixed mixture weight of WM policy (0..1)
    - softmax_beta:   RL inverse temperature; scaled x10 internally
    - wm_leak:        Leak toward uniform on WM weights per trial (0..1)
    - beta_wm_base:   Base inverse temperature for WM retrieval (>0)
    - load_slope:     Load scaling; beta_wm_eff = beta_wm_base / (1 + load_slope*(nS-3))

    Set-size impact:
    - Larger set size decreases WM precision via beta_wm_eff, making choices rely more
      on RL even when WM is stored.
    """
    lr, wm_weight, softmax_beta, wm_leak, beta_wm_base, load_slope = model_parameters

    softmax_beta *= 10.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))

        # WM weights: row per state; start uniform
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM beta
        beta_wm_eff = beta_wm_base / (1.0 + max(0.0, load_slope * (nS - 3)))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            # WM policy with load-adjusted precision
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # One-shot WM pulse toward chosen action; stronger on reward
            pulse = np.zeros(nA)
            pulse[a] = 1.0
            eta = 0.2 + 0.8 * r  # 1.0 on reward, 0.2 on no-reward
            w[s, :] = (1.0 - eta) * w[s, :] + eta * pulse

            # Normalize to a probability simplex to stabilize the policy
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] = w[s, :] / row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Recency-weighted episodic WM with load-dependent forgetting and confidence scaling.

    Summary:
    - RL updates Q-values with a fixed learning rate and softmax policy.
    - WM accumulates a recency-weighted trace of chosen actions per state (episodic buffer),
      with exponential forgetting that increases with set size.
    - WM policy precision is scaled by a confidence parameter; arbitration is fixed by wm_weight.

    Parameters (model_parameters):
    - lr:            RL learning rate (0..1)
    - wm_weight:     Mixture weight assigned to WM policy (0..1)
    - softmax_beta:  RL inverse temperature; scaled x10 internally
    - mem_lambda:    Recency gain for the WM trace (0..1); how much a new event updates memory
    - conf_temp:     Scales WM inverse temperature; softmax_beta_wm = 1 + 50*conf_temp
    - load_forget:   Load-dependent forgetting; leak = 1 - exp(-load_forget * nS)

    Set-size impact:
    - Larger set size increases the WM leak (forgetting) as leak = 1 - exp(-load_forget * nS),
      weakening episodic traces under higher load.
    """
    lr, wm_weight, softmax_beta, mem_lambda, conf_temp, load_forget = model_parameters

    softmax_beta *= 10.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))

        # WM episodic traces, start uniform
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent leak
        leak = 1.0 - np.exp(-max(0.0, load_forget) * float(nS))
        leak = np.clip(leak, 0.0, 1.0)

        # WM precision from confidence
        softmax_beta_wm = 1.0 + 50.0 * conf_temp

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM forgetting toward uniform, stronger with load
            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

            # Recency-weighted episodic pulse for chosen action, reward-scaled
            pulse = np.zeros(nA)
            pulse[a] = 1.0
            gain = mem_lambda * (0.5 + 0.5 * r)  # half-strength on no-reward, full on reward
            w[s, :] = (1.0 - gain) * w[s, :] + gain * pulse

            # Normalize row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] = w[s, :] / row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
