def cognitive_model1(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL + decaying WM with set-size–penalized arbitration.

    Mechanism
    - RL: tabular Q-learning with a single learning rate and softmax policy.
    - WM store: cache of the last rewarded action per state that passively decays toward uniform.
    - Arbitration: mixture of WM and RL policies where the WM weight is penalized as a power-law
      function of set size.

    Parameters
    - lr: learning rate for RL Q-updates (0..1).
    - wm_weight_base: baseline mixture weight of WM vs RL (0..1).
    - softmax_beta: inverse temperature for RL; internally scaled by 10 (>0).
    - wm_decay: per-trial decay of WM toward uniform (0..1), higher means faster forgetting.
    - ss_exponent: exponent controlling how WM weight scales with set size (>0). Effective WM weight
      per block is wm_weight_base / ( (nS/3)^ss_exponent ). Larger ss_exponent => stronger penalty at set size 6.

    Returns
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, ss_exponent = parameters
    softmax_beta *= 10.0  # match template's larger dynamic range for RL policy
    softmax_beta_wm = 50.0  # WM policy assumed near-deterministic when available

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values and WM cache
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size–penalized WM mixture weight for this block
        wm_weight_eff = wm_weight_base / max((nS / 3.0) ** max(ss_exponent, 0.0), 1e-6)
        wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax policy
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(z_rl) / np.sum(np.exp(z_rl))
            p_rl = max(pi_rl[a], 1e-12)

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM softmax policy from cached weights
            z_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(z_wm) / np.sum(np.exp(z_wm))
            p_wm = max(pi_wm[a], 1e-12)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # 1) Passive decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # 2) If rewarded, overwrite with a one-hot cache of the chosen action
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL + Bayesian WM counts with set-size–dependent mixture shrinkage.

    Mechanism
    - RL: tabular Q-learning with softmax policy.
    - WM store: per-state Dirichlet-like counts for action correctness (success/failure counts
      updated from observed feedback). The WM policy uses the posterior mean of the reward
      probability for each action at the current state.
    - Arbitration: mixture of WM and RL policies. The WM weight is shrunk as set size increases
      via an exponential factor.

    Parameters
    - lr: RL learning rate (0..1).
    - softmax_beta: inverse temperature for RL; internally scaled by 10 (>0).
    - wm_weight: baseline WM mixture weight (0..1).
    - wm_alpha0: WM prior pseudo-count for successes and failures (>=0). Larger values reduce WM volatility.
    - ss_weight_decay: set-size penalty rate on WM weight (>=0). Effective WM weight is
      wm_weight * exp(-ss_weight_decay * (nS - 3)).

    Returns
    - Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta, wm_weight, wm_alpha0, ss_weight_decay = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))

        # WM counts: for each state-action, maintain successes and failures
        succ = wm_alpha0 * np.ones((nS, nA))
        fail = wm_alpha0 * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # used if needed as a fallback prior

        # Set-size–dependent WM weight
        wm_weight_eff = wm_weight * np.exp(-max(ss_weight_decay, 0.0) * (nS - 3.0))
        wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # RL policy
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(z_rl) / np.sum(np.exp(z_rl))
            p_rl = max(pi_rl[a], 1e-12)

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM posterior mean of reward probability per action: E[p] = succ / (succ + fail)
            wm_p_mean = succ[s, :] / np.maximum(succ[s, :] + fail[s, :], 1e-12)
            # Convert to a softmax policy with high precision
            z_wm = softmax_beta_wm * (wm_p_mean - np.max(wm_p_mean))
            pi_wm = np.exp(z_wm) / np.sum(np.exp(z_wm))
            p_wm = max(pi_wm[a], 1e-12)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Update WM counts for the chosen action
            succ[s, a] += r
            fail[s, a] += (1.0 - r)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, parameters):
    """
    RL with eligibility traces + WM with probabilistic retrieval limited by set size.

    Mechanism
    - RL: tabular Q-learning with eligibility traces (replacing traces within a state).
      The trace emphasizes recently chosen actions when updating Q with the PE.
    - WM store: cache of last rewarded action per state (like a one-item map). The WM policy is
      near-deterministic, but retrieval succeeds only with a probability that declines with set size.
    - Arbitration: mixture of WM and RL policies weighted by the product wm_weight * p_recall,
      where p_recall depends on set size.

    Parameters
    - lr: RL learning rate (0..1).
    - softmax_beta: inverse temperature for RL; internally scaled by 10 (>0).
    - wm_weight: baseline WM mixture weight (0..1).
    - trace_lambda: eligibility trace decay for the current state's actions (0..1).
    - recall_base: baseline WM retrieval probability at set size 3 (0..1).
    - recall_slope: linear penalty on retrieval per additional item beyond 3 (>=0).
      Effective p_recall = clip(recall_base / (1 + recall_slope * (nS - 3)), 0, 1).

    Returns
    - Negative log-likelihood of the observed choices.
    """
    lr, softmax_beta, wm_weight, trace_lambda, recall_base, recall_slope = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values and eligibility traces (per state-action)
        q = (1.0 / nA) * np.ones((nS, nA))
        e = np.zeros((nS, nA))

        # WM cache
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size–dependent recall probability
        p_recall = recall_base / max(1.0 + max(recall_slope, 0.0) * (nS - 3.0), 1e-6)
        p_recall = float(np.clip(p_recall, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(z_rl) / np.sum(np.exp(z_rl))
            p_rl = max(pi_rl[a], 1e-12)

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # If WM has a peaked item at this state, use it; otherwise fallback to uniform
            if np.max(W_s) > (1.0 / nA) + 1e-6:
                z_wm = softmax_beta_wm * (W_s - np.max(W_s))
                pi_wm = np.exp(z_wm) / np.sum(np.exp(z_wm))
            else:
                pi_wm = w_0[s, :]

            p_wm = max(pi_wm[a], 1e-12)

            # Arbitration with probabilistic retrieval
            wm_mix = wm_weight * p_recall
            wm_mix = float(np.clip(wm_mix, 0.0, 1.0))
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with eligibility traces
            # Replace trace for chosen action in the current state, decay others in the same state
            e[s, :] *= trace_lambda
            e[s, a] = 1.0
            delta = r - Q_s[a]
            q += lr * delta * e  # broadcasted over all states/actions; only current state's traces are nonzero

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Passive maintenance: no decay term here; WM either stores or stays uniform
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -blocks_log_p
