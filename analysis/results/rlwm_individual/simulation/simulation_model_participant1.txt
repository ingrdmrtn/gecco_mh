def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL with eligibility traces and action stickiness + WM with load-driven noise.

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, lambda_trace, wm_weight, softmax_beta, kappa, wm_noise]

    Returns:
    - simulated_actions: array of simulated choices (int)
    - simulated_rewards: array of simulated rewards (0/1)
    """
    import numpy as np

    lr, lambda_trace, wm_weight_base, softmax_beta, kappa, wm_noise = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]

        nA = 3
        # Use provided set size per block (assumed constant within block)
        nS = int(set_sizes[block_indices][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Eligibility traces
        e = np.zeros((nS, nA))

        # WM load-driven noise (per block)
        noise = 1.0 - np.exp(-wm_noise * max(1, nS))
        noise = float(np.clip(noise, 0.0, 1.0))

        last_action = None

        # Build state -> correct action mapping (assumes states are 0..nS-1)
        # Take the first encountered correct action for each state
        state_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t in range(len(block_states)):
            s = int(block_states[t])
            if 0 <= s < nS and not seen[s]:
                state_correct[s] = int(block_correct[t])
                seen[s] = True

        for t in range(len(block_states)):
            idx = block_indices[t]
            s = int(block_states[t])

            # RL policy with stickiness
            Q_s = q[s, :].copy()
            if last_action is not None:
                bias = np.zeros(nA)
                bias[int(last_action)] = kappa
                Q_s = Q_s + bias

            # Stable softmax for RL
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy (clean)
            W_s = w[s, :].copy()
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_wm_clean = np.exp(wm_logits)
            p_wm_clean /= np.sum(p_wm_clean)

            # Load-driven corruption towards uniform
            p_uniform = np.ones(nA) / nA
            p_wm = (1.0 - noise) * p_wm_clean + noise * p_uniform

            # Mixture policy
            wm_weight = wm_weight_base
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[idx] = a

            # Outcome
            correct_a = int(state_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[idx] = r

            # Learning updates
            # Eligibility traces
            e *= lambda_trace
            e[s, a] += 1.0

            # RL update with traces
            pe = r - q[s, a]
            q += lr * pe * e

            # WM update
            if r > 0.5:
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps
            else:
                relax = 0.05
                w[s, :] = (1.0 - relax) * w[s, :] + relax * w_0[s, :]

            last_action = a

    return simulated_actions, simulated_rewards
