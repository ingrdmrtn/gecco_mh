def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL with stay bias + WM with load-dependent confusion and delta learning

    Inputs:
    - stimulus: array of state/stimulus identifiers for each trial
    - blocks: array of block indices for each trial
    - set_sizes: array of set sizes for each trial (constant within a block)
    - correct_answer: array of correct action for each trial
    - parameters: tuple/list (lr, wm_weight, softmax_beta, stay_bias, wm_lr, confusion_rate)

    Returns:
    - simulated_actions: array of sampled actions (int)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, stay_bias, wm_lr, confusion_rate = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        nS = int(set_sizes[block_idx][0])

        # Build state -> correct action mapping (assumes one correct action per state within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of this state in the block and take its correct action
            match = np.where(block_states == s)[0]
            if len(match) > 0:
                correct_actions[s] = block_correct[match[0]]
            else:
                # Fallback if state not encountered (should not happen); default to 0
                correct_actions[s] = 0

        # Initialize value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        # Effective confusion increases with set size
        conf_eff = 1.0 - (1.0 - np.clip(confusion_rate, 0.0, 1.0)) ** max(1, nS)

        for t in range(len(block_states)):
            idx = block_idx[t]
            s = block_states[t]

            # RL policy with stay bias
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stay_bias
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy (high inverse temperature)
            W_s = w[s, :].copy()
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[idx] = a

            # Generate reward from environment
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[idx] = r

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM delta update toward reward-contingent target with confusion
            if r > 0.5:
                target = np.ones(nA) * (conf_eff / (nA - 1.0))
                target[a] = 1.0 - conf_eff
            else:
                target = w_0[s, :].copy()
            w[s, :] = (1.0 - wm_lr) * w[s, :] + wm_lr * target

            # Update stay memory
            last_action[s] = a

            tr += 1

    return simulated_actions, simulated_rewards
