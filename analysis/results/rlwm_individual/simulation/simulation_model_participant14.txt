import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the Uncertainty-gated RL+WM mixture model.

    Inputs:
    - stimulus: state index on each trial (array of ints, 0..nS-1 within a block)
    - blocks: block index per trial (array of ints)
    - set_sizes: set size of the current block for each trial (array of ints; e.g., 3 or 6)
    - correct_answer: correct action on each trial (array of ints, in {0,1,2})
    - parameters: [lr, softmax_beta_param, gate_bias, gate_size, wm_decay, wm_learn]

    Returns:
    - simulated_actions: array of chosen actions (ints)
    - simulated_rewards: array of binary rewards (ints)
    """
    lr, softmax_beta_param, gate_bias, gate_size, wm_decay, wm_learn = parameters

    softmax_beta = softmax_beta_param * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            st_idx = np.where(block_states == st)[0][0]
            correct_actions[st] = block_correct[st_idx]

        # Initialize RL and WM policies
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Compute RL and WM choice distributions
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Entropies for uncertainty gating
            H_rl = -np.sum(p_rl * np.log(np.clip(p_rl, eps, 1.0)))
            H_wm = -np.sum(p_wm * np.log(np.clip(p_wm, eps, 1.0)))

            # Gating weight (higher when RL is more uncertain than WM)
            size_term = gate_size * (3.5 - nS)
            wm_weight_t = 1.0 / (1.0 + np.exp(-(gate_bias + size_term + (H_rl - H_wm))))

            # Mixture policy
            p_total = wm_weight_t * p_wm + (1.0 - wm_weight_t) * p_rl

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for the state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM imprint on current state
            w[s, :] = (1.0 - wm_learn) * w[s, :]
            w[s, a] += wm_learn * r

            tr += 1

    return simulated_actions, simulated_rewards
