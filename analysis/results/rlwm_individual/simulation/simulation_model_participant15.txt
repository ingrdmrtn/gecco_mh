import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with set-size-specific arbitration weights and WM decay + global lapse.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial (0..nA-1)
    - parameters: [lr, wm_weight_small, wm_weight_large, softmax_beta, wm_decay, epsilon]

    Returns:
    - simulated_actions: int array of chosen actions per trial
    - simulated_rewards: int array of rewards per trial (0/1)
    """
    lr, wm_weight_small, wm_weight_large, softmax_beta, wm_decay, epsilon = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    tiny = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])
        wm_weight = wm_weight_small if nS == 3 else wm_weight_large

        # Map each state to its correct action using first occurrence in this block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = block_correct[idx[0]]
            else:
                # Fallback (shouldn't happen if all states appear): assign a random correct action
                correct_actions[s] = np.random.randint(nA)

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA), dtype=float)
        w = (1.0 / nA) * np.ones((nS, nA), dtype=float)
        w_0 = (1.0 / nA) * np.ones((nS, nA), dtype=float)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            logits_rl = softmax_beta * Q_s
            logits_rl -= np.max(logits_rl)
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # WM softmax (very deterministic)
            logits_wm = softmax_beta_wm * W_s
            logits_wm -= np.max(logits_wm)
            p_wm = np.exp(logits_wm)
            p_wm /= np.sum(p_wm)

            # Arbitration and lapse
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - epsilon) * p_total + epsilon * (1.0 / nA)

            # Numerical safety
            p_total = np.clip(p_total, tiny, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward from task rule
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update
            if r > 0.5:
                w[s, :] = tiny
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Normalize WM row and enforce tiny floor
            w[s, :] = np.maximum(w[s, :], tiny)
            w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
