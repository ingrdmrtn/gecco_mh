def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL with Pearce-Hall associability and recency-based WM that is load-sensitive.

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block ids per trial
    - set_sizes: array of set size per trial (block-constant)
    - correct_answer: array of correct action per trial
    - parameters: [lr_base, beta_rl, wm_weight, wm_decay_base, xi_load_sensitivity]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    import numpy as np

    lr_base, beta_rl, wm_weight, wm_decay_base, xi_load_sensitivity = parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_actions = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (take first occurrence)
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            st_mask = np.where(block_states == st)[0]
            if len(st_mask) > 0:
                correct_actions[st] = block_correct_actions[st_mask[0]]
            else:
                correct_actions[st] = 0  # fallback if a state does not appear

        # Initialize RL and WM values and associability
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        alpha = 0.5 * np.ones((nS, nA))
        alpha_decay = 0.9  # passive decay of associability toward 0.5

        # Load-dependent WM decay (same across trials within block)
        load_factor = 1.0 + xi_load_sensitivity * max(0, nS - 3)
        wm_decay = 1.0 - (1.0 - wm_decay_base) / load_factor

        for t in range(len(block_states)):
            s = int(block_states[t])
            Q_s = q[s, :]
            W_s = w[s, :]

            # Compute policy components
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Simulate reward from correct action
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update with Pearce-Hall associability
            delta = r - Q_s[a]
            q[s, a] += lr_base * alpha[s, a] * delta

            # Small decay of Q toward uniform baseline
            q[s, :] = 0.995 * q[s, :] + 0.005 * (1.0 / nA)

            # Associability dynamics: global decay toward 0.5, then local update based on |PE|
            alpha = alpha_decay * alpha + (1.0 - alpha_decay) * 0.5
            alpha[s, a] = 0.9 * alpha[s, a] + 0.1 * min(1.0, abs(delta))

            # WM decay toward baseline
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Reward-contingent WM refresh (partial overwrite)
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.8 * w[s, :] + 0.2 * one_hot

            tr += 1

    return simulated_actions, simulated_rewards
