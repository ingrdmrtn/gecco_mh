def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate actions and rewards from the RL+WM WSLS-like heuristic with set-size-dependent lapse.

    Inputs:
    - stimulus: array of state indices per trial (int)
    - blocks: array of block indices per trial (int)
    - set_sizes: array of set size per trial (int) [only the first per block is used]
    - correct_answer: array of correct action per trial (int, in {0,1,2})
    - parameters: [lr, wm_weight, softmax_beta, wsls_strength, epsilon_size_slope]

    Returns:
    - simulated_actions: array of chosen actions (int)
    - simulated_rewards: array of received rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wsls_strength, epsilon_size_slope = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        # Determine set size for the block
        nS = int(set_sizes[block_indices][0])
        nA = 3

        # Build per-state correct action mapping assuming states are 0..nS-1
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            # find first occurrence of this state in the block to get its correct action
            idxs = np.where(block_states == st)[0]
            if len(idxs) > 0:
                correct_actions[st] = int(block_correct[idxs[0]])
            else:
                # if a state doesn't appear (edge case), default to 0
                correct_actions[st] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # not used for updates but kept for parity

        # Lapse for this block as a function of set size
        eps = epsilon_size_slope * max(0.0, float(nS) - 3.0)
        eps = min(0.5, max(0.0, eps))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute RL policy
            Q_s = q[s, :]
            Q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_shift)
            p_rl /= np.sum(p_rl)

            # Compute WM policy (near-deterministic)
            W_s = w[s, :]
            W_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_shift)
            p_wm /= np.sum(p_wm)

            # Mixture and lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - eps) * p_mix + eps * (1.0 / nA)
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward from correct action mapping
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update (WSLS-like)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wsls_strength) * w[s, :] + wsls_strength * one_hot
            else:
                anti = np.ones(nA) / (nA - 1.0)
                anti[a] = 0.0
                w[s, :] = (1.0 - wsls_strength) * w[s, :] + wsls_strength * anti

            tr += 1

    return simulated_actions, simulated_rewards
