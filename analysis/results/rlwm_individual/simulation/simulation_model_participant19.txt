def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with set-size-dependent binding interference (simulation).

    Inputs:
    - stimulus: sequence of states for each trial (array-like of ints)
    - blocks: block index for each trial (array-like)
    - set_sizes: block-dependent set sizes on each trial (array-like of ints)
    - correct_answer: correct action on each trial (array-like of ints in [0, nA-1])
    - parameters: [lr, softmax_beta, wm_weight_base, wm_learn, binding_noise_base, wm_beta]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, softmax_beta, wm_weight_base, wm_learn, binding_noise_base, wm_beta = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # base determinism for WM
    beta_wm_eff = softmax_beta_wm * max(0.0, wm_beta)

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes, dtype=int)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    tr = 0

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_set_sizes = set_sizes[block_indices]
        block_correct = correct_answer[block_indices]

        # Determine set size for the block as specified by inputs
        nS = int(block_set_sizes[0])

        # Map per-state correct action using first occurrence within block
        # Assumes states are labeled 0..nS-1
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                state_correct[s] = int(block_correct[idx[0]])
            else:
                # If state not encountered (edge case), default to 0
                state_correct[s] = 0

        # Initialize RL and WM policy representations
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent arbitration and binding interference
        wm_weight = np.clip(wm_weight_base * (3.0 / max(1, nS)), 0.0, 1.0)
        bind = np.clip(binding_noise_base * max(0, nS - 3) / 3.0, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy
            if beta_wm_eff > 0:
                p_wm = np.exp(beta_wm_eff * (W_s - np.max(W_s)))
                p_wm /= np.sum(p_wm)
            else:
                p_wm = np.ones(nA) / nA

            # Arbitration
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            correct_a = state_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update
            if r == 1:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                down = wm_learn * 0.5
                w[s, a] = max(0.0, w[s, a] * (1.0 - down))
                # renormalize row
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = 1.0 / nA
                else:
                    w[s, :] /= row_sum

            # Binding interference (swap-like diffusion)
            if nS > 1:
                mean_other = (np.sum(w, axis=0) - w[s, :]) / (nS - 1)
                w[s, :] = (1.0 - bind) * w[s, :] + bind * mean_other

            # Ensure probabilities well-formed
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
