def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with PE-gated WM reliance and set-sizeâ€“dependent lapses (simulation).

    Parameters:
    - stimulus: sequence of stimuli (states) per trial (array-like, int-coded)
    - blocks: block index for each trial (array-like)
    - set_sizes: block-dependent set sizes on each trial (array-like; used for reference but inferred from stimulus)
    - correct_answer: correct action per trial (array-like)
    - parameters: [lr, softmax_beta, wm_weight_base, pe_gate_slope, lapse_base, lapse_setsize_slope]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, softmax_beta, wm_weight_base, pe_gate_slope, lapse_base, lapse_setsize_slope = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus)[block_idx].astype(int)
        block_correct = np.asarray(correct_answer)[block_idx].astype(int)

        nA = 3
        # Infer number of unique states in this block
        unique_states = np.unique(block_states)
        # Remap states to 0..nS-1 if needed
        state_map = {s: i for i, s in enumerate(unique_states)}
        remapped_states = np.array([state_map[s] for s in block_states], dtype=int)
        nS = len(unique_states)

        # Build a mapping from internal state id to its correct action
        correct_actions = np.zeros(nS, dtype=int)
        for s_ext, s_int in state_map.items():
            # pick first occurrence's correct action
            first_idx = np.where(block_states == s_ext)[0][0]
            correct_actions[s_int] = block_correct[first_idx]

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent lapse (per block)
        extra_load = max(0, nS - 3)
        lapse = 1.0 - np.exp(-(lapse_base + lapse_setsize_slope * extra_load))
        lapse = np.clip(lapse, 0.0, 1.0)

        for t in range(len(block_idx)):
            s = remapped_states[t]
            correct_a = correct_actions[s]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy softmax
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy softmax (nearly deterministic)
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Compute action-dependent WM gates using hypothetical RPE for each action
            # r_a = 1 if a == correct_action else 0
            r_vec = np.zeros(nA)
            r_vec[correct_a] = 1.0
            rpe_vec = r_vec - Q_s  # elementwise r - Q(s,a)
            gate_vec = 1.0 / (1.0 + np.exp(-pe_gate_slope * rpe_vec))
            wm_w_eff_vec = wm_weight_base * (1.0 - lapse) * gate_vec
            wm_w_eff_vec = np.clip(wm_w_eff_vec, 0.0, 1.0)

            # Mixture for each action
            p_mix = wm_w_eff_vec * p_wm + (1.0 - wm_w_eff_vec) * p_rl
            p_mix = np.maximum(p_mix, 1e-12)
            p_mix /= np.sum(p_mix)

            # Sample action
            a = np.random.choice(nA, p=p_mix)
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM update (supervised; decay toward uniform when unrewarded)
            if r > 0.5:
                w[s, :] = 0.0 * w[s, :] + w_0[s, :]
                w[s, a] = 1.0
            else:
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
