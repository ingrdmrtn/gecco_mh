def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with probabilistic recall policy and load-dependent recall plus WM decay.

    Parameters:
    - stimulus: sequence of stimuli for each trial (array of ints; state indices within block)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array) [used to determine nS per block]
    - correct_answer: correct action for each trial (array)
    - parameters: [lr, wm_weight, softmax_beta, rec_base, rec_sensitivity, wm_decay_base]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr, wm_weight, softmax_beta, rec_base, rec_sensitivity, wm_decay_base = parameters
    softmax_beta *= 10  # match fitting scale

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct_actions = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use the provided block set size (as in fitting code)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumes each state has a consistent correct action in the block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            mask = block_states == s
            # Fallback if state s not present at this point: default to first seen mapping
            if np.any(mask):
                correct_actions[s] = block_correct_actions[mask][0]
            else:
                # If a state doesn't appear in trials (edge case), default uniformly to 0
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent parameters (constant within block)
        load = max(0.0, (float(nS) - 3.0) / 3.0)
        rec_base_load = np.clip(rec_base * (1.0 - load), 0.0, 1.0)
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) ** (1.0 + load)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy: softmax over Q
            Q_center = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_center)
            p_rl /= np.sum(p_rl)

            # WM recall probability depends on base (load) and confidence (top-second)
            sorted_W = np.sort(W_s)[::-1]
            top = sorted_W[0]
            second = sorted_W[1] if nA > 1 else 0.0
            conf = max(0.0, top - second)
            p_recall = np.clip(rec_base_load + rec_sensitivity * conf, 0.0, 1.0)

            # WM policy: mixture of recall-delta on argmax and uniform guess
            argmax_w = int(np.argmax(W_s))
            p_wm_recall = np.zeros(nA)
            p_wm_recall[argmax_w] = 1.0  # deterministic recall
            p_wm_guess = np.ones(nA) / nA
            p_wm = p_recall * p_wm_recall + (1.0 - p_recall) * p_wm_guess

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward from task contingencies
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # Reward-gated WM consolidation
            if r > 0.0:
                alpha = min(1.0, 0.5 + 0.5 * conf)  # stronger consolidation when confident
                w[s, :] *= (1.0 - alpha)
                w[s, a] += alpha
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :].copy()

            tr += 1

    return simulated_actions, simulated_rewards
