def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate actions and rewards using the RL + Precision-limited WM model with set-sizeâ€“dependent retrieval lapses.

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta]

    Returns:
    - simulated_actions: array of chosen actions
    - simulated_rewards: array of obtained rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Set size follows fitting code (taken from set_sizes)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumed consistent within block)
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                state_to_correct[s] = block_correct[idx[0]]
            else:
                state_to_correct[s] = 0  # fallback if a state is never presented

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute set-size dependent parameters
        beta_wm_eff = softmax_beta_wm * (3.0 / max(1.0, float(nS)))
        lambda_wm = wm_weight * max(0.0, (float(nS) - 3.0) / max(1.0, float(nS)))
        decay_base = lr * (1.0 + max(0.0, (float(nS) - 3.0) / max(1.0, float(nS))) * wm_weight)
        decay_base = min(max(decay_base, 0.0), 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            x_rl = softmax_beta * Q_s
            x_rl -= np.max(x_rl)
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)

            # WM policy with retrieval lapse and set-size dependent precision
            x_wm = beta_wm_eff * W_s
            x_wm -= np.max(x_wm)
            p_wm_soft = np.exp(x_wm)
            p_wm_soft /= np.sum(p_wm_soft)
            p_uniform = np.ones(nA) / nA
            p_wm = (1.0 - lambda_wm) * p_wm_soft + lambda_wm * p_uniform

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for this state
            correct_a = int(state_to_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward prior, then reward-scaled bump to chosen action
            w[s, :] = (1.0 - decay_base) * w[s, :] + decay_base * w_0[s, :]
            w[s, a] += wm_weight * (0.5 + 0.5 * r)

            tr += 1

    return simulated_actions, simulated_rewards
