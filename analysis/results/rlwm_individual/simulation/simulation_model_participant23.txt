def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with interference-by-time (ISI) scaling of WM weight, set-size noise on WM precision,
    and global choice stickiness.

    Parameters:
    - stimulus: sequence of stimuli (state indices) for each trial (array-like of ints)
    - blocks: block index for each trial (array-like)
    - set_sizes: block-dependent set sizes on each trial (array-like of ints)
    - correct_answer: correct action for each trial (array-like of ints in [0, nA-1])
    - parameters: [lr, wm_weight, softmax_beta, decay, nu, stick]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_weight, softmax_beta, decay, nu, stick = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes, dtype=int)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size per block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # not used for updates, kept for parity

        # Track last time each state was seen for ISI-based WM mixing
        last_seen = -1 * np.ones(nS, dtype=int)

        # Global previous action for stickiness
        prev_action_global = -1

        # Set-size dependent WM precision
        beta_wm_eff = softmax_beta_wm * np.exp(-max(nu, 0.0) * max(nS - 3, 0))

        # Base WM mixture
        wm_mix_base = np.clip(wm_weight, 0.0, 1.0)

        for t_in_block, tr in enumerate(block_indices):
            s = block_states[t_in_block]

            # Compute ISI in trials since last visit of this state
            if last_seen[s] < 0:
                isi = 1e6  # effectively very large if never seen
            else:
                isi = t_in_block - last_seen[s]

            # ISI-scaled WM mixture
            wm_mix = wm_mix_base * np.exp(-max(decay, 0.0) * float(isi))
            wm_mix = np.clip(wm_mix, 0.0, 1.0)

            # Effective action values with global stickiness bias
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            if prev_action_global >= 0:
                bias = np.zeros(nA)
                bias[prev_action_global] = stick
                Q_eff = Q_s + bias / max(softmax_beta, 1e-8)
                W_eff = W_s + bias / max(beta_wm_eff, 1e-8)
            else:
                Q_eff = Q_s
                W_eff = W_s

            # RL softmax
            logits_rl = softmax_beta * (Q_eff - np.max(Q_eff))
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # WM softmax (set-size dependent precision)
            logits_wm = beta_wm_eff * (W_eff - np.max(W_eff))
            p_wm = np.exp(logits_wm)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Reward from correct mapping
            correct_a = block_correct[t_in_block]
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update: one-shot on correct reward, no overwrite on errors
            if r > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Bookkeeping
            last_seen[s] = t_in_block
            prev_action_global = a

    return simulated_actions, simulated_rewards
