def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM mixture with power-law set-size gating and time-based WM retention.
    - RL uses a single learning rate.
    - WM contribution is gated down by set size via a power-law factor nS^(-phi).
    - WM retention depends on time since last visit to the state via exponential retention lam^ISI.
    - WM encodes toward a one-hot on rewarded trials with strength eta_wm.

    Parameters:
    - stimulus: sequence of states for each trial (array of ints)
    - blocks: block index for each trial (array of ints)
    - set_sizes: set size on each trial (array of ints) [used to get nS per block]
    - correct_answer: correct action on each trial (array of ints)
    - parameters: [lr, wm_weight, softmax_beta, phi, lam, eta_wm]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_weight, softmax_beta, phi, lam, eta_wm = parameters
    softmax_beta *= 10.0  # match fitting code scaling
    softmax_beta_wm = 50.0  # very deterministic WM

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        # Per fitting code, nS is the set size for the block (first trial's set size)
        nS = int(block_set_sizes[0])

        # Map states to 0..nS-1 if not already; assume states are 0..nS-1
        # Build per-state correct action assuming it is consistent within a block
        correct_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            mask_s = (block_states == s)
            if np.any(mask_s):
                correct_by_state[s] = int(block_correct[mask_s][0])
            else:
                # If a state index is not present (shouldn't happen), default to 0
                correct_by_state[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last visit per state for WM decay
        last_visit = -np.ones(nS, dtype=int)

        # Set-size gate for WM mixture
        gate = (nS ** (-phi)) if nS > 0 else 1.0
        wm_weight_eff = wm_weight * gate

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Time-based WM decay only when the state is visited, using ISI since last visit
            if last_visit[s] >= 0:
                isi = t - last_visit[s]
                decay = (lam ** isi)
                w[s, :] = decay * w[s, :] + (1.0 - decay) * w_0[s, :]
            last_visit[s] = t

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            r = 1 if a == int(correct_by_state[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            q[s, a] += lr * (r - Q_s[a])

            # WM encoding on reward
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * target

            tr += 1

    return simulated_actions, simulated_rewards
