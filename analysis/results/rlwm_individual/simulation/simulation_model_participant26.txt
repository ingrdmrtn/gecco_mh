import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with RPE-gated arbitration and set-size cost (simulation).

    Parameters:
    - stimulus: sequence of stimuli (states) per trial (array of ints)
    - blocks: block index per trial (array of ints)
    - set_sizes: block set-size per trial (array of ints)
    - correct_answer: correct action per trial (array of ints)
    - parameters: [lr, wm_base, softmax_beta, wm_decay, arbi_rpe_gain, ss_cost_gain]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_base, softmax_beta, wm_decay, arbi_rpe_gain, ss_cost_gain = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_actions = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided block set size (constant within block, as in fitting code)
        nS = int(block_set_sizes[0])

        # Build state->correct action map within block from first occurrence of each state
        correct_actions = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t_idx, s in enumerate(block_states):
            s_int = int(s)
            if not seen[s_int]:
                correct_actions[s_int] = int(block_correct_actions[t_idx])
                seen[s_int] = True

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Arbitration baseline with set-size cost
        wm_weight_base = wm_base - ss_cost_gain * max(0, nS - 3)

        # Track last unsigned RPE per state to gate arbitration during decision
        last_abs_rpe = np.zeros(nS)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Arbitration weight using previous unsigned RPE for this state
            wm_weight_dyn = wm_weight_base + arbi_rpe_gain * (1.0 - last_abs_rpe[s])
            wm_weight_dyn = np.clip(wm_weight_dyn, 0.0, 1.0)

            # Softmax policies
            logits_rl = softmax_beta * Q_s
            logits_wm = softmax_beta_wm * W_s

            # Stable softmax
            max_rl = np.max(logits_rl)
            exp_rl = np.exp(logits_rl - max_rl)
            p_rl = exp_rl / np.sum(exp_rl)

            max_wm = np.max(logits_wm)
            exp_wm = np.exp(logits_wm - max_wm)
            p_wm = exp_wm / np.sum(exp_wm)

            # Mixture policy
            p_total = wm_weight_dyn * p_wm + (1.0 - wm_weight_dyn) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # Compute RPE (for updates and next-trial arbitration gating)
            rpe = r - Q_s[a]
            last_abs_rpe[s] = np.abs(rpe)

            # RL update
            q[s, a] += lr * rpe

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM rewarded storage
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - lr) * w[s, :] + lr * onehot

            tr += 1

    return simulated_actions, simulated_rewards
