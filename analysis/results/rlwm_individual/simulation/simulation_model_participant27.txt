import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards from the RL+WM model with PE-gated arbitration and set-size scaling.

    Inputs:
    - stimulus: array of state IDs per trial (ints)
    - blocks: array of block IDs per trial (ints)
    - set_sizes: array of set size per trial (ints; constant within a block)
    - correct_answer: array of correct action per trial (ints in [0,1,2])
    - parameters: (lr, wm_weight_init, softmax_beta, alpha_wm, pe_gain)

    Returns:
    - simulated_actions: array of chosen actions per trial (ints in [0,1,2])
    - simulated_rewards: array of rewards per trial (0/1)
    """

    lr, wm_weight_init, softmax_beta, alpha_wm, pe_gain = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    g_base = max(pe_gain, 0.0) / 2.0
    xi_size = min(max(pe_gain, 0.0) / 2.0, 4.0)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])


        # Map each state to its correct action (assumed consistent within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                correct_actions[s] = int(block_correct[idx_s[0]])
            else:
                # If a state is not encountered (edge case), default to 0
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Initialize arbitration weight (sigmoid-squashed)
        wm_weight_dyn = 1.0 / (1.0 + np.exp(-5.0 * (wm_weight_init - 0.5)))

        # Set-sizeâ€“scaled PE sensitivity (constant within block)
        pe_sens = g_base * (3.0 / float(nS)) ** xi_size

        for t_local, tr in enumerate(block_indices):
            s = int(block_states[t_local])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax probabilities

            logits_rl = softmax_beta * Q_s
            logits_wm = softmax_beta_wm * W_s

            max_rl = np.max(logits_rl)
            exp_rl = np.exp(logits_rl - max_rl)
            p_rl = exp_rl / np.sum(exp_rl)

            max_wm = np.max(logits_wm)
            exp_wm = np.exp(logits_wm - max_wm)
            p_wm = exp_wm / np.sum(exp_wm)

            # Mixture policy
            p_total = wm_weight_dyn * p_wm + (1.0 - wm_weight_dyn) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # Arbitration update (PE-gated, reward-modulated)
            pe = abs(r - Q_s[a])
            target_wm = r * np.exp(-pe_sens * pe)
            wm_weight_dyn = wm_weight_dyn + (1.0 - wm_weight_dyn) * target_wm - wm_weight_dyn * (1.0 - target_wm)
            wm_weight_dyn = min(max(wm_weight_dyn, 1e-6), 1.0 - 1e-6)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update (reward-gated Hebbian-like)
            eta = max(alpha_wm, 0.0) * (0.5 + 0.5 * r)
            if r > 0:
                tgt = np.zeros(nA)
                tgt[a] = 1.0
            else:
                tgt = w_0[s, :]
            w[s, :] = (1.0 - eta) * w[s, :] + eta * tgt

            # Normalize WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

    return simulated_actions, simulated_rewards
