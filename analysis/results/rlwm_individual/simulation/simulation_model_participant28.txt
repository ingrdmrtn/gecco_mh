def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with set-sizeâ€“dependent swap errors and RL temperature load-sensitivity.

    Inputs:
    - stimulus: state index on each trial (array of ints)
    - blocks: block index for each trial (array of ints)
    - set_sizes: set size for the block on each trial (array of ints)
    - correct_answer: correct action on each trial (array of ints)
    - parameters: [lr, wm_weight, softmax_beta, swap_base, beta_load_slope]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints: 0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, swap_base, beta_load_slope = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]
        block_correct = correct_answer[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (take first occurrence within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                correct_actions[s] = block_correct[idx_s[0]]
            else:
                # Fallback if a state is not present (should not happen); pick arbitrary
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent components
        swap = 1.0 - np.exp(-swap_base * max(0.0, float(nS) - 1.0))
        beta_base = softmax_beta
        beta_eff = beta_base / (1.0 + beta_load_slope * (float(nS) - 3.0))
        beta_eff = max(1e-6, beta_eff)

        for t in range(len(block_states)):
            global_t = block_indices[t]
            s = int(block_states[t])

            # RL policy
            Q_s = q[s, :].copy()
            z_rl = Q_s - np.max(Q_s)
            p_rl = np.exp(beta_eff * z_rl)
            p_rl /= np.sum(p_rl)

            # WM policy with swap
            W_s = w[s, :].copy()
            z_wm_self = W_s - np.max(W_s)
            p_wm_self = np.exp(softmax_beta_wm * z_wm_self)
            p_wm_self /= np.sum(p_wm_self)

            if nS > 1:
                p_others_list = []
                for j in range(nS):
                    if j == s:
                        continue
                    W_j = w[j, :].copy()
                    z_wm_j = W_j - np.max(W_j)
                    p_wm_j = np.exp(softmax_beta_wm * z_wm_j)
                    p_wm_j /= np.sum(p_wm_j)
                    p_others_list.append(p_wm_j)
                p_wm_others = np.mean(p_others_list, axis=0) if len(p_others_list) > 0 else (np.ones(nA) / nA)
            else:
                p_wm_others = np.ones(nA) / nA

            p_wm = (1.0 - swap) * p_wm_self + swap * p_wm_others

            # Mixture and action sampling
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, None)
            p_total /= np.sum(p_total)
            a = int(np.random.choice(nA, p=p_total))

            simulated_actions[global_t] = a

            # Reward from environment
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[global_t] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update: store last rewarded action for this state
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

    return simulated_actions, simulated_rewards
