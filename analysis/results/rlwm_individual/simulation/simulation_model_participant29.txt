import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + Hebbian choice kernel as WM with load-dependent interference.

    Parameters:
    - stimulus: sequence of stimuli (state indices) for each trial (array of ints)
    - blocks: block index for each trial (array of ints)
    - set_sizes: block-dependent set sizes on each trial (array of ints)
    - correct_answer: correct action for each trial (array of ints)
    - parameters: [lr, wm_weight, softmax_beta, hebb_eta, reward_gain, load_interf]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_weight, softmax_beta, hebb_eta, reward_gain, load_interf = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size per block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within this block
        # Assumes that for each state, the correct action is consistent within the block
        unique_states = np.unique(block_states)
        # Create a mapping from state index (0..nS-1) to correct action
        state_to_correct = np.zeros(nS, dtype=int)
        for st in unique_states:
            idx = np.where(block_states == st)[0][0]
            state_to_correct[st] = int(block_correct[idx])

        # Effective WM learning rate given load
        wm_eta_eff = hebb_eta / (1.0 + max(0.0, load_interf * float(nS - 3)))

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Compute policy components
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl = p_rl / np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm = p_wm / np.sum(p_wm)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for current state
            w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

            # Hebbian pulse toward chosen action with reward-modulated gain
            pulse = np.zeros(nA)
            pulse[a] = 1.0
            gain = 1.0 + reward_gain * (2.0 * r - 1.0)  # >1 on reward if reward_gain>0; <1 on no-reward

            w[s, :] = w[s, :] + wm_eta_eff * gain * (pulse - w[s, :])

            # Renormalize WM row to keep it a proper distribution
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] = w[s, :] / row_sum
            else:
                w[s, :] = w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
