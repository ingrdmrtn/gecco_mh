def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Hybrid RL + recall-based WM with lapses and asymmetric RL learning.

    Parameters
    ----------
    stimulus : array-like of int
        State index on each trial (assumed to be 0..nS-1 within a block).
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size for each trial (used to set recall probability; assumed constant within a block).
    correct_answer : array-like of int
        Correct action (0..2) for each trial.
    parameters : sequence
        lr_pos : float in (0,1)
            RL learning rate for positive prediction errors.
        wm_weight : float in [0,1]
            Base mixture weight on WM policy.
        softmax_beta : float >= 0
            Inverse-temperature for RL softmax (scaled up internally).
        lr_neg : float in (0,1)
            RL learning rate for negative prediction errors.
        wm_lapse : float in [0,1]
            Lapse probability in WM retrieval (adds uniform noise).
        recall_slope : float >= 0
            Linear decrement of WM recall from set size 3 to 6:
            recall_prob = max(0, 1 - recall_slope*(set_size - 3)).

    Returns
    -------
    simulated_actions : array of int
        Simulated action choices (0..2) for each trial.
    simulated_rewards : array of int
        Simulated binary rewards (0/1) for each trial.
    """
    import numpy as np

    lr_pos = parameters[0]
    wm_weight = parameters[1]
    softmax_beta = parameters[2] * 10.0  # scale as in fitting code
    lr_neg = parameters[3] if len(parameters) > 3 else lr_pos
    wm_lapse = parameters[4] if len(parameters) > 4 else 0.1
    recall_slope = parameters[5] if len(parameters) > 5 else 0.3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    wm_weight_eff = float(np.clip(wm_weight, 0.0, 1.0))

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus)[block_idx].astype(int)
        block_correct = np.asarray(correct_answer)[block_idx].astype(int)
        block_set_sizes = np.asarray(set_sizes)[block_idx].astype(int)

        # Number of states assumed to equal the set size within the block (as in fitting code)
        nS = int(block_set_sizes[0])

        # Initialize RL values and WM store
        q = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        has_mem = np.zeros(nS, dtype=bool)
        mem_act = np.zeros(nS, dtype=int)

        for t in range(len(block_states)):
            global_t = block_idx[t]
            s = int(block_states[t])

            # RL softmax policy
            Q_s = q[s, :]
            logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(logits)
            p_rl /= np.sum(p_rl)

            # WM recall probability depends on set size at this trial
            set_size_t = int(block_set_sizes[t])
            recall_prob = max(0.0, 1.0 - recall_slope * (set_size_t - 3))
            eff_recall = recall_prob * (1.0 - wm_lapse)

            # WM policy: uniform unless a memory exists; then add mass to stored action
            if has_mem[s]:
                p_wm = (1.0 - eff_recall) * w_0[s, :].copy()
                p_wm[int(mem_act[s])] += eff_recall
            else:
                p_wm = w_0[s, :].copy()

            # Mixture of WM and RL
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[global_t] = a

            # Generate reward from correct answer
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[global_t] = r

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            eta = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += eta * pe

            # WM memory update: store on reward; forget if the stored action is proven wrong
            if r >= 0.5:
                has_mem[s] = True
                mem_act[s] = a
            else:
                if has_mem[s] and mem_act[s] == a:
                    has_mem[s] = False

    return simulated_actions, simulated_rewards
