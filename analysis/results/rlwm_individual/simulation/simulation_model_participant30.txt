def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using:
    - RL: single learning rate; softmax policy.
    - WM: Dirichlet-like counts per state decayed toward a symmetric prior.
    - Interference: WM decay increases with set size: wm_decay_eff = wm_decay_base * (nS/3), clipped to [0,1].
    - Perseveration bias toward the last action in a state.
    - Mixture of WM and RL policies.

    Inputs:
    - stimulus: array of state indices per trial (ints)
    - blocks: array of block indices per trial
    - set_sizes: array with set size per trial (ints)
    - correct_answer: array of correct action per trial (ints in [0..2])
    - parameters: [lr, wm_weight, softmax_beta, wm_decay_base, wm_alpha, pers_bias]

    Returns:
    - simulated_actions: int array of chosen actions per trial
    - simulated_rewards: int array of rewards (0/1) per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay_base, wm_alpha, pers_bias = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct_ans = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # In fitting, nS is taken from set size constant within block
        nS = int(block_set_sizes[0])

        # Map each encountered state to its correct action (assume consistent within block)
        # States are assumed to be labeled 0..nS-1
        state_ids = np.unique(block_states)
        # Build per-state correct action lookup
        correct_actions = np.zeros(nS, dtype=int)
        for s in state_ids:
            first_idx = np.where(block_states == s)[0][0]
            correct_actions[s] = block_correct_ans[first_idx]

        # Initialize RL and WM structures
        q = (1.0 / nA) * np.ones((nS, nA))
        counts = wm_alpha * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        # Effective WM decay scales with set size
        wm_decay_eff = wm_decay_base * (float(nS) / 3.0)
        wm_decay_eff = np.clip(wm_decay_eff, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            Q_s = q[s, :]
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy from normalized counts with perseveration bias
            c_s = counts[s, :].copy()
            c_sum = np.sum(c_s)
            if c_sum <= 0:
                W_s = w_0[s, :].copy()
            else:
                W_s = c_s / c_sum

            W_logits = W_s.copy()
            if last_action[s] >= 0:
                W_logits[last_action[s]] += pers_bias

            wm_logits = softmax_beta_wm * (W_logits - np.max(W_logits))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward prior for current state only
            counts[s, :] = (1.0 - wm_decay_eff) * counts[s, :] + wm_decay_eff * wm_alpha

            # WM update given outcome
            if r > 0:
                counts[s, a] += 1.0
            else:
                penal = min(0.2, counts[s, a] - 1e-6)
                counts[s, a] -= penal

            # Update perseveration tracker
            last_action[s] = a

            tr += 1

    return simulated_actions, simulated_rewards
