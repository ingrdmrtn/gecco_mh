def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using:
    - RL with asymmetric learning rates (lr_pos, lr_neg)
    - WM cache that stores the last rewarded action per state (no leak)
    - Arbitration between WM and RL with set-sizeâ€“specific WM weights

    Parameters
    - stimulus: array of state IDs per trial
    - blocks: array of block IDs per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct actions per trial
    - parameters: [lr_pos, lr_neg, softmax_beta, wm_weight_small, wm_weight_large]

    Returns
    - simulated_actions: array of simulated actions per trial
    - simulated_rewards: array of simulated binary rewards per trial
    """
    lr_pos, lr_neg, softmax_beta, wm_weight_small, wm_weight_large = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map from state -> correct action assumed stationary within block
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            st_mask = (block_states == st)
            # assume each state appears; fall back to first if needed
            if np.any(st_mask):
                correct_actions[st] = int(block_correct[st_mask][0])
            else:
                correct_actions[st] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))

        wm_weight_eff = np.clip(wm_weight_small if nS <= 3 else wm_weight_large, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(z_rl)
            pi_rl /= np.sum(pi_rl)

            # WM policy
            z_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(z_wm)
            pi_wm /= np.sum(pi_wm)

            # Mixture
            p_total = wm_weight_eff * pi_wm + (1.0 - wm_weight_eff) * pi_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # RL update (asymmetric learning rates)
            delta = r - Q_s[a]
            lr = lr_pos if delta >= 0.0 else lr_neg
            q[s, a] += lr * delta

            # WM update: cache last rewarded action, no leak
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

            tr += 1

    return simulated_actions, simulated_rewards
