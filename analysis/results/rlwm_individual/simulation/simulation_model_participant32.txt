def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the RL+WM model with rehearsal, set-sizeâ€“dependent decay, and state-wise stickiness.

    Inputs:
    - stimulus: array of state indices per trial (assumed 0..nS-1 within block)
    - blocks: array of block index per trial
    - set_sizes: array of set size per trial (constant within each block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_rehearsal, wm_decay, stickiness]

    Returns:
    - simulated_actions: array of simulated action choices
    - simulated_rewards: array of simulated binary rewards
    """
    lr, wm_weight, softmax_beta, wm_rehearsal, wm_decay, stickiness = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map per-state correct action (assumes consistent mapping within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first occurrence of this state in the block to get its correct action
            mask = (block_states == s)
            if np.any(mask):
                correct_actions[s] = int(block_correct[mask][0])
            else:
                # If a state doesn't occur (edge case), default to 0
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax
            q_logits = softmax_beta * Q_s
            q_logits = q_logits - np.max(q_logits)
            p_rl = np.exp(q_logits)
            p_rl /= np.sum(p_rl)

            # WM with stickiness bias on logits, then near-deterministic softmax
            wm_logits = W_s.copy()
            if last_action[s] >= 0:
                bias_vec = np.zeros(nA)
                bias_vec[last_action[s]] = stickiness
                wm_logits = wm_logits + bias_vec
            wm_logits = softmax_beta_wm * wm_logits
            wm_logits = wm_logits - np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, eps)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for state
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM global decay toward uniform, scaled by set size
            decay_eff = min(1.0, wm_decay * (float(nS) / 3.0))
            w = (1.0 - decay_eff) * w + decay_eff * w_0

            # WM rehearsal for the visited state
            onehot = np.zeros(nA)
            onehot[a] = 1.0
            w[s, :] = (1.0 - wm_rehearsal) * w[s, :] + wm_rehearsal * onehot

            # Additional reinforcement on rewarded trials
            if r > 0:
                w[s, :] = 0.5 * w[s, :] + 0.5 * onehot

            # Ensure valid probabilities for the visited state
            w[s, :] = np.clip(w[s, :], eps, None)
            w[s, :] /= np.sum(w[s, :])

            # Update stickiness memory
            last_action[s] = a

            tr += 1

    return simulated_actions, simulated_rewards
