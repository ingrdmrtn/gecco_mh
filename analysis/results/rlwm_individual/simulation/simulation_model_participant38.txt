def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL with Q-value forgetting + WM with set-size-dependent interference (simulation).

    Parameters:
    - stimulus: array of state identifiers for each trial (can be any labels; remapped within block)
    - blocks: array of block indices for each trial
    - set_sizes: array giving set size per trial (constant within block)
    - correct_answer: array of correct action per trial (0..nA-1)
    - parameters: [lr, q_forget, softmax_beta, wm_weight, wm_interf, wm_gain]

    Returns:
    - simulated_actions: array of sampled actions (0..2)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, q_forget, softmax_beta, wm_weight, wm_interf, wm_gain = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate blocks; keep original trial order by writing back to true indices
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_indices]
        block_correct_actions_raw = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]
        nA = 3

        # Remap arbitrary state labels within this block to 0..nS-1
        unique_states = np.unique(block_states_raw)
        # Use provided set size to match the fitted model's interference scaling
        nS = int(block_set_sizes[0])
        # In case labels are fewer/more than provided nS, we still map on observed states
        state_to_idx = {s_val: i for i, s_val in enumerate(sorted(unique_states))}
        # Build per-state correct action mapping
        correct_actions = np.zeros(len(unique_states), dtype=int)
        for s_val in unique_states:
            s_mask = (block_states_raw == s_val)
            correct_actions[state_to_idx[s_val]] = int(block_correct_actions_raw[s_mask][0])

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((len(unique_states), nA))
        w = (1.0 / nA) * np.ones((len(unique_states), nA))
        w_0 = (1.0 / nA) * np.ones((len(unique_states), nA))

        # WM interference depends on intended set size (as in fitting code)
        interf_factor = (nS - 1.0) / nS if nS > 0 else 0.0
        wm_interf_block = wm_interf * interf_factor

        for t, tr_idx in enumerate(block_indices):
            # Current state (remapped)
            s = int(state_to_idx[block_states_raw[t]])
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            x_rl = softmax_beta * Q_s
            x_rl -= np.max(x_rl)
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (very deterministic)
            x_wm = softmax_beta_wm * W_s
            x_wm -= np.max(x_wm)
            p_wm = np.exp(x_wm)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            # Normalize for safety
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr_idx] = a

            # Reward based on correct action for this state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr_idx] = r

            # RL update with forgetting toward uniform
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * w_0[s, :]

            # WM global interference: rows move toward their grand mean
            mean_row = np.mean(w, axis=0)
            w = (1.0 - wm_interf_block) * w + wm_interf_block * np.tile(mean_row, (w.shape[0], 1))

            # WM update on current state's row
            if r > 0.5:
                # Move mass toward chosen action
                w[s, :] = (1.0 - wm_gain) * w[s, :]
                w[s, a] += wm_gain
            else:
                # Move mass away from chosen action, distribute to others
                dec = min(wm_gain, w[s, a])
                w[s, a] -= dec
                inc = dec / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += inc

            # Normalize WM row; fallback to uniform if degenerate
            row_sum = np.sum(w[s, :])
            if row_sum > 1e-12:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

    return simulated_actions, simulated_rewards
