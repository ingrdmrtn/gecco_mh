import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards for the RL+WM mixture model with capacity-limited WM and set-size dependent forgetting.

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, lr_wm, rho, k_capacity]

    Returns:
    - simulated_actions: array of simulated actions per trial
    - simulated_rewards: array of simulated binary rewards per trial
    """
    lr, wm_weight, softmax_beta, lr_wm, rho, k_capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (take first occurrence)
        unique_states = np.unique(block_states)
        # Ensure states are 0..nS-1; if not, create a mapping
        if unique_states.min() != 0 or unique_states.max() != nS - 1 or len(unique_states) != nS:
            # Remap to 0..nS-1
            state_map = {s: i for i, s in enumerate(sorted(unique_states))}
            remapped_states = np.array([state_map[s] for s in block_states], dtype=int)
        else:
            remapped_states = block_states

        correct_actions = np.zeros(nS, dtype=int)
        filled = np.zeros(nS, dtype=bool)
        for t_idx, s in enumerate(remapped_states):
            if not filled[s]:
                correct_actions[s] = block_correct[t_idx]
                filled[s] = True

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent scaling
        cap_factor = min(1.0, max(0.0, float(k_capacity)) / float(nS))
        wm_weight_block = wm_weight * cap_factor
        rho_eff = min(0.999, max(0.0, float(rho)) * (float(nS) / 3.0))

        # Iterate trials in block
        for t in range(len(block_states)):
            s = remapped_states[t]

            # Apply WM decay toward uniform baseline
            w = (1.0 - rho_eff) * w + rho_eff * w_0

            # Current state value vectors
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Policies (softmax with numerical stability)
            Q_s_shift = Q_s - np.max(Q_s)
            W_s_shift = W_s - np.max(W_s)
            p_rl = np.exp(softmax_beta * Q_s_shift)
            p_rl /= np.sum(p_rl)
            p_wm = np.exp(softmax_beta_wm * W_s_shift)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Reward based on correct action for this state
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update (using pre-update W_s)
            w[s, a] += lr_wm * (r - W_s[a])

            tr += 1

    return simulated_actions, simulated_rewards
