import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with reward-gated WM writes, set-size–modulated WM decay, and choice stickiness.

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_decay_base, eta_size, kappa]

    Returns:
    - simulated_actions: array of simulated action choices
    - simulated_rewards: array of simulated binary rewards
    """
    lr, wm_weight, softmax_beta, wm_decay_base, eta_size, kappa = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    wm_weight = np.clip(wm_weight, 0.0, 1.0)
    eta_size = max(0.0, eta_size)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    eps = 1e-12

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        nS = int(set_sizes[block_idx][0])

        # Initialize value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size–modulated effective WM decay
        wm_decay_eff = 1.0 - (1.0 - np.clip(wm_decay_base, 0.0, 1.0)) ** (1.0 + eta_size * max(0, nS - 3))

        # Map each state to its correct action within the block (assumes states indexed 0..nS-1)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first occurrence of this state in the block to read its correct action
            where_s = np.where(block_states == s)[0]
            if len(where_s) > 0:
                correct_actions[s] = int(block_correct[where_s[0]])
            else:
                # Fallback (should not happen if states are present): default to 0
                correct_actions[s] = 0

        prev_a = None

        for t, gi in enumerate(block_idx):
            s = int(stimulus[gi])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Add stickiness to the last chosen action
            if prev_a is not None:
                Q_s[prev_a] += kappa
                W_s[prev_a] += kappa

            # RL policy (softmax)
            x_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(x_rl)
            p_rl /= max(np.sum(p_rl), eps)

            # WM policy (high inverse temperature)
            x_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(x_wm)
            p_wm /= max(np.sum(p_wm), eps)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, 0.0)
            p_total /= max(np.sum(p_total), eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[gi] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[gi] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform for the current state
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # Reward-gated WM write
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                write_strength = wm_decay_eff
                w[s, :] = (1.0 - write_strength) * w[s, :] + write_strength * one_hot

            # Update stickiness memory
            prev_a = a

    return simulated_actions, simulated_rewards
