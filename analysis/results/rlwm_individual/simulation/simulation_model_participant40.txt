import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with capacity scaling and WM decay/store strength.

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, capacity_K, wm_decay, wm_store]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_weight, softmax_beta, capacity_K, wm_decay, wm_store = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within this block
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                state_to_correct[s] = block_correct[idx[0]]

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity scaling of WM weight
        wm_scale = min(1.0, max(0.0, capacity_K / float(nS)))
        eff_wm_weight = wm_weight * wm_scale

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy (more deterministic)
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = eff_wm_weight * p_wm + (1.0 - eff_wm_weight) * p_rl

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Outcome
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_store) * w[s, :] + wm_store * one_hot

            tr += 1

    return simulated_actions, simulated_rewards
