def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + graded WM mixture with capacity- and decay-adjusted WM (simulation).

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (used to infer nS per block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_capacity]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Per fitting code, set size is constant within a block; take the first
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumes consistency within block)
        # States are expected to be 0..nS-1 as in fitting
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            s_mask = block_states == s
            if np.any(s_mask):
                state_to_correct[s] = int(block_correct[s_mask][0])
            else:
                # If a state does not occur (unlikely), default to action 0
                state_to_correct[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM arbitration weight for this block
        wm_weight_eff = wm_weight * min(1.0, wm_capacity / max(1.0, nS))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl /= np.sum(prl)

            # WM softmax (high determinism)
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm /= np.sum(pwm)

            # Mixture policy
            p_total = wm_weight_eff * pwm + (1.0 - wm_weight_eff) * prl

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for the state
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform, then WM update with same lr
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            W_s = w[s, :]  # refreshed after decay
            delta_wm = r - W_s[a]
            w[s, a] += lr * delta_wm

            tr += 1

    return simulated_actions, simulated_rewards
