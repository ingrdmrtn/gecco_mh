def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards for the RLWM model with capacity-limited, decaying one-shot working memory and RL integration.

    Inputs:
    - stimulus: array of state indices per trial (int)
    - blocks: array of block indices per trial (int)
    - set_sizes: array of set size per trial (int) [used to determine nS per block]
    - correct_answer: array of correct action per trial (int)
    - parameters: list [lr, wm_weight_base, softmax_beta, wm_decay, wm_capacity]

    Returns:
    - simulated_actions: array of sampled actions (int)
    - simulated_rewards: array of binary rewards (int; 1 if correct, 0 otherwise)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, wm_decay, wm_capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus[block_idx], dtype=int)
        block_correct = np.asarray(correct_answer[block_idx], dtype=int)
        block_set_sizes = np.asarray(set_sizes[block_idx], dtype=int)

        nA = 3
        # Use block set size as in the fitting code
        nS = int(block_set_sizes[0])

        # Build a mapping from state -> correct action (assumes one correct action per state in a block)
        state_correct = np.zeros(nS, dtype=int)
        # If states are guaranteed 0..nS-1 this will work directly
        for s_val in np.unique(block_states):
            first_idx = np.where(block_states == s_val)[0][0]
            state_correct[s_val] = block_correct[first_idx]

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight for this block (capacity-limited)
        wm_weight_eff_block = wm_weight_base * min(1.0, float(wm_capacity) / max(1.0, float(nS)))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Decay WM table toward uniform before making the choice
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            q_centered = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(q_centered)
            p_rl /= np.sum(p_rl)

            # WM softmax
            w_centered = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(w_centered)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_eff_block * p_wm + (1.0 - wm_weight_eff_block) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            correct_a = int(state_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            q[s, a] += lr * (r - Q_s[a])

            # WM update: store rewarded action deterministically when rewarded
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
