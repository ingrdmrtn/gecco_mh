def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with WM decay and set-size-dependent WM engagement.

    Inputs:
    - stimulus: array of states per trial (integers)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial (integers 0..2)
    - parameters: (lr, wm_weight, softmax_beta, wm_decay, wm_neg)

    Returns:
    - simulated_actions: array of simulated choices per trial
    - simulated_rewards: array of simulated rewards (0/1) per trial
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_neg = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        wm_weight_eff = wm_weight * (3.0 / max(1.0, nS))

        for t in range(len(block_idx)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl /= np.sum(prl)

            # WM policy (near-deterministic)
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm /= np.sum(pwm)

            # Mixture
            p_total = wm_weight_eff * pwm + (1.0 - wm_weight_eff) * prl
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            q[s, a] += lr * (r - Q_s[a])

            # WM decay toward uniform for the visited state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM feedback update
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot
            else:
                w[s, :] = (1.0 - wm_neg) * w[s, :] + wm_neg * w_0[s, :]

    return simulated_actions, simulated_rewards
