def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with load-scaled WM weight and decay (simulation).

    Inputs:
    - stimulus: array of state indices per trial (assumed 0..nS-1 within a block)
    - blocks: array of block indices per trial
    - set_sizes: array indicating set size on each trial (constant within a block)
    - correct_answer: array of correct action per trial (0..2)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, load_sensitivity]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    lr, wm_weight, softmax_beta, wm_decay, load_sensitivity = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct_actions = correct_answer[block_indices].astype(int)

        nA = 3
        # Use provided set size for the block
        nS = int(set_sizes[block_indices][0])

        # Map per-state correct action from first occurrence within the block
        # Assumes states are labeled 0..nS-1 within a block
        per_state_correct = np.zeros(nS, dtype=int)
        for s_id in range(nS):
            # find first trial in block where state equals s_id
            idx = np.where(block_states == s_id)[0]
            if len(idx) > 0:
                per_state_correct[s_id] = int(block_correct_actions[idx[0]])
            else:
                # if a state is never shown (rare), default to 0
                per_state_correct[s_id] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight for this block based on set size
        wm_weight_eff = wm_weight * np.exp(-load_sensitivity * (nS - 3))
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute RL policy
            Q_s = q[s, :]
            Q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_shift)
            p_rl /= np.sum(p_rl)

            # Compute WM policy
            W_s = w[s, :]
            W_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_shift)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Determine reward from ground-truth mapping
            correct_a = per_state_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM one-shot encoding on rewarded trials (store last rewarded action)
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
