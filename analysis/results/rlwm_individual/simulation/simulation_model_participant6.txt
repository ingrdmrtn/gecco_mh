import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards for:
    RL (valence-asymmetric) + WM with prediction-error-based gating and load-dependent WM forgetting.

    Inputs:
    - stimulus: array of state/stimulus indices per trial (int)
    - blocks: array of block indices per trial (int)
    - set_sizes: array of set size for the block on each trial (int)
    - correct_answer: array of correct action per trial (int in [0..nA-1])
    - parameters: [lr_pos, lr_neg, softmax_beta, wm_weight, pe_sensitivity, wm_forgetting]

    Returns:
    - simulated_actions: array of chosen actions (int)
    - simulated_rewards: array of rewards (0/1)
    """
    lr_pos, lr_neg, softmax_beta, wm_weight, pe_sensitivity, wm_forgetting = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use block-declared set size (consistent with fitting code)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM forgetting (effective per block)
        load_scale = max(1.0, nS / 3.0)
        wm_forgetting_eff = np.clip(wm_forgetting * load_scale, 0.0, 1.0)

        for t in range(len(block_states)):
            s = block_states[t]
            correct_a = block_correct[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probabilities
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM softmax probabilities (near-deterministic)
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Action-specific PE-based gating (uses counterfactual reward per action)
            # r_a = 1 if action equals correct action, else 0
            r_vec = np.zeros(nA)
            r_vec[correct_a] = 1.0
            pe_vec = r_vec - Q_s  # elementwise PE per action
            wm_weight_eff_vec = wm_weight * np.exp(-pe_sensitivity * np.abs(pe_vec))

            # Mixture policy per action
            p_total = wm_weight_eff_vec * p_wm + (1.0 - wm_weight_eff_vec) * p_rl
            # Normalize to guard against small numerical drift
            p_total = np.clip(p_total, 1e-12, None)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Realized reward and PE for chosen action
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r
            pe = r - Q_s[a]

            # RL update (valence asymmetric)
            if pe >= 0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM forgetting
            w = (1.0 - wm_forgetting_eff) * w + wm_forgetting_eff * w_0

            # WM update on rewarded trials: store the correct association deterministically
            if r > 0:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0

            tr_global += 1

    return simulated_actions, simulated_rewards
