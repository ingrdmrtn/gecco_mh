def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the RL+WM model with load-induced interference
    and graded error-based encoding (mirrors cognitive_model1).

    Inputs:
    - stimulus: array of states per trial (0..nS-1 within block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (3 or 6)
    - correct_answer: array of correct action per trial (0..2)
    - parameters: (lr, wm_weight_base, softmax_beta, interference, wm_beta_scale)

    Returns:
    - simulated_actions: array of simulated actions (0..2)
    - simulated_rewards: array of simulated rewards (0/1)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, interference, wm_beta_scale = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 10.0 * max(1e-6, wm_beta_scale)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    trial_idx = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]
        block_correct = correct_answer[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # assume each state appears at least once per block
            correct_actions[s] = block_correct[block_states == s][0]

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # initialized but not used further, as in fitting code

        # Compute load-dependent mixture and interference
        load_factor = 1.0 / (1.0 + max(0.0, interference) * max(0, nS - 3))
        wm_weight_eff = np.clip(wm_weight_base * load_factor, 0.0, 1.0)

        for t in range(len(block_states)):
            tr = block_indices[t]
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Generate reward based on correct action for this state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM interference (blend toward block mean, stronger at higher load)
            w_mean = np.mean(w, axis=0)
            blend_strength = 1.0 - load_factor
            w[s, :] = (1.0 - blend_strength) * w[s, :] + blend_strength * w_mean

            # WM graded encoding
            onehot = np.zeros(nA)
            onehot[a] = 1.0
            if r > 0:
                enc_pos = 0.6 * load_factor + 0.2
                w[s, :] = (1.0 - enc_pos) * w[s, :] + enc_pos * onehot
            else:
                enc_neg = 0.2 * load_factor + 0.05
                anti = (1.0 - onehot) / (nA - 1)
                w[s, :] = (1.0 - enc_neg) * w[s, :] + enc_neg * anti

            # Normalize WM state distribution
            w[s, :] = np.clip(w[s, :], 1e-9, None)
            w[s, :] /= np.sum(w[s, :])

    return simulated_actions, simulated_rewards
