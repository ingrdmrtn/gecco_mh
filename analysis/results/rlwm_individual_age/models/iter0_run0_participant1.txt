def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with set-size and age-dependent WM gating and decay.

    Idea:
    - Choices are generated by a mixture of a model-free RL system and a one-shot Working Memory (WM) system.
    - WM stores deterministic stimulus-action associations upon reward and decays toward uniform with time.
    - The contribution of WM is reduced under higher set-size (6) and for older participants.
    - RL updates Q-values with a single learning rate.

    Parameters
    - model_parameters: list or array with 6 parameters
        0. lr: RL learning rate in [0,1]
        1. wm_weight: base mixture weight of WM in [0,1]
        2. softmax_beta: inverse temperature for RL softmax; internally multiplied by 10
        3. wm_decay: per-trial decay of WM toward uniform in [0,1]
        4. wm_setsize_cost: non-negative; how strongly WM degrades when set size increases (from 3 to 6)
        5. wm_age_cost: in [0,1]; multiplicative WM reduction for older group

    Inputs
    - states: array-like of state indices per trial
    - actions: array-like of chosen action indices per trial (0..2)
    - rewards: array-like of rewards per trial (0/1)
    - blocks: array-like block index per trial
    - set_sizes: array-like set size (3 or 6) per trial
    - age: array-like with constant age per trial

    Returns
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_setsize_cost, wm_age_cost = model_parameters

    # Temperature scaling as specified
    softmax_beta *= 10

    # Age group coding: 0 = young (<=45), 1 = old (>45)
    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    # Clamp key parameters to reasonable ranges without importing special functions
    lr = min(max(lr, 0.0), 1.0)
    wm_weight = min(max(wm_weight, 0.0), 1.0)
    wm_decay = min(max(wm_decay, 0.0), 1.0)
    wm_setsize_cost = max(wm_setsize_cost, 0.0)
    wm_age_cost = min(max(wm_age_cost, 0.0), 1.0)

    softmax_beta_wm = 50.0  # very deterministic WM
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL Q-values and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: probability of chosen action a under softmax
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy: near-deterministic over WM vector for the state
            # Normalize W_s to a distribution and compute softmax prob of chosen action
            W_s = W_s / max(np.sum(W_s), eps)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Effective WM weight shrinks with set size and (if older) age
            # Larger set size -> smaller weight; older group -> scaled down by (1 - wm_age_cost)
            wm_weight_size = wm_weight * np.exp(-wm_setsize_cost * (nS_t - 3.0))
            wm_weight_age = wm_weight_size * (1.0 - wm_age_cost * age_group)
            wm_weight_eff = min(max(wm_weight_age, 0.0), 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay depends on set size and age: more decay with larger set size or older age
            # We convert the base decay to an effective decay for this trial:
            # decay_eff = 1 - (1 - base_decay) * exp(-wm_setsize_cost*(nS-3)) * (1 - wm_age_cost*age_group)
            decay_scale = np.exp(-wm_setsize_cost * (nS_t - 3.0))
            age_scale = 1.0 - wm_age_cost * age_group
            decay_eff = 1.0 - (1.0 - wm_decay) * decay_scale * age_scale
            decay_eff = min(max(decay_eff, 0.0), 1.0)

            # Apply decay toward uniform
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]

            # WM storage on rewarded trials: store a one-hot association
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning rates + WM gating by memory strength and perseveration bias.
    
    Idea:
    - RL uses separate learning rates for positive and negative prediction errors.
    - WM stores rewarded associations (one-shot) and contributes when memory strength for the state is high.
    - Memory strength decays over time and is reduced by set size and age.
    - Perseveration bias adds a choice stickiness term within a state; older age amplifies perseveration.

    Parameters
    - model_parameters: list or array with 6 parameters
        0. lr_pos: RL learning rate for positive prediction errors in [0,1]
        1. lr_neg: RL learning rate for negative prediction errors in [0,1]
        2. wm_weight: base WM mixture weight in [0,1]
        3. softmax_beta: inverse temperature for RL softmax; internally multiplied by 10
        4. perseveration: base perseveration strength (>=0)
        5. age_effect: in [0,1]; scales both WM reduction and perseveration increase for older adults

    Inputs
    - states, actions, rewards, blocks, set_sizes, age: arrays as described in model1

    Returns
    - Negative log-likelihood of the observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, perseveration, age_effect = model_parameters

    softmax_beta *= 10.0

    # Age group coding
    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    # Clamp parameters
    lr_pos = min(max(lr_pos, 0.0), 1.0)
    lr_neg = min(max(lr_neg, 0.0), 1.0)
    wm_weight = min(max(wm_weight, 0.0), 1.0)
    perseveration = max(perseveration, 0.0)
    age_effect = min(max(age_effect, 0.0), 1.0)

    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))
        # WM table and a scalar memory strength per state m[s] in [0,1]
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        m = np.zeros(nS)

        # Last choice per state for perseveration
        last_choice = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]

            # Perseveration bias: add bias to the previously chosen action within a state
            # Older age increases perseveration via (1 + age_effect*age_group)
            persev_eff = perseveration * (1.0 + age_effect * age_group)
            bias = np.zeros(nA)
            if last_choice[s] >= 0:
                bias[last_choice[s]] = persev_eff

            # RL policy with bias
            pref = softmax_beta * Q_s + bias
            # probability of chosen action under biased softmax (compute via shift by chosen pref)
            denom_rl = np.sum(np.exp(pref - pref[a]))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            W_s = w[s, :]
            W_s = W_s / max(np.sum(W_s), eps)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # WM gating by memory strength m[s], which decays more with larger set size and in older age
            # Update memory strength decay before using it
            # Base decay per trial for m is implicit via set size and age factor
            # We define m_decay_eff = 1 - exp(-k*(nS-3)) scaled by age; use k = 0.5 for sensitivity
            k = 0.5
            m_decay_eff = 1.0 - np.exp(-k * (nS_t - 3.0))  # 0 at set size 3, increases at 6
            m_decay_eff = min(max(m_decay_eff, 0.0), 1.0)
            # Older participants have higher effective decay
            m_decay_eff = 1.0 - (1.0 - m_decay_eff) * (1.0 - 0.5 * age_effect * age_group)
            m_decay_eff = min(max(m_decay_eff, 0.0), 1.0)
            # Apply decay to memory strength
            m[s] = (1.0 - m_decay_eff) * m[s]

            # Effective WM weight combines base wm_weight and current memory strength
            wm_weight_eff = wm_weight * m[s]
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with asymmetry
            pe = r - Q_s[a]
            if pe >= 0.0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM decay toward uniform for the current state (tracks content precision)
            # Decays faster with set size and age using the same m_decay_eff
            w[s, :] = (1.0 - m_decay_eff) * w[s, :] + m_decay_eff * w_0[s, :]

            # On rewarded trials: strengthen WM content and memory strength
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0
                # Memory strength jumps up; capped at 1
                m[s] = min(1.0, m[s] + 0.7)

            # Update last choice for perseveration
            last_choice[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with decay + slot-limited WM and uncertainty arbitration.

    Idea:
    - WM has a limited number of slots K: it can store deterministic mappings for up to K states per block.
      If a new state appears when full, the oldest is evicted (FIFO).
    - K decreases with larger set size and in older age.
    - RL has a learning rate and a per-trial decay toward uniform, which increases with set size and age.
    - Arbitration weight depends on WM certainty (1 - entropy) and is scaled by wm_weight and set size/age.

    Parameters
    - model_parameters: list or array with 6 parameters
        0. lr: RL learning rate in [0,1]
        1. wm_weight: base WM mixture weight in [0,1]
        2. softmax_beta: inverse temperature for RL softmax; internally multiplied by 10
        3. q_decay: base RL decay toward uniform per trial in [0,1]
        4. K_base: base WM capacity (slots) at set size 3 for young participants (>=1)
        5. age_sensitivity: in [0,1]; how strongly age reduces K and increases RL decay

    Inputs
    - states, actions, rewards, blocks, set_sizes, age: arrays as described in model1

    Returns
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, q_decay, K_base, age_sensitivity = model_parameters

    softmax_beta *= 10.0

    # Age group coding
    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    # Clamp parameters
    lr = min(max(lr, 0.0), 1.0)
    wm_weight = min(max(wm_weight, 0.0), 1.0)
    q_decay = min(max(q_decay, 0.0), 1.0)
    K_base = max(K_base, 1.0)
    age_sensitivity = min(max(age_sensitivity, 0.0), 1.0)

    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))
        q_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM table and slot management
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # FIFO queue for stored states in WM
        wm_queue = []

        # Effective capacity K decreases with set size and age
        # K_eff = max(1, floor( K_base * size_factor * age_factor ))
        size_factor = 1.0 / (1.0 + 0.8 * (nS - 3.0))  # halves approximately when going 3->6
        age_factor = 1.0 - 0.5 * age_sensitivity * age_group
        K_eff = int(max(1.0, np.floor(K_base * size_factor * age_factor)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL decay toward uniform depends on set size and age
            # decay_eff = 1 - (1 - q_decay) * size_factor * age_factor
            size_factor_t = 1.0 / (1.0 + 0.8 * (nS_t - 3.0))
            age_factor_t = 1.0 - 0.5 * age_sensitivity * age_group
            decay_eff = 1.0 - (1.0 - q_decay) * size_factor_t * age_factor_t
            decay_eff = min(max(decay_eff, 0.0), 1.0)
            # Apply decay to Q-values of the current state before choice/update
            q[s, :] = (1.0 - decay_eff) * q[s, :] + decay_eff * q_0[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy uses current W_s
            W_s = W_s / max(np.sum(W_s), eps)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Arbitration based on WM certainty: weight scales with (1 - entropy)
            # Entropy of W_s (base e); max entropy ln(3)
            entropy = -np.sum(W_s * np.log(np.maximum(W_s, eps)))
            max_entropy = np.log(nA)
            certainty = 1.0 - (entropy / max_entropy)  # in [0,1]
            # Additional reduction with set size and age
            wm_weight_eff = wm_weight * certainty * size_factor_t * age_factor_t
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM slot-limited update
            # If rewarded, attempt to store deterministically in WM
            if r > 0.0:
                # If s not in queue, consider adding it
                if s not in wm_queue:
                    # Evict if full
                    if len(wm_queue) >= K_eff:
                        evict_s = wm_queue.pop(0)
                        # Reset evicted state's WM to uniform (forget)
                        w[evict_s, :] = w_0[evict_s, :].copy()
                    wm_queue.append(s)
                # Store one-hot association for s
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # On non-reward, no new storage; small passive decay toward uniform for stored states
                if s in wm_queue:
                    w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
