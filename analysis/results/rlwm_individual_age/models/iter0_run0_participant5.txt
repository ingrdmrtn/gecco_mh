Below are three standalone cognitive models that implement hybrid Reinforcement Learning (RL) plus Working Memory (WM) accounts for the RLWM task. Each function:

- Follows the provided template structure
- Returns the negative log-likelihood of the observed choices
- Uses all parameters meaningfully (≤ 6 per model)
- Incorporates age group (young = 0, old = 1) and set size effects
- Uses a WM policy and WM value updating, along with RL Q-learning

Note: Assumes numpy is already imported as np (no imports included).



def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Cognitive Model 1: Hybrid RL + Reward-gated WM with set size and age-modulated WM weighting.
    
    Description:
    - RL: Q-learning with softmax action selection.
    - WM: Reward-gated one-hot memory that decays toward uniform baseline; policy via near-deterministic softmax.
    - Mixture: Action probability is a convex combination of WM and RL policies.
    - Set size effect: WM mixture weight decreases with larger set sizes via a power-law factor.
    - Age effect: Older group has reduced WM contribution.

    Parameters (list):
    - lr: RL learning rate (0..1)
    - wm_weight_base: Base mixture weight for WM (0..1), modulated by set size and age
    - softmax_beta: Inverse temperature for RL (scaled up internally)
    - wm_lr: WM learning rate towards one-hot after reward (0..1)
    - ss_slope: Exponent controlling how strongly set size reduces WM weight (>=0)
    
    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_lr, ss_slope = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound, as per template
    
    # Age group: 0 for young, 1 for old
    age_group = 0 if age[0] <= 45 else 1
    
    softmax_beta_wm = 50.0  # very deterministic WM policy
    blocks_log_p = 0.0
    
    # Fixed WM passive decay toward uniform for current state
    wm_decay = 0.2  # modest decay toward uniform
    
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]
        
        nA = 3
        nS = int(block_set_sizes[0])
        
        # Initialize RL Q-values and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        
        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])
            
            Q_s = q[s, :]
            W_s = w[s, :]
            
            # RL policy: numerically stable softmax probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            
            # WM policy: softmax over WM values with high beta (near-deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            
            # Mixture weight modulation by set size and age
            # Base scaling by set size via power-law (3/nS)**ss_slope; reduce for older adults
            ss_factor = (3.0 / nS) ** max(ss_slope, 0.0)
            wm_weight_eff = wm_weight_base * ss_factor * (1.0 - 0.3 * age_group)
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)
            
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
            
            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            
            # WM update:
            # 1) Passive decay toward uniform for current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # 2) Reward-gated one-hot update: if rewarded, move toward chosen action
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - wm_lr) * w[s, :] + wm_lr * onehot
        
        blocks_log_p += log_p
    
    return -blocks_log_p



def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Cognitive Model 2: Capacity-limited WM + RL with age-dependent learning rate.
    
    Description:
    - RL: Q-learning with softmax policy; learning rate depends on age group.
    - WM: Deterministic storage of last rewarded action per state, with decay toward uniform.
    - Capacity effect: WM mixture weight is high when set size <= K_cap; reduced otherwise.
    - Age effect: Implemented via separate learning rates for young vs older participants.

    Parameters (list):
    - lr_y: RL learning rate for young participants
    - lr_o: RL learning rate for older participants
    - softmax_beta: Inverse temperature for RL (scaled internally)
    - wm_weight_base: Base WM mixture weight (0..1)
    - K_cap: WM capacity in number of states (e.g., around 3-4)
    - wm_decay: Per-trial decay of WM toward uniform (0..1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_y, lr_o, softmax_beta, wm_weight_base, K_cap, wm_decay = model_parameters
    softmax_beta *= 10.0
    
    age_group = 0 if age[0] <= 45 else 1
    lr = lr_y if age_group == 0 else lr_o
    
    softmax_beta_wm = 50.0  # near-deterministic WM readout
    blocks_log_p = 0.0
    
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]
        
        nA = 3
        nS = int(block_set_sizes[0])
        
        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        
        # Capacity-dependent WM mixture weight
        if nS <= K_cap:
            wm_weight_eff = wm_weight_base
        else:
            wm_weight_eff = wm_weight_base * (K_cap / float(nS))
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)
        
        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])
            
            Q_s = q[s, :]
            W_s = w[s, :]
            
            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            
            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
            
            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            
            # WM update
            # Passive decay for current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # On reward: store a one-hot memory of the correct action
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # Strong update toward the rewarded action
                w[s, :] = 0.9 * w[s, :] + 0.1 * onehot  # small learning step; decay handles persistence
        
        blocks_log_p += log_p
    
    return -blocks_log_p



def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Cognitive Model 3: RL with choice stickiness + Bayesian-like WM counts with age and set-size-dependent decay.
    
    Description:
    - RL: Q-learning with softmax and a choice stickiness component that biases toward the most recent action in a state.
    - WM: Dirichlet-like counts per state-action that form a categorical memory; softmax policy over normalized counts.
    - WM decay increases with set size; age increases both decay and stickiness.
    - Mixture: Convex combination of WM and RL policies.

    Parameters (list):
    - lr: RL learning rate
    - softmax_beta: RL inverse temperature (scaled internally)
    - wm_weight_base: Base weight for WM policy
    - wm_decay_base: Base WM decay toward uniform (0..1)
    - stickiness_base: Baseline tendency to repeat the last action (added to Q preference)
    - age_mod: Scales how much age amplifies stickiness and WM decay (>0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_base, wm_decay_base, stickiness_base, age_mod = model_parameters
    softmax_beta *= 10.0
    
    age_group = 0 if age[0] <= 45 else 1
    
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]
        
        nA = 3
        nS = int(block_set_sizes[0])
        
        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))
        # WM as Dirichlet-like counts; start with uniform prior alpha0=1
        counts = np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        
        # Track previous action per state for stickiness
        prev_action = -1 * np.ones(nS, dtype=int)
        
        # Age and set-size-dependent parameters
        # WM decay increases with set size; additionally amplified by age
        wm_decay = wm_decay_base + (nS - 3) * 0.05
        wm_decay = wm_decay * (1.0 + age_mod * age_group)
        wm_decay = np.clip(wm_decay, 0.0, 1.0)
        
        # Stickiness amplified by age
        stickiness = stickiness_base * (1.0 + age_mod * age_group)
        
        # WM weight decreases with larger set size
        wm_weight_eff = wm_weight_base * (3.0 / nS)
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)
        
        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])
            
            Q_s = q[s, :].copy()
            
            # Apply stickiness to RL preferences for the previous action in this state
            if prev_action[s] >= 0:
                Q_s[prev_action[s]] += stickiness
            
            # RL policy with stickiness-adjusted preferences
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            
            # WM policy from counts -> probabilities -> softmax
            probs_wm = counts[s, :] / np.sum(counts[s, :])
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (probs_wm - probs_wm[a])))
            
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
            
            # RL update
            base_Q = q[s, :]
            delta = r - base_Q[a]
            q[s, a] += lr * delta
            
            # WM update: decay counts toward uniform and then update with outcome
            # Decay: convex combination with uniform prior mass
            total_mass = np.sum(counts[s, :])
            uniform_counts = (total_mass / nA) * np.ones(nA)
            counts[s, :] = (1.0 - wm_decay) * counts[s, :] + wm_decay * uniform_counts
            
            # Outcome-dependent increment: reward increases chosen action count strongly,
            # non-reward gives a small diffuse increment to all (weak evidence).
            if r > 0.0:
                counts[s, a] += 1.0
            else:
                counts[s, :] += 0.05  # slight increase to keep probabilities well-defined
            
            # Update previous action for stickiness
            prev_action[s] = a
        
        blocks_log_p += log_p
    
    return -blocks_log_p

Notes on parameter influences by set size and age group:
- Model 1: WM mixture weight decreases with set size via (3/nS)^ss_slope and is reduced for older adults.
- Model 2: WM weight is high when nS ≤ K_cap, but scales down as K_cap/nS otherwise; age modulates RL learning via separate lr_y and lr_o.
- Model 3: WM decay increases with set size and is amplified by age; stickiness is also amplified by age; WM weight decreases with set size (3/nS).
