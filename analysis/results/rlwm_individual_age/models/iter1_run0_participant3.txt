def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL(λ) + decaying WM with set-size-driven interference; age increases WM decay.

    Mechanisms:
    - RL system: softmax choice; Q-learning with eligibility traces (λ).
    - WM system: associative matrix that decays toward uniform each trial; on reward,
      memory for the state is sharpened toward the chosen action.
    - Arbitration: mixture of RL and WM policies with fixed WM weight.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - softmax_beta: RL inverse temperature; internally scaled by 10
    - wm_weight0: baseline WM mixture weight (0-1)
    - wm_decay: base WM decay rate toward uniform per trial (0-1)
    - trace_lambda: RL eligibility trace parameter λ (0-1)
    - inter_set: additional WM decay per extra item beyond 3 (>=0)

    Age group:
    - age_group = 0 if age <= 45 (young), 1 if age > 45 (old)
    - Effective WM decay: wm_decay_eff = wm_decay + 0.2*age_group
      (older adults show more decay)
    Set size effect:
    - Effective per-trial WM decay: wm_decay_eff + inter_set * max(0, (nS - 3)/3)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight0, wm_decay, trace_lambda, inter_set = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Eligibility traces for RL
        e = np.zeros((nS, nA))

        # Compute effective WM decay including set-size and age
        wm_decay_eff = wm_decay + 0.2 * age_group
        wm_decay_eff += inter_set * max(0.0, (nS - 3.0) / 3.0)
        wm_decay_eff = min(max(wm_decay_eff, 0.0), 1.0)

        wm_weight = wm_weight0  # fixed mixture weight in this model

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy for chosen action probability
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over w[s,:]
            W_s = w[s, :]
            p_wm_vec = np.exp(softmax_beta_wm * (W_s - W_s.max()))
            p_wm_vec /= (p_wm_vec.sum() + 1e-12)
            p_wm = p_wm_vec[a]

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with eligibility traces
            # Decay traces
            e *= trace_lambda
            # Set replacing trace for visited state-action
            e[s, :] *= 0.0
            e[s, a] = 1.0
            delta = r - Q_s[a]
            q += lr * delta * e

            # WM decay toward uniform
            w = (1.0 - wm_decay_eff) * w + wm_decay_eff * w_0

            # WM sharpening on rewarded trials toward chosen action
            if r > 0.5:
                # Move row s toward one-hot on a by complement
                sharpen = np.zeros(nA)
                sharpen[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * sharpen
                # Renormalize to a distribution
                w[s, :] /= (w[s, :].sum() + 1e-12)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Uncertainty-gated RL+WM: arbitration depends on RL confidence, set size, and age.

    Mechanisms:
    - RL system: softmax policy; single learning rate.
    - WM system: stores perfect associations on reward; small leak each trial.
    - Arbitration: WM weight is a sigmoid of (baseline + k_uncertain*(1 - RL confidence)
      - k_set*(set size effect) - k_age*(age_group)), so WM is favored when RL is uncertain,
      but penalized by larger set sizes and older age.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - softmax_beta: RL inverse temperature; internally scaled by 10
    - w0: baseline arbitration bias toward WM (real-valued)
    - k_uncertain: weight on 1 - RL confidence (>=0)
    - k_set: penalty of set size on WM arbitration (>=0)
    - k_age: penalty of old age on WM arbitration (>=0)

    Age group:
    - age_group = 0 if age <= 45 (young), 1 if age > 45 (old)
      Contributes via the k_age term in the arbitration.

    Set size effect:
    - Enters arbitration weight via -k_set * max(0, (nS - 3)/3).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, w0, k_uncertain, k_set, k_age = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        set_penalty = k_set * max(0.0, (nS - 3.0) / 3.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy and confidence
            Q_s = q[s, :]
            # probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            # RL confidence: margin between best and second-best probabilities
            logits = softmax_beta * (Q_s - Q_s.max())
            probs = np.exp(logits)
            probs /= probs.sum() + 1e-12
            sorted_probs = np.sort(probs)
            rl_conf = sorted_probs[-1] - sorted_probs[-2]  # in [0,1]

            # WM policy: softmax over current WM weights
            W_s = w[s, :]
            p_wm_vec = np.exp(softmax_beta_wm * (W_s - W_s.max()))
            p_wm_vec /= (p_wm_vec.sum() + 1e-12)
            p_wm = p_wm_vec[a]

            # Arbitration weight via sigmoid of linear combination
            z = w0 + k_uncertain * (1.0 - rl_conf) - set_penalty - k_age * age_group
            wm_weight = 1.0 / (1.0 + np.exp(-z))

            # Mixture probability
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM leak and store on reward
            # Small leak increases with set size (non-parameterized base)
            leak = 0.02 * nS
            leak = min(leak, 0.5)
            w = (1.0 - leak) * w + leak * w_0
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Capacity-limited WM + RL with perseveration bias; age reduces WM capacity and weight.

    Mechanisms:
    - RL system: softmax with an added perseveration bias favoring the last action in a state.
    - WM system: stochastic storage with capacity limit; on reward, the association is stored
      with probability p_store that decreases with set size and age; memory decays toward uniform.
    - Arbitration: fixed baseline WM weight reduced by age.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - softmax_beta: RL inverse temperature; internally scaled by 10
    - wm_weight0: baseline WM mixture weight (0-1)
    - capacity: WM capacity scaling (0-1), controls p_store with set size
    - persev: perseveration bias added to the last action in a state (>=0)
    - age_gain: scales the age penalty on WM capacity and weight (>=0)

    Age group:
    - age_group = 0 if age <= 45 (young), 1 if age > 45 (old)
      - wm_weight_eff = wm_weight0 * (1 - 0.2 * age_gain * age_group)
      - capacity_eff = capacity * (1 - 0.5 * age_gain * age_group)

    Set size effect:
    - p_store = min(1, capacity_eff * 3 / nS) decreases as set size increases.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight0, capacity, persev, age_gain = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    # Age-adjusted parameters
    wm_weight_age = wm_weight0 * (1.0 - 0.2 * age_gain * age_group)
    wm_weight_age = min(max(wm_weight_age, 0.0), 1.0)
    capacity_age = capacity * (1.0 - 0.5 * age_gain * age_group)
    capacity_age = min(max(capacity_age, 0.0), 1.0)

    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Track last action per state for perseveration
        last_action = -1 * np.ones(nS, dtype=int)

        # Set-size dependent storage prob
        p_store = min(1.0, capacity_age * 3.0 / max(1.0, float(nS)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with perseveration bias
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += persev
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            W_s = w[s, :]
            p_wm_vec = np.exp(softmax_beta_wm * (W_s - W_s.max()))
            p_wm_vec /= (p_wm_vec.sum() + 1e-12)
            p_wm = p_wm_vec[a]

            # Mixture with age-reduced WM weight
            p_total = wm_weight_age * p_wm + (1.0 - wm_weight_age) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update (without perseveration in learning)
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # Update last action for perseveration
            last_action[s] = a

            # WM decay slightly toward uniform
            leak = 0.05
            w = (1.0 - leak) * w + leak * w_0

            # WM storage on reward with capacity-limited probability
            if r > 0.5:
                if np.random.rand() < p_store:
                    w[s, :] = 0.0
                    w[s, a] = 1.0
                # else: if storage fails, memory remains as is

        blocks_log_p += log_p

    return -blocks_log_p
