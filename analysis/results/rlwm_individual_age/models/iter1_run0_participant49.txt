def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with asymmetric RL learning rates and set-size/age-dependent WM interference.

    Idea
    - RL: Q-learning with separate positive and negative learning rates (captures older adultsâ€™
      reduced learning from negative outcomes).
    - WM: fast one-shot encoding of the last rewarded action per state, but subject to interference
      that increases with set size and age.
    - Mixture: action probability is a convex combination of RL softmax and WM softmax.
      The WM weight is the base weight scaled down by interference that depends on set size and age.

    Parameters
    ----------
    states : array-like
        State index per trial.
    actions : array-like
        Chosen action (0..2).
    rewards : array-like
        Binary reward (0/1).
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6).
    age : array-like
        Participant age (same value repeated).
    model_parameters : list or array
        [lr_pos, lr_neg, wm_base, beta_base, interference]
        - lr_pos: RL learning rate for positive prediction errors (0..1).
        - lr_neg: RL learning rate for negative prediction errors (0..1).
        - wm_base: base weight of WM mixture (0..1).
        - beta_base: base inverse temperature for RL (scaled internally).
        - interference: WM interference strength per excess item beyond 3 (>=0).

    Returns
    -------
    float
        Negative log-likelihood of choices.
    """
    lr_pos, lr_neg, wm_base, beta_base, interference = model_parameters

    # Age group: 0 if young (<=45), 1 if old (>45)
    age_group = 0 if age[0] <= 45 else 1

    # Older adults typically exhibit lower effective inverse temperature (more noise)
    # Apply a multiplicative reduction for older adults.
    softmax_beta = beta_base * 10.0 * (1.0 - 0.35 * age_group)

    softmax_beta_wm = 50.0  # deterministic WM choice when memory intact
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Set-size and age-dependent WM interference reduces effective WM contribution
            # Interference per excess item beyond 3; scaled further for older adults.
            excess = max(0, nS_t - 3)
            interference_factor = 1.0 / (1.0 + interference * excess * (1.0 + 0.5 * age_group))
            wm_weight_eff = np.clip(wm_base * interference_factor, 0.0, 1.0)

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - q[s, a]
            lr = lr_pos if pe >= 0 else lr_neg
            q[s, a] += lr * pe

            # WM decay via interference to uniform (global interference each trial)
            decay = 1.0 - interference_factor
            w = (1.0 - decay) * w + decay * w_0

            # WM encoding: after reward, store the chosen action as correct (one-shot)
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = target
            else:
                # No reward: leave as is (already decayed above)
                pass

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with probabilistic WM availability (win-stay store) controlled by set size and age.

    Idea
    - RL: standard Q-learning with a single learning rate.
    - WM: a simple "win-stay" store. When a state-action is rewarded, WM stores that action.
      WM is available on a trial with probability that decreases with set size and age.
    - Policy: When WM is available, use a near-deterministic WM softmax; otherwise use RL softmax.
      This is implemented as an expected-mixture with an availability weight.
    - Availability decreases sigmoidally with (set size - 3) and age group.

    Parameters
    ----------
    states : array-like
        State index per trial.
    actions : array-like
        Chosen action (0..2).
    rewards : array-like
        Binary reward (0/1).
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6).
    age : array-like
        Participant age (same value repeated).
    model_parameters : list or array
        [lr, beta, wm_base, forget_base, forget_setsize_w, forget_age_w]
        - lr: RL learning rate (0..1).
        - beta: RL inverse temperature (scaled internally).
        - wm_base: base WM availability at set size 3 for young participants (0..1).
        - forget_base: base forgetting drive (>=0).
        - forget_setsize_w: weight scaling forgetting with set size (>0 increases forgetting at 6).
        - forget_age_w: additional forgetting for older group (>=0).

    Returns
    -------
    float
        Negative log-likelihood of choices.
    """
    lr, beta, wm_base, forget_base, forget_setsize_w, forget_age_w = model_parameters

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta = beta * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))

        # WM store: for each state, a one-hot vector for last rewarded action (else uniform)
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM availability as a function of set size and age
            # Forgetting drive = forget_base + w_setsize*(excess) + w_age*(age_group)
            excess = max(0, nS_t - 3)
            forgetting_drive = forget_base + forget_setsize_w * excess + forget_age_w * age_group
            # Convert to an availability in (0,1) via a logistic transform anchored by wm_base
            # availability = wm_base * sigmoid(-forgetting_drive)
            availability = wm_base / (1.0 + np.exp(forgetting_drive))
            wm_weight_eff = np.clip(availability, 0.0, 1.0)

            # WM policy (deterministic if available)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Expected mixture with availability
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM update: win-stay store; if rewarded, set one-hot; otherwise revert slightly to uniform
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = target
            else:
                # partial relaxation toward uniform reflecting uncertainty when not rewarded
                relax = 0.25
                w[s, :] = (1.0 - relax) * w[s, :] + relax * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with uncertainty-gated WM contribution, age-scaled WM baseline, and lapse.

    Idea
    - RL: standard Q-learning.
    - WM: graded value map that encodes the currently believed best action, with mild precision.
      WM decays toward uniform each trial; decay increases with set size, and the WM baseline weight
      is reduced in older adults.
    - Gating: the effective WM mixture weight increases when RL is uncertain (high entropy), and
      decreases when RL is confident (low entropy). This captures strategic reliance on WM early.
    - Lapse: a small set-size- and age-independent lapse probability adds uniform choice noise.

    Parameters
    ----------
    states : array-like
        State index per trial.
    actions : array-like
        Chosen action (0..2).
    rewards : array-like
        Binary reward (0/1).
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6).
    age : array-like
        Participant age (same value repeated).
    model_parameters : list or array
        [lr, beta, wm_base, wm_decay, gate_sensitivity, lapse]
        - lr: RL learning rate (0..1).
        - beta: RL inverse temperature (scaled internally).
        - wm_base: baseline WM weight at set size 3 for young (0..1).
        - wm_decay: base WM decay per trial toward uniform (0..1).
        - gate_sensitivity: how strongly RL uncertainty up-weights WM (>=0).
        - lapse: mixture with uniform random choice (0..1).

    Returns
    -------
    float
        Negative log-likelihood of choices.
    """
    lr, beta, wm_base, wm_decay, gate_sensitivity, lapse = model_parameters

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta = beta * 10.0
    softmax_beta_wm = 25.0  # WM is sharp but not fully deterministic
    eps = 1e-12

    # Age reduces baseline WM contribution
    wm_base_age = np.clip(wm_base * (1.0 - 0.4 * age_group), 0.0, 1.0)

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # RL uncertainty (entropy of RL softmax over all actions)
            logits = softmax_beta * (Q_s - np.max(Q_s))
            prl_vec = np.exp(logits)
            prl_vec /= np.sum(prl_vec)
            entropy = -np.sum(prl_vec * np.log(np.maximum(prl_vec, eps)))  # in nats
            max_entropy = np.log(nA)
            uncert = entropy / max_entropy  # 0..1

            # Set-size dependent WM decay (more decay at larger set sizes)
            decay_eff = np.clip(wm_decay * (1.0 + 0.75 * max(0, nS_t - 3)), 0.0, 1.0)

            # Uncertainty-gated WM weight
            wm_weight_eff = np.clip(wm_base_age * (1.0 + gate_sensitivity * (uncert - 0.5)), 0.0, 1.0)

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Combine WM and RL, then apply lapse
            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM global decay toward uniform
            w = (1.0 - decay_eff) * w + decay_eff * w_0

            # WM encoding: bias state toward chosen action, stronger if rewarded
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                alpha_w = 1.0  # full overwrite on reward
            else:
                # If not rewarded, only a mild nudge toward the chosen action
                target = 0.5 * w[s, :] + 0.5 * w_0[s, :]
                alpha_w = 0.3
            w[s, :] = (1.0 - alpha_w) * w[s, :] + alpha_w * target

        blocks_log_p += log_p

    return -blocks_log_p
