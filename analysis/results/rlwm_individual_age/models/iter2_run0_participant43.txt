def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + eligibility traces mixed with capacity-limited WM; WM precision declines with set size and age.

    Mechanism
    - Policy is a convex combination of RL softmax and WM softmax.
    - RL uses a standard TD update with an eligibility trace over state-action pairs to allow some credit
      assignment spillover across time. The trace decays by lambda each trial.
    - WM encodes the last rewarded mapping in a one-shot manner but decays/leaks toward uniform. Its precision
      (effective inverse temperature) declines with set size and is further reduced in older adults.
    - The WM mixture weight is fixed within a block (base wm_weight) but is scaled down in larger set sizes.

    Parameters
    ----------
    states : array-like of int
        State index at each trial, zero-based within each block.
    actions : array-like of int
        Chosen action at each trial (0..2).
    rewards : array-like of int
        Reward feedback (0/1) at each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for the block of each trial.
    age : array-like
        Participant's age (repeated). Age group: 0=young (<=45), 1=old (>45).
    model_parameters : list or array
        [lr, wm_weight, softmax_beta, trace_lambda, wm_decay, age_wm_noise]
        - lr: RL learning rate (0..1)
        - wm_weight: base mixture weight for WM in the policy (0..1)
        - softmax_beta: RL inverse temperature; scaled by 10 internally
        - trace_lambda: eligibility trace decay (0..1)
        - wm_decay: WM decay per trial toward uniform (0..1)
        - age_wm_noise: multiplicative factor increasing WM imprecision in older adults

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, trace_lambda, wm_decay, age_wm_noise = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0  # base high precision for WM
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Eligibility traces for RL
        e = np.zeros((nS, nA))

        # WM weight penalized for larger set sizes
        wm_w_block = np.clip(wm_weight * (1.0 if nS == 3 else 0.5), 0.0, 1.0)

        # WM precision penalty for set size and age
        wm_precision_scale = 1.0 / (1.0 + 0.5 * (nS - 3))  # 1.0 for 3, ~0.67 for 6
        wm_precision_age = 1.0 / (1.0 + age_wm_noise * age_group)
        beta_wm_eff = softmax_beta_wm * wm_precision_scale * wm_precision_age

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            Q_center = Q_s - np.max(Q_s)
            p_vec_rl = np.exp(softmax_beta * Q_center)
            p_vec_rl /= np.sum(p_vec_rl)
            p_rl = p_vec_rl[a]

            # WM policy [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_center = W_s - np.max(W_s)
            p_vec_wm = np.exp(beta_wm_eff * W_center)
            p_vec_wm /= np.sum(p_vec_wm)
            p_wm = p_vec_wm[a]

            p_total = wm_w_block * p_wm + (1.0 - wm_w_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update with eligibility traces
            pe = r - q[s, a]
            # Update traces: replace for chosen SA, decay for others
            e *= trace_lambda
            e[s, a] = 1.0
            q += lr * pe * e

            # WM decay and gated one-shot update on reward
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0.0:
                row = np.full(nA, 1e-6)
                row[a] = 1.0
                row = row / np.sum(row)
                w[s, :] = row

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with UCB-like exploration bonus + WM with graded updates; age reduces exploration bonus.

    Mechanism
    - Policy mixes RL and WM.
    - RL Q-values are augmented by an uncertainty bonus: exploration favors actions with fewer visits in a state.
      The strength of this bonus is reduced in older adults.
    - WM is updated in a graded manner on every trial toward the chosen action with step proportional to reward,
      and decays toward uniform. WM influence is reduced in larger set sizes.

    Parameters
    ----------
    states : array-like of int
        State index at each trial, zero-based within each block.
    actions : array-like of int
        Chosen action at each trial (0..2).
    rewards : array-like of int
        Reward feedback (0/1) at each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for the block of each trial.
    age : array-like
        Participant's age (repeated). Age group: 0=young (<=45), 1=old (>45).
    model_parameters : list or array
        [lr, wm_weight, softmax_beta, ucb_bonus, age_ucb_penalty, wm_decay]
        - lr: RL learning rate
        - wm_weight: base policy mixture weight for WM
        - softmax_beta: RL inverse temperature; scaled by 10 internally
        - ucb_bonus: scale of the UCB-like exploration bonus
        - age_ucb_penalty: fractional reduction of ucb_bonus in older adults (0..1)
        - wm_decay: WM decay per trial toward uniform

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, ucb_bonus, age_ucb_penalty, wm_decay = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0
    eps = 1e-12

    # Age reduces exploration bonus
    bonus_scale = 1.0 - age_ucb_penalty * age_group
    ucb_bonus_eff = max(0.0, ucb_bonus * bonus_scale)

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Action visit counts per state for UCB
        n_sa = np.ones((nS, nA))  # start at 1 to avoid division by zero

        # WM weight penalized for larger set sizes
        wm_w_block = np.clip(wm_weight * (1.0 if nS == 3 else 0.5), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL with UCB-like bonus
            Q_s = q[s, :].copy()
            bonus = ucb_bonus_eff * (1.0 / np.sqrt(n_sa[s, :]))
            Q_aug = Q_s + bonus

            Q_center = Q_aug - np.max(Q_aug)
            p_vec_rl = np.exp(softmax_beta * Q_center)
            p_vec_rl /= np.sum(p_vec_rl)
            p_rl = p_vec_rl[a]

            # WM policy [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :].copy()
            W_center = W_s - np.max(W_s)
            p_vec_wm = np.exp(softmax_beta_wm * W_center)
            p_vec_wm /= np.sum(p_vec_wm)
            p_wm = p_vec_wm[a]

            p_total = wm_w_block * p_wm + (1.0 - wm_w_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # Update counts for exploration
            n_sa[s, a] += 1.0

            # WM graded update + decay
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Move WM toward a one-hot on the chosen action with strength proportional to reward
            target = np.full(nA, 1e-6)
            target[a] = 1.0
            target /= np.sum(target)
            # Decay toward uniform first
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # Reward-gated attraction toward chosen action
            if r > 0.0:
                # Move halfway toward the target on reward (implicit step = 0.5)
                w[s, :] = 0.5 * w[s, :] + 0.5 * target

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Dynamic arbitration RL-WM model with age-biased action prior and reward sensitivity.

    Mechanism
    - Policy mixes RL and WM with a dynamically updated arbitration weight (wm_mix), which adapts online
      based on which system better predicted the chosen action on the previous trial. Larger set sizes
      reduce the baseline wm_mix. 
    - RL uses a reward sensitivity parameter that scales the effective reward prediction error.
    - WM decays toward uniform and is updated on all trials, with update strength proportional to reward.
    - An action bias prior favors action 0 in older adults, mixed into both RL and WM probabilities.

    Parameters
    ----------
    states : array-like of int
        State index at each trial, zero-based within each block.
    actions : array-like of int
        Chosen action at each trial (0..2).
    rewards : array-like of int
        Reward feedback (0/1) at each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for the block of each trial.
    age : array-like
        Participant's age (repeated). Age group: 0=young (<=45), 1=old (>45).
    model_parameters : list or array
        [lr, softmax_beta, wm_forgetting, arb_lr, bias_old, reward_sensitivity]
        - lr: RL learning rate
        - softmax_beta: RL inverse temperature; scaled by 10 internally
        - wm_forgetting: WM decay per trial toward uniform (0..1)
        - arb_lr: learning rate for updating wm_mix (0..1)
        - bias_old: strength of action-0 prior in older adults (0..1), mixed into action probabilities
        - reward_sensitivity: scales reward before RL/WM updates via r_eff = sigmoid(k)*r

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, softmax_beta, wm_forgetting, arb_lr, bias_old, reward_sensitivity = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0
    eps = 1e-12

    # Age-dependent action prior favoring action 0 when older
    # Prior distribution: mix a spike on action 0 with uniform based on bias_old*age_group
    def action_prior(nA):
        if age_group == 0 or bias_old <= 0.0:
            return np.ones(nA) / nA
        prior = (1.0 - bias_old) * (np.ones(nA) / nA)
        prior[0] += bias_old
        prior /= np.sum(prior)
        return prior

    prior_cache = {3: action_prior(3)}

    # Reward sensitivity
    k = reward_sensitivity
    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))
    reward_scale = sigmoid(k)

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Cache prior (same nA across blocks in this task)
        prior = prior_cache[nA]

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Baseline wm_mix reduced in larger set size
        wm_mix = 0.7 if nS == 3 else 0.35  # starting arbitration weight within block

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r_raw = float(block_rewards[t])
            r = reward_scale * r_raw  # effective reward

            # RL policy
            Q_s = q[s, :].copy()
            Q_center = Q_s - np.max(Q_s)
            p_vec_rl = np.exp(softmax_beta * Q_center)
            p_vec_rl /= np.sum(p_vec_rl)

            # WM policy [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :].copy()
            W_center = W_s - np.max(W_s)
            p_vec_wm = np.exp(softmax_beta_wm * W_center)
            p_vec_wm /= np.sum(p_vec_wm)

            # Mix in age-dependent prior to both systems
            p_vec_rl = 0.9 * p_vec_rl + 0.1 * prior
            p_vec_rl /= np.sum(p_vec_rl)
            p_vec_wm = 0.9 * p_vec_wm + 0.1 * prior
            p_vec_wm /= np.sum(p_vec_wm)

            p_rl = p_vec_rl[a]
            p_wm = p_vec_wm[a]

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM decay and reward-weighted update toward chosen action
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            w[s, :] = (1.0 - wm_forgetting) * w[s, :] + wm_forgetting * w_0[s, :]
            if r_raw > 0.0:
                target = np.full(nA, 1e-6)
                target[a] = 1.0
                target /= np.sum(target)
                # Update strength proportional to effective reward
                eta_w = min(1.0, 0.5 * r)  # bounded step size
                w[s, :] = (1.0 - eta_w) * w[s, :] + eta_w * target

            # Arbitration update: move wm_mix toward source that better predicted the choice
            err_rl = -np.log(max(p_rl, eps))
            err_wm = -np.log(max(p_wm, eps))
            # If WM predicted better (lower surprise), increase wm_mix; else decrease
            direction = 1.0 if err_wm < err_rl else 0.0
            wm_mix = wm_mix + arb_lr * (direction - wm_mix)
            wm_mix = float(np.clip(wm_mix, 0.0, 1.0))

        blocks_log_p += log_p

    return -float(blocks_log_p)
