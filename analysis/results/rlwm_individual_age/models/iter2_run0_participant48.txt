def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL–WM arbitration by uncertainty (entropy). Age reduces WM precision and load reduces both systems' precision.

    Mechanism:
    - RL: tabular Q-learning with softmax.
    - WM: fast, one-shot storage with decay toward uniform. Policy is a sharp softmax (precision parameterized).
    - Arbitration: trial-by-trial mixture weight g is a logistic function of the entropy difference H_rl - H_wm.
      When WM is more certain (lower entropy), g increases and the policy leans on WM; otherwise it leans on RL.
    - Load effects: larger set size (6) reduces effective precision of both RL and WM.
    - Age effect: older adults have reduced WM precision (beta_wm) regardless of load.

    Parameters
    ----------
    states : array-like of int
        Stimulus/state index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of int
        Binary outcomes {0,1}.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size for each trial's block (3 or 6).
    age : array-like of int
        Participant age (same value repeated). Age group is 0 if <=45 else 1.
    model_parameters : list or array
        [lr, beta_rl_base, beta_wm_base, wm_decay, age_wm_penalty, load_beta_penalty]
        - lr: RL learning rate (0..1)
        - beta_rl_base: base RL inverse temperature (scaled by 10 internally)
        - beta_wm_base: base WM inverse temperature (scaled by 10 internally)
        - wm_decay: WM decay per visit of a state toward uniform (0..1)
        - age_wm_penalty: reduction in WM precision for older adults (subtracted from beta_wm)
        - load_beta_penalty: how much larger set size reduces RL precision

    Returns
    -------
    float
        Negative log-likelihood of the observed action sequence.
    """
    lr, beta_rl_base, beta_wm_base, wm_decay, age_wm_penalty, load_beta_penalty = model_parameters

    beta_rl_base *= 10.0
    beta_wm_base *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    nA = 3
    eps = 1e-12

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0

        # Effective precisions given load and age
        # RL precision decreases with load (6 vs 3) via load_beta_penalty
        load_factor = (nS - 3) / 3.0  # 0 for 3, 1 for 6
        beta_rl_eff = beta_rl_base * np.exp(-load_beta_penalty * load_factor)

        # WM precision reduced by load and additionally penalized in older adults
        beta_wm_eff = beta_wm_base * (3.0 / nS) - age_wm_penalty * age_group
        beta_wm_eff = max(1.0, beta_wm_eff)  # keep positive and reasonably sharp

        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # Compute RL softmax policy (full vector for entropy)
            Q_s = q[s, :]
            Q_s_center = Q_s - np.max(Q_s)
            exps_rl = np.exp(beta_rl_eff * Q_s_center)
            p_rl_vec = exps_rl / np.sum(exps_rl)
            p_rl = p_rl_vec[a]

            # Compute WM softmax policy (full vector for entropy)
            W_s = w[s, :]
            W_s_center = W_s - np.max(W_s)
            exps_wm = np.exp(beta_wm_eff * W_s_center)
            p_wm_vec = exps_wm / np.sum(exps_wm)
            p_wm = p_wm_vec[a]

            # Entropy-based arbitration
            H_rl = -np.sum(p_rl_vec * np.log(np.clip(p_rl_vec, eps, 1.0)))
            H_wm = -np.sum(p_wm_vec * np.log(np.clip(p_wm_vec, eps, 1.0)))

            # Weight g increases when WM is more certain (lower entropy)
            # Use beta_wm_eff as a scale for sensitivity of arbitration to entropy difference
            k = 0.5 * np.log1p(beta_wm_eff)  # increasing function of WM precision
            g = 1.0 / (1.0 + np.exp(-k * (H_rl - H_wm)))
            g = np.clip(g, 0.0, 1.0)

            p_total = g * p_wm + (1.0 - g) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay and one-shot storage on reward
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r == 1:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL–WM fixed-mixture with load- and age-dependent WM interference/decay.

    Mechanism:
    - RL: tabular Q-learning with softmax (precision relatively stable).
    - WM: one-shot storage with decay toward uniform. The decay grows with set size according to a load exponent.
      Older adults exhibit stronger load-driven decay (greater exponent), reducing effective WM contribution.
    - Mixture: constant per-block weight for WM determined by set size via the same exponent.

    Parameters
    ----------
    states : array-like of int
        Stimulus/state index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of int
        Binary outcomes {0,1}.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size for each trial's block (3 or 6).
    age : array-like of int
        Participant age (same value repeated). Age group is 0 if <=45 else 1.
    model_parameters : list or array
        [lr, beta, wm_weight_base, wm_decay_base, gamma_load, age_gamma_shift]
        - lr: RL learning rate (0..1)
        - beta: RL inverse temperature (scaled by 10 internally)
        - wm_weight_base: WM mixture weight at set size 3 (0..1)
        - wm_decay_base: WM decay per visit at set size 3 (0..1)
        - gamma_load: exponent controlling how set size scales WM interference/decay
        - age_gamma_shift: additional exponent added for older adults (>=0)

    Returns
    -------
    float
        Negative log-likelihood of the observed action sequence.
    """
    lr, beta, wm_weight_base, wm_decay_base, gamma_load, age_gamma_shift = model_parameters

    beta *= 10.0
    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    nA = 3
    eps = 1e-12

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nS = int(block_set_sizes[0])

        # Load exponent effective for this block (older adults amplify load)
        gamma_eff = gamma_load + age_gamma_shift * age_group

        # WM weight scales down with load via exponent
        wm_weight = wm_weight_base * (3.0 / nS) ** gamma_eff
        wm_weight = np.clip(wm_weight, 0.0, 1.0)

        # WM decay increases with load via exponent mapping
        # Effective decay toward uniform for visited states
        wm_decay_eff = 1.0 - (1.0 - wm_decay_base) * (3.0 / nS) ** gamma_eff
        wm_decay_eff = np.clip(wm_decay_eff, 0.0, 1.0)

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0

        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            Q_s_center = Q_s - np.max(Q_s)
            exps_rl = np.exp(beta * Q_s_center)
            p_rl_vec = exps_rl / np.sum(exps_rl)
            p_rl = p_rl_vec[a]

            # WM policy
            W_s = w[s, :]
            # High precision WM policy (deterministic limit)
            beta_wm = 50.0
            W_s_center = W_s - np.max(W_s)
            exps_wm = np.exp(beta_wm * W_s_center)
            p_wm_vec = exps_wm / np.sum(exps_wm)
            p_wm = p_wm_vec[a]

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay and one-shot storage on reward
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]
            if r == 1:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with reward-sensitivity and WM recall gating by state-specific recency; age speeds WM forgetting.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate, but prediction errors are scaled by a reward-sensitivity ρ:
        effective lr = lr * ρ for rewards, lr * (2-ρ) for no-reward, keeping a single parameter (not two alphas).
    - WM: one-shot storage upon reward with deterministic policy when recalled. WM recall probability for a state
      decays exponentially with the number of trials since last rewarded encoding of that state.
    - Mixture: on each trial, mixture weight equals the recall probability for that state.
    - Load/age effects: larger set size and older age reduce the recency time constant, accelerating forgetting.

    Parameters
    ----------
    states : array-like of int
        Stimulus/state index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of int
        Binary outcomes {0,1}.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size for each trial's block (3 or 6).
    age : array-like of int
        Participant age (same value repeated). Age group is 0 if <=45 else 1.
    model_parameters : list or array
        [lr, beta, rho, wm_recall_base, tau_recency, age_recall_shift]
        - lr: base RL learning rate (0..1)
        - beta: RL inverse temperature (scaled by 10 internally)
        - rho: reward sensitivity (0..2). >1 emphasizes learning from rewards; <1 emphasizes non-rewards
        - wm_recall_base: base recall probability at zero lag (0..1)
        - tau_recency: base recency time constant in trials for set size 3
        - age_recall_shift: reduction of the effective time constant in older adults (>=0)

    Returns
    -------
    float
        Negative log-likelihood of the observed action sequence.
    """
    lr, beta, rho, wm_recall_base, tau_recency, age_recall_shift = model_parameters

    beta *= 10.0
    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    nA = 3
    eps = 1e-12

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nS = int(block_set_sizes[0])

        # Effective recency time constant: decreases with load and age
        tau_eff = max(1e-3, tau_recency * (3.0 / nS) - age_recall_shift * age_group)

        # Initialize RL and WM memory
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))  # WM table for recalled policy
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Track trials since last rewarded encoding for each state
        since_reward = np.full(nS, np.inf)

        log_p = 0.0

        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            Q_s_center = Q_s - np.max(Q_s)
            exps_rl = np.exp(beta * Q_s_center)
            p_rl_vec = exps_rl / np.sum(exps_rl)
            p_rl = p_rl_vec[a]

            # WM recall probability from recency
            lag = since_reward[s]
            # At first encounter (lag=inf), recall is zero
            if np.isfinite(lag):
                p_recall = wm_recall_base * np.exp(-lag / tau_eff)
            else:
                p_recall = 0.0
            p_recall = np.clip(p_recall, 0.0, 1.0)

            # WM policy (deterministic limit)
            beta_wm = 50.0
            W_s = w[s, :]
            W_s_center = W_s - np.max(W_s)
            exps_wm = np.exp(beta_wm * W_s_center)
            p_wm_vec = exps_wm / np.sum(exps_wm)
            p_wm = p_wm_vec[a]

            # Mixture weighted by recall probability
            p_total = p_recall * p_wm + (1.0 - p_recall) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with reward-sensitivity
            if r == 1:
                lr_eff = lr * rho
            else:
                lr_eff = lr * (2.0 - rho)
            delta = r - q[s, a]
            q[s, a] += lr_eff * delta

            # WM update: on reward, store perfect association; otherwise leave as is
            if r == 1:
                w[s, :] = 0.0
                w[s, a] = 1.0
                since_reward[s] = 0.0
            else:
                # increment recency counters for all states that are finite
                pass

            # Increment recency counters for all states at the end of trial
            finite_mask = np.isfinite(since_reward)
            since_reward[finite_mask] += 1.0

        blocks_log_p += log_p

    return -blocks_log_p
