def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + WM with capacity-gated WM usage and RL eligibility traces.

    Idea
    - RL: Q-learning uses eligibility traces to propagate prediction errors within a block.
    - WM: one-shot storage of rewarded actions; WM decays over time with decay increasing with set size and age.
    - Gating: The contribution of WM is controlled by a capacity gate that depends on set size and age.

    Parameters
    ----------
    states : array-like
        State index per trial (0..nS-1).
    actions : array-like
        Chosen action per trial (0..2).
    rewards : array-like
        Binary reward (0/1).
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6).
    age : array-like
        Participant age (same value repeated).
    model_parameters : list or array
        [lr, beta_base, wm_gate, lam, wm_decay_base, cap_age]
        - lr: RL learning rate (0..1).
        - beta_base: base inverse temperature for RL; internally scaled to a higher range.
        - wm_gate: sensitivity of WM gating to capacity-load mismatch (higher -> sharper gating).
        - lam: eligibility trace parameter (0..1); higher spreads learning more to recent state-actions.
        - wm_decay_base: base WM decay per trial (>=0), scaled by set size and age.
        - cap_age: reduction of effective WM capacity for older adults (>=0).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, beta_base, wm_gate, lam, wm_decay_base, cap_age = model_parameters

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0  # near-deterministic WM when intact
    softmax_beta = beta_base * 10.0  # scaled RL inverse temperature
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL values and eligibility traces
        q = (1.0 / nA) * np.ones((nS, nA))
        e = np.zeros((nS, nA))  # eligibility traces

        # Initialize WM values
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM gating: effective capacity vs load
            # Define a notional capacity that is reduced for older adults by cap_age.
            # Use a smooth gate based on the mismatch capacity - load.
            effective_capacity = 4.0 - cap_age * age_group  # ~4 for young, reduced for older if cap_age>0
            load = float(nS_t)
            gate_input = effective_capacity - load
            # squash to [0,1] via logistic function
            wm_weight_eff = 1.0 / (1.0 + np.exp(-wm_gate * gate_input))
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # WM policy (softmax over WM weights)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with eligibility trace
            pe = r - q[s, a]
            # Update eligibilities: chosen (s,a)=1, others decay
            e *= lam
            e[s, :] *= 0.0
            e[s, a] = 1.0
            q += lr * pe * e

            # WM decay scales with set size and age (more decay for larger set and older age)
            decay = wm_decay_base * (nS_t / 6.0) * (1.0 + 0.5 * age_group)
            decay = np.clip(decay, 0.0, 1.0)
            w = (1.0 - decay) * w + decay * w_0

            # WM update: rewarded actions are stored one-shot, otherwise leave as is
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + WM with novelty-driven exploration and age-dependent novelty sensitivity.

    Idea
    - RL: standard Q-learning.
    - Novelty bonus: add a bonus to actions that have been chosen less often in a state
      (count-based exploration). Older adults receive a reduced novelty bonus.
    - WM: one-shot storage of rewarded actions; WM forgets more with larger set size and age.
    - Mixture: WM/RL combination where WM weight decays with load and age.

    Parameters
    ----------
    states : array-like
        State index per trial.
    actions : array-like
        Chosen action per trial.
    rewards : array-like
        Binary reward per trial.
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6).
    age : array-like
        Participant age (same value repeated).
    model_parameters : list or array
        [lr, beta_base, wm_base, wm_forget, novelty_w, age_explore]
        - lr: RL learning rate (0..1).
        - beta_base: base RL inverse temperature (scaled internally).
        - wm_base: base WM mixture weight (0..1).
        - wm_forget: base WM forgetting rate (>=0) per trial; scaled by set size and age.
        - novelty_w: strength of count-based novelty bonus (>=0).
        - age_explore: fraction by which older adults reduce novelty (0..1).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, beta_base, wm_base, wm_forget, novelty_w, age_explore = model_parameters

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta = beta_base * 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Action selection counts per state-action
        counts = np.zeros((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            # Novelty-adjusted Q
            # Count-based bonus: higher for less-chosen actions
            bonus = novelty_w / np.sqrt(counts[s, :] + 1.0)
            # Older adults have reduced novelty seeking
            bonus *= (1.0 - age_explore * age_group)

            Q_aug = q[s, :] + bonus

            # RL policy on augmented values
            denom_rl = np.sum(np.exp(softmax_beta * (Q_aug - Q_aug[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM mixing weight: decays with set size and age
            load_factor = 1.0 / (1.0 + max(0, nS_t - 3))
            wm_weight_eff = np.clip(wm_base * load_factor * (1.0 - 0.4 * age_group), 0.0, 1.0)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (w[s, :] - w[s, a])))
            p_wm = 1.0 / max(denom_wm, eps)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # Update counts
            counts[s, a] += 1.0

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM decay scales with set size and age
            decay = wm_forget * (nS_t / 6.0) * (1.0 + 0.5 * age_group)
            decay = np.clip(decay, 0.0, 1.0)
            w = (1.0 - decay) * w + decay * w_0

            # WM update: store rewarded action; on no-reward, small soft reset toward uniform on that state
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = target
            else:
                # small nudge toward uncertainty when an expected mapping is contradicted
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning + WM go/avoid cache and load/age-dependent temperatures.

    Idea
    - RL: different learning rates for positive and negative outcomes.
    - Temperature: RL inverse temperature decreases with set size and further with age.
    - WM: maintains a 'go' cache for rewarded actions and an 'avoid' cache for punished actions.
      WM policy uses go minus weighted avoid to discourage previously incorrect actions.
    - Mixture: WM contribution declines with set size and age; WM caches decay with load/age.

    Parameters
    ----------
    states : array-like
        State per trial.
    actions : array-like
        Chosen action (0..2).
    rewards : array-like
        Reward (0/1).
    blocks : array-like
        Block index.
    set_sizes : array-like
        Set size (3 or 6).
    age : array-like
        Participant age (same value repeated).
    model_parameters : list or array
        [lr_pos, lr_neg, beta_base, wm_base, wm_prune, avoid_weight]
        - lr_pos: RL learning rate for positive prediction errors (0..1).
        - lr_neg: RL learning rate for negative prediction errors (0..1).
        - beta_base: base RL inverse temperature (scaled internally).
        - wm_base: base WM mixture weight (0..1).
        - wm_prune: WM decay/pruning rate (>=0), scaled by set size and age.
        - avoid_weight: weight on the 'avoid' cache in WM policy (>=0).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr_pos, lr_neg, beta_base, wm_base, wm_prune, avoid_weight = model_parameters

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))

        # WM go and avoid caches
        w_go = (1.0 / nA) * np.ones((nS, nA))
        w_avoid = np.zeros((nS, nA))
        w_uniform = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]
            nS_t = int(block_set_sizes[t])

            # RL temperature decreases with set size and age
            beta_eff = beta_base * 10.0 / (1.0 + 0.5 * max(0, nS_t - 3) + 0.5 * age_group)

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(beta_eff * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM mixture weight declines with load and age
            wm_weight_eff = wm_base / (1.0 + max(0, nS_t - 3))
            wm_weight_eff *= (1.0 - 0.4 * age_group)
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # WM policy uses go minus weighted avoid
            wm_pref = w_go[s, :] - avoid_weight * w_avoid[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (wm_pref - wm_pref[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - q[s, a]
            lr_use = lr_pos if pe >= 0 else lr_neg
            q[s, a] += lr_use * pe

            # WM decay/pruning scales with load and age
            decay = wm_prune * (nS_t / 6.0) * (1.0 + 0.5 * age_group)
            decay = np.clip(decay, 0.0, 1.0)
            w_go = (1.0 - decay) * w_go + decay * w_uniform
            w_avoid = (1.0 - decay) * w_avoid  # avoid cache decays toward zero

            # WM updates
            if r >= 0.5:
                # strengthen go cache as one-hot for rewarded action
                target = np.zeros(nA)
                target[a] = 1.0
                w_go[s, :] = target
                # also reduce avoidance for the chosen action
                w_avoid[s, a] = 0.0
            else:
                # tag the chosen action as to-be-avoided
                w_avoid[s, a] = 1.0
                # soften go cache for that state to reflect uncertainty
                w_go[s, :] = 0.9 * w_go[s, :] + 0.1 * w_uniform[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
