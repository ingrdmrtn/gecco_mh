def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Capacity-limited RL-WM mixture with age-adjusted exploration temperature.

    Decision policy:
    - Choice is a mixture of RL softmax and a high-precision WM softmax.
    - WM reliability is capacity-limited: weight scales down with larger set sizes.
    - Younger adults get a boost to effective RL temperature; older adults get a reduction.

    Learning/updating:
    - RL: Rescorla-Wagner update with single learning rate.
    - WM: Per-visit leak toward uniform (wm_forget), and reward-contingent write (wm_write)
      that pushes WM toward the chosen action. Both operate every trial.

    Parameters (list/tuple of length 6):
    - lr: RL learning rate (0..1).
    - softmax_beta: base RL inverse temperature; internally scaled by 10.
    - wm_weight_base: baseline WM mixture weight (0..1) at set size 3 for young adults.
    - wm_forget: WM leak per visit toward uniform (0..1).
    - wm_write: WM write strength toward chosen action per trial (0..1); stronger trace if rewarded.
    - age_temp_bias: additive bias to RL inverse temperature for young vs old (positive favors young).
      Effective beta = softmax_beta + age_temp_bias*(1 - 2*age_group).

    Age and set-size effects:
    - WM mixture weight is scaled by (3/nS) and mildly boosted for young adults:
        wm_weight_eff = clip(wm_weight_base * (3/nS) * (1 + 0.2*(1 - age_group)), 0, 1)
    - RL temperature gets an age bias via age_temp_bias.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_base, wm_forget, wm_write, age_temp_bias = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_group = 0 if age[0] <= 45 else 1

    eps = 1e-12
    blocks_log_p = 0.0

    # Age-adjusted RL temperature
    softmax_beta_eff = max(softmax_beta + age_temp_bias * (1 - 2 * age_group), eps)

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-limited WM mixture weight
        wm_weight_eff = np.clip(wm_weight_base * (3.0 / nS) * (1.0 + 0.2 * (1 - age_group)), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # WM decay toward uniform each time state is visited
            w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]

            # RL policy for chosen action a
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta_eff * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy for chosen action a
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward-contingent write toward chosen action
            write_strength = wm_write * (0.5 + 0.5 * r)  # stronger when rewarded
            w[s, :] = (1.0 - write_strength) * w[s, :]
            w[s, a] += write_strength
            # renormalize to keep a valid distribution
            w[s, :] = w[s, :] / np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL-WM mixture with novelty-augmented RL and age-dependent novelty bias.

    Decision policy:
    - Choice is a mixture of RL softmax and WM softmax.
    - RL values are augmented by a novelty bonus for actions that are weakly represented in WM.
      Novelty encourages exploration in large set sizes and is age-modulated.

    Learning/updating:
    - RL: Rescorla-Wagner update with a single learning rate.
    - WM: Decay toward uniform each state visit; if rewarded, WM shifts strongly toward chosen action.

    Parameters (list/tuple of length 6):
    - lr: RL learning rate (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_weight_base: baseline WM mixture weight (0..1) at set size 3, young.
    - novelty_bonus: scale of novelty added to RL action values (>=0).
    - wm_decay: WM decay toward uniform per visit (0..1).
    - age_novelty_bias: signed bias on novelty for young vs old; novelty *= (1 + age_novelty_bias*(1-2*age_group)).

    Age and set-size effects:
    - WM mixture weight decreases with set size: multiplied by (3/nS).
    - Novelty scales up with set size and is reduced for older adults via age_novelty_bias.
      Effective novelty = novelty_bonus * ((nS - 2)/3) * (1 + age_novelty_bias*(1 - 2*age_group)).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_base, novelty_bonus, wm_decay, age_novelty_bias = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_group = 0 if age[0] <= 45 else 1

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        wm_weight_eff = np.clip(wm_weight_base * (3.0 / nS), 0.0, 1.0)
        novelty_scale = novelty_bonus * ((nS - 2.0) / 3.0) * (1.0 + age_novelty_bias * (1 - 2 * age_group))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Compute WM policy
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Novelty vector: higher bonus for actions with lower WM weight
            wm_inv = 1.0 - W_s  # novelty associated with weak memory
            nov_vec = wm_inv - np.mean(wm_inv)  # zero-mean to avoid biasing all options
            Q_aug = q[s, :] + novelty_scale * nov_vec

            # RL policy with novelty augmentation
            denom_rl = np.sum(np.exp(softmax_beta * (Q_aug - Q_aug[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # Mixture of WM and RL
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update (based on true reward)
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM update: stronger write when rewarded; otherwise just decay already applied
            if r > 0.5:
                write = min(1.0, 0.8 + 0.1 * (3.0 / nS))  # slightly stronger in small sets
                w[s, :] = (1.0 - write) * w[s, :]
                w[s, a] += write
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Time-varying WM reliance with lapse and reward-weighted WM writes.

    Decision policy:
    - Mixture of RL and WM, with WM mixture weight collapsing from a start to an end value
      over the block (indexed by trial number within block). Younger adults start and stay
      more WM-reliant via an age shift.
    - Final choice has a lapse component: with probability lapse_rate, choose uniformly.

    Learning/updating:
    - RL: Rescorla-Wagner update.
    - WM: Mild decay each visit; reward-weighted write (stronger for reward, weaker otherwise),
      and write strength is capacity-limited by set size.

    Parameters (list/tuple of length 6):
    - lr: RL learning rate (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_weight_start: initial WM mixture weight at the start of a block (0..1) for nS=3.
    - wm_weight_end: terminal WM mixture weight by the end of a block (0..1) for nS=3.
    - lapse_rate: probability of a random choice (0..1).
    - age_wm_shift: additive shift applied to the WM weight trajectory for young vs old:
      wm_weight_logit += age_wm_shift * (1 - 2*age_group).

    Age and set-size effects:
    - WM weights are reduced in larger set sizes via a logit shift: +logit_scale*(3/nS - 1).
    - Young adults get a positive shift if age_wm_shift > 0; older get a negative shift.
    - WM write strength is scaled by (3/nS).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_start, wm_weight_end, lapse_rate, age_wm_shift = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_group = 0 if age[0] <= 45 else 1

    eps = 1e-12
    blocks_log_p = 0.0

    # Helper: logit/sigmoid
    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    def logit(p):
        p = np.clip(p, 1e-6, 1 - 1e-6)
        return np.log(p / (1.0 - p))

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])
        T = len(block_states)

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute logit shifts for set size and age
        setsize_shift = logit(3.0 / max(nS, 1.0)) - logit(1.0)  # negative for nS>3, zero for nS=3
        age_shift = age_wm_shift * (1 - 2 * age_group)

        log_p = 0.0
        for t in range(T):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # WM decay (mild, implicit via averaging with uniform)
            decay = 0.1
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # Interpolate WM weight over the block (0..1). Then apply set size and age on logit scale.
            tau = 0.0 if T <= 1 else t / (T - 1)
            wm_w_raw = (1.0 - tau) * wm_weight_start + tau * wm_weight_end
            wm_logit = logit(wm_w_raw) + setsize_shift + age_shift
            wm_weight = sigmoid(wm_logit)
            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            # RL policy prob of chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy prob of chosen action
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture with lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse_rate) * p_mix + lapse_rate * (1.0 / nA)
            log_p += np.log(max(p_total, eps))

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM reward-weighted write; scaled by capacity
            write_strength = (0.2 + 0.6 * r) * (3.0 / nS)
            w[s, :] = (1.0 - write_strength) * w[s, :]
            w[s, a] += write_strength
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
