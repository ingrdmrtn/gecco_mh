def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + WM with uncertainty-based arbitration modulated by set size and age.

    Idea:
    - Choices are a mixture of an RL policy (Q-learning) and a WM policy (state-specific memory).
    - Arbitration weight for WM increases when WM distribution for the state is sharp (low entropy),
      and decreases with higher set size and for older adults.
    - RL uses separate learning rates for positive and negative prediction errors.

    Parameters (model_parameters):
    - lr_pos: learning rate for positive prediction errors (0..1).
    - lr_neg: learning rate for negative prediction errors (0..1).
    - softmax_beta: base inverse temperature for RL action selection; internally scaled x10.
    - wm_decay: decay rate of WM toward uniform each trial within a block (0..1).
    - arb_bias: baseline bias toward WM in the sigmoid arbitration (can be negative or positive).
    - ent_weight: weight of WM entropy in arbitration (>=0), higher values decrease WM weight when entropy is high.

    Inputs:
    - states: array of state indices per trial (int).
    - actions: array of chosen actions per trial (int in [0,2]).
    - rewards: array of rewards per trial (0/1).
    - blocks: array of block indices per trial.
    - set_sizes: array of set sizes per trial (3 or 6).
    - age: array with a single repeated value of participant's age.
    
    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr_pos, lr_neg, softmax_beta, wm_decay, arb_bias, ent_weight = model_parameters
    softmax_beta *= 10.0  # higher upper bound per template

    # Age group coding: 0 = young (<=45), 1 = old (>45)
    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0  # very deterministic WM policy by design
    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute load term: 0 for set size 3, 1 for set size 6
        load = max(0.0, (nS - 3.0) / 3.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Entropy of WM for arbitration: H in [0, log(nA)], normalize to [0,1]
            H = -np.sum(W_s * np.log(np.clip(W_s, eps, 1.0)))
            H_norm = H / np.log(nA)

            # Arbitration weight: sigmoid of (bias - ent_weight*entropy - load_term - age_term)
            # Age and load reduce reliance on WM additively.
            wm_logit = (
                arb_bias
                - ent_weight * H_norm
                - 1.0 * load
                - 0.75 * age_group  # older adults rely less on WM
            )
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update with signed learning rates
            pe = r - Q_s[a]
            lr = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += lr * pe

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM imprint on rewarded trials; on no-reward slightly suppress chosen action
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            if r > 0.0:
                # Sharpen toward the rewarded action
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot
            else:
                # Mild push away from the chosen action to reflect error tagging
                w[s, a] = 0.5 * w[s, a] + 0.5 * (1.0 / nA)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Capacity-limited WM slots with reliability and load/age-modulated lapses.

    Idea:
    - WM has a limited capacity K_cap (in number of states). If set size > K_cap, only a fraction of states
      can be actively maintained. The effective WM availability for a state is approx p_wm_avail = min(1, K_eff/nS).
    - K_eff is reduced for older adults via age_cap_drop (captures reduced capacity with age).
    - WM, when available, proposes a policy with precision set by wm_reliability.
    - A load- and age-modulated lapse diverts some probability mass to uniform random choice.
    - RL (Q-learning) runs in parallel and is mixed with WM when WM is available.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - softmax_beta: base inverse temperature for RL softmax; internally scaled x10.
    - K_cap: nominal WM capacity in number of states (>=0).
    - wm_reliability: scales WM inverse temperature (0..1 mapped to [5,50]).
    - lap_gain: controls how much lapse increases with set size and age (>=0).
    - age_cap_drop: fraction by which capacity is reduced for older adults (0..1).

    Inputs:
    - states, actions, rewards, blocks, set_sizes, age: as specified.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, softmax_beta, K_cap, wm_reliability, lap_gain, age_cap_drop = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective capacity decreases with age
        K_eff = max(0.0, K_cap * (1.0 - age_cap_drop * age_group))
        # Availability probability for any given state under uniform allocation
        p_wm_avail = min(1.0, K_eff / float(nS)) if nS > 0 else 0.0

        # WM precision based on reliability parameter
        softmax_beta_wm = 5.0 + 45.0 * np.clip(wm_reliability, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy candidate
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture using availability as weight
            p_mix = p_wm_avail * p_wm + (1.0 - p_wm_avail) * p_rl

            # Lapse increases with load and age
            load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6
            lapse = np.clip(lap_gain * (0.5 * load + 0.5 * age_group), 0.0, 0.3)
            p_uniform = 1.0 / nA
            p_total = (1.0 - lapse) * p_mix + lapse * p_uniform

            log_p += np.log(max(p_total, eps))

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM update: imprint on reward, light decay otherwise
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.4 * w[s, :] + 0.6 * one_hot
            else:
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Meta-learned WM-RL mixture driven by ongoing prediction errors, with WM forgetting.

    Idea:
    - Maintain a dynamic WM mixture weight per block that is adjusted online:
      mix_t+1 = sigmoid(logit(mix_t) + pe_mix_gain * PE_t - mix_penalty),
      where mix_penalty grows with set size and age (reducing WM reliance under load/aging).
    - RL uses standard Q-learning.
    - WM is a leaky memory updated toward the chosen action on reward, with a forget_rate.
    - The policy is a mixture of RL and WM at each trial.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - softmax_beta: base inverse temperature for RL; internally scaled x10.
    - base_mix: initial WM mixture weight at block start (0..1).
    - pe_mix_gain: how strongly PE modulates WM mixture (positive increases WM after positive PE).
    - forget_rate: WM forgetting rate toward uniform per trial (0..1).
    - age_mix_shift: additional penalty factor applied to WM mixture for older adults and higher load.

    Inputs:
    - states, actions, rewards, blocks, set_sizes, age: as specified.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, softmax_beta, base_mix, pe_mix_gain, forget_rate, age_mix_shift = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0  # high precision WM
    eps = 1e-12
    blocks_log_p = 0.0

    def clip01(x):
        return min(1.0, max(0.0, x))

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Initialize mixture
        wm_mix = clip01(base_mix)

        # Load penalty (0 for 3, 1 for 6)
        load = max(0.0, (nS - 3.0) / 3.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Apply current mixture
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM forgetting and update
            w[s, :] = (1.0 - forget_rate) * w[s, :] + forget_rate * w_0[s, :]
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot

            # Meta-update of mixture:
            # Increase with positive PE, decrease with negative PE; also apply penalty from load and age.
            # Update in logit space for stability.
            wm_mix = clip01(wm_mix)
            logit = np.log(max(wm_mix, eps) / max(1.0 - wm_mix, eps))
            penalty = age_mix_shift * (0.5 * load + 0.5 * age_group)
            logit = logit + pe_mix_gain * pe - penalty
            wm_mix = 1.0 / (1.0 + np.exp(-logit))

        blocks_log_p += log_p

    return -blocks_log_p
