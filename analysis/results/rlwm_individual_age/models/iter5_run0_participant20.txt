def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL–WM mixture with WM decay and age-dependent interference.

    Policy
    - Choice probability is a convex mixture of RL softmax and WM softmax.
    - RL uses a standard delta rule.
    - WM stores one-hot associations after rewards and decays toward uniform otherwise.
    - WM reliability is degraded by set-size driven interference that is amplified in older adults.

    Age use
    - Age group increases WM interference; here age_group=0 (young), so minimal interference.

    Parameters
    ----------
    model_parameters : list or array-like
        [lr, wm_weight_base, softmax_beta, wm_decay, wm_interference_age]
        - lr: RL learning rate (0..1).
        - wm_weight_base: base mixture weight for WM (0..1).
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - wm_decay: WM learning/decay step toward target on each trial (0..1).
        - wm_interference_age: coefficient controlling WM interference per extra item (nS-3),
                               multiplied by age_group (0 or 1). Larger -> more flattening.

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, wm_interference_age = model_parameters
    softmax_beta *= 10.0  # as specified in the template

    # age group coding
    try:
        age_group = 0 if age[0] <= 45 else 1
    except Exception:
        age_group = 0 if age <= 45 else 1

    softmax_beta_wm = 50.0  # very deterministic WM
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        idx = (blocks == b)

        block_actions = np.asarray(actions)[idx].astype(int)
        block_rewards = np.asarray(rewards)[idx].astype(float)
        block_states = np.asarray(states)[idx].astype(int)
        block_set_sizes = np.asarray(set_sizes)[idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL values and WM weights
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (softmax with temperature)
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            exp_rl = np.exp(logits_rl)
            p_rl = exp_rl[a] / np.sum(exp_rl)

            # WM policy (softmax over WM weights)
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            exp_wm = np.exp(logits_wm)
            p_wm = exp_wm[a] / np.sum(exp_wm)

            # Set-size reduction of WM weight; age-dependent interference weakens WM further
            # base scaling with set size: 3/nS
            base_scale = 3.0 / nS
            # interference factor: stronger with larger set and for older age group
            interference = wm_interference_age * age_group * max(0, (nS - 3) / 3.0)
            wm_weight = np.clip(wm_weight_base * base_scale * (1.0 - interference), 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward = move toward one-hot; no-reward = mild decay toward uniform
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
            else:
                target = w_0[s, :]
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * target
            # Normalize to valid probability vector
            w[s, :] = np.maximum(w[s, :], 1e-12)
            w[s, :] /= np.sum(w[s, :])

            # Global interference after each observation: flatten all WM rows slightly
            if nS > 3 and age_group == 1:
                lam = np.clip(wm_interference_age * (nS - 3) / 3.0, 0.0, 1.0)
                w = (1.0 - lam) * w + lam * w_0
                # renormalize row-wise
                w = np.maximum(w, 1e-12)
                w = (w.T / np.sum(w, axis=1)).T

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Asymmetric RL + WM with reliability-based arbitration and lapse rate.

    Policy
    - Mixture of RL and WM policies, weighted by WM reliability (confidence).
    - WM reliability = max(W_s) scaled; further reduced by set size.
    - Final policy includes a small lapse probability.

    Learning
    - RL uses separate learning rates for positive and negative prediction errors.
    - WM learns from reward (toward one-hot) with rate wm_learn; after non-reward, it partially
      relaxes toward uniform.

    Age use
    - Lapse rate increases with age group (older -> more lapses).
      For this participant (young), lapse is lower.

    Parameters
    ----------
    model_parameters : list or array-like
        [lr_pos, lr_neg, softmax_beta, wm_base, wm_learn, lapse]
        - lr_pos: RL learning rate for positive PE.
        - lr_neg: RL learning rate for negative PE.
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - wm_base: baseline WM contribution to arbitration (0..1).
        - wm_learn: WM encoding/decay step size (0..1).
        - lapse: baseline lapse probability mixed with uniform (0..1).

    Returns
    -------
    nll : float
        Negative log-likelihood.
    """
    lr_pos, lr_neg, softmax_beta, wm_base, wm_learn, lapse = model_parameters
    softmax_beta *= 10.0

    # age group coding
    try:
        age_group = 0 if age[0] <= 45 else 1
    except Exception:
        age_group = 0 if age <= 45 else 1

    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    # age-modulated lapse
    lapse_eff = np.clip(lapse * (1.0 + 0.5 * age_group), 0.0, 0.3)  # cap lapses to keep stability

    for b in np.unique(blocks):
        idx = (blocks == b)

        block_actions = np.asarray(actions)[idx].astype(int)
        block_rewards = np.asarray(rewards)[idx].astype(float)
        block_states = np.asarray(states)[idx].astype(int)
        block_set_sizes = np.asarray(set_sizes)[idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            exp_rl = np.exp(logits_rl)
            p_rl_vec = exp_rl / np.sum(exp_rl)
            p_rl = p_rl_vec[a]

            # WM policy
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            exp_wm = np.exp(logits_wm)
            p_wm_vec = exp_wm / np.sum(exp_wm)
            p_wm = p_wm_vec[a]

            # Reliability-based arbitration:
            # reliability ~ how peaked WM is; use max prob scaled to [0,1]
            reliability = (np.max(W_s) - (1.0 / nA)) / (1.0 - (1.0 / nA))
            reliability = np.clip(reliability, 0.0, 1.0)
            # downscale with set size
            reliability *= (3.0 / nS)
            wm_weight = np.clip(wm_base * reliability, 0.0, 1.0)

            # Mixture with lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse_eff) * p_mix + lapse_eff * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            lr_eff = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += lr_eff * pe

            # WM update
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                # partial relaxation toward uniform when not rewarded
                w[s, :] = (1.0 - 0.5 * wm_learn) * w[s, :] + (0.5 * wm_learn) * w_0[s, :]

            w[s, :] = np.maximum(w[s, :], 1e-12)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with forgetting + WM slot-limited storage.

    Policy
    - Mixture of RL softmax and WM softmax.
    - WM contribution scales with the fraction of states that can be stored (slot limit).
      If slots >= set size, WM is strong; otherwise WM is diluted.

    Learning
    - RL has separate learning rates for positive/negative PE and a small forgetting term.
    - WM stores one-hot associations after reward; otherwise, it softly decays toward uniform.
    - WM weight is boosted when the state appears to be stored (i.e., WM is peaked for that state).

    Age use
    - Older age reduces the number of effective WM slots by a fixed amount (age penalty).

    Parameters
    ----------
    model_parameters : list or array-like
        [alpha_pos, alpha_neg, softmax_beta, wm_weight_base, wm_slots_base, rl_forgetting]
        - alpha_pos: RL learning rate when PE >= 0.
        - alpha_neg: RL learning rate when PE < 0.
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - wm_weight_base: baseline WM mixture weight (0..1).
        - wm_slots_base: nominal number of WM slots (e.g., around 3–4).
        - rl_forgetting: per-trial multiplicative forgetting (0..0.1) applied to Q toward uniform.

    Returns
    -------
    nll : float
        Negative log-likelihood.
    """
    alpha_pos, alpha_neg, softmax_beta, wm_weight_base, wm_slots_base, rl_forgetting = model_parameters
    softmax_beta *= 10.0

    # age group coding
    try:
        age_group = 0 if age[0] <= 45 else 1
    except Exception:
        age_group = 0 if age <= 45 else 1

    # Age reduces effective WM slots by 0.7 slot for older participants (no reduction for young)
    wm_slots_eff_global = max(0.0, wm_slots_base - 0.7 * age_group)

    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        idx = (blocks == b)

        block_actions = np.asarray(actions)[idx].astype(int)
        block_rewards = np.asarray(rewards)[idx].astype(float)
        block_states = np.asarray(states)[idx].astype(int)
        block_set_sizes = np.asarray(set_sizes)[idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Slot-based dilution factor: fraction of items that can be maintained
        slot_fraction = np.clip(wm_slots_eff_global / max(1, nS), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            # Apply small RL forgetting toward uniform each trial
            if rl_forgetting > 0.0:
                q = (1.0 - rl_forgetting) * q + rl_forgetting * (1.0 / nA)

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            exp_rl = np.exp(logits_rl)
            p_rl = exp_rl[a] / np.sum(exp_rl)

            # WM policy
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            exp_wm = np.exp(logits_wm)
            p_wm = exp_wm[a] / np.sum(exp_wm)

            # State-specific WM reliability: more peaked = more reliable
            rel_state = (np.max(W_s) - (1.0 / nA)) / (1.0 - (1.0 / nA))
            rel_state = np.clip(rel_state, 0.0, 1.0)

            # Combine global slot fraction and state reliability; also scale by 3/nS
            wm_weight = np.clip(wm_weight_base * slot_fraction * rel_state * (3.0 / nS), 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric rates
            pe = r - Q_s[a]
            lr_eff = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += lr_eff * pe

            # WM update: store after reward; otherwise decay a bit
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                step = 0.6  # strong encoding on reward
            else:
                target = w_0[s, :]
                step = 0.3  # weaker decay on non-reward
            w[s, :] = (1.0 - step) * w[s, :] + step * target
            w[s, :] = np.maximum(w[s, :], 1e-12)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -float(blocks_log_p)
