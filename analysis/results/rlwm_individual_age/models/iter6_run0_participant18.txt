def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM arbitration by RL uncertainty and WM capacity with age-dependent capacity penalty.

    Idea:
    - RL system: standard Q-learning with inverse temperature softmax_beta.
    - WM system: near-deterministic table (softmax_beta_wm fixed high) that rapidly stores recent correct associations,
      but values decay over time at rate wm_decay.
    - Arbitration: trial-wise WM weight increases when RL is uncertain (high entropy => down-weights RL) and when set size
      is within WM capacity. Capacity is penalized for older adults.

    Parameters
    - lr: RL learning rate
    - softmax_beta: RL inverse temperature (internally scaled up)
    - wm_capacity: capacity in number of items that WM can hold with high fidelity
    - wm_decay: decay rate (0..1) applied to WM table each trial for the current state (larger = faster decay)
    - age_capacity_penalty: reduction in effective capacity for older adults (in items)

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_capacity, wm_decay, age_capacity_penalty = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly selective WM policy

    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM capacity (reduced in larger sets and for older adults)
        eff_capacity = wm_capacity - age_group * age_capacity_penalty
        # Load factor: 0 when nS <= eff_capacity; increases when nS exceeds capacity
        load_over = max(0.0, nS - eff_capacity)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            Z_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(Z_rl, 1e-12)

            # WM policy (near-deterministic)
            W_s = w[s, :]
            Z_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm_clean = 1.0 / max(Z_wm, 1e-12)

            # Arbitration: combine (i) RL uncertainty and (ii) load over capacity
            # RL uncertainty via normalized entropy (0..1)
            pi_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            pi_rl = pi_rl / np.sum(pi_rl)
            entropy_rl = -np.sum(pi_rl * np.log(np.clip(pi_rl, 1e-12, 1.0))) / np.log(nA)

            # WM weight reduces with load_over (linear), increases with RL entropy (needs WM help)
            # Map to [0,1] using a sigmoid on a simple linear combination.
            wm_logit = (+2.5 * (entropy_rl - 0.5)) + (-1.0 * load_over)
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

            # Mixture policy
            p_total = wm_weight * p_wm_clean + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay and update
            # Decay current state's WM row toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # Reward-dependent encoding: if rewarded, write a strong pattern; if not, weakly suppress chosen action
            if r > 0:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0
            else:
                w[s, a] = 0.5 * w[s, a] + 0.5 * w_0[s, a]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning rates + WM gating by surprise and load, with age-dependent gating bias.

    Idea:
    - RL: Q-learning with separate learning rates for positive and negative prediction errors.
    - WM: near-deterministic table that encodes more strongly when outcomes are surprising (|PE| large).
    - Arbitration: WM weight is a logistic gate driven by surprise (|PE|), reduced by set size (3 vs 6) load,
      and shifted by age (older adults show weaker WM gating).

    Parameters
    - lr_pos: RL learning rate for positive PE
    - lr_neg: RL learning rate for negative PE
    - softmax_beta: RL inverse temperature (internally scaled up)
    - wm_gate_base: base logit controlling WM involvement
    - age_gate_effect: additive logit shift for older adults (negative values reduce WM gate for older adults)

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, softmax_beta, wm_gate_base, age_gate_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load penalty factor reduces WM gating in larger set size
        load_penalty = (nS - 3) / 3.0  # 0 for 3, 1 for 6

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            Z_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(Z_rl, 1e-12)

            # WM policy
            W_s = w[s, :]
            Z_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(Z_wm, 1e-12)

            # Surprise (absolute PE) from RL, normalized to [0,1] by range of reward [0,1]
            pe = r - Q_s[a]
            surprise = min(1.0, abs(pe))

            # WM gate logit: base + surprise drive - load penalty + age shift for older adults
            wm_logit = wm_gate_base + 2.0 * (surprise - 0.5) - 1.0 * load_penalty + (age_gate_effect if age_group == 1 else 0.0)
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            if pe >= 0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM encoding scaled by surprise: more surprising => stronger write
            # Implement as interpolation between keeping current row and writing a one-hot.
            encode_strength = surprise  # 0..1
            # Mild decay first toward uniform
            w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]
            # Then encode
            w[s, :] = (1.0 - encode_strength) * w[s, :] + encode_strength * w_0[s, :]
            if r > 0:
                w[s, a] = (1.0 - encode_strength) * w[s, a] + encode_strength * 1.0
            else:
                # For errors, avoid confidently marking the chosen action as correct
                w[s, a] = 0.5 * w[s, a] + 0.5 * w_0[s, a]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with set-size- and age-modulated RL learning rate and independent WM temperature.

    Idea:
    - RL: Q-learning with inverse temperature softmax_beta. The learning rate is meta-adapted by set size (slower in larger sets)
      and by age group (older adults learn more slowly/faster depending on age_lr_shift).
    - WM: fast, high-temperature system with its own inverse temperature wm_beta and standard decay per visit.
    - Mixture: fixed WM weight parameter (sigmoid-transformed) combining WM and RL policies.

    Parameters
    - lr_base: baseline RL learning rate
    - lr_ss_slope: linear slope reducing (if positive) RL learning rate as set size increases from 3 to 6
    - age_lr_shift: additive shift to learning rate for older adults (e.g., negative -> slower learning in older group)
    - softmax_beta: RL inverse temperature (scaled internally)
    - wm_beta: WM inverse temperature (directly used)
    - wm_weight_base: base logit for WM mixture weight (sigmoid-transformed to 0..1)

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr_base, lr_ss_slope, age_lr_shift, softmax_beta, wm_beta, wm_weight_base = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Learning rate adapted by set size and age
        ss_factor = (nS - 3) / 3.0  # 0 for 3, 1 for 6
        lr = lr_base + lr_ss_slope * ss_factor + (age_lr_shift if age_group == 1 else 0.0)
        lr = np.clip(lr, 0.0, 1.0)

        # Mixture weight
        wm_weight = 1.0 / (1.0 + np.exp(-wm_weight_base))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # RL policy
            Q_s = q[s, :]
            Z_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(Z_rl, 1e-12)

            # WM policy with its own inverse temperature
            W_s = w[s, :]
            Z_wm = np.sum(np.exp(wm_beta * (W_s - W_s[a])))
            p_wm = 1.0 / max(Z_wm, 1e-12)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay and update (simple decay + reward-driven write)
            w[s, :] = 0.6 * w[s, :] + 0.4 * w_0[s, :]
            if r > 0:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0
            else:
                # On errors, slightly suppress chosen action to avoid reinforcing wrong mapping
                w[s, a] = 0.7 * w[s, a] + 0.3 * w_0[s, a]

        blocks_log_p += log_p

    return -blocks_log_p
