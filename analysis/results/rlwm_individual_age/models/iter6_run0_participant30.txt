def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Mixture of model-free RL and capacity-limited WM with set-size-scaled WM influence and age-dependent lapses.

    Mechanism
    - RL: standard delta rule per state-action value.
    - WM: fast-learning cache with interference-based decay that increases with set size (3 vs 6).
    - Arbitration: fixed WM weight scaled by set-size (less WM influence for 6) and mildly reduced by age.
    - Lapses: age-dependent lapse rate mixes in a uniform-random choice component.

    Parameters
    ----------
    states : array-like
        State index per trial within block (0..nS-1).
    actions : array-like
        Chosen action per trial (0..2).
    rewards : array-like
        Reward per trial (0 or 1).
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6), constant within block.
    age : array-like
        Participant age repeated for each trial.
    model_parameters : list or array
        [lr, beta, wm_strength, wm_lr, lapse_base, age_lapse_boost]
        - lr: RL learning rate (0..1)
        - beta: inverse temperature base (>0), scaled by 10 internally
        - wm_strength: base WM contribution at set size 3 (0..1); scales down with larger set size
        - wm_lr: WM learning rate (0..1)
        - lapse_base: baseline lapse probability (0..0.5)
        - age_lapse_boost: additive multiplier for older group on lapse (>=0)

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, beta, wm_strength, wm_lr, lapse_base, age_lapse_boost = model_parameters
    softmax_beta = beta * 10.0

    age_group = 0 if age[0] <= 45 else 1
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent WM interference/decay (no extra parameter)
        # 0 for 3, 0.5 for 6
        wm_interference = max(0.0, (nS - 3) / 6.0) + 0.0  # gentle, 0 for 3, 0.5 for 6

        # WM mixture weight scales down with set size and slightly with age
        wm_weight = np.clip(wm_strength * (3.0 / nS) * (1.0 - 0.2 * age_group), 0.0, 1.0)

        # Age-dependent lapse
        lapse = np.clip(lapse_base * (1.0 + age_lapse_boost * age_group), 0.0, 0.5)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic softmax over WM weights)
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)

            log_p += np.log(max(p_total, 1e-12))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM interference toward uniform + fast update on chosen action
            w = (1.0 - wm_interference) * w + wm_interference * w_0
            w[s, a] += wm_lr * (r - w[s, a])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Asymmetric RL with set-size-modulated negativity and one-shot WM encoding; arbitration by entropy.

    Mechanism
    - RL: base learning rate is split into positive/negative via a ratio; when set size is 6, negative learning is
      amplified (helping to avoid perseverating on wrong actions in high load).
    - WM: one-shot encoding on rewarded trials; decays via set-size-dependent interference.
    - Arbitration: WM vs RL weight determined by relative certainty (inverse entropy of their action distributions).
    - Age: increases RL noise (reduces beta) and reduces WM certainty weight via noisier WM softmax.

    Parameters
    ----------
    model_parameters : list or array
        [alpha_base, pos_neg_ratio, beta, wm_one_shot, age_noise_gain, size_neg_ratio]
        - alpha_base: baseline learning rate (0..1)
        - pos_neg_ratio: scales alpha_pos = alpha_base * pos_neg_ratio (>=0)
        - beta: base inverse temperature (>0), scaled by 10 internally
        - wm_one_shot: strength of one-shot WM encoding on reward (0..1)
        - age_noise_gain: increases decision noise for older group (>=0)
        - size_neg_ratio: multiplicative boost to negative learning when set size is 6 (>=1)

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    alpha_base, pos_neg_ratio, beta, wm_one_shot, age_noise_gain, size_neg_ratio = model_parameters
    softmax_beta = beta * 10.0

    age_group = 0 if age[0] <= 45 else 1
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL params: asymmetry with set-size-adjusted negative learning
        alpha_pos = np.clip(alpha_base * pos_neg_ratio, 0.0, 1.0)
        neg_boost = 1.0 if nS == 3 else size_neg_ratio
        alpha_neg = np.clip(alpha_base * neg_boost, 0.0, 1.0)

        # Age increases noise (reduces effective beta)
        beta_eff = softmax_beta / (1.0 + age_noise_gain * age_group)

        # WM structures
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        q = (1.0 / nA) * np.ones((nS, nA))

        # WM decay by set size (0 for 3, 0.5 for 6)
        wm_decay = max(0.0, (nS - 3) / 6.0)

        # Age adds noise to WM policy by lowering its beta slightly
        beta_wm = 50.0 / (1.0 + 0.3 * age_group)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Build RL policy distribution via relative softmax
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(beta_eff * (Q_s - Q_s[a])))

            # WM policy
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(beta_wm * (W_s - W_s[a])))

            # Compute entropies to weight arbitration (lower entropy => higher weight)
            # Get full distributions by softmax for entropy computation
            rl_probs = np.exp(beta_eff * (Q_s - np.max(Q_s)))
            rl_probs = rl_probs / np.sum(rl_probs)
            wm_probs = np.exp(beta_wm * (W_s - np.max(W_s)))
            wm_probs = wm_probs / np.sum(wm_probs)

            ent_rl = -np.sum(rl_probs * np.log(np.clip(rl_probs, 1e-12, 1.0)))
            ent_wm = -np.sum(wm_probs * np.log(np.clip(wm_probs, 1e-12, 1.0)))

            inv_ent_rl = 1.0 / (1.0 + ent_rl)  # 0.5..1-ish
            inv_ent_wm = 1.0 / (1.0 + ent_wm)

            wm_weight = inv_ent_wm / (inv_ent_wm + inv_ent_rl)
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update with asymmetry
            pe = r - Q_s[a]
            if pe >= 0:
                q[s, a] += alpha_pos * pe
            else:
                q[s, a] += alpha_neg * pe

            # WM decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # One-shot encoding on rewarded trials; mild anti-encoding on no-reward
            if r > 0.5:
                # move row toward one-hot on chosen action
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_one_shot) * w[s, :] + wm_one_shot * target
            else:
                # small corrective push away from chosen action confidence
                w[s, a] = (1.0 - 0.5 * wm_one_shot) * w[s, a] + 0.5 * wm_one_shot * (1.0 / nA)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Uncertainty-bonus RL with dynamic WM gating by recency and age-modulated WM decay.

    Mechanism
    - RL: standard delta learning; policy uses an uncertainty bonus (UCB-style) that increases in larger set sizes
      and for older participants; bonus decays with visits per state-action.
    - WM: fast-learning buffer with age- and set-size-modulated decay; WM gate weight is higher early in a block and
      declines as the state is visited more (recency/novelty).
    - Arbitration: convex combination of WM and RL policies using the dynamic WM gate.

    Parameters
    ----------
    model_parameters : list or array
        [lr, beta, wm_lr, wm_decay_base, age_wm_penalty, bonus_scale]
        - lr: RL learning rate (0..1)
        - beta: base inverse temperature (>0), scaled by 10 internally
        - wm_lr: WM learning rate (0..1)
        - wm_decay_base: baseline WM decay toward uniform each trial (0..1)
        - age_wm_penalty: additive increase in WM decay for older group (>=0)
        - bonus_scale: scales the RL uncertainty bonus (>=0)

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, beta, wm_lr, wm_decay_base, age_wm_penalty, bonus_scale = model_parameters
    softmax_beta = beta * 10.0

    age_group = 0 if age[0] <= 45 else 1
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Visit counts for uncertainty bonus (start small to avoid div by zero)
        counts = 0.5 * np.ones((nS, nA))

        # WM decay effective: base + age penalty + set-size penalty
        wm_decay_eff = np.clip(wm_decay_base + age_wm_penalty * age_group + 0.3 * max(0.0, (nS - 3) / 3.0), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL uncertainty bonus (state-action specific)
            # Higher for larger set sizes and older group; decays with sqrt(count)
            bonus_amp = bonus_scale * (1.0 + 0.5 * age_group + 0.5 * max(0.0, (nS - 3) / 3.0))
            u_bonus = bonus_amp / np.sqrt(np.maximum(counts[s, :], 1e-6))

            # Policy uses Q + bonus for softmax; keep Q_s intact for learning update
            Q_aug = q[s, :] + u_bonus
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_aug - Q_aug[a])))

            # WM policy
            beta_wm = 50.0
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(beta_wm * (W_s - W_s[a])))

            # Dynamic WM gate: stronger early/low-visit, weaker later/high-visit; diminished with set size
            visits_state = np.sum(counts[s, :])  # total visits to this state (starts at 1.5)
            # Map visits to gate in (0,1): gate = base/(1 + visits/k)
            k_vis = 3.0 if nS == 3 else 5.0
            base_gate = 0.9 * (3.0 / nS)  # less WM under higher set size
            wm_weight = np.clip(base_gate / (1.0 + (visits_state - 1.5) / k_vis), 0.0, 1.0)
            # Age slightly reduces WM gate
            wm_weight *= (1.0 - 0.2 * age_group)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL learning
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # Update counts
            counts[s, a] += 1.0

            # WM decay + learning
            w = (1.0 - wm_decay_eff) * w + wm_decay_eff * w_0
            w[s, a] += wm_lr * (r - w[s, a])

        blocks_log_p += log_p

    return -blocks_log_p
