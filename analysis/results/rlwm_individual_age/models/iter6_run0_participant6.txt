def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with load- and age-adjusted WM reliance and WM leak.

    Mechanism:
    - RL: standard delta rule and softmax (provided by template).
    - WM policy: deterministic readout from a WM trace; mixture weight downscales with set size
      and with age; WM trace leaks toward uniform with load- and age-dependent rate.
    - WM update: strong write on rewarded trials, weak corrective write on non-reward.

    Parameters (up to 6 total):
    - lr: RL learning rate (0..1).
    - wm_weight: base arbitration weight on WM at set size 3 for young adults (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_leak_base: base WM leak rate per trial at set size 3 for young (0..1).
    - wm_write_strength: write strength into WM on rewarded trials (0..1).
    - age_wm_penalty: proportional reduction in WM reliance and increase in leak for older adults (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta = model_parameters[:3]
    extra = list(model_parameters[3:])
    # Defaults if not provided
    wm_leak_base = extra[0] if len(extra) > 0 else 0.1
    wm_write_strength = extra[1] if len(extra) > 1 else 0.8
    age_wm_penalty = extra[2] if len(extra) > 2 else 0.5

    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load- and age-adjusted WM parameters for this block
        leak = wm_leak_base * (nS / 3.0) * (1.0 + age_wm_penalty * age_group)
        leak = float(np.clip(leak, 0.0, 0.95))
        wm_weight_eff_block = wm_weight * (3.0 / nS) * (1.0 - age_wm_penalty * age_group)
        wm_weight_eff_block = float(np.clip(wm_weight_eff_block, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability (fixed by template)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Deterministic softmax over WM values
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Arbitration with block-effective WM weight
            p_total = wm_weight_eff_block * p_wm + (1 - wm_weight_eff_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update (fixed by template)
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Rewarded: strong write toward chosen action
            if r > 0.5:
                w[s, :] = (1 - wm_write_strength) * w[s, :]
                w[s, a] += wm_write_strength
            else:
                # Unrewarded: weak corrective push away from chosen action, toward uniform
                weak = 0.2 * wm_write_strength
                w[s, a] = (1 - weak) * w[s, a]
                redistribute = weak
                others = [i for i in range(nA) if i != a]
                w[s, others] += redistribute / (nA - 1)

            # Normalize row
            w[s, :] = np.clip(w[s, :], 0, None)
            w[s, :] /= np.sum(w[s, :])

            # Leak toward uniform baseline with load- and age-adjustment
            w[s, :] = (1 - leak) * w[s, :] + leak * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with surprise-gated WM arbitration and load-adjusted WM forgetting.

    Mechanism:
    - RL: standard delta rule and softmax (template).
    - WM policy: deterministic WM readout; arbitration weight is reduced when trial-level surprise is high.
      Surprise is computed from RL's prediction error magnitude, and the gate is also penalized by load and age.
    - WM update: reward-contingent write scaled by (1 - surprise), and uniform forgetting otherwise.

    Parameters (up to 6 total):
    - lr: RL learning rate (0..1).
    - wm_weight: baseline WM weight at low surprise, set size 3, young (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - gate_mid: gate midpoint parameter controlling overall gating bias (real).
    - gate_slope: sensitivity of gating to surprise (>=0).
    - wm_forget: baseline WM forgetting toward uniform per trial at set size 3 (0..1).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta = model_parameters[:3]
    extra = list(model_parameters[3:])
    gate_mid = extra[0] if len(extra) > 0 else 0.0
    gate_slope = extra[1] if len(extra) > 1 else 3.0
    wm_forget = extra[2] if len(extra) > 2 else 0.2

    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability (fixed by template)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Trial-level surprise from RL prediction error magnitude (before update)
            pe_mag = abs(r - Q_s[a])

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Deterministic WM readout
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Surprise-gated WM weight: higher surprise -> lower WM reliance
            # Sigmoid gate in [0,1], then scaled by load and age
            gate_input = gate_mid - gate_slope * pe_mag
            gate_prob = 1.0 / (1.0 + np.exp(-gate_input))
            wm_weight_eff = wm_weight * gate_prob * (3.0 / nS) * (1.0 - 0.3 * age_group)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update (fixed by template)
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Forgetting rate increases with load and age
            forget = wm_forget * (nS / 3.0) * (1.0 + 0.5 * age_group)
            forget = float(np.clip(forget, 0.0, 0.95))

            # Reward-contingent consolidation scaled by (1 - surprise): predictable reward writes more strongly
            if r > 0.5:
                write_strength = (1.0 - pe_mag)  # in [0,1]
                write_strength = float(np.clip(write_strength, 0.0, 1.0))
                # Write to chosen action
                w[s, :] = (1 - write_strength) * w[s, :]
                w[s, a] += write_strength
            else:
                # Unrewarded: forget toward uniform to reduce misleading traces
                pass  # handled below by forgetting

            # Normalize then apply forgetting toward uniform
            w[s, :] = np.clip(w[s, :], 0, None)
            w[s, :] /= np.sum(w[s, :])
            w[s, :] = (1 - forget) * w[s, :] + forget * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with uncertainty-based arbitration and WM temperature smoothing by age.

    Mechanism:
    - RL: standard delta rule and softmax (template).
    - WM policy: WM action distribution is sharpened or smoothed depending on age via a temperature on W.
    - Arbitration: WM weight decreases with the entropy (uncertainty) of the WM distribution; also reduced by load and age.
    - WM update: reward-based delta to the WM row with leaky decay to uniform.

    Parameters (up to 6 total):
    - lr: RL learning rate (0..1).
    - wm_weight: base WM weight at zero entropy, set size 3, young (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_unc_slope: scale translating WM entropy to down-weighting of WM (>=0).
    - wm_leak_base: base WM leak per trial at set size 3 (0..1).
    - age_temp_scale: additional WM temperature (smoothing) factor applied in older adults (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta = model_parameters[:3]
    extra = list(model_parameters[3:])
    wm_unc_slope = extra[0] if len(extra) > 0 else 3.0
    wm_leak_base = extra[1] if len(extra) > 1 else 0.15
    age_temp_scale = extra[2] if len(extra) > 2 else 1.0

    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # base deterministic; we will adjust effective temp
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load- and age-modulated leak
        leak = wm_leak_base * (nS / 3.0) * (1.0 + 0.5 * age_group)
        leak = float(np.clip(leak, 0.0, 0.95))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability (fixed by template)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Compute WM distribution with an age-dependent smoothing temperature
            # Effective temperature: older adults have higher temperature via age_temp_scale
            tau = 1.0 + age_group * age_temp_scale  # tau >= 1
            # Implement temperature by raising probabilities to 1/tau (sharpen if tau<1, smooth if tau>1)
            W_norm = W_s / max(np.sum(W_s), eps)
            W_temp = np.power(np.clip(W_norm, eps, 1.0), 1.0 / tau)
            W_temp = W_temp / np.sum(W_temp)

            # Deterministic softmax on the temperature-adjusted WM distribution
            logits_wm = softmax_beta_wm * (W_temp - np.max(W_temp))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Entropy-based arbitration: higher entropy -> lower WM weight
            entropy = -np.sum(W_temp * np.log(np.clip(W_temp, eps, 1.0))) / np.log(nA)  # normalized in [0,1]
            wm_weight_eff = wm_weight * np.exp(-wm_unc_slope * entropy) * (3.0 / nS) * (1.0 - 0.3 * age_group)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update (fixed by template)
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Reward-based delta update on WM with leaky decay
            alpha_w = 0.8 if r > 0.5 else 0.2  # stronger move on reward
            # Move WM row toward a one-hot on chosen action
            target = np.zeros(nA)
            target[a] = 1.0
            w[s, :] = (1 - alpha_w) * w[s, :] + alpha_w * target

            # Leak toward uniform
            w[s, :] = (1 - leak) * w[s, :] + leak * w_0[s, :]

            # Normalize
            w[s, :] = np.clip(w[s, :], 0, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
