def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with eligibility traces and PE-gated Working Memory (WM), with set-size interference and age-modulated trace.

    Mechanism
    - RL: Q-learning with replacing eligibility traces. Trace decay is age-dependent.
    - WM: Near-deterministic policy readout but subject to set-size interference. WM contribution is gated by the
      absolute prediction error (PE): larger surprise increases the WM mixture weight on the next choice.
    - Arbitration: mixture of RL and WM policies on each trial.
    - Set size: reduces the baseline WM mixture weight (3/nS).
    - Age: older adults have shorter eligibility persistence (lower lambda), implemented via age-specific lambda.

    Parameters (6)
    - alpha_rl: RL learning rate (0..1)
    - softmax_beta: RL inverse temperature; internally scaled by *10*
    - wm_weight_base: base WM mixture weight (0..1)
    - lambda_y: eligibility-trace decay for young (0..1)
    - lambda_o: eligibility-trace decay for old (0..1)
    - k_size_gate: gating gain controlling how much |PE| modulates WM mixture (>=0)

    Inputs
    - states, actions, rewards: trial-wise arrays
    - blocks: block index per trial
    - set_sizes: set size on each trial (block-constant)
    - age: array with the same scalar value; <=45 -> young (0), >45 -> old (1)
    - model_parameters: list of 6 parameters as above

    Returns
    - Negative log-likelihood of observed choices under the model.
    """
    alpha_rl, softmax_beta, wm_weight_base, lambda_y, lambda_o, k_size_gate = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    lam = lambda_y if age_group == 0 else lambda_o

    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        e = np.zeros((nS, nA))  # eligibility traces

        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Baseline WM mixture weight reduced by set size
        wm_mix_base = max(0.0, min(1.0, wm_weight_base)) * (3.0 / float(nS))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy (probability of chosen action)
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = max(p_rl, 1e-12)

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = max(p_wm, 1e-12)

            # PE-gated WM mixture: stronger WM after surprising outcomes
            pe = r - Q_s[a]
            # Gate via logistic transform centered near |PE|=0.5, scaled by k_size_gate
            gate = 1.0 / (1.0 + np.exp(-k_size_gate * (abs(pe) - 0.5)))
            wm_mix = (1.0 - gate) * wm_mix_base + gate * min(1.0, wm_mix_base * 1.5)  # boost when surprising

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL with replacing eligibility traces
            e *= lam
            e[s, :] *= 0.0
            e[s, a] = 1.0
            q += alpha_rl * pe * e

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay/interference toward uniform each trial; stronger with bigger set sizes
            # Interference factor: fraction beyond size 3
            interf = (float(nS) - 3.0) / 3.0
            interf = max(0.0, min(1.0, interf))
            decay = 1.0 - 0.5 * interf  # more interference -> faster decay
            w[s, :] = decay * w[s, :] + (1.0 - decay) * w_0[s, :]

            if r > 0.0:
                # Successful outcome: store one-shot action template in WM
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with valence-asymmetric learning and WM with age- and set-size-dependent lapses (mixture with uniform).

    Mechanism
    - RL: separate learning rates for rewards vs. non-rewards.
    - WM: near-deterministic readout but corrupted by a lapse that increases with set size and with age.
           Lapses mix WM policy with uniform, and also weaken WM storage strength.
    - Arbitration: fixed base WM weight, scaled down by set size (3/nS).
    - Set size: increases WM lapses; reduces baseline WM mixture.
    - Age: older group has larger WM lapse propensity.

    Parameters (6)
    - alpha_pos: RL learning rate for positive PE (rewarded trials) (0..1)
    - alpha_neg: RL learning rate for negative PE (unrewarded trials) (0..1)
    - softmax_beta: RL inverse temperature; internally scaled by *10*
    - wm_weight_base: base WM mixture weight (0..1)
    - k_lapse_age: added WM lapse component for older group (>=0)
    - k_size_noise: WM lapse growth per extra item beyond 3 (>=0)

    Inputs/Outputs as in cognitive_model1.
    """
    alpha_pos, alpha_neg, softmax_beta, wm_weight_base, k_lapse_age, k_size_noise = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1
    age_lapse = k_lapse_age * float(age_group)

    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))

        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        wm_mix = max(0.0, min(1.0, wm_weight_base)) * (3.0 / float(nS))

        # Lapse rate depends on set size and age
        size_term = max(0.0, (float(nS) - 3.0) / 3.0)
        lapse = max(0.0, min(1.0, k_size_noise * size_term + age_lapse))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy probability of chosen action
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = max(p_rl, 1e-12)

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Compute WM policy, then mix with uniform due to lapse
            W_s = w[s, :]
            p_wm_det = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm_det = max(p_wm_det, 1e-12)
            p_uniform = 1.0 / nA
            p_wm = (1.0 - lapse) * p_wm_det + lapse * p_uniform
            p_wm = max(p_wm, 1e-12)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with valence-asymmetric learning rate
            pe = r - Q_s[a]
            alpha = alpha_pos if pe >= 0 else alpha_neg
            q[s, a] += alpha * pe

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Lapse also weakens storage strength; first decay toward uniform
            # Stronger decay with larger lapse
            decay = 1.0 - 0.5 * lapse
            w[s, :] = decay * w[s, :] + (1.0 - decay) * w_0[s, :]

            # Reward-gated storage; storage strength scales with (1 - lapse)
            if r > 0.0:
                strength = (1.0 - lapse)
                w[s, :] = (1.0 - strength) * w_0[s, :]
                w[s, a] += strength  # concentrates mass on chosen action
                # renormalize (safety)
                w[s, :] = w[s, :] / np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL-WM arbitration via entropy-based confidence, with WM decay and age-biased arbitration.

    Mechanism
    - RL: standard Q-learning with single learning rate.
    - WM: decays toward uniform each trial; reward causes one-shot storage. Decay increases with set size.
    - Arbitration: mixture weight is computed dynamically from the difference in entropies (confidence) of WM vs RL.
        w_mix = sigmoid(wm_bias_base + k_entropy * (H_rl - H_wm) - size_penalty) scaled by age bias.
      Lower RL entropy (more confident) reduces WM weight; lower WM entropy increases WM weight.
    - Set size: increases decay and adds a penalty term to WM arbitration.
    - Age: older group down-weights WM via age_bias_scale.

    Parameters (6)
    - alpha_rl: RL learning rate (0..1)
    - softmax_beta: RL inverse temperature; internally scaled by *10*
    - wm_bias_base: baseline arbitration bias toward WM (real-valued)
    - k_entropy: gain mapping entropy difference to WM weight (>=0)
    - decay_wm: base WM decay toward uniform per trial (0..1)
    - age_bias_scale: multiplicative down-weighting of WM for older group (0..1)

    Inputs/Outputs as in cognitive_model1.
    """
    alpha_rl, softmax_beta, wm_bias_base, k_entropy, decay_wm, age_bias_scale = model_parameters
    softmax_beta *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))

        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Size penalty for arbitration and stronger WM decay with larger set sizes
        size_frac = max(0.0, (float(nS) - 3.0) / 3.0)
        size_penalty = np.log(1.0 + size_frac)  # small positive penalty if nS > 3
        decay_size = 1.0 - (1.0 - decay_wm) * (1.0 - 0.5 * size_frac)
        decay_size = max(0.0, min(1.0, decay_size))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy vector (for entropy) and chosen probability
            Q_s = q[s, :]
            Q_shift = Q_s - np.max(Q_s)
            expQ = np.exp(softmax_beta * Q_shift)
            p_rl_vec = expQ / np.sum(expQ)
            p_rl = max(p_rl_vec[a], 1e-12)

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :]
            W_shift = W_s - np.max(W_s)
            expW = np.exp(softmax_beta_wm * W_shift)
            p_wm_vec = expW / np.sum(expW)
            p_wm = max(p_wm_vec[a], 1e-12)

            # Entropy-based arbitration
            H_rl = -np.sum(p_rl_vec * np.log(np.maximum(p_rl_vec, 1e-12)))
            H_wm = -np.sum(p_wm_vec * np.log(np.maximum(p_wm_vec, 1e-12)))

            logit = wm_bias_base + k_entropy * (H_rl - H_wm) - size_penalty
            wm_mix = 1.0 / (1.0 + np.exp(-logit))

            # Age reduces WM reliance multiplicatively for older group
            wm_mix *= (1.0 - age_bias_scale * float(age_group))
            wm_mix = max(0.0, min(1.0, wm_mix))

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += alpha_rl * pe

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform each trial (size-dependent)
            w[s, :] = decay_size * w[s, :] + (1.0 - decay_size) * w_0[s, :]

            # Reward-gated one-shot storage
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p
