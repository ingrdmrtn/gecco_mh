def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with eligibility-like recency and slot-limited WM with refresh; arbitration with bias and lapse.

    Overview:
    - Policy is a mixture of RL and WM policies with an age- and load-sensitive arbitration and a small lapse.
    - RL uses a delta rule and a mild recency factor (like a simple eligibility-like persistence).
    - WM is slot-limited: when set size exceeds capacity, WM precision is diluted.
    - WM is refreshed each trial toward a one-hot at the last rewarded action in that state.
    - Arbitration weight favors WM when set size is small and for younger adults; includes a bias parameter.

    Parameters (model_parameters):
    - lr: RL learning rate in [0,1]
    - beta_rl: RL inverse temperature; internally scaled by 10
    - wm_slot_cap: effective WM slots (>=1). Determines dilution at larger set sizes
    - wm_refresh: WM refresh/strengthening rate in [0,1] when reward is delivered
    - arb_bias: baseline arbitration bias toward WM in [0,1]
    - lapse_base: base lapse probability in [0,1]; increases with set size and age

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, beta_rl, wm_slot_cap, wm_refresh, arb_bias, lapse_base = model_parameters
    beta_rl *= 10.0
    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0  # deterministic WM channel
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Recency factor for RL (eligibility-like persistence stronger for smaller set sizes and younger)
        recency = np.clip(0.1 * (3.0 / nS) * (1.2 - 0.2 * age_group), 0.0, 0.3)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(beta_rl * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM precision diluted by load relative to slot capacity
            load_ratio = float(nS) / max(wm_slot_cap, 1e-6)
            wm_precision_scale = 1.0 / (1.0 + max(0.0, load_ratio - 1.0))  # 1 if nS<=cap; drops if nS>cap
            beta_wm_eff = softmax_beta_wm * wm_precision_scale

            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Arbitration: favor WM for small set size and younger adults, plus bias
            wm_weight = np.clip(arb_bias * (3.0 / nS) * (1.1 - 0.1 * age_group), 0.0, 1.0)

            # Lapse increases with load and age
            lapse = np.clip(lapse_base * (1.0 + 0.3 * (nS == 6) + 0.4 * age_group), 0.0, 0.5)

            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with recency bump to chosen action
            delta = r - Q_s[a]
            q[s, a] = Q_s[a] + lr * delta + recency * (1.0 / nA - Q_s[a])  # gentle pull to baseline for stability

            # WM drift toward uniform
            w[s, :] = 0.95 * w[s, :] + 0.05 * w_0[s, :]

            # WM refresh on reward: move toward one-hot of chosen action
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_refresh) * w[s, :] + wm_refresh * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning and uncertainty-gated WM via encoding probability.

    Overview:
    - RL: separate learning rates for positive and negative prediction errors.
    - WM: one-shot encoding with probability p_enc that decreases with set size and age.
          Encoded items are represented as sharp one-hots; otherwise remain near-uniform.
    - Arbitration: WM weight scales with RL uncertainty (higher entropy â†’ more WM) and with encoding success.
    - Both RL and WM operate via softmax; WM softmax has its own temperature.

    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive PE in [0,1]
    - lr_neg: RL learning rate for negative PE in [0,1]
    - beta_rl: RL inverse temperature; internally scaled by 10
    - beta_wm: WM inverse temperature; internally scaled by 10
    - p_enc_base: base encoding probability in [0,1]
    - age_penalty: additional reduction in p_enc for older adults in [0,1]

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, beta_rl, beta_wm, p_enc_base, age_penalty = model_parameters
    beta_rl *= 10.0
    beta_wm *= 10.0
    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(beta_rl * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM policy
            denom_wm = np.sum(np.exp(beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # RL uncertainty via softmax entropy (normalized)
            probs_rl = np.exp(beta_rl * (Q_s - np.max(Q_s)))
            probs_rl = probs_rl / max(np.sum(probs_rl), 1e-12)
            entropy_rl = -np.sum(probs_rl * np.log(np.clip(probs_rl, 1e-12, 1.0)))
            max_entropy = np.log(nA)
            uncert = entropy_rl / max(max_entropy, 1e-12)  # in [0,1]

            # Encoding probability: lower with larger set and for older adults
            p_enc = p_enc_base * (3.0 / nS) * (1.0 - age_penalty * age_group)
            p_enc = np.clip(p_enc, 0.0, 1.0)

            # Effective WM weight gates on uncertainty and expected encoding
            wm_weight = np.clip(uncert * p_enc, 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetry
            pe = r - Q_s[a]
            lr_use = lr_pos if pe >= 0 else lr_neg
            q[s, a] = Q_s[a] + lr_use * pe

            # WM updating: encode one-shot with prob p_enc when reward arrives, else slight decay to uniform
            # We approximate stochastic encoding by using its expectation.
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - p_enc) * w[s, :] + p_enc * one_hot
            else:
                # decay toward uniform when not rewarded
                w[s, :] = 0.98 * w[s, :] + 0.02 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with decay and stickiness + WM traces with interference; meta-control via WM cost.

    Overview:
    - RL: delta rule with value decay and choice stickiness in policy.
    - WM: recency-weighted traces strengthened by reward; interference increases with set size.
    - Arbitration: meta-controller down-weights WM when its subjective cost is high.
      WM cost increases with set size and more for older adults.
    - Final policy mixes RL (with stickiness) and WM probabilities.

    Parameters (model_parameters):
    - lr: RL learning rate in [0,1]
    - beta_rl: RL inverse temperature; internally scaled by 10
    - wm_gain_base: base WM gain in [0,1] before interference and cost
    - wm_interf_rate: interference rate per extra item (>=0)
    - wm_cost_base: base WM subjective cost (>=0) modulating arbitration
    - stickiness: choice stickiness parameter added to chosen action logits (>=0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, beta_rl, wm_gain_base, wm_interf_rate, wm_cost_base, stickiness = model_parameters
    beta_rl *= 10.0
    age_group = 0 if age[0] <= 45 else 1

    beta_wm = 40.0  # high precision WM channel
    decay_q = 0.01  # gentle RL decay toward uniform

    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_action = None  # for stickiness

        # Precompute WM interference factor as function of set size
        extra_items = max(0, nS - 3)
        wm_interf = wm_interf_rate * extra_items  # 0 at set size 3, increases for 6
        wm_interf = np.clip(wm_interf, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy with stickiness
            rl_logits = beta_rl * Q_s
            if last_action is not None:
                rl_logits[int(last_action)] += stickiness
            rl_logits -= np.max(rl_logits)
            probs_rl = np.exp(rl_logits)
            probs_rl = probs_rl / max(np.sum(probs_rl), 1e-12)
            p_rl = probs_rl[a]

            # WM policy
            wm_logits = beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            probs_wm = np.exp(wm_logits)
            probs_wm = probs_wm / max(np.sum(probs_wm), 1e-12)
            p_wm = probs_wm[a]

            # Meta-control arbitration: WM gain reduced by interference and cost
            wm_cost = wm_cost_base * (nS / 3.0) * (1.0 + 0.5 * age_group)
            wm_gain = wm_gain_base * (1.0 - wm_interf) / (1.0 + wm_cost)
            wm_weight = np.clip(wm_gain, 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with decay
            pe = r - q[s, a]
            q[s, a] += lr * pe
            # small decay toward uniform to prevent runaway values
            q[s, :] = (1.0 - decay_q) * q[s, :] + decay_q * (1.0 / nA)

            # WM update: reward-based strengthening with interference-driven diffusion
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                strengthen = 0.6 * (1.0 - wm_interf)  # less strengthening under higher interference
                w[s, :] = (1.0 - strengthen) * w[s, :] + strengthen * one_hot
            else:
                # Under no reward, interference causes diffusion toward uniform
                diffuse = 0.05 + 0.25 * wm_interf
                w[s, :] = (1.0 - diffuse) * w[s, :] + diffuse * w_0[s, :]

            last_action = a

        blocks_log_p += log_p

    return -blocks_log_p
