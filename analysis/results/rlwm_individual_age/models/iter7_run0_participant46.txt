def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + capacity-limited Working Memory (WM) with load- and age-gated encoding.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate.
    - WM: one-shot encoding of rewarded state-action pairs that decays over time toward uniform.
      WM contributes a policy mixed with RL; its weight is reduced by higher set size and by older age.
    - Mixing: final choice probability is a convex combination of WM and RL softmax policies.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight_base: baseline WM mixture weight (0..1), before gating by set size and age.
    - softmax_beta: RL inverse temperature; scaled by 10 internally.
    - wm_temp: WM inverse temperature; scaled by 10 internally.
    - wm_decay: per-trial decay of WM toward uniform (0..1), larger means faster decay.
    - gate_slope: strength of set size and age penalty on WM weight (real; >0 increases penalty).

    Age and load effects:
    - WM weight is multiplicatively gated: wm_weight = wm_weight_base * sigmoid(-(gate_slope)*((nS-3)+age_group))
      so older adults and larger set sizes reduce WM contribution.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_temp, wm_decay, gate_slope = model_parameters
    softmax_beta *= 10.0
    wm_temp *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))
        # WM values: start at uniform; will store sharp peaks when encoded
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute WM mix gate for this block based on set size and age
        x = (nS - 3.0) + float(age_group)
        wm_weight = wm_weight_base * (1.0 / (1.0 + np.exp(gate_slope * x)))  # sigmoid(-k*x)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL softmax prob for the chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM softmax prob for the chosen action
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(wm_temp * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM decay toward uniform (applies to all states each step)
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM encoding: one-shot if rewarded, with age- and load-sensitive sharpness
            if r > 0.0:
                # Encoding sharpness increases as set size decreases and for younger adults
                sharp = 1.0 + 2.0 / (1.0 + np.exp(gate_slope * x))  # in [1,3) depending on gate
                # Set the chosen action high, others low, then renormalize
                w_row = np.ones(nA) * 1.0
                w_row[a] = sharp
                w_row = w_row / np.sum(w_row)
                w[s, :] = w_row

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Asymmetric RL + outcome-gated WM + state-specific stickiness.

    Mechanism:
    - RL: tabular Q-learning with separate learning rates for positive vs negative prediction errors.
    - WM: fast mapping stored when rewarded; decays over time. WM weight increases after reward trials.
      WM reliance is reduced for older adults and larger set sizes (fixed functional penalty).
    - Stickiness: within-state tendency to repeat the last chosen action, modeled as an additive bias.

    Parameters (model_parameters):
    - lr_pos: learning rate for positive prediction errors (0..1).
    - lr_neg: learning rate for negative prediction errors (0..1).
    - softmax_beta: RL inverse temperature; scaled by 10 internally.
    - wm_weight_base: base WM mixture weight (0..1).
    - wm_reward_boost: multiplicative boost of WM weight on rewarded trials (>=0).
    - stickiness: additive bias (in value units) to repeat the last action in the same state (can be +/-).

    Age and load effects:
    - WM weight is divided by (1 + (nS-3) + age_group), reducing it under higher load and in older adults.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, softmax_beta, wm_weight_base, wm_reward_boost, stickiness = model_parameters
    softmax_beta *= 10.0
    wm_temp = 50.0  # deterministic WM policy

    age_group = 0 if age[0] <= 45 else 1

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last action per state for stickiness
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Outcome-gated WM weight, with age+load penalty
            denom_penalty = 1.0 + max(0.0, (nS - 3.0)) + float(age_group)
            wm_weight = wm_weight_base / denom_penalty
            if r > 0.0:
                wm_weight = min(1.0, wm_weight * (1.0 + wm_reward_boost))

            # RL softmax with within-state stickiness bias
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stickiness
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM policy
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(wm_temp * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            lr_eff = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += lr_eff * pe

            # WM decay
            w = 0.9 * w + 0.1 * w_0  # mild default decay per trial
            # WM encoding only if rewarded
            if r > 0.0:
                w_row = np.zeros(nA)
                w_row[a] = 1.0
                w[s, :] = w_row

            # Update stickiness memory
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + capacity-limited WM cache: only a subset of states are stored in WM.

    Mechanism:
    - RL: tabular Q-learning with learning rate and softmax choice.
    - WM: a cache that can store a limited number of states per block.
      If a state is cached, its action mapping is stored sharply after reward and decays otherwise.
      The probability that a state is in the cache is proportional to (capacity / set size).
    - Mixing: when the state is cached, WM policy is mixed with RL; otherwise, rely on RL only.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - softmax_beta: RL inverse temperature; scaled by 10 internally.
    - wm_weight_base: WM mixture weight when the state is cached (0..1).
    - cache_capacity: baseline WM capacity in number of states (>=0).
    - age_capacity_penalty: capacity decrement for older adults (>=0).
    - wm_decay: per-trial decay rate of WM toward uniform (0..1).

    Age and load effects:
    - Effective capacity = max(0, cache_capacity - age_capacity_penalty * age_group).
      Probability a state is cached = min(1, effective_capacity / nS).
      Thus older adults and larger set sizes reduce WM availability.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_base, cache_capacity, age_capacity_penalty, wm_decay = model_parameters
    softmax_beta *= 10.0
    wm_temp = 50.0  # sharp WM policy

    age_group = 0 if age[0] <= 45 else 1

    rng = np.random.RandomState(0)  # deterministic caching mask per block for likelihood evaluation

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Determine which states are in WM cache for this block
        eff_cap = max(0.0, cache_capacity - age_capacity_penalty * float(age_group))
        p_cache = min(1.0, eff_cap / max(1.0, float(nS)))

        # Create a binary cache mask per state (fixed for the block for likelihood reproducibility)
        cache_mask = (rng.rand(nS) < p_cache).astype(int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM policy (only if cached)
            if cache_mask[s] == 1:
                W_s = w[s, :]
                denom_wm = np.sum(np.exp(wm_temp * (W_s - W_s[a])))
                p_wm = 1.0 / max(denom_wm, 1e-12)
                p_total = wm_weight_base * p_wm + (1.0 - wm_weight_base) * p_rl
            else:
                p_total = p_rl

            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM decay
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM encoding if cached and rewarded
            if cache_mask[s] == 1 and r > 0.0:
                w_row = np.zeros(nA)
                w_row[a] = 1.0
                w[s, :] = w_row

        blocks_log_p += log_p

    return -blocks_log_p
