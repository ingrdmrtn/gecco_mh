def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with capacity-limited WM and load/age-scaled decay.

    Mechanism:
    - RL: standard delta rule and softmax (template).
    - WM policy: near-deterministic readout from WM distribution.
    - Arbitration: WM weight is scaled by effective slot capacity (K/nS) and age penalty.
    - WM update: reward-contingent write (one-hot toward chosen action) with probability proportional
      to capacity; global decay toward uniform per trial scales with set size and age.

    Parameters (<=6 total):
    - lr: RL learning rate (0..1).
    - wm_weight: base arbitration weight (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - K_slots: WM capacity in item slots; determines probability of successful WM write.
    - decay_wm: baseline WM decay toward uniform at set size 3 (0..1).
    - age_slot_penalty: multiplicative penalty on WM capacity for older group (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, K_slots, decay_wm, age_slot_penalty = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Effective arbitration weight scales with capacity and age
        cap = min(1.0, max(0.0, K_slots) / float(nS))
        cap *= (1.0 - age_slot_penalty * age_group)  # penalize in older group
        cap = float(np.clip(cap, 0.0, 1.0))
        wm_weight_eff_block = float(np.clip(wm_weight * cap, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability of the chosen action (per template)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: near-deterministic softmax over WM weights
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Mixture
            p_total = wm_weight_eff_block * p_wm + (1 - wm_weight_eff_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: reward-contingent write with capacity-limited probability
            # and global decay toward uniform that scales with load and age
            decay = decay_wm * (nS / 3.0) * (1.0 + 0.5 * age_group)
            decay = float(np.clip(decay, 0.0, 0.95))

            w[s, :] = (1 - decay) * w[s, :] + decay * w_0[s, :]

            if r > 0.5:
                # Write strength proportional to capacity cap (already age-adjusted)
                write = cap
                if write > 0:
                    w[s, :] = (1 - write) * w[s, :]
                    w[s, a] += write

            # Renormalize to be safe
            w[s, :] = np.clip(w[s, :], 0, None)
            if np.sum(w[s, :]) > 0:
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with asymmetric RL learning rates and uncertainty-gated WM arbitration plus WM inhibition.

    Mechanism:
    - RL: softmax (template) with separate learning rates for positive/negative prediction errors.
    - WM policy: near-deterministic softmax over WM distribution.
    - Arbitration: WM weight increases when RL is uncertain (high entropy), and decreases with set size and age.
    - WM update: after reward, promote chosen action; after no reward, inhibit chosen and redistribute mass to others.

    Parameters (<=6 total):
    - lr_pos: RL learning rate for positive PE (0..1).
    - lr_neg: RL learning rate for negative PE (0..1).
    - wm_weight: base arbitration weight (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_inhibit: strength of WM inhibition on non-rewarded trials (0..1).
    - age_unc_slope: exponent controlling sensitivity of arbitration to RL uncertainty; also penalized by age.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, wm_inhibit, age_unc_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL chosen-action probability (template)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # RL full policy to compute uncertainty (entropy)
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(logits_rl)
            pi_rl = pi_rl / np.sum(pi_rl)
            entropy = -np.sum(pi_rl * np.log(np.clip(pi_rl, eps, 1.0)))
            H_norm = float(entropy / np.log(nA))  # normalize to [0,1]

            # WM policy
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Arbitration: WM up-weighted when RL is uncertain; penalized by load and age
            wm_weight_eff = wm_weight * (H_norm ** max(0.0, age_unc_slope)) * (3.0 / nS) * (1.0 - 0.25 * age_group)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update with asymmetric learning rates
            delta = r - Q_s[a]
            lr_used = lr_pos if delta >= 0 else lr_neg
            q[s][a] += lr_used * delta

            # WM update: reinforce on reward, inhibit chosen on no-reward
            if r > 0.5:
                # Promote chosen action toward 1
                gain = 1.0 - W_s[a]
                w[s, :] = W_s  # copy
                w[s, a] += gain
            else:
                inh = float(np.clip(wm_inhibit, 0.0, 1.0))
                # Reduce chosen action and redistribute to others uniformly
                reduce = inh * W_s[a]
                w[s, a] = W_s[a] - reduce
                w[s, :] += (reduce / (nA - 1))  # spread to all actions
                w[s, a] -= (reduce / (nA - 1))  # keep net redistribution only to others

            # Renormalize and clip
            w[s, :] = np.clip(w[s, :], 0, None)
            if np.sum(w[s, :]) > 0:
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with choice-trace WM (stickiness) and load/age-scaled WM precision.

    Mechanism:
    - RL: standard delta rule and softmax (template).
    - WM policy: fast choice-trace that favors recently chosen actions within a state;
      precision (beta) is scaled down by set size and age.
    - Arbitration: base wm_weight scaled down by set size and age.
    - WM update: leaky integrator toward recent choice; reward amplifies the trace.

    Parameters (<=6 total):
    - lr: RL learning rate (0..1).
    - wm_weight: base arbitration weight (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - trace_decay: per-trial leak of the WM trace toward uniform (0..1).
    - stickiness_strength: amount added to the chosen action each trial (>0).
    - age_temp_bias: scales WM precision by exp(-age_temp_bias * modifiers); larger => lower WM beta under load/age.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, trace_decay, stickiness_strength, age_temp_bias = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # base WM precision (very deterministic)
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Block-wise arbitration scaling by load and age
        wm_weight_eff_block = wm_weight * (3.0 / nS) * (1.0 - 0.3 * age_group)
        wm_weight_eff_block = float(np.clip(wm_weight_eff_block, 0.0, 1.0))

        # WM precision scaling by load and age via an exponential temperature penalty
        wm_beta_scale = np.exp(-age_temp_bias * ((nS / 3.0) - 1.0 + 0.5 * age_group))
        wm_beta_eff = max(1.0, softmax_beta_wm * wm_beta_scale)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL chosen-action probability
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy using adjusted precision
            logits_wm = wm_beta_eff * (W_s - np.max(W_s))
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Mixture
            p_total = wm_weight_eff_block * p_wm + (1 - wm_weight_eff_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: leaky choice trace with reward amplification
            leak = float(np.clip(trace_decay, 0.0, 1.0))
            w[s, :] = (1 - leak) * w[s, :] + leak * w_0[s, :]

            add = stickiness_strength * (1.0 + 0.5 * r)  # amplify if rewarded
            w[s, a] += add

            # Normalize and clip
            w[s, :] = np.clip(w[s, :], 0, None)
            if np.sum(w[s, :]) > 0:
                w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
