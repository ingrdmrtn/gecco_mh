def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + capacity-limited WM with age- and set-size-modulated decay.

    Idea:
    - RL: standard delta rule with softmax policy.
    - WM: one-shot storage of rewarded state-action associations; when no reward,
      the WM trace decays toward uniform. The effective WM contribution (weight)
      and decay both depend on set size (3 vs 6) and age group.
    - Arbitration: mixture of WM and RL policies with a dynamically scaled WM weight.

    Parameters
    ----------
    model_parameters : [lr, wm_weight_base, softmax_beta, wm_decay_base, age_wm_penalty, setsize_capacity_slope]
        - lr: RL learning rate (0..1).
        - wm_weight_base: base mixture weight for WM (0..1) before modulations.
        - softmax_beta: inverse temperature for RL (scaled x10 internally).
        - wm_decay_base: base WM decay rate toward uniform on non-reward (0..1).
        - age_wm_penalty: multiplicative penalty on WM engagement/maintenance for older adults (>=0).
        - setsize_capacity_slope: how strongly set size (6 vs 3) reduces WM capacity/weight and increases decay (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay_base, age_wm_penalty, setsize_capacity_slope = model_parameters
    beta = softmax_beta * 10.0

    age_group = 0 if age[0] <= 45 else 1
    beta_wm = 50.0  # deterministic WM policy

    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL values and WM store
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size factor: 0 for 3, 1 for 6
        setsize_factor = max(0.0, (nS - 3) / 3.0)

        # Effective WM engagement weight, reduced by set size and age
        # Uses a smooth capacity-like transform
        cap_reduction = 1.0 / (1.0 + np.exp(-3.0 * (1.0 - setsize_factor)))  # ~0 for 3, ~0.95 for 6
        wm_weight_eff = wm_weight_base * (1.0 - setsize_capacity_slope * cap_reduction)
        wm_weight_eff *= (1.0 / (1.0 + age_wm_penalty * age_group))
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        # WM decay on non-reward increases with set size and age
        wm_decay = wm_decay_base * (1.0 + setsize_capacity_slope * setsize_factor) * (1.0 + age_wm_penalty * age_group)
        wm_decay = np.clip(wm_decay, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy (softmax with very high beta)
            denom_wm = np.sum(np.exp(beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM update: rewarded trials stored one-shot; else decay towards uniform
            if r >= 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with uncertainty-adaptive inverse temperature + noisy WM with age-by-set-size interference.

    Idea:
    - RL: standard delta rule, but the softmax temperature adapts to local action-value
      uncertainty (variance across the state's Q values). Higher uncertainty reduces
      effective beta. Strength of this adaptation is parameterized.
    - WM: one-shot storage on reward, but WM readout is noisy; the effective WM inverse
      temperature is reduced by a noise term that increases with set size and age (interference).
    - Arbitration: mixture with a WM weight reduced by age-by-set-size interference.

    Parameters
    ----------
    model_parameters : [lr, wm_weight_base, softmax_beta, beta_unc_slope, wm_noise_base, age_size_interference]
        - lr: RL learning rate.
        - wm_weight_base: base WM mixture weight (0..1).
        - softmax_beta: base inverse temperature for RL (scaled x10 internally).
        - beta_unc_slope: scales the reduction of RL beta with Q-variance in the current state (>=0).
        - wm_noise_base: base WM noise factor reducing WM beta (>=0).
        - age_size_interference: multiplicative factor that amplifies interference for larger set sizes and older adults (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, beta_unc_slope, wm_noise_base, age_size_interference = model_parameters
    beta_base = softmax_beta * 10.0
    age_group = 0 if age[0] <= 45 else 1
    eps = 1e-12

    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        setsize_factor = max(0.0, (nS - 3) / 3.0)

        # Age-by-set-size interference term
        interf = (1.0 + age_group * age_size_interference) * (1.0 + setsize_factor * age_size_interference)

        # Effective WM weight reduced by interference
        wm_weight_eff = wm_weight_base / interf
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        # WM readout noise reduces WM inverse temperature
        beta_wm = 50.0 / (1.0 + wm_noise_base * interf)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL beta adapted by Q-variance (uncertainty)
            q_var = np.var(Q_s)
            beta_rl = beta_base / (1.0 + beta_unc_slope * q_var)

            denom_rl = np.sum(np.exp(beta_rl * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            denom_wm = np.sum(np.exp(beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM update: rewarded trials stored; else mild decay to uniform
            relax = 0.1 * interf  # more interference -> faster relaxation
            relax = np.clip(relax, 0.0, 1.0)
            if r >= 0.5 and pe > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - relax) * w[s, :] + relax * w_0[s, :]

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + surprise-gated WM mixture with age- and set-size penalties.

    Idea:
    - RL: standard delta rule with softmax policy.
    - WM: one-shot storage on reward; otherwise decays toward uniform.
    - Arbitration: WM weight on each trial is a logistic function of surprise
      (unsigned prediction error). Age and set size penalize WM engagement.

    Parameters
    ----------
    model_parameters : [lr, softmax_beta, wm_weight_base, surprise_gain, age_wm_penalty, setsize_wm_penalty]
        - lr: RL learning rate.
        - softmax_beta: inverse temperature for RL (scaled x10 internally).
        - wm_weight_base: base offset in the logistic for WM engagement.
        - surprise_gain: slope linking unsigned PE to WM engagement (>=0).
        - age_wm_penalty: subtractive penalty (in logit space) for older adults (>=0).
        - setsize_wm_penalty: subtractive penalty (in logit space) for larger set sizes (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight_base, surprise_gain, age_wm_penalty, setsize_wm_penalty = model_parameters
    beta = softmax_beta * 10.0
    age_group = 0 if age[0] <= 45 else 1
    beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        setsize_factor = max(0.0, (nS - 3) / 3.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            denom_wm = np.sum(np.exp(beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Surprise-based WM engagement (logistic on unsigned PE with penalties)
            pe_unsigned = abs(r - q[s, a])
            logit = wm_weight_base + surprise_gain * pe_unsigned
            logit -= age_wm_penalty * age_group
            logit -= setsize_wm_penalty * setsize_factor
            wm_weight_eff = 1.0 / (1.0 + np.exp(-logit))
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM update: rewarded trials stored; else gentle decay
            if r >= 0.5 and pe > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                relax = 0.15 + 0.35 * setsize_factor  # faster decay in larger set sizes
                if age_group == 1:
                    relax *= (1.0 + 0.5 * 1.0)  # additional decay for older adults
                relax = np.clip(relax, 0.0, 1.0)
                w[s, :] = (1.0 - relax) * w[s, :] + relax * w_0[s, :]

        blocks_log_p += log_p

    return -float(blocks_log_p)
