def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with reward-driven WM encoding, load-scaled WM decay, and age-biased RL temperature.

    Mechanism:
    - RL: standard delta-rule with softmax; older age reduces RL inverse temperature.
    - WM policy: deterministic softmax over a state-action WM matrix.
    - Arbitration: fixed baseline WM weight scaled by set size; mixture of RL and WM policies.
    - WM update: reward-contingent encoding toward one-hot; per-trial decay toward uniform that increases with load and age.

    Parameters:
    - lr: RL learning rate (0..1).
    - wm_weight: baseline WM mixture weight at set size 3 (0..1).
    - softmax_beta: RL inverse temperature base; internally scaled by 10.
    - wm_encoding_rate: WM write strength on rewarded trials (0..1).
    - wm_decay_base: baseline WM decay toward uniform per trial at set size 3 (0..1).
    - age_beta_mult: multiplicative reduction of RL temperature for older group (0..1); RL beta *= (1 - age_beta_mult*age_group).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_encoding_rate, wm_decay_base, age_beta_mult = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    # Age reduces RL temperature (more exploration in older group)
    softmax_beta *= (1.0 - age_beta_mult * age_group)

    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action (softmax)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over W_s with high precision
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(wm_logits)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Arbitration: scale WM weight by set size (capacity pressure)
            wm_weight_eff = wm_weight * (3.0 / nS)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update:
            # 1) Reward-contingent encoding toward one-hot for the chosen action
            if r > 0.5:
                enc = float(np.clip(wm_encoding_rate * (3.0 / nS), 0.0, 1.0))
                w[s, :] = (1.0 - enc) * w[s, :]
                w[s, a] += enc

            # 2) Load- and age-scaled decay toward uniform
            decay = wm_decay_base * (nS / 3.0) * (1.0 + 0.5 * age_group)
            decay = float(np.clip(decay, 0.0, 0.95))
            # Normalize and decay
            w[s, :] = np.clip(w[s, :], 0.0, None)
            w[s, :] /= np.sum(w[s, :])
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLWM with entropy-based arbitration and Pearce-Hall attention to RL learning.

    Mechanism:
    - RL: delta-rule with attention-modulated learning rate based on surprise |PE|.
      Higher surprise increases learning rate (Pearce-Hall), without changing temperature.
    - WM policy: softmax over WM weights, but mixed with uniform noise that grows with load and age.
    - Arbitration: WM weight scales with WM certainty via entropy and with set size.
      When WM is uncertain (high entropy), arbitration shifts toward RL.

    Parameters:
    - lr_base: baseline RL learning rate (0..1).
    - wm_weight_base: baseline WM mixture weight at set size 3 (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - attention_gain: scales the PE-driven boost to RL learning rate (>=0).
    - wm_noise_base: baseline WM noise (mixture with uniform) at set size 3 (0..1).
    - age_wm_noise_mult: multiplicative increase of WM noise for older group (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_base, wm_weight_base, softmax_beta, attention_gain, wm_noise_base, age_wm_noise_mult = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # very deterministic base before noise mix
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action (softmax)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Compute PE and attention-modulated learning rate
            pe = r - Q_s[a]
            lr_t = lr_base * (1.0 + max(0.0, attention_gain) * abs(pe))
            lr_t = float(np.clip(lr_t, 0.0, 1.0))

            # WM policy: softmax, then noise-mix with uniform
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm_det = np.exp(wm_logits)
            pi_wm_det = pi_wm_det / np.sum(pi_wm_det)

            wm_noise = wm_noise_base * (nS / 3.0) * (1.0 + age_wm_noise_mult * age_group)
            wm_noise = float(np.clip(wm_noise, 0.0, 0.95))
            pi_wm = (1.0 - wm_noise) * pi_wm_det + wm_noise * (1.0 / nA)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Arbitration weight via entropy (certainty) and load
            # Normalize entropy by log(nA) to [0,1]
            ent = -np.sum(pi_wm * np.log(np.clip(pi_wm, eps, 1.0)))
            ent_norm = float(ent / np.log(nA))
            wm_weight_eff = wm_weight_base * (1.0 - ent_norm) * (3.0 / nS)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update with attention-modulated learning rate
            q[s][a] += lr_t * pe

            # WM update: write on reward; forgetting tied to the same noise factor
            if r > 0.5:
                # Strong, fast write to the chosen action
                w[s, :] *= 0.0
                w[s, a] = 1.0

            # Normalize and apply forgetting toward uniform using wm_noise as decay
            w[s, :] = np.clip(w[s, :], 0.0, None)
            if np.sum(w[s, :]) <= 0:
                w[s, :] = w_0[s, :].copy()
            else:
                w[s, :] /= np.sum(w[s, :])

            w[s, :] = (1.0 - wm_noise) * w[s, :] + wm_noise * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL plus elimination-learning WM with lapse.

    Mechanism:
    - RL: standard delta-rule with softmax (template).
    - WM policy: for each state, maintain a candidate set distribution over actions.
      Negative feedback reduces probability of the chosen action (elimination),
      positive feedback concentrates probability on the chosen action.
      Per-trial leak toward uniform increases with load and age.
    - Arbitration: fixed baseline WM weight scaled by set size.
    - Lapse: final mixture with uniform choices.

    Parameters:
    - lr: RL learning rate (0..1).
    - wm_weight: baseline WM mixture weight at set size 3 (0..1).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - elim_leak_base: baseline WM leak toward uniform per trial at set size 3 (0..1).
    - age_elim_penalty: multiplicative increase of WM leak for older group (>=0).
    - lapse_rate: probability of a random choice irrespective of beliefs (0..1).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, elim_leak_base, age_elim_penalty, lapse_rate = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50  # very deterministic when reading WM, though we use distributions directly
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_actions = actions[blocks == b].astype(int)
        block_rewards = rewards[blocks == b].astype(float)
        block_states = states[blocks == b].astype(int)
        block_set_sizes = set_sizes[blocks == b].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        # WM "candidate" distribution
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action (softmax)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: use current candidate distribution directly (with high-precision softmax)
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            pi_wm = np.exp(wm_logits)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = float(np.clip(pi_wm[a], eps, 1.0))

            # Arbitration weight scaled by set size
            wm_weight_eff = wm_weight * (3.0 / nS)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            # Apply lapse
            p_total = (1.0 - lapse_rate) * p_mix + lapse_rate * (1.0 / nA)
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM elimination-learning update
            leak = elim_leak_base * (nS / 3.0) * (1.0 + age_elim_penalty * age_group)
            leak = float(np.clip(leak, 0.0, 0.95))

            if r < 0.5:
                # Eliminate (down-weight) the chosen action; keep others as is, then renormalize
                w[s, a] *= (1.0 - (1.0 - leak))  # equivalent to multiply by leak
            else:
                # Concentrate on the rewarded action (push toward one-hot on a)
                enc = 1.0 - leak
                w[s, :] = (1.0 - enc) * w[s, :]
                w[s, a] += enc

            # Normalize and apply leak toward uniform
            w[s, :] = np.clip(w[s, :], 0.0, None)
            if np.sum(w[s, :]) <= 0:
                w[s, :] = w_0[s, :].copy()
            else:
                w[s, :] /= np.sum(w[s, :])

            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
