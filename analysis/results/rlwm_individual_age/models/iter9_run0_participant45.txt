def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + capacity-limited WM with age and set-size dependent arbitration and WM forgetting.

    Idea:
    - RL: standard Q-learning with softmax policy.
    - WM: state-action template that is set to the last rewarded action and otherwise forgets (drifts to uniform).
    - Arbitration: fixed WM weight reduced by (a) larger set sizes and (b) older age.
    - Age group: older adults receive an additional penalty on WM contribution.

    Parameters
    ----------
    states : array-like of int
        State identity on each trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of float/int
        Feedback on each trial (0 or 1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for the block on each trial (3 or 6).
    age : array-like (single repeated value)
        Participant age (repeated). Age group: 0 if <=45 (young), 1 otherwise (old).
    model_parameters : list or array
        [lr, beta_base, wm_weight_base, wm_forget, ss6_penalty, age_wm_delta]
        - lr: RL learning rate (0..1).
        - beta_base: RL inverse temperature (scaled internally by 10).
        - wm_weight_base: baseline WM influence (0..1).
        - wm_forget: WM drift toward uniform when no rewarded update occurs (0..1).
        - ss6_penalty: WM weight penalty applied when set size == 6 (>=0).
        - age_wm_delta: additional WM penalty applied to older adults (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, beta_base, wm_weight_base, wm_forget, ss6_penalty, age_wm_delta = model_parameters

    softmax_beta = beta_base * 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    age_group = 1 if age[0] > 45 else 0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute block-level WM weight with penalties for large set size and age
        base = np.clip(wm_weight_base, 0.0, 1.0)
        size_penalty = ss6_penalty if nS == 6 else 0.0
        age_penalty = age_group * age_wm_delta
        wm_weight_block = np.clip(base - size_penalty - age_penalty, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Qs = q[s, :].copy()
            Qs = Qs - np.max(Qs)
            exp_rl = np.exp(softmax_beta * Qs)
            p_rl_vec = exp_rl / np.sum(exp_rl)
            p_rl = max(p_rl_vec[a], 1e-12)

            # WM policy: reward sets a one-hot memory; otherwise it drifts toward uniform
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]
            w[s, :] = w[s, :] / np.maximum(np.sum(w[s, :]), 1e-12)

            Ws = w[s, :].copy()
            Ws = Ws - np.max(Ws)
            exp_wm = np.exp(softmax_beta_wm * Ws)
            p_wm_vec = exp_wm / np.sum(exp_wm)
            p_wm = max(p_wm_vec[a], 1e-12)

            # Arbitration: convex combination with block-level WM weight
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL value update
            pe = r - q[s, a]
            q[s, a] += lr * pe

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + WM with learned arbitration gate per block (meta-learner), age-biased initialization.

    Idea:
    - RL: standard Q-learning with softmax policy.
    - WM: state-action template set by rewarded choices; no explicit decay here.
    - Arbitration: a latent gate g_t in [0,1] updated trial-by-trial via delta rule:
        g <- g + g_alpha * (r - expected_r_from_mix) with projection to [0,1].
      This lets the model learn whether to rely more on WM or RL within a block.
    - Age group: older adults start with a lower initial g (less WM reliance).

    Parameters
    ----------
    states : array-like of int
        State identity on each trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of float/int
        Feedback on each trial (0 or 1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for the block on each trial (3 or 6).
    age : array-like (single repeated value)
        Participant age (repeated). Age group: 0 if <=45 (young), 1 otherwise (old).
    model_parameters : list or array
        [lr, beta_base, beta_wm, g_alpha, g_init_base, g_old_delta]
        - lr: RL learning rate (0..1).
        - beta_base: RL inverse temperature for RL (scaled internally by 10).
        - beta_wm: WM inverse temperature (scaled internally by 10 as well).
        - g_alpha: gate learning rate (0..1).
        - g_init_base: baseline initial gate value (0..1).
        - g_old_delta: subtraction from initial gate for older adults (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, beta_base, beta_wm, g_alpha, g_init_base, g_old_delta = model_parameters

    softmax_beta = beta_base * 10.0
    softmax_beta_wm = beta_wm * 10.0

    age_group = 1 if age[0] > 45 else 0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Initialize gate with age bias; optionally allow mild dependence on set size via prior scaling
        g0 = np.clip(g_init_base - age_group * g_old_delta, 0.0, 1.0)
        # Optional weak prior adjustment: smaller initial g in larger set sizes
        g = np.clip(g0 * (3.0 / float(nS)), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Qs = q[s, :].copy()
            Qs = Qs - np.max(Qs)
            exp_rl = np.exp(softmax_beta * Qs)
            p_rl_vec = exp_rl / np.sum(exp_rl)
            p_rl = max(p_rl_vec[a], 1e-12)

            # WM update rule: set to one-hot on reward, otherwise keep the current template
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            w[s, :] = w[s, :] / np.maximum(np.sum(w[s, :]), 1e-12)

            # WM policy
            Ws = w[s, :].copy()
            Ws = Ws - np.max(Ws)
            exp_wm = np.exp(softmax_beta_wm * Ws)
            p_wm_vec = exp_wm / np.sum(exp_wm)
            p_wm = max(p_wm_vec[a], 1e-12)

            # Mix and compute likelihood
            p_mix = np.clip(g, 0.0, 1.0) * p_wm + (1.0 - np.clip(g, 0.0, 1.0)) * p_rl
            p_mix = max(p_mix, 1e-12)
            log_p += np.log(p_mix)

            # Gate meta-update: use reward prediction from mixture as target
            # Expected reward under the mixture policy approximated by chosen action prob
            r_hat = p_mix
            g += g_alpha * (r - r_hat)
            g = float(np.clip(g, 0.0, 1.0))

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with value forgetting + WM with limited slots and probabilistic encoding; age reduces slots.

    Idea:
    - RL: Q-learning with softmax and value forgetting toward uniform each visit.
    - WM: only a subset of states can be actively stored (slots K). On a rewarded trial,
      the state is encoded with probability proportional to available capacity. Encoded states
      yield a high-precision WM policy; non-encoded states yield a low-precision policy near uniform.
    - Arbitration: if a state is encoded, WM dominates (high weight); otherwise WM contributes weakly.
    - Age group: older adults have fewer effective slots (K reduced by age_slots_delta).

    Parameters
    ----------
    states : array-like of int
        State identity on each trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of float/int
        Feedback on each trial (0 or 1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for the block on each trial (3 or 6).
    age : array-like (single repeated value)
        Participant age (repeated). Age group: 0 if <=45 (young), 1 otherwise (old).
    model_parameters : list or array
        [lr, beta_base, q_forget, wm_slots, age_slots_delta, enc_precision]
        - lr: RL learning rate (0..1).
        - beta_base: RL inverse temperature (scaled internally by 10).
        - q_forget: RL forgetting rate toward uniform per visit (0..1).
        - wm_slots: baseline number of WM slots (>=0; capped at 6).
        - age_slots_delta: slot reduction applied to older adults (>=0).
        - enc_precision: WM precision for encoded states (inverse temp multiplier, >=1).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices.
    """
    lr, beta_base, q_forget, wm_slots, age_slots_delta, enc_precision = model_parameters

    softmax_beta = beta_base * 10.0
    softmax_beta_wm = float(enc_precision) * 10.0  # precision for encoded states

    age_group = 1 if age[0] > 45 else 0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        idx = (blocks == b)
        block_actions = actions[idx]
        block_rewards = rewards[idx]
        block_states = states[idx]
        block_set_sizes = set_sizes[idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Determine effective slots after age penalty
        K_eff = max(0.0, float(wm_slots) - age_group * float(age_slots_delta))
        K_eff = float(np.clip(K_eff, 0.0, float(nS)))

        # Track which states are currently encoded in WM (boolean mask)
        encoded = np.zeros(nS, dtype=bool)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with forgetting toward uniform on each visit
            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * (1.0 / nA)
            Qs = q[s, :].copy()
            Qs = Qs - np.max(Qs)
            exp_rl = np.exp(softmax_beta * Qs)
            p_rl_vec = exp_rl / np.sum(exp_rl)
            p_rl = max(p_rl_vec[a], 1e-12)

            # WM encoding rule on reward: encode with probability proportional to capacity usage
            # Approximate probability: p_enc = min(1, (K_eff - current_load + 1) / nS) when rewarded
            current_load = float(np.sum(encoded))
            if r > 0.0:
                if K_eff <= 0.0:
                    will_encode = False
                else:
                    p_enc = np.clip((K_eff - current_load + 1.0) / float(nS), 0.0, 1.0)
                    # Deterministic surrogate using expectation: treat p_enc as WM weight for setting strength
                    will_encode = (p_enc > 0.5)
                if will_encode:
                    encoded[s] = True
                    w[s, :] = 0.0
                    w[s, a] = 1.0
                else:
                    # Weak trace even if not formally encoded
                    w[s, :] = 0.5 * w[s, :] + 0.5 * w_0[s, :]
            else:
                # No reward: do not encode; small relaxation toward uniform for non-encoded states
                if not encoded[s]:
                    w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]

            w[s, :] = w[s, :] / np.maximum(np.sum(w[s, :]), 1e-12)

            # WM policy depends on encoding status
            if encoded[s]:
                Ws = w[s, :].copy()
                Ws = Ws - np.max(Ws)
                exp_wm = np.exp(softmax_beta_wm * Ws)
                p_wm_vec = exp_wm / np.sum(exp_wm)
                wm_weight = 0.85  # WM dominates when encoded
            else:
                # Low-precision, near-uniform WM for non-encoded states
                Ws = w[s, :].copy()
                Ws = Ws - np.max(Ws)
                exp_wm = np.exp(2.0 * Ws)  # very low precision fallback
                p_wm_vec = exp_wm / np.sum(exp_wm)
                wm_weight = 0.15  # weak WM influence when not encoded

            p_wm = max(p_wm_vec[a], 1e-12)

            # Arbitration
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

        blocks_log_p += log_p

    return -blocks_log_p
