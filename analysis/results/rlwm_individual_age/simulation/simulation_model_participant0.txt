def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + WM with PE-based arbitration and capacity-limited encoding (simulation).

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array of ints starting at 0 per block)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, wm_logit, softmax_beta, k_pe, rho, age_penalty]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, wm_logit, softmax_beta, k_pe, rho, age_penalty = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        nS = int(set_sizes[block_idx][0])
        nA = 3

        # Map each state to its correct action (assumes state indices are 0..nS-1 within block)
        correct_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first correct action observed for this state in the block
            mask_s = (block_states == s)
            if np.any(mask_s):
                correct_by_state[s] = block_correct[mask_s][0]
            else:
                # fallback (should not happen if stimuli are complete)
                correct_by_state[s] = 0

        # Initialize RL and WM stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))

        # Store last PE per state for arbitration; start at 0 => high confidence
        last_pe = np.zeros(nS)

        # Precompute effective WM overwrite strength for the block
        cap = (3.0 / float(nS)) * (1.0 - float(age_group) * max(0.0, age_penalty))
        rho_eff = float(np.clip(rho * cap, 0.0, 1.0))

        for t in range(len(block_states)):
            s = block_states[t]

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Arbitration weight from previous PE (confidence signal)
            conf = 1.0 - np.minimum(1.0, np.maximum(0.0, abs(last_pe[s])))
            wm_weight = 1.0 / (1.0 + np.exp(-(wm_logit + k_pe * conf)))

            # RL softmax
            q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * q_shift)
            p_rl /= np.sum(p_rl)

            # WM softmax (very sharp)
            w_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * w_shift)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, eps)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Outcome
            correct_a = correct_by_state[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # PE before update (using Q_s cached above)
            pe = r - Q_s[a]
            last_pe[s] = pe  # stored for next arbitration in this state

            # RL update
            q[s, a] += lr * pe

            # WM update with capacity-limited overwrite
            if rho_eff > 0.0:
                if r > 0:
                    target = np.zeros(nA)
                    target[a] = 1.0
                    w[s, :] = (1.0 - rho_eff) * w[s, :] + rho_eff * target
                else:
                    anti = np.ones(nA) / nA
                    anti[a] = 0.0
                    anti = anti / np.sum(anti)
                    w[s, :] = (1.0 - 0.5 * rho_eff) * w[s, :] + (0.5 * rho_eff) * anti

                # Keep w a proper probability distribution
                w[s, :] = np.maximum(w[s, :], eps)
                w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
