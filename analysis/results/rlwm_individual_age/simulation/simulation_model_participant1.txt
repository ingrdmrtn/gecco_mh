def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for:
    RL with eligibility traces + capacity-limited WM and lapse.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, lambda_et, K_base, K_age_drop, lapse]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, softmax_beta, lambda_et, K_base, K_age_drop, lapse = parameters

    # Parameter constraints and scaling (match fitting code)
    lr = min(max(lr, 0.0), 1.0)
    softmax_beta = max(softmax_beta, 1e-6) * 10.0
    lambda_et = min(max(lambda_et, 0.0), 1.0)
    K_base = max(K_base, 1e-6)
    K_age_drop = max(K_age_drop, 0.0)
    lapse = min(max(lapse, 0.0), 0.2)

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    nA = 3
    eps = 1e-12

    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)

        # Determine number of states in this block
        nS = int(block_set_sizes[0])

        # Initialize RL and WM state for the block
        q = (1.0 / nA) * np.ones((nS, nA))        # Q-values
        e = np.zeros((nS, nA))                    # eligibility traces

        stored_action = -np.ones(nS, dtype=int)   # WM slots, -1 means empty
        recency_counter = np.zeros(nS)            # for LRU eviction

        # Precompute per-state correct action mapping
        # Assumes state labels are 0..nS-1 and consistent across trials
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if idx_s.size > 0:
                state_correct[s] = block_correct[idx_s[0]]

        for t in range(len(block_states)):
            s = int(block_states[t])
            nS_t = int(block_set_sizes[t])

            # Update recency counters (LRU)
            recency_counter += 1.0
            recency_counter[s] = 0.0

            # Compute effective WM capacity and availability weight
            K_eff_cont = max(K_base - K_age_drop * age_group, 0.0) / (1.0 + max(nS_t - 3, 0))
            wm_availability_w = min(1.0, K_eff_cont / max(nS_t, 1))

            # WM policy
            if stored_action[s] >= 0:
                p_wm = np.zeros(nA)
                p_wm[stored_action[s]] = 1.0
                w_wm = wm_availability_w
            else:
                p_wm = np.ones(nA) / nA
                w_wm = 0.0

            # RL policy (softmax)
            Q_s = q[s, :]
            Q_shift = Q_s - np.max(Q_s)
            exp_Q = np.exp(softmax_beta * Q_shift)
            p_rl = exp_Q / max(np.sum(exp_Q), eps)

            # Arbitration and lapse
            p_total = w_wm * p_wm + (1.0 - w_wm) * p_rl
            p_total = (1.0 - lapse) * p_total + lapse * (1.0 / nA)

            # Normalize for safety
            p_total = np.clip(p_total, eps, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr + t] = a

            # Outcome based on correct mapping
            correct_a = int(state_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr + t] = r

            # Eligibility trace update
            e *= lambda_et
            e[s, a] = 1.0

            # RL update
            delta = r - Q_s[a]
            q += lr * delta * e

            # WM storage/update on reward
            if r > 0:
                if stored_action[s] == -1:
                    current_K = np.sum(stored_action >= 0)
                    K_cap = int(np.floor(K_eff_cont + 1e-9))
                    if K_cap < 0:
                        K_cap = 0
                    if current_K >= K_cap and K_cap >= 0:
                        if K_cap == 0 and current_K > 0:
                            # Clear all if no capacity
                            stored_action[:] = -1
                        else:
                            candidates = np.where(stored_action >= 0)[0]
                            if candidates.size > 0:
                                evict_idx = candidates[np.argmax(recency_counter[candidates])]
                                stored_action[evict_idx] = -1
                # Store (or refresh) the current association
                stored_action[s] = a

        tr += len(block_states)

    return simulated_actions, simulated_rewards
