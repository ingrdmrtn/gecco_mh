def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + Capacity-weighted WM mixture with age bias and noisy encoding.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: set size for each trial (array; used per block)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [alpha_rl, beta_rl, wm_weight0, gamma_ss, age_bias, eps_encode]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    alpha_rl, beta_rl, wm_weight0, gamma_ss, age_bias, eps_encode = parameters
    beta_rl *= 10.0
    beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        # Set size is constant within block per the task design (use first)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM arrays
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))

        # Compute block-wise WM mixture weight
        ss_penalty = gamma_ss * max(nS - 3, 0)
        age_penalty = age_bias * (1 if age_group == 1 else 0)
        wm_weight_logit = wm_weight0 - ss_penalty - age_penalty
        wm_weight = 1.0 / (1.0 + np.exp(-wm_weight_logit))

        for t in range(len(block_states)):
            s = block_states[t]
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            rl_logits = beta_rl * (Q_s - np.max(Q_s))
            rl_exp = np.exp(rl_logits)
            p_rl = rl_exp / np.sum(rl_exp)

            # WM policy
            wm_logits = beta_wm * (W_s - np.max(W_s))
            wm_exp = np.exp(wm_logits)
            p_wm = wm_exp / np.sum(wm_exp)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Reward from correct answer on this trial
            r = 1 if a == block_correct[t] else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] = Q_s[a] + alpha_rl * delta

            # WM leak toward uniform
            leak = np.clip(eps_encode * (nS / 6.0), 0.0, 1.0)
            w[s, :] = (1.0 - leak) * W_s + leak * (1.0 / nA)

            # WM encoding on reward (overwrites leaked row when rewarded)
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - eps_encode) * one_hot + eps_encode * (1.0 / nA)

            tr += 1

    return simulated_actions, simulated_rewards
