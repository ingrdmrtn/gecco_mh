import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + surprise-gated WM (mixture) simulation.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action for each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, wm_gate_base, wm_pe_gain, ss_gate_penalty, age_gate_penalty]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """

    lr, softmax_beta, wm_gate_base, wm_pe_gain, ss_gate_penalty, age_gate_penalty = parameters

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_answers = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block (assume stationary per state)
        correct_actions_per_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first occurrence's correct answer for this state
            mask = (block_states == s)
            if np.any(mask):
                correct_actions_per_state[s] = int(block_correct_answers[mask][0])
            else:
                # if a state is never visited (rare), default to 0
                correct_actions_per_state[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL and WM policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            rl_logits = softmax_beta * Q_s
            rl_logits = rl_logits - np.max(rl_logits)
            rl_exp = np.exp(rl_logits)
            p_rl = rl_exp / np.sum(rl_exp)

            # WM softmax (high beta)
            wm_logits = softmax_beta_wm * W_s
            wm_logits = wm_logits - np.max(wm_logits)
            wm_exp = np.exp(wm_logits)
            wm_pol = wm_exp / np.sum(wm_exp)

            # Before we choose an action, we need the gate, which depends on delta.
            # Use expected delta for each candidate action to define a mixture?
            # The fitted model uses the realized delta after observing reward.
            # For simulation, we approximate by using current best estimate for delta with the chosen action.
            # We first compute a provisional action distribution with equal arbitration (or use RL-only to get delta).
            # To stay faithful to the fitted model's trial flow, we compute gate using current state's
            # predicted delta for the action we will take, which requires sampling action with current mixture.
            # However, mixture weight equals gate, which depends on delta. In the fitted likelihood, gate is computed
            # after seeing reward using the realized delta. For simulation, we will:
            # 1) sample an action from RL policy (p_rl) to compute delta used for gating,
            # 2) then compute gate and final mixture and resample action accordingly.
            # This keeps the gate driven by surprise relative to RL value, as in the fitted update step.

            # Step 1: provisional action from RL to evaluate delta for gating
            a_rl = int(np.random.choice(nA, p=p_rl))
            # Reward if we took a_rl:
            r_rl = 1 if a_rl == correct_actions_per_state[s] else 0
            delta_prov = r_rl - Q_s[a_rl]

            # Compute gate (WM weight)
            penalty = ss_gate_penalty * max(0, nS - 3) + age_gate_penalty * age_group
            gate_input = wm_gate_base + wm_pe_gain * np.abs(delta_prov) - penalty
            wm_weight = 1.0 / (1.0 + np.exp(-gate_input))
            wm_weight = float(np.clip(wm_weight, 0.0, 1.0))

            # Final mixture policy
            p_total = wm_weight * wm_pol + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample final action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Realized reward
            r = 1 if a == correct_actions_per_state[s] else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_weight) * w[s, :] + wm_weight * target
            else:
                w[s, a] = (1.0 - 0.5 * wm_weight) * w[s, a]

            # WM decay toward uniform, with load and age effects
            decay = 0.05 + 0.05 * max(0, nS - 3) + 0.05 * age_group
            decay = np.clip(decay, 0.0, 0.5)
            w = (1.0 - decay) * w + decay * w_0

            # Normalize WM rows to be valid distributions
            row_sums = np.sum(w, axis=1, keepdims=True)
            row_sums = np.clip(row_sums, 1e-12, None)
            w = w / row_sums

            tr += 1

    return simulated_actions, simulated_rewards
