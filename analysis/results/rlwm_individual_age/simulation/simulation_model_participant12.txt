def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for:
    RL with asymmetric learning, capacity-limited episodic WM retrieval, and age-dependent perseveration.

    Arguments:
    - stimulus: sequence of stimuli (state indices) per trial (array-like)
    - blocks: block index per trial (array-like)
    - set_sizes: block-dependent set sizes per trial (array-like) [not strictly needed beyond sanity]
    - correct_answer: correct action per trial (array-like)
    - age: participant's age (array-like, single repeated value)
    - parameters: [lr_pos, lr_neg, beta, persev_y, persev_o, capacity_C]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr_pos, lr_neg, beta, persev_y, persev_o, capacity_C = parameters
    softmax_beta = beta * 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1
    persev_bonus = persev_y if age_group == 0 else persev_o
    age_factor = 1.0 - 0.3 * age_group  # older group retrieves less

    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = np.array(stimulus)[block_indices]
        block_correct = np.array(correct_answer)[block_indices]

        nA = 3
        # Determine number of states in this block
        unique_states = np.unique(block_states)
        nS = int(len(unique_states))

        # Map raw state ids (which might not be 0..nS-1) to consecutive indices
        state_map = {s: i for i, s in enumerate(unique_states)}
        mapped_block_states = np.array([state_map[s] for s in block_states])

        # Build per-state correct action mapping using first occurrence within the block
        correct_actions = np.zeros(nS, dtype=int)
        for s_raw, s_idx in state_map.items():
            first_occ = np.where(block_states == s_raw)[0][0]
            correct_actions[s_idx] = block_correct[first_occ]

        # Initialize RL and WM/episodic memory
        q = (1.0 / nA) * np.ones((nS, nA))
        epi_action = -np.ones(nS, dtype=int)  # -1 means no stored episodic action
        epi_strength = np.zeros(nS)           # in [0,1]
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_action = None

        for t in range(len(block_indices)):
            s = mapped_block_states[t]

            # RL policy with perseveration bias
            Q_s = q[s, :].copy()
            if last_action is not None:
                Q_s[last_action] += persev_bonus
            Q_s_centered = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_s_centered)
            p_rl = p_rl / np.maximum(np.sum(p_rl), eps)

            # WM retrieval probability and WM policy
            cap_weight = min(1.0, max(0.0, capacity_C) / max(1.0, float(nS)))
            retrieve_p = age_factor * cap_weight * np.clip(epi_strength[s], 0.0, 1.0)
            retrieve_p = np.clip(retrieve_p, 0.0, 1.0)

            if epi_action[s] >= 0:
                one_hot = np.zeros(nA)
                one_hot[epi_action[s]] = 1.0
                w[s, :] = (1.0 - epi_strength[s]) * w_0[s, :] + epi_strength[s] * one_hot
            else:
                w[s, :] = w_0[s, :]

            W_s = w[s, :]
            W_s_centered = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_s_centered)
            p_wm = p_wm / np.maximum(np.sum(p_wm), eps)

            wm_weight = retrieve_p
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.maximum(np.sum(p_total), eps)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for this state
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_indices[t]] = r

            # RL update (asymmetric learning rates)
            pe = r - q[s, a]
            lr_use = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += lr_use * pe

            # Episodic memory update (store rewarded action; strengthen/decay strength)
            if r > 0.5:
                epi_action[s] = a
                epi_strength[s] = epi_strength[s] + (1.0 - epi_strength[s]) * 0.5
            else:
                epi_strength[s] = epi_strength[s] * 0.5

            # Update perseveration
            last_action = a

            tr += 1

    return simulated_actions, simulated_rewards
