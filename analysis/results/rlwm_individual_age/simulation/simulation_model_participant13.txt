def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL + one-shot WM model with retrieval noise
    and set-size/age effects on RL precision.

    Arguments:
    - stimulus: sequence of stimuli (state indices) per trial (array-like, ints)
    - blocks: block index for each trial (array-like, ints)
    - set_sizes: set size for each trial (array-like, ints; constant within block)
    - correct_answer: correct action for each trial (array-like, ints)
    - age: participant's age (array-like with single repeated value)
    - parameters: list with 6 parameters
        0) lr: RL learning rate (0..1)
        1) wm_weight: WM mixture weight (0..1)
        2) beta_base: base RL inverse temperature (scaled by 10 internally)
        3) wm_eta: WM retrieval noise (>0). WM beta = 1 / wm_eta
        4) size_beta_scale: scales how much set size reduces RL beta
        5) age_beta_shift: additive beta shift for older group (scaled by 10 internally)

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, wm_weight, beta_base, wm_eta, size_beta_scale, age_beta_shift = parameters
    beta_base *= 10.0
    age_beta_shift *= 10.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate over blocks
    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.array(stimulus)[block_idx].astype(int)
        block_correct = np.array(correct_answer)[block_idx].astype(int)
        block_set_sizes = np.array(set_sizes)[block_idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])  # follow fitting code: set size defined per block

        # RL inverse temperature with set-size and age adjustments
        beta_sz = beta_base / (1.0 + size_beta_scale * max(0, nS - 3))
        softmax_beta = beta_sz + (age_beta_shift if age_group == 1 else 0.0)

        # WM inverse temperature from retrieval noise
        softmax_beta_wm = 1.0 / max(1e-6, wm_eta)

        # Build mapping from state -> correct action within block
        # Assume states are in [0, nS-1]
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of this state in the block to read correct action
            mask_s = (block_states == s)
            if np.any(mask_s):
                correct_actions[s] = block_correct[mask_s][0]
            else:
                # if a state is absent (shouldn't happen), default to 0
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))

        # Step through trials in this block
        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(z_rl)
            p_rl /= np.sum(p_rl)

            # WM policy
            z_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(z_wm)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_idx[t]] = a

            # Generate reward from correct action
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM one-shot win-stay with no decay; only on rewarded trials
            if r > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
