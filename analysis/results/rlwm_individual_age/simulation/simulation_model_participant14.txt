def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL + WSLS-like WM model with age/set-size gate.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of state IDs)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [alpha_rl, beta_rl, wm_weight0, wm_wsls_bias, neg_mult]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    alpha_rl, beta_rl, wm_weight0, wm_wsls_bias, neg_mult = parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_correct_ans = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        # Map states in this block to 0..nS-1 (in case input states are not zero-based/contiguous)
        unique_states = np.unique(block_states_raw)
        state_to_local = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_local[s] for s in block_states_raw], dtype=int)

        nA = 3
        # Prefer set size provided per block; assume constant within block
        nS = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else len(unique_states)
        # Safety: ensure nS matches observed unique states
        nS = max(nS, len(unique_states))

        # Derive per-state correct action (assumed constant within a block)
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for s_local in range(len(unique_states)):
            first_idx = np.where(block_states == s_local)[0][0]
            correct_actions_by_state[s_local] = int(block_correct_ans[first_idx])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # not used in updates but kept for parity

        last_action = -np.ones(nS, dtype=int)
        last_reward = -np.ones(nS, dtype=int)

        # Compute effective WM weight for this block
        age_scale = 1.1 if age_group == 0 else 0.9
        size_scale = 3.0 / float(nS)
        wm_weight_eff = np.clip(wm_weight0 * age_scale * size_scale, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            q_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(q_logits)
            p_rl /= np.sum(p_rl)

            # WM WSLS heuristic
            wm_pref = np.ones(nA) / nA
            if last_action[s] != -1:
                a_last = last_action[s]
                if last_reward[s] == 1:
                    # Win-stay: concentrate probability on last action
                    wm_pref = np.ones(nA) * (1.0 - 1.0 / nA) / (nA - 1)
                    wm_pref[a_last] = 1.0 / nA + (1.0 - 1.0 / nA)
                elif last_reward[s] == 0:
                    # Lose-shift: zero on last action, uniform on others
                    wm_pref = np.ones(nA) / (nA - 1)
                    wm_pref[a_last] = 0.0

            wsls_vec = wm_pref
            logits_wm = wm_wsls_bias * (wsls_vec - 1.0 / nA)
            wm_read = W_s + logits_wm

            wm_logits = softmax_beta_wm * (wm_read - np.max(wm_read))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture policy and action sampling
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward generation (deterministic correct/incorrect)
            correct_a = correct_actions_by_state[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update with amplified negative prediction errors
            delta = r - Q_s[a]
            delta_eff = delta if delta >= 0.0 else neg_mult * delta
            q[s, a] += alpha_rl * delta_eff

            # Update last action/reward for WSLS
            last_action[s] = a
            last_reward[s] = r

            # WM value update toward WSLS preference
            w[s, :] = 0.5 * w[s, :] + 0.5 * wsls_vec

            tr += 1

    return simulated_actions, simulated_rewards
