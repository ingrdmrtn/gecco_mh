def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from the RL + leaky WM model with reward-rate-driven temperature
    and load/reward-dependent WM mixture.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of state indices)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action for each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, beta_gain, wm_weight_base, tau_meta, wm_leak]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, softmax_beta, beta_gain, wm_weight_base, tau_meta, wm_leak = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        # Use provided block set size (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action using first occurrence
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = block_correct[idx[0]]
            else:
                # Fallback: if this state doesn't appear (rare), assign a default
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Running reward-rate estimate
        m = 0.5

        for t in range(len(block_states)):
            s = int(block_states[t])
            set_size = int(block_set_sizes[t])

            # Effective RL inverse temperature with meta-learning and age attenuation
            beta_gain_eff = beta_gain * (1.0 - 0.3 * age_group)
            beta_rl_eff = softmax_beta * (1.0 + beta_gain_eff * (m - 0.5))
            beta_rl_eff = max(0.0, beta_rl_eff)

            # RL policy
            Q_s = q[s, :]
            q_shift = Q_s - np.max(Q_s)
            pi_rl = np.exp(np.clip(beta_rl_eff * q_shift, -700, 700))
            pi_rl = pi_rl / np.sum(pi_rl)

            # WM policy
            W_s = w[s, :]
            w_shift = W_s - np.max(W_s)
            pi_wm = np.exp(np.clip(softmax_beta_wm * w_shift, -700, 700))
            pi_wm = pi_wm / np.sum(pi_wm)

            # Mixture weight depends on load and recent reward-rate
            load_factor = 3.0 / float(set_size)
            reward_factor = 0.8 + 0.4 * m
            wm_weight_eff = np.clip(wm_weight_base * load_factor * reward_factor, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * pi_wm + (1.0 - wm_weight_eff) * pi_rl
            p_total = np.maximum(p_total, eps)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Deliver reward according to correct action mapping
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: learn from reward, then leak toward uniform
            wm_learn = np.clip(1.0 - wm_leak, 0.0, 1.0)
            if r > 0:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target

            leak_eff = np.clip(wm_leak * (set_size / 3.0) * (1.0 + 0.5 * age_group), 0.0, 1.0)
            w[s, :] = (1.0 - leak_eff) * w[s, :] + leak_eff * w_0[s, :]

            # Update running reward-rate estimate after observing outcome
            m = (1.0 - tau_meta) * m + tau_meta * r

            tr += 1

    return simulated_actions, simulated_rewards
