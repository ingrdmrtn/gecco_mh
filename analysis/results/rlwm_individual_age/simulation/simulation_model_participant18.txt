def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for:
    RL with asymmetric learning rates + WM gating by surprise and load, with age-dependent gating bias.

    Arguments:
    - stimulus: sequence of stimuli (state indices) per trial (array-like, ints)
    - blocks: block index per trial (array-like, ints)
    - set_sizes: set size per trial (array-like, ints; constant within each block, e.g., 3 or 6)
    - correct_answer: correct action per trial (array-like, ints in [0,1,2])
    - age: participant age (array-like with single repeated value)
    - parameters: [lr_pos, lr_neg, softmax_beta, wm_gate_base, age_gate_effect]

    Returns:
    - simulated_actions: array of chosen actions (ints)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr_pos, lr_neg, softmax_beta, wm_gate_base, age_gate_effect = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus)[block_indices].astype(int)
        block_correct = np.asarray(correct_answer)[block_indices].astype(int)
        block_set_sizes = np.asarray(set_sizes)[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])  # assumes states are 0..nS-1 within block
        load_penalty = (nS - 3) / 3.0  # 0 for 3, 1 for 6

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track previous surprise per state to set the WM gate for the next visit
        prev_surprise = 0.5 * np.ones(nS)  # neutral initial surprise

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute WM gate weight using previous surprise for this state
            wm_logit = wm_gate_base + 2.0 * (prev_surprise[s] - 0.5) - 1.0 * load_penalty + (age_gate_effect if age_group == 1 else 0.0)
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

            # RL policy
            Q_s = q[s, :]
            rl_scores = softmax_beta * Q_s
            rl_scores -= np.max(rl_scores)
            p_rl = np.exp(rl_scores)
            p_rl /= np.sum(p_rl)

            # WM policy (near-deterministic)
            W_s = w[s, :]
            wm_scores = softmax_beta_wm * W_s
            wm_scores -= np.max(wm_scores)
            p_wm = np.exp(wm_scores)
            p_wm /= np.sum(p_wm)

            # Arbitration
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[tr] = r

            # RL update
            pe = r - Q_s[a]
            if pe >= 0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # Compute current-trial surprise (for WM encoding and next-trial gating)
            surprise = min(1.0, abs(pe))

            # WM decay towards baseline (row-wise)
            w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]

            # Surprise-dependent partial reset of the row towards baseline
            w[s, :] = (1.0 - surprise) * w[s, :] + surprise * w_0[s, :]

            # Encode outcome in WM for chosen action
            if r > 0:
                w[s, a] = (1.0 - surprise) * w[s, a] + surprise * 1.0
            else:
                w[s, a] = 0.5 * w[s, a] + 0.5 * w_0[s, a]

            # Update gating driver for this state
            prev_surprise[s] = surprise

            tr += 1

    return simulated_actions, simulated_rewards
