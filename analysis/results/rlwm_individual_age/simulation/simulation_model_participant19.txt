def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using RL + decaying WM with age- and load-modulated WM precision.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, size6_noise, age_wm_boost]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, size6_noise, age_wm_boost = parameters

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # base WM precision

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action using the first occurrence within the block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idxs = np.where(block_states == s)[0]
            if len(idxs) > 0:
                correct_actions[s] = block_correct[idxs[0]]
            else:
                # Fallback (should not happen if each state appears at least once)
                correct_actions[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size penalty and age modulation (constant within block)
        size_penalty = (1.0 - size6_noise) if nS == 6 else 1.0
        age_mult = 1.0 + age_wm_boost * (1 - 2 * age_group)
        wm_beta_eff = max(1e-6, softmax_beta_wm * size_penalty * age_mult)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(z_rl)
            p_rl /= np.sum(p_rl)

            # WM policy
            z_wm = wm_beta_eff * (W_s - np.max(W_s))
            p_wm = np.exp(z_wm)
            p_wm /= np.sum(p_wm)

            # Arbitration
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for this state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM strengthening on rewarded trials
            if r > 0:
                strengthen = size_penalty  # less strengthening for set size 6
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - strengthen) * w[s, :] + strengthen * onehot

            tr += 1

    return simulated_actions, simulated_rewards
