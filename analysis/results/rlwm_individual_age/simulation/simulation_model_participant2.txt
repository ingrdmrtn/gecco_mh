def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulation for:
    RL with adaptive arbitration by surprise (prediction error) + WM win-stay cache.

    Inputs:
    - stimulus: sequence of stimuli for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block dependent set sizes on each trial (array)
    - correct_answer: correct action (0..2) on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_rl, beta_wm, wm_gate_base, pe_gate_gain, age_gate_penalty]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr, beta_rl, beta_wm, wm_gate_base, pe_gate_gain, age_gate_penalty = parameters
    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = max(1e-6, beta_wm)

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_actions = correct_answer[block_indices]

        nA = 3
        # Determine number of states in this block
        nS = int(set_sizes[block_indices][0])

        # Build mapping from state id to its correct action within the block
        # Assume stimuli are already coded 0..nS-1
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            # find first trial of this state to pick its correct action
            idx = np.where(block_states == st)[0][0]
            correct_actions[st] = int(block_correct_actions[idx])

        # Initialize RL and WM structures
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        wm_cache = -1 * np.ones(nS, dtype=int)

        # Maintain last absolute PE per state for arbitration (proxy for current-trial surprise)
        last_abs_pe = np.zeros(nS)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            Q_s = q[s, :].copy()
            Q_s_centered = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_s_centered)
            p_rl /= np.sum(p_rl)

            # WM policy: use cached deterministic preference if available, else baseline w row
            W_s = w[s, :].copy()
            if wm_cache[s] >= 0:
                cached = wm_cache[s]
                W_s[:] = (1.0 - 0.9) / (nA - 1)
                W_s[cached] = 0.9
            W_s_centered = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_s_centered)
            p_wm /= np.sum(p_wm)

            # Arbitration weight based on last absolute PE (state-specific), set size, and age
            size_penalty = (float(nS) / 3.0 - 1.0)  # 0 at set size 3, 1 at set size 6
            wm_weight_raw = wm_gate_base + pe_gate_gain * last_abs_pe[s]
            wm_weight_raw -= age_gate_penalty * age_group * (1.0 + size_penalty)
            wm_weight = np.clip(wm_weight_raw, 0.0, 1.0)

            # Mixed policy and action sampling
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Environment feedback (deterministic mapping)
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # Update WM cache on rewarded trials
            if r > 0:
                wm_cache[s] = a

            # WM leak toward baseline
            leak = np.clip(0.03 + 0.04 * (float(nS) / 3.0) + 0.03 * age_group, 0.0, 1.0)
            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

            # Update last abs PE for future arbitration
            last_abs_pe[s] = np.abs(pe)

            tr += 1

    return simulated_actions, simulated_rewards
