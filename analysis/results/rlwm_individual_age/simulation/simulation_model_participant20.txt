def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate actions and rewards for the RLâ€“WM mixture with WM decay and age-dependent interference.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array-like of ints, state indices)
    - blocks: block index for each trial (array-like)
    - set_sizes: block-dependent set sizes on each trial (array-like)
    - correct_answer: correct action for each trial (array-like of ints)
    - age: participant's age (array-like or scalar)
    - parameters: [lr, wm_weight_base, softmax_beta, wm_decay, wm_interference_age]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, wm_decay, wm_interference_age = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM

    try:
        age_group = 0 if age[0] <= 45 else 1
    except Exception:
        age_group = 0 if age <= 45 else 1

    stimulus = np.asarray(stimulus).astype(int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes).astype(int)
    correct_answer = np.asarray(correct_answer).astype(int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        nA = 3
        # Set size as specified for the block (matching fitting code)
        nS = int(set_sizes[block_indices[0]])

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Build mapping from state to its correct action within the block
        # Assumes each state has a fixed correct action in a block
        state_to_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for idx_local, s in enumerate(block_states):
            if not seen[s]:
                state_to_correct[s] = block_correct[idx_local]
                seen[s] = True

        # Precompute mixture weight terms that depend only on nS and age
        base_scale = 3.0 / nS
        interference = wm_interference_age * age_group * max(0.0, (nS - 3) / 3.0)
        wm_weight = np.clip(wm_weight_base * base_scale * (1.0 - interference), 0.0, 1.0)

        for j, trial_idx in enumerate(block_indices):
            s = block_states[j]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            exp_rl = np.exp(logits_rl)
            p_rl = exp_rl / np.sum(exp_rl)

            # WM policy
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            exp_wm = np.exp(logits_wm)
            p_wm = exp_wm / np.sum(exp_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[trial_idx] = a

            # Generate reward based on correct action
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[trial_idx] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update with decay toward target
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
            else:
                target = w_0[s, :]
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * target

            # Normalize and floor
            w[s, :] = np.maximum(w[s, :], 1e-12)
            w[s, :] /= np.sum(w[s, :])

            # Additional age-dependent interference across all states in high set sizes
            if nS > 3 and age_group == 1:
                lam = np.clip(wm_interference_age * (nS - 3) / 3.0, 0.0, 1.0)
                w = (1.0 - lam) * w + lam * w_0
                w = np.maximum(w, 1e-12)
                w = (w.T / np.sum(w, axis=1)).T

    return simulated_actions, simulated_rewards
