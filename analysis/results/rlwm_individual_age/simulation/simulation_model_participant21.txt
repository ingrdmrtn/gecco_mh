def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with RL forgetting, WM limited precision, and action perseveration.
    Simulates actions and rewards following the logic of cognitive_model3.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: list of 6 fitted parameter values

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    eps = 1e-12

    # Unpack and transform parameters to match fitting code
    alpha, beta, wm_w0, wm_precision, q_forget, persev = parameters
    alpha = 1.0 / (1.0 + np.exp(-alpha))
    q_forget = 1.0 / (1.0 + np.exp(-q_forget))
    wm_fidelity = 1.0 / (1.0 + np.exp(-wm_precision))
    softmax_beta = abs(beta) * 10.0
    softmax_beta_wm = 50.0

    # Age effects
    age_group = 0 if age[0] <= 45 else 1
    if age_group == 1:
        q_forget = np.clip(q_forget * 1.25, 0.0, 1.0)
        wm_fidelity *= 0.8

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Use provided set sizes consistent with fitting code
        nS = int(block_set_sizes[0])

        # Map a correct action per state (assumes fixed mapping within block)
        # If states are 0..nS-1 within block, this works directly
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                correct_actions[s] = block_correct[idx_s[0]]
            else:
                # Fallback to a default valid action if state not observed yet
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w0 = (1.0 / nA) * np.ones((nS, nA))

        # WM mixture weight scaled by load
        wm_weight_block = (1.0 / (1.0 + np.exp(-wm_w0))) * (3.0 / float(nS))
        wm_weight_block = np.clip(wm_weight_block, 0.0, 1.0)

        # Perseveration: previous action per state
        prev_action = -1 * np.ones(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with perseveration bias
            Q_s = q[s, :].copy()
            if prev_action[s] >= 0:
                Q_s[prev_action[s]] += persev
            # Softmax for RL
            x_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(x_rl)
            p_rl = p_rl / max(np.sum(p_rl), eps)

            # WM policy (deterministic-ish)
            W_s = w[s, :].copy()
            x_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(x_wm)
            p_wm = p_wm / max(np.sum(p_wm), eps)

            # Mixture
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = p_total / max(np.sum(p_total), eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward based on correct action for this state
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_idx[t]] = r

            # RL update: Q-learning with forgetting toward uniform
            pe = r - q[s, a]
            q[s, a] += alpha * pe
            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * (1.0 / nA)

            # WM update: on reward, store noisy one-hot; else reset to uniform
            if r == 1:
                w[s, :] = ((1.0 - wm_fidelity) / (nA - 1)) * np.ones(nA)
                w[s, a] = wm_fidelity
            else:
                w[s, :] = w0[s, :].copy()

            # Update perseveration memory
            prev_action[s] = a

            tr += 1

    return simulated_actions, simulated_rewards
