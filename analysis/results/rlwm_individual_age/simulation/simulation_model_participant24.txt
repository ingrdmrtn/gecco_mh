def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the RL + WM model with global perseveration bias
    and load-sensitive WM strength (converted from the provided fitting code).

    Inputs:
    - stimulus: state index on each trial (array of ints, 0..nS-1 within a block)
    - blocks: block index for each trial (array)
    - set_sizes: set size for each block, repeated on block trials (array)
    - correct_answer: correct action on each trial (array of ints, 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_y, beta_o, perseveration, wm_eta, load_sensitivity]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, beta_y, beta_o, perseveration, wm_eta, load_sensitivity = parameters

    age_group = 0 if age[0] <= 45 else 1
    beta = beta_y if age_group == 0 else beta_o
    beta *= 10.0
    beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)

        nA = 3
        nS = int(set_sizes[block_indices][0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w0 = (1.0 / nA) * np.ones((nS, nA))

        last_action_global = -1

        setsize_factor = max(nS - 3, 0) / 3.0
        wm_leak = np.clip(load_sensitivity * setsize_factor, 0.0, 1.0)

        for i, tr in enumerate(block_indices):
            s = int(block_states[i])

            Q_s = q[s, :].copy()
            bias = np.zeros(nA)
            if last_action_global >= 0:
                bias[last_action_global] += perseveration

            rl_logits = beta * (Q_s + bias - np.max(Q_s + bias))
            exp_rl = np.exp(rl_logits)
            pi_rl = exp_rl / np.sum(exp_rl)

            W_s = w[s, :]
            wm_logits = beta_wm * (W_s - np.max(W_s))
            exp_wm = np.exp(wm_logits)
            pi_wm = exp_wm / np.sum(exp_wm)

            wm_conf = np.max(W_s) - (1.0 / nA)
            wm_conf = np.clip(wm_conf / (1.0 - 1.0 / nA), 0.0, 1.0)

            wm_weight = np.clip(0.5 + wm_eta * (wm_conf - load_sensitivity * setsize_factor), 0.0, 1.0)

            p_total = wm_weight * pi_wm + (1.0 - wm_weight) * pi_rl
            p_total = p_total / np.sum(p_total)

            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            r = 1 if a == int(block_correct[i]) else 0
            simulated_rewards[tr] = r

            pe = r - q[s, a]
            q[s, a] += lr * pe

            w = (1.0 - wm_leak) * w + wm_leak * w0

            if r > 0.5:
                w[s, :] = (1.0 - wm_eta) * w[s, :]
                w[s, a] += wm_eta
                w[s, :] /= np.sum(w[s, :])

            last_action_global = a

    return simulated_actions, simulated_rewards
