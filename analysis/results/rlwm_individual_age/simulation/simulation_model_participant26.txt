def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for:
    RL with eligibility traces + WM with interference across states.

    Arguments:
    - stimulus: sequence of stimuli (state IDs) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action for each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, wm_weight_base, interference, age_interference_boost, trace_lambda]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, softmax_beta, wm_weight_base, interference, age_interference_boost, trace_lambda = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)

        nA = 3
        # Use provided set size for the block (consistent with fitting code)
        nS = int(set_sizes[block_indices][0])

        # Build per-state correct action map from the trials in this block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            mask_s = np.where(block_states == s)[0]
            if len(mask_s) > 0:
                correct_actions[s] = block_correct[mask_s[0]]
            else:
                # Fallback if a state is not observed (shouldn't happen), set arbitrary action
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        e = np.zeros((nS, nA))

        # Block-level arbitration and interference
        wm_weight_block = wm_weight_base * (3.0 / max(1.0, float(nS)))
        wm_weight_block = min(1.0, max(0.0, wm_weight_block))

        inter_base = interference * (float(nS) / 3.0)
        inter_eff = inter_base * (1.0 + age_interference_boost * age_group)
        inter_eff = min(1.0, max(0.0, inter_eff))

        for t in range(len(block_states)):
            s = block_states[t]

            # Softmax for RL
            Q_s = q[s, :].copy()
            x_rl = softmax_beta * Q_s
            x_rl -= np.max(x_rl)
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)

            # Softmax for WM
            W_s = w[s, :].copy()
            x_wm = softmax_beta_wm * W_s
            x_wm -= np.max(x_wm)
            p_wm = np.exp(x_wm)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = np.clip(p_total, eps, None)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Reward from ground truth mapping
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_indices[t]] = r

            # RL update with eligibility traces
            pe = r - Q_s[a]
            e *= trace_lambda
            e[s, a] += 1.0
            q += lr * pe * e

            # WM decay toward baseline for current state (interference-driven)
            w[s, :] = (1.0 - 0.5 * inter_eff) * w[s, :] + (0.5 * inter_eff) * w_0[s, :]

            # WM storage with interference spillover upon reward
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0

                gamma_main = 1.0 - 0.5 * inter_eff
                w[s, :] = (1.0 - gamma_main) * w[s, :] + gamma_main * target

                if nS > 1:
                    spill = inter_eff / max(1, nS - 1)
                    for s_other in range(nS):
                        if s_other == s:
                            continue
                        w[s_other, :] = (1.0 - spill) * w[s_other, :] + spill * target

                # Normalize per-state WM distributions
                for s_norm in range(nS):
                    w[s_norm, :] = np.clip(w[s_norm, :], eps, None)
                    w[s_norm, :] /= np.sum(w[s_norm, :])

            tr += 1

    return simulated_actions, simulated_rewards
