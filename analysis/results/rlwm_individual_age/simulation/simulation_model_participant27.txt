def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the cognitive_model3 dynamics.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of ints)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct answer on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_rl_base, beta_wm_base, arb_bias, arb_load_slope, arb_age_shift]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, beta_rl_base, beta_wm_base, arb_bias, arb_load_slope, arb_age_shift = parameters

    beta_rl = beta_rl_base * 10.0
    beta_wm_base = beta_wm_base * 10.0

    age_group = 0 if age[0] <= 45 else 1
    age_term = (arb_age_shift if age_group == 1 else -arb_age_shift)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    eps = 1e-12
    tr = 0

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct_ans = correct_answer[block_indices].astype(int)
        nS = int(set_sizes[block_indices][0])

        # Map per-state correct action from first occurrence in block
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for st in range(nS):
            idx = np.where(block_states == st)[0]
            if len(idx) > 0:
                correct_actions_by_state[st] = block_correct_ans[idx[0]]
            else:
                correct_actions_by_state[st] = 0

        # Initialize RL and WM
        Q = (1.0 / nA) * np.ones((nS, nA))
        W = (1.0 / nA) * np.ones((nS, nA))
        U = (1.0 / nA) * np.ones((nS, nA))

        # Arbitration and WM settings for this block
        load_term = max(0, nS - 3)
        z = arb_bias + arb_load_slope * load_term + age_term
        wm_weight_block = 1.0 / (1.0 + np.exp(-z))
        wm_weight_block = np.clip(wm_weight_block, 0.0, 1.0)

        beta_wm = beta_wm_base / (1.0 + 0.2 * load_term)
        wm_decay = 0.02 + 0.01 * load_term

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            Q_s = Q[s, :]
            logits_rl = beta_rl * (Q_s - np.max(Q_s))
            p_rl_vec = np.exp(logits_rl)
            p_rl_vec = p_rl_vec / max(np.sum(p_rl_vec), eps)

            # WM policy
            W_s = W[s, :]
            logits_wm = beta_wm * (W_s - np.max(W_s))
            p_wm_vec = np.exp(logits_wm)
            p_wm_vec = p_wm_vec / max(np.sum(p_wm_vec), eps)

            # Arbitration mixture
            p_total = wm_weight_block * p_wm_vec + (1.0 - wm_weight_block) * p_rl_vec
            p_total = p_total / max(np.sum(p_total), eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Simulate reward based on correct action for this state
            correct_a = int(correct_actions_by_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            pe = r - Q[s, a]
            Q[s, a] += lr * pe

            # WM decay toward uniform
            W = (1.0 - wm_decay) * W + wm_decay * U

            # Reward-triggered WM consolidation
            if r > 0.5:
                target = np.full(nA, 1e-6)
                target[a] = 1.0 - (nA - 1) * 1e-6
                W[s, :] = 0.4 * W[s, :] + 0.6 * target
            else:
                W[s, a] = 0.9 * W[s, a] + 0.1 * (1.0 / nA)

            tr += 1

    return simulated_actions, simulated_rewards
