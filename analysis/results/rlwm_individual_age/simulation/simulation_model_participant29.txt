def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL with eligibility traces + Surprise-gated WM, with load/age-adjusted gating threshold.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, lambda_trace, wm_weight_base, surprise_threshold_base, age_size_threshold_gain]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, softmax_beta, lambda_trace, wm_weight_base, surprise_threshold_base, age_size_threshold_gain = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)

        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action using first occurrence within the block
        # Assumes states are labeled 0..nS-1
        correct_actions_per_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                correct_actions_per_state[s] = block_correct[idx_s[0]]
            else:
                # If a state doesn't appear (edge case), default to 0
                correct_actions_per_state[s] = 0

        # Initialize RL and WM values and eligibility traces
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        E = np.zeros((nS, nA))

        # Threshold increases with load beyond 3 and with age group
        extra_items = max(0, nS - 3)
        thr = surprise_threshold_base * (1.0 + age_size_threshold_gain * (age_group + extra_items))

        for t in range(len(block_states)):
            s = block_states[t]

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            logits_rl = softmax_beta * Q_s
            logits_rl -= np.max(logits_rl)
            prl_all = np.exp(logits_rl)
            prl_all_sum = max(np.sum(prl_all), eps)
            prl_all /= prl_all_sum

            # WM policy
            logits_wm = softmax_beta_wm * W_s
            logits_wm -= np.max(logits_wm)
            pwm_all = np.exp(logits_wm)
            pwm_all_sum = max(np.sum(pwm_all), eps)
            pwm_all /= pwm_all_sum

            # For surprise computation we need q[s, a] pre-update; will compute after sampling a.
            # Arbitration requires wm_weight_eff depending on surprise, which depends on a and r.
            # We thus first construct mixture weights conditional on a by computing wm_weight after observing r.
            # To do this faithfully, we sample an action from the RL policy first for computing surprise?
            # In the original fitting, arbitration used surprise based on the chosen action and obtained reward.
            # We'll sample using a provisional equal mixture, then recompute mixture using the same a for sampling.
            # To avoid bias, we sample action using RL policy (baseline) to determine surprise gate, then resample with gated mixture.

            # Sample provisional action according to RL (could also use average mixture, but RL is the default system)
            a_prov = np.random.choice(nA, p=prl_all)

            # Determine reward for the provisional action
            correct_action = correct_actions_per_state[s]
            r_prov = 1 if a_prov == correct_action else 0

            # Compute surprise for the provisional choice
            delta_pre = r_prov - Q_s[a_prov]
            surprise = abs(delta_pre)

            # Compute effective WM weight based on surprise gate
            wm_weight_eff = wm_weight_base if surprise > thr else 0.0
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # Final action sampling using gated mixture policy at current state
            p_total = wm_weight_eff * pwm_all + (1.0 - wm_weight_eff) * prl_all
            p_total = np.maximum(p_total, eps)
            p_total /= np.sum(p_total)

            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Realized reward under task contingencies
            correct_action = correct_actions_per_state[s]
            r = 1 if a == correct_action else 0
            simulated_rewards[block_indices[t]] = r

            # Eligibility trace update
            E *= lambda_trace
            E[s, :] *= 0.0
            E[s, a] = 1.0

            # RL update with traces
            delta = r - q[s, a]
            q += lr * delta * E

            # WM update: decay toward uniform, then strengthen rewarded action for this state
            w[s, :] = 0.9 * w[s, :] + 0.1 * (1.0 / nA)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot

            tr += 1

    return simulated_actions, simulated_rewards
