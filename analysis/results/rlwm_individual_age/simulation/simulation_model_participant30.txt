def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Surprise-gated WM with shared decay and age-modulated temperature.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of ints: 0..nS-1 within block)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array, ints 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_base, wm_beta, wm_gate_surprise, decay, age_temp_shift]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, beta_base, wm_beta, wm_gate_surprise, decay, age_temp_shift = parameters
    softmax_beta = beta_base * 10.0
    softmax_beta_wm = wm_beta
    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]

        nA = 3
        nS = int(np.unique(set_sizes[block_idx])[0])  # block set size

        # Map each state to its correct action (assumes stationary within block)
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            state_trials = np.where(block_states == s)[0]
            state_correct[s] = int(block_correct[state_trials[0]])

        # Effective temperatures
        beta_eff = softmax_beta * np.exp(-age_temp_shift * age_group)
        beta_wm_eff = softmax_beta_wm

        # Initialize RL and WM values and priors
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        q_0 = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Maintain a per-state arbitration weight to use for action selection.
        # It will be updated after observing reward (initialized to 0.5).
        wm_weight_state = 0.5 * np.ones(nS)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute system policies
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Softmax for RL
            logits_rl = beta_eff * (Q_s - np.max(Q_s))
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # Softmax for WM
            logits_wm = beta_wm_eff * (W_s - np.max(W_s))
            p_wm = np.exp(logits_wm)
            p_wm /= np.sum(p_wm)

            # Arbitration weight for choice: use last available state-specific weight
            wm_weight_choice = np.clip(wm_weight_state[s], 0.0, 1.0)

            # Mixture policy and action sampling
            p_total = wm_weight_choice * p_wm + (1.0 - wm_weight_choice) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Simulate reward from correct answer
            r = 1 if a == state_correct[s] else 0
            simulated_rewards[block_idx[t]] = r

            # Compute gating signal based on observed outcome (as in fitting code)
            pe = abs(r - Q_s[a])  # prediction error magnitude based on pre-update Q
            size_scale = (nS - 3) / 3.0  # 0 for 3, 1 for 6
            gate_input = (r - 0.5) + pe - 0.5 * size_scale - 0.2 * age_group
            wm_weight = 1.0 / (1.0 + np.exp(-wm_gate_surprise * gate_input))
            wm_weight = float(np.clip(wm_weight, 0.0, 1.0))

            # Apply decay toward priors
            q = (1.0 - decay) * q + decay * q_0
            w = (1.0 - decay) * w + decay * w_0

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update (gated)
            w[s, a] += wm_weight * (r - W_s[a])

            # Store gating weight for this state for next choice
            wm_weight_state[s] = wm_weight

            tr += 1

    return simulated_actions, simulated_rewards
