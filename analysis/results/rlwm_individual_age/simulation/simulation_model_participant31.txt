import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL with PE-modulated exploration and PE-gated WM encoding (simulation).

    Arguments:
    - stimulus: sequence of stimuli for each trial (array; states 0..nS-1)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array; actions 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_base, beta_pe_gain, wm_weight_base, wm_pe_gate, wm_decay_base]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """

    lr, beta_base, beta_pe_gain, wm_weight_base, wm_pe_gate, wm_decay_base = parameters

    age_group = 0 if age[0] <= 45 else 1

    softmax_beta_wm = 50.0  # very deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_actions = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            idxs = np.where(block_states == st)[0]
            if len(idxs) == 0:
                # Fallback in unlikely case of missing state
                correct_actions[st] = 0
            else:
                correct_actions[st] = int(block_correct_actions[idxs[0]])

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective arbitration and WM parameters (set-size and age adjusted)
        wm_weight_eff = wm_weight_base * (3.0 / nS) * (1.0 - 0.25 * age_group)
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        wm_decay_eff = np.clip(wm_decay_base * (nS / 3.0) * (1.0 + 0.5 * age_group), 0.0, 1.0)

        wm_pe_gate_eff = wm_pe_gate * (nS / 3.0) * (1.0 + 0.5 * age_group)

        # Track last absolute PE per state to modulate next-trial beta (causal approximation)
        last_abs_pe = np.zeros(nS)

        # Iterate through trials in the block
        for t_local, tr in enumerate(block_indices):
            s = int(block_states[t_local])

            Q_s = q[s, :]
            W_s = w[s, :]

            # PE-modulated exploration: use last absolute PE for this state (causal for simulation)
            beta_dyn = beta_base / (1.0 + beta_pe_gain * last_abs_pe[s])
            beta_dyn = max(beta_dyn, 1e-3)
            softmax_beta = beta_dyn * 10.0

            # RL policy
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)  # numerical stability
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Arbitration
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # Compute PE and update RL
            pe = r - q[s, a]
            abs_pe = abs(pe)
            q[s, a] += lr * pe

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # WM encoding gated by surprise and reward
            if abs_pe >= wm_pe_gate_eff and r > 0.0:
                w[s, a] += (1.0 - w[s, a]) * 0.9
                w[s, :] /= np.sum(w[s, :])

            # Update last absolute PE for this state
            last_abs_pe[s] = abs_pe

    return simulated_actions, simulated_rewards
