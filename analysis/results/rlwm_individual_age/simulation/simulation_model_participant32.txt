def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from the RL+WM model with valence-specific learning,
    set-sizeâ€“dependent decision noise, and reward-gated WM contribution.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array of int)
    - blocks: block index for each trial (array of int)
    - set_sizes: block-dependent set sizes on each trial (array of int)
    - correct_answer: correct action on each trial (array of int)
    - age: participant's age (array, single repeated value)
    - parameters: [alpha_pos, alpha_neg, wm_weight, softmax_beta, beta_set_cost, gate]

    Returns:
    - simulated_actions: simulated action choices (array of int)
    - simulated_rewards: simulated binary rewards (array of int)
    """
    import numpy as np

    alpha_pos, alpha_neg, wm_weight, softmax_beta, beta_set_cost, gate = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    age_group = 0 if age[0] <= 45 else 1
    if age_group == 1:
        alpha_pos *= 0.8
        alpha_neg *= 0.8
        softmax_beta *= 0.8

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        nA = 3
        # Use provided set size (consistent within block)
        nS = int(set_sizes[block_idx][0])

        # Map each state to its correct action (assumes stationary mapping within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # first occurrence of this state in the block gives its correct action
            s_mask = (block_states == s)
            if np.any(s_mask):
                correct_actions[s] = int(block_correct[s_mask][0])
            else:
                # If a state index doesn't appear (edge case), default to 0
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last reward outcome per state to drive WM gating (initialize as non-reward)
        last_reward_state = np.zeros(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Effective RL temperature adjusted by set-size cost
            beta_eff = softmax_beta / (1.0 + beta_set_cost * max(0, nS - 3))

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax for RL and WM policies
            # RL
            x_rl = beta_eff * (Q_s - np.max(Q_s))
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)
            # WM
            x_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(x_wm)
            p_wm /= np.sum(p_wm)

            # WM weight: reduced with larger set size; gated by last outcome for this state
            base_wm_w = wm_weight / (1.0 + max(0, nS - 3))
            gate_t = gate if last_reward_state[s] > 0 else (1.0 - gate)
            wm_weight_eff = np.clip(base_wm_w * gate_t, 0.0, 1.0)

            # Mixture policy and action sampling
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome from correct action mapping
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            alpha = alpha_pos if delta > 0 else alpha_neg
            q[s, a] += alpha * delta

            # WM update: load-dependent forgetting + reward-triggered consolidation
            forget = 0.1 + 0.2 * max(0, nS - 3) / 3.0  # 0.1 at set-size 3; 0.3 at 6
            w[s, :] = (1.0 - forget) * w[s, :] + forget * w_0[s, :]
            if r > 0:
                consolidate = 0.7
                w[s, :] = (1.0 - consolidate) * w[s, :]
                w[s, a] += consolidate
                # Normalize WM row
                w_row_sum = np.sum(w[s, :])
                if w_row_sum > 0:
                    w[s, :] /= w_row_sum

            # Update last reward state flag for gating on next encounter
            last_reward_state[s] = r

            tr += 1

    return simulated_actions, simulated_rewards
