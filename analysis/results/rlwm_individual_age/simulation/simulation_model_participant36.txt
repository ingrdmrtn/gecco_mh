def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + retrieval-based WM with capacity and global stickiness.

    Arguments:
    - stimulus: sequence of state/stimulus IDs for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action for each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_base, wm_retr_base, C, stickiness, alpha_decay]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, beta_base, wm_retr_base, C, stickiness, alpha_decay = parameters
    softmax_beta = beta_base * 10.0
    softmax_beta_wm = 50.0

    age_group = 1 if age[0] > 45 else 0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    tr = 0

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        # Set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Build state-wise correct action mapping (assumes states are 0..nS-1)
        correct_actions = np.zeros(nS, dtype=int)
        for s_id in range(nS):
            idx = np.where(block_states == s_id)[0]
            if len(idx) > 0:
                correct_actions[s_id] = int(block_correct[idx[0]])
            else:
                # Fallback (should not happen if each state appears): default to 0
                correct_actions[s_id] = 0

        # Effective RL temperature with age and load scaling
        softmax_beta_eff = softmax_beta * ((1.0 - 0.3 * age_group) / (1.0 + 0.4 * (nS - 3.0)))

        # WM retrieval probability for the block
        base_ret = 1.0 / (1.0 + np.exp(-wm_retr_base))
        load_factor = max(0.0, 1.0 - max(0.0, nS - C) / max(1.0, C))
        p_retrieve = np.clip(base_ret * load_factor * (1.0 - 0.25 * age_group), 0.0, 1.0)

        # WM decay rate for the block
        decay = np.clip(alpha_decay * (1.0 + 0.5 * age_group + 0.5 * max(0.0, nS - C) / max(1.0, C)), 0.0, 1.0)

        last_global_action = -1

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy with global stickiness
            logits_rl = softmax_beta_eff * Q_s
            if last_global_action >= 0:
                logits_rl[last_global_action] += stickiness
            logits_rl -= np.max(logits_rl)
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (near-deterministic choice)
            logits_wm = softmax_beta_wm * W_s
            logits_wm -= np.max(logits_wm)
            p_wm = np.exp(logits_wm)
            p_wm /= np.sum(p_wm)

            # Mixture of controllers
            p_total = p_retrieve * p_wm + (1.0 - p_retrieve) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Generate reward
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM update
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot
            else:
                w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            last_global_action = a
            tr += 1

    return simulated_actions, simulated_rewards
