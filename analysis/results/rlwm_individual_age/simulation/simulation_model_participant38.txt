def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL with asymmetric learning and Pearce-Hall associability; WM gating by recent PE vs threshold.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of ints: state indices within block)
    - blocks: block index for each trial (array)
    - set_sizes: block dependent set sizes on each trial (array) [not used directly; inferred per block]
    - correct_answer: correct action on each trial (array of ints 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: [lr_pos, lr_neg, beta, gate_sensitivity, age_gate_shift, attn_decay]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr_pos, lr_neg, beta, gate_sensitivity, age_gate_shift, attn_decay = parameters
    beta_eff = beta * 10.0
    beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct_actions = correct_answer[block_indices].astype(int)

        nA = 3
        nS = int(len(np.unique(block_states)))

        # Map each state to its correct action within the block
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            # assume each state appears at least once in the block
            idx = np.where(block_states == st)[0][0]
            correct_actions[st] = block_correct_actions[idx]

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w0 = (1.0 / nA) * np.ones((nS, nA))

        # Pearce-Hall associability per state
        alpha = 0.5 * np.ones(nS)

        # Gating threshold depends on set size and age
        base_threshold = 0.3 + age_gate_shift * (nS / 3.0) * (1.0 + 0.5 * age_group)

        # Use most recent PE per state to gate (approximation of fitting logic)
        last_pe = np.zeros(nS)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax choice probabilities for RL and WM
            # RL
            q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(beta_eff * q_shift)
            p_rl_sum = np.sum(p_rl)
            if p_rl_sum <= 0:
                p_rl = np.ones(nA) / nA
            else:
                p_rl /= p_rl_sum

            # WM
            w_shift = W_s - np.max(W_s)
            p_wm = np.exp(beta_wm * w_shift)
            p_wm_sum = np.sum(p_wm)
            if p_wm_sum <= 0:
                p_wm = np.ones(nA) / nA
            else:
                p_wm /= p_wm_sum

            # Gating by recent PE vs threshold (logistic)
            wm_weight = 1.0 / (1.0 + np.exp(-gate_sensitivity * (last_pe[s] - base_threshold)))
            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Outcome (deterministic correct-answer reward)
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_indices[t]] = r

            # Prediction error
            pe = r - Q_s[a]

            # RL update with asymmetric learning rates scaled by associability
            lr_eff = (lr_pos if pe >= 0.0 else lr_neg) * alpha[s]
            q[s, a] += lr_eff * pe

            # Update associability (Pearce-Hall)
            alpha[s] = (1.0 - attn_decay) * alpha[s] + attn_decay * np.abs(pe)
            alpha[s] = np.clip(alpha[s], 0.0, 1.0)

            # WM update: reinforced one-hot vs decay toward baseline
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot
            else:
                w[s, :] = 0.9 * w[s, :] + 0.1 * w0[s, :]

            # Update recent PE for gating on subsequent encounters with this state
            last_pe[s] = pe

            tr += 1

    return simulated_actions, simulated_rewards
