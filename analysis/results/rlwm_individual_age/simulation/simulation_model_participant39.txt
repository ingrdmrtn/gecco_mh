def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL with choice stickiness and lapse + WM as episodic recall (age- and load-modulated).

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block dependent set sizes on each trial (array) [not directly used; block size inferred per block]
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, wm_weight, softmax_beta, stickiness, lapse_base, recall_base]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr, wm_weight, softmax_beta, stickiness, lapse_base, recall_base = parameters
    softmax_beta *= 10.0  # per template
    softmax_beta_wm = 50.0  # unused (episodic WM is rule-based)

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]

        nA = 3
        # Determine set size from the block (as in fitting: constant within block)
        nS = int(len(np.unique(block_states)))

        # Map each state to its correct action (assumed stable within block)
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            first_occ = np.where(block_states == st)[0][0]
            correct_actions[st] = int(block_correct[first_occ])

        # Initialize RL values, WM placeholders, last choice, and episodic memory
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))  # unused for choice; kept for template consistency
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_choice = -1 * np.ones(nS, dtype=int)
        episodic = -1 * np.ones(nS, dtype=int)  # last rewarded action per state

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with stickiness
            Q_s = q[s, :].copy()
            if last_choice[s] >= 0:
                Q_s[last_choice[s]] += stickiness
            # Stable softmax
            Q_centered = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_centered)
            p_rl = p_rl / np.sum(p_rl)

            # WM episodic recall probability (age- and load-modulated)
            load_factor = 3.0 / max(float(nS), 1.0)  # 1 at nS=3, 0.5 at nS=6
            recall_prob = recall_base * load_factor * (1.0 - 0.4 * age_group)
            recall_prob = np.clip(recall_prob, 0.0, 1.0)

            # Construct WM policy
            if episodic[s] >= 0:
                p_wm = np.ones(nA) * ((1.0 - recall_prob) / nA)
                p_wm[episodic[s]] += recall_prob
            else:
                p_wm = np.ones(nA) / nA

            # Mix RL and WM
            p_mixed = wm_weight * p_wm + (1.0 - wm_weight) * p_rl

            # Lapse to uniform
            lapse = lapse_base * (nS / 6.0) * (1.0 + 0.5 * age_group)
            lapse = np.clip(lapse, 0.0, 0.5)
            p_total = (1.0 - lapse) * p_mixed + lapse * (np.ones(nA) / nA)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward delivery
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # Episodic memory update
            if r == 1:
                episodic[s] = a

            # Update last choice
            last_choice[s] = a

            # WM value decay towards baseline (for completeness; not used in policy)
            w[s, :] = 0.95 * w[s, :] + 0.05 * w_0[s, :]
            w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
