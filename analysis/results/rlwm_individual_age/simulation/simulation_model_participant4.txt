def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using:
    - RL with asymmetric learning rates
    - Choice perseveration kernel
    - WM with set-size driven interference
    - Mixture of WM and RL policies with WM weight reduced by set size and age

    Arguments:
    - stimulus: sequence of stimuli/states for each trial (array of ints)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array; assumed constant within a block)
    - correct_answer: correct action on each trial (array of ints in [0..nA-1])
    - age: participant's age (array, single repeated value)
    - parameters: [lr_pos, lr_neg, wm_weight, softmax_beta, persev_weight, setsize_sensitivity]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints: 0/1)
    """
    import numpy as np

    lr_pos, lr_neg, wm_weight, softmax_beta, persev_weight, setsize_sensitivity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # deterministic WM policy

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Loop over blocks
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]
        nA = 3
        nS = int(block_set_sizes[0])  # assumed constant within block

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Perseveration kernel
        C = np.zeros(nA)
        kernel_decay = 0.9
        last_action = None

        # Set-size- and age-dependent parameters
        sens_eff = setsize_sensitivity * (1.0 + age_group)
        wm_weight_eff = wm_weight / (1.0 + sens_eff * max(0, nS - 1))
        base_wm_decay = 0.05
        wm_decay = base_wm_decay + sens_eff * 0.10

        # Iterate over trials in this block
        for idx_local, tr in enumerate(block_indices):
            s = int(block_states[idx_local])

            # Decay perseveration kernel and add previous action bias
            C = kernel_decay * C
            if last_action is not None:
                C[last_action] += 1.0

            # RL policy
            Q_s = q[s, :]
            prefs = softmax_beta * Q_s + persev_weight * C
            prefs -= np.max(prefs)
            exp_prefs = np.exp(prefs)
            p_rl = exp_prefs / np.sum(exp_prefs)

            # WM policy
            W_s = w[s, :]
            wm_prefs = softmax_beta_wm * W_s
            wm_prefs -= np.max(wm_prefs)
            exp_wm = np.exp(wm_prefs)
            p_wm = exp_wm / np.sum(exp_wm)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Determine reward from correct answer
            correct_a = int(block_correct[idx_local])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update (asymmetric learning rates)
            pe = r - Q_s[a]
            lr = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += lr * pe

            # WM update: decay then overwrite with one-hot on rewarded trials
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

            # Clip and renormalize the WM row for current state
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

            # Update perseveration memory
            last_action = a

    return simulated_actions, simulated_rewards
