def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL with age/load-reduced temperature + WM with precision gating via a dynamic gain and decay.

    Arguments:
    - stimulus: sequence of stimuli (states) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set size for each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, wm_weight, softmax_beta, lambda_trace, wm_gain, temp_penalty]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr, wm_weight, softmax_beta, lambda_trace, wm_gain, temp_penalty = parameters
    softmax_beta *= 10.0

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate by blocks
    unique_blocks = np.unique(blocks)
    for b in unique_blocks:
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use block-provided set size (as in fitting code)
        nS = int(block_set_sizes[0])

        # Map each state id to its correct action (take first occurrence)
        # States are assumed to be 0..nS-1 within block
        state_to_correct = {}
        for idx, s in enumerate(block_states):
            if s not in state_to_correct:
                state_to_correct[int(s)] = int(block_correct[idx])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Initialize WM precision per state
        kappa = np.ones(nS)

        # Compute load and temperature scaling constants for this block
        load = age_group + max(nS - 3, 0) / 3.0
        beta_eff = softmax_beta * np.exp(-temp_penalty * load)

        kappa_decay = float(np.clip(lambda_trace * (1.0 + load), 0.0, 1.0))
        eta_wm = 0.5 * float(np.clip(lambda_trace, 0.0, 1.0))

        # Trial loop within block (respect original trial order)
        for local_t, tr in enumerate(block_indices):
            s = int(block_states[local_t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            logits_rl = beta_eff * Q_s
            max_rl = np.max(logits_rl)
            prl = np.exp(logits_rl - max_rl)
            prl /= np.sum(prl)

            # WM policy with precision-gated temperature
            beta_wm_eff = softmax_beta_wm * max(kappa[s], 1e-6)
            logits_wm = beta_wm_eff * W_s
            max_wm = np.max(logits_wm)
            pwm = np.exp(logits_wm - max_wm)
            pwm /= np.sum(pwm)

            # Arbitration
            p_total = wm_weight * pwm + (1.0 - wm_weight) * prl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correctness
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM precision update (kappa)
            kappa[s] = (1.0 - kappa_decay) * kappa[s] + wm_gain * max(r, 0.0)
            kappa[s] = float(np.clip(kappa[s], 0.0, 100.0))

            # WM table update
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * onehot
            else:
                # Move slightly away from chosen action starting from uniform
                uni = w_0[s, :].copy()
                anti = uni.copy()
                anti[a] = max(0.0, anti[a] - 1.0 / (nA * 2.0))
                anti = anti / anti.sum()
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * anti

            # Normalize WM row
            row_sum = w[s, :].sum()
            if row_sum > 0:
                w[s, :] = w[s, :] / row_sum

    return simulated_actions, simulated_rewards
