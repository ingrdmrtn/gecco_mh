def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL with asymmetric learning rates and PE-gated WM mixture,
    with age/load reducing WM gating.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of ints; state indices within block)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array; unused for simulation other than sanity)
    - correct_answer: correct action on each trial (array of ints in {0,1,2})
    - age: participant's age (array, single repeated value)
    - parameters: [lr_pos, lr_neg, wm_base, gate_sensitivity, age_size_tradeoff, softmax_beta]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr_pos, lr_neg, wm_base, gate_sens, age_size_tradeoff, softmax_beta = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]

        nA = 3
        # Determine number of states in this block and map each state to its correct action
        unique_states = np.unique(block_states)
        nS = len(unique_states)

        # Build mapping from state id to index 0..nS-1 if states are not contiguous
        # Here we assume states are 0..nS-1 within block as in the fitting code
        # but to be safe we construct a direct mapping
        state_map = {s: i for i, s in enumerate(unique_states)}
        # Determine correct action per mapped state
        correct_actions = np.zeros(nS, dtype=int)
        for s_raw in unique_states:
            s_idx = state_map[s_raw]
            # pick the first occurrence's correct action for that state in this block
            first_idx = np.where(block_states == s_raw)[0][0]
            correct_actions[s_idx] = int(block_correct[first_idx])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last PE per state for WM gating (initialized to 0 => baseline wm_base)
        pe_prev = np.zeros(nS)

        for t in range(len(block_states)):
            s_raw = block_states[t]
            s = state_map[s_raw]

            Q_s = q[s, :]
            W_s = w[s, :]

            # Compute WM gating weight using last PE experienced in this state
            wm_logit = wm_base + gate_sens * abs(pe_prev[s]) - age_size_tradeoff * (age_group + (nS - 3))
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

            # Policies
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl /= np.sum(prl)

            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm /= np.sum(pwm)

            p_total = wm_weight * pwm + (1.0 - wm_weight) * prl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_indices[t]] = r

            # Compute PE for gating update (using pre-update Q)
            pe = r - Q_s[a]
            pe_prev[s] = pe

            # RL update (asymmetric)
            if pe >= 0:
                q[s, a] = Q_s[a] + lr_pos * pe
            else:
                q[s, a] = Q_s[a] + lr_neg * pe

            # WM decay toward baseline, then reinforcement on rewarded trials
            decay = 0.8
            w[s, :] = decay * w[s, :] + (1.0 - decay) * w_0[s, :]
            if r > 0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
