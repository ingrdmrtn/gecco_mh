def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate actions and rewards using the RL(Î±) with eligibility traces, age-modulated decay,
    stochastic WM gating under load, and action stickiness.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of ints: state indices 0..nS-1)
    - blocks: block index for each trial (array)
    - set_sizes: set size for each trial (array, constant within a block: 3 or 6)
    - correct_answer: correct action for each trial (array of ints 0..2; constant per state within block)
    - age: participant's age (array, single repeated value)
    - parameters: [alpha, softmax_beta, lambda_trace, beta_wm_gate, sigma_gate_age, kappa_stick]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    alpha, softmax_beta, lambda_trace, beta_wm_gate, sigma_gate_age, kappa_stick = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block (assumes stationary mapping per state)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first occurrence of this state within the block
            first_idx = np.where(block_states == s)[0][0]
            correct_actions[s] = block_correct[first_idx]

        # Initialize RL, eligibility traces, and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        elig = np.zeros((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age-modulated eligibility trace decay
        lam_eff = np.clip(lambda_trace * (0.8 ** age_group), 0.0, 1.0)

        last_action = None

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Stickiness bias
            bias = np.zeros(nA)
            if last_action is not None:
                bias[last_action] += kappa_stick

            # RL policy
            Qb = Q_s + bias
            rl_logits = softmax_beta * (Qb - np.max(Qb))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy
            Wb = W_s + bias
            wm_logits = softmax_beta_wm * (Wb - np.max(Wb))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Stochastic WM gate: depends on load (nS-3) and age noise
            load = float(nS - 3)  # 0 for set=3, 3 for set=6
            noise_unit = ((t + 1) % 5) / 4.0 - 0.5  # in [-0.5, 0.5]
            noise = sigma_gate_age * age_group * noise_unit
            gate_bias = -(beta_wm_gate * load) + noise
            wm_weight = 1.0 / (1.0 + np.exp(-gate_bias))
            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Generate reward from correct action mapping
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update with eligibility traces
            pe = r - Q_s[a]
            elig *= lam_eff
            elig[s, a] += 1.0
            q += alpha * pe * elig

            # WM decay and overwrite on reward
            w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            last_action = a
            tr += 1

    return simulated_actions, simulated_rewards
