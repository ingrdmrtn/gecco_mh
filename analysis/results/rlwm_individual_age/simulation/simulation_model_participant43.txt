def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the RL + eligibility traces mixed with capacity-limited WM model.

    Arguments:
    - stimulus: sequence of stimuli (state ids) for each trial (array-like, zero-based within block if possible)
    - blocks: block index for each trial (array-like)
    - set_sizes: block-dependent set sizes on each trial (array-like)
    - correct_answer: correct action on each trial (array-like)
    - age: participant's age (array-like, single repeated value)
    - parameters: [lr, wm_weight, softmax_beta, trace_lambda, wm_decay, age_wm_noise]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, trace_lambda, wm_decay, age_wm_noise = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    eps = 1e-12

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states_raw = np.array(stimulus)[block_indices]
        block_correct_ans = np.array(correct_answer)[block_indices]
        block_set_sizes = np.array(set_sizes)[block_indices]

        # Determine action/state space sizes
        nA = 3
        # Use given set size for the block if provided; fallback to observed unique states
        nS_block = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else len(np.unique(block_states_raw))

        # Map possibly arbitrary state labels to 0..nS-1
        unique_states = np.unique(block_states_raw)
        state_to_idx = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_idx[s] for s in block_states_raw])

        # If provided set size exceeds observed states (e.g., some states not shown), cap by observed
        nS = max(len(unique_states), nS_block)

        # Build per-state correct action lookup (for observed states)
        correct_actions = np.zeros(len(unique_states), dtype=int)
        for i, s in enumerate(unique_states):
            mask = (block_states_raw == s)
            # take the first occurrence's correct answer
            correct_actions[i] = int(block_correct_ans[mask][0])

        # Initialize RL, WM, and eligibility trace
        q = (1.0 / nA) * np.ones((len(unique_states), nA))
        w = (1.0 / nA) * np.ones((len(unique_states), nA))
        w_0 = (1.0 / nA) * np.ones((len(unique_states), nA))
        e = np.zeros((len(unique_states), nA))

        # Block-level WM mixture scaling by set size (3 vs 6)
        wm_w_block = np.clip(wm_weight * (1.0 if nS == 3 else 0.5), 0.0, 1.0)

        # WM precision scaling by set size and age
        wm_precision_scale = 1.0 / (1.0 + 0.5 * (nS - 3))  # 1.0 for 3, ~0.67 for 6
        wm_precision_age = 1.0 / (1.0 + age_wm_noise * age_group)
        beta_wm_eff = softmax_beta_wm * wm_precision_scale * wm_precision_age

        # Iterate through trials in this block
        for t in range(len(block_states)):
            s = int(block_states[t])
            correct_a = int(correct_actions[s])

            # Current policy from RL
            Q_s = q[s, :].copy()
            Q_center = Q_s - np.max(Q_s)
            p_vec_rl = np.exp(softmax_beta * Q_center)
            p_vec_rl /= np.sum(p_vec_rl)

            # Current policy from WM
            W_s = w[s, :].copy()
            W_center = W_s - np.max(W_s)
            p_vec_wm = np.exp(beta_wm_eff * W_center)
            p_vec_wm /= np.sum(p_vec_wm)

            # Mixture policy
            p_total = wm_w_block * p_vec_wm + (1.0 - wm_w_block) * p_vec_rl
            p_total = np.clip(p_total, eps, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Simulate reward
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL TD error and eligibility trace update
            pe = float(r) - q[s, a]
            e *= trace_lambda
            e[s, a] = 1.0
            q += lr * pe * e

            # WM decay toward uniform; then one-shot encode on reward
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0:
                row = np.full(nA, 1e-6)
                row[a] = 1.0
                row = row / np.sum(row)
                w[s, :] = row

            tr += 1

    return simulated_actions, simulated_rewards
