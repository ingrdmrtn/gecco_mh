def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL + asymmetric learning + recency-based WM model.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array-like of ints)
    - blocks: block index for each trial (array-like of ints)
    - set_sizes: set size for each trial (array-like of ints)
    - correct_answer: correct action on each trial (array-like of ints)
    - age: participant's age (array-like, single repeated value)
    - parameters: [alpha_pos, alpha_neg, softmax_beta, decay_base, size_decay_gain, age_decay_gain]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    alpha_pos, alpha_neg, softmax_beta, decay_base, size_decay_gain, age_decay_gain = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]

        nA = 3
        # Determine number of states in this block. Assume states are 0..nS-1.
        nS = int(np.max(block_states) + 1)

        # For each state, get the correct action (from the first occurrence of that state in the block)
        correct_actions = np.zeros(nS, dtype=int)
        for st in range(nS):
            st_idx = np.where(block_states == st)[0]
            if len(st_idx) == 0:
                correct_actions[st] = 0
            else:
                correct_actions[st] = int(block_correct[st_idx[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Recency tracking for WM
        last_enc_time = -1e9 * np.ones(nS)
        time_counter = 0

        # Compute decay for this block given load and age
        load_factor = 1.0 + size_decay_gain * max(0, nS - 3)
        age_factor = 1.0 + age_decay_gain * age_group
        decay = decay_base * load_factor * age_factor

        for t in range(len(block_states)):
            s = block_states[t]

            # Get RL and WM policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM softmax (high precision)
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Compute effective WM weight from recency decay
            if last_enc_time[s] < -1e8:
                wm_weight_eff = 0.0
            else:
                elapsed = max(0.0, time_counter - last_enc_time[s])
                wm_weight_eff = np.exp(-decay * elapsed)
                wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward: 1 if matches correct action for the current state
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_indices[t]] = r

            # RL update with asymmetric learning rates
            pe = r - q[s, a]
            q[s, a] += (alpha_pos if pe >= 0 else alpha_neg) * pe

            # WM update on rewarded trials: store the winning action deterministically
            if r > 0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0
                w[s, :] /= np.sum(w[s, :])
                last_enc_time[s] = time_counter

            # Advance time
            time_counter += 1
            tr += 1

    return simulated_actions, simulated_rewards
