def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from the RL + capacity-limited WM model with age and set-size dependent arbitration and WM forgetting.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of ints; state IDs 0..nS-1 within block)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set size per trial (array; values like 3 or 6)
    - correct_answer: correct action on each trial (array of ints 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_base, wm_weight_base, wm_forget, ss6_penalty, age_wm_delta]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints 0/1)
    """
    import numpy as np

    lr, beta_base, wm_weight_base, wm_forget, ss6_penalty, age_wm_delta = parameters

    softmax_beta = beta_base * 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    age_group = 1 if age[0] > 45 else 0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        nA = 3
        # Set size for this block (number of unique states equals set size)
        nS = int(set_sizes[block_indices][0])

        # Derive per-state correct action mapping from the block data
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # pick the first occurrence's correct action for this state
            first_idx = np.where(block_states == s)[0][0]
            correct_actions[s] = int(block_correct[first_idx])

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Arbitration weight for this block
        base = np.clip(wm_weight_base, 0.0, 1.0)
        size_penalty = ss6_penalty if nS == 6 else 0.0
        age_penalty = age_group * age_wm_delta
        wm_weight_block = np.clip(base - size_penalty - age_penalty, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            Qs = q[s, :].copy()
            Qs -= np.max(Qs)
            exp_rl = np.exp(softmax_beta * Qs)
            p_rl_vec = exp_rl / np.sum(exp_rl)

            # WM policy
            Ws = w[s, :].copy()
            Ws -= np.max(Ws)
            exp_wm = np.exp(softmax_beta_wm * Ws)
            p_wm_vec = exp_wm / np.sum(exp_wm)

            # Arbitration mixture
            p_total_vec = wm_weight_block * p_wm_vec + (1.0 - wm_weight_block) * p_rl_vec
            p_total_vec = p_total_vec / np.sum(p_total_vec)

            # Sample action
            a = int(np.random.choice(nA, p=p_total_vec))
            simulated_actions[block_indices[t]] = a

            # Reward from task rule
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[block_indices[t]] = r

            # WM update
            if r > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]
            # Normalize WM row to avoid numerical drift
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = (1.0 / nA)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            tr += 1

    return simulated_actions, simulated_rewards
