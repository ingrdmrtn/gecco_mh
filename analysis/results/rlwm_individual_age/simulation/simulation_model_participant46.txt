def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for:
    Asymmetric RL + outcome-gated WM + state-specific stickiness.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr_pos, lr_neg, softmax_beta, wm_weight_base, wm_reward_boost, stickiness]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr_pos, lr_neg, softmax_beta, wm_weight_base, wm_reward_boost, stickiness = parameters
    softmax_beta *= 10.0
    wm_temp = 50.0  # deterministic WM policy

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct_actions_seq = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        nS = int(block_set_sizes[0])

        # Determine the correct action per state within block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = int(block_correct_actions_seq[idx[0]])
            else:
                # Fallback (should not happen if all states appear)
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_action = -1 * np.ones(nS, dtype=int)

        # Precompute age/load penalty for WM
        denom_penalty = 1.0 + max(0.0, (nS - 3.0)) + float(age_group)
        base_wm_weight = wm_weight_base / denom_penalty

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with stickiness
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stickiness
            Q_s_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_s_shift)
            p_rl /= np.sum(p_rl)

            # WM policy
            W_s = w[s, :].copy()
            W_s_shift = W_s - np.max(W_s)
            p_wm = np.exp(wm_temp * W_s_shift)
            p_wm /= max(np.sum(p_wm), 1e-12)

            # Outcome-gated WM weight (depends on whether action would be rewarded)
            correct_a = correct_actions[s]
            wm_weight_vec = base_wm_weight * np.ones(nA)
            wm_weight_vec[correct_a] = min(1.0, wm_weight_vec[correct_a] * (1.0 + wm_reward_boost))

            # Mixture policy with action-dependent WM weight
            p_total = wm_weight_vec * p_wm + (1.0 - wm_weight_vec) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Simulate reward
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update (asymmetric)
            pe = r - q[s, a]
            lr_eff = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += lr_eff * pe

            # WM decay
            w = 0.9 * w + 0.1 * w_0

            # WM update on reward: store one-hot mapping for rewarded pair
            if r > 0:
                w_row = np.zeros(nA)
                w_row[a] = 1.0
                w[s, :] = w_row

            # Update stickiness memory
            last_action[s] = a

            tr += 1

    return simulated_actions, simulated_rewards
