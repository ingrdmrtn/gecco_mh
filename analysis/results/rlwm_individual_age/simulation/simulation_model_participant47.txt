def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate actions and rewards using:
    RL with asymmetric learning rates + one-shot WM with forgetting.
    - WM stores the last rewarded action per state via replacement and decays toward uniform.
    - WM weight declines with larger set sizes; older adults forget faster.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array-like, int)
    - blocks: block index for each trial (array-like, int)
    - set_sizes: block-dependent set sizes on each trial (array-like, int)
    - correct_answer: correct action on each trial (array-like, int in [0..2])
    - age: participant's age (array-like, single repeated value)
    - parameters: [alpha_pos, alpha_neg, softmax_beta, wm_weight_base, phi_forget]

    Returns:
    - simulated_actions: simulated action choices (array of int)
    - simulated_rewards: simulated binary rewards (array of int)
    """
    import numpy as np

    alpha_pos, alpha_neg, softmax_beta, wm_weight_base, phi_forget = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 1 if age[0] > 45 else 0
    age_forget_factor = 1.3 if age_group == 1 else 1.0
    phi_forget_age = np.clip(phi_forget * age_forget_factor, 0.0, 1.0)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus)[block_idx].astype(int)
        block_correct = np.asarray(correct_answer)[block_idx].astype(int)
        block_set_sizes = np.asarray(set_sizes)[block_idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action using the first occurrence in the block
        # Assumes states are indexed 0..nS-1, consistent with fitting code.
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            first_occ = np.where(block_states == s)[0]
            if len(first_occ) > 0:
                correct_actions[s] = block_correct[first_occ[0]]
            else:
                # Fallback (should not occur if state appears in block): set to 0
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM mixture weight scales down with larger set sizes
        wm_size_scale = 3.0 / float(nS)  # 1 for set size 3, 0.5 for set size 6
        wm_weight_block = np.clip(wm_weight_base * wm_size_scale, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies (stable)
            q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * q_shift)
            p_rl /= np.sum(p_rl)

            w_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * w_shift)
            p_wm /= np.sum(p_wm)

            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Generate reward from task rule
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            alpha = alpha_pos if delta > 0 else alpha_neg
            q[s, a] += alpha * delta

            # WM forgetting toward uniform for current state
            w[s, :] = (1.0 - phi_forget_age) * w[s, :] + phi_forget_age * w_0[s, :]

            # One-shot WM overwrite on reward
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
