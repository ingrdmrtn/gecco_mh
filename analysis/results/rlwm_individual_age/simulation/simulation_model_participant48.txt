def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with perseveration bias; age increases perseveration and reduces WM weight under load.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array, ints 0..nS-1 within block)
    - blocks: block index for each trial (array)
    - set_sizes: set size on each trial (array; should be constant within block, values like 3 or 6)
    - correct_answer: correct action on each trial (array, ints 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, wm_weight_base, wm_decay, perseveration_beta, age_effect]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, softmax_beta, wm_weight_base, wm_decay, perseveration_beta, age_effect = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    if age[0] <= 45:
        age_group = 0
    else:
        age_group = 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its (first-seen) correct action within the block
        correct_actions = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for i in range(len(block_states)):
            s = block_states[i]
            if 0 <= s < nS and not seen[s]:
                correct_actions[s] = block_correct[i]
                seen[s] = True

        # Initialize RL, WM, and perseveration
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        pers = np.zeros((nS, nA))

        # Compute block-wise WM weight and effective perseveration strength
        load_scale = 3.0 / nS
        wm_weight = wm_weight_base * load_scale - age_effect * age_group
        wm_weight = np.clip(wm_weight, 0.0, 1.0)
        pers_beta_eff = perseveration_beta * (1.0 + age_effect * age_group)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with perseveration bias
            Q_s = q[s, :].copy()
            Q_s_bias = Q_s + pers_beta_eff * pers[s, :]
            z_rl = softmax_beta * Q_s_bias
            z_rl -= np.max(z_rl)
            p_rl = np.exp(z_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (near-deterministic)
            W_s = w[s, :].copy()
            z_wm = softmax_beta_wm * W_s
            z_wm -= np.max(z_wm)
            p_wm = np.exp(z_wm)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome based on correct action mapping
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM decay toward uniform, then store if rewarded
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r == 1:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Update perseveration to favor repeating chosen action in this state
            pers[s, :] = 0.0
            pers[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
