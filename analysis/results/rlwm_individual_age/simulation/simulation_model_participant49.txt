def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Surprise-gated WM writing + asymmetric RL learning, with load/age penalties on gating.

    Arguments:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr_pos, lr_neg, beta_rl, gate_base, age_gate_drop, beta_wm]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr_pos, lr_neg, beta_rl, gate_base, age_gate_drop, beta_wm = parameters

    softmax_beta = beta_rl * 10.0
    softmax_beta_wm = max(1e-3, beta_wm)

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block
        unique_states = np.unique(block_states)
        state_to_correct = {}
        for st in unique_states:
            # pick the first occurrence's correct action for this state
            state_to_correct[int(st)] = int(block_correct[np.where(block_states == st)[0][0]])

        # Initialize RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])
            nS_t = int(block_set_sizes[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Gating for action selection (pre-outcome; no surprise term yet)
            load_pen = age_gate_drop * (age_group * 1.0 + 0.5 * max(0, nS_t - 3))
            g_raw_pred = gate_base - load_pen
            wm_weight_eff_act = 1.0 / (1.0 + np.exp(-g_raw_pred))

            p_total = wm_weight_eff_act * p_wm + (1.0 - wm_weight_eff_act) * p_rl

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update (asymmetric learning)
            pe = r - q[s, a]
            lr_use = lr_pos if pe >= 0 else lr_neg
            q[s, a] += lr_use * pe

            # WM decay (age- and load-dependent)
            decay = 0.03 * (1.0 + 0.7 * age_group) * (1.0 + 0.5 * max(0, nS_t - 3))
            w = (1.0 - decay) * w + decay * w_0

            # Surprise-gated WM writing (post-outcome)
            surprise = abs(pe)
            g_raw = gate_base + 3.0 * surprise - load_pen
            wm_weight_eff_write = 1.0 / (1.0 + np.exp(-g_raw))

            if wm_weight_eff_write > 0.0:
                if r >= 0.5:
                    target = np.zeros(nA)
                    target[a] = 1.0
                    w[s, :] = (1.0 - wm_weight_eff_write) * w[s, :] + wm_weight_eff_write * target
                else:
                    anti = np.ones(nA) / (nA - 1)
                    anti[a] = 0.0
                    w[s, :] = (1.0 - 0.5 * wm_weight_eff_write) * w[s, :] + 0.5 * wm_weight_eff_write * anti

            tr += 1

    return simulated_actions, simulated_rewards
