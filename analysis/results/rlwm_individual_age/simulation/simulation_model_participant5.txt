def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate behavior under Cognitive Model 1 (Hybrid RL + Reward-gated WM with set size and age-modulated WM weighting).

    Arguments:
    - stimulus: sequence of stimuli for each trial (array, int-coded states)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action for each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, wm_weight_base, softmax_beta, wm_lr, ss_slope]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr, wm_weight_base, softmax_beta, wm_lr, ss_slope = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    wm_decay = 0.2
    nA = 3

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        # Determine set size for the block (assumed constant within block)
        nS = int(set_sizes[block_indices][0])

        # Map each state to its correct action
        # Assumes states are 0..nS-1 within block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of state s in this block
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = int(block_correct[idx[0]])
            else:
                # Fallback (should not happen if all states appear): default to 0
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            # RL policy
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (very deterministic)
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(logits_wm)
            p_wm /= np.sum(p_wm)

            # Set size modulation and age effect on WM mixture weight
            ss_factor = (3.0 / nS) ** max(ss_slope, 0.0)
            wm_weight_eff = wm_weight_base * ss_factor * (1.0 - 0.3 * age_group)
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, None)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correctness
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward baseline
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Reward-gated WM update (store one-hot after reward)
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - wm_lr) * w[s, :] + wm_lr * onehot

            tr += 1

    return simulated_actions, simulated_rewards
