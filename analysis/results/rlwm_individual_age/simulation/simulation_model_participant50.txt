def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using:
    RL + WM capacity-gated mixture with state-wise choice-kernel perseveration.

    Inputs:
    - stimulus: state index on each trial (array of ints; 0..nS-1 within block)
    - blocks: block index per trial (array of ints)
    - set_sizes: block set size per trial (array of ints; e.g., 3 or 6)
    - correct_answer: correct action index per trial (array of ints; 0..2)
    - age: participant's age (array with single repeated value)
    - parameters: [lr, beta_base, K_young, K_old, eta_ck, ck_decay]

    Returns:
    - simulated_actions: array of sampled actions (ints)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, beta_base, K_young, K_old, eta_ck, ck_decay = parameters
    eps = 1e-12
    age_group = 1 if age[0] > 45 else 0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        softmax_beta = beta_base * 10.0
        softmax_beta_wm = 50.0  # near-deterministic WM

        K = K_old if age_group == 1 else K_young
        wm_weight = float(np.clip(K / max(1.0, nS), 0.0, 1.0))

        # Map each state to its correct action (first occurrence within the block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idxs = np.where(block_states == s)[0]
            if len(idxs) > 0:
                correct_actions[s] = block_correct[idxs[0]]
            else:
                # Fallback if a state is missing (shouldn't happen): assign a random valid action
                correct_actions[s] = np.random.randint(nA)

        # Initialize value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        ck = np.zeros((nS, nA))  # choice-kernel

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            CK_s = ck[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy with centered choice-kernel
            CK_centered = CK_s - np.mean(CK_s)
            prefs_rl = softmax_beta * (Q_s - np.max(Q_s)) + eta_ck * CK_centered
            pref_shift_rl = prefs_rl - np.max(prefs_rl)
            pi_rl = np.exp(pref_shift_rl)
            pi_rl = pi_rl / (np.sum(pi_rl) + eps)

            # WM policy
            prefs_wm = softmax_beta_wm * (W_s - np.max(W_s))
            pref_shift_wm = prefs_wm - np.max(prefs_wm)
            pi_wm = np.exp(pref_shift_wm)
            pi_wm = pi_wm / (np.sum(pi_wm) + eps)

            # Mixture policy
            p_total = wm_weight * pi_wm + (1.0 - wm_weight) * pi_rl
            p_total = p_total / (np.sum(p_total) + eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # Choice-kernel update (state-wise perseveration)
            ck[s, :] *= ck_decay
            ck[s, a] += (1.0 - ck_decay)

            # WM update: store correct action on rewarded trials only
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            # else: no change to WM on errors (as in fitting code)

            tr += 1

    return simulated_actions, simulated_rewards
