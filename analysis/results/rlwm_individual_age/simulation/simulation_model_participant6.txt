def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate actions and rewards under RLWM with surprise-gated WM arbitration and load-adjusted WM forgetting.

    Arguments:
    - stimulus: sequence of stimuli for each trial (array of state indices)
    - blocks: block index for each trial (array)
    - set_sizes: set size for each trial (array; per block, all trials share the same set size)
    - correct_answer: correct action on each trial (array; integer 0..2)
    - age: participant's age (array, single repeated value)
    - parameters: list with up to 6 parameters:
        [lr, wm_weight, softmax_beta, gate_mid=0.0, gate_slope=3.0, wm_forget=0.2]

    Returns:
    - simulated_actions: simulated action choices (array, int)
    - simulated_rewards: simulated binary rewards (array, int)
    """
    import numpy as np

    # Unpack parameters
    lr, wm_weight, softmax_beta = parameters[:3]
    extra = list(parameters[3:])
    gate_mid = extra[0] if len(extra) > 0 else 0.0
    gate_slope = extra[1] if len(extra) > 1 else 3.0
    wm_forget = extra[2] if len(extra) > 2 else 0.2

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    # Age group
    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    eps = 1e-12

    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct_actions = correct_answer[block_indices].astype(int)

        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(set_sizes[block_indices][0])

        # Map each state in this block to its correct action (use first occurrence)
        # Ensure we cover all 0..nS-1 states
        correct_actions = np.zeros(nS, dtype=int)
        for s_idx in range(nS):
            # find first trial with this state
            mask = np.where(block_states == s_idx)[0]
            if len(mask) > 0:
                correct_actions[s_idx] = block_correct_actions[mask[0]]
            else:
                # Fallback if state not present (shouldn't happen): default to 0
                correct_actions[s_idx] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Maintain a per-state surprise/PE tracker for causal gating
        pe_state = np.zeros(nS, dtype=float)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy (softmax)
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(logits_rl)
            p_rl = p_rl / np.sum(p_rl)

            # WM policy (near-deterministic softmax)
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(logits_wm)
            p_wm = p_wm / np.sum(p_wm)

            # Surprise-gated WM arbitration using previous PE magnitude for this state
            pe_mag_for_gate = float(np.clip(np.abs(pe_state[s]), 0.0, 1.0))
            gate_input = gate_mid - gate_slope * pe_mag_for_gate
            gate_prob = 1.0 / (1.0 + np.exp(-gate_input))
            wm_weight_eff = wm_weight * gate_prob * (3.0 / nS) * (1.0 - 0.3 * age_group)
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr + t] = a

            # Determine reward from ground-truth correct action
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr + t] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM forgetting factor (load- and age-adjusted)
            forget = wm_forget * (nS / 3.0) * (1.0 + 0.5 * age_group)
            forget = float(np.clip(forget, 0.0, 0.95))

            # WM write if rewarded; write strength scales with (1 - surprise)
            pe_mag_current = abs(delta)
            if r > 0.5:
                write_strength = float(np.clip(1.0 - pe_mag_current, 0.0, 1.0))
                w[s, :] = (1.0 - write_strength) * w[s, :]
                w[s, a] += write_strength
            else:
                # no explicit punitive write; handled by forgetting below
                pass

            # Normalize and apply forgetting toward uniform
            w[s, :] = np.clip(w[s, :], 0.0, None)
            w[s, :] = w[s, :] / max(np.sum(w[s, :]), eps)
            w[s, :] = (1.0 - forget) * w[s, :] + forget * w_0[s, :]

            # Update PE tracker for causal gating on next encounter of this state
            pe_state[s] = pe_mag_current

        tr += len(block_states)

    return simulated_actions, simulated_rewards
