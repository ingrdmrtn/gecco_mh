import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RLWM with confidence-based arbitration (gating) simulation.

    Arguments:
    - stimulus: sequence of stimuli (states) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, softmax_beta, theta_wm, gamma_gate, alpha_wm]

    Returns:
    - simulated_actions: simulated action choices (array, int)
    - simulated_rewards: simulated binary rewards (array, int)
    """
    lr, softmax_beta, theta_wm, gamma_gate, alpha_wm = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_group = 0 if age[0] <= 45 else 1
    theta_eff_shift = 0.1 if age_group == 1 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_correct_ans = correct_answer[block_idx]

        # Map raw state labels to 0..nS-1 indices
        unique_states = np.unique(block_states_raw)
        nS = int(set_sizes[block_idx][0]) if len(block_idx) > 0 else len(unique_states)
        state_to_idx = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_idx[s] for s in block_states_raw], dtype=int)

        # Derive per-state correct action from first occurrence
        nA = 3
        per_state_correct = np.zeros(len(unique_states), dtype=int)
        for s_val, s_idx in state_to_idx.items():
            first_occ = np.where(block_states_raw == s_val)[0][0]
            per_state_correct[s_idx] = block_correct_ans[first_occ]

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((len(unique_states), nA))
        w = (1.0 / nA) * np.ones((len(unique_states), nA))

        theta_eff = (theta_wm + theta_eff_shift)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            Qc = Q_s - np.max(Q_s)
            exp_rl = np.exp(softmax_beta * Qc)
            denom_rl = np.sum(exp_rl)
            p_rl = exp_rl / max(1e-12, denom_rl)

            # WM policy
            Lc = W_s - np.max(W_s)
            exp_wm = np.exp(softmax_beta_wm * Lc)
            denom_wm = np.sum(exp_wm)
            p_wm = exp_wm / max(1e-12, denom_wm)

            # WM confidence and gate
            sorted_W = np.sort(W_s)[::-1]
            top1 = sorted_W[0]
            top2 = sorted_W[1] if nA > 1 else 0.0
            conf = top1 - top2
            conf_scaled = conf / (float(nS) / 3.0)
            gate = 1.0 / (1.0 + np.exp(-gamma_gate * (conf_scaled - theta_eff)))
            gate = float(np.clip(gate, 0.0, 1.0))

            # Mixture policy
            p_total = gate * p_wm + (1.0 - gate) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward from environment
            correct_a = int(per_state_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            rpe = r - Q_s[a]
            q[s, a] += lr * rpe

            # WM update toward target
            target = np.ones(nA) / nA
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
            w[s, :] += alpha_wm * (target - W_s)
            w[s, :] = np.clip(w[s, :], 1e-8, None)
            w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
