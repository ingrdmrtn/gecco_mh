def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the model:
    RL with decaying choice bias and WM recency cache; WM weight equals memory strength.

    Arguments:
    - stimulus: sequence of stimuli (state indices) per trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action (0..2) on each trial (array)
    - age: participant's age (array, single repeated value)
    - parameters: [lr, beta_base, bias_gain, bias_decay, wm_interf, beta_age_shift]

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    import numpy as np

    lr, beta_base, bias_gain, bias_decay, wm_interf, beta_age_shift = parameters
    age_group = 0 if age[0] <= 45 else 1

    softmax_beta = beta_base * 10.0 * (1.0 - beta_age_shift * age_group)
    softmax_beta = max(softmax_beta, 1e-3)  # keep positive
    softmax_beta_wm = 50.0  # very deterministic WM readout

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its (assumed constant) correct action within the block
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                correct_actions_by_state[s] = block_correct[idx_s[0]]
            else:
                correct_actions_by_state[s] = 0  # fallback if state not present

        # Initialize RL Q-values, WM structures, and choice bias
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))   # used only to retrieve cached action via argmax
        w_0 = (1.0 / nA) * np.ones((nS, nA)) # default WM distribution when cache weak
        bias = np.zeros((nS, nA))

        time_since = np.full(nS, 1e6)  # large means "unknown"/long ago
        cached = np.zeros(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Advance WM time counters
            time_since += 1.0

            # RL with decaying choice bias
            Q_s = q[s, :] + bias[s, :]
            Q_s_centered = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_s_centered)
            p_rl /= np.sum(p_rl)

            # WM strength and policy
            s_m = np.exp(-wm_interf * max(0.0, float(nS) - 1.0) * (time_since[s] / max(1.0, float(nS))))
            s_m = float(np.clip(s_m, 0.0, 1.0))

            cached_a = int(np.argmax(w[s, :]))  # the last rewarded action for this state (if any)
            onehot_cached = np.zeros(nA)
            onehot_cached[cached_a] = 1.0
            W_s_vec = (1.0 - s_m) * w_0[s, :] + s_m * onehot_cached

            W_s_centered = W_s_vec - np.max(W_s_vec)
            p_wm = np.exp(softmax_beta_wm * W_s_centered)
            p_wm /= np.sum(p_wm)

            # Arbitration: WM weight equals memory strength
            wm_weight = s_m
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Simulate reward from correct action for this state
            correct_a = int(correct_actions_by_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # Choice bias update (decay globally, then add bias for chosen action in current state)
            bias *= (1.0 - bias_decay)
            bias[s, a] += bias_gain

            # WM cache update on rewarded trials
            if r > 0:
                cached[s] = a
                w[s, :] = 0.0
                w[s, a] = 1.0
                time_since[s] = 0.0

            tr += 1

    return simulated_actions, simulated_rewards
