def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL with Pearce–Hall attention and reward-gated WM with set-size interference (simulation).

    Arguments:
    - stimulus: sequence of stimuli (states) for each trial (array of ints)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array of ints in [0,1,2])
    - age: participant's age (array, single repeated value)
    - parameters: [alpha_base, k_attn, softmax_beta, wm_weight_base, wm_interf]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    alpha_base, k_attn, softmax_beta, wm_weight_base, wm_interf = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM readout

    age_group = 0 if age[0] <= 45 else 1

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Build per-state correct action (assumes consistent mapping within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            inds = np.where(block_states == s)[0]
            if len(inds) > 0:
                correct_actions[s] = block_correct[inds[0]]
            else:
                # Fallback if a state is not encountered (shouldn't happen in typical blocks)
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM mixture weight with age and set-size modulation
        wm_mix = max(0.0, min(1.0, wm_weight_base)) * (3.0 / float(nS))
        if age_group == 1:
            wm_mix *= 0.7

        # Set-size interference -> decay toward uniform
        interf = max(0.0, wm_interf) * ((float(nS) - 3.0) / 3.0)
        decay = max(0.0, min(1.0, 1.0 - interf))

        for local_t, tr in enumerate(block_idx):
            s = int(block_states[local_t])

            # Softmax over RL values
            Q_s = q[s, :]
            Q_shift = Q_s - np.max(Q_s)
            expQ = np.exp(softmax_beta * Q_shift)
            p_rl = expQ / np.sum(expQ)

            # Softmax over WM values
            W_s = w[s, :]
            W_shift = W_s - np.max(W_s)
            expW = np.exp(softmax_beta_wm * W_shift)
            p_wm = expW / np.sum(expW)

            # Mixture policy
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # RL update with Pearce–Hall attention
            pe = float(r) - Q_s[a]
            alpha_eff = alpha_base + k_attn * abs(pe)
            alpha_eff = max(0.0, min(1.0, alpha_eff))
            q[s, a] += alpha_eff * pe

            # WM decay toward uniform
            w[s, :] = decay * w[s, :] + (1.0 - decay) * w_0[s, :]

            # Reward-gated WM write
            if r > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0

    return simulated_actions, simulated_rewards
