def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + capacity-limited, decaying Working Memory (WM) mixture model simulation.

    Parameters:
    - stimulus: array of state indices for each trial
    - blocks: array of block indices for each trial
    - set_sizes: array of block-dependent set size (nS) for each trial
    - correct_answer: array of correct action indices for each trial
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, lapse]

    Returns:
    - simulated_actions: array of simulated action choices (int)
    - simulated_rewards: array of simulated binary rewards (int)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, lapse = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 100.0  # near-deterministic WM
    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use the block's declared set size (as in the fitting code)
        nS = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else int(np.max(block_states) + 1)

        # Map each state in this block to its correct action (first occurrence)
        # If some states are not encountered, they will not be queried anyway.
        state_to_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t in range(len(block_states)):
            s = int(block_states[t])
            if 0 <= s < nS and not seen[s]:
                state_to_correct[s] = int(block_correct[t])
                seen[s] = True

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight scales with capacity limit
        wm_weight_eff = wm_weight * (3.0 / max(1.0, float(nS)))
        wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Default to uniform choice if out-of-range (defensive)
            if not (0 <= s < nS):
                a = np.random.choice(nA)
                r = 0
                simulated_actions[block_indices[t]] = a
                simulated_rewards[block_indices[t]] = r
                # Apply WM decay even if invalid
                w = (1 - wm_decay) * w + wm_decay * w_0
                continue

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl /= np.sum(prl)

            # WM softmax (near-deterministic)
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm /= np.sum(pwm)

            # Mixture and lapse
            pmix = wm_weight_eff * pwm + (1.0 - wm_weight_eff) * prl
            p_total = (1.0 - lapse) * pmix + lapse * (1.0 / nA)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Generate reward based on correct action mapping
            correct_a = int(state_to_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w = (1 - wm_decay) * w + wm_decay * w_0

            # WM update on reward: store perfect association for this state
            if r > 0:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0
                w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
