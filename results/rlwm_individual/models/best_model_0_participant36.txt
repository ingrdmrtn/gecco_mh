def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with valence-dependent learning, action stickiness, and WM mixture.
    - RL: two learning rates for positive/negative prediction errors.
    - Stickiness: bias to repeat the last action taken in the same state (adds to RL logits).
    - WM: writes the rewarded action (one-shot) and decays; stronger in small set size blocks.
    - Mixture between WM and RL policies; no additional lapse (stickiness absorbs some noise).

    Parameters
    ----------
    model_parameters : list or array-like of length 6
        alpha_pos : float
            RL learning rate for positive PE (0..1).
        alpha_neg : float
            RL learning rate for negative PE (0..1).
        wm_weight : float
            Base WM mixture weight; scaled by set size inside the model.
        softmax_beta : float
            RL inverse temperature (scaled by 10 internally).
        tau_stick : float
            Strength of action stickiness added to the last action's logit.
        wm_decay : float
            WM decay toward uniform (0..1).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    alpha_pos, alpha_neg, wm_weight_base, softmax_beta, tau_stick, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)  # -1 means no prior action in this state

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += tau_stick  # add bias to repeat

            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            set_size_scale = 3.0 / nS
            wm_weight = np.clip(wm_weight_base * set_size_scale, 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            pe = r - q[s, a]
            lr = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += lr * pe

            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
