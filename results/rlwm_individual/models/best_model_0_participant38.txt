def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with Q-value forgetting + WM with set-size-dependent interference.

    Idea:
    - RL: standard delta rule plus forgetting toward uniform (helps when set size is large/volatile).
    - WM: per-state policy that suffers global interference proportional to set size:
      all WM rows are nudged toward their grand average each trial, then the current state's row is
      adjusted toward the chosen action (rewarded) or away (unrewarded).
    - Set size affects:
      - WM interference strength via a multiplicative factor proportional to (nS-1)/nS.
      - Mixture weight is constant; interference degrades WM more for larger sets.

    Parameters
    - model_parameters[0] = lr (float in [0,1]): RL learning rate.
    - model_parameters[1] = q_forget (float in [0,1]): Forgetting rate pulling Q toward uniform each trial.
    - model_parameters[2] = softmax_beta (float > 0): RL inverse temperature (internally scaled by 10).
    - model_parameters[3] = wm_weight (float in [0,1]): Mixture weight for WM policy.
    - model_parameters[4] = wm_interf (float in [0,1]): Base WM interference rate (scaled by (nS-1)/nS).
    - model_parameters[5] = wm_gain (float in [0,1]): Strength of WM update toward/away from chosen action in current state.

    Returns
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, q_forget, softmax_beta, wm_weight, wm_interf, wm_gain = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        interf_factor = (nS - 1.0) / nS
        wm_interf_block = wm_interf * interf_factor

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            delta = r - Q_s[a]
            q[s][a] += lr * delta

            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * w_0[s, :]

            mean_row = np.mean(w, axis=0)
            w = (1.0 - wm_interf_block) * w + wm_interf_block * np.tile(mean_row, (nS, 1))

            if r > 0.5:

                w[s, :] = (1.0 - wm_gain) * w[s, :]
                w[s, a] += wm_gain
            else:

                dec = min(wm_gain, w[s, a])
                w[s, a] -= dec
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += dec / (nA - 1)

            row_sum = np.sum(w[s, :])
            if row_sum > 1e-12:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p
