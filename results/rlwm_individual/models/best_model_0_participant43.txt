def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with surprise-gated WM encoding and load-modulated forgetting.

    Idea:
    - RL learns action values via delta rule and softmax.
    - WM stores a one-shot policy per state when outcomes are surprising and rewarding.
    - Surprise is approximated from the current state's RL certainty: surprise â‰ˆ 1 - max(Q_s).
    - WM decay increases with load (larger set size => faster forgetting).
    - Arbitration between WM and RL is gated by surprise: more surprise => rely more on WM.

    Parameters:
    - lr: RL learning rate in [0,1].
    - wm_weight_base: Base weight for WM contribution in [0,1].
    - softmax_beta: RL inverse temperature (scaled internally x10).
    - wm_forget: Baseline WM forgetting rate in [0,1]. Larger => faster decay toward uniform.
    - pe_sensitivity: Surprise sensitivity (>=0). Scales the sigmoid that maps surprise to WM arbitration and storage strength.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_forget, pe_sensitivity = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        wm_forget_eff = np.clip(wm_forget * (1.0 + load), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            surprise = 1.0 - np.max(Q_s)

            attention = 1.0 / (1.0 + np.exp(-pe_sensitivity * surprise))

            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            wm_weight_eff = np.clip(wm_weight_base * attention, 0.0, 1.0)

            p_total = p_wm * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            delta = r - Q_s[a]
            q[s][a] += lr * delta



            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0

                encode_strength = np.clip(attention * wm_forget_eff, 0.0, 1.0)
                w[s, :] = (1.0 - encode_strength) * w[s, :] + encode_strength * target
            else:
                w[s, :] = (1.0 - wm_forget_eff) * w[s, :] + wm_forget_eff * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
