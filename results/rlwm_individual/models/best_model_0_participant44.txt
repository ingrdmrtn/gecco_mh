def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + surprise-gated WM arbitration with age-adjusted learning speed.

    Model idea:
    - RL learns Q-values with a single learning rate, but older adults have a slowdown factor on learning.
    - WM stores a peaked distribution over actions per state; on reward it moves toward a one-hot code,
      on no-reward it decays toward uniform.
    - The arbitration weight for WM is stronger when:
        (a) set size is small (low load), and
        (b) the current trial has high surprise (|reward - Q| larger), reflecting that WM engages more when
            there is a notable discrepancy that can be resolved by explicit memory.
      Older adults receive a penalty to WM contribution.
    - Action selection is a mixture of RL softmax and WM softmax.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight_base: baseline WM mixing weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - surprise_sensitivity: scales how much surprise gates WM (>=0)
    - age_rl_slowdown: factor that reduces RL learning in older adults (0..1; effective lr *= (1 - age_rl_slowdown) if older)

    Inputs:
    - states, actions, rewards: arrays per trial
    - blocks: block id per trial
    - set_sizes: set size per trial (3 or 6)
    - age: array-like with single participant age

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, surprise_sensitivity, age_rl_slowdown = model_parameters
    softmax_beta *= 10.0  # scale as in template
    softmax_beta_wm = 50.0  # very deterministic WM policy
    age = age[0]
    older = 1.0 if age >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))  # WM distribution per state
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # uniform prior WM

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            load_scale = 3.0 / float(nS)  # 1.0 for setsize=3, 0.5 for setsize=6
            delta = r - Q_s[a]
            surprise = abs(delta)
            wm_mix = wm_weight_base * load_scale * (1.0 - 0.4 * older) * (1.0 + surprise_sensitivity * surprise)
            wm_mix = np.clip(wm_mix, 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            eff_lr = lr * (1.0 - age_rl_slowdown * older)
            q[s, a] += eff_lr * (r - q[s, a])

            if r > 0.5:

                alpha_w = np.clip(0.5 + 0.5 * surprise, 0.0, 1.0)
                peaked = (1.0 / nA) * np.ones(nA)
                peaked[a] = 1.0
                peaked /= np.sum(peaked)
                w[s, :] = (1.0 - alpha_w) * W_s + alpha_w * peaked
            else:

                decay = 0.3 + 0.4 * (nS - 3) / 3.0  # 0.3 for setsize=3, 0.7 for setsize=6
                w[s, :] = (1.0 - decay) * W_s + decay * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
