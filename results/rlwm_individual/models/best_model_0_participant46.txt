def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with slot-limited capacity and decay (mixture policy).

    Idea:
    - Choices are a mixture of model-free RL and a fast, capacity-limited working memory (WM).
    - WM encodes state-action pairs quickly when rewarded, but decays toward uniform.
    - WM mixture weight is reduced when set size grows beyond a capacity parameter.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1)
    - wm_weight: baseline mixture weight on WM (0..1)
    - softmax_beta: inverse temperature for RL softmax (scaled up internally)
    - wm_decay: decay rate of WM toward uniform each time the state is encountered (0..1)
    - wm_learn: one-shot WM learning rate when reward is received (0..1)
    - wm_capacity: effective WM capacity in number of items (e.g., 1..6). Reduces WM influence when nS > capacity.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_learn, wm_capacity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        cap_scale = min(1.0, wm_capacity / max(1.0, float(nS)))
        wm_weight_block = np.clip(wm_weight * cap_scale, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))


            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            log_p += np.log(max(p_total, eps))

            delta = r - Q_s[a]
            q[s, a] += lr * delta


            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            if r > 0.0:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target

        blocks_log_p += log_p

    return -blocks_log_p
