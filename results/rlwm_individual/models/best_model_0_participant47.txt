def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with capacity-scaling and WM decay.

    Idea:
    - Choices arise from a mixture of a model-free RL policy and a working-memory (WM) policy.
    - WM is a fast, one-shot encoder that stores rewarded S-A associations but decays toward a uniform baseline.
    - The effective WM contribution scales down with set size via a capacity parameter K (i.e., WM_weight_eff = wm_weight * min(1, K/nS)).

    Parameters (tuple):
    - lr: RL learning rate (0..1)
    - wm_weight: Base mixture weight for WM contribution (0..1)
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally; higher => more deterministic)
    - phi: WM decay toward baseline per trial (0..1)
    - eta_wm: WM encoding strength on rewarded trials (0..1)
    - K: WM capacity (in number of states; positive real). Effective WM weight is scaled by min(1, K/nS)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, phi, eta_wm, K = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        cap_scale = min(1.0, max(0.0, K / float(nS)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            wm_weight_eff = wm_weight * cap_scale
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)
            wm_weight = wm_weight_eff  # ensure the template line uses the effective weight

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)  # numerical guard
            log_p += np.log(p_total)

            delta = r - Q_s[a]
            q[s][a] += lr * delta


            w[s, :] = (1.0 - phi) * w[s, :] + phi * w_0[s, :]

            if r > 0.0:
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm

        blocks_log_p += log_p

    return -blocks_log_p
