def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with choice stickiness + WM availability governed by capacity (sigmoid) and WM temperature scaling.

    Idea:
    - RL uses standard delta learning, but the policy includes choice stickiness within-state
      by temporarily boosting the last chosen action's value at decision time.
    - WM stores one-shot associations; availability of WM is a sigmoid function of set size relative to capacity.
    - WM softmax temperature is scaled by a parameter to allow less-than-perfect WM.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight: Baseline WM mixture weight (0..1).
    - softmax_beta: Inverse temperature for RL policy (scaled internally by 10).
    - capacity_k: WM capacity threshold (in items) around which WM availability drops.
    - capacity_slope: Positive slope for the sigmoid controlling how sharply WM availability declines with set size.
    - stickiness: Choice perseveration bias added (temporarily) to the last chosen action's Q in that state.

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, capacity_k, capacity_slope, stickiness = model_parameters

    softmax_beta *= 10.0
    base_softmax_beta_wm = 50.0  # baseline; we will keep it but availability gates WM contribution

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_action = -1 * np.ones(nS, dtype=int)


        avail = 1.0 / (1.0 + np.exp(capacity_slope * (float(nS) - capacity_k)))
        wm_weight_eff = wm_weight * avail

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stickiness

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            W_s = w[s, :]

            softmax_beta_wm = base_softmax_beta_wm
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            p_total = p_wm * wm_weight_eff + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            pe = r - q[s, a]
            q[s, a] += lr * pe

            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0

                write_rate = min(1.0, 0.5 + 0.5 * avail)
                w[s, :] = (1.0 - write_rate) * w[s, :] + write_rate * one_hot
            else:

                decay_rate = min(1.0, 0.5 + 0.5 * (1.0 - avail))
                w[s, :] = (1.0 - decay_rate) * w[s, :] + decay_rate * w_0[s, :]

            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
