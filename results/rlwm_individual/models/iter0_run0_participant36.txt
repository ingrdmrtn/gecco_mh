def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with decaying WM values and lapses.
    - RL system: tabular Q-learning with softmax choice.
    - WM system: fast-learning action weights per state that decay toward uniform.
      WM learns toward a target vector where the chosen action gets value r (0/1) and others 0.
    - Policy mixture: convex combination of WM and RL with a global WM weight.
    - Lapse: small probability of a uniform random choice to capture noise/exploration.

    Parameters
    ----------
    model_parameters : list or array-like of length 6
        lr : float
            RL learning rate (0..1).
        wm_weight : float
            Mixture weight for WM policy in action selection (0..1).
        softmax_beta : float
            Inverse temperature for RL softmax (scaled by 10 internally).
        wm_decay : float
            Per-trial decay of WM values toward uniform (0..1).
        wm_learning : float
            WM learning rate toward target vector (0..1).
        lapse : float
            Lapse probability for random choice (0..1).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_learning, lapse = model_parameters
    softmax_beta *= 10.0  # higher upper bound
    softmax_beta_wm = 50.0  # very deterministic WM
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: probability of chosen action a via softmax trick
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Working Memory policy: softmax over WM action weights
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with lapse (uniform random choice with prob=lapse)
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)

            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay of current state's weights toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM learning toward target vector (chosen action gets r, others 0)
            target = np.zeros(nA)
            target[a] = r
            w[s, :] += wm_learning * (target - w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with uncertainty-based WM gating and set-size scaling.
    - RL: tabular Q-learning with softmax.
    - WM: one-shot storage of rewarded action; decays toward uniform otherwise.
    - Dynamic mixture:
        * Base WM weight scaled by set size (capacity pressure): weight * (3 / nS).
        * Additional gating by RL uncertainty: more WM when RL is uncertain (high entropy).
          Implemented via a sigmoid on (H_norm - 0.5) with gate_beta controlling sensitivity.
    - Lapse for random choices.

    Parameters
    ----------
    model_parameters : list or array-like of length 6
        lr : float
            RL learning rate (0..1).
        wm_weight : float
            Base WM weight before dynamic scaling (0..1).
        softmax_beta : float
            RL inverse temperature (scaled by 10 internally).
        wm_decay : float
            WM decay toward uniform per trial (0..1).
        gate_beta : float
            Sensitivity of WM gating to RL uncertainty (higher => stronger gating).
        lapse : float
            Lapse probability for random choice (0..1).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, gate_beta, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax choice prob for chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # RL uncertainty (entropy of RL softmax over Q_s)
            logits = softmax_beta * (Q_s - np.max(Q_s))
            pi = np.exp(logits)
            pi /= np.sum(pi)
            H = -np.sum(pi * np.log(np.maximum(pi, eps)))  # nats
            H_norm = H / np.log(nA)  # normalize to [0,1]

            # Dynamic WM weight = base * set-size scaling * sigmoid gating by uncertainty
            set_size_scale = 3.0 / nS  # less WM influence when nS=6 than when nS=3
            gate = 1.0 / (1.0 + np.exp(-gate_beta * (H_norm - 0.5)))
            wm_weight = np.clip(wm_weight_base * set_size_scale * gate, 0.0, 1.0)

            # Mixture with lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)

            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM one-shot storage on rewarded trials; otherwise just decay
            if r > 0.5:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]  # small additional stabilizing decay
                w[s, :] = 0.0
                w[s, a] = 1.0  # store the correct action deterministically

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with valence-dependent learning, action stickiness, and WM mixture.
    - RL: two learning rates for positive/negative prediction errors.
    - Stickiness: bias to repeat the last action taken in the same state (adds to RL logits).
    - WM: writes the rewarded action (one-shot) and decays; stronger in small set size blocks.
    - Mixture between WM and RL policies; no additional lapse (stickiness absorbs some noise).

    Parameters
    ----------
    model_parameters : list or array-like of length 6
        alpha_pos : float
            RL learning rate for positive PE (0..1).
        alpha_neg : float
            RL learning rate for negative PE (0..1).
        wm_weight : float
            Base WM mixture weight; scaled by set size inside the model.
        softmax_beta : float
            RL inverse temperature (scaled by 10 internally).
        tau_stick : float
            Strength of action stickiness added to the last action's logit.
        wm_decay : float
            WM decay toward uniform (0..1).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    alpha_pos, alpha_neg, wm_weight_base, softmax_beta, tau_stick, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)  # -1 means no prior action in this state

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL with stickiness: augment Q with stickiness bias on last action
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += tau_stick  # add bias to repeat

            W_s = w[s, :]

            # RL choice probability for chosen action a
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Set-size dependent WM weight (stronger for small set size)
            set_size_scale = 3.0 / nS
            wm_weight = np.clip(wm_weight_base * set_size_scale, 0.0, 1.0)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update with valence-dependent learning rates
            pe = r - q[s, a]
            lr = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += lr * pe

            # WM decay and one-shot storage on reward
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Update last action for stickiness
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
