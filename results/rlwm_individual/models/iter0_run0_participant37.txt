Here are three candidate cognitive models for the RLWM task. Each model is a standalone Python function that follows the provided template and returns the negative log-likelihood of the observed choices. All parameters are used and capped at six per model.

Note: Assume numpy as np is already imported in your environment.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with capacity-limited WM and decay.
    - RL: standard Q-learning with softmax.
    - WM: one-shot storage of rewarded action with decay toward uniform; WM retrieval probability scales with capacity vs. set size.

    Parameters
    ----------
    model_parameters : tuple
        lr : float
            RL learning rate (0..1).
        wm_weight : float
            Mixture weight for WM vs RL (0..1), applied after WM retrieval uncertainty.
        softmax_beta : float
            RL inverse temperature (scaled internally by 10).
        lr_wm : float
            WM learning rate for storing/adjusting the state-action association.
        decay : float
            Global WM decay toward uniform (0..1) applied each trial.
        K : float
            Effective WM capacity; retrieval probability scales as phi = min(1, K / set_size).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, lr_wm, decay, K = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM retrieval probability based on capacity vs. set size
        phi = min(1.0, max(0.0, K / max(1.0, nS)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy: choice probability of chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(eps, denom_rl)

            # WM policy: softmax over WM values, tempered by retrieval probability phi
            W_s = w[s, :]
            beta_wm_eff = softmax_beta_wm * phi  # less deterministic when phi < 1
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm_soft = 1.0 / max(eps, denom_wm)
            p_wm = phi * p_wm_soft + (1.0 - phi) * (1.0 / nA)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(eps, p_total))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward baseline (global)
            w = (1.0 - decay) * w + decay * w_0

            # WM update: one-shot storage if rewarded; redistribute otherwise
            if r > 0.5:
                # Move WM for this state toward a one-hot at chosen action
                w[s, :] = (1.0 - lr_wm) * w[s, :]
                w[s, a] += lr_wm
            else:
                # Penalize chosen action, redistribute mass to other actions
                w[s, a] = (1.0 - lr_wm) * w[s, a]
                inc = lr_wm / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] = (1.0 - 0.0) * w[s, aa] + inc  # add inc to others

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with confidence-gated WM policy and reward-gated storage.
    - RL: standard Q-learning with softmax.
    - WM: stores only rewarded associations; WM policy becomes near-uniform when WM is not confident.
      Confidence is the margin between the top two WM values; if margin < theta, WM is unreliable.

    Parameters
    ----------
    model_parameters : tuple
        lr : float
            RL learning rate (0..1).
        wm_weight : float
            Mixture weight for WM vs RL (0..1).
        softmax_beta : float
            RL inverse temperature (scaled internally by 10).
        lr_wm : float
            WM learning rate for storing rewarded associations.
        decay : float
            Global WM decay toward uniform (0..1) each trial.
        theta : float
            Confidence threshold (0..1) used to reduce WM influence when WM is uncertain.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, lr_wm, decay, theta = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(eps, denom_rl)

            # Compute WM confidence as margin between best and second-best WM values
            W_s = w[s, :]
            sorted_vals = np.sort(W_s)[::-1]
            margin = sorted_vals[0] - sorted_vals[1] if nA >= 2 else sorted_vals[0]

            # Confidence-to-weight mapping: gamma in [0,1]
            # If margin <= theta -> gamma ~ 0; if margin >= 1 -> gamma ~ 1
            if margin <= theta:
                gamma = 0.0
            else:
                gamma = min(1.0, (margin - theta) / max(eps, 1.0 - theta))

            beta_wm_eff = softmax_beta_wm * gamma
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm_soft = 1.0 / max(eps, denom_wm)
            # When not confident (gamma small), add uniform noise
            p_wm = gamma * p_wm_soft + (1.0 - gamma) * (1.0 / nA)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(eps, p_total))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w = (1.0 - decay) * w + decay * w_0

            # WM update: reward-gated storage (store only when rewarded)
            if r > 0.5:
                w[s, :] = (1.0 - lr_wm) * w[s, :]
                w[s, a] += lr_wm
            # else: just decay already applied; no explicit negative update

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with set-size-dependent WM mixture weight (dynamic gating).
    - RL: standard Q-learning with softmax.
    - WM: fast one-shot storage with decay toward uniform.
    - Mixture weight is dynamically adjusted by set size via a logistic transform:
        wm_weight_eff = sigmoid(wm_w0 + wm_k * (s_ref - set_size)),
      where s_ref=3 emphasizes higher WM weight at smaller set sizes.

    Parameters
    ----------
    model_parameters : tuple
        lr : float
            RL learning rate (0..1).
        wm_w0 : float
            Bias term for the logistic gating of WM weight.
        softmax_beta : float
            RL inverse temperature (scaled internally by 10).
        lr_wm : float
            WM learning rate (0..1).
        decay : float
            WM decay toward uniform per trial (0..1).
        wm_k : float
            Slope for set-size modulation of WM weight (positive -> more WM at 3 vs 6).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_w0, softmax_beta, lr_wm, decay, wm_k = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12
    s_ref = 3.0  # reference small set size

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Dynamic WM weight based on set size
        wm_weight_eff = 1.0 / (1.0 + np.exp(-(wm_w0 + wm_k * (s_ref - float(nS)))))

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(eps, denom_rl)

            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(eps, denom_wm)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            log_p += np.log(max(eps, p_total))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay and update
            w = (1.0 - decay) * w + decay * w_0
            if r > 0.5:
                w[s, :] = (1.0 - lr_wm) * w[s, :]
                w[s, a] += lr_wm
            else:
                # Mild negative update: suppress chosen action a bit
                w[s, a] = (1.0 - lr_wm) * w[s, a]
                incr = lr_wm / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += incr

        blocks_log_p += log_p

    return -blocks_log_p

Explanation of model differences (brief):
- Model 1: WM contribution internally down-weights under high load using a capacity parameter K (via retrieval probability), while the external mixture weight wm_weight stays fixed.
- Model 2: WM policy is confidence-gated; when WM is unreliable (small margin), it is softened toward uniform before mixing with RL.
- Model 3: The WM/RL mixture weight itself is dynamically modulated by set size through a logistic function, yielding stronger WM usage in small set sizes and weaker in large set sizes.
