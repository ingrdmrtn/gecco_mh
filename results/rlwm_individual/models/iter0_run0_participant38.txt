def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with load-sensitive WM capacity and decay.

    The model mixes a standard RL softmax policy with a working-memory (WM) policy.
    WM has limited capacity and decays toward uniform under load. WM weight is
    reduced when set size exceeds capacity.

    Parameters (tuple):
    - lr: RL learning rate in [0,1].
    - wm_weight: Baseline weight of WM in [0,1] (maximal contribution when not capacity-limited).
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally by 10).
    - wm_decay: WM decay rate in [0,1]; effective decay grows with set size.
    - wm_alpha: WM learning strength in [0,1] for updating remembered mapping and suppressing wrong action.
    - wm_capacity: WM capacity (in number of stateâ€“action pairs). Effective WM weight scales with min(1, capacity/nS).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_alpha, wm_capacity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM contribution is capacity-limited
        cap_factor = min(1.0, max(0.0, wm_capacity / max(1, nS)))
        wm_weight_eff = wm_weight * cap_factor

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: probability of chosen action via softmax trick
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Apply load-sensitive WM decay toward uniform before computing policy
            # Effective decay increases with set size (multiplicative interference)
            eff_decay = 1.0 - (1.0 - wm_decay) ** max(1, nS)
            W_s = (1.0 - eff_decay) * W_s + eff_decay * w_0[s, :]

            # WM policy via near-deterministic softmax over WM associative strengths
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture of WM and RL with capacity-adjusted WM weight
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: Hebbian strengthen correct action, suppress incorrect, then renormalize
            if r > 0.5:
                # Move probability mass toward chosen action
                W_s = (1.0 - wm_alpha) * W_s
                W_s[a] += wm_alpha
            else:
                # Suppress chosen action when incorrect
                W_s[a] = (1.0 - wm_alpha) * W_s[a]
            # Ensure valid probability distribution
            W_s = np.clip(W_s, 1e-12, None)
            W_s = W_s / np.sum(W_s)
            w[s, :] = W_s

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with dynamic WM gating that adapts to load.

    The model mixes RL with a WM controller whose state-specific gate g_s (0..1) determines
    the weight of WM vs RL for each state. Gating increases with rewarded experience and
    decays with interference that grows with set size.

    Parameters (tuple):
    - lr: RL learning rate in [0,1].
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally by 10).
    - wm_learn: Increment to WM gate following reward (how fast WM gains control), in [0,1].
    - wm_decay: Baseline decay of WM gate each encounter, in [0,1].
    - wm_interference: Additional decay per unit load beyond 3, modulating decay by set size, in [0,1].
    - wm_weight_baseline: Initial WM gate level (baseline) before learning, in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_learn, wm_decay, wm_interference, wm_weight_baseline = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # State-specific WM gates, initialized lower under higher load
        init_gate = wm_weight_baseline * (3.0 / max(1, nS))
        g = np.clip(init_gate, 0.0, 1.0) * np.ones(nS)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy prob of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM associative strengths softly decay toward uniform each encounter
            W_s = (1.0 - wm_decay) * W_s + wm_decay * w_0[s, :]

            # WM policy via near-deterministic softmax
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # State-specific mixture controlled by gate g[s]
            gate_s = np.clip(g[s], 0.0, 1.0)
            p_total = gate_s * p_wm + (1.0 - gate_s) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM content update: strengthen chosen action on reward
            if r > 0.5:
                # Sharpen toward chosen action
                W_s = 0.5 * W_s
                W_s[a] += 0.5
            # Renormalize
            W_s = np.clip(W_s, 1e-12, None)
            W_s = W_s / np.sum(W_s)
            w[s, :] = W_s

            # Gate update: decay with load-dependent interference, increase with reward
            load_term = max(0.0, (nS - 3.0) / 3.0)  # 0 for nS=3, 1 for nS=6
            eff_decay = np.clip(wm_decay + wm_interference * load_term, 0.0, 1.0)
            g[s] = (1.0 - eff_decay) * g[s] + wm_learn * r
            g[s] = np.clip(g[s], 0.0, 1.0)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + win-stay/lose-shift working-memory heuristic with load-sensitive lapses.

    WM stores a recent correct action per state (win-stay); after losses it avoids the last
    action (lose-shift). Confidence in the WM trace decays and is lower under higher load.
    The final policy is an epsilon-greedy mixture where epsilon increases with set size.

    Parameters (tuple):
    - lr: RL learning rate in [0,1].
    - wm_weight: Weight on WM policy in [0,1] (before epsilon).
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally by 10).
    - eps0: Baseline lapse/undirected exploration probability in [0,1].
    - eps_load: Additional lapse when moving from set size 3 to 6 (load sensitivity), in [0,1].
    - wm_decay: Decay rate of WM confidence per encounter in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, eps0, eps_load, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM memory of last action and reward per state
        last_a = -1 * np.ones(nS, dtype=int)
        last_r = -1.0 * np.ones(nS)
        conf = np.zeros(nS)  # confidence/strength of WM trace in [0,1]

        # Load-sensitive lapse probability
        load_term = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6
        eps = np.clip(eps0 + eps_load * load_term, 0.0, 0.5)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # Construct WM policy distribution from win-stay/lose-shift rule
            if last_a[s] >= 0:
                if last_r[s] > 0.5:
                    # Win-stay: concentrate on last correct action
                    R = np.zeros(nA)
                    R[last_a[s]] = 1.0
                else:
                    # Lose-shift: avoid last action, uniform over alternatives
                    R = np.ones(nA) / (nA - 1.0)
                    R[last_a[s]] = 0.0
            else:
                # No memory yet
                R = w_0[s, :].copy()

            # Confidence-weighted WM distribution (decays toward uniform when low confidence)
            W_s = conf[s] * R + (1.0 - conf[s]) * w_0[s, :]

            # For record-keeping (not strictly necessary for policy), store in w
            w[s, :] = W_s

            # WM softmax policy probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # RL softmax probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Mixture of WM and RL, then apply undirected exploration (epsilon)
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - eps) * p_mix + eps * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM confidence update: decay then boost on reward
            conf[s] = (1.0 - wm_decay) * conf[s] + wm_decay * r
            conf[s] = np.clip(conf[s], 0.0, 1.0)

            # Update memory trace
            last_a[s] = a
            last_r[s] = r

        blocks_log_p += log_p

    return -blocks_log_p
