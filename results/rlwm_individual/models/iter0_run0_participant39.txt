def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM capacity-and-forgetting model (slot-like WM + model-free RL).
    
    The choice policy is a mixture of a reinforcement-learning (RL) policy and a working-memory (WM) policy.
    - RL: standard delta-rule on Q-values with softmax choice.
    - WM: slot-like memory of correct action per state with decay toward uniform; high inverse temperature for WM.
    - Arbitration: WM weight is scaled by an effective capacity term K/nS (capped at 1), so WM contributes less in larger set sizes.

    Parameters
    ----------
    model_parameters: tuple/list of 5 floats
        lr               : RL learning rate (0..1).
        wm_capacity      : WM capacity in number of items (e.g., 1..6); effective WM weight scales as min(1, wm_capacity / set_size).
        softmax_beta     : Base inverse temperature for RL (rescaled by 10 inside the model).
        wm_decay         : WM decay toward uniform per trial (0..1). Larger = faster forgetting.
        lapse            : Lapse rate for WM policy (0..1), mixing a uniform policy into WM to avoid overconfidence.

    Returns
    -------
    nll : float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_capacity, softmax_beta, wm_decay, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM mixture weight scales with capacity relative to set size
        cap_scale = min(1.0, max(0.0, wm_capacity) / max(1.0, nS))
        wm_weight_eff = cap_scale  # use capacity-scaled WM dominance inside the block

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy:
            # Interpret W_s as a probability vector (memory). Add lapse to ensure support.
            p_wm_a = (1.0 - lapse) * W_s[a] + lapse * (1.0 / nA)
            p_wm_a = np.clip(p_wm_a, eps, 1.0 - eps)

            # Mixture policy
            p_total = wm_weight_eff * p_wm_a + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0 - eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM updating:
            # If rewarded, store a strong one-hot memory for the chosen action
            if r > 0.5:
                # Move probability mass toward the chosen action
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # Overwrite strongly by projecting onto one-hot while preserving some residual
                w[s, :] = 0.0 * w[s, :] + one_hot
            else:
                # If not rewarded, suppress the chosen action slightly by renormalizing away from it
                # Move a small fraction from the chosen action to others; here re-normalize to keep sum to 1
                drop = 0.2  # small fixed suppression for error trials
                moved = min(drop, w[s, a] - 1e-6)
                w[s, a] -= moved
                redistribute = moved / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += redistribute

            # Ensure numerical stability
            w[s, :] = np.clip(w[s, :], eps, 1.0)
            w[s, :] = w[s, :] / np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with valence-asymmetric learning + WM one-shot binding with interference.
    
    - RL: Two learning rates for positive vs negative RPEs; softmax policy.
    - WM: On rewarded trials, stores a strong one-shot association (winner-take-all) for that state.
           Between trials, memory suffers interference/forgetting proportional to set size.
    - Arbitration: Fixed WM-RL mixture weight scaled by a strength parameter, attenuated by interference from larger set sizes.

    Parameters
    ----------
    model_parameters: tuple/list of 6 floats
        lr_pos            : RL learning rate for positive prediction errors (0..1).
        lr_neg            : RL learning rate for negative prediction errors (0..1).
        wm_strength       : Baseline WM mixture strength (0..1) before interference.
        softmax_beta      : Base inverse temperature for RL (rescaled by 10 inside the model).
        lapse             : Lapse rate for WM policy (0..1), mixing uniform into WM.
        interference      : Interference/forgetting gain (>0). Effective forgetting scales with interference / set_size.

    Returns
    -------
    nll : float
        Negative log-likelihood of the observed choices under the model.
    """
    lr_pos, lr_neg, wm_strength, softmax_beta, lapse, interference = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        # WM store starts uniform; we'll "write" one-hot on rewarded trials.
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Interference-based attenuation of WM within block
        # Higher set size => stronger forgetting => lower effective WM weight.
        forget_rate = 1.0 - np.exp(-max(1e-8, interference) / max(1.0, nS))
        wm_weight_eff = (1.0 - forget_rate) * wm_strength

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy from stored memory with lapse
            p_wm_a = (1.0 - lapse) * W_s[a] + lapse * (1.0 / nA)
            p_wm_a = np.clip(p_wm_a, eps, 1.0 - eps)

            # Mixture
            p_total = wm_weight_eff * p_wm_a + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0 - eps)
            log_p += np.log(p_total)

            # RL update with valence asymmetry
            pe = r - Q_s[a]
            if pe >= 0.0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM interference/decay toward uniform each trial for the current state
            w[s, :] = (1.0 - forget_rate) * w[s, :] + forget_rate * w_0[s, :]

            # WM update (one-shot binding on reward; clear if not rewarded)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot
            else:
                # On error, partially clear memory for this state toward uniform
                w[s, :] = 0.5 * w[s, :] + 0.5 * w_0[s, :]

            # Stability
            w[s, :] = np.clip(w[s, :], eps, 1.0)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with uncertainty-based arbitration.
    
    - RL: standard delta-rule with softmax policy.
    - WM: fast-learning associative memory (delta-rule over action values) with decay.
          Choice from WM via a high-temperature softmax on its action values.
    - Arbitration: Trial-wise WM weight is a sigmoid of a bias plus a term that increases when
      (a) set size is small and (b) RL policy is uncertain (high entropy).
      This yields more WM reliance when capacity allows and RL is unsure.

    Parameters
    ----------
    model_parameters: tuple/list of 6 floats
        lr                : RL learning rate (0..1).
        softmax_beta      : Base inverse temperature for RL (rescaled by 10 inside the model).
        wm_learn          : WM learning rate (0..1) for its own value table.
        wm_decay          : WM decay toward uniform per trial (0..1).
        wm_bias           : Arbitration bias term (can be negative/positive).
        arbitrate_slope   : Sensitivity of arbitration to (3/set_size - RL entropy) (positive increases WM reliance).

    Returns
    -------
    nll : float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, softmax_beta, wm_learn, wm_decay, wm_bias, arbitrate_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))  # RL values
        w = (1.0 / nA) * np.ones((nS, nA))  # WM values (interpreted as action values, not probabilities)
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax policy from WM values
            # High inverse temperature makes WM policy near-deterministic on its current max
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute RL policy entropy (using a stable softmax)
            logits = softmax_beta * (Q_s - np.max(Q_s))
            pi = np.exp(logits)
            pi = pi / np.sum(pi)
            entropy_rl = -np.sum(pi * np.log(np.clip(pi, eps, 1.0))) / np.log(nA)  # normalized 0..1

            # Arbitration: more WM when set size small and RL uncertain (high entropy)
            ss_term = 3.0 / max(1.0, nS)  # 1.0 for set size 3, 0.5 for set size 6
            arb_input = wm_bias + arbitrate_slope * (ss_term - entropy_rl)
            wm_weight = 1.0 / (1.0 + np.exp(-arb_input))  # sigmoid in [0,1]

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, eps, 1.0 - eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform (value forgetting)
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM learning: fast delta-rule on the chosen action toward observed reward
            w_pe = r - W_s[a]
            w[s, a] += wm_learn * w_pe

        blocks_log_p += log_p

    return -blocks_log_p
