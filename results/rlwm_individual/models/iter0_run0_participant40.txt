def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with one-shot WM storage, decay, and set-size-dependent precision.

    Parameters
    ----------
    states : array-like of int
        State indices on each trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen actions on each trial (0..nA-1).
    rewards : array-like of int
        Binary feedback (0/1) for each trial.
    blocks : array-like of int
        Block index per trial; states reset across blocks.
    set_sizes : array-like of int
        Set size (number of distinct states in block) per trial.
    model_parameters : iterable of 6 floats
        lr : RL learning rate in [0,1].
        wm_weight : Mixture weight for WM vs RL (0..1); larger = more WM.
        softmax_beta : Inverse temperature for RL policy; gets scaled by 10 inside.
        wm_decay : Rate of WM decay towards uniform each trial (0..1).
        wm_store : One-shot WM learning strength when rewarded (0..1).
        wm_capacity_exp : Exponent controlling how WM precision scales with set size.
            Effective WM inverse temperature is beta_wm * (3/nS) ** wm_capacity_exp.

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_store, wm_capacity_exp = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: set-size-dependent precision
            # Reduce WM determinism as nS grows via exponent
            beta_wm_eff = softmax_beta_wm * (3.0 / float(nS)) ** wm_capacity_exp
            logits_wm = beta_wm_eff * W_s
            logits_wm = logits_wm - np.max(logits_wm)  # stability
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = pi_wm[a]

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # 1) Global decay of WM towards uniform for this state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # 2) One-shot storage on positive feedback: move WM for state s toward chosen action
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_store) * w[s, :] + wm_store * one_hot

            # 3) Renormalize to a proper probability vector
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with recency-based WM, interference by set size, and action perseveration bias embedded in WM.

    Parameters
    ----------
    states : array-like of int
        State indices on each trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen actions on each trial (0..nA-1).
    rewards : array-like of int
        Binary feedback (0/1) for each trial.
    blocks : array-like of int
        Block index per trial; states reset across blocks.
    set_sizes : array-like of int
        Set size (number of distinct states in block) per trial.
    model_parameters : iterable of 6 floats
        lr : RL learning rate in [0,1].
        wm_weight : Mixture weight for WM vs RL (0..1); larger = more WM.
        softmax_beta : Inverse temperature for RL policy; gets scaled by 10 inside.
        wm_recency : Strength of adding the most recent chosen action into WM (0..1).
        wm_interference : Rate of WM decay towards uniform per trial, scaled by set size (0..1).
        perseveration : Bias added to the WM logits favoring the last chosen action in that state (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_recency, wm_interference, perseveration = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Track the last action taken in each state for perseveration (initialize to -1 = none)
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: chosen action probability
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy:
            # Interference scales with set size; more states => faster decay / more noise.
            decay_eff = wm_interference * (float(nS) / 3.0)
            decay_eff = np.clip(decay_eff, 0.0, 1.0)

            # Build logits that include WM content and perseveration to last action in that state
            beta_wm_eff = softmax_beta_wm  # kept high; interference handled by decay
            logits_wm = beta_wm_eff * W_s

            if last_action[s] >= 0:
                pa = np.zeros(nA)
                pa[last_action[s]] = 1.0
                logits_wm = logits_wm + perseveration * pa

            # Softmax for WM
            logits_wm = logits_wm - np.max(logits_wm)
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / np.sum(pi_wm)
            p_wm = pi_wm[a]

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # 1) Interference-driven decay towards uniform at the current state
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]

            # 2) Recency: push WM toward the most recent chosen action (task set maintenance)
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            w[s, :] = (1.0 - wm_recency) * w[s, :] + wm_recency * one_hot

            # 3) Optional small penalty on incorrectly reinforced actions: if r=0, slightly reduce chosen prob
            if r < 0.5:
                w[s, a] = max(w[s, a] - 0.1 * wm_recency, 0.0)

            # 4) Renormalize
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

            # Update last action memory for perseveration
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with arbitration-like WM reliability and capacity-limited interference.

    The WM policy itself becomes less reliable as set size increases (capacity limit),
    and more relied upon when RL is uncertain (high-entropy RL policy). Mixture weight
    in the template is fixed; arbitration is implemented by making the WM policy itself a
    mixture between uniform and WM-softmax depending on RL uncertainty.

    Parameters
    ----------
    states : array-like of int
        State indices on each trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen actions on each trial (0..nA-1).
    rewards : array-like of int
        Binary feedback (0/1) for each trial.
    blocks : array-like of int
        Block index per trial; states reset across blocks.
    set_sizes : array-like of int
        Set size (number of distinct states in block) per trial.
    model_parameters : iterable of 6 floats
        lr : RL learning rate in [0,1].
        wm_weight : Mixture weight for WM vs RL (0..1); larger = more WM.
        softmax_beta : Inverse temperature for RL policy; gets scaled by 10 inside.
        wm_decay : Base WM decay rate towards uniform each trial (0..1).
        wm_learn : WM learning strength toward rewarded action (0..1).
        arb_sensitivity : Sensitivity converting RL policy entropy into WM reliability (>=0).

    Returns
    -------
    nll : float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_learn, arb_sensitivity = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy for chosen action
            # Also compute full RL policy to get entropy (uncertainty)
            qlogits = softmax_beta * Q_s
            qlogits = qlogits - np.max(qlogits)
            pi_rl_vec = np.exp(qlogits)
            pi_rl_vec = pi_rl_vec / np.sum(pi_rl_vec)
            p_rl = pi_rl_vec[a]

            # RL entropy (0 to log(nA)); higher means more uncertainty
            H_rl = -np.sum(pi_rl_vec * (np.log(pi_rl_vec + eps)))

            # WM policy:
            # Capacity-induced noise: decay increases with set size (more items => more interference)
            decay_eff = np.clip(wm_decay * (float(nS) / 3.0), 0.0, 1.0)
            beta_wm_eff = softmax_beta_wm  # base determinism; reliability will be mixed with uniform below

            logits_wm = beta_wm_eff * W_s
            logits_wm = logits_wm - np.max(logits_wm)
            pi_wm_det = np.exp(logits_wm)
            pi_wm_det = pi_wm_det / np.sum(pi_wm_det)

            # Arbitration on WM reliability: blend WM-softmax with uniform based on RL entropy
            # Map entropy (0..log nA) to [0,1] via a sigmoid around half max entropy
            H_max = np.log(nA)
            centered = (H_rl - 0.5 * H_max)
            g = 1.0 / (1.0 + np.exp(-arb_sensitivity * centered))  # higher when RL is uncertain
            pi_wm = g * pi_wm_det + (1.0 - g) * (1.0 / nA) * np.ones(nA)
            p_wm = pi_wm[a]

            # Mixture of WM and RL
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # 1) Global capacity-limited decay for current state
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]

            # 2) On reward, store the chosen action in WM (one-shot)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * one_hot
            else:
                # On no reward, slightly flatten WM at this state
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

            # 3) Renormalize
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p
