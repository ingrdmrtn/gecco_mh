def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with capacity scaling and WM decay/overwrite.

    Model overview:
    - Two systems contribute to choice:
      1) Reinforcement Learning (RL): tabular Q-learning with softmax choice.
      2) Working Memory (WM): fast, one-shot storage of the last correct action per state,
         with decay toward a uniform prior and partial overwrite.
    - Mixture weight for WM is scaled by set size via a capacity parameter K, so WM is
      more influential at set size 3 than at 6.

    Parameters (6 total):
    - lr: RL learning rate in [0,1]
    - wm_weight: Base mixture weight for WM in [0,1] (before capacity scaling)
    - softmax_beta: Inverse temperature for RL softmax (will be scaled by 10 internally)
    - wm_decay: Per-trial decay of WM toward a uniform baseline in [0,1]
    - wm_capacity_K: WM capacity parameter; effective WM weight scales as min(1, K / set_size)
    - wm_replace: One-shot overwrite strength for WM when rewarded, in [0,1]
                  If reward=1, WM moves toward a one-hot at chosen action with this strength.
                  If reward=0, WM slightly suppresses the chosen action by this factor.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity_K, wm_replace = model_parameters
    softmax_beta *= 10.0  # higher upper bound for RL inverse temperature
    softmax_beta_wm = 50.0  # very deterministic WM policy
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables and baseline
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: probability of chosen action a under softmax
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over WM values with very high beta (near-deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Effective WM weight scaled by set size via capacity K
            nS_curr = int(block_set_sizes[t])
            cap_scale = min(1.0, wm_capacity_K / max(1.0, float(nS_curr)))
            wm_weight_eff = np.clip(wm_weight * cap_scale, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0 - eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward baseline then overwrite/suppress
            # Decay for the current state's WM row
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            if r >= 0.5:
                # Move WM toward a one-hot for the chosen action
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_replace) * w[s, :] + wm_replace * one_hot
            else:
                # If incorrect, suppress the chosen action a bit toward the baseline
                w[s, a] = (1.0 - wm_replace) * w[s, a] + wm_replace * w_0[s, a]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with capacity scaling, WM decay, and undirected lapses that increase with set size.

    Model overview:
    - RL system: tabular Q-learning with softmax.
    - WM system: fast storage of the last correct action per state with decay to uniform.
    - Mixture weight for WM scales with set size via capacity K.
    - Additionally, a set-size-dependent lapse probability injects uniform-random choices:
        p_total = (1 - lapse) * [wm_mix * p_wm + (1 - wm_mix) * p_rl] + lapse * (1/nA)
      where lapse increases with set size (higher at 6 than 3) controlled by a single parameter.

    Parameters (6 total):
    - lr: RL learning rate in [0,1]
    - wm_weight: Base mixture weight for WM in [0,1] (before capacity scaling)
    - softmax_beta: Inverse temperature for RL softmax (scaled by 10 internally)
    - wm_decay: Per-trial WM decay toward uniform in [0,1]
    - wm_capacity_K: WM capacity parameter; effective WM weight scales as min(1, K / set_size)
    - lapse_beta: Controls how strongly lapses increase with set size; lapse = sigmoid(lapse_beta) * ((set_size - 3) / 3)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity_K, lapse_beta = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    # Precompute sigmoid of lapse_beta to bound in (0,1)
    lapse_amp = 1.0 / (1.0 + np.exp(-lapse_beta))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WM mixture weight with capacity scaling
            nS_curr = int(block_set_sizes[t])
            cap_scale = min(1.0, wm_capacity_K / max(1.0, float(nS_curr)))
            wm_weight_eff = np.clip(wm_weight * cap_scale, 0.0, 1.0)

            # Lapse probability increases with set size (0 at 3, up to lapse_amp at 6)
            lapse = lapse_amp * max(0.0, (nS_curr - 3.0) / 3.0)
            lapse = np.clip(lapse, 0.0, 1.0)

            # Mixture with lapses
            mix_prob = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * mix_prob + lapse * (1.0 / nA)
            p_total = np.clip(p_total, eps, 1.0 - eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay then overwrite on correct, slight suppression on incorrect
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r >= 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot  # full overwrite on correct to keep model parsimonious
            else:
                # Gentle push back toward uniform for the chosen action when incorrect
                w[s, a] = 0.5 * w[s, a] + 0.5 * w_0[s, a]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with soft gating by set size (logistic gating) and WM decay.

    Model overview:
    - RL: tabular Q-learning with softmax.
    - WM: fast, one-shot-like memory with decay to uniform.
    - WM contribution is gated by a logistic function of set size:
        gate_prob = sigmoid(g0 + g1 * (3 - set_size))
      so that WM is more likely to contribute at set size 3 than 6.
      Effective mixture weight = wm_weight * gate_prob.

    Parameters (6 total):
    - lr: RL learning rate in [0,1]
    - wm_weight: Base WM mixture weight in [0,1]; scaled by gate_prob
    - softmax_beta: RL inverse temperature (scaled by 10 internally)
    - wm_decay: Per-trial WM decay toward uniform in [0,1]
    - g0: Gate bias parameter (set-size-independent component)
    - g1: Gate slope parameter (how strongly set size modulates gating)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, g0, g1 = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Soft gating based on set size (higher for small set sizes)
            nS_curr = int(block_set_sizes[t])
            gate_arg = g0 + g1 * (3.0 - float(nS_curr))
            gate_prob = 1.0 / (1.0 + np.exp(-gate_arg))
            wm_weight_eff = np.clip(wm_weight * gate_prob, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0 - eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay then overwrite on correct, partial reset on incorrect
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r >= 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot
            else:
                # If incorrect, clear some confidence by moving halfway toward uniform
                w[s, :] = 0.5 * w[s, :] + 0.5 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
