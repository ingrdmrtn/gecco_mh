def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with capacity-limited WM and decay toward uniform.
    - RL: single learning rate (lr) and softmax inverse temperature (softmax_beta).
    - WM: item-based store per state with one-shot storage for rewarded actions,
          elimination for unrewarded actions, and decay toward uniform (wm_decay).
    - Mixture weight depends on set size via a capacity parameter K:
      effective_wm_weight = wm_weight * min(1, K / nS). Thus, larger set sizes reduce WM influence.

    Parameters (tuple):
    - lr: scalar in [0,1], RL learning rate for Q-values.
    - wm_weight: scalar in [0,1], base weight on WM policy in the choice mixture.
    - softmax_beta: scalar >= 0, RL inverse temperature (internally scaled by 10).
    - K: positive scalar, effective WM capacity (number of items that can be strongly maintained).
    - wm_decay: scalar in [0,1], per-decision decay of WM values toward uniform (applied on visited state).

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, K, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM when strong memory is present

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent WM weight via capacity
        cap_factor = min(1.0, float(K) / float(nS))
        wm_w_eff = np.clip(wm_weight * cap_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Decay WM on current state toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # RL policy
            Q_s = q[s, :].copy()
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy (softmax over WM weights)
            W_s = w[s, :].copy()
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update
            if r > 0.5:
                # Store the correct action as a strong item (one-shot)
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Eliminate the incorrect action from WM beliefs for this state
                w[s, a] = 0.0
                if np.all(w[s, :] <= eps):
                    # If everything got eliminated, reset to uniform
                    w[s, :] = w_0[s, :].copy()
                else:
                    # Renormalize remaining mass
                    w[s, :] = w[s, :] / max(np.sum(w[s, :]), eps)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL (dual learning rates) + deterministic WM (win-stay/lose-eliminate) with set-size-dependent mixture via logistic.
    - RL: separate learning rates for positive and negative prediction errors (alpha_pos, alpha_neg),
          softmax inverse temperature (softmax_beta).
    - WM: per-state cache with one-shot storage on reward and elimination on non-reward; decay toward uniform.
    - Mixture weight depends on set size via a logistic function of set size:
        wm_w_eff = 1 / (1 + exp(-(wm_bias + wm_slope * (s0 - nS))))
      where s0 is a fixed anchor (here 4.5), so that larger nS tends to reduce WM weight when wm_slope > 0.

    Parameters (tuple):
    - alpha_pos: RL learning rate for positive prediction errors (in [0,1]).
    - alpha_neg: RL learning rate for negative prediction errors (in [0,1]).
    - softmax_beta: RL inverse temperature (internally scaled by 10).
    - wm_bias: intercept for the logistic mapping of WM weight.
    - wm_slope: slope for set-size dependence in the logistic WM weight.
    - wm_decay: scalar in [0,1], decay of WM toward uniform on the visited state.

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    alpha_pos, alpha_neg, softmax_beta, wm_bias, wm_slope, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    eps = 1e-12
    blocks_log_p = 0.0
    s0 = 4.5  # anchor between 3 and 6 to allow monotonic set-size effect

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Logistic WM weight as a function of set size
        z = wm_bias + wm_slope * (s0 - float(nS))
        wm_w_eff = 1.0 / (1.0 + np.exp(-z))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Decay WM on visited state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # RL policy
            Q_s = q[s, :].copy()
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            W_s = w[s, :].copy()
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with dual learning rates
            pe = r - q[s, a]
            if pe >= 0:
                q[s, a] += alpha_pos * pe
            else:
                q[s, a] += alpha_neg * pe

            # WM update: win-store, lose-eliminate
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, a] = 0.0
                if np.all(w[s, :] <= eps):
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] = w[s, :] / max(np.sum(w[s, :]), eps)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM mixture with set-size-scaled WM weight and choice perseveration.
    - RL: single learning rate and softmax inverse temperature; includes a stickiness bias (kappa)
          toward repeating the last action taken in the same state.
    - WM: decaying cache as before (win-store, lose-eliminate with decay to uniform).
    - Set-size effect on WM weight via a power-law scaling: wm_w_eff = wm_weight * (nS ** -gamma),
      clipped to [0,1], so WM de-emphasized more strongly as set size increases when gamma > 0.

    Parameters (tuple):
    - lr: RL learning rate in [0,1].
    - wm_weight: base WM mixture weight in [0,1].
    - gamma: positive scalar controlling how quickly WM weight decays with set size.
    - wm_decay: scalar in [0,1], WM decay toward uniform on the visited state.
    - softmax_beta: RL inverse temperature (internally scaled by 10).
    - kappa: stickiness parameter added to the value of the last chosen action in the same state.

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, gamma, wm_decay, softmax_beta, kappa = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last action per state for stickiness
        last_action = -1 * np.ones(nS, dtype=int)

        # Set-size dependent WM weight via power-law
        wm_w_eff = wm_weight * (float(nS) ** (-gamma))
        wm_w_eff = float(np.clip(wm_w_eff, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Decay WM for current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # RL policy with stickiness
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += kappa
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy
            W_s = w[s, :].copy()
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Mixture
            p_total = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, a] = 0.0
                if np.all(w[s, :] <= eps):
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] = w[s, :] / max(np.sum(w[s, :]), eps)

            # Update last action for stickiness
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p

Notes on set-size effects:
- Model1: WM influence scales with min(1, K/nS), modeling limited capacity; larger sets reduce WM contribution.
- Model2: WM influence decreases with set size via a logistic function parameterized by wm_bias and wm_slope.
- Model3: WM influence scales as nS^-gamma; larger sets reduce WM contribution more strongly for larger gamma.

All models reset learning within blocks, as required by the task structure.
