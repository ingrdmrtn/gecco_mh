def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with slot-limited capacity and decay (mixture policy).

    Idea:
    - Choices are a mixture of model-free RL and a fast, capacity-limited working memory (WM).
    - WM encodes state-action pairs quickly when rewarded, but decays toward uniform.
    - WM mixture weight is reduced when set size grows beyond a capacity parameter.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1)
    - wm_weight: baseline mixture weight on WM (0..1)
    - softmax_beta: inverse temperature for RL softmax (scaled up internally)
    - wm_decay: decay rate of WM toward uniform each time the state is encountered (0..1)
    - wm_learn: one-shot WM learning rate when reward is received (0..1)
    - wm_capacity: effective WM capacity in number of items (e.g., 1..6). Reduces WM influence when nS > capacity.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_learn, wm_capacity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-adjusted WM weight for this block
        cap_scale = min(1.0, wm_capacity / max(1.0, float(nS)))
        wm_weight_block = np.clip(wm_weight * cap_scale, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: probability of chosen action a under softmax
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy: softmax of W_s with very high beta (almost argmax). If W_s is uniform, this becomes ~uniform.
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture of WM and RL
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # WM decays toward uniform whenever the state is visited
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, one-shot encode the chosen action into WM
            if r > 0.0:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with uncertainty-weighted mixture and set-size modulation.

    Idea:
    - Choices are a mixture of RL and WM.
    - WM is fast but decays; RL learns gradually.
    - The mixture weight increases with WM certainty on the current state (how peaked W_s is),
      and decreases with set size via a power-law.
    - This links WM contribution to both load (set size) and local memory strength.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: baseline WM weight (0..1)
    - softmax_beta: RL inverse temperature (scaled up internally)
    - wm_decay: WM decay toward uniform per state visit (0..1)
    - gamma: set-size sensitivity exponent; higher gamma reduces WM weight more at larger set sizes (>=0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, gamma = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size modulation factor: reference set size 6
        ss_factor = (max(1.0, float(nS)) / 6.0) ** (-max(0.0, gamma))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WM certainty for current state: how peaked is W_s
            # Range ~[0,1]: 0 when uniform, 1 when one-hot
            wm_certainty = max(0.0, np.max(W_s) - np.min(W_s))
            wm_weight_t = np.clip(wm_weight * ss_factor * wm_certainty, 0.0, 1.0)

            p_total = wm_weight_t * p_wm + (1.0 - wm_weight_t) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform on each state visit
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, strengthen chosen action in WM (fast imprint)
            if r > 0.0:
                # Use a strong push proportional to certainty gap complement
                gap = 1.0 - wm_certainty
                target = np.zeros(nA)
                target[a] = 1.0
                imprint = np.clip(0.5 + 0.5 * gap, 0.0, 1.0)  # more imprint when current memory is uncertain
                w[s, :] = (1.0 - imprint) * w[s, :] + imprint * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with PE-gated WM encoding and load-dependent WM interference.

    Idea:
    - WM encoding is gated by the magnitude of the RL prediction error (PE) on rewarded trials:
      larger surprise -> stronger WM storage.
    - WM is interfered by set size: more items -> more interference, reducing WM mixture weight.
    - Choices mix WM and RL policies.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: baseline WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature (scaled up internally)
    - gate_slope: slope of the logistic gating on |PE| (higher -> more sensitive gating)
    - interference: load-driven reduction in WM weight per extra item beyond 3 (0..1)
    - wm_learn: base strength for WM encoding when gated in (0..1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, gate_slope, interference, wm_learn = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM weight reduction relative to a 3-item baseline
        load_factor = max(0.0, 1.0 - np.clip(interference, 0.0, 1.0) * max(0, nS - 3) / max(1, 3))
        wm_weight_block = np.clip(wm_weight * load_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy (deterministic read-out of W_s)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Interference-like leak toward uniform upon each visit
            # Leak scaled slightly by set size to reflect more interference at higher load
            leak = np.clip(interference * (nS / 6.0), 0.0, 1.0)
            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

            # Gated WM encoding: only on rewarded trials, strength set by logistic on |PE|
            if r > 0.0:
                gate = 1.0 / (1.0 + np.exp(-gate_slope * abs(delta)))
                strength = np.clip(wm_learn * gate, 0.0, 1.0)
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - strength) * w[s, :] + strength * target

        blocks_log_p += log_p

    return -blocks_log_p
