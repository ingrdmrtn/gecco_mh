def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with capacity-scaling and WM decay.

    Idea:
    - Choices arise from a mixture of a model-free RL policy and a working-memory (WM) policy.
    - WM is a fast, one-shot encoder that stores rewarded S-A associations but decays toward a uniform baseline.
    - The effective WM contribution scales down with set size via a capacity parameter K (i.e., WM_weight_eff = wm_weight * min(1, K/nS)).

    Parameters (tuple):
    - lr: RL learning rate (0..1)
    - wm_weight: Base mixture weight for WM contribution (0..1)
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally; higher => more deterministic)
    - phi: WM decay toward baseline per trial (0..1)
    - eta_wm: WM encoding strength on rewarded trials (0..1)
    - K: WM capacity (in number of states; positive real). Effective WM weight is scaled by min(1, K/nS)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, phi, eta_wm, K = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM mixture weight scaled by capacity vs set size.
        cap_scale = min(1.0, max(0.0, K / float(nS)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Use capacity-scaled WM weight
            wm_weight_eff = wm_weight * cap_scale
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)
            wm_weight = wm_weight_eff  # ensure the template line uses the effective weight

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)  # numerical guard
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay WM toward baseline each trial
            w[s, :] = (1.0 - phi) * w[s, :] + phi * w_0[s, :]
            # On rewarded trials, encode chosen action with strength eta_wm (one-shot push toward one-hot)
            if r > 0.0:
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with WM strength controlled by recency/decay within each state (retrieval-weighted arbitration).

    Idea:
    - RL contributes via standard softmax over Q-values.
    - WM stores the most recently rewarded action for each state and decays in strength over time.
    - The mixture weight for WM is modulated on each trial by the current WM strength m_s for that state:
        wm_weight_eff(s,t) = wm_weight * m_s
      so that when memory is fresh/strong, WM dominates; when stale, RL takes over.
    - WM associative values also decay to baseline and are refreshed on rewarded trials.

    Parameters (tuple):
    - lr: RL learning rate (0..1)
    - wm_weight: Base mixture weight for WM contribution (0..1)
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally)
    - lam: Decay rate of WM strength m_s per encounter (0..1)
    - eta_wm: WM encoding strength that refreshes both association and strength on rewarded trials (0..1)
    - alpha_wm_decay: Decay toward baseline for WM associative values per encounter (0..1)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, lam, eta_wm, alpha_wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # WM strength per state in [0,1], reflects retrievability/recency
        m_strength = np.zeros(nS)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy from associative memory w
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Retrieval-weighted arbitration: scale wm_weight by current memory strength for this state
            wm_weight_eff = wm_weight * m_strength[s]
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)
            wm_weight = wm_weight_eff  # so the template mixture uses the state-dependent weight

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay WM associative values and memory strength for the encountered state
            w[s, :] = (1.0 - alpha_wm_decay) * w[s, :] + alpha_wm_decay * w_0[s, :]
            m_strength[s] = (1.0 - lam) * m_strength[s]

            # On rewarded trials: encode the chosen action strongly and refresh WM strength
            if r > 0.0:
                # Push association toward one-hot at chosen action
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm
                # Refresh memory strength upward toward 1
                m_strength[s] = m_strength[s] + eta_wm * (1.0 - m_strength[s])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning rates + capacity-limited WM slots.

    Idea:
    - RL uses separate learning rates for positive vs negative prediction errors.
    - WM stores up to C states (capacity) per block. If a state is in WM, its policy is near-deterministic.
    - A rewarded encounter attempts to place/update the state in WM. If capacity is exceeded, evict the oldest state (FIFO).
    - When the queried state s is in WM, the mixture gives WM more influence (via wm_weight); otherwise RL dominates.

    Parameters (tuple):
    - lr_pos: RL learning rate for positive prediction errors (0..1)
    - lr_neg: RL learning rate for negative prediction errors (0..1)
    - wm_weight: Mixture weight when the state is currently stored in WM (0..1)
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally)
    - C: WM capacity (number of states storable per block; >=0)
    - eta_wm: WM encoding strength applied to associative values on rewarded trials (0..1)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, C, eta_wm = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track which states are in WM (FIFO queue of state indices)
        wm_queue = []
        in_wm = np.zeros(nS, dtype=bool)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy uses associative vector; if state s is in WM, it will be sharp, else near-uniform baseline
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: if s is in WM, use full wm_weight; else downweight WM to 0
            wm_weight_eff = wm_weight if in_wm[s] else 0.0
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)
            wm_weight = wm_weight_eff  # to be used by the template mixture

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            lr_use = lr_pos if pe >= 0.0 else lr_neg
            q[s][a] += lr_use * pe

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # If rewarded, encode state-action in WM
            if r > 0.0:
                # Update associative values toward one-hot on chosen action
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm
                # Add/refresh state in WM FIFO
                if not in_wm[s]:
                    # Insert new state, evict if beyond capacity C
                    wm_queue.append(s)
                    in_wm[s] = True
                    # Enforce capacity
                    while C < len(wm_queue):
                        evict = wm_queue.pop(0)
                        in_wm[evict] = False
                        # When evicted, reset its WM association to baseline
                        w[evict, :] = w_0[evict, :]
                else:
                    # Refresh recency by moving s to the end of the queue
                    # Remove first occurrence and append
                    for idx_i, st in enumerate(wm_queue):
                        if st == s:
                            wm_queue.pop(idx_i)
                            break
                    wm_queue.append(s)

        blocks_log_p += log_p

    return -blocks_log_p
