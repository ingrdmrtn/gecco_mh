Below are three standalone cognitive models that combine reinforcement learning (RL) with working memory (WM). Each function follows the provided template structure, returns the negative log-likelihood of the observed choices, and uses all parameters meaningfully. The models differ in how WM contributes under varying set sizes and how WM is updated.

Note: Assume numpy (np) is already imported.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with capacity-sensitive gating and decay.

    Idea:
    - RL learns cached values per state-action via a standard delta rule.
    - WM stores a supervised, one-shot association for rewarded actions with decay toward uniform.
    - WM weight is reduced as set size increases (capacity pressure).

    Parameters (model_parameters):
    - lr: RL learning rate (0..1) for Q-values.
    - wm_weight: Baseline mixture weight for WM versus RL (0..1).
    - softmax_beta: Inverse temperature for RL policy (scaled internally by 10).
    - wm_decay: WM update/decay rate (0..1). Higher -> faster overwriting/decay.
    - wm_setsize_sensitivity: Scales how much WM weight drops with set size beyond 3.
                              Effective weight = wm_weight / (1 + wm_setsize_sensitivity * (nS - 3))

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_setsize_sensitivity = model_parameters

    softmax_beta *= 10.0  # higher dynamic range for RL
    softmax_beta_wm = 50.0  # very deterministic WM policy

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent WM weight (capacity pressure)
        wm_weight_eff = wm_weight / (1.0 + wm_setsize_sensitivity * max(0, nS - 3))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over WM weights (deterministic)
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Mixture
            p_total = p_wm * wm_weight_eff + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)  # numerical safety
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: rewarded -> move toward one-hot of chosen action; unrewarded -> decay to uniform
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * one_hot
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning + WM gated by state-specific reliability and set size.

    Idea:
    - RL uses separate learning rates for positive and negative outcomes.
    - WM stores one-shot associations with decay to uniform.
    - WM mixture weight is scaled by:
        (a) State-specific "reliability" (EMA of recent rewards in that state).
        (b) Set size factor (3/nS) to reflect capacity dilution.

    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive reward prediction errors (0..1).
    - lr_neg: RL learning rate for negative reward prediction errors (0..1).
    - wm_weight: Baseline WM mixture weight (0..1).
    - softmax_beta: Inverse temperature for RL policy (scaled internally by 10).
    - rel_rate: Exponential averaging rate for WM reliability per state (0..1). Higher -> faster adaptation.

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, rel_rate = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Initialize state-specific reliability (how confident WM is for that state)
        reliability = 0.5 * np.ones(nS)  # start uncertain

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Effective WM mixture weight: baseline * reliability_s * capacity factor
            capacity_factor = 3.0 / max(3.0, float(nS))
            wm_weight_eff = wm_weight * reliability[s] * capacity_factor

            p_total = p_wm * wm_weight_eff + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            lr_use = lr_pos if pe >= 0 else lr_neg
            q[s, a] += lr_use * pe

            # WM update (same supervised rule with decay to uniform on errors)
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # stronger commitment on reward
                w[s, :] = (1.0 - rel_rate) * w[s, :] + rel_rate * one_hot
            else:
                w[s, :] = (1.0 - rel_rate) * w[s, :] + rel_rate * w_0[s, :]

            # Update state-specific reliability via EMA of reward
            reliability[s] = (1.0 - rel_rate) * reliability[s] + rel_rate * r

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with choice stickiness + WM availability governed by capacity (sigmoid) and WM temperature scaling.

    Idea:
    - RL uses standard delta learning, but the policy includes choice stickiness within-state
      by temporarily boosting the last chosen action's value at decision time.
    - WM stores one-shot associations; availability of WM is a sigmoid function of set size relative to capacity.
    - WM softmax temperature is scaled by a parameter to allow less-than-perfect WM.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight: Baseline WM mixture weight (0..1).
    - softmax_beta: Inverse temperature for RL policy (scaled internally by 10).
    - capacity_k: WM capacity threshold (in items) around which WM availability drops.
    - capacity_slope: Positive slope for the sigmoid controlling how sharply WM availability declines with set size.
    - stickiness: Choice perseveration bias added (temporarily) to the last chosen action's Q in that state.

    Returns:
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, capacity_k, capacity_slope, stickiness = model_parameters

    softmax_beta *= 10.0
    base_softmax_beta_wm = 50.0  # baseline; we will keep it but availability gates WM contribution

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last action per state for stickiness (-1 = none)
        last_action = -1 * np.ones(nS, dtype=int)

        # WM availability as sigmoid of set size relative to capacity
        # availability ~ 1 for nS << capacity_k, ~ 0 for nS >> capacity_k
        avail = 1.0 / (1.0 + np.exp(capacity_slope * (float(nS) - capacity_k)))
        wm_weight_eff = wm_weight * avail

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with choice stickiness: boost the last chosen action in this state
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stickiness

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with temperature and stored associations
            W_s = w[s, :]
            # Keep very deterministic WM but respect availability through mixture weight, not temperature
            softmax_beta_wm = base_softmax_beta_wm
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Mixture with availability-gated WM weight
            p_total = p_wm * wm_weight_eff + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM update: reward -> one-hot encoding; no reward -> decay to uniform
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # Use a moderate overwrite rate tied to availability (more availability -> stronger write)
                write_rate = min(1.0, 0.5 + 0.5 * avail)
                w[s, :] = (1.0 - write_rate) * w[s, :] + write_rate * one_hot
            else:
                # Decay rate inversely related to availability: with low WM availability, faster decay
                decay_rate = min(1.0, 0.5 + 0.5 * (1.0 - avail))
                w[s, :] = (1.0 - decay_rate) * w[s, :] + decay_rate * w_0[s, :]

            # Update last action for stickiness
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
