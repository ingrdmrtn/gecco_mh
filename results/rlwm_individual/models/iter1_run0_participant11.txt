def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with eligibility traces + WM fast mapping, arbitration penalized by load, and state-wise action stickiness.

    Idea:
    - RL uses eligibility traces (lambda) within a state to spread credit to recently chosen actions.
    - WM learns rapidly from reward and decays toward uniform; it's more influential in small set sizes.
    - Arbitration weight is scaled by load factor (3 / nS).
    - RL policy includes a state-wise stickiness bias toward the last action chosen in that state.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - lam: Eligibility trace decay (0..1), larger spreads credit over longer horizons within a state
    - wm_weight: Base working-memory weight mixed with RL, scaled by load (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - wm_decay: WM decay toward uniform on each update step (0..1)
    - stickiness: Bias added to the last action taken in the same state (>0 favors repeating) (0..inf)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, lam, wm_weight, softmax_beta, wm_decay, stickiness = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Eligibility traces and last action per state
        e = np.zeros((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy with stickiness bias toward last action in this state
            bias = np.zeros(nA)
            if last_action[s] >= 0:
                bias[last_action[s]] = stickiness
            logits_rl = Q_s + bias
            # Compute prob of chosen action under RL softmax
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (logits_rl - logits_rl[a])))

            # WM policy: deterministic softmax over WM weights
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: load penalty (more WM weight in small sets)
            load_scale = min(1.0, 3.0 / float(nS))
            wm_weight_eff = wm_weight * load_scale
            wm_weight_eff = max(0.0, min(1.0, wm_weight_eff))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with eligibility trace
            delta = r - Q_s[a]
            # Decay traces within the same state; replacing trace for chosen action
            e[s, :] *= lam
            e[s, a] = 1.0
            q[s, :] += lr * delta * e[s, :]

            # WM update: reward drives one-shot encoding; non-reward decays to uniform
            if r > 0.5:
                w[s, :] = (1.0 - wm_decay) * w[s, :]
                w[s, a] += wm_decay
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Normalize WM row for numerical stability
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

            # Update last action for stickiness
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Surprise-gated WM arbitration + RL, with WM encoding and load-induced interference.

    Idea:
    - RL updates with a single learning rate.
    - WM encodes rewarded action via wm_alpha; non-reward induces decay to uniform.
    - Arbitration weight depends on unsigned prediction error (Bayesian surprise proxy),
      via a sigmoid with parameter pe_scale, and a base weight wm_weight.
    - WM suffers interference that increases with set size beyond 3, controlled by intf.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: Base WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - wm_alpha: WM encoding/decay step size (0..1)
    - pe_scale: Scales influence of unsigned prediction error on WM gating (>0)
    - intf: WM interference factor with load; mixes WM with uniform as set size grows (0..1)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_alpha, pe_scale, intf = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last unsigned PE per state to gate WM on subsequent encounters
        last_abs_pe = np.zeros(nS)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM effective representation under load-induced interference
            load_excess = max(0.0, (float(nS) - 3.0) / 3.0)  # 0 at nS<=3, up to 1 at nS=6
            rho = max(0.0, min(1.0, intf * load_excess))
            W_eff = (1.0 - rho) * W_s + rho * w_0[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            # Arbitration: sigmoid of last unsigned PE at this state
            # wm_gate in [0,1], increases with surprise
            wm_gate = 1.0 / (1.0 + np.exp(-pe_scale * last_abs_pe[s]))
            wm_weight_eff = wm_weight * wm_gate
            wm_weight_eff = max(0.0, min(1.0, wm_weight_eff))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # Compute current PE (pre-update) and store abs PE for next time
            delta = r - Q_s[a]
            last_abs_pe[s] = abs(delta)

            # RL update
            q[s, a] += lr * delta

            # WM update: encode with wm_alpha on reward; decay to uniform on non-reward
            if r > 0.5:
                w[s, :] = (1.0 - wm_alpha) * w[s, :]
                w[s, a] += wm_alpha
            else:
                w[s, :] = (1.0 - wm_alpha) * w[s, :] + wm_alpha * w_0[s, :]

            # Normalize WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Count-based exploratory RL + capacity-limited WM with decay; arbitration by effective capacity.

    Idea:
    - RL policy adds a count-based exploration bonus to under-sampled actions (optimistic exploration).
    - WM stores rewarded mappings with decay; effective maintenance is limited by a capacity C
      relative to set size, which also scales arbitration weight.
    - As nS grows beyond capacity, WM representations are pulled toward uniform (capacity sharing).

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - eta: Exploration bonus strength; bonus ~ eta / sqrt(1 + N_sa) added to RL logits (>=0)
    - wm_weight: Base WM mixture weight, scaled by effective capacity (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - capacity_C: WM capacity in number of items (0..6); arbitration scales by min(1, C/nS)
    - wm_decay: WM decay toward uniform on each trial/update (0..1)

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, eta, wm_weight, softmax_beta, capacity_C, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Visit counts for count-based exploration
        n_sa = np.zeros((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy with count-based exploration bonus on logits
            bonus = eta / np.sqrt(1.0 + n_sa[s, :])
            logits_rl = Q_s + bonus
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (logits_rl - logits_rl[a])))

            # WM policy: capacity sharing pulls toward uniform as capacity is exceeded
            cap_scale = min(1.0, max(0.0, float(capacity_C)) / max(1.0, float(nS)))
            W_eff = cap_scale * W_s + (1.0 - cap_scale) * w_0[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            # Arbitration weighted by effective capacity
            wm_weight_eff = wm_weight * cap_scale
            wm_weight_eff = max(0.0, min(1.0, wm_weight_eff))

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update and increment counts
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            n_sa[s, a] += 1.0

            # WM update: decay toward uniform, then boost chosen action on reward
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                # Add an extra push to the rewarded action, then renormalize
                w[s, a] += wm_decay
            # Normalize WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
