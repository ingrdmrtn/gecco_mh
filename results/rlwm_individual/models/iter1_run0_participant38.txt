def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with set-size-specific WM decay and weight scaling.

    Idea:
    - RL: single learning rate with softmax choice rule.
    - WM: per-state policy table that decays toward uniform each trial.
      - Rewarded trials push WM toward a one-hot policy favoring the chosen action.
      - Unrewarded trials push WM away from the chosen action (redistribute mass to the others).
    - Set size affects:
      - WM decay rate via separate parameters for 3 vs 6 (wm_decay_3 vs wm_decay_6).
      - Effective WM mixture weight via a 3/nS scaling (less WM influence at nS=6).

    Parameters
    - model_parameters[0] = lr (float in [0,1]): RL learning rate for Q-values.
    - model_parameters[1] = wm_weight_base (float in [0,1]): Base WM mixture weight (scaled by 3/nS per block).
    - model_parameters[2] = softmax_beta (float > 0): Inverse temperature for RL softmax (internally scaled by 10).
    - model_parameters[3] = wm_decay_3 (float in [0,1]): WM decay rate toward uniform for set size 3.
    - model_parameters[4] = wm_decay_6 (float in [0,1]): WM decay rate toward uniform for set size 6.
    - model_parameters[5] = wm_gain (float in [0,1]): Strength of WM update toward/away from the chosen action.

    Returns
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight_base, softmax_beta, wm_decay_3, wm_decay_6, wm_gain = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size effects
        wm_decay = wm_decay_3 if nS == 3 else wm_decay_6
        wm_w_block = wm_weight_base * (3.0 / nS)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of the chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WORKING MEMORY POLICY: softmax over WM row (deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_w_block * p_wm + (1.0 - wm_w_block) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update: reward -> one-hot toward chosen; no-reward -> push away from chosen
            if r > 0.5:
                # Pull toward one-hot at chosen action
                w[s, :] = (1.0 - wm_gain) * w[s, :]
                w[s, a] += wm_gain  # add mass to chosen
            else:
                # Reduce chosen action probability; redistribute to others
                dec = min(wm_gain, w[s, a])
                w[s, a] -= dec
                # Distribute removed mass equally among others
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += dec / (nA - 1)

            # Ensure numerical stability: renormalize row
            row_sum = np.sum(w[s, :])
            if row_sum > 1e-12:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with surprise-modulated learning + WM one-shot binding with set-size-specific precision.

    Idea:
    - RL: learning rate is dynamically scaled by unsigned prediction error (Pearce-Hall style):
      lr_t = clip(lr_base + kappa * |delta|, 0, 1).
    - WM: on rewarded trials, store a one-hot action for that state (fast mapping);
      on unrewarded trials, mildly suppress the chosen action.
    - Set size affects WM precision (beta_wm) with separate parameters for 3 vs 6.
      WM mixture weight is constant; precision reduction at set size 6 makes WM less decisive.

    Parameters
    - model_parameters[0] = lr_base (float in [0,1]): Baseline RL learning rate.
    - model_parameters[1] = kappa (float >= 0): Surprise gain scaling |delta| into additional learning.
    - model_parameters[2] = wm_weight (float in [0,1]): Mixture weight for WM vs RL policy.
    - model_parameters[3] = softmax_beta (float > 0): RL inverse temperature (internally scaled by 10).
    - model_parameters[4] = wm_beta_3 (float > 0): WM inverse temperature at set size 3.
    - model_parameters[5] = wm_beta_6 (float > 0): WM inverse temperature at set size 6.

    Returns
    - Negative log-likelihood of the observed choices under the model.
    """
    lr_base, kappa, wm_weight, softmax_beta, wm_beta_3, wm_beta_6 = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # placeholder; will be overridden per block below

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size-specific WM precision
        softmax_beta_wm = wm_beta_3 if nS == 3 else wm_beta_6

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update with surprise-modulated lr
            delta = r - Q_s[a]
            lr_t = lr_base + kappa * abs(delta)
            lr_t = min(max(lr_t, 0.0), 1.0)
            q[s][a] += lr_t * delta

            # WM update:
            # No explicit decay parameter here; keep last memory unless overwritten by outcomes
            if r > 0.5:
                # Store one-hot for the rewarded action
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Suppress chosen action slightly; move mass to others
                dec = min(0.5, 0.25)  # fixed mild suppression; keep stable
                dec = 0.25
                dec = min(dec, w[s, a])
                w[s, a] -= dec
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += dec / (nA - 1)

            # Renormalize for safety
            row_sum = np.sum(w[s, :])
            if row_sum > 1e-12:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with Q-value forgetting + WM with set-size-dependent interference.

    Idea:
    - RL: standard delta rule plus forgetting toward uniform (helps when set size is large/volatile).
    - WM: per-state policy that suffers global interference proportional to set size:
      all WM rows are nudged toward their grand average each trial, then the current state's row is
      adjusted toward the chosen action (rewarded) or away (unrewarded).
    - Set size affects:
      - WM interference strength via a multiplicative factor proportional to (nS-1)/nS.
      - Mixture weight is constant; interference degrades WM more for larger sets.

    Parameters
    - model_parameters[0] = lr (float in [0,1]): RL learning rate.
    - model_parameters[1] = q_forget (float in [0,1]): Forgetting rate pulling Q toward uniform each trial.
    - model_parameters[2] = softmax_beta (float > 0): RL inverse temperature (internally scaled by 10).
    - model_parameters[3] = wm_weight (float in [0,1]): Mixture weight for WM policy.
    - model_parameters[4] = wm_interf (float in [0,1]): Base WM interference rate (scaled by (nS-1)/nS).
    - model_parameters[5] = wm_gain (float in [0,1]): Strength of WM update toward/away from chosen action in current state.

    Returns
    - Negative log-likelihood of the observed choices under the model.
    """
    lr, q_forget, softmax_beta, wm_weight, wm_interf, wm_gain = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size-dependent interference factor
        interf_factor = (nS - 1.0) / nS
        wm_interf_block = wm_interf * interf_factor

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update with forgetting toward uniform
            delta = r - Q_s[a]
            q[s][a] += lr * delta
            # Forgetting for the active state s (can be thought of as leakage each visit)
            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * w_0[s, :]

            # WM global interference: rows nudge toward their grand average
            mean_row = np.mean(w, axis=0)
            w = (1.0 - wm_interf_block) * w + wm_interf_block * np.tile(mean_row, (nS, 1))

            # WM update for current state
            if r > 0.5:
                # Move current state's WM toward one-hot at chosen action
                w[s, :] = (1.0 - wm_gain) * w[s, :]
                w[s, a] += wm_gain
            else:
                # Push away from chosen action, redistribute to others
                dec = min(wm_gain, w[s, a])
                w[s, a] -= dec
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += dec / (nA - 1)

            # Normalize current state's WM row for stability
            row_sum = np.sum(w[s, :])
            if row_sum > 1e-12:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p
