def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RLWM model with capacity-limited, decaying one-shot working memory and RL integration.
    
    Policy: Mixture of RL softmax and WM softmax.
    - RL: tabular Q-learning with single learning rate and inverse temperature.
    - WM: one-shot storage of rewarded action for each state (deterministic policy),
          with trial-wise decay toward uniform and effective WM weight reduced by set size.

    Parameters
    - lr: scalar in [0,1], learning rate for RL Q-learning.
    - wm_weight: base mixture weight for WM (0..1). The effective WM weight is
                 wm_weight * min(1, wm_capacity / set_size) (reduced under higher load).
    - softmax_beta: inverse temperature for RL policy; internally scaled by 10 for a wide range.
    - wm_decay: decay (0..1) of WM table toward uniform each trial; higher means faster forgetting.
    - wm_capacity: capacity (in items) controlling how set size modulates the WM influence.

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, wm_capacity = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight for this block based on capacity and set size
        wm_weight_eff_block = wm_weight_base * min(1.0, wm_capacity / max(1.0, float(nS)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Compute RL choice probability for chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # Working memory policy:
            # - Before policy eval, apply global decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Compute WM probability for chosen action
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture
            p_total = p_wm * wm_weight_eff_block + (1.0 - wm_weight_eff_block) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update
            delta_q = r - Q_s[a]
            q[s, a] += lr * delta_q

            # WM update: one-shot storage on rewarded trials
            # If reward, store the chosen action deterministically for this state
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            # If no reward, we rely on decay and leave the current WM trace as is

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RLWM model with parallel RL and noisy WM-Q, with decay and set-size-dependent WM noise.
    
    Policy: Mixture of RL softmax and WM softmax.
    - RL: tabular Q-learning with single learning rate and inverse temperature.
    - WM: separate value table with its own learning rate and decay, acting as a fast-learning system.
          Its inverse temperature decreases as set size increases.

    Parameters
    - lr: scalar in [0,1], RL learning rate for Q-learning.
    - wm_weight: base mixture weight for WM (0..1). Used uniformly within a block.
    - softmax_beta: inverse temperature for RL policy; internally scaled by 10.
    - lr_wm: scalar in [0,1], WM learning rate for its value table.
    - wm_decay: decay (0..1) of WM value table toward uniform each trial; higher means faster forgetting.
    - wm_noise_scale: nonnegative; increases WM stochasticity with set size via:
                      beta_wm_eff = softmax_beta_wm / (1 + wm_noise_scale * (set_size - 1)).

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, lr_wm, wm_decay, wm_noise_scale = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm_base = 50.0  # high determinism at small set size
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Value tables
        q = (1.0 / nA) * np.ones((nS, nA))  # RL
        w = (1.0 / nA) * np.ones((nS, nA))  # WM values
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent WM temperature
        beta_wm_eff = softmax_beta_wm_base / (1.0 + wm_noise_scale * max(0.0, float(nS) - 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy probability for chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # Apply decay to WM values toward uniform before policy eval
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM policy probability for chosen action
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update
            delta_q = r - Q_s[a]
            q[s, a] += lr * delta_q

            # WM update: fast value learning with its own learning rate
            delta_w = r - W_s[a]
            w[s, a] += lr_wm * delta_w

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WSLS working-memory heuristic with set-size-modulated RL learning rates.
    
    Policy: Mixture of RL softmax and WM softmax.
    - RL: tabular Q-learning with separate learning rates for positive vs negative outcomes.
          Learning rates are reduced under higher set size.
    - WM: Win-stay/lose-shift heuristic stored per state: prefers repeating the last rewarded action,
          and avoids the last action if it was not rewarded. Implemented as a preference vector
          fed to a high-temperature softmax.

    Parameters
    - lr_pos: RL learning rate when reward=1.
    - lr_neg: RL learning rate when reward=0.
    - wm_weight: mixture weight (0..1) for the WM heuristic.
    - softmax_beta: inverse temperature for RL policy; internally scaled by 10.
    - wsls_bias: magnitude of WM preference for stay vs shift (>0 increases WM determinism).
    - setsize_lr_suppression: nonnegative; scales down both RL learning rates as:
                              lr_eff = lr * 1 / (1 + setsize_lr_suppression * (set_size - 1)).

    Returns
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, wsls_bias, setsize_lr_suppression = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic mapping from WM preferences to choices
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values
        q = (1.0 / nA) * np.ones((nS, nA))

        # WSLS memory per state
        last_action = -1 * np.ones(nS, dtype=int)
        last_reward = np.zeros(nS)

        log_p = 0.0
        # Set-size dependent scaling for RL learning rates
        lr_scale = 1.0 / (1.0 + setsize_lr_suppression * max(0.0, float(nS) - 1.0))

        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy probability for chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM WSLS preference vector for this state
            pref = np.zeros(nA)
            if last_action[s] >= 0:
                if last_reward[s] > 0.5:
                    # Win-stay: boost the last rewarded action
                    pref[last_action[s]] = wsls_bias
                else:
                    # Lose-shift: penalize the last action
                    pref[last_action[s]] = -wsls_bias
            # Softmax over preferences to get WM policy prob of chosen action
            denom_wm = np.sum(np.exp(softmax_beta_wm * (pref - pref[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update with valence-specific, set-size-scaled learning rate
            lr_eff = (lr_pos if r > 0.5 else lr_neg) * lr_scale
            delta_q = r - Q_s[a]
            q[s, a] += lr_eff * delta_q

            # Update WSLS memory
            last_action[s] = a
            last_reward[s] = r

        blocks_log_p += log_p

    return -blocks_log_p
