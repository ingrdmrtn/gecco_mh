Below are three standalone cognitive models that follow your template, differ in mechanisms and parameterizations, and return negative log-likelihoods. Each model uses all of its parameters and includes a WM component that behaves differently under set-size load.

Note: No imports are included; assume numpy as np is available.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 1: Load-gated WM with leak and state-specific perseveration bias.
    - RL: Q-learning with softmax.
    - WM: leaky store toward uniform each trial; stores rewarded action; policy includes perseveration bias to repeat the last action in the same state.
    - Arbitration: Base WM weight is multiplicatively gated by set size using a logistic gate.

    Parameters:
      lr:               RL learning rate (0..1)
      wm_weight0:       Base mixture weight for WM before gating (0..1)
      softmax_beta:     RL inverse temperature (scaled internally; >=0)
      wm_leakage:       WM leak toward uniform per encounter (0..1); also sets storage strength via 1-exp(-5*wm_leakage)
      wm_gate_slope:    Slope (>0) of the logistic load gate that reduces WM weight as set size increases
      perseveration:    Strength (>=0) of same-state perseveration bias added to WM policy for the last chosen action
    Returns:
      Negative log-likelihood of observed actions.
    """
    lr, wm_weight0, softmax_beta, wm_leakage, wm_gate_slope, perseveration = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = block_set_sizes[0]

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)  # for perseveration within-state

        # Logistic load gate: smaller WM weight when nS is large (center at 4.5)
        gate = 1.0 / (1.0 + np.exp(wm_gate_slope * (nS - 4.5)))
        wm_weight_eff = np.clip(wm_weight0 * gate, 0.0, 1.0)
        store_strength = 1.0 - np.exp(-5.0 * max(0.0, wm_leakage))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: add perseveration bias to last action in this state
            wm_logits = W_s.copy()
            if last_action[s] >= 0:
                wm_logits[last_action[s]] += perseveration

            # Softmax for WM policy with fixed high temperature
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (wm_logits - wm_logits[a])))

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM leak toward uniform, then store on reward
            w[s, :] = (1.0 - wm_leakage) * w[s, :] + wm_leakage * w_0[s, :]
            if r > 0:
                # Move distribution toward one-hot on chosen action
                w[s, :] = (1.0 - store_strength) * w[s, :]
                w[s, a] += store_strength
                # Normalize
                w[s, :] = np.clip(w[s, :], 1e-12, None)
                w[s, :] /= np.sum(w[s, :])

            # Update last action for perseveration
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 2: Asymmetric RL learning rates with load-attenuated WM weight and lapse.
    - RL: Q-learning with separate learning rates for positive vs negative prediction errors.
    - WM: one-shot store on reward with smoothing; small decay toward uniform on no-reward.
    - Arbitration: WM mixture weight decays exponentially with set size; overall policy includes a lapse to uniform.

    Parameters:
      lr_pos:        RL learning rate for positive prediction errors (0..1)
      lr_neg:        RL learning rate for negative prediction errors (0..1)
      wm_weight:     Base WM mixture weight before load attenuation (0..1)
      softmax_beta:  RL inverse temperature (scaled internally; >=0)
      gamma_load:    Load sensitivity (>0); larger values reduce WM influence more strongly at larger set sizes
      lapse:         Lapse probability (0..1): with this probability, choices are uniform; also used as WM smoothing
    Returns:
      Negative log-likelihood of observed actions.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, gamma_load, lapse = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = block_set_sizes[0]

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Exponential attenuation of WM weight with set size (0 at 3, stronger at 6)
        wm_weight_eff = wm_weight * np.exp(-gamma_load * max(0.0, nS - 3))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability for chosen action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy softmax (deterministic) over W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture of WM and RL, then lapse to uniform
            p_mix = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = (1 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = np.maximum(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            delta = r - Q_s[a]
            eta = lr_pos if delta >= 0 else lr_neg
            q[s][a] += eta * delta

            # WM update:
            # - On reward: store near one-hot with smoothing by 'lapse' to avoid zeros.
            # - On no reward: slight decay toward uniform (use same smoothing).
            if r > 0:
                w[s, :] = lapse * w_0[s, :]  # smoothing baseline
                w[s, a] += (1.0 - lapse)
            else:
                w[s, :] = (1.0 - lapse) * w[s, :] + lapse * w_0[s, :]
            # Normalize for safety
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 3: Entropy-gated WM, RL forgetting, and WM retrieval noise increasing with load.
    - RL: Q-learning with softmax and per-trial forgetting toward uniform.
    - WM: stores graded action preferences; retrieval is noisy and more so under larger set sizes.
    - Arbitration: WM weight is boosted when RL is uncertain (high entropy) and set size is small.

    Parameters:
      lr:               RL learning rate (0..1)
      wm_weight_base:   Base WM mixture weight (0..1)
      softmax_beta:     RL inverse temperature (scaled internally; >=0)
      q_forget:         RL forgetting rate toward uniform per encounter (0..1)
      gate_gain:        Strength (>=0) of entropyÃ—(low load) boost of WM weight; higher boosts WM when RL is uncertain and load is low
      wm_noise:         WM retrieval noise (0..1); blends WM values with uniform more strongly at higher set sizes
    Returns:
      Negative log-likelihood of observed actions.
    """
    lr, wm_weight_base, softmax_beta, q_forget, gate_gain, wm_noise = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic base (modulated by wm_noise below)
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = block_set_sizes[0]

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            # RL forgetting toward uniform before computing choice/update
            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * w_0[s, :]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL chosen-action probability
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # RL entropy for arbitration (normalized 0..1)
            rl_probs = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            rl_probs = rl_probs / np.sum(rl_probs)
            entropy = -np.sum(rl_probs * np.log(np.maximum(rl_probs, 1e-12))) / np.log(nA)

            # Load factor: 1 at high load (nS=6), 0 at low load (nS=3)
            load_high = max(0.0, (nS - 3) / 3.0)

            # WM weight boosted when entropy high and load low
            boost = gate_gain * entropy * (1.0 - load_high)
            wm_weight_eff = np.clip(wm_weight_base * (1.0 + boost), 0.0, 1.0)

            # WM retrieval noise increases with load: blend W with uniform
            noise_scale = wm_noise * (nS / 6.0)
            W_eff = (1.0 - noise_scale) * W_s + noise_scale * w_0[s, :]

            # Effective WM temperature reduced by retrieval noise (more noise -> lower beta)
            beta_wm_eff = softmax_beta_wm * max(1e-3, (1.0 - noise_scale))

            # WM chosen-action probability
            p_wm = 1 / np.sum(np.exp(beta_wm_eff * (W_eff - W_eff[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update:
            # - On reward: move distribution toward one-hot for chosen action (faster when less noise)
            # - On no reward: mild repulsion from chosen action (stronger when more noise)
            pos_rate = max(1e-3, 1.0 - noise_scale)  # faster consolidation at low load
            neg_rate = 0.5 * noise_scale            # more repulsion at high load

            if r > 0:
                w[s, :] = (1.0 - pos_rate) * w[s, :] + pos_rate * w_0[s, :]
                w[s, a] += pos_rate  # increase chosen action
            else:
                dec = min(neg_rate, w[s, a] - 1e-12)
                w[s, a] -= dec
                redistribute = dec / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += redistribute

            # Normalize
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
