def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with probabilistic availability and load-dependent forgetting.

    Idea
    - RL updates state-action Q-values via a single learning rate.
    - WM stores the most likely correct action per state when rewarded, but access is capacity-limited.
      In larger set sizes, WM is only probabilistically available (C/nS), otherwise devolves to uniform.
    - WM representation decays toward uniform at each trial (forgetting) and only encodes on rewarded trials.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for each trial.
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, wm_capacity, wm_forget)
        - lr: RL learning rate in [0,1]
        - wm_weight: mixture weight of WM policy vs RL policy in [0,1]
        - softmax_beta: RL inverse temperature (>0); internally scaled up
        - wm_capacity: effective WM capacity (in items), 0..6
        - wm_forget: WM decay toward uniform per trial in [0,1]

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_capacity, wm_forget = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Probability that WM is accessible for a given item under load
        p_avail = min(1.0, max(0.0, wm_capacity) / max(1.0, float(nS)))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: if available, softmax over WM weights; if not, effectively uniform
            p_wm_soft = 1.0 / np.sum(np.exp(softmax_beta_wm*(W_s-W_s[a])))
            p_wm = p_avail * p_wm_soft + (1.0 - p_avail) * (1.0 / nA)

            p_total = wm_weight*p_wm + (1.0-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decays toward uniform globally; encode on rewarded trials for the visited state
            w = (1.0 - wm_forget) * w + wm_forget * w_0
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # Move the current state's WM row toward the encoded one-hot
                enc_rate = min(1.0, 2.0 * (1.0 - wm_forget))  # tie encoding to forgetting (bounded)
                w[s, :] = (1.0 - enc_rate) * w[s, :] + enc_rate * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning rates + WM with surprise-gated weighting and load-dependent decay.

    Idea
    - RL uses different learning rates after reward vs no reward (α+ and α−).
    - WM encodes rewarded action for the current state; decays globally toward uniform each trial.
    - The weight on WM vs RL is dynamically gated by unsigned prediction error |δ|:
        large surprise -> rely more on RL exploration (down-weight WM),
        small surprise -> rely more on WM exploitation (up-weight WM).
    - WM also weakens with larger set size via a load penalty on its decay rate.

    Parameters
    ----------
    states : array-like of int
    actions : array-like of int
    rewards : array-like of {0,1}
    blocks : array-like of int
    set_sizes : array-like of int
    model_parameters : tuple
        (alpha_pos, alpha_neg, softmax_beta, wm_weight, wm_load_penalty, surprise_slope)
        - alpha_pos: RL learning rate after reward (δ>0)
        - alpha_neg: RL learning rate after no reward (δ<=0)
        - softmax_beta: RL inverse temperature (>0); internally scaled
        - wm_weight: baseline WM mixture weight in [0,1]
        - wm_load_penalty: scales decay with set size; higher means faster WM decay when nS is large
        - surprise_slope: sensitivity of WM weight to unsigned prediction error |δ|

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    alpha_pos, alpha_neg, softmax_beta, wm_weight, wm_load_penalty, surprise_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM decay per step
        base_decay = 1.0 / (1.0 + np.exp(-wm_load_penalty * (nS - 3.0)))  # 0..1 increasing with load
        wm_decay = base_decay  # use directly as the per-step decay toward uniform

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm*(W_s-W_s[a])))

            # Surprise-gated mixture: more WM when |delta| is small
            delta_peek = r - Q_s[a]
            w_gate = wm_weight * (1.0 / (1.0 + np.exp(surprise_slope * np.abs(delta_peek))))
            p_total = w_gate*p_wm + (1.0-w_gate)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update with asymmetry
            delta = r-Q_s[a]
            lr_eff = alpha_pos if delta > 0.0 else alpha_neg
            q[s][a] += lr_eff*delta

            # WM decay globally, encode on rewarded trials for current state
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                enc = min(1.0, 1.5 * (1.0 - wm_decay))
                w[s, :] = (1.0 - enc) * w[s, :] + enc * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with time-based WM decay, entropy-gated WM reliance, and load-dependent lapse.

    Idea
    - RL uses a single learning rate.
    - WM encodes rewarded action; decays toward uniform over time with a time-constant rho_time.
      To approximate time-based decay within this template, we decay globally each trial by rho_time.
    - The WM mixture weight is increased when RL is uncertain (high action entropy) to reflect
      reliance on WM when RL is diffuse, but is tempered by set-size load.
    - Lapse probability increases with set size (reflecting higher cognitive load).

    Parameters
    ----------
    states : array-like of int
    actions : array-like of int
    rewards : array-like of {0,1}
    blocks : array-like of int
    set_sizes : array-like of int
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, rho_time, load_slope, epsilon_base)
        - lr: RL learning rate
        - wm_weight: baseline WM weight in [0,1]
        - softmax_beta: RL inverse temperature (>0); internally scaled
        - rho_time: WM decay rate per trial in [0,1] (higher -> faster decay)
        - load_slope: controls how load (nS) reduces WM reliance and increases lapse
        - epsilon_base: base lapse scale in [0,1]; multiplied by load transform

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, rho_time, load_slope, epsilon_base = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load transforms
        load_factor = 1.0 / (1.0 + np.exp(-load_slope * (nS - 3.0)))  # 0 at low load, ->1 at high
        lapse = epsilon_base * load_factor
        wm_load_downweight = 1.0 - 0.7 * load_factor  # reduce WM under high load

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]

            # RL policy and entropy proxy from Q_s (softmax distribution)
            logits = softmax_beta * (Q_s - np.max(Q_s))
            pi = np.exp(logits) / np.sum(np.exp(logits))
            p_rl = 1.0 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # Entropy of RL policy (0..ln nA), normalize to 0..1
            H = -np.sum(pi * np.log(np.maximum(pi, 1e-12)))
            H_norm = H / np.log(nA)

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm*(W_s-W_s[a])))

            # Entropy-gated WM weight, tempered by load
            w_eff = wm_weight * (0.5 + 0.5 * H_norm) * wm_load_downweight
            p_mix = w_eff * p_wm + (1.0 - w_eff) * p_rl

            # Load-dependent lapse
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # Time-based WM decay and reward-triggered encoding
            w = (1.0 - rho_time) * w + rho_time * w_0
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # Encode strongly on reward
                enc_strength = min(1.0, 1.0 - 0.5 * rho_time)
                w[s, :] = (1.0 - enc_strength) * w[s, :] + enc_strength * onehot

        blocks_log_p += log_p

    return -blocks_log_p
