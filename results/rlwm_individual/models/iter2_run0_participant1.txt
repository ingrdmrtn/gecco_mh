def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with decaying Q-values + WM with load-dependent retrieval and deterministic storage on reward.

    Policy:
    - Action probability is a mixture of RL softmax and WM softmax.
    - WM retrieval probability declines sigmoidally with set size: p_ret = sigmoid(theta0 - theta1 * nS).
      If WM fails to retrieve, WM contributes a uniform policy.
    - RL Q-values decay toward uniform baseline each trial (capturing forgetting/volatility).
    
    Parameters:
    - model_parameters[0]: lr (float in [0,1]) RL learning rate for prediction errors.
    - model_parameters[1]: wm_weight (float in [0,1]) Base mixture weight for WM policy.
    - model_parameters[2]: softmax_beta (float >= 0) RL inverse temperature (internally scaled by 10).
    - model_parameters[3]: theta0 (float) WM retrieval intercept; larger increases WM usage.
    - model_parameters[4]: theta1 (float >= 0) WM retrieval load sensitivity; larger reduces WM usage with set size.
    - model_parameters[5]: q_decay (float in [0,1]) Per-trial Q decay toward uniform baseline.
    
    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, wm_weight_base, softmax_beta, theta0, theta1, q_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM retrieval probability
        # Sigmoid to keep it in (0,1)
        def sigmoid(x):
            return 1.0 / (1.0 + np.exp(-x))

        p_ret = sigmoid(theta0 - theta1 * nS)
        p_ret = float(np.clip(p_ret, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with retrieval failures: if retrieve fails, WM contributes uniform
            W_s = w[s, :]
            p_wm_det = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = p_ret * p_wm_det + (1.0 - p_ret) * (1.0 / nA)

            # Mixture
            wm_weight = wm_weight_base
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = float(np.clip(p_total, 1e-12, 1.0))
            log_p += np.log(p_total)

            # RL update with decay toward uniform
            pe = r - Q_s[a]
            q[s, a] += lr * pe
            # global decay toward uniform for this state's Q
            q[s, :] = (1.0 - q_decay) * q[s, :] + q_decay * (1.0 / nA)

            # WM updating:
            # If rewarded, store a near one-hot code for the chosen action (fast encoding).
            # If not rewarded, no overwrite; keep current memory.
            if r > 0.5:
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with eligibility traces and action stickiness + WM with load-driven noise.

    Policy:
    - Action probability is a mixture of RL softmax and WM softmax.
    - RL includes an eligibility trace for state-action updates (lambda_trace) and action stickiness (kappa).
      Stickiness adds a bias to repeat the previous action in the RL value comparison.
    - WM is a fast store of rewarded associations; at decision time, WM is corrupted by load-driven noise,
      blending the WM policy with uniform: p_wm_eff = (1 - noise) * softmax(W) + noise * uniform,
      where noise = 1 - exp(-wm_noise * nS).

    Parameters:
    - model_parameters[0]: lr (float in [0,1]) Base RL learning rate.
    - model_parameters[1]: lambda_trace (float in [0,1]) Eligibility trace decay for the chosen state-action.
    - model_parameters[2]: wm_weight (float in [0,1]) Mixture weight for WM policy.
    - model_parameters[3]: softmax_beta (float >= 0) RL inverse temperature (internally scaled by 10).
    - model_parameters[4]: kappa (float) Action stickiness strength added to last chosen action.
    - model_parameters[5]: wm_noise (float >= 0) Load-driven WM noise strength.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, lambda_trace, wm_weight_base, softmax_beta, kappa, wm_noise = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Eligibility traces for RL per state-action
        e = np.zeros((nS, nA))

        # Load-dependent WM noise (probability to replace WM policy with uniform)
        noise = 1.0 - np.exp(-wm_noise * max(1, nS))
        noise = float(np.clip(noise, 0.0, 1.0))

        last_action = None
        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with action stickiness bias
            Q_s = q[s, :].copy()
            if last_action is not None:
                bias = np.zeros(nA)
                bias[last_action] = kappa
                Q_s = Q_s + bias
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with load-driven noise
            W_s = w[s, :]
            p_wm_clean = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = (1.0 - noise) * p_wm_clean + noise * (1.0 / nA)

            # Mixture
            wm_weight = wm_weight_base
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = float(np.clip(p_total, 1e-12, 1.0))
            log_p += np.log(p_total)

            # RL update with eligibility traces:
            # Decay all traces
            e *= lambda_trace
            # Increment trace for current state-action
            e[s, a] += 1.0
            # TD error computed on the acted pair
            pe = r - q[s, a]
            # Update all Qs proportional to their eligibility
            q += lr * pe * e

            # WM updating:
            # Rewarded trials store one-hot; unrewarded gently relax toward uniform baseline.
            if r > 0.5:
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps
            else:
                relax = 0.05
                w[s, :] = (1.0 - relax) * w[s, :] + relax * w_0[s, :]

            last_action = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Uncertainty- and load-based arbitration + RL with asymmetric value decay + WM refreshing.

    Policy:
    - Mixture of RL and WM policies with dynamic arbitration based on:
      (a) Set size load (reduces WM influence), and
      (b) RL uncertainty (higher RL entropy increases WM influence).
      wm_weight_eff = sigmoid(w0 + w_unc*(Hmax - H_rl) - w_load * nS),
      where H_rl is the action-entropy of the RL policy for the current state.
    - RL uses a decay toward uniform that is stronger after negative outcomes (asymmetric decay).
    - WM uses fast storage on reward and a "refresh" operation on state visits that sharpens
      the currently best WM action slightly even without reward.

    Parameters:
    - model_parameters[0]: lr (float in [0,1]) RL learning rate for PE.
    - model_parameters[1]: softmax_beta (float >= 0) RL inverse temperature (internally scaled by 10).
    - model_parameters[2]: w0 (float) Arbitration intercept (higher favors WM).
    - model_parameters[3]: w_unc (float) Sensitivity of arbitration to RL certainty (Hmax - H_rl).
    - model_parameters[4]: w_load (float >= 0) Load penalty on WM weight per item.
    - model_parameters[5]: decay_neg (float in [0,1]) Extra RL decay toward uniform applied after negative outcomes.
    
    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, softmax_beta, w0, w_unc, w_load, decay_neg = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy for current state and its entropy
            Q_s = q[s, :]
            logits = softmax_beta * (Q_s - np.max(Q_s))
            exp_logits = np.exp(logits)
            pi_rl = exp_logits / np.sum(exp_logits)
            p_rl = float(pi_rl[a])

            # Entropy of RL policy (natural log)
            H_rl = -np.sum(pi_rl * (np.log(pi_rl + 1e-12)))
            H_max = np.log(nA)

            # WM policy
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration weight: higher when RL is uncertain (high entropy) and load is low
            wm_weight = sigmoid(w0 + w_unc * (H_max - H_rl) - w_load * nS)

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = float(np.clip(p_total, 1e-12, 1.0))
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe
            # Asymmetric decay after negative outcome: push Q toward uniform more when r == 0
            if r < 0.5:
                q[s, :] = (1.0 - decay_neg) * q[s, :] + decay_neg * (1.0 / nA)

            # WM updating: store on reward; refresh (sharpen) on any visit
            if r > 0.5:
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps
            else:
                # Refresh current best WM action slightly toward a sharper distribution
                refresh = 0.05
                best_a = int(np.argmax(w[s, :]))
                sharpen = np.full(nA, (1.0 - refresh) / (nA - 1))
                sharpen[best_a] = refresh
                # Move WM toward sharpened version
                w[s, :] = 0.9 * w[s, :] + 0.1 * sharpen

        blocks_log_p += log_p

    return -blocks_log_p
