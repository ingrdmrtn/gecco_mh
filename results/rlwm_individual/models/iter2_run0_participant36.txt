def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM mixture with eligibility-gated WM and load scaling.

    Overview:
    - RL system learns Q-values with a standard delta rule and softmax policy.
    - WM system stores action values per state, decays toward uniform, and overwrites on reward.
    - The WM mixture weight is dynamically up-weighted for recently visited states (eligibility)
      and down-weighted by set size (resource dilution).

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1).
    - wm_weight_base: Baseline weight on WM policy (0..1).
    - softmax_beta: Inverse temperature for RL softmax (scaled by 10 internally).
    - lambda_et: Eligibility-gating factor (>=0). Also controls how long recent visits keep boosting WM usage.
    - wm_forget: WM decay toward uniform per encounter (0..1).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, lambda_et, wm_forget = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Per-state eligibility trace to up-weight WM after recent visits
        e = np.zeros(nS)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta*(Q_s - Q_s[a])))

            # WM policy: deterministic softmax over WM values
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm*(W_s - W_s[a])))

            # Eligibility-gated and load-scaled WM weight
            # - Eligibility e[s] boosts WM usage immediately after visiting s (scaled by lambda_et).
            # - Load scaling via factor 3/nS (resource dilution with set size).
            wm_boost = (1.0 + lambda_et * e[s]) / (1.0 + lambda_et)
            wm_weight_eff = wm_weight_base * wm_boost * (3.0 / nS)
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay toward uniform, then overwrite on reward (fast write)
            w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot

            # Update eligibility: decay all, refresh current state
            e *= (1.0 - np.clip(lambda_et, 0.0, 1.0))
            e[s] += 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with resource-rational precision and leaky WM updates.

    Overview:
    - RL system: delta-rule Q-learning with softmax policy (beta scaled by 10).
    - WM system: softmax policy whose precision (beta_wm) and mixture weight both diminish
                 with set size via a power law; WM values updated via leaky integration.
    - WM "precision scale" also governs write strength on rewarded trials.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1).
    - wm_weight: Base WM mixture weight at set size 3 (0..1).
    - softmax_beta: Inverse temperature for RL softmax (scaled by 10 internally).
    - wm_beta_scale: Scales WM precision and write strength (>=0).
    - wm_leak: Per-encounter leak toward uniform for WM values (0..1).
    - alpha_power: Exponent for how WM degrades with set size (>=0); larger means sharper drop.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_beta_scale, wm_leak, alpha_power = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # base; will be scaled by resource precision
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute load-dependent scalars
        load_factor = (3.0 / nS) ** max(0.0, alpha_power)
        wm_beta_eff_scale = wm_beta_scale * load_factor
        wm_weight_eff_base = np.clip(wm_weight * load_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta*(Q_s - Q_s[a])))

            # WM policy with load-adjusted precision
            beta_wm_eff = softmax_beta_wm * max(0.0, wm_beta_eff_scale)
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff*(W_s - W_s[a])))

            # Mixture: load-diminished WM contribution
            wm_weight_eff = wm_weight_eff_base
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM leaky update: decay toward uniform, then reward-contingent write
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                write_strength = np.clip(wm_leak * max(0.0, wm_beta_eff_scale), 0.0, 1.0)
                w[s, :] = (1.0 - write_strength) * w[s, :] + write_strength * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with win-stay memory and delay- + load-dependent retention, plus lapses.

    Overview:
    - RL system: delta-rule Q-learning with softmax policy.
    - WM system: stores the last rewarded action per state; retention decays with the delay
                 since last successful storage and with set size (load drop).
      On each trial, WM induces a peaked distribution over the stored action (if retained),
      otherwise uniform. WM and RL are mixed and a lapse rate adds random choice noise.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1).
    - wm_weight: Base WM mixture weight at low load (0..1).
    - softmax_beta: Inverse temperature for RL softmax (scaled by 10 internally).
    - retention: Decay rate of WM retention per trial gap (>=0).
    - load_drop: Load sensitivity for WM usage and retention (>=0).
    - lapse: Lapse probability of random choice, independent of policies (0..1).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, retention, load_drop, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Memory of last rewarded action and last trial index for each state
        stored_action = -np.ones(nS, dtype=int)
        last_success_t = -np.ones(nS, dtype=int)

        # Effective WM weight decreases with load
        wm_weight_load = wm_weight * np.exp(-max(0.0, load_drop) * max(0, nS - 3))
        wm_weight_load = np.clip(wm_weight_load, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta*(Q_s - Q_s[a])))

            # Build WM-based distribution from retained memory (ignore current w and construct policy)
            if stored_action[s] >= 0:
                dt = t - last_success_t[s] if last_success_t[s] >= 0 else 1e6
                # Retention decays with delay and load
                retain_prob = np.exp(-max(0.0, retention) * dt * (1.0 + max(0.0, load_drop) * max(0, nS - 3)))
                retain_prob = np.clip(retain_prob, 0.0, 1.0)
                pref = np.ones(nA) / nA
                pref[stored_action[s]] = (1.0 - (nA - 1) / nA) * retain_prob + (1.0 / nA)
                # Now convert to a WM softmax probability of chosen action
                # By constructing a pseudo W_s that has the desired distribution as softmax output with high beta
                pseudo_W = pref
            else:
                pseudo_W = np.ones(nA) / nA

            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (pseudo_W - pseudo_W[a])))

            # Mix RL and WM, then add lapses
            p_mix = wm_weight_load * p_wm + (1.0 - wm_weight_load) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # Update WM store: on reward, store action with full strength; update timing
            if r > 0.0:
                stored_action[s] = a
                last_success_t[s] = t
                # Keep w consistent with the stored memory for this state
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot
            else:
                # On non-reward, let w drift toward uniform for that state
                drift = 0.5  # moderate drift to keep w from becoming stale; constant to avoid extra params
                w[s, :] = (1.0 - drift) * w[s, :] + drift * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
