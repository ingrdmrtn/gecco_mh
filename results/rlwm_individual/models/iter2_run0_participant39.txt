def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with load-dependent forgetting and probabilistic encoding.

    Mechanism
    - RL: vanilla Q-learning.
    - WM: stores rewarded state-action associations as a deterministic policy row (one-hot) when encoded.
          WM globally forgets toward uniform, with forgetting increasing when set size exceeds capacity.
    - Mixture: choice probability is a convex combination of WM and RL policies.

    Parameters
    ----------
    model_parameters : tuple
        - lr: RL learning rate (0..1), applied to Q-values per prediction error.
        - wm_weight: mixture weight of WM vs RL (0..1).
        - softmax_beta: RL inverse temperature; internally scaled by 10 for range.
        - capacity_K: WM capacity (in number of state-action pairs that can be stably maintained).
        - enc: WM encoding strength upon reward (0..1); moves WM row toward a one-hot for the rewarded action.
        - forget_rate: base WM forgetting rate (0..1); effective forgetting scales with load beyond capacity.

    Set-size effects
    ----------------
    - WM forgetting per trial within a block: forget_eff = forget_rate * max(0, (nS - capacity_K)/nS).
      Thus, when set size nS exceeds capacity, WM decays faster toward uniform.
    """
    lr, wm_weight, softmax_beta, capacity_K, enc, forget_rate = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50  # deterministic WM when not decayed
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent WM forgetting for this block
        overload = max(0.0, (float(nS) - float(capacity_K))) / max(1.0, float(nS))
        forget_eff = np.clip(forget_rate * overload, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic softmax over current WM row)
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM global forgetting toward uniform (load-dependent)
            if forget_eff > 0:
                w = (1 - forget_eff) * w + forget_eff * w_0

            # WM encoding on reward: move row toward one-hot of chosen action
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s] = (1 - enc) * w[s] + enc * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with choice-trace (perseveration) and load-dependent WM lapses.

    Mechanism
    - RL: vanilla Q-learning (as in template).
    - WM: maintains a fast policy per state; policy is blended with a per-state choice-trace that biases
          toward the most recent action in that state (perseveration).
    - WM lapses: with probability epsilon(load), WM outputs a uniform random choice; implemented as
                 a mixture between WM-softmax and uniform.
    - Mixture: WM and RL policies are combined by wm_weight.

    Parameters
    ----------
    model_parameters : tuple
        - lr: RL learning rate (0..1).
        - wm_weight: mixture weight of WM vs RL (0..1).
        - softmax_beta: RL inverse temperature; internally scaled by 10 for range.
        - stickiness: strength (0..1) of per-state choice-trace added to WM policy.
        - eps_base: baseline WM lapse rate at nS=3 (0..1); also controls WM reset on errors.
        - eps_slope: additional lapse per extra item beyond 3 (>=0); epsilon = clip(eps_base + eps_slope*(nS-3), 0..1).

    Set-size effects
    ----------------
    - WM lapse epsilon increases with load: epsilon = eps_base + eps_slope*(nS - 3).
      Larger set sizes cause more WM lapses (more uniform responding from WM).
    """
    lr, wm_weight, softmax_beta, stickiness, eps_base, eps_slope = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Per-state last choice tracker (initialize to None -> uniform)
        last_choice = -1 * np.ones(nS, dtype=int)

        # Load-dependent WM lapse
        epsilon = np.clip(eps_base + eps_slope * (float(nS) - 3.0), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Add per-state choice-trace to WM row
            if last_choice[s] >= 0:
                onehot_trace = np.zeros(nA)
                onehot_trace[int(last_choice[s])] = 1.0
                W_s_eff = (1 - stickiness) * W_s + stickiness * onehot_trace
            else:
                W_s_eff = W_s

            # Deterministic WM softmax
            p_wm_soft = 1 / np.sum(np.exp(softmax_beta_wm * (W_s_eff - W_s_eff[a])))
            # Lapse-adjusted WM policy: mixture with uniform due to load
            p_wm = (1 - epsilon) * p_wm_soft + epsilon * (1.0 / nA)

            # Mixture with RL
            p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM update: fast learning on rewards; mild reset on errors tied to eps_base
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # Strong consolidation into WM on reward
                w[s] = 0.0 * w[s] + 1.0 * onehot
            else:
                # On error, probabilistic reset toward uniform controlled by eps_base
                if eps_base > 0:
                    w[s] = (1 - eps_base) * w[s] + eps_base * w_0[s]

            # Update per-state last choice
            last_choice[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with load-dependent value decay + WM with Hebbian recency and global action interference.

    Mechanism
    - RL: Q-learning plus passive decay of Q-values toward uniform each trial; decay increases with set size.
    - WM: recency-weighted Hebbian update toward the chosen action when rewarded; additionally, each choice
          induces global action interference that slightly biases all states' WM toward the chosen action,
          mimicking confusion/interference among items at higher load.
    - Mixture: WM and RL policies are combined by wm_weight.

    Parameters
    ----------
    model_parameters : tuple
        - lr: RL learning rate (0..1).
        - wm_weight: mixture weight of WM vs RL (0..1).
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - q_decay_base: base RL decay toward uniform (0..1); effective decay scales with set size.
        - wm_interf: strength (0..1) of global action interference per choice (applied to all states).
        - wm_alpha: WM learning rate toward one-hot on rewarded trials (0..1).

    Set-size effects
    ----------------
    - RL decay: q_decay_eff = q_decay_base * (nS / 6). Larger set size yields stronger decay of Q-values.
      This captures reduced RL stability under higher cognitive load.
    """
    lr, wm_weight, softmax_beta, q_decay_base, wm_interf, wm_alpha = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent RL decay for this block
        q_decay_eff = np.clip(q_decay_base * (float(nS) / 6.0), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta
            # RL decay toward uniform (load-dependent)
            if q_decay_eff > 0:
                q = (1 - q_decay_eff) * q + q_decay_eff * w_0

            # WM recency/Hebbian update on reward
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s] = (1 - wm_alpha) * w[s] + wm_alpha * onehot

            # Global action interference: bias all states weakly toward the chosen action
            if wm_interf > 0:
                onehot_a = np.zeros((nS, nA))
                onehot_a[:, a] = 1.0
                w = (1 - wm_interf) * w + wm_interf * onehot_a

        blocks_log_p += log_p

    return -blocks_log_p
