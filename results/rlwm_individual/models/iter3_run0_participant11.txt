def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Gated WM storage by prediction-error with capacity scaling; RL with single learning rate.

    Idea:
    - RL learns state-action values with a single learning rate.
    - WM encodes rewarded mappings only when unsigned prediction error exceeds a gating threshold.
    - WM contribution scales down with larger set sizes via a capacity parameter.
    - WM traces decay toward uniform with a forgetting rate.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight0: Base mixture weight for WM policy before load scaling (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - gate_theta: Gating threshold on |prediction error| for committing to WM (>=0)
    - wm_capacity: Effective item capacity governing load scaling (>=1)
    - wm_forget: WM decay rate toward uniform per trial (0..1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight0, softmax_beta, gate_theta, wm_capacity, wm_forget = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM weight via capacity scaling
        load_scale = min(1.0, float(wm_capacity) / float(nS))
        wm_weight_eff_block = max(0.0, min(1.0, wm_weight0 * load_scale))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic softmax over WM weights)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff_block * p_wm + (1.0 - wm_weight_eff_block) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]

            # WM storage gated by unsigned PE and reward
            if (abs(delta) >= gate_theta) and (r > 0.5):
                # Strongly bias memory toward the rewarded action
                # Implement by adding a focused boost to chosen action then renormalize
                w[s, a] += (1.0 - np.sum(w[s, :])) + 1e-9  # ensure positivity
                # Normalize row
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Recency-based WM with perseveration bias and enhanced learning from non-rewards.

    Idea:
    - RL learns with a learning-rate amplification on non-rewarded outcomes (punishment sensitivity).
    - WM tracks the most recent action per state (independent of reward) via a recency parameter.
    - WM policy also includes an on-policy perseveration bias favoring the last action in the same state.
    - WM mixture weight scales down with set size (load).

    Parameters (model_parameters):
    - lr: Base RL learning rate (0..1)
    - wm_weight: Base WM mixture weight before load scaling (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - wm_recency: WM recency/learning rate toward last chosen action (0..1)
    - punish_kappa: Multiplier for learning on non-reward trials (>=0). Effective lr becomes lr*(1 + kappa) when r=0.
    - perseveration: Additive bias in WM logits toward the previous action in the same state (>=0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_recency, punish_kappa, perseveration = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        prev_action = -1 * np.ones(nS, dtype=int)  # last action per state

        # Load scaling for WM influence
        load_scale = min(1.0, 3.0 / float(nS))
        wm_weight_block = max(0.0, min(1.0, wm_weight * load_scale))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :].copy()

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with perseveration bias in logits
            prev_vec = np.zeros(3)
            if prev_action[s] >= 0:
                prev_vec[prev_action[s]] = 1.0
            W_eff = W_s + perseveration * prev_vec
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            # Mixture
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with enhanced non-reward learning
            alpha = lr * (1.0 + punish_kappa * (1.0 - r))
            delta = r - Q_s[a]
            q[s, a] += alpha * delta

            # WM recency update: move toward last chosen action (irrespective of reward)
            # Decay toward uniform then add recency mass to chosen action
            w[s, :] = (1.0 - wm_recency) * w[s, :] + wm_recency * w_0[s, :]
            w[s, a] += wm_recency
            # Normalize
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

            # Update previous action for perseveration on next visit
            prev_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Bayesian-like WM with confidence- and load-gated arbitration; RL with single learning rate.

    Idea:
    - RL learns values with a single rate.
    - WM approximates a Dirichlet memory over actions per state (rewarded actions increase counts).
      We implement this by maintaining a probability vector per state that decays toward uniform and
      shifts mass to the rewarded action.
    - Arbitration weight is modulated by set size via a logistic gate and by WM confidence (max probability).

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight0: Maximum WM contribution scaling factor (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - bias_load: Intercept for the arbitration logistic gate (real)
    - slope_load: Slope on set size (more negative -> less WM at nS=6) (real)
    - wm_decay: WM decay/mixing rate toward uniform and for incorporating new evidence (0..1)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight0, softmax_beta, bias_load, slope_load, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM confidence (0..1): normalized margin over uniform
            max_p = np.max(W_s)
            conf = (max_p - 1.0 / nA) / (1.0 - 1.0 / nA)
            conf = max(0.0, min(1.0, conf))

            # Load + confidence gated WM weight
            z = bias_load + slope_load * float(3 - nS) + 2.0 * (conf - 0.5)
            wm_weight_eff = wm_weight0 * sigmoid(z)
            wm_weight_eff = max(0.0, min(1.0, wm_weight_eff))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform, then add reward-driven mass to chosen action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, a] += wm_decay  # add evidence for rewarded action
            # Normalize
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
