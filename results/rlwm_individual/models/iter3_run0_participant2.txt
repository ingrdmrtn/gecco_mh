def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with uncertainty-gated WM mixture, fast WM encoding, and WM decay.

    Idea:
    - The policy is a mixture of an RL softmax and a WM softmax.
    - The contribution of WM is dynamically gated by RL uncertainty (entropy).
      When RL is uncertain (high entropy), WM weight increases; when RL is confident, WM weight decreases.
    - WM stores rewarded associations quickly (kappa_store) and decays toward uniform at a constant rate.

    Parameters (model_parameters):
    - lr: scalar in (0,1). RL learning rate.
    - wm_weight: scalar in (0,1). Base WM weight before gating.
    - softmax_beta: positive scalar. Inverse temperature for RL softmax (internally scaled x10).
    - kappa_store: scalar in (0,1). One-shot WM encoding strength on rewarded trials.
    - wm_decay: scalar in (0,1). Trial-wise decay of WM toward uniform.
    - gate_gain: positive scalar. Controls sensitivity of the WM gate to RL entropy.

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    lr, wm_weight, softmax_beta, kappa_store, wm_decay, gate_gain = model_parameters
    softmax_beta *= 10.0  # keep as in template
    softmax_beta_wm = 50.0  # very deterministic WM policy
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probability of the chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax probability of the chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Uncertainty-gated mixture: gate increases with RL entropy
            # Compute RL policy (for entropy), stable via subtract-max trick
            logits = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(logits)
            pi_rl /= np.sum(pi_rl)
            # Entropy of RL
            H_rl = -np.sum(pi_rl * np.log(np.clip(pi_rl, eps, 1.0)))
            H_max = np.log(nA)
            # Gate in [0,1] via sigmoid centered at mid-entropy
            center = 0.5 * H_max
            gate = 1.0 / (1.0 + np.exp(-gate_gain * (H_rl - center)))
            wm_eff = np.clip(wm_weight * gate, 0.0, 1.0)

            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM encoding (only if rewarded): pull probability mass toward chosen action
            if r > 0.5:
                w[s, :] = (1.0 - kappa_store) * w[s, :]
                w[s, a] += kappa_store

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning and WM with valence-sensitive plasticity plus load-dependent lapses.

    Idea:
    - RL uses separate learning rates for positive vs negative prediction errors.
    - WM policy is mixed with RL; WM experiences a lapse toward uniform that increases with set size (load).
    - WM plasticity is valence-sensitive:
        - On reward: shift WM toward the chosen action (attraction).
        - On no-reward: shift WM away from the chosen action (repulsion), spreading mass to alternatives.

    Parameters (model_parameters):
    - alpha_pos: scalar in (0,1). RL learning rate for positive PE (reward).
    - alpha_neg: scalar in (0,1). RL learning rate for negative PE (no reward).
    - wm_weight: scalar in (0,1). Mixture weight of WM vs RL.
    - softmax_beta: positive scalar. Inverse temperature for RL softmax (internally scaled x10).
    - eta_wm: scalar in (0,1). Magnitude of WM plasticity per trial.
    - lapse_gain: scalar >= 0. Scales the WM lapse with set size; effective lapse = lapse_gain * (nS-3)/3, clipped to [0, 0.5].

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    alpha_pos, alpha_neg, wm_weight, softmax_beta, eta_wm, lapse_gain = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent lapse for WM
        lapse = np.clip(lapse_gain * max(0.0, (nS - 3.0) / 3.0), 0.0, 0.5)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax prob of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax, then apply lapse toward uniform
            p_wm_core = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = (1.0 - lapse) * p_wm_core + lapse * (1.0 / nA)

            wm_eff = np.clip(wm_weight, 0.0, 1.0)
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            alpha = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += alpha * pe

            # WM valence-sensitive plasticity: mass-conserving update
            if r > 0.5:
                # Attraction to chosen action
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm
            else:
                # Repulsion from chosen action: move probability from chosen to others equally
                take = eta_wm * w[s, a] + eta_wm * (1.0 / nA)  # ensure some movement even if w[s,a] small
                take = np.clip(take, 0.0, w[s, a]) if take > w[s, a] else take
                w[s, a] = w[s, a] - take
                give_each = take / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += give_each

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with global WM interference across states under load.

    Idea:
    - Standard RL softmax is mixed with a WM softmax.
    - WM decays toward uniform and encodes rewarded actions quickly.
    - Under higher set size, rewarded encoding creates global interference:
      a small fraction of the WM update for the current state 'bleeds' into all other states,
      capturing proactive interference that grows with load.

    Parameters (model_parameters):
    - lr: scalar in (0,1). RL learning rate.
    - wm_weight: scalar in (0,1). Mixture weight of WM vs RL.
    - softmax_beta: positive scalar. Inverse temperature for RL softmax (internally scaled x10).
    - eta_wm: scalar in (0,1). WM encoding strength on rewarded trials for the current state.
    - wm_decay: scalar in (0,1). WM decay toward uniform on each visit.
    - interference: scalar in (0,1). Base fraction of WM encoding that globally interferes; effective interference scales with set size as interference * (nS/3).

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    lr, wm_weight, softmax_beta, eta_wm, wm_decay, interference = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective interference grows with set size
        xi = np.clip(interference * (nS / 3.0), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL probability for chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM probability for chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            wm_eff = np.clip(wm_weight, 0.0, 1.0)
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay on the visited state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM encoding plus global interference if rewarded
            if r > 0.5:
                # Primary encoding on the visited state
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm

                # Interference: bleed a fraction of the update into all other states
                if xi > 0.0 and nS > 1:
                    bleed = xi * eta_wm
                    for s_other in range(nS):
                        if s_other == s:
                            continue
                        w[s_other, :] = (1.0 - bleed) * w[s_other, :]
                        w[s_other, a] += bleed

        blocks_log_p += log_p

    return -blocks_log_p
