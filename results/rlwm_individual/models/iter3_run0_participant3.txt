def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with forgetting and WM precision degraded by set-size interference.

    Mechanism:
    - RL: delta-rule learning with per-state forgetting toward uniform (rl_forget) applied each trial.
    - WM: fast store that is near-deterministic but whose effective precision decreases with set size
      via an interference factor. WM also consolidates more strongly after reward.
    - Arbitration: fixed mixture weight wm_weight between WM and RL.

    Parameters
    ----------
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, rl_forget, wm_precision, interference_kappa)
        - lr: RL learning rate in [0,1].
        - wm_weight: mixture weight for WM vs RL in [0,1].
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10.
        - rl_forget: RL forgetting toward uniform for the visited state per trial in [0,1].
        - wm_precision: baseline WM precision scaling (>0), higher => sharper WM policy.
        - interference_kappa: how much set size degrades WM precision (>=0).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, rl_forget, wm_precision, interference_kappa = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with load-dependent effective precision
            load_factor = 1.0 + interference_kappa * max(0.0, float(nS) - 1.0)
            beta_wm_eff = softmax_beta_wm * wm_precision / load_factor
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture
            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with forgetting toward uniform on the visited state
            q[s, :] = (1.0 - rl_forget) * q[s, :] + rl_forget * (1.0 / nA)
            delta = r - q[s][a]
            q[s][a] += lr * delta

            # WM update:
            # - If rewarded, strengthen one-hot memory; else decay toward uniform.
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            target = r * one_hot + (1.0 - r) * w_0[s, :]
            # The same wm_precision that sharpens policy also makes updates more decisive
            alpha_wm = np.tanh(wm_precision)  # map >0 to (0,1)
            w[s, :] = (1.0 - alpha_wm) * w[s, :] + alpha_wm * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with surprise- and load-based arbitration and WM decay.

    Mechanism:
    - RL: delta-rule learning; no forgetting.
    - WM: fast memory that decays toward uniform at rate wm_decay.
    - Arbitration: the WM weight is computed each trial via a logistic transform of:
        baseline (wm_weight) + load term (favoring small set size) + surprise term (|PE|).
      This allows the model to rely more on WM in low-load and high-surprise situations.

    Parameters
    ----------
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, load_gain, pe_gain, wm_decay)
        - lr: RL learning rate in [0,1].
        - wm_weight: baseline arbitration bias (logit-space) for WM reliance.
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10.
        - load_gain: gain on load signal (positive values increase WM weight in small sets).
        - pe_gain: gain on absolute prediction error (>=0), higher => more WM reliance with surprise.
        - wm_decay: WM decay rate toward uniform per trial in [0,1].

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, load_gain, pe_gain, wm_decay = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration weight: sigmoid of baseline + load + surprise
            # Load signal: higher for smaller sets (3 vs 6)
            load_signal = float(3 - (nS - 3)) if nS >= 3 else 3.0  # equals 3 for nS=3, 0 for nS=6
            # Surprise term uses absolute PE from current Q (before updating)
            pe = abs(r - Q_s[a])
            logit_w = wm_weight + load_gain * load_signal + pe_gain * pe
            w_eff = 1.0 / (1.0 + np.exp(-logit_w))
            w_eff = np.clip(w_eff, 0.0, 1.0)

            p_total = p_wm * w_eff + (1.0 - w_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s][a]
            q[s][a] += lr * delta

            # WM update: decay toward uniform, then incorporate current outcome
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            # Reward strengthens the chosen action in WM; no-reward leaves decayed state
            if r > 0.0:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * one_hot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with inhibitory WM updates on errors and set-size-dependent lapses.

    Mechanism:
    - RL: standard delta-rule learning.
    - WM: rewarded actions are stored (strengthened) probabilistically via a gating parameter.
          Unrewarded trials inhibit the chosen action in WM (reduce its weight), encouraging
          shift. WM distribution is softly renormalized toward uniform.
    - Lapses: an action-independent lapse probability increases with set size and blends
              the final policy with the uniform distribution.

    Parameters
    ----------
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, gate_prob, inhibit_strength, lapse_slope)
        - lr: RL learning rate in [0,1].
        - wm_weight: mixture weight for WM vs RL in [0,1].
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10.
        - gate_prob: probability/strength of committing WM after reward in [0,1].
        - inhibit_strength: fraction by which the chosen action weight is reduced in WM on no-reward in [0,1].
        - lapse_slope: how much lapses increase per additional item beyond 3 (>=0).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, gate_prob, inhibit_strength, lapse_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture (before lapse)
            p_mix = p_wm * wm_weight + (1.0 - wm_weight) * p_rl

            # Lapse increases with set size beyond 3; cap to avoid degeneracy
            eps = max(0.0, lapse_slope * max(0, nS - 3))
            eps = min(eps, 0.25)  # cap at 25% lapse
            p_total = (1.0 - eps) * p_mix + eps * (1.0 / nA)

            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s][a]
            q[s][a] += lr * delta

            # WM update with reward-gated storage and error-driven inhibition
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            if r > 0.0:
                # Commit to the rewarded action with strength gate_prob
                w[s, :] = (1.0 - gate_prob) * w[s, :] + gate_prob * one_hot
            else:
                # Inhibit the chosen (incorrect) action and softly renormalize toward uniform
                w[s, a] = (1.0 - inhibit_strength) * w[s, a]
                # Soft renormalization: mix with uniform to avoid zeros and maintain distribution
                w[s, :] = 0.9 * (w[s, :] / np.sum(w[s, :])) + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
