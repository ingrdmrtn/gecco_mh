def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 1: Dynamic arbitration via WM confidence and load penalty + PE-gated WM storage.
    - RL: standard Q-learning with single learning rate.
    - WM: probabilistic preference vector per state that decays toward uniform; storage is gated
      by the signed prediction error (PE). Reward strengthens the chosen action; no-reward gives
      mild demotion.
    - Arbitration: trial-wise WM weight is a sigmoid of WM confidence minus a load penalty.
      WM confidence is 1 - normalized entropy of WM policy for the current state.

    Parameters:
      lr:                RL learning rate (0..1)
      softmax_beta:      RL inverse temperature (scaled internally by 10; >=0)
      wm_decay:          WM leak toward uniform per encounter (0..1)
      gate_gain:         Sensitivity of WM update gate to signed PE (higher = more selective)
      arb_gain:          Gain on arbitration sigmoid mapping confidence-load to WM weight

    Returns:
      Negative log-likelihood of observed actions.
    """
    lr, softmax_beta, wm_decay, gate_gain, arb_gain = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (softmax over WM weights)
            wm_logits = W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (wm_logits - wm_logits[a])))

            # Arbitration: confidence minus load penalty through sigmoid
            eps = 1e-12
            ent = -np.sum(np.clip(W_s, eps, 1.0) * np.log(np.clip(W_s, eps, 1.0)))
            conf = 1.0 - ent / np.log(nA)  # 0..1
            load_penalty = (nS - 3) / max(1.0, (6 - 3))  # 0 for set size 3, 1 for 6
            wm_weight_eff = 1 / (1 + np.exp(-arb_gain * (conf - load_penalty)))
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # PE-gated WM storage: gate opens more with positive PE
            gate = 1 / (1 + np.exp(-gate_gain * delta))
            if r > 0:
                # Strengthen chosen action proportional to gate; renormalize
                add = gate
                w[s, :] = (1.0 - add) * w[s, :]
                w[s, a] += add
            else:
                # Mild demotion of chosen action when no reward; redistribute to others
                demote = 0.5 * gate
                take = min(demote, w[s, a] * 0.999)
                w[s, a] -= take
                redistribute = take / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += redistribute

            # Ensure proper normalization and bounds
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 2: Set-sizeâ€“attenuated WM mixture + asymmetric RL learning + replace/inhibit WM.
    - RL: Q-learning with asymmetric learning rates for positive vs negative prediction errors.
    - WM: winner-take-all replacement on reward; inhibition of the chosen action on no-reward.
      WM leaks toward uniform each time the state is visited.
    - Arbitration: a base WM mixture weight attenuated by set size via an interference factor:
        wm_weight_eff = mixing_base / (1 + interference * (nS - 1))

    Parameters:
      lr_pos:         RL learning rate for positive PE (0..1)
      lr_neg:         RL learning rate for negative PE (0..1)
      softmax_beta:   RL inverse temperature (scaled internally by 10; >=0)
      mixing_base:    Base WM mixture weight before set-size attenuation (0..1)
      interference:   Degree to which larger set sizes reduce WM contribution (>=0)
      alpha_store:    WM replacement strength when rewarded (0..1)

    Returns:
      Negative log-likelihood of observed actions.
    """
    lr_pos, lr_neg, softmax_beta, mixing_base, interference, alpha_store = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size attenuated WM weight
        wm_weight_eff = mixing_base / (1.0 + max(0.0, interference) * max(0, nS - 1))
        wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

        # Choose a mild WM leak dependent on set size as well (optional implicit effect)
        wm_leak = np.clip(0.15 * (nS / 6.0), 0.0, 0.5)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            wm_logits = W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (wm_logits - wm_logits[a])))

            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            delta = r - Q_s[a]
            lr = lr_pos if delta >= 0 else lr_neg
            q[s][a] += lr * delta

            # WM leak toward uniform
            w[s, :] = (1.0 - wm_leak) * w[s, :] + wm_leak * w_0[s, :]

            # WM replace/inhibit update
            if r > 0:
                # Replace toward one-hot on chosen action with strength alpha_store
                alpha = np.clip(alpha_store, 0.0, 1.0)
                w[s, :] = (1.0 - alpha) * w[s, :]
                w[s, a] += alpha
            else:
                # Inhibit chosen; redistribute to other actions
                inhib = 0.25 * (1.0 - np.mean(W_s))  # stronger when WM is uncertain
                take = min(inhib, w[s, a] * 0.999)
                w[s, a] -= take
                redistribute = take / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += redistribute

            # Normalize and bound
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 3: Surprise-based arbitration + one-shot WM strengthening + lapse.
    - RL: standard Q-learning with single learning rate.
    - WM: decays toward uniform; on reward, one-shot strengthening of the chosen action with
      configurable store strength; no-reward yields mild anti-chosen nudge.
    - Arbitration: WM weight increases when surprise (|PE|) is small (suggesting a known mapping):
        wm_weight = sigmoid(arb_gain * (1 - |PE|)), computed online per trial.
      This makes WM dominate once RL/WMs are consistent and prediction becomes stable.
    - Lapse: final policy includes a lapse component mixing in uniform random responding.

    Parameters:
      lr:                 RL learning rate (0..1)
      softmax_beta:       RL inverse temperature (scaled internally by 10; >=0)
      wm_decay:           WM leak toward uniform per encounter (0..1)
      arb_gain:           Gain controlling sensitivity of WM weight to surprise
      lapse:              Probability of uniform random responding (0..0.2 typical)
      wm_store_strength:  One-shot WM strengthening amount when r=1 (0..1)

    Returns:
      Negative log-likelihood of observed actions.
    """
    lr, softmax_beta, wm_decay, arb_gain, lapse, wm_store_strength = model_parameters

    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            wm_logits = W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (wm_logits - wm_logits[a])))

            # Compute PE for arbitration (before RL update)
            delta = r - Q_s[a]
            wm_weight_eff = 1 / (1 + np.exp(-arb_gain * (1.0 - abs(delta))))
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            p_mix = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = (1 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            q[s][a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update
            if r > 0:
                add = np.clip(wm_store_strength, 0.0, 1.0)
                w[s, :] = (1.0 - add) * w[s, :]
                w[s, a] += add
            else:
                # Mild anti-chosen nudge to encourage exploration in WM
                demote = 0.2 * (1.0 / nA)
                take = min(demote, w[s, a] * 0.999)
                w[s, a] -= take
                redistribute = take / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += redistribute

            # Normalize and bound
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p
