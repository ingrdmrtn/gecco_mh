def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL(Î») with WM precision reduced by load (interference-based) and global decay.

    Idea
    - RL uses an eligibility trace to allow faster credit assignment across recently visited items.
    - WM encodes rewarded state-action pairs toward a one-hot, but its precision is reduced as set size grows
      via an interference parameter (phi_interf). Lower precision both:
        (a) reduces the effective WM inverse temperature, and
        (b) increases global decay toward uniform.
    - Mixture policy combines WM and RL with a fixed wm_weight.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for each trial.
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, lambda_e, phi_interf)
        - lr: RL learning rate [0,1]
        - wm_weight: mixture weight of WM policy vs RL [0,1]
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10
        - lambda_e: eligibility trace decay (0..1); higher spreads learning across recent visits
        - phi_interf: WM interference per extra item; larger values reduce WM precision with larger nS

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, lambda_e, phi_interf = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Eligibility traces over state-action pairs
        e = np.zeros((nS, nA))

        # WM precision and decay depend on load
        wm_precision = 1.0 / (1.0 + phi_interf * max(0, nS - 3))  # in (0,1]
        wm_beta_eff = softmax_beta_wm * wm_precision
        wm_decay = 1.0 - wm_precision  # more items -> more decay toward uniform

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy with fixed softmax
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: higher load reduces precision (via wm_beta_eff)
            p_wm = 1.0 / np.sum(np.exp(wm_beta_eff * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with eligibility trace
            # Update trace for the chosen state-action; decay others
            e *= lambda_e
            e[s, a] += 1.0
            delta = r - Q_s[a]
            q += lr * delta * e  # credit assignment spread by eligibility

            # WM global decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # Reward-triggered encoding toward one-hot
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # Encoding strength tied to available precision
                enc_rate = wm_precision
                w[s, :] = (1.0 - enc_rate) * w[s, :] + enc_rate * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with load-adjusted WM mixture weight and persistent WM traces.

    Idea
    - RL uses a standard delta rule.
    - WM stores rewarded associations with an encoding strength; traces persist across trials with a persistence factor.
    - The influence of WM on choice is down-weighted by set size using a logistic transform:
        wm_weight_eff = sigmoid(logit(wm_weight_base) - load_gain * (nS - 3)).
      This captures reduced WM use under higher load while remaining within [0,1].

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for each trial.
    model_parameters : tuple
        (lr, wm_weight_base, softmax_beta, load_gain, enc_strength, wm_persist)
        - lr: RL learning rate [0,1]
        - wm_weight_base: base WM mixture weight in [0,1] at set size 3
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10
        - load_gain: how strongly larger set sizes reduce WM weight (>=0)
        - enc_strength: WM encoding rate toward one-hot on reward [0,1]
        - wm_persist: WM trace persistence each trial (0..1); lower -> faster decay to uniform

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, load_gain, enc_strength, wm_persist = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    # Helper: logistic and logit without importing
    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    def logit(p):
        eps = 1e-9
        p = np.clip(p, eps, 1.0 - eps)
        return np.log(p / (1.0 - p))

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-adjusted WM weight (constant within block given fixed nS)
        wm_logit = logit(wm_weight_base) - load_gain * max(0, nS - 3)
        wm_weight_eff = sigmoid(wm_logit)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM policies
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with load-adjusted WM weight
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM persistence decay toward uniform
            w = wm_persist * w + (1.0 - wm_persist) * w_0
            # Reward-triggered WM encoding
            if r > 0.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - enc_strength) * w[s, :] + enc_strength * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with load-modulated exploration and WM confirmation counts.

    Idea
    - RL uses a delta rule, but its effective inverse temperature decreases with load,
      capturing noisier RL choices under higher set size.
      effective_beta = (softmax_beta * 10) / (1 + load_temp_slope * (nS - 3)).
    - WM tracks confirmation counts of rewarded actions per state. As evidence accumulates,
      WM distribution concentrates on the most confirmed action; once a count reaches k_confirm,
      WM becomes nearly deterministic for that state.
    - Policy is a mixture of RL and WM with a fixed wm_weight.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size (3 or 6) for each trial.
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, load_temp_slope, k_confirm)
        - lr: RL learning rate [0,1]
        - wm_weight: mixture weight of WM policy vs RL [0,1]
        - softmax_beta: base RL inverse temperature (>0); internally scaled by 10
        - load_temp_slope: how strongly larger set sizes lower RL temperature (>=0)
        - k_confirm: confirmation threshold (>=1). Larger values require more rewards to lock in WM

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, load_temp_slope, k_confirm = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):

        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # WM confirmation counts
        counts = np.zeros((nS, nA))

        # RL temperature reduced by load (constant per block)
        beta_rl_eff = softmax_beta / (1.0 + load_temp_slope * max(0, nS - 3))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy with load-adjusted beta
            p_rl = 1.0 / np.sum(np.exp(beta_rl_eff * (Q_s - Q_s[a])))

            # WM policy: from confirmation counts
            # If any action reaches k_confirm, push W_s to one-hot on the best action
            if np.max(counts[s, :]) >= k_confirm:
                best_a = int(np.argmax(counts[s, :]))
                W_s = np.zeros(nA)
                W_s[best_a] = 1.0
                w[s, :] = W_s.copy()

            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # Update WM counts and distribution
            if r > 0.0:
                counts[s, a] += 1.0
            # Convert counts to probabilities with a small additive smoothing toward uniform
            smoothed = counts[s, :] + 1e-3
            if np.sum(smoothed) > 0:
                w[s, :] = smoothed / np.sum(smoothed)
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
