def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with capacity-limited WM and lapse.

    Mechanism:
    - RL: delta-rule learning over Q-values with softmax action selection.
    - WM: near-deterministic store for the currently correct action in each state.
    - Capacity limit: the WM contribution is down-weighted when set size exceeds a
      continuous capacity C by a factor min(1, C/nS).
    - Lapse: with probability epsilon, choices are uniform (captures random lapses).

    Parameters
    ----------
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, capacity_C, wm_decay, epsilon)
        - lr: RL learning rate in [0,1].
        - wm_weight: baseline WM mixture weight in [0,1].
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10.
        - capacity_C: effective WM capacity (>=1). When nS > C, WM influence scales ~ C/nS.
        - wm_decay: WM update rate toward its target in [0,1].
        - epsilon: lapse probability (0..0.2 typical) mixing policy with uniform.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, capacity_C, wm_decay, epsilon = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm*(W_s - W_s[a])))

            # Capacity-limited WM weight
            cap_scale = min(1.0, max(0.0, float(capacity_C)) / float(nS))
            wm_eff = wm_weight * cap_scale

            p_total = p_wm*wm_eff + (1-wm_eff)*p_rl
            # Lapse mixture with uniform
            p_total = (1 - epsilon) * p_total + epsilon * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            target = r * one_hot + (1.0 - r) * w_0[s, :]
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with surprise-gated WM reliance, WM volatility decay, and WM noise.

    Mechanism:
    - RL: delta-rule Q-learning with softmax.
    - WM policy: softmax over a WM map; however, the map is smoothed by wm_noise
      (mixture with uniform) to capture imprecision.
    - Surprise gating: High absolute reward prediction error (surprise) reduces the
      instantaneous reliance on WM (trust RL more). The reduction scales with
      surprise_sensitivity.
    - WM update: fast update toward one-hot if rewarded, otherwise decay to uniform
      at rate wm_vol_decay.

    Parameters
    ----------
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, surprise_sensitivity, wm_vol_decay, wm_noise)
        - lr: RL learning rate in [0,1].
        - wm_weight: baseline WM mixture weight in [0,1].
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10.
        - surprise_sensitivity: scales how much abs(PE) suppresses WM reliance (>=0).
        - wm_vol_decay: WM update/decay rate toward target in [0,1].
        - wm_noise: amount of WM noise (0..1) mixing W with uniform before softmax.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, surprise_sensitivity, wm_vol_decay, wm_noise = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s_raw = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM noise smoothing
            W_s = (1.0 - wm_noise) * W_s_raw + wm_noise * (1.0 / nA)
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm*(W_s - W_s[a])))

            # Surprise-gated WM reliance
            pe = r - Q_s[a]
            surprise = abs(pe)
            wm_suppression = 1.0 / (1.0 + surprise_sensitivity * surprise)
            wm_eff = wm_weight * wm_suppression

            p_total = p_wm*wm_eff + (1-wm_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            target = r * one_hot + (1.0 - r) * w_0[s, :]
            w[s, :] = (1.0 - wm_vol_decay) * w[s, :] + wm_vol_decay * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with load-triggered WM interference and lapse.

    Mechanism:
    - RL: delta-rule learning with softmax.
    - WM: softmax over a maintained W; precision drops as set size exceeds a
      threshold via an interference factor.
    - Interference: when set size nS surpasses threshold_k, WM precision is
      reduced: beta_wm_eff = beta_wm / (1 + wm_interference * max(0, nS - threshold_k)).
      Additionally, WM contents are pulled toward uniform by an interference
      contraction, modeling cross-item interference.
    - Lapse: mixture with uniform at rate lapse_rate.

    Parameters
    ----------
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, wm_interference, threshold_k, lapse_rate)
        - lr: RL learning rate in [0,1].
        - wm_weight: mixture weight for WM vs RL in [0,1].
        - softmax_beta: RL inverse temperature (>0); internally scaled by 10.
        - wm_interference: strength of load-induced WM degradation (>=0).
        - threshold_k: set-size threshold beyond which WM interference escalates (>=1).
        - lapse_rate: mixture with uniform choice (0..0.2 typical).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_interference, threshold_k, lapse_rate = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s_base = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            overload = max(0.0, float(nS) - float(threshold_k))
            beta_wm_eff = softmax_beta_wm / (1.0 + wm_interference * overload)
            # Interference contraction toward uniform (state-specific)
            contraction = wm_interference * (overload / max(1.0, float(nS)))
            W_s = (1.0 - contraction) * W_s_base + contraction * (1.0 / nA)
            p_wm = 1 / np.sum(np.exp(beta_wm_eff*(W_s - W_s[a])))

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            # Lapse
            p_total = (1 - lapse_rate) * p_total + lapse_rate * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta
            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            # Reward-driven reinforcement; mild decay to uniform otherwise
            reinforce = r * one_hot + (1.0 - r) * w_0[s, :]
            # Use a fast WM update proportional to interference-adjusted precision
            wm_update_rate = min(1.0, 1.0 / (1.0 + wm_interference * overload))
            w[s, :] = (1.0 - wm_update_rate) * w[s, :] + wm_update_rate * reinforce

        blocks_log_p += log_p

    return -blocks_log_p
