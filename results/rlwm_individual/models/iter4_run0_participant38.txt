def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Capacity-limited WM with uncertainty-weighted arbitration.

    Mechanism:
    - RL learns action values per state via Rescorla-Wagner.
    - WM stores a fast, supervised mapping for rewarded actions and decays toward uniform otherwise.
    - Arbitration weight for WM is the product of:
        (a) a base WM engagement parameter,
        (b) a capacity term reflecting a fixed number of WM slots (relative to set size),
        (c) an uncertainty term: RL's policy entropy (higher entropy -> more WM reliance).

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - wm_base: baseline WM engagement weight in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - slots: number of items WM can actively maintain (>=0).
    - wm_learn: WM supervised learning rate toward one-hot after reward in [0,1].
    - wm_decay: WM decay rate toward uniform after non-reward in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_base, softmax_beta, slots, wm_learn, wm_decay = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability of the observed action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Working memory policy: near-deterministic softmax over W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute RL policy to estimate entropy (uncertainty)
            logits = softmax_beta * Q_s
            logits -= np.max(logits)
            prl_vec = np.exp(logits)
            prl_vec /= np.sum(prl_vec)
            eps = 1e-12
            entropy = -np.sum(prl_vec * np.log(np.clip(prl_vec, eps, 1.0)))
            max_entropy = np.log(nA)
            unc = entropy / max_entropy  # in [0,1]

            # Capacity term: probability the state is within WM slots
            cap = min(1.0, float(slots) / max(1.0, nS))

            wm_weight_t = np.clip(wm_base * cap * unc, 0.0, 1.0)

            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward -> supervised one-hot; no reward -> decay
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with adaptive arbitration learned online and load penalty.

    Mechanism:
    - RL learns via Rescorla-Wagner.
    - WM stores a fast mapping; updated toward one-hot on reward.
    - The arbitration weight (between WM and RL) is controlled by a latent logit z that is updated online:
        z <- z + eta_w * (r - Q_s[a]) * (p_wm(a|s) - p_rl(a|s)) - load_penalty * load_term
      where load_term increases with set size (higher load reduces reliance on WM).
    - WM's inverse temperature is also modulated by a scale factor.

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_beta_scale: multiplicative factor for WM determinism (>=0), applied to a high base beta.
    - eta_w: learning rate for arbitration logit z (>=0).
    - load_penalty: strength of load-driven reduction of WM weight (>=0).
    - wm_learn: WM supervised learning rate toward one-hot after reward in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_beta_scale, eta_w, load_penalty, wm_learn = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # base; will be scaled
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Arbitration logit z per block, initialized at 0 (sigmoid -> 0.5)
        z = 0.0

        # Load term constant within block
        load_term = np.clip((nS - 3.0) / 3.0, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability of the observed action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with scaled determinism
            beta_wm_eff = softmax_beta_wm * max(0.0, wm_beta_scale)
            p_wm = 1 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Current WM weight
            wm_weight_t = 1.0 / (1.0 + np.exp(-z))
            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update (supervised on reward; weak decay otherwise toward uniform)
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                # mild implicit decay: small fraction of wm_learn to baseline
                decay = 0.25 * wm_learn
                w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

            # Update arbitration logit z: move toward the system that better predicted the outcome,
            # and penalize WM under higher load.
            # We need p_rl(a|s) and p_wm(a|s) explicitly for the chosen action; already computed.
            z += eta_w * delta * (p_wm - p_rl) - load_penalty * load_term

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Dual-learning-rate RL + WM with load-induced interference and choice stickiness.

    Mechanism:
    - RL uses separate learning rates for positive vs. negative prediction errors.
    - WM stores fast mappings; reward drives one-hot consolidation, non-reward induces decay.
    - Load-induced interference: non-visited states drift toward uniform each trial, proportionally to set size.
    - Choice stickiness: RL policy includes a bias toward the last chosen action within the same state.
    - Arbitration weight for WM decays with load but is anchored by a gain parameter.

    Parameters (total 6):
    - alpha_plus: RL learning rate for positive prediction errors in [0,1].
    - alpha_minus: RL learning rate for negative prediction errors in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_gain: baseline WM influence and WM learning strength in [0,1].
    - interference: cross-state WM interference rate (>=0), scaled by load each trial.
    - stick: choice stickiness added to the value of the previously chosen action within state (can be >=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    alpha_plus, alpha_minus, softmax_beta, wm_gain, interference, stick = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Last chosen action per state for stickiness (-1 means none yet)
        last_a = -1 * np.ones(nS, dtype=int)

        # Precompute load factor for arbitration and interference scaling
        load_term = np.clip((nS - 3.0) / 3.0, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL Q with stickiness bias on the previously chosen action in this state
            Q_s = q[s, :].copy()
            if last_a[s] >= 0:
                Q_s[last_a[s]] += stick

            W_s = w[s, :]

            # RL choice probability of the observed action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: near-deterministic softmax over W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: WM weight decreases with load, anchored by wm_gain
            wm_base = np.clip(wm_gain, 0.0, 1.0)
            wm_weight_t = np.clip(wm_base * (1.0 - 0.8 * load_term), 0.0, 1.0)

            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with separate learning rates
            pe = r - q[s, a]
            lr = alpha_plus if pe >= 0.0 else alpha_minus
            q[s, a] += lr * pe

            # WM update: reward consolidates, no-reward decays
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_gain) * w[s, :] + wm_gain * target
            else:
                decay = 0.5 * wm_gain
                w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # Cross-state interference scaled by load: non-visited states drift toward uniform
            intrude = interference * load_term
            if intrude > 0.0:
                for j in range(nS):
                    if j == s:
                        continue
                    w[j, :] = (1.0 - intrude) * w[j, :] + intrude * w_0[j, :]

            # Normalize WM rows
            for j in range(nS):
                w[j, :] = np.clip(w[j, :], 1e-8, 1.0)
                w[j, :] /= np.sum(w[j, :])

            # Update last chosen action for stickiness
            last_a[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
