def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with load-dependent WM reliance, reward-gated encoding, lapse, and choice perseveration.

    Mechanism
    - RL: Standard Q-learning.
    - WM: Stores rewarded mappings; on reward, the WM row for the state is pulled toward the chosen action.
      No change on non-reward (to avoid false encoding).
    - Mixture: A convex combination of RL and WM policies. The WM reliance is reduced by set size.
    - Perseveration: A choice-stickiness bias adds a fixed bonus to the previously chosen action’s logit.
    - Lapse: With some probability, choices are random (uniform), independent of value.

    Parameters
    ----------
    model_parameters : tuple
        - lr: RL learning rate (0..1).
        - wm_weight: Base mixture weight for WM vs RL (0..1). Effective WM weight decreases with set size.
        - softmax_beta: RL inverse temperature; internally scaled by 10 for sharper policies.
        - wm_store: WM encoding step size on rewarded trials (0..1).
        - lapse: Lapse probability (0..0.5), mixed with uniform choice.
        - pers: Choice perseveration strength added to last chosen action’s logit (>=0).

    Set-size effects
    ----------------
    - Effective WM mixture weight: wm_weight_eff = wm_weight * (3 / nS). Thus, WM contributes less under higher load (nS=6).
    """
    lr, wm_weight, softmax_beta, wm_store, lapse, pers = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        wm_weight_eff = np.clip(wm_weight * (3.0 / float(nS)), 0.0, 1.0)
        prev_a = None

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            # Add perseveration bias to logits
            stick = np.zeros(nA)
            if prev_a is not None:
                stick[prev_a] += pers

            Q_s = q[s, :] + stick
            W_s = w[s, :] + stick

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with lapse
            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = (1 - lapse) * p_total + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r - q[s][a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s] = (1 - wm_store) * w[s] + wm_store * onehot

            prev_a = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with entropy-based arbitration and load-dependent WM noise.

    Mechanism
    - RL: Standard Q-learning; choice distribution derived from Q via softmax.
    - WM: Fast, deterministic store that is corrupted by load-dependent noise.
    - Arbitration: The WM mixture weight is dynamically determined by RL uncertainty (entropy).
      When RL is uncertain (high entropy), the model leans more on WM, modulated by set size.
    - WM learning: Reward-dependent supervised update toward the chosen action; on non-reward,
      the WM row partially relaxes toward uniform (forgetting).

    Parameters
    ----------
    model_parameters : tuple
        - lr: RL learning rate (0..1).
        - wm_weight: Base WM weight (0..1), converted to a logit base for arbitration.
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - arbit_k: Arbitration sensitivity (>0). Higher values increase the impact of RL entropy and set size on reducing WM reliance.
        - wm_noise_base: Base WM noise level (>=0). Effective WM noise increases with set size.
        - wm_alpha: WM learning rate toward target (0..1).

    Set-size effects
    ----------------
    - WM noise: noise_eff = wm_noise_base * (nS / 3), which lowers the effective WM inverse temperature:
        beta_wm_eff = softmax_beta_wm / (1 + noise_eff).
    - Arbitration: wm_weight_eff = sigmoid(logit(wm_weight) - arbit_k * H_rl * (nS / 6)),
      so higher load and higher RL entropy both reduce WM reliance.
    """
    lr, wm_weight, softmax_beta, arbit_k, wm_noise_base, wm_alpha = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Precompute constants for arbitration
        wm_weight_clip = np.clip(wm_weight, 1e-6, 1 - 1e-6)
        base_logit = np.log(wm_weight_clip) - np.log(1 - wm_weight_clip)
        noise_eff = wm_noise_base * (float(nS) / 3.0)
        beta_wm_eff = softmax_beta_wm / (1.0 + noise_eff)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: full softmax to compute entropy
            logits_rl = softmax_beta * Q_s
            logits_rl -= np.max(logits_rl)
            probs_rl = np.exp(logits_rl)
            probs_rl /= np.sum(probs_rl)
            p_rl = probs_rl[a]

            # Entropy of RL to modulate arbitration
            H_rl = -np.sum(probs_rl * np.log(np.clip(probs_rl, 1e-12, 1.0)))
            wm_weight_eff = 1.0 / (1.0 + np.exp(-(base_logit - arbit_k * H_rl * (float(nS) / 6.0))))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            logits_wm = beta_wm_eff * W_s
            logits_wm -= np.max(logits_wm)
            probs_wm = np.exp(logits_wm)
            probs_wm /= np.sum(probs_wm)
            p_wm = probs_wm[a]

            p_total = wm_weight_eff * p_wm + (1 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r - q[s][a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            if r > 0:
                target = np.zeros(nA)
                target[a] = 1.0
            else:
                target = w_0[s].copy()
            w[s] = (1 - wm_alpha) * w[s] + wm_alpha * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with eligibility traces + WM with load-dependent eviction interference.

    Mechanism
    - RL: Q-learning with eligibility traces. An eligibility matrix (same shape as Q) decays each trial;
      the current state-action sets eligibility to 1, and Q is updated proportionally to eligibility.
    - WM: Reward-based encoding toward the chosen action for the current state.
    - Interference: Higher set size increases global eviction pressure that pulls all WM rows toward uniform,
      mimicking item interference/overwriting under higher load.
    - Mixture: WM and RL policies are combined via wm_weight.

    Parameters
    ----------
    model_parameters : tuple
        - lr: RL learning rate (0..1).
        - wm_weight: Mixture weight of WM vs RL (0..1).
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - trace_lambda: Eligibility decay (0..1). Larger values prolong credit assignment to recent choices.
        - evict_base: Base eviction/interference rate (0..1). Effective eviction increases with set size.
        - wm_enc: WM encoding rate toward chosen one-hot on rewarded trials (0..1).

    Set-size effects
    ----------------
    - Interference/eviction: evict_eff = evict_base * ((nS - 3) / 3), i.e., zero at set size 3 and strongest at 6.
    """
    lr, wm_weight, softmax_beta, trace_lambda, evict_base, wm_enc = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        elig = np.zeros((nS, nA))

        evict_eff = np.clip(evict_base * ((float(nS) - 3.0) / 3.0), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update with eligibility traces
            delta = r - q[s][a]
            elig *= trace_lambda
            elig[s, :] *= 0.0
            elig[s, a] = 1.0
            q += lr * delta * elig

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s] = (1 - wm_enc) * w[s] + wm_enc * onehot

            # Global eviction/interference pulling WM toward uniform
            if evict_eff > 0:
                w = (1 - evict_eff) * w + evict_eff * w_0

        blocks_log_p += log_p

    return -blocks_log_p
