def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + surprise-gated WM arbitration with age-adjusted learning speed.

    Model idea:
    - RL learns Q-values with a single learning rate, but older adults have a slowdown factor on learning.
    - WM stores a peaked distribution over actions per state; on reward it moves toward a one-hot code,
      on no-reward it decays toward uniform.
    - The arbitration weight for WM is stronger when:
        (a) set size is small (low load), and
        (b) the current trial has high surprise (|reward - Q| larger), reflecting that WM engages more when
            there is a notable discrepancy that can be resolved by explicit memory.
      Older adults receive a penalty to WM contribution.
    - Action selection is a mixture of RL softmax and WM softmax.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight_base: baseline WM mixing weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - surprise_sensitivity: scales how much surprise gates WM (>=0)
    - age_rl_slowdown: factor that reduces RL learning in older adults (0..1; effective lr *= (1 - age_rl_slowdown) if older)

    Inputs:
    - states, actions, rewards: arrays per trial
    - blocks: block id per trial
    - set_sizes: set size per trial (3 or 6)
    - age: array-like with single participant age

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight_base, softmax_beta, surprise_sensitivity, age_rl_slowdown = model_parameters
    softmax_beta *= 10.0  # scale as in template
    softmax_beta_wm = 50.0  # very deterministic WM policy
    age = age[0]
    older = 1.0 if age >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))  # WM distribution per state
        w_0 = (1.0 / nA) * np.ones((nS, nA))  # uniform prior WM

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy (softmax with "selector" trick)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (softmax on current WM distribution)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: load and surprise gating with age penalty
            load_scale = 3.0 / float(nS)  # 1.0 for setsize=3, 0.5 for setsize=6
            delta = r - Q_s[a]
            surprise = abs(delta)
            wm_mix = wm_weight_base * load_scale * (1.0 - 0.4 * older) * (1.0 + surprise_sensitivity * surprise)
            wm_mix = np.clip(wm_mix, 0.0, 1.0)

            # Mixture policy
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update with age-adjusted learning speed
            eff_lr = lr * (1.0 - age_rl_slowdown * older)
            q[s, a] += eff_lr * (r - q[s, a])

            # WM update: rewarded trials strengthen memory, non-reward causes decay toward uniform
            if r > 0.5:
                # Move W_s toward one-hot on chosen action proportionally to surprise
                alpha_w = np.clip(0.5 + 0.5 * surprise, 0.0, 1.0)
                peaked = (1.0 / nA) * np.ones(nA)
                peaked[a] = 1.0
                peaked /= np.sum(peaked)
                w[s, :] = (1.0 - alpha_w) * W_s + alpha_w * peaked
            else:
                # Decay toward uniform; more decay under higher load
                decay = 0.3 + 0.4 * (nS - 3) / 3.0  # 0.3 for setsize=3, 0.7 for setsize=6
                w[s, :] = (1.0 - decay) * W_s + decay * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL mixed with a win-stay/lose-shift WM heuristic; age affects shift tendency.

    Model idea:
    - RL learns Q-values (single learning rate).
    - WM implements a fast heuristic:
        - After a rewarded choice in a state, prefer repeating that action (win-stay).
        - After a non-rewarded choice, bias away from the last action (lose-shift).
      The strength of this heuristic is controlled by win_stay_strength, while shifting away
      on losses is reduced in older adults by age_shift_bias.
    - WM contribution is higher under low load (set size 3).
    - Action selection is a mixture of RL softmax and WM heuristic softmax.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: baseline WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - win_stay_strength: strength of win-stay preference (>=0)
    - age_shift_bias: reduces lose-shift effect for older adults (0..1; larger means less shifting in older)

    Inputs:
    - states, actions, rewards, blocks, set_sizes, age

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, win_stay_strength, age_shift_bias = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]
    older = 1.0 if age >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))  # WM distribution per state (used to store heuristic policy)
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_action = -1 * np.ones(nS, dtype=int)
        last_reward = -np.ones(nS)  # -1 means no history yet

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Construct WM heuristic distribution W_s based on last outcome in this state
            W_s = w_0[s, :].copy()
            if last_action[s] >= 0 and last_reward[s] >= 0.0:
                if last_reward[s] > 0.5:
                    # Win-stay: move mass to last_action
                    W_s = (1.0 / nA) * np.ones(nA)
                    W_s[last_action[s]] = 1.0 + win_stay_strength
                    W_s = W_s / np.sum(W_s)
                else:
                    # Lose-shift: move mass away from last_action
                    shift_strength = max(0.0, 1.0 - age_shift_bias * older) * win_stay_strength
                    W_s = (1.0 / nA) * np.ones(nA)
                    # reduce probability of last action, increase others
                    W_s[last_action[s]] = max(0.0, 1.0 - shift_strength)
                    # Normalize to sum to 1
                    W_s = W_s / np.sum(W_s)

            # WM policy softmax
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Load-dependent arbitration
            load_scale = 3.0 / float(nS)
            wm_mix = np.clip(wm_weight * load_scale * (1.0 - 0.3 * older), 0.0, 1.0)

            # Mixture
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # Update heuristic memory traces
            last_action[s] = a
            last_reward[s] = r
            # Also store W_s so next trial can use the constructed distribution as w[s]
            w[s, :] = W_s

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with novelty bonus and episodic WM; age reduces directed exploration.

    Model idea:
    - RL learns Q-values with a novelty (count-based) bonus to encourage exploration of
      less-tried actions; older adults show reduced novelty-driven exploration.
    - WM stores the last rewarded action per state as a sharp distribution; on non-reward it
      partially decays to uniform.
    - WM contribution is stronger in low load (set size 3).
    - Policy is a mixture of RL (with novelty bonus) and WM.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: baseline WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally by 10)
    - novelty_bonus: base strength of exploration bonus (>=0)
    - age_explore_decay: multiplicative reduction of novelty bonus for older adults (0..1)

    Inputs:
    - states, actions, rewards, blocks, set_sizes, age

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, novelty_bonus, age_explore_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]
    older = 1.0 if age >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask].astype(int)
        block_rewards = rewards[block_mask].astype(float)
        block_states = states[block_mask].astype(int)
        block_set_sizes = set_sizes[block_mask].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Count visits per state-action for novelty
        counts = np.zeros((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Compute novelty bonuses for actions in state s
            bonus_scale = novelty_bonus * (1.0 - age_explore_decay * older)
            bonus_vec = bonus_scale / np.sqrt(counts[s, :] + 1.0)

            Q_s = q[s, :].copy() + bonus_vec
            W_s = w[s, :].copy()

            # RL (with novelty) policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: stronger WM under low load; mild age penalty
            load_scale = 3.0 / float(nS)
            wm_mix = np.clip(wm_weight * load_scale * (1.0 - 0.3 * older), 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update (no age modulation here; age acts via exploration term)
            q[s, a] += lr * (r - q[s, a])

            # WM update: store rewarded action as sharp; otherwise decay toward uniform
            if r > 0.5:
                peaked = (1.0 / nA) * np.ones(nA)
                peaked[a] = 1.0
                peaked /= np.sum(peaked)
                w[s, :] = peaked
            else:
                decay = 0.4 + 0.3 * (nS - 3) / 3.0  # more decay under higher load
                w[s, :] = (1.0 - decay) * W_s + decay * w_0[s, :]

            # Update counts after observing action
            counts[s, a] += 1.0

        blocks_log_p += log_p

    return -blocks_log_p
