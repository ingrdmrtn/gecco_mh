def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 1: RL + WM mixture with set-size-scaled WM temperature and plasticity
    - RL: tabular Q-learning with softmax policy.
    - WM: probability map over actions per state that is sharpened by reward and slightly repelled by non-reward.
    - Arbitration: convex mixture of WM and RL policies; WM softmax temperature degrades with set size.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate for Q updates.
    - wm_weight0: float in [0,1], baseline WM mixture weight.
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - wm_temp_base: float > 0, base inverse temperature for WM (larger = more deterministic WM).
    - ss_exponent: float, exponent for set-size scaling of WM temperature: beta_wm_eff = wm_temp_base * (3/nS)^ss_exponent.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight0, softmax_beta, wm_temp_base, ss_exponent = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic (will be scaled by set size below)
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: softmax over W_s with set-size-scaled inverse temperature
            scale = (3.0 / float(nS)) ** ss_exponent
            beta_wm_eff = max(1e-6, wm_temp_base * scale)
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture
            wm_weight_eff = np.clip(wm_weight0, 0.0, 1.0)
            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM updating:
            # - Reward: sharpen toward one-hot on chosen action (more when scale is high, i.e., small set size)
            # - No reward: push probability away from chosen action slightly
            if r > 0.5:
                # Move distribution toward one-hot on a with step proportional to scale
                step = np.clip(0.5 * scale, 0.0, 1.0)
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1 - step) * W_s + step * target
            else:
                # Reduce chosen action prob, renormalize; magnitude shrinks with set size
                step = np.clip(0.1 * scale, 0.0, 0.2)
                w[s, a] = max(0.0, W_s[a] - step)
                row_sum = np.sum(w[s, :])
                if row_sum <= 1e-8:
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 2: RL with asymmetric learning + WM gating by surprise and load-dependent decay
    - RL: Q-learning with separate learning rates for positive vs negative prediction errors.
    - WM: categorical store that locks onto rewarded associations; decays toward uniform with set-size-dependent leak.
    - Arbitration: WM-RL mixture; WM weight is increased when WM is confident (low entropy) and when surprise is high.

    Parameters (model_parameters):
    - alpha_pos: float in [0,1], RL learning rate for positive PE (r - Q > 0).
    - alpha_neg: float in [0,1], RL learning rate for negative PE (r - Q < 0).
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - wm_weight0: float in [0,1], baseline mixture weight on WM.
    - gate_slope: float >= 0, slope controlling upweighting of WM by surprise and confidence.
    - wm_decay: float in [0,1], base WM leak per trial toward uniform (scaled up under larger set size).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    alpha_pos, alpha_neg, softmax_beta, wm_weight0, gate_slope, wm_decay = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy (deterministic softmax on W_s)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute WM confidence (low entropy -> high confidence)
            eps = 1e-12
            entropy = -np.sum(W_s * np.log(W_s + eps))
            max_entropy = np.log(nA)
            conf = 1.0 - (entropy / max_entropy)  # in [0,1]

            # Compute RL surprise: absolute PE if reward observed
            pe = r - Q_s[a]
            surprise = min(1.0, abs(pe))  # normalized in [0,1] because r in {0,1} and Q in [0,1] typically

            # Effective WM weight: baseline + sigmoid upweighting by conf*surprise
            up = 1.0 / (1.0 + np.exp(-gate_slope * (conf * surprise)))
            wm_weight_eff = np.clip(wm_weight0 * up, 0.0, 1.0)

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update with asymmetric learning rates
            if pe >= 0:
                q[s][a] += alpha_pos * pe
            else:
                q[s][a] += alpha_neg * pe

            # WM decay toward uniform, stronger for larger set sizes
            scale_leak = float(nS) / 3.0  # >1 for nS=6
            leak = np.clip(wm_decay * scale_leak, 0.0, 1.0)
            w = (1.0 - leak) * w + leak * w_0

            # WM gated storage: on reward and positive surprise, commit to one-hot; else slight demotion
            if r > 0.5 and pe > 0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Small demotion of the chosen action; magnitude reduced when WM already confident
                demote = (0.1 * (1.0 - conf)) / scale_leak
                w[s, a] = max(0.0, w[s, a] - demote)
                row_sum = np.sum(w[s, :])
                if row_sum <= 1e-8:
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 3: RL + WM with rehearsal and set-size-dependent noise and lapses
    - RL: tabular Q-learning with softmax policy.
    - WM: probability map that benefits from rehearsal when a state repeats and is noisier under larger set sizes.
    - Arbitration: mixture with a small lapse to uniform responding; WM temperature degrades with set size.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - wm_weight0: float in [0,1], baseline WM mixture weight.
    - rehearsal_gain: float in [0,1], extra sharpening of WM when the same state repeats on consecutive trials.
    - noise_ss: float >= 0, scales how much WM temperature declines with set size: beta_wm_eff = 50 / (1 + noise_ss*(nS-3)).
    - lapse: float in [0,0.2], lapse probability mixing with uniform to capture random errors.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight0, rehearsal_gain, noise_ss, lapse = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic (will be adjusted by noise_ss and set size)
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        prev_s = None
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: softmax over W_s with set-size-dependent noise
            beta_wm_eff = softmax_beta_wm / (1.0 + noise_ss * max(0, nS - 3))
            beta_wm_eff = max(1e-6, beta_wm_eff)
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture with lapse
            wm_weight_eff = np.clip(wm_weight0, 0.0, 1.0)
            p_mix = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM baseline decay slightly toward uniform, stronger under larger set size
            decay = np.clip(0.05 * (1.0 + noise_ss * max(0, nS - 3)), 0.0, 1.0)
            w = (1.0 - decay) * w + decay * w_0

            # WM learning: reward sharpens to one-hot; no-reward demotes chosen action
            if r > 0.5:
                # sharpen
                step = 0.6
                target = np.zeros(nA); target[a] = 1.0
                w[s, :] = (1 - step) * W_s + step * target
            else:
                demote = 0.1 / (1.0 + noise_ss * max(0, nS - 3))
                w[s, a] = max(0.0, w[s, a] - demote)
                row_sum = np.sum(w[s, :])
                if row_sum <= 1e-8:
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= row_sum

            # Rehearsal: if the same state repeats, further sharpen current WM trace
            if prev_s is not None and s == prev_s:
                gain = np.clip(rehearsal_gain, 0.0, 1.0)
                w[s, :] = (1 - gain) * w[s, :] + gain * np.eye(nA)[a]

            prev_s = s

        blocks_log_p += log_p

    return -blocks_log_p
