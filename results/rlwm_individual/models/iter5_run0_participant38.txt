def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + capacity-limited WM with confidence-gated arbitration.

    Mechanism:
    - RL learns action values per state via a Rescorla-Wagner rule (single learning rate).
    - WM stores a probabilistic mapping for each state that is pushed toward one-hot on reward
      and relaxes via decay otherwise (single parameter controls both consolidation and decay).
    - Arbitration weight for WM is reduced under higher set-size (capacity limit) and increased
      when the WM distribution for the current state is peaked (confidence-gated).

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - wm_weight0: Baseline WM arbitration weight in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - capacity_C: WM capacity (in number of states; e.g., between 1 and 6).
    - wm_decay: WM rate in [0,1]; toward one-hot on reward and toward uniform on no-reward.
    - conf_slope: Slope (>0) of confidence gating; larger means more boost when WM is peaked.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight0, softmax_beta, capacity_C, wm_decay, conf_slope = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Capacity-based downweighting (harder to use WM when nS > capacity)
        cap_factor = min(1.0, max(0.0, float(capacity_C) / float(nS)))  # in [0,1]

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Confidence gating based on how peaked WM is for state s
            peak = np.max(W_s)
            baseline_peak = 1.0 / nA
            conf = 1.0 / (1.0 + np.exp(-conf_slope * (peak - baseline_peak)))  # in (0,1)

            wm_weight_t = wm_weight0 * cap_factor * conf
            wm_weight_t = max(0.0, min(1.0, wm_weight_t))

            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: consolidate on reward, relax to uniform on no-reward
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * target
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Normalize for numerical stability
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with UCB-like directed exploration + WM with load-driven interference.

    Mechanism:
    - RL: standard Rescorla-Wagner learning, but the choice policy uses Q augmented by a
      visit-count uncertainty bonus (UCB-like), promoting exploration of less-sampled actions.
    - WM: one-shot supervised learning toward the rewarded action; across trials, WM suffers
      load-driven interference that drifts all states toward uniform as set size increases.
    - Arbitration: fixed WM weight per block (parameter), mixing WM and RL policies.

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - wm_weight: Mixture weight in [0,1] for WM policy.
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - explore_c: Strength (>0) of directed exploration bonus for RL (UCB-like).
    - gamma_int: Base interference rate for WM in [0,1], scaled by load each trial.
    - wm_learn: WM supervised learning rate toward one-hot on reward in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, explore_c, gamma_int, wm_learn = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Visit counts for UCB-like exploration
        n_counts = np.zeros((nS, nA)) + 1e-6  # avoid div by zero

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with UCB-like bonus
            Q_s = q[s, :].copy()
            bonus = explore_c / np.sqrt(n_counts[s, :] + 1.0)
            Q_s_eff = Q_s + bonus
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s_eff - Q_s_eff[a])))

            # WM policy from current WM map
            W_s = w[s, :]
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            wm_weight_t = max(0.0, min(1.0, wm_weight))
            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update uses true Q (without bonus)
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update: one-shot toward correct when rewarded
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target

            # Load-driven global interference each trial (affects all states)
            # Scale interference with load beyond 3 items
            load_scale = max(0.0, (nS - 3.0) / 3.0)  # 0 for nS<=3, up to 1 for nS=6
            int_rate = max(0.0, min(1.0, gamma_int * (1.0 + load_scale)))
            w = (1.0 - int_rate) * w + int_rate * w_0

            # Renormalize WM distributions
            w = np.clip(w, 1e-8, 1.0)
            w /= np.sum(w, axis=1, keepdims=True)

            # Update counts
            n_counts[s, a] += 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with recency-gated arbitration and load scaling.

    Mechanism:
    - RL learns per-state via Rescorla-Wagner (single learning rate).
    - WM stores mappings; after reward, WM moves toward one-hot; after non-reward, it decays.
    - Arbitration weight for WM depends on how recently the state was seen (exponential recency),
      and is further scaled down under higher load (nS).

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - wm_weight: Baseline WM weight in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - recency_tau: Time constant (>0) for exponential recency gating (in trials).
    - wm_learn: WM learning rate toward one-hot after reward in [0,1].
    - wm_decay: WM decay rate toward uniform after non-reward in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, recency_tau, wm_learn, wm_decay = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Track last-seen trial index per state for recency gating
        last_seen = -1 * np.ones(nS, dtype=int)

        # Load scaling (reduce WM reliance when nS is larger)
        load_scale = min(1.0, 3.0 / float(nS))  # 1 for nS=3, 0.5 for nS=6

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Recency gating: more WM when the same state was seen recently
            if last_seen[s] >= 0:
                dt = max(0, t - int(last_seen[s]))
                rec = np.exp(-float(dt) / max(1e-6, recency_tau))
            else:
                rec = 0.0

            wm_weight_t = wm_weight * rec * load_scale
            wm_weight_t = max(0.0, min(1.0, wm_weight_t))

            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: learn on reward, decay on non-reward
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

            # Update last seen time for this state
            last_seen[s] = t

        blocks_log_p += log_p

    return -blocks_log_p
