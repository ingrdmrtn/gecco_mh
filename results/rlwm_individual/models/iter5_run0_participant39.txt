def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + graded WM with confidence- and load-dependent arbitration, asymmetric RL learning.

    Mechanisms
    - RL: delta rule with asymmetric learning rates (positive vs negative prediction errors).
    - WM: graded action traces per state that decay over same-state revisits; WM policy is softmax
      with its own precision.
    - Arbitration: trial-wise WM weight is a logistic function of WM confidence (state-specific)
      penalized by set size (higher load reduces WM control).
    - Load effect: both arbitration and WM precision are sensitive to set size.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Feedback per trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size per trial (constant within a block).
    model_parameters : tuple/list of 6 floats
        lr_pos : RL learning rate for positive prediction errors in [0,1].
        lr_neg : RL learning rate for negative prediction errors in [0,1].
        softmax_beta : Base inverse temperature for RL (rescaled internally by *10).
        wm_decay : WM decay applied to the active state's WM vector on each visit in [0,1].
        arb_slope : Gain of sigmoid arbitration that maps WM confidence to WM weight (>0).
        wm_beta0 : Baseline WM inverse temperature (precision) for WM softmax (>0).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, softmax_beta, wm_decay, arb_slope, wm_beta0 = model_parameters
    softmax_beta *= 10  # higher upper bound as in template
    softmax_beta_wm = 50  # base deterministic WM policy (unused directly; we use wm_beta0)

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: graded vector; use its own precision scaled by load
            # Effective WM precision decreases with set size
            beta_wm_eff = wm_beta0 / max(1.0, (nS / 3.0))
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Arbitration weight: logistic on WM confidence, penalized by load
            # Confidence: how peaked is W_s (range in [0,1]): max - mean
            conf = float(np.max(W_s) - np.mean(W_s))
            # Load penalty grows with set size; mapped to [0,1] roughly
            load_pen = max(0.0, (nS - 3) / 3.0)
            wm_weight = 1.0 / (1.0 + np.exp(-arb_slope * (conf - load_pen)))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            delta = r - Q_s[a]
            eta = lr_pos if delta >= 0 else lr_neg
            q[s][a] += eta * delta

            # WM update on the visited state
            # Decay toward uniform, then incorporate current outcome
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                # Move WM toward a one-hot memory of the rewarded action
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * onehot

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WSLS-style WM with load-scaled WM control and action stickiness in WM policy.

    Mechanisms
    - RL: standard delta rule (single learning rate) with softmax.
    - WM (WSLS): working memory implements a within-state win-stay/lose-shift heuristic.
      If last outcome in the state was reward, prefer repeating last action; else prefer
      switching away from it. This is implemented as a WM softmax over a bias vector.
    - Arbitration: fixed base WM weight reduced monotonically by set size via an exponential.
    - Stickiness in WM: additional bias to repeat the last action even after losses.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Feedback per trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size per trial (constant within a block).
    model_parameters : tuple/list of 6 floats
        lr : RL learning rate in [0,1].
        softmax_beta : RL inverse temperature (rescaled by *10).
        wm_weight0 : Base WM weight in [0,1] before load scaling.
        wsls_beta : WM inverse temperature controlling strength of WSLS bias (>0).
        load_slope : Positive slope; effective WM weight is wm_weight0 * exp(-load_slope*(nS-3)).
        stickiness : Additive WM bias for the last action regardless of outcome (>=0).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_weight0, wsls_beta, load_slope, stickiness = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50  # defined by template; WM precision uses wsls_beta

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))  # unused as a store; we'll still maintain a bias vector shape
        w_0 = (1 / nA) * np.ones((nS, nA))

        last_action = -1 * np.ones(nS, dtype=int)
        last_reward = np.zeros(nS)

        # Load-scaled WM mixture weight
        wm_weight = wm_weight0 * np.exp(-load_slope * max(0, nS - 3))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Construct WSLS WM bias vector for this state
            W_s = np.zeros(nA)
            if last_action[s] >= 0:
                # Start with stickiness toward last action
                W_s[last_action[s]] += stickiness
                if last_reward[s] > 0.5:
                    # Win-stay: boost last action
                    W_s[last_action[s]] += 1.0
                else:
                    # Lose-shift: penalize last action and softly boost alternatives
                    W_s[last_action[s]] -= 1.0
                    alt_idx = [i for i in range(nA) if i != last_action[s]]
                    for j in alt_idx:
                        W_s[j] += 0.5 / (nA - 1)
            else:
                # No memory yet: use uniform baseline (encoded by zeros here)
                pass

            # WM softmax policy with its own precision
            p_wm = 1.0 / np.sum(np.exp(wsls_beta * (W_s - W_s[a])))

            # Mixture policy with load-scaled WM weight
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM "update": store last action and outcome for WSLS
            last_action[s] = a
            last_reward[s] = r

            # Also keep w matrix consistent with template variables by encoding W_s in w
            # (not used in policy beyond this trial)
            w[s, :] = w_0[s, :] + W_s

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with uncertainty-adaptive temperature + WM gated by surprise and load.

    Mechanisms
    - RL: delta rule with fixed learning rate. RL softmax temperature adapts to
      state-visit uncertainty: early (uncertain) visits use lower inverse temperature,
      later visits use higher inverse temperature.
    - WM: one-shot storage of rewarded action per state that decays when revisited.
      WM influence is upregulated by surprise (absolute prediction error) and downregulated
      by set size.
    - Arbitration: WM weight is wm_weight0 * (1/nS) scaled by a sigmoid of surprise.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Feedback per trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size per trial (constant within a block).
    model_parameters : tuple/list of 6 floats
        lr : RL learning rate in [0,1].
        softmax_beta0 : Baseline RL inverse temperature (rescaled by *10).
        wm_weight0 : Base WM weight scale in [0,1].
        surprise_gain : Gain of sigmoid that maps absolute PE to extra WM control (>0).
        wm_forget : WM decay on the visited state's WM vector in [0,1].
        beta_unc_gain : How much RL inverse temperature grows with visits (>0).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, softmax_beta0, wm_weight0, surprise_gain, wm_forget, beta_unc_gain = model_parameters
    softmax_beta = softmax_beta0 * 10
    softmax_beta_wm = 50  # base; WM policy uses this high precision on stored mappings

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        visits = np.zeros(nS)  # count visits per state

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            visits[s] += 1.0

            # RL policy with uncertainty-adaptive inverse temperature
            # More visits -> higher beta; first visit uses baseline
            beta_rl_eff = softmax_beta * (1.0 + beta_unc_gain * (visits[s] - 1.0))
            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(beta_rl_eff * (Q_s - Q_s[a])))

            # WM policy: use WM softmax over the WM vector
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute surprise (absolute PE) for arbitration
            pe = r - Q_s[a]
            surprise = abs(pe)
            # Base WM weight penalized by load and boosted by surprise
            wm_weight_base = (wm_weight0 / max(1.0, float(nS)))
            wm_weight = wm_weight_base * (1.0 / (1.0 + np.exp(-surprise_gain * (surprise - 0.5))))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            q[s][a] += lr * pe

            # WM update on visited state: decay then set if rewarded
            # Decay toward uniform (forgetting)
            w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # Overwrite more strongly on reward to store mapping
                w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * onehot

        blocks_log_p += log_p

    return -blocks_log_p
