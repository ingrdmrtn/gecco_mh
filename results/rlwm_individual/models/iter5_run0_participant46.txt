def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 1: RL + WM mixture with surprise-gated WM and load-adjusted WM precision
    - RL: tabular Q-learning with softmax policy (provided in template).
    - WM: fast, one-shot associative store, with leak and set-size-adjusted precision.
    - Arbitration: WM weight is baseline-scaled and gated by unsigned prediction error (surprise).
      WM precision increases in small set sizes and decreases in large set sizes.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate for Q updates.
    - wm_weight: float in [0,1], baseline WM mixture weight.
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - wm_decay: float in [0,1], per-trial WM leak toward uniform; stronger leak yields weaker WM.
    - wm_temp3: float >= 0, scales WM softmax precision at set size 3; precision scales with (3/nS).
    - gate_tau: float >= 0, steepness of surprise gating; higher values make gating more sensitive to PE.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_temp3, gate_tau = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound (RL)
    softmax_beta_wm = 50  # base WM precision; will be rescaled by set size and wm_temp3
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: set-size-adjusted precision; sharper at small nS
            beta_wm_eff = softmax_beta_wm * max(0.0, wm_temp3) * (3.0 / float(nS))
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Surprise-gated WM arbitration: larger unsigned PE => more WM
            pe = abs(r - Q_s[a])
            gate = 1.0 / (1.0 + np.exp(-gate_tau * (pe - 0.5)))  # centered around 0.5
            wm_weight_eff = np.clip(wm_weight * gate, 0.0, 1.0)

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM updating with leak and error-based shaping
            # Global leak toward uniform (scaled by set size to mimic load)
            leak = np.clip(wm_decay * (float(nS) / 3.0), 0.0, 1.0)
            w = (1.0 - leak) * w + leak * w_0

            # If rewarded: commit one-shot to the chosen action
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # If not rewarded: shift probability mass away from a, renormalize
                move = 0.25 * (3.0 / float(nS))  # smaller move under higher load
                w[s, a] = max(0.0, w[s, a] - move)
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 2: RL with asymmetric learning + capacity-limited WM + perseverative WM trace
    - RL: tabular Q-learning with asymmetric learning rates for reward vs no-reward.
    - WM: capacity/share-limited store that favors the last action (perseveration) and
           locks in one-shot when rewarded. Effective WM availability declines with set size.
    - Arbitration: mixture of WM and RL, with WM weight scaled by effective capacity.

    Parameters (model_parameters):
    - lr_pos: float in [0,1], RL learning rate when r = 1.
    - wm_weight: float in [0,1], baseline WM mixture weight before capacity scaling.
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - lr_neg: float in [0,1], RL learning rate when r = 0.
    - capacity_share: float in [0,1], fraction of states effectively encoded in WM at set size 3.
                      Effective inclusion scales as min(1, capacity_share * 3/nS).
    - perseveration: float in [0,1], strength of updating WM toward the most recent chosen action
                     on error trials (perseverative tilt).

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr_pos, wm_weight, softmax_beta, lr_neg, capacity_share, perseveration = model_parameters
    softmax_beta *= 10  # RL temperature
    softmax_beta_wm = 50  # WM is relatively deterministic when strongly committed
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))  # WM distribution per state
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Effective probability that a state is within WM given capacity and set size
        p_incl = min(1.0, max(0.0, capacity_share) * (3.0 / float(nS)))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: if state is included, use w[s] distribution with high precision; otherwise, near-uniform
            # Implement inclusion effect by softmax temperature scaling
            beta_wm_eff = softmax_beta_wm * p_incl
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Arbitration: WM weight scaled by inclusion probability
            wm_weight_eff = np.clip(wm_weight * p_incl, 0.0, 1.0)

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # Asymmetric RL update
            if r > 0.5:
                q[s][a] += lr_pos * (1.0 - Q_s[a])
            else:
                q[s][a] += lr_neg * (0.0 - Q_s[a])

            # WM leak toward uniform increases when not included strongly
            leak = np.clip((1.0 - p_incl), 0.0, 1.0)
            w = (1.0 - leak) * w + leak * w_0

            # WM update:
            # - If rewarded: commit one-shot to chosen action
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # - If not rewarded: perseverative tilt toward the last chosen action
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - perseveration) * w[s, :] + perseveration * onehot
                # renormalize for safety
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 3: RL with decay + WM as count-based uncertainty-driven policy
    - RL: tabular Q-learning with decay-forgetting and softmax choice.
    - WM: tracks visit counts per state-action and favors uncertain (less-sampled) actions,
           implemented via a softmax over uncertainty bonuses 1/sqrt(N+1).
    - Arbitration: mixture of WM (directed exploration) and RL, with WM weight increasing under load.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], baseline WM mixture weight at set size 3.
    - softmax_beta: float >= 0, inverse temperature for RL softmax (internally scaled by 10).
    - decay_rl: float in [0,1], multiplicative decay of Q values per trial (forgetting).
    - lambda_unc: float >= 0, scale on the WM uncertainty bonus strength.
    - wm_temp_base: float >= 0, scales WM softmax precision; interacts with set size.

    Returns:
    - Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, decay_rl, lambda_unc, wm_temp_base = model_parameters
    softmax_beta *= 10  # RL temperature
    softmax_beta_wm = 50  # base scale for WM; combined with wm_temp_base
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))  # unused for storage here, kept for template consistency
        w_0 = (1 / nA) * np.ones((nS, nA))
        # Count matrix for uncertainty (WM component)
        counts = np.zeros((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            # Uncertainty vector: higher for less-sampled actions
            U_s = lambda_unc * (1.0 / np.sqrt(counts[s, :] + 1.0))

            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: softmax over uncertainty (directed exploration)
            beta_wm_eff = softmax_beta_wm * max(0.0, wm_temp_base) * (float(nS) / 3.0)
            # Reuse the same chosen-centered probability computation
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (U_s - U_s[a])))

            # Arbitration: WM weight increases with set size (more exploration pressure)
            wm_weight_eff = np.clip(wm_weight * (float(nS) / 3.0), 0.0, 1.0)

            p_total = p_wm*wm_weight_eff + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL decay-forgetting then update
            q *= (1.0 - decay_rl)
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # Update counts (WM uncertainty tracker)
            counts[s, a] += 1.0

            # Keep w consistent (not used as memory here): softly track uncertainty distribution
            # Map uncertainty U_s to a normalized distribution for interpretability
            u_row = U_s.copy()
            u_row = np.maximum(u_row, 1e-8)
            u_row /= np.sum(u_row)
            w[s, :] = 0.5 * w[s, :] + 0.5 * u_row

        blocks_log_p += log_p

    return -blocks_log_p
