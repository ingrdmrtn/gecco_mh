def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with uncertainty-weighted arbitration and load-dependent WM decay.

    Idea:
    - Choices are a mixture of RL and WM policies.
    - Arbitration weight for WM depends on WM certainty (low entropy => higher weight),
      and is penalized by higher set size (load).
    - WM contents decay toward uniform at a rate that increases with load.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_bias: real, baseline logit for WM weight in the arbitration.
    - softmax_beta: float >= 0, base inverse temperature for RL; internally scaled by 10.
    - entropy_gain: real >= 0, how strongly low WM entropy increases WM weight.
      Effective term: wm_weight = sigmoid(wm_bias + entropy_gain*(H_ref - H_wm) - load_term)
      where H_wm is entropy of WM policy and H_ref is reference max entropy.
    - decay_load: float >= 0, how strongly WM decays with load (set size).
      Effective decay per state encounter: wm_decay_eff = clip(decay_load * ((nS - 3)/3), 0, 1).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_bias, softmax_beta, entropy_gain, decay_load = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        # Load-dependent terms
        load_scaled = max(0.0, (nS - 3) / 3.0)
        wm_decay_eff = min(1.0, max(0.0, decay_load * load_scaled))
        H_ref = np.log(nA)  # maximum entropy reference (uniform)

        # Value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl

            # WM policy and its entropy (confidence)
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            wm_probs = np.exp(wm_logits)
            wm_probs /= np.sum(wm_probs)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm

            # Entropy of WM distribution
            H_wm = -np.sum(wm_probs * np.log(np.clip(wm_probs, 1e-12, 1.0)))

            # Arbitration: higher weight when WM is confident (low entropy), reduced by load
            wm_logit = wm_bias + entropy_gain * (H_ref - H_wm) - load_scaled
            wm_weight_eff = 1.0 / (1.0 + np.exp(-wm_logit))

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform (load dependent)
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]

            # WM storage: on reward, store chosen action as the dominant association
            if r > 0.0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with load-limited WM precision and within-state refresh.

    Idea:
    - Choices are a mixture of RL and WM policies.
    - WM precision (inverse temperature) decreases with larger set sizes.
    - When the same state repeats, WM is "refreshed" by sharpening its distribution toward
      its current best action (even without external feedback).

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - wm_weight: float in [0,1], mixture weight on WM.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - wm_inv_cap: float >= 0, load sensitivity of WM precision.
      Effective WM beta: beta_wm_eff = 50 / (1 + wm_inv_cap * (nS - 3)/3).
    - wm_refresh: float in [0,1], sharpening toward WM's current best action when the
      same state repeats consecutively (models attention-based refresh).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_inv_cap, wm_refresh = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm_base = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        # Load-limited WM precision
        load_scaled = max(0.0, (nS - 3) / 3.0)
        beta_wm_eff = softmax_beta_wm_base / (1.0 + wm_inv_cap * load_scaled)

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        last_state = None

        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Refresh if same state repeats: sharpen W_s toward its current best action
            if last_state is not None and s == last_state:
                best_a = int(np.argmax(w[s, :]))
                sharpen = (1.0 - wm_refresh) * w[s, :] + wm_refresh * np.eye(nA)[best_a]
                w[s, :] = sharpen

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl

            # WM policy with load-limited precision
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm

            wm_weight_eff = wm_weight
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: store rewarded associations strongly
            if r > 0.0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0

            last_state = s

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL+WM with load-reduced RL learning and WM cross-state interference.

    Idea:
    - Choices are a mixture of RL and WM policies.
    - RL learning rate decreases under higher load (more states).
    - WM updates for a rewarded state bleed (interfere) into other states' WM,
      increasing confusion under load.

    Parameters (model_parameters):
    - lr: float in [0,1], base RL learning rate at low load.
    - wm_weight: float in [0,1], mixture weight on WM.
    - softmax_beta: float >= 0, RL inverse temperature; internally scaled by 10.
    - lr_load: float >= 0, load sensitivity reducing RL learning under larger set sizes.
      Effective lr: lr_eff = lr / (1 + lr_load * (nS - 3)/3).
    - wm_interference: float in [0,1], magnitude of cross-state WM interference that
      scales with load: eff = wm_interference * (nS - 3)/3.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, lr_load, wm_interference = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        load_scaled = max(0.0, (nS - 3) / 3.0)
        lr_eff = lr / (1.0 + lr_load * load_scaled)
        wm_interf_eff = min(1.0, max(0.0, wm_interference * load_scaled))

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm

            wm_weight_eff = wm_weight
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with load-reduced learning
            delta = r - Q_s[a]
            q[s, a] += lr_eff * delta

            # WM update: store rewarded item in current state
            if r > 0.0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0

                # Cross-state interference: bleed into other states
                if wm_interf_eff > 0.0:
                    for s2 in range(nS):
                        if s2 == s:
                            continue
                        # Move other states slightly toward uniform, then add a small bias to action a
                        w[s2, :] = (1.0 - wm_interf_eff) * w[s2, :] + wm_interf_eff * w_0[s2, :]
                        w[s2, a] = min(1.0, w[s2, a] + wm_interf_eff / nA)

        blocks_log_p += log_p

    return -blocks_log_p
