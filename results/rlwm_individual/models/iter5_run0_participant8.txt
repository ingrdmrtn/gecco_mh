def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + error-driven WM gating with set-size interference.

    Idea
    - RL: standard Q-learning with softmax.
    - WM: a cached mapping per state; policy is near-deterministic readout from WM.
    - Arbitration: WM weight increases with surprise (|prediction error|) on the current trial,
      and decreases linearly with set size (3/nS).
    - WM interference/forgetting increases with set size.

    Parameters
    ----------
    model_parameters : (lr, wm_weight, softmax_beta, wm_gate_err, interference_rate)
        lr : float
            RL learning rate (0-1).
        wm_weight : float
            Base WM mixture weight (0-1).
        softmax_beta : float
            RL inverse temperature (scaled by 10 internally).
        wm_gate_err : float
            Sensitivity of WM engagement to absolute RL prediction error (>=0).
            Larger values strengthen WM when the outcome is surprising.
        interference_rate : float
            Rate of WM decay/interference that scales with set size (>=0).

    Returns
    -------
    float
        Negative log-likelihood of observed actions under the model.
    """
    lr, wm_weight, softmax_beta, wm_gate_err, interference_rate = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy (deterministic readout of cached mapping)
            p_wm_core = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Error-driven WM engagement: larger |delta| => stronger WM
            delta_peek = r - Q_s[a]
            gate_err = 1.0 - np.exp(-wm_gate_err * abs(delta_peek))
            # Set-size penalty (linear): 3/nS
            ss_scale = 3.0 / max(3.0, float(nS))
            wm_weight_eff = np.clip(wm_weight * gate_err * ss_scale, 0.0, 1.0)

            p_total = p_wm_core * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            log_p += np.log(p_total + 1e-12)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Set-sizeâ€“dependent forgetting/interference
            forget = 1.0 - np.exp(-interference_rate * (float(nS) / 3.0))

            # Baseline decay toward uniform for the visited state
            w[s, :] = (1.0 - forget) * W_s + forget * w_0[s, :]

            # If rewarded, strengthen the chosen mapping in WM
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # Blend toward one-hot mapping after reward
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot

            # Light global interference to other states increases with set size
            if nS > 3:
                leak = 0.1 * forget
                for s2 in range(nS):
                    if s2 == s:
                        continue
                    w[s2, :] = (1.0 - leak) * w[s2, :] + leak * w_0[s2, :]

            # Renormalize
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + entropy-based WM arbitration with precision-scaled WM readout and decay.

    Idea
    - RL: standard Q-learning with softmax.
    - WM: a cached mapping whose readout sharpness is scaled by a WM precision parameter.
      WM decays at a fixed rate.
    - Arbitration: WM weight increases when WM is confident (low entropy) and decreases
      with set size (3/nS). WM precision also controls how strongly rewards update WM.

    Parameters
    ----------
    model_parameters : (lr, wm_weight, softmax_beta, wm_precision, decay_wm)
        lr : float
            RL learning rate (0-1).
        wm_weight : float
            Base WM mixture weight (0-1).
        softmax_beta : float
            RL inverse temperature (scaled by 10 internally).
        wm_precision : float
            Scales WM inverse temperature for policy and strength of WM learning (>=0).
        decay_wm : float
            Per-visit WM decay rate toward uniform (0-1).

    Returns
    -------
    float
        Negative log-likelihood of observed actions under the model.
    """
    lr, wm_weight, softmax_beta, wm_precision, decay_wm = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic (will be scaled by wm_precision)
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Precision-scaled WM readout
            beta_wm_eff = softmax_beta_wm * max(0.0, wm_precision)
            p_wm_core = 1 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Entropy-based arbitration (lower entropy => higher WM weight)
            W_safe = np.clip(W_s, 1e-12, 1.0)
            H = -np.sum(W_safe * np.log(W_safe)) / np.log(nA)  # normalized entropy in [0,1]
            confidence = 1.0 - H
            ss_scale = 3.0 / max(3.0, float(nS))
            wm_weight_eff = np.clip(wm_weight * confidence * ss_scale, 0.0, 1.0)

            p_total = p_wm_core * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            log_p += np.log(p_total + 1e-12)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform for the visited state
            w[s, :] = (1.0 - decay_wm) * W_s + decay_wm * w_0[s, :]

            # Reward-driven WM strengthening scaled by precision
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                k = 1.0 - np.exp(-max(0.0, wm_precision))  # in (0,1) increasing with precision
                w[s, :] = (1.0 - k) * w[s, :] + k * one_hot

            # Renormalize
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with global WM leak, set-size down-weighting, and action perseveration in RL policy.

    Idea
    - RL: standard Q-learning with softmax, augmented by a perseveration bonus added to
      the previously chosen action in the same state (choice stickiness).
    - WM: leaky memory that drifts toward uniform each trial; reward strengthens the
      chosen mapping. WM contribution is reduced in larger set sizes.
    - Arbitration: fixed base WM weight scaled by set size factor 1/(1 + ss_bonus * (nS-3)).

    Parameters
    ----------
    model_parameters : (lr, wm_weight, softmax_beta, tau_wm, perseveration, ss_bonus)
        lr : float
            RL learning rate (0-1).
        wm_weight : float
            Base WM mixture weight (0-1).
        softmax_beta : float
            RL inverse temperature (scaled by 10 internally).
        tau_wm : float
            Global WM leak toward uniform applied every trial (0-1).
        perseveration : float
            Additive bonus to the Q-value of the last chosen action in the current state (any real).
        ss_bonus : float
            Degree to which larger set sizes down-weight WM: weight /= (1 + ss_bonus*(nS-3)) (>=0).

    Returns
    -------
    float
        Negative log-likelihood of observed actions under the model.
    """
    lr, wm_weight, softmax_beta, tau_wm, perseveration, ss_bonus = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Track last action taken per state for perseveration
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Apply global WM leak every trial (all states)
            if tau_wm > 0:
                w = (1.0 - tau_wm) * w + tau_wm * w_0

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy with perseveration bonus on last action for this state
            Q_eff = Q_s.copy()
            if last_action[s] >= 0:
                Q_eff[last_action[s]] = Q_eff[last_action[s]] + perseveration
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_eff - Q_eff[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm_core = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Set-size down-weighting of WM
            denom = 1.0 + max(0.0, ss_bonus) * max(0.0, float(nS) - 3.0)
            wm_weight_eff = np.clip(wm_weight / denom, 0.0, 1.0)

            p_total = p_wm_core * wm_weight_eff + (1 - wm_weight_eff) * p_rl
            log_p += np.log(p_total + 1e-12)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Reward strengthens the chosen mapping with a strength linked to tau_wm (learn_strength)
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                learn_strength = np.clip(2.0 * max(0.0, tau_wm), 0.0, 1.0)
                w[s, :] = (1.0 - learn_strength) * W_s + learn_strength * one_hot
            else:
                # No reward: maintain current WM after global leak
                w[s, :] = w[s, :]

            # Renormalize
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

            # Update last action
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
