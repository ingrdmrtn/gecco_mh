def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with eligibility traces and load-driven interference.

    Mechanism:
    - RL: delta-rule with eligibility traces across state-action pairs, allowing
      recent choices to receive more credit (trace decays each trial).
    - WM: near one-hot associative memory per state. WM precision degrades with set size
      (softmax precision scales down) and suffers global interference that blends the WM
      table toward uniform each trial as a function of load.
    - Policy: mixture of RL and WM using fixed wm_weight from the template.

    Parameters:
    - lr: [0,1] RL learning rate for value updates.
    - wm_weight: [0,1] policy mixture weight on WM (1 = all WM, 0 = all RL).
    - softmax_beta: inverse temperature for RL softmax (internally scaled by 10).
    - trace_lambda: [0,1] eligibility trace decay. Higher values spread credit over recent trials.
    - interference_beta: >=0, scales load-driven WM interference and reduces WM precision with set size.
    - wm_learn: [0,1] WM learning rate toward a one-hot code for rewarded actions (and away for unrewarded).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, trace_lambda, interference_beta, wm_learn = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Eligibility traces for RL
        e = np.zeros((nS, nA))

        # Load-dependent factors
        load_factor = max(0.0, nS - 1)  # 0 for nS=1; larger as set size increases
        # Decrease WM softmax precision with load
        beta_wm_eff = softmax_beta_wm / (1.0 + interference_beta * load_factor)
        # Global interference strength per trial (toward uniform)
        glob_interf = 1.0 - np.exp(-interference_beta * load_factor / max(nS, 1.0))
        glob_interf = np.clip(glob_interf, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM policy: sharpen row for current state with load-degraded precision
            W_s = w[s, :].copy()
            denom_wm = np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update with eligibility trace
            delta = r - Q_s[a]
            # Decay all eligibilities and set current to 1 (replacing trace)
            e *= trace_lambda
            e[s, :] *= 0.0
            e[s, a] = 1.0
            q += lr * delta * e

            # WM global interference toward uniform (load-driven)
            w = (1.0 - glob_interf) * w + glob_interf * w_0

            # WM associative update: move chosen action toward one-hot if rewarded,
            # and away from chosen if not rewarded (credit/anti-credit assignment).
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            if r > 0.5:
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * one_hot
            else:
                # Push probability mass away from chosen toward the other actions
                not_a = np.ones(nA) / (nA - 1.0)
                not_a[a] = 0.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * not_a

            # Keep rows normalized to sum to 1 for numerical stability
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with load-adaptive arbitration embedded in the WM policy and fast WM encoding.

    Mechanism:
    - RL: standard delta-rule with softmax.
    - WM: fast encoding of recent correct actions with learning rate wm_eta.
      The WM policy internally blends with the RL policy depending on set size via a logistic
      function (so WM relies more on RL under high load).
    - Policy: template mixture uses wm_weight; here p_wm already contains arbitration.

    Parameters:
    - lr: [0,1] RL learning rate.
    - wm_weight: [0,1] external policy mixture with the RL policy (template).
                 Note: p_wm itself already blends WM and RL by load.
    - softmax_beta: RL inverse temperature (scaled by 10).
    - load_midpoint: midpoint of the logistic in set-size units (around which WM reliance halves).
    - load_slope: >0, steepness of the logistic that maps set size to WM reliance.
    - wm_eta: [0,1] WM learning rate to encode outcomes into the WM table.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, load_midpoint, load_slope, wm_eta = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent arbitration within the WM policy
        # WM reliance decreases with set size via logistic
        # a_load in [0,1], high for small nS, low for large nS
        a_load = 1.0 / (1.0 + np.exp(load_slope * (nS - load_midpoint)))

        # Mild passive decay of WM to uniform per trial scales with load (optional)
        decay_load = 1.0 - np.exp(-0.1 * max(0, nS - 1))
        decay_load = np.clip(decay_load, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # Pure WM policy based on w
            W_s = w[s, :].copy()
            denom_wm_pure = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm_pure = 1.0 / max(denom_wm_pure, 1e-12)

            # Embed arbitration into WM policy: under higher load, defer toward p_rl
            p_wm = a_load * p_wm_pure + (1.0 - a_load) * p_rl

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM passive decay with load
            w = (1.0 - decay_load) * w + decay_load * w_0

            # WM fast encoding of outcome at current state
            one_hot = np.zeros(nA)
            if r > 0.5:
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_eta) * w[s, :] + wm_eta * one_hot
            else:
                # If incorrect, bias away from chosen action
                anti = np.ones(nA) / (nA - 1.0)
                anti[a] = 0.0
                w[s, :] = (1.0 - wm_eta) * w[s, :] + wm_eta * anti

            # Normalize WM row for stability
            ssum = np.sum(w[s, :])
            if ssum > 0:
                w[s, :] /= ssum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Dirichlet-like WM with PE-gated strengthening and load-scaled decay.

    Mechanism:
    - RL: standard delta-rule.
    - WM: each state's row acts like Dirichlet concentration parameters. The WM policy uses the
      expected categorical probabilities from the Dirichlet (row normalized), sharpened by a
      deterministic softmax. Concentrations decay toward a symmetric prior alpha0 uniformly, with
      decay increasing under higher load. Positive prediction error (surprise) gates stronger WM
      consolidation.
    - Policy: mixture of RL and WM via fixed wm_weight (template).

    Parameters:
    - lr: [0,1] RL learning rate.
    - wm_weight: [0,1] policy mixture weight on WM.
    - softmax_beta: RL inverse temperature (scaled by 10).
    - pe_sensitivity: >=0, scales how |prediction error| amplifies WM strengthening.
    - dirichlet_alpha0: >0, symmetric prior strength for WM concentrations (higher = flatter/less certain).
    - decay_wm: [0,1], base WM decay rate toward the prior each trial, scaled by load.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, pe_sensitivity, dirichlet_alpha0, decay_wm = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        # WM concentrations; start from symmetric prior
        w = dirichlet_alpha0 * np.ones((nS, nA))
        w_0 = dirichlet_alpha0 * np.ones((nS, nA))

        # Load-scaled decay factor toward prior
        load_scale = (nS - 1.0) / max(1.0, nS)  # 0 for 1, approaches 1 as nS grows
        decay_eff = np.clip(decay_wm * (0.25 + 0.75 * load_scale), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, 1e-12)

            # WM policy from Dirichlet expectations
            W_row = w[s, :]
            W_expect = W_row / max(np.sum(W_row), 1e-12)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_expect - W_expect[a])))
            p_wm = 1.0 / max(denom_wm, 1e-12)

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, 1e-12))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward symmetric prior (load-scaled)
            w = (1.0 - decay_eff) * w + decay_eff * w_0

            # PE-gated WM strengthening:
            # Larger |delta| increases the increment added to the chosen action.
            pe_gain = 1.0 + pe_sensitivity * abs(delta)
            if r > 0.5:
                # Strengthen chosen action concentration
                w[s, a] += pe_gain
            else:
                # For incorrect outcome, slightly increase non-chosen actions (exclusion)
                inc = pe_gain / (nA - 1.0)
                for ai in range(nA):
                    if ai != a:
                        w[s, ai] += inc

            # Keep concentrations positive and bounded away from 0
            w[s, :] = np.maximum(w[s, :], 1e-6)

        blocks_log_p += log_p

    return -blocks_log_p
