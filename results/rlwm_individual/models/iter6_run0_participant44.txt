def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with load-sensitive forgetting + capacity-limited WM with load-driven interference.

    Rationale:
    - RL values decay toward uniform more under higher set size (load), capturing poorer retention.
    - WM contributes a near-deterministic policy that decays toward uniform and suffers extra
      interference as set size increases.
    - Mixture of WM and RL determines action probabilities.

    Parameters
    ----------
    model_parameters : tuple/list of length 6
      lr               : RL learning rate (0..1).
      wm_weight_base   : Baseline WM mixture weight (0..1).
      softmax_beta     : RL inverse temperature (scaled internally by 10).
      rl_forget_base   : Baseline RL forgetting toward uniform per trial (0..1).
      load_sensitivity : Scales both RL forgetting and WM interference with set size (>=0).
      wm_decay         : Baseline WM decay toward uniform per trial (0..1).

    Returns
    -------
    neg_log_likelihood : float
      Negative log-likelihood of the observed action sequence.
    """
    lr, wm_weight_base, softmax_beta, rl_forget_base, load_sensitivity, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states  = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q   = (1.0 / nA) * np.ones((nS, nA))
        w   = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent rates
        load_level = max(nS - 3, 0) / 3.0  # 0 for 3-set, 1 for 6-set
        rl_forget = np.clip(rl_forget_base * (1.0 + load_sensitivity * load_level), 0.0, 1.0)
        wm_interf = np.clip(wm_decay * (1.0 + load_sensitivity * load_level), 0.0, 1.0)
        wm_weight = np.clip(wm_weight_base * (1.0 - 0.5 * load_level), 0.0, 1.0)  # less WM influence at high load

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probability for chosen action (as per template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM softmax policy (deterministic-ish)
            logits_wm = softmax_beta_wm * W_s
            max_wm = np.max(logits_wm)
            exp_wm = np.exp(logits_wm - max_wm)
            p_wm_vec = exp_wm / np.sum(exp_wm)
            p_wm = p_wm_vec[a]

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with learning and forgetting toward uniform
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            q = (1.0 - rl_forget) * q + rl_forget * w_0  # global small forgetting per trial

            # WM decay + interference, then update toward recent reward structure
            w = (1.0 - wm_interf) * w + wm_interf * w_0
            if r > 0.0:
                # On reward, strengthen current state-action in WM
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.6 * w[s, :] + 0.4 * one_hot
            else:
                # On no reward, gentle suppression of chosen action
                suppress = 0.2 * w[s, a]
                w[s, a] -= suppress
                redistribute = suppress / (nA - 1)
                for ai in range(nA):
                    if ai != a:
                        w[s, ai] += redistribute
                # Normalize row
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with action stickiness (state-specific perseveration) + WM misbinding under load.

    Rationale:
    - RL choice is biased toward repeating the last action taken in that state (stickiness).
    - WM suffers misbinding under higher load: the policy and updates partially reference
      a wrong state, leading to degraded WM contribution when nS is large.

    Parameters
    ----------
    model_parameters : tuple/list of length 6
      lr             : RL learning rate (0..1).
      wm_weight      : Base WM mixture weight (0..1).
      softmax_beta   : RL inverse temperature (scaled internally by 10).
      stickiness     : Additive bias to the last chosen action's value in the same state (>=0).
      misbind_rate   : Base probability of WM misbinding (0..1), scaled up by load.
      wm_decay       : WM decay toward uniform per trial (0..1).

    Returns
    -------
    neg_log_likelihood : float
      Negative log-likelihood of the observed action sequence.
    """
    lr, wm_weight, softmax_beta, stickiness, misbind_rate, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    rng = np.random.RandomState(0)  # deterministic pseudo-randomness for misbinding index selection

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states  = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q   = (1.0 / nA) * np.ones((nS, nA))
        w   = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Keep track of last action per state for stickiness
        last_action = -1 * np.ones(nS, dtype=int)

        load_level = max(nS - 3, 0) / 3.0  # 0 for 3-set, 1 for 6-set
        mb = np.clip(misbind_rate * (1.0 + load_level), 0.0, 1.0)  # misbinding increases with load
        wm_weight_eff = np.clip(wm_weight * (1.0 - 0.3 * load_level), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Apply stickiness to Q_s before RL policy: add bias to last action in this state
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stickiness

            # RL softmax probability for chosen action (as per template), using biased Q_s
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with potential misbinding: use a mixture of correct and a random other state
            if nS > 1:
                other_states = [i for i in range(nS) if i != s]
                mb_state = rng.choice(other_states)
            else:
                mb_state = s
            W_eff = (1.0 - mb) * w[s, :] + mb * w[mb_state, :]

            logits_wm = softmax_beta_wm * W_eff
            max_wm = np.max(logits_wm)
            exp_wm = np.exp(logits_wm - max_wm)
            p_wm_vec = exp_wm / np.sum(exp_wm)
            p_wm = p_wm_vec[a]

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # Update last action for stickiness
            last_action[s] = a

            # WM decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM update with misbinding: sometimes update the wrong state's WM
            target_state = s if rng.rand() > mb else mb_state
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[target_state, :] = 0.5 * w[target_state, :] + 0.5 * one_hot
            else:
                # Penalize the chosen action slightly in the (possibly misbound) state
                reduce = 0.2 * w[target_state, a]
                w[target_state, a] -= reduce
                if nA > 1:
                    gain = reduce / (nA - 1)
                    for ai in range(nA):
                        if ai != a:
                            w[target_state, ai] += gain
                row_sum = np.sum(w[target_state, :])
                if row_sum > 0:
                    w[target_state, :] /= row_sum
                else:
                    w[target_state, :] = w_0[target_state, :].copy()

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Meta-control: WM gating by performance and load + load-sensitive RL temperature.

    Rationale:
    - RL inverse temperature decreases with set size (noisier decisions under load).
    - WM contribution is dynamically gated by a logistic function of recent performance and load:
      better performance engages WM more; higher load suppresses WM engagement.

    Parameters
    ----------
    model_parameters : tuple/list of length 6
      lr                  : RL learning rate (0..1).
      wm_weight0          : Baseline WM weight prior to gating (0..1).
      softmax_beta        : Base RL inverse temperature (scaled internally by 10).
      gate_intercept      : Logistic intercept for WM gate (real).
      gate_slope          : Logistic slope on running performance accuracy (>=0).
      beta_load_sensitivity : Sensitivity of RL temperature to load (>=0), larger -> lower beta at high load.

    Returns
    -------
    neg_log_likelihood : float
      Negative log-likelihood of the observed action sequence.
    """
    lr, wm_weight0, softmax_beta, gate_intercept, gate_slope, beta_load_sensitivity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0

    eps = 1e-12

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states  = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q   = (1.0 / nA) * np.ones((nS, nA))
        w   = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-adjust RL beta
        load_level = max(nS - 3, 0) / 3.0  # 0 for 3-set, 1 for 6-set
        beta_rl = softmax_beta * np.exp(-beta_load_sensitivity * load_level)

        # Running performance estimate (exponential average of rewards)
        perf = 0.5  # start neutral
        perf_alpha = 0.2  # fixed smoothing within block

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability (replace softmax_beta with beta_rl for this block)
            # Keep the template form by reusing the symbol via a local alias
            soft_beta_local = beta_rl
            p_rl = 1.0 / np.sum(np.exp(soft_beta_local * (Q_s - Q_s[a])))

            # WM policy
            logits_wm = softmax_beta_wm * W_s
            max_wm = np.max(logits_wm)
            exp_wm = np.exp(logits_wm - max_wm)
            p_wm_vec = exp_wm / np.sum(exp_wm)
            p_wm = p_wm_vec[a]

            # WM gating via logistic on performance and load
            gate_input = gate_intercept + gate_slope * (perf - 0.5) - 2.0 * load_level
            wm_gate = 1.0 / (1.0 + np.exp(-gate_input))
            wm_mix = np.clip(wm_weight0 * wm_gate, 0.0, 1.0)

            # Mixture
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform + simple Hebbian update on reward
            w = 0.9 * w + 0.1 * w_0  # mild decay each trial
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.4 * w[s, :] + 0.6 * one_hot
            else:
                # No-reward: slight softening toward uniform to reduce overconfidence
                w[s, :] = 0.95 * w[s, :] + 0.05 * w_0[s, :]

            # Update performance estimate
            perf = (1.0 - perf_alpha) * perf + perf_alpha * r

        blocks_log_p += log_p

    return -blocks_log_p
