def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with interference-limited WM write, retrieval noise, and load-gated arbitration.

    Idea:
    - RL learns Q-values via delta rule.
    - WM stores near one-shot associations but suffers set-size dependent interference.
    - Arbitration weight for WM is dynamically adjusted by load within block (we write it into wm_weight
      before mixing), while respecting the template mixture equation.
    - WM retrieval has a small noise that increases with set size.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight: Baseline WM weight that we dynamically adjust by load (0..1).
    - softmax_beta: RL inverse temperature (scaled internally by 10).
    - wm_write: Baseline WM write strength on reward (0..1).
    - interference_k: Interference scale; higher means less set-size penalty on WM writing (>0).
    - noise_eps: WM retrieval noise level (0..0.5), mixed with uniform and scaled by set size.

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_write, interference_k, noise_eps = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50  # very deterministic base

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Precompute a load-based availability factor for this block
        # Strong WM at small set size, weaker at large set size. Smooth hyperbolic form.
        wm_avail = 1.0 / (1.0 + (max(nS, 1) - 1) / max(interference_k, 1e-6))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:].copy()
            W_s = w[s,:].copy()

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s - Q_s[a])))

            # WM policy with retrieval noise increasing with set size
            # Mix W_s with uniform by an epsilon that grows with set size
            eps = min(0.49, noise_eps * (nS / 6.0))
            W_mix = (1 - eps) * W_s + eps * (1.0 / nA) * np.ones(nA)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_mix - W_mix[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Dynamically gate wm_weight by load (availability)
            wm_weight = np.clip(wm_avail * wm_weight, 0.0, 1.0)

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr*delta

            # WM update: reward-driven write with set-size interference,
            # and mild decay to default after errors.
            if r > 0:
                one_hot = np.zeros(nA); one_hot[a] = 1.0
                write_rate = np.clip(wm_write * wm_avail, 0.0, 1.0)
                w[s, :] = (1 - write_rate) * w[s, :] + write_rate * one_hot
            else:
                # Set-size dependent decay (more decay at larger set sizes)
                decay_rate = np.clip((1 - wm_avail) * 0.5, 0.0, 0.5)
                w[s, :] = (1 - decay_rate) * w[s, :] + decay_rate * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + hypothesis-testing WM with confirmation/forgetting and load-scaled consolidation.

    Idea:
    - RL uses standard delta learning.
    - WM represents a "hypothesized correct action" per state; upon reward, it consolidates toward
      a one-hot code; upon errors, it forgets toward uniform.
    - Consolidation and forgetting rates are modulated by set size (load).
    - We dynamically adjust wm_weight by load per block (write into wm_weight before mixing).

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight: Baseline WM mixture weight (0..1).
    - softmax_beta: RL inverse temperature (scaled internally by 10).
    - wm_confirm: Base consolidation rate toward the rewarded action (0..1).
    - wm_confuse: Base forgetting rate toward uniform on error (0..1).
    - load_scale: Positive factor; higher values make consolidation drop faster with set size (>0).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_confirm, wm_confuse, load_scale = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50  # very deterministic WM readout

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load factor: consolidation drops with set size; forgetting increases
        load_factor = 1.0 / (1.0 + load_scale * max(nS - 3.0, 0.0))
        confirm_eff = np.clip(wm_confirm * load_factor, 0.0, 1.0)
        confuse_eff = np.clip(wm_confuse * (1.0 / max(load_factor, 1e-6) - 1.0 + 1.0), 0.0, 1.0)  # grows when load_factor small

        # Adjust arbitration by load
        wm_weight = np.clip(wm_weight * load_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:].copy()
            W_s = w[s,:].copy()

            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s - Q_s[a])))

            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr*delta

            # WM hypothesis update
            if r > 0:
                one_hot = np.zeros(nA); one_hot[a] = 1.0
                w[s, :] = (1 - confirm_eff) * w[s, :] + confirm_eff * one_hot
            else:
                w[s, :] = (1 - confuse_eff) * w[s, :] + confuse_eff * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with surprise-driven arbitration and load-dependent WM decay.

    Idea:
    - RL learns via delta rule.
    - Arbitration: WM contribution increases when the unsigned RL prediction error is large
      (surprise), encouraging reliance on WM early; but it is reduced under higher load.
      We implement this by overwriting wm_weight within-trial based on surprise and load,
      while keeping the given mixture equation.
    - WM updates: PE-gated write on rewards and set-size-modulated decay on each visit.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1).
    - wm_weight_base: Baseline WM mixture weight (0..1) modulated by surprise and load.
    - softmax_beta: RL inverse temperature (scaled internally by 10).
    - surprise_gain: Positive gain mapping |PE| to WM arbitration weight (>0).
    - wm_decay_rate: Base decay of WM toward uniform on each visit (0..1).
    - setsize_slope: Slope controlling how decay and arbitration depend on set size (>0).

    Returns:
    - Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, surprise_gain, wm_decay_rate, setsize_slope = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50  # deterministic WM readout

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Precompute a load suppression factor for arbitration and a load-amplified decay
        load_suppr = 1.0 / (1.0 + np.exp(setsize_slope * (nS - 3.5)))  # ~1 for small set, ~0 for large set
        visit_decay = np.clip(wm_decay_rate * (1.0 - load_suppr), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:].copy()
            W_s = w[s,:].copy()

            # Compute surprise from RL before updating
            pe = r - Q_s[a]
            surprise = abs(pe)
            wm_weight = np.clip(wm_weight_base * load_suppr * (1.0 / (1.0 + np.exp(-surprise_gain * (surprise - 0.0)))), 0.0, 1.0)

            # RL policy
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s - Q_s[a])))

            # WM policy
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            p_total = p_wm*wm_weight + (1-wm_weight)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            q[s][a] += lr * pe

            # WM update: PE-gated write on reward, and load-dependent decay each visit
            if r > 0:
                one_hot = np.zeros(nA); one_hot[a] = 1.0
                # Write strength grows with surprise (early learning) but saturates
                write_rate = np.clip(0.5 + 0.5 * (1.0 / (1.0 + np.exp(-surprise_gain * (surprise - 0.5)))), 0.0, 1.0)
                w[s, :] = (1 - write_rate) * w[s, :] + write_rate * one_hot
            # Apply decay on each visit (stronger under load)
            w[s, :] = (1 - visit_decay) * w[s, :] + visit_decay * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
