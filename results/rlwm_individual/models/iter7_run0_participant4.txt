def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + load-dependent WM strength and decay.

    Mechanism
    - RL: standard Rescorla–Wagner update (as in template).
    - WM policy: constructed from a memory probability map W; action prob is softmax with a
      tunable WM sharpness that is reduced with higher set sizes (load).
    - WM update: leak (decay) toward uniform grows with set size; rewarded outcomes write a
      one-hot trace into W with strength controlled by wm_temp_scale.

    Parameters
    ----------
    model_parameters : list/tuple of 5 floats
        lr              : RL learning rate in (0,1].
        wm_weight       : Mixture weight of WM policy in [0,1]; used in arbitration line.
        softmax_beta    : RL inverse temperature; scaled by *10 internally.
        wm_temp_scale   : Scales WM sharpness (>0). Larger => crisper WM policy.
        decay_load      : Controls how fast WM traces decay with set size (>=0). Larger => more decay.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_temp_scale, decay_load = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    eps = 1e-12
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent parameters
        # tau reduces WM strength with load; leak grows with load
        tau = 1.0 / (1.0 + decay_load * max(0, nS - 1))
        leak = 1.0 - np.exp(-decay_load * max(1.0, float(nS)))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action (given by template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Build a load-weakened WM map by mixing with uniform; sharpen with wm_temp_scale
            W_eff = tau * W_s + (1.0 - tau) * w_0[s, :]
            # Scale the logits before the deterministic WM softmax to emulate WM temperature
            logits_wm = softmax_beta_wm * wm_temp_scale * (W_eff - W_eff[a])
            p_wm = 1.0 / np.sum(np.exp(logits_wm))

            p_total = p_wm * wm_weight + (1 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Passive decay toward uniform increases with load
            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]
            # Rewarded outcomes write a one-hot trace into WM with strength tied to wm_temp_scale
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta_wm = 1.0 - np.exp(-wm_temp_scale)  # in (0,1); larger wm_temp_scale => stronger write
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * one_hot
            # Ensure normalization for numerical stability
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with cross-state interference and forgetting.

    Mechanism
    - RL: standard Rescorla–Wagner update (as in template).
    - WM policy: softmax over a learned mapping W. WM softmax temperature is tunable.
    - WM update:
        * Rewarded trials store a near one-hot vector for the current state.
        * Cross-state interference: rewarded action also bleeds into other states' WM
          with strength 'interference' (models confusion across items under load).
        * Per-trial forgetting (leak) toward uniform controlled by 'forgetting'.

    Parameters
    ----------
    model_parameters : list/tuple of 6 floats
        lr             : RL learning rate in (0,1].
        wm_weight      : Mixture weight of WM policy in [0,1]; used in arbitration line.
        softmax_beta   : RL inverse temperature; scaled by *10 internally.
        wm_temp        : WM softmax temperature scaling (>0). Larger => crisper WM policy.
        interference   : Cross-state spillover for rewarded action in [0,1]. 0=no spillover.
        forgetting     : Per-trial WM leak toward uniform in [0,1]. Larger => faster forgetting.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_temp, interference, forgetting = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    eps = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Adjustable WM temperature via scaling of W_s before softmax
            W_eff = W_s.copy()
            logits_wm = softmax_beta_wm * wm_temp * (W_eff - W_eff[a])
            p_wm = 1.0 / np.sum(np.exp(logits_wm))

            p_total = p_wm * wm_weight + (1 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Forgetting (leak) toward uniform for the current state each trial
            leak = np.clip(forgetting, 0.0, 1.0)
            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

            if r > 0:
                # Encode near one-hot in current state
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta = 1.0 - np.exp(-wm_temp)  # stronger with larger wm_temp
                w[s, :] = (1.0 - eta) * w[s, :] + eta * one_hot

                # Cross-state interference: reward for (s,a) boosts action a in other states
                spill = np.clip(interference, 0.0, 1.0) / max(1, nS - 1)
                if spill > 0 and nS > 1:
                    for s2 in range(nS):
                        if s2 == s:
                            continue
                        # Mild leak first to avoid runaway accumulation
                        w[s2, :] = (1.0 - 0.5 * leak) * w[s2, :] + 0.5 * leak * w_0[s2, :]
                        # Additive boost towards action a
                        w[s2, a] = (1.0 - spill) * w[s2, a] + spill * 1.0
                        # Renormalize row
                        row_sum = np.sum(w[s2, :])
                        if row_sum > 0:
                            w[s2, :] /= row_sum
                        else:
                            w[s2, :] = w_0[s2, :]

            # Normalize current state's WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + uncertainty-modulated WM expression.

    Mechanism
    - RL: standard Rescorla–Wagner update (as in template).
    - WM policy: softmax over W with an effective WM expression controlled by RL uncertainty.
      When RL is uncertain (high entropy), WM is expressed more strongly; when RL is confident,
      WM expression is reduced. This is implemented by mixing W with uniform before softmax.
    - WM update: leak toward uniform (wm_leak) and reward-based encoding toward one-hot.
      WM temperature base controls how sharply WM expresses its map.

    Parameters
    ----------
    model_parameters : list/tuple of 6 floats
        lr             : RL learning rate in (0,1].
        wm_weight      : Mixture weight of WM policy in [0,1]; used in arbitration line.
        softmax_beta   : RL inverse temperature; scaled by *10 internally.
        wm_temp0       : Baseline WM temperature scaling (>0).
        ent_slope      : Sensitivity of WM expression to RL entropy (>0).
        wm_leak        : Per-trial WM leak toward uniform in [0,1].

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_temp0, ent_slope, wm_leak = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    eps = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        H_max = np.log(nA)

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action (given by template)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Compute RL policy distribution to estimate entropy (uncertainty)
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            exps = np.exp(logits_rl)
            pi = exps / np.sum(exps)
            H = -np.sum(pi * np.log(np.clip(pi, eps, 1.0)))  # entropy in [0, H_max]

            # Map entropy to an expression strength alpha in [0,1]
            # More uncertainty (higher H) => larger alpha (stronger WM expression).
            norm_H = H / max(H_max, eps)
            alpha = 1.0 / (1.0 + np.exp(-ent_slope * (norm_H - 0.5)))

            # Build effective WM map and apply baseline temperature scaling
            W_eff = alpha * W_s + (1.0 - alpha) * w_0[s, :]
            logits_wm = softmax_beta_wm * wm_temp0 * (W_eff - W_eff[a])
            p_wm = 1.0 / np.sum(np.exp(logits_wm))

            p_total = p_wm * wm_weight + (1 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Leak toward uniform
            leak = np.clip(wm_leak, 0.0, 1.0)
            w[s, :] = (1.0 - leak) * w[s, :] + leak * w_0[s, :]

            # Reward-based encoding toward one-hot
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # Encoding rate modulated by current uncertainty alpha: store more when RL is uncertain
                eta = 0.5 * (alpha + 1.0 - np.exp(-wm_temp0))  # in (0,1)
                eta = np.clip(eta, 0.0, 1.0)
                w[s, :] = (1.0 - eta) * w[s, :] + eta * one_hot

            # Normalize for stability
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
