def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with entropy-gated arbitration and capacity penalty.

    Mechanism:
    - RL: Rescorla–Wagner update with softmax choice.
    - WM: Supervised mapping that one-shots toward rewarded action and decays toward uniform after non-reward.
    - Arbitration: Trial-wise WM weight increases with RL uncertainty (entropy) and decreases with load (larger set size).

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - wm_weight: Baseline WM arbitration weight in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - entropy_slope: Strength by which RL entropy increases WM reliance (>=0).
    - wm_decay: Decay rate toward uniform after non-reward in [0,1].
    - wm_learn: One-shot WM learning rate toward one-hot after reward in [0,1].

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, entropy_slope, wm_decay, wm_learn = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action a (softmax trick as in template)
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic softmax over current WM weights)
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: WM weight increases with RL entropy, decreases with load
            # Compute RL softmax policy to estimate entropy
            logits = softmax_beta * (Q_s - np.max(Q_s))
            pi_rl = np.exp(logits)
            pi_rl = pi_rl / np.sum(pi_rl)
            eps = 1e-12
            H_rl = -np.sum(pi_rl * np.log(np.clip(pi_rl, eps, 1.0))) / np.log(nA)  # normalized [0,1]

            load_penalty = max(0.0, (nS - 3.0) / 3.0)  # 0 for nS=3, 1 for nS=6
            # Logit space combination: baseline + entropy gain - capacity penalty
            base_logit = np.log(np.clip(wm_weight, eps, 1 - eps)) - np.log(1 - np.clip(wm_weight, eps, 1 - eps))
            wm_logit = base_logit + entropy_slope * H_rl - load_penalty
            wm_weight_t = 1.0 / (1.0 + np.exp(-wm_logit))
            wm_weight_t = np.clip(wm_weight_t, 0.0, 1.0)

            p_total = wm_weight_t * p_wm + (1 - wm_weight_t) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward => push to one-hot; no-reward => decay to uniform
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Keep WM rows normalized and bounded
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with asymmetric learning and WM baseline reduced by load.

    Mechanism:
    - RL: Rescorla–Wagner with asymmetric learning (negative outcomes scaled by neg_mult).
    - WM: Supervised mapping with learning on reward only; no-reward leaves WM unchanged.
    - Arbitration: Baseline WM weight multiplicatively suppressed by load (larger nS).

    Parameters (total 6):
    - lr: RL base learning rate for positive PE in [0,1].
    - wm_weight: Baseline WM arbitration weight in [0,1] at low load (nS=3).
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - neg_mult: Multiplier for negative PE learning (>=0). Effective LR- = lr*neg_mult.
    - wm_learn: WM supervised learning rate toward one-hot on reward in [0,1].
    - load_sensitivity: Strength of multiplicative WM suppression with load (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, neg_mult, wm_learn, load_sensitivity = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load-dependent WM baseline
        load_term = max(0.0, (nS - 3.0) / 3.0)
        wm_weight_block = wm_weight * np.exp(-load_sensitivity * load_term)
        wm_weight_block = np.clip(wm_weight_block, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy prob of chosen action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_weight_block * p_wm + (1 - wm_weight_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL asymmetric update
            pe = r - Q_s[a]
            eff_lr = lr if pe >= 0 else lr * neg_mult
            q[s, a] += eff_lr * pe

            # WM update: learn only from reward
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with surprise-amplified WM learning and load-induced global interference.

    Mechanism:
    - RL: Rescorla–Wagner update with softmax choice.
    - WM: State-specific mapping w[s,:] and a global action popularity vector g (captures misbinding/interference).
      On reward, WM learning rate is increased by outcome surprise; under higher load, interference mixes g into the
      state-specific WM policy, reducing selectivity.
    - Arbitration: Fixed baseline WM weight.

    Parameters (total 6):
    - lr: RL learning rate in [0,1].
    - wm_weight: Baseline WM arbitration weight in [0,1].
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - pe_gain: Gain by which absolute RL prediction error boosts WM learning (>=0).
    - wm_learn_base: Baseline WM learning rate on reward in [0,1].
    - interference: Strength with which load mixes global popularity into WM policy (>=0).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, pe_gain, wm_learn_base, interference = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        # Global action popularity (captures interference across states)
        g = (1 / nA) * np.ones(nA)

        # Load-dependent interference mix (0 at nS=3; grows with nS and 'interference')
        load_term = max(0.0, (nS - 3.0) / 3.0)
        mix_global = np.clip(interference * load_term, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy prob of chosen action
            p_rl = 1 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Effective WM policy is a mix of state-specific WM and global popularity under load
            W_eff = (1.0 - mix_global) * W_s + mix_global * g
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_eff - W_eff[a])))

            # Fixed arbitration
            wm_w = np.clip(wm_weight, 0.0, 1.0)
            p_total = wm_w * p_wm + (1 - wm_w) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # Surprise-amplified WM learning on reward
            if r > 0.5:
                wm_lr_t = np.clip(wm_learn_base + pe_gain * abs(pe), 0.0, 1.0)
                # Update state-specific WM toward one-hot
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_lr_t) * w[s, :] + wm_lr_t * target
                # Update global popularity toward rewarded action
                g = (1.0 - wm_lr_t) * g + wm_lr_t * target
            else:
                # Light implicit stabilization toward uniform for both w[s] and g when no reward
                k = 0.05
                w[s, :] = (1.0 - k) * w[s, :] + k * w_0[s, :]
                g = (1.0 - k) * g + k * (1.0 / nA) * np.ones(nA)

            # Renormalize
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])
            g = np.clip(g, 1e-8, 1.0)
            g /= np.sum(g)

        blocks_log_p += log_p

    return -blocks_log_p
