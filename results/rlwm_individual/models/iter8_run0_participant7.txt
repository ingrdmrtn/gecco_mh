def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with entropy-gated WM under load and graded WM learning.

    Core ideas
    - RL: standard delta-learning with softmax.
    - WM: probabilistic action memory per state; updated toward one-hot on reward,
      and gently away from chosen action on non-reward; ongoing decay toward uniform.
    - Load effect: WM contribution is down-weighted by set size (nS) via an inverse-capacity term.
    - Entropy gate: when WM for the current state is uncertain (high entropy), the model leans
      more on RL by reducing WM's mixture weight.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within each block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size on each trial (3 or 6).
    model_parameters : tuple
        (lr, wm_mix_base, softmax_beta, inv_capacity, wm_learn_rate)
        - lr: RL learning rate in [0,1].
        - wm_mix_base: baseline WM mixture weight in [0,1].
        - softmax_beta: RL inverse temperature (>0); internally scaled up by 10.
        - inv_capacity: strength of load penalty on WM (>=0). Higher reduces WM at larger set sizes.
        - wm_learn_rate: WM learning rate (0..1) that sets how strongly WM updates on each trial.

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, wm_mix_base, softmax_beta, inv_capacity, wm_learn_rate = model_parameters
    softmax_beta *= 10.0  # follow template scaling
    softmax_beta_wm = 50.0  # deterministic WM policy per template

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-induced down-weighting of WM
        load_factor = 1.0 / (1.0 + max(0.0, inv_capacity) * max(0, nS - 3))

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Entropy-based gate: high entropy => reduce WM weight
            eps = 1e-12
            H = -np.sum(W_s * np.log(W_s + eps)) / np.log(nA)  # normalized entropy in [0,1]
            wm_weight_eff = np.clip(wm_mix_base * load_factor * (1.0 - H), 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay toward uniform (stronger under load implicitly via reduced mixture, not values)
            decay = 0.05 + 0.10 * (nS - 3) / 3.0 if nS > 3 else 0.05
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # WM learning: reward pulls to one-hot; non-reward gently pushes to anti-chosen distribution
            onehot = np.zeros(nA)
            onehot[a] = 1.0
            if r > 0.0:
                eta = wm_learn_rate
                w[s, :] = (1.0 - eta) * w[s, :] + eta * onehot
            else:
                eta = 0.5 * wm_learn_rate
                anti = (1.0 - onehot) / (nA - 1)
                w[s, :] = (1.0 - eta) * w[s, :] + eta * anti

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with misbinding noise under load and choice perseveration kernel.

    Core ideas
    - RL: standard delta-learning.
    - WM: one-shot Hebbian-like update on rewards, but with load-dependent misbinding noise
      that dilutes the stored association. WM also undergoes mild forgetting.
    - Perseveration: a decaying choice kernel biases choices toward recently chosen actions,
      affecting both RL and WM policies.
    - Load effect: misbinding probability increases with set size (more at nS=6 than nS=3),
      reducing WM precision in larger sets.

    Parameters
    ----------
    states : array-like of int
    actions : array-like of int
    rewards : array-like of {0,1}
    blocks : array-like of int
    set_sizes : array-like of int
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, misbind_base, wm_decay, perseveration)
        - lr: RL learning rate [0,1].
        - wm_weight: WM mixture weight [0,1].
        - softmax_beta: RL inverse temperature (>0); scaled by 10 internally.
        - misbind_base: baseline misbinding rate per extra item beyond 3 (>=0).
        - wm_decay: per-trial decay of WM toward uniform (0..1).
        - perseveration: strength of the choice kernel bias (>=0).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, misbind_base, wm_decay, perseveration = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Choice kernel (perseveration): per-state action tendency
        k = np.zeros((nS, nA))
        kernel_decay = 0.8

        # Load-dependent misbinding probability
        misbind = np.clip(misbind_base * max(0, nS - 3), 0.0, 0.9)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Preferences incorporate perseveration kernel
            Q_pref = q[s, :] + perseveration * k[s, :]
            W_pref = w[s, :] + perseveration * k[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_pref - Q_pref[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_pref - W_pref[a])))

            p_total = np.clip(wm_weight, 0.0, 1.0) * p_wm + (1.0 - np.clip(wm_weight, 0.0, 1.0)) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM Hebbian-like update on reward with misbinding dilution
            if r > 0.0:
                # Intended memory for chosen action
                onehot = np.zeros(nA); onehot[a] = 1.0
                # Misbinding modeled as mixing with uniform over actions
                imprint = (1.0 - misbind) * onehot + misbind * (1.0 / nA) * np.ones(nA)
                # Strong overwrite at reward
                eta = 0.8
                w[s, :] = (1.0 - eta) * w[s, :] + eta * imprint
            else:
                # On errors, very mild adjustment away from chosen action
                eta = 0.1
                anti = np.ones(nA) / nA
                w[s, :] = (1.0 - eta) * w[s, :] + eta * anti

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

            # Update choice kernel (perseveration)
            k[s, :] *= kernel_decay
            k[s, a] += 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM with reward-rate adaptive gating and load-scaled WM precision.

    Core ideas
    - RL: standard delta-learning with softmax.
    - WM: action probabilities per state with decay; updated toward chosen action on rewards.
    - Mixture control: WM contribution is higher when recent reward rate is low (need for control)
      and lower when reward rate is high (habitual RL suffices). WM is also downscaled by load.
    - Load scaling: WM mixture is multiplied by (3/nS)^load_exponent to capture reduced precision
      at larger set sizes.

    Parameters
    ----------
    states : array-like of int
    actions : array-like of int
    rewards : array-like of {0,1}
    blocks : array-like of int
    set_sizes : array-like of int
    model_parameters : tuple
        (lr, wm_weight_base, softmax_beta, reward_trace_tau, load_exponent, wm_decay)
        - lr: RL learning rate [0,1].
        - wm_weight_base: baseline WM mixture weight [0,1].
        - softmax_beta: RL inverse temperature (>0); scaled by 10 internally.
        - reward_trace_tau: timescale for running reward-rate (>=1). Larger = slower adaptation.
        - load_exponent: exponent controlling how strongly load reduces WM (>=0).
        - wm_decay: per-trial decay of WM toward uniform (0..1).

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, wm_weight_base, softmax_beta, reward_trace_tau, load_exponent, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # WM assumed precise when it is used

    # Convert tau to exponential smoothing coefficient
    tau = max(1.0, reward_trace_tau)
    alpha_rbar = 1.0 / tau

    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Running reward rate within block
        rbar = 0.5

        # Load-based WM downscaling
        load_scale = (3.0 / float(nS)) ** max(0.0, load_exponent)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Reward-rate adaptive WM gating: more WM when rbar is low
            wm_need = 1.0 - np.clip(rbar, 0.0, 1.0)
            wm_weight_eff = np.clip(wm_weight_base * load_scale * wm_need, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM learning: strong imprint on reward, minimal change on non-reward
            if r > 0.0:
                eta = 0.7
                onehot = np.zeros(nA); onehot[a] = 1.0
                w[s, :] = (1.0 - eta) * w[s, :] + eta * onehot
            else:
                # Light uncertainty-spreading on errors
                eta = 0.1
                w[s, :] = (1.0 - eta) * w[s, :] + eta * (1.0 / nA) * np.ones(nA)

            # Normalize WM row
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

            # Update running reward rate
            rbar = (1.0 - alpha_rbar) * rbar + alpha_rbar * r

        blocks_log_p += log_p

    return -blocks_log_p
