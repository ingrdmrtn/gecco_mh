def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + one-shot WM with set-size-dependent encoding and cross-talk interference.

    Mechanism
    - RL: Rescorla–Wagner update (provided in template).
    - WM: a fast, near-deterministic store of rewarded associations (per state).
      On positive feedback, the WM mapping for the current state is updated toward
      a one-hot choice, with:
        • encoding probability decreasing with set size,
        • cross-talk interference that injects small action-wise noise across all actions.
    - Policy: mixture of WM and RL where mixture weight scales down with set size.

    Parameters
    ----------
    model_parameters : list/tuple of 5 floats
        lr                 : RL learning rate in (0,1].
        wm_weight          : Base WM mixture weight in [0,1]; scaled by set size internally.
        softmax_beta       : RL inverse temperature; scaled by *10 internally.
        encode_prob_base   : Base probability of WM encoding after reward (0..1).
                             Effective encoding prob = encode_prob_base / nS (clipped to [0,1]).
        interference_gain  : Cross-talk magnitude added to non-chosen actions during WM update
                             (suggested small, e.g., 0..0.2). Scales with set size.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, encode_prob_base, interference_gain = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    eps = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Set-size-dependent WM availability and encoding probability
        wm_weight_eff = np.clip(wm_weight * (3.0 / max(3.0, nS)), 0.0, 1.0)
        p_encode = np.clip(encode_prob_base / max(1.0, nS), 0.0, 1.0)
        x_talk = np.clip(interference_gain * (nS - 3) / max(1.0, nS - 3 + 1e-9), 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (near-deterministic softmax over WM weights)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM updating:
            # 1) mild decay toward uniform each trial (load-insensitive stabilization)
            decay = 0.05 * (nS / 6.0)  # slightly stronger decay in larger sets
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # 2) On reward, probabilistic encoding with cross-talk interference
            if r > 0:
                # Encode with probability p_encode by moving toward one-hot target
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta = p_encode  # effective learning toward one-hot
                w[s, :] = (1.0 - eta) * w[s, :] + eta * one_hot

                # Cross-talk interference: inject small uniform mass to non-chosen actions
                # scaled by x_talk, which grows with load
                noise_vec = np.ones(nA) / nA
                noise_vec[a] = 0.0  # keep chosen action minimally affected
                if np.sum(noise_vec) > 0:
                    noise_vec = noise_vec / np.sum(noise_vec)
                w[s, :] = (1.0 - x_talk) * w[s, :] + x_talk * noise_vec

            # Re-normalize to avoid drift
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + WM arbitration by confidence with load-dependent down-weighting and anti-Hebbian error.

    Mechanism
    - RL: Rescorla–Wagner update.
    - WM: stores state-action values updated strongly after rewards; decays each trial.
      Incorrect feedback triggers anti-Hebbian suppression of the chosen action in WM.
    - Arbitration: WM mixture weight scales with WM confidence (margin between best and second-best
      in WM for the current state), and is down-weighted as set size increases.

    Parameters
    ----------
    model_parameters : list/tuple of 5 floats
        lr            : RL learning rate in (0,1].
        wm_weight     : Base WM mixture weight (0..1), modulated by confidence/load.
        softmax_beta  : RL inverse temperature; scaled by *10 internally.
        lambda_load   : Load penalty on WM arbitration (>=0). Larger => stronger down-weighting
                        with set size (applied from set size 3 as baseline).
        wm_decay      : Per-trial WM decay toward uniform (0..1), applied to the active state.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, lambda_load, wm_decay = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    eps = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Load factor relative to size 3 baseline
        load_factor = max(0.0, (nS - 3) / 3.0)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Confidence from WM: margin between top-1 and top-2 (0..1)
            sorted_W = np.sort(W_s)[::-1]
            top1 = sorted_W[0]
            top2 = sorted_W[1] if nA > 1 else 0.0
            margin = np.clip(top1 - top2, 0.0, 1.0)

            # Arbitration: sigmoid of margin, penalized by load
            conf = 1.0 / (1.0 + np.exp(-10.0 * (margin - 0.1)))  # ~0 if low margin, ~1 if high
            wm_weight_eff = wm_weight * np.clip(1.0 - lambda_load * load_factor, 0.0, 1.0) * conf

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay toward uniform for current state
            decay = np.clip(wm_decay, 0.0, 1.0)
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # WM update:
            if r > 0:
                # Hebbian push toward one-hot on reward
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta_pos = 0.7  # strong consolidation
                w[s, :] = (1.0 - eta_pos) * w[s, :] + eta_pos * one_hot
            else:
                # Anti-Hebbian: suppress chosen action, redistribute to others
                eta_neg = 0.3
                suppress = np.zeros(nA)
                suppress[a] = 1.0
                w[s, :] = w[s, :] * (1.0 - eta_neg * suppress)
                # Renormalize within state
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + recency-gated WM with age-dependent retrieval and lapse.

    Mechanism
    - RL: Rescorla–Wagner update.
    - WM: per-state episodic mapping with a recency 'age'. Positive feedback stores a
      one-hot action and resets its age; on every trial, age increases and WM trace
      decays toward uniform. Retrieval probability decreases with age and with set size.
    - Policy: mixture of WM and RL plus a small lapse probability to uniform random choice.

    Parameters
    ----------
    model_parameters : list/tuple of 5 floats
        lr            : RL learning rate in (0,1].
        wm_weight     : Base WM mixture weight in [0,1].
        softmax_beta  : RL inverse temperature; scaled by *10 internally.
        recency_tau   : Time constant controlling how quickly WM retrieval decays with age (>0).
        lapse         : Lapse probability (0..0.2) mixed into final policy.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, recency_tau, lapse = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    eps = 1e-12

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))
        ages = np.full(nS, 1e6)  # large age indicates no reliable WM yet

        # Load scaling: greater nS reduces WM availability
        load_scale = 3.0 / max(3.0, nS)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Age-dependent WM retrieval probability
            age = ages[s]
            p_retrieve = np.exp(-max(0.0, age) / max(1e-6, recency_tau))
            wm_weight_eff = np.clip(wm_weight * load_scale * p_retrieve, 0.0, 1.0)

            # Mixture with lapse
            mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * mix + lapse * (1.0 / nA)
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # WM decay for current state as a function of age (older => closer to uniform)
            # Increase age each trial for the current state
            ages[s] += 1.0
            decay = 1.0 - np.exp(-1.0 / max(1e-6, recency_tau))
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # WM update on reward: store and reset age
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta = 0.9  # strong overwrite on success
                w[s, :] = (1.0 - eta) * w[s, :] + eta * one_hot
                ages[s] = 0.0  # reset recency on successful learning

            # Safety renormalization
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
