def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL + Capacity-limited WM with binding asymmetry and load-sensitive availability.

    Mechanism
    - RL: delta-rule learning with softmax policy.
    - WM: associative table that decays toward uniform and binds chosen action to the
      current state. Rewarded trials drive more focused binding than non-rewarded
      trials (binding asymmetry).
    - Load sensitivity: WM availability declines as set size grows via a capacity
      scaling factor. Arbitration multiplies the baseline wm_weight by this availability.

    Parameters
    ----------
    model_parameters: tuple/list of 6 floats
        lr: float
            RL learning rate (0..1).
        wm_weight: float
            Baseline mixture weight for WM (0..1) before load adjustment.
        softmax_beta: float
            RL inverse temperature; internally scaled by 10.
        wm_decay: float
            Per-trial decay of WM table toward uniform (0..1).
        cap_scale: float
            Capacity scaling that governs WM availability; higher means WM is more
            available at larger set sizes (0..2 recommended).
        bind_asym: float
            Asymmetry for WM binding on non-rewarded trials; 0=no binding on errors,
            1=symmetric binding like rewards.

    Returns
    -------
    nll: float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, cap_scale, bind_asym = model_parameters
    softmax_beta *= 10.0  # beta has a higher upper bound
    softmax_beta_wm = 50.0  # very deterministic
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-sensitive availability: more load => less WM influence
        # cap_scale>1 can partially protect WM at nS=6, cap_scale<1 weakens it further
        wm_available = np.clip(cap_scale * (3.0 / max(3.0, float(nS))), 0.0, 1.5)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (soft-argmax over W_s)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: baseline wm_weight scaled by availability from capacity under load
            wm_mix = np.clip(wm_weight * np.clip(wm_available, 0.0, 1.0), 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform each trial
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM associative binding at the experienced state
            # Rewarded: move strongly toward one-hot on chosen action
            # Non-rewarded: bind with reduced strength (bind_asym factor)
            if r > 0.5:
                # Push row toward one-hot on a
                w[s, :] = (1.0 - lr) * w[s, :]
                w[s, a] += lr
            else:
                effective_lr = lr * np.clip(bind_asym, 0.0, 1.0)
                w[s, :] = (1.0 - effective_lr) * w[s, :]
                w[s, a] += effective_lr

            # Normalize WM probabilities for the state
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL (valence-asymmetric) + Surprise-gated WM encoding with load-dependent leak.

    Mechanism
    - RL: separate learning rates for positive and negative prediction errors.
    - WM: associative table that encodes the current (state, chosen action) proportionally
      to surprise (absolute RL prediction error). Under higher load (nS), WM leaks faster.
    - Arbitration: convex mixture of RL and WM with a fixed baseline wm_weight.

    Parameters
    ----------
    model_parameters: tuple/list of 6 floats
        alpha_pos: float
            RL learning rate for positive PE (0..1).
        alpha_neg: float
            RL learning rate for negative PE (0..1).
        wm_weight: float
            Mixture weight for WM contribution (0..1).
        softmax_beta: float
            RL inverse temperature; internally scaled by 10.
        kappa_surprise: float
            Scales surprise-based encoding strength into WM (0..5 recommended).
        wm_leak: float
            Baseline WM leak toward uniform per trial; effective leak grows with set size.

    Returns
    -------
    nll: float
        Negative log-likelihood of observed choices.
    """
    alpha_pos, alpha_neg, wm_weight, softmax_beta, kappa_surprise, wm_leak = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM leak (higher nS => more leak)
        leak = np.clip(wm_leak * (1.0 + max(0.0, (nS - 3.0) / 3.0)), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration with fixed baseline weight (WM degraded implicitly by leak)
            wm_mix = np.clip(wm_weight, 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with valence asymmetry
            pe = r - Q_s[a]
            lr_eff = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += lr_eff * pe

            # WM leak toward uniform
            w = (1.0 - leak) * w + leak * w_0

            # Surprise-gated encoding of current choice into WM
            # Encoding strength scales with absolute RL PE and is reduced by load via leak already
            encode_strength = np.clip(kappa_surprise * abs(pe), 0.0, 1.0)
            if encode_strength > 0.0:
                w[s, :] = (1.0 - encode_strength) * w[s, :]
                w[s, a] += encode_strength

            # Normalize WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    RL with perseveration + WM retrieval noise increasing with load.

    Mechanism
    - RL: delta-rule learning with softmax decision augmented by action perseveration
      (stickiness) within state.
    - WM: associative table used for near-deterministic retrieval, but its effective
      precision is degraded by a retrieval noise that scales with set size.
    - Arbitration: fixed baseline wm_weight; load primarily weakens WM by increasing
      its retrieval noise.

    Parameters
    ----------
    model_parameters: tuple/list of 5 floats
        lr: float
            RL learning rate (0..1).
        wm_weight: float
            Mixture weight for WM contribution (0..1).
        softmax_beta: float
            RL inverse temperature; internally scaled by 10.
        persev: float
            Perseveration strength added to the last chosen action in each state (0..5).
        wm_noise: float
            Base retrieval noise scaling for WM; effective noise grows with set size.

    Returns
    -------
    nll: float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, persev, wm_noise = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    blocks_log_p = 0.0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last chosen action per state for perseveration
        last_action = -1 * np.ones(nS, dtype=int)

        # WM retrieval noise increases with load
        noise = np.clip(wm_noise * max(0.0, (nS - 3.0) / 3.0), 0.0, 10.0)
        beta_wm_eff = softmax_beta_wm / (1.0 + noise)

        log_p = 0.0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :]

            # Add perseveration bias to RL logits for the last action in this state
            if last_action[s] >= 0:
                # Implement by shifting the chosen last action's value upward
                Q_s[last_action[s]] += persev / 10.0  # modest scaling to keep numerics stable

            # RL policy (with perseveration-shifted Q_s)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with degraded precision under load
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Fixed arbitration weight; WM weakened through beta_wm_eff under load
            wm_mix = np.clip(wm_weight, 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update (without perseveration in learning, only in policy)
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM: slight decay to uniform and reward-driven binding
            # Use a small implicit decay dependent on noise so that higher load also
            # increases forgetting, complementing retrieval noise.
            decay = np.clip(0.02 + 0.08 * (noise / (1.0 + noise)), 0.0, 1.0)
            w = (1.0 - decay) * w + decay * w_0

            if r > 0.5:
                w[s, :] = (1.0 - lr) * w[s, :]
                w[s, a] += lr
            else:
                # On errors, only a mild update toward chosen action (encodes hypothesis)
                w[s, :] = (1.0 - 0.25 * lr) * w[s, :]
                w[s, a] += 0.25 * lr

            # Normalize WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

            # Update last action for perseveration
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
