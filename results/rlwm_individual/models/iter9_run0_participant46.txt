def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 1: RL + WM with load-specific arbitration weights and load-dependent decay

    Idea:
    - RL: tabular Q-learning with softmax (template).
    - WM: fast-learning associative weights per state that move toward one-hot for rewarded action and
      decay more under higher load.
    - Arbitration: separate WM mixture weights for low load (set size 3) vs high load (set size 6).

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - softmax_beta: float >= 0, RL inverse temperature (internally scaled by 10).
    - wm_weight3: float in [0,1], WM mixture weight when nS == 3.
    - wm_weight6: float in [0,1], WM mixture weight when nS == 6.
    - wm_alpha: float in [0,1], WM learning/decay rate (drives update toward one-hot and decay to uniform).

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, softmax_beta, wm_weight3, wm_weight6, wm_alpha = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # choose the arbitration weight based on load
        wm_weight = wm_weight3 if nS <= 3 else wm_weight6

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: softmax over WM weights (deterministic with high beta)
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Load-dependent effective weight (extra decay of WM influence when nS is larger)
            load_scale = 3.0 / float(nS)
            wm_weight_eff = np.clip(wm_weight * load_scale, 0.0, 1.0)

            p_total = wm_weight_eff* p_wm + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM update:
            # 1) global decay toward uniform, stronger under higher load
            decay = wm_alpha * (1.0 - load_scale)
            if decay > 0:
                w = (1 - decay) * w + decay * w_0

            # 2) state-specific learning toward one-hot for rewarded action
            if r > 0.5:
                # move row toward one-hot on chosen action
                w[s,:] = (1 - wm_alpha) * w[s,:]
                w[s,a] += wm_alpha
            else:
                # penalize chosen action slightly to reduce its WM weight
                w[s,a] = (1 - wm_alpha) * w[s,a]
                # renormalize row to avoid drift to zero
                row_sum = np.sum(w[s,:])
                if row_sum <= 0:
                    w[s,:] = w_0[s,:].copy()
                else:
                    w[s,:] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 2: RL + episodic WM with confidence traces and noise

    Idea:
    - RL: tabular Q-learning with softmax (template).
    - WM: stores a one-hot guess for each state's best action when rewarded; retrieval is noisy.
           Each state has a confidence c[s] that increases with reward and decays via a trace parameter.
    - Arbitration: WM weight scales with baseline wm_weight0 times confidence and is dampened by load.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - softmax_beta: float >= 0, RL inverse temperature (internally scaled by 10).
    - wm_weight0: float in [0,1], baseline WM mixture weight.
    - trace_lambda: float in [0,1], confidence persistence (higher = slower decay).
    - wm_noise: float in [0,1], mixture with uniform that injects retrieval noise into WM policy.

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, softmax_beta, wm_weight0, trace_lambda, wm_noise = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # episodic store: for each state, preferred action index; initialize with None (-1)
        preferred = -1 * np.ones(nS, dtype=int)
        # confidence per state
        c = np.zeros(nS)

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Build a WM distribution from the episodic store with noise
            if preferred[s] >= 0:
                wm_vec = np.zeros(3)
                wm_vec[preferred[s]] = 1.0
            else:
                wm_vec = np.ones(3) / 3.0

            wm_vec = (1 - wm_noise) * wm_vec + wm_noise * (np.ones(3) / 3.0)
            # merge with associative w for smoothness
            W_s = 0.5 * w[s, :] + 0.5 * wm_vec

            Q_s = q[s, :]
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy softmax on W_s
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: WM weight based on confidence and load
            load_scale = 3.0 / float(nS)
            wm_weight_eff = np.clip(wm_weight0 * c[s] * load_scale, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # Confidence trace update: decay then increment on reward
            c *= trace_lambda
            if r > 0.5:
                c[s] = 1.0 - (1.0 - c[s]) * (1.0 - trace_lambda)  # push toward 1 after decay
                preferred[s] = a
            else:
                # if negative outcome, reduce confidence in this state
                c[s] *= 0.5

            # Update associative WM weights toward episodic target for stability
            target = np.ones(3) / 3.0
            if preferred[s] >= 0 and r > 0.5:
                target = np.zeros(3)
                target[preferred[s]] = 1.0
            w[s, :] = 0.7 * w[s, :] + 0.3 * target

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Model 3: RL + WM with entropy-based arbitration and load-modulated forgetting

    Idea:
    - RL: tabular Q-learning with softmax (template).
    - WM: graded associative weights per state; updated toward one-hot on rewards using wm_lr; 
      decays to uniform proportionally to load each trial.
    - Arbitration: weight WM more when RL policy is uncertain (high entropy) and WM is confident (low entropy).
      A bias term shifts the arbitration and load further scales WM influence.

    Parameters (model_parameters):
    - lr: float in [0,1], RL learning rate.
    - softmax_beta: float >= 0, RL inverse temperature (internally scaled by 10).
    - wm_weight0: float in [0,1], base WM propensity.
    - entropy_bias: float (can be positive or negative), scales (H_RL - H_WM) into arbitration.
    - wm_lr: float in [0,1], WM learning rate toward one-hot on rewarded pairs (also used in decay magnitude).

    Returns:
    - Negative log-likelihood of observed actions.
    """
    lr, softmax_beta, wm_weight0, entropy_bias, wm_lr = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s,:]
            W_s = w[s,:]

            # RL action prob via softmax trick in template
            p_rl = 1 / np.sum(np.exp(softmax_beta*(Q_s-Q_s[a])))

            # WM policy: softmax on W_s
            p_wm = 1 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute full RL and WM distributions to estimate entropy
            # RL distribution
            q_shift = Q_s - np.max(Q_s)
            rl_probs = np.exp(softmax_beta * q_shift)
            rl_probs /= max(np.sum(rl_probs), 1e-12)
            # WM distribution
            w_shift = W_s - np.max(W_s)
            wm_probs = np.exp(softmax_beta_wm * w_shift)
            wm_probs /= max(np.sum(wm_probs), 1e-12)

            # Entropy (natural log)
            def entropy(p):
                p_safe = np.clip(p, 1e-12, 1.0)
                return -np.sum(p_safe * np.log(p_safe))

            H_rl = entropy(rl_probs)
            H_wm = entropy(wm_probs)

            # Arbitration: sigmoid of base + entropy contrast, scaled by load
            load_scale = 3.0 / float(nS)
            x = wm_weight0 + entropy_bias * (H_rl - H_wm)
            wm_weight_eff = 1.0 / (1.0 + np.exp(-x))
            wm_weight_eff = np.clip(wm_weight_eff * load_scale, 0.0, 1.0)

            p_total = wm_weight_eff*p_wm + (1-wm_weight_eff)*p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)
      
            # RL update
            delta = r-Q_s[a]
            q[s][a] += lr*delta

            # WM decay to uniform increases with load
            decay = wm_lr * (1.0 - load_scale)
            if decay > 0:
                w = (1 - decay) * w + decay * w_0

            # WM learning toward one-hot on rewarded action
            if r > 0.5:
                w[s,:] = (1 - wm_lr) * w[s,:]
                w[s,a] += wm_lr
            else:
                # slight suppression of chosen action if not rewarded
                w[s,a] = (1 - wm_lr) * w[s,a]
                row_sum = np.sum(w[s,:])
                if row_sum <= 0:
                    w[s,:] = w_0[s,:].copy()
                else:
                    w[s,:] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p
