def cognitive_model1(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Hybrid RL + WM with load-dependent WM precision and decay.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate.
    - WM: actively stores rewarded state-action associations; decays toward uniform, and both WM precision and decay worsen with larger set size.
    - Arbitration: fixed WM mixture weight.

    Parameters:
    - lr: RL learning rate (0..1)
    - wm_weight: mixture weight for WM contribution (0..1)
    - softmax_beta: RL inverse temperature; internally scaled by 10 for numerical range
    - wm_decay: base WM decay rate toward uniform per trial (0..1); scaled up with set size
    - wm_precision_base: base WM inverse temperature (>=0), modulated down by load
    - load_penalty: how much WM precision degrades with set size (>=0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_precision_base, load_penalty = model_parameters
    softmax_beta *= 10.0  # RL precision
    softmax_beta_wm = 50.0  # placeholder; replaced below by load-dependent precision
    blocks_log_p = 0.0

    eps = 1e-15
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM precision and decay
        # Precision decreases with set size via a simple divisive rule
        softmax_beta_wm_eff = wm_precision_base / (1.0 + load_penalty * max(0, nS - 3))
        softmax_beta_wm = max(1e-3, softmax_beta_wm_eff)

        # Effective decay increases with set size via compounding
        wm_forget = 1.0 - (1.0 - wm_decay) ** max(1, (nS - 1))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            # RL policy (softmax in ratio form for chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :].copy()
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(p_total + eps)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Global decay toward uniform, stronger at larger set sizes
            w = (1.0 - wm_forget) * w + wm_forget * w_0
            # Reward-contingent one-shot storage
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Hybrid RL + WM with confidence-based arbitration and load-scaled WM decay.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate.
    - WM: rewarded associations stored with high precision; decays toward uniform.
    - Arbitration: trial-wise weight depends on WM confidence (margin between best and 2nd-best WM action),
                  scaled by set size (less WM reliance under high load).

    Parameters:
    - lr: RL learning rate (0..1)
    - wm_weight: baseline WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature; internally scaled by 10
    - arbitration_slope: sensitivity of the arbitration to WM confidence (>=0)
    - wm_decay: base WM decay factor (0..1), scaled up with set size

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, arbitration_slope, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # high precision for WM-selected actions
    blocks_log_p = 0.0
    eps = 1e-15

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-scaled WM decay
        wm_forget = 1.0 - (1.0 - wm_decay) ** max(1, (nS - 1))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            W_s = w[s, :].copy()
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Confidence-based arbitration:
            # confidence = top1 - top2 of WM action values (bounded in [0,1])
            sorted_W = np.sort(W_s)
            top1 = sorted_W[-1]
            top2 = sorted_W[-2] if nA >= 2 else sorted_W[-1]
            confidence = max(0.0, top1 - top2)

            # Map confidence to [0,1] via a logistic controlled by arbitration_slope.
            # Reduce WM weight under larger set sizes by a 3/nS scaling.
            conf_weight = 1.0 / (1.0 + np.exp(-arbitration_slope * (confidence - 0.25)))
            omega = np.clip(wm_weight * conf_weight * (3.0 / nS), 0.0, 1.0)

            p_total = omega * p_wm + (1.0 - omega) * p_rl
            log_p += np.log(p_total + eps)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform, stronger with set size
            w = (1.0 - wm_forget) * w + wm_forget * w_0
            # Reward-gated storage
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, model_parameters):
    """
    Hybrid RL + capacity-limited WM with interference and recency-sensitive precision.

    Mechanism:
    - RL: tabular Q-learning with a single learning rate.
    - WM: capacity-K storage of rewarded state-action mappings; states beyond capacity receive lower WM precision.
          Interference causes global decay; eviction from WM wipes stored associations for evicted states.
          Recency boosts WM precision for recently updated states.
    - Arbitration: fixed WM mixture weight.

    Parameters:
    - lr: RL learning rate (0..1)
    - wm_weight: WM mixture weight (0..1)
    - softmax_beta: RL inverse temperature; internally scaled by 10
    - capacity_K: effective WM capacity in number of states (1..6, real-valued allowed; rounded internally)
    - interference: global decay strength toward uniform per trial (0..1)
    - recency_beta: multiplicative precision boost factor for recently updated states (>=0)

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, capacity_K, interference, recency_beta = model_parameters
    softmax_beta *= 10.0
    base_beta_wm_in = 12.0  # lower precision when out of capacity
    base_beta_wm_incap = 50.0  # high precision when in capacity
    blocks_log_p = 0.0
    eps = 1e-15

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        K = int(np.clip(np.round(capacity_K), 1, nS))
        in_cap = np.zeros(nS, dtype=bool)
        recency_tick = np.zeros(nS, dtype=int)  # last updated time; higher = more recent
        time_counter = 0

        # A simple recency-ordered list of states in WM
        wm_list = []

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])
            time_counter += 1

            Q_s = q[s, :].copy()
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Determine precision depending on capacity membership and recency
            if in_cap[s]:
                # Recency-based precision boost
                # More recent updates get higher precision; we normalize by time scale using a simple exponential-like rule
                # recency_age = current time - last update time
                age = max(1, time_counter - recency_tick[s])
                boost = 1.0 + recency_beta / np.sqrt(age)
                beta_wm_eff = base_beta_wm_incap * boost
            else:
                beta_wm_eff = base_beta_wm_in

            W_s = w[s, :].copy()
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(p_total + eps)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Global interference-driven decay toward uniform; stronger when set size exceeds K
            scale = (nS / max(1, K))
            lam = 1.0 - (1.0 - interference) ** scale
            w = (1.0 - lam) * w + lam * w_0

            # Reward-triggered WM update with capacity management
            if r > 0.0:
                # Store as one-hot
                w[s, :] = 0.0
                w[s, a] = 1.0
                recency_tick[s] = time_counter

                # Maintain recency-ordered capacity list
                if in_cap[s]:
                    # Move s to the end (most recent)
                    wm_list = [x for x in wm_list if x != s]
                    wm_list.append(s)
                else:
                    # Add s; if overflow, evict oldest
                    wm_list.append(s)
                    in_cap[s] = True
                    if len(wm_list) > K:
                        evict = wm_list.pop(0)
                        in_cap[evict] = False
                        # Eviction wipe: remove strong memory by resetting toward uniform
                        w[evict, :] = w_0[evict, :]

        blocks_log_p += log_p

    return -blocks_log_p
