def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the dual-learning-rate RL model with
    set-size-dependent exploration and age-dependent perseveration (stickiness).

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct action indices per trial (0..2)
    - age: participant's age in years (float)

    Returns:
    - simulated_actions: array of simulated action choices per trial
    - simulated_rewards: array of simulated binary rewards per trial
    """

    alpha_pos, alpha_neg, beta_small, beta_large, kappa_young, kappa_old = parameters

    # Apply the same scaling used during fitting
    beta_small = max(1e-6, beta_small) * 10.0
    beta_large = max(1e-6, beta_large) * 10.0

    is_old = 1 if (np.asarray(age)[0] >= 45 if np.ndim(age) > 0 else age >= 45) else 0
    kappa = kappa_old if is_old else kappa_young

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_indices = np.arange(n_trials)

    for b in np.unique(blocks):
        mask = (blocks == b)
        idx = tr_indices[mask]

        block_states = np.asarray(stimulus)[mask].astype(int)
        block_correct = np.asarray(correct_answer)[mask].astype(int)
        block_set_sizes = np.asarray(set_sizes)[mask].astype(int)

        nA = 3
        # Number of states in this block inferred from set size value on trials in this block
        nS = int(block_set_sizes[0])

        # Initialize Q-values and last-action tracker per state
        Q = np.ones((nS, nA)) / nA
        last_action_per_state = -1 * np.ones(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])
            correct_a = int(block_correct[t])
            nS_block = int(block_set_sizes[t])

            beta = beta_small if nS_block <= 3 else beta_large

            # Preferences with stickiness toward last action in this state
            pref = beta * Q[s, :].copy()
            if last_action_per_state[s] >= 0:
                pref[last_action_per_state[s]] += kappa

            # Softmax
            pref -= np.max(pref)
            expv = np.exp(pref)
            p_vec = expv / np.sum(expv)

            # Sample action
            a = int(np.random.choice(nA, p=p_vec))
            simulated_actions[idx[t]] = a

            # Outcome based on correctness
            r = 1 if a == correct_a else 0
            simulated_rewards[idx[t]] = r

            # RL update with dual learning rates
            pe = float(r) - Q[s, a]
            lr = alpha_pos if pe >= 0.0 else alpha_neg
            Q[s, a] += lr * pe

            # Update stickiness memory for this state
            last_action_per_state[s] = a

    return simulated_actions, simulated_rewards
