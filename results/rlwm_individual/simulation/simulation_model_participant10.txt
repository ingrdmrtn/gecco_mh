def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using:
    - RL with asymmetric learning rates (separate for rewards and no-rewards)
    - WM with interference that increases with set size
    - Fixed WM/RL mixture per block
    - WM policy noisiness depends on set size via interference

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of block-dependent set sizes per trial
    - correct_answer: array of correct actions per trial
    - parameters: [lr_pos, lr_neg, wm_weight, softmax_beta, gamma]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, gamma = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use the provided set size for the block (as in fitting code)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM interference increases with set size
        phi = 1.0 - np.exp(-gamma * float(nS))
        # Effective WM inverse temperature decreases with interference
        eff_beta_wm = max(1.0, softmax_beta_wm * (1.0 - 0.8 * phi))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (softmax)
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy (softmax with interference-adjusted beta)
            p_wm = np.exp(eff_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Generate reward from correct action mapping
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[tr] = r

            # RL update with asymmetric learning rates
            lr = lr_pos if r > 0.5 else lr_neg
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform with interference phi
            w[s, :] = (1.0 - phi) * w[s, :] + phi * w_0[s, :]

            # If rewarded, store perfect memory of chosen action
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Keep WM distribution well-formed
            w[s, :] = np.maximum(w[s, :], 1e-12)
            w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
