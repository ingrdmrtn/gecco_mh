def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from the RL + capacity-limited WM mixture model
    (converted from the provided likelihood code).

    Inputs:
    - stimulus: within-block state index for each trial (array of ints starting at 0)
    - blocks: block index for each trial (array)
    - set_sizes: block set size (3 or 6) for each trial (array)
    - correct_answer: correct action (0,1,2) for each trial (array)
    - age: participant's age (float)

    Returns:
    - simulated_actions: simulated action choices (array of ints in {0,1,2})
    - simulated_rewards: simulated binary rewards (array of ints in {0,1})
    """

    alpha, beta, K, phi, w0, epsilon = parameters

    beta *= 10.0
    age_val = float(age)
    is_older = 1 if age_val >= 45 else 0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]
        block_correct = correct_answer[block_idx].astype(int)

        # Determine number of states in this block from provided set size (as in fit code)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM to uniform
        Q = (1.0 / nA) * np.ones((nS, nA))
        WM = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM capacity adjusted by age
        K_eff = max(1.0, K + (1.0 if is_older == 0 else -1.0))

        # Build per-state correct action (assumed fixed within block per state)
        # Fallback: use first occurrence for each state
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            mask = (block_states == s)
            if np.any(mask):
                state_correct[s] = int(block_correct[mask][0])
            else:
                state_correct[s] = 0  # default (should not happen if states appear)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            Q_s = Q[s, :]
            z = Q_s - np.max(Q_s)
            expz = np.exp(beta * z)
            p_rl = expz / np.sum(expz)

            # WM decay toward uniform for the current state (as in fit code)
            decay_strength = min(1.0, 0.05 * (nS / 3.0))
            WM[s, :] = (1.0 - decay_strength) * WM[s, :] + decay_strength * (1.0 / nA)

            # WM policy is the current WM distribution over actions for state s
            p_wm = WM[s, :].copy()
            # Ensure numerical stability
            p_wm = np.clip(p_wm, 1e-12, 1.0)
            p_wm /= np.sum(p_wm)

            # Age- and load-adjusted WM mixture weight
            load_ratio = nS / max(1.0, K_eff)
            wm_weight = w0 * (1.0 / (1.0 + (load_ratio - 1.0) * (2.0 if is_older else 1.0)))
            wm_weight = float(np.clip(wm_weight, 0.0, 1.0))

            # Mixture with lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - epsilon) * p_mix + epsilon * (1.0 / nA)
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action for state s
            r = 1 if a == state_correct[s] else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            Q[s, a] += alpha * delta

            # WM refresh on rewarded trials
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                WM[s, :] = (1.0 - phi) * WM[s, :] + phi * onehot

            tr += 1

    return simulated_actions, simulated_rewards
