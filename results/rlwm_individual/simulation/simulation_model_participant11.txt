def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate actions and rewards using the "Surprise-gated WM encoding with confidence-weighted arbitration and load scaling" model.

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (block-constant)
    - correct_answer: array of correct actions per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_alpha, surprise_gate, load_beta]

    Returns:
    - simulated_actions: array of simulated action choices
    - simulated_rewards: array of simulated binary rewards
    """
    lr, wm_weight, softmax_beta, wm_alpha, surprise_gate, load_beta = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size (as in fitting code)
        nS = int(block_set_sizes[0])

        # Build per-state correct action mapping (assumes stationary within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # Find first occurrence of this state within the block
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = int(block_correct[idx[0]])
            else:
                # If state not present (rare), default to 0
                correct_actions[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load scaling of base WM weight
        load_scale = (3.0 / float(nS)) ** max(0.0, load_beta)
        wm_weight_block = max(0.0, min(1.0, wm_weight * min(1.0, load_scale)))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Softmax probabilities for RL and WM
            # RL
            q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * q_shift)
            p_rl /= np.sum(p_rl)
            # WM
            w_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * w_shift)
            p_wm /= np.sum(p_wm)

            # Arbitration weight with surprise-gated boost by PE and WM confidence
            # Compute current PE relative to chosen action's Q after choice; for arbitration we need a PE estimate.
            # Use expected PE if choosing stochastically is not defined; instead follow fitting logic:
            # compute PE after observing outcome of selected action. Here we must choose action first,
            # but arbitration depends on PE; in fitting it used observed action to get PE.
            # Therefore, compute p_total with base wm_weight_block for sampling the action,
            # then after observing r, compute PE and update wm_weight_t for logging; in simulation we need action sampling now.
            # To match fitting where arbitration weight is computed prior to sampling, we approximate by using PE based on current best guess:
            # However, to adhere closely, we will compute arbitration using PE from the last trial's Q for the chosen action.
            # We therefore first compute a temporary mixture using current confidence but with zero PE (no boost), sample action,
            # then compute the actual boost for updates (does not affect sampling in this implementation step).

            # Confidence from WM (difference between top two values)
            sorted_W = np.sort(W_s)[::-1]
            conf = max(0.0, sorted_W[0] - (sorted_W[1] if len(sorted_W) > 1 else 0.0))

            # Use base mixture for sampling (boost requires PE which needs action and reward)
            wm_weight_t_sample = wm_weight_block
            p_total = wm_weight_t_sample * p_wm + (1.0 - wm_weight_t_sample) * p_rl
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # Compute PE for chosen action
            pe = r - Q_s[a]

            # Compute surprise-gated boost now (used conceptually in fitting before observing, but updates unaffected)
            wm_boost = surprise_gate * abs(pe) * conf
            wm_weight_t = max(0.0, min(1.0, wm_weight_block + wm_boost))

            # RL update
            q[s, a] += lr * pe

            # WM update
            if r > 0.5:
                # Strong encode chosen action, shrink others
                w[s, :] = (1.0 - wm_alpha) * w[s, :]
                w[s, a] += wm_alpha
            else:
                # Error-driven redistribution: move mass from chosen to others
                reduce = wm_alpha * min(1.0, w[s, a])
                w[s, a] = max(0.0, w[s, a] - reduce)
                increment = reduce / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += increment

            # Renormalize WM row; if degenerate, reset to uniform prior
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
