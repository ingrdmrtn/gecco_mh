def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using:
      - RL: tabular Q-learning with softmax; Q-values globally forget toward uniform.
      - WM: one-shot storage upon reward; competitive decay scales up with set size (interference).
      - Mixture: constant wm_weight scaled down by interference from set size.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within a block)
    - blocks: array of block index per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial (0..nA-1)
    - parameters: [lr, softmax_beta, wm_weight, wm_one_shot, interference, q_forget]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, softmax_beta, wm_weight, wm_one_shot, interference, q_forget = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate blocks
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        # Set size for this block (assumed constant within block, as in fitting code)
        nS = int(set_sizes[block_indices][0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Block-level effective mixture weight and base WM decay (interference-scaled)
        wm_weight_eff_block = wm_weight / (1.0 + interference * max(0.0, float(nS - 1)))
        wm_weight_eff_block = np.clip(wm_weight_eff_block, 0.0, 1.0)
        base_wm_decay = np.clip(interference * (float(nS) / 6.0), 0.0, 1.0)

        for t, tr in enumerate(block_indices):
            s = int(block_states[t])

            # Current policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax probabilities
            q_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(q_logits)
            p_rl /= np.sum(p_rl)

            # WM softmax probabilities
            w_logits = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(w_logits)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_eff_block * p_wm + (1.0 - wm_weight_eff_block) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Determine reward (correct if chosen action equals correct answer on this trial)
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[tr] = r

            # RL forgetting toward uniform, then TD update
            q[s, :] = (1.0 - q_forget) * q[s, :] + q_forget * (1.0 / nA)
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM competitive decay, then one-shot write on rewarded trials
            w[s, :] = (1.0 - base_wm_decay) * w[s, :] + base_wm_decay * w_0[s, :]
            if r > 0:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_one_shot) * w[s, :] + wm_one_shot * target

    return simulated_actions, simulated_rewards
