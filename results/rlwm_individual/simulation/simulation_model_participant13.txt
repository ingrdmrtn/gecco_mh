def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM arbitration via prediction error and set-size-dependent WM lapses (simulation).

    Parameters:
    - stimulus: sequence of stimuli/states for each trial (array of ints, 0..nS-1)
    - blocks: block index for each trial (array)
    - set_sizes: set size for each trial (array; constant within a block)
    - correct_answer: correct action for each trial (array of ints, 0..nA-1)
    - parameters: [lr_pos, lr_neg, wm_weight0, softmax_beta, lapse0]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr_pos, lr_neg, wm_weight0, softmax_beta, lapse0 = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]
        block_correct = correct_answer[block_indices].astype(int)

        # Determine number of states in block and map each state's correct action
        nS = int(block_set_sizes[0])
        # Map each state to its correct action (assumes consistency within block)
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                state_correct[s] = block_correct[idx[0]]
            else:
                # If a state is not observed (unlikely), default to 0
                state_correct[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size dependent WM lapse
        if nS <= 3:
            lapse = 0.0
        else:
            lapse = np.clip(lapse0 * (float(nS - 3) / max(1.0, 6.0 - 3.0)), 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])
            correct_a = int(state_correct[s])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            Q_center = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_center)
            p_rl = p_rl / np.maximum(np.sum(p_rl), eps)

            # WM core policy
            W_center = W_s - np.max(W_s)
            p_wm_core = np.exp(softmax_beta_wm * W_center)
            p_wm_core = p_wm_core / np.maximum(np.sum(p_wm_core), eps)

            # WM lapse mixture with uniform
            p_uniform = np.ones(nA) / nA
            p_wm = (1.0 - lapse) * p_wm_core + lapse * p_uniform

            # Action-specific arbitration weight using predicted PE for each action
            # Expectation: r_hat = 1 for correct action, else 0
            r_hat = np.zeros(nA)
            r_hat[correct_a] = 1.0
            pe_hat = r_hat - Q_s  # vector of predicted PEs if each action were chosen
            eff_w_vec = np.clip(wm_weight0 * (1.0 - np.abs(pe_hat)), 0.0, 1.0)

            # Elementwise arbitration and normalization
            p_total = eff_w_vec * p_wm + (1.0 - eff_w_vec) * p_rl
            p_total_sum = np.sum(p_total)
            if p_total_sum <= 0:
                p_total = np.ones(nA) / nA
            else:
                p_total = p_total / p_total_sum

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            if r > Q_s[a]:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM update
            if r > 0.0:
                alpha_pos = 1.0
                w[s, :] = (1.0 - alpha_pos) * w[s, :]
                w[s, a] += alpha_pos  # sets to one-hot on chosen action
            else:
                alpha_neg = 0.2
                w[s, a] = max(eps, (1.0 - alpha_neg) * w[s, a])
                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum

            tr += 1

    return simulated_actions, simulated_rewards
