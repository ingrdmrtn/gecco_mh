def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Surprise-gated RL+WM with load-dependent arbitration and load-driven WM decay (simulation).

    Parameters:
    - stimulus: sequence of stimuli (state indices) for each trial (array-like, int)
    - blocks: block index for each trial (array-like, int)
    - set_sizes: set size on each trial (array-like, int); assumed constant within a block
    - correct_answer: correct action on each trial (array-like, int)
    - parameters: [lr, softmax_beta, wm_base, surprise_gain, load_scale, wm_learn]

    Returns:
    - simulated_actions: simulated action choices (array, int)
    - simulated_rewards: simulated binary rewards (array, int)
    """
    lr, softmax_beta, wm_base, surprise_gain, load_scale, wm_learn = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumes consistency within state)
        unique_states = np.unique(block_states)
        # Ensure states are 0..nS-1; if not, remap
        if not np.array_equal(unique_states, np.arange(nS)):
            # Create a mapping from observed state labels to 0..nS-1
            state_map = {s_old: i for i, s_old in enumerate(unique_states)}
            mapped_states = np.array([state_map[s] for s in block_states], dtype=int)
            # Build correct action per internal state
            correct_actions = np.zeros(nS, dtype=int)
            for s_old, s_new in state_map.items():
                correct_actions[s_new] = block_correct[block_states == s_old][0]
        else:
            mapped_states = block_states.copy()
            correct_actions = np.zeros(nS, dtype=int)
            for s in range(nS):
                correct_actions[s] = block_correct[block_states == s][0]

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        load_term = max(0.0, (nS - 3))
        wm_decay = np.clip(load_scale * (load_term / max(1.0, nS)), 0.0, 1.0)

        for t in range(len(block_states)):
            s = mapped_states[t]

            # Global WM decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax
            q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * q_shift)
            p_rl /= np.sum(p_rl)

            # WM softmax (near-deterministic)
            w_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * w_shift)
            p_wm /= np.sum(p_wm)

            # Compute action-specific arbitration weight using predicted surprise
            correct_a = correct_actions[s]
            wm_weights_per_a = np.zeros(nA)
            mix = np.zeros(nA)
            for a in range(nA):
                r_hat = 1.0 if a == correct_a else 0.0
                surprise = abs(r_hat - Q_s[a])
                wm_logit = wm_base + surprise_gain * surprise - load_scale * load_term
                wm_weight_a = 1.0 / (1.0 + np.exp(-wm_logit))
                wm_weights_per_a[a] = wm_weight_a
                mix[a] = wm_weight_a * p_wm[a] + (1.0 - wm_weight_a) * p_rl[a]

            # Normalize mixture to obtain valid probabilities
            mix_sum = np.sum(mix)
            if mix_sum <= 0 or not np.isfinite(mix_sum):
                # Fallback to RL policy if numerical issues arise
                p_total = p_rl
            else:
                p_total = mix / mix_sum

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Realized reward
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update uses the arbitration weight corresponding to chosen action
            wm_weight_chosen = wm_weights_per_a[a]
            if r > 0:
                target = w_0[s, :].copy()
                target[a] = 1.0
                target /= np.sum(target)
                eta = wm_learn * wm_weight_chosen
                w[s, :] = (1.0 - eta) * w[s, :] + eta * target
            else:
                eta = 0.25 * wm_learn * (1.0 + wm_weight_chosen)
                w[s, :] = (1.0 - eta) * w[s, :] + eta * w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
