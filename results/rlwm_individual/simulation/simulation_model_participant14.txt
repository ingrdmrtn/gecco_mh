import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards under the RL+WM mixture model with capacity- and decay-modulated WM.

    Inputs:
    - stimulus: sequence of state indices per trial (array of int; states are 0..nS-1 within block)
    - blocks: block index for each trial (array of int)
    - set_sizes: set size for each trial (array of int; constant within a block)
    - correct_answer: correct action per trial (array of int; used to define the state->correct action mapping)
    - parameters: [lr, wm_weight, softmax_beta, K, wm_decay]

    Returns:
    - simulated_actions: simulated action choices (array of int)
    - simulated_rewards: simulated binary rewards (array of int)
    """
    lr, wm_weight, softmax_beta, K, wm_decay = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]
        block_correct = correct_answer[block_idx].astype(int)

        # Determine set size for this block
        nS = int(block_set_sizes[0])

        # Build state -> correct action map (use first occurrence of each state)
        state_to_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t in range(len(block_states)):
            s = block_states[t]
            if not seen[s]:
                state_to_correct[s] = block_correct[t]
                seen[s] = True

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Apply WM decay to the current state's WM values before computing choice
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax policy
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM softmax policy (very sharp)
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # WM availability scaled by capacity relative to current set size
            nS_curr = float(nS)
            avail = min(1.0, max(0.0, K / nS_curr))
            wm_eff = wm_weight * avail

            # Mixture policy
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_idx[t]] = a

            # Determine reward from correct action mapping
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: strong write of rewarded action after decay
            if r > 0.5:
                # reset to nearly one-hot for chosen action
                eps = 1e-6
                w[s, :] = eps
                w[s, a] = 1.0 - (nA - 1) * eps

            tr += 1

    return simulated_actions, simulated_rewards
