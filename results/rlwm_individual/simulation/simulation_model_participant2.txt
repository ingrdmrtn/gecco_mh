def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using:
    RL with asymmetric learning + WM retrieval noise increasing with set size.

    Inputs:
    - stimulus: array of state indices (assumed 0..nS-1 within block)
    - blocks: array of block indices (same length as stimulus)
    - set_sizes: array of set sizes per trial (used to get nS per block; constant within block)
    - correct_answer: array of correct actions per trial
    - parameters: [rl_alpha_pos, rl_alpha_neg, softmax_beta, wm_mix, noise_gain]

    Returns:
    - simulated_actions: array of chosen actions
    - simulated_rewards: array of binary rewards
    """
    rl_alpha_pos, rl_alpha_neg, softmax_beta, wm_mix, noise_gain = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        # Set size for this block (assumed constant within block)
        nA = 3
        nS = int(set_sizes[block_indices][0])

        # Build per-state correct action lookup from first occurrence
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = int(block_correct[idx[0]])
            else:
                # If a state never appears (edge case), default to 0
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM retrieval noise as a function of set size
        gamma = 1.0 / (1.0 + np.exp(-noise_gain * (nS - 3.0)))
        gamma = np.clip(gamma, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            Q_s = q[s, :]
            Q_shift = Q_s - np.max(Q_s)
            expQ = np.exp(softmax_beta * Q_shift)
            p_rl_vec = expQ / np.sum(expQ)

            # WM retrieval with load-dependent noise, then policy
            W_clean = w[s, :]
            W_retrieved = (1.0 - gamma) * W_clean + gamma * w_0[s, :]
            W_shift = W_retrieved - np.max(W_retrieved)
            expW = np.exp(softmax_beta_wm * W_shift)
            p_wm_vec = expW / np.sum(expW)

            # Mixture policy
            wm_weight = np.clip(wm_mix, 0.0, 1.0)
            p_total = wm_weight * p_wm_vec + (1.0 - wm_weight) * p_rl_vec

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward delivery based on correct action
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update (asymmetric learning rates)
            alpha = rl_alpha_pos if r > 0.5 else rl_alpha_neg
            delta = r - Q_s[a]
            q[s, a] += alpha * delta

            # WM update with mild compression/recovery and normalization
            if r > 0.5:
                w[s, :] = 0.9 * w[s, :]
                w[s, a] += 0.1
            else:
                w[s, :] = 0.95 * w[s, :] + 0.05 * w_0[s, :]

            w[s, :] = np.clip(w[s, :], 0.0, None)
            w_sum = np.sum(w[s, :])
            if w_sum > 0:
                w[s, :] /= w_sum
            else:
                w[s, :] = w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
