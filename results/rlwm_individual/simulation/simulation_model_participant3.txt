def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the RL+WM model with outcome-gated WM reliance
    and unified WM update/decay rate.

    Inputs:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - parameters: (lr, wm_weight, softmax_beta, win_boost, loss_suppression, kappa)

    Returns:
    - simulated_actions: simulated action choices (array)
    - simulated_rewards: simulated binary rewards (array)
    """
    lr, wm_weight, softmax_beta, win_boost, loss_suppression, kappa = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_indices]
        block_correct_ans = correct_answer[block_indices]
        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(set_sizes[block_indices][0])

        # Map raw state labels to 0..nS-1 to ensure valid indexing
        unique_states = np.unique(block_states_raw)
        state_map = {s_raw: i for i, s_raw in enumerate(unique_states)}
        # Build correct action per mapped state (use first occurrence)
        correct_actions = np.zeros(len(unique_states), dtype=int)
        for s_raw in unique_states:
            idx = np.where(block_states_raw == s_raw)[0][0]
            correct_actions[state_map[s_raw]] = int(block_correct_ans[idx])

        # Initialize RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Initialize gate
        g = 0.0

        for t in range(len(block_states_raw)):
            s_mapped = state_map[block_states_raw[t]]
            # In case mapped states are fewer than nS, indexing still valid
            Q_s = q[s_mapped, :]
            W_s = w[s_mapped, :]

            # RL policy
            Q_center = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_center)
            p_rl /= np.sum(p_rl)

            # WM policy
            W_center = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_center)
            p_wm /= np.sum(p_wm)

            # Effective WM weight via logistic gating
            eps = 1e-12
            wm_w = min(max(wm_weight, eps), 1.0 - eps)
            logit_w = np.log(wm_w) - np.log(1.0 - wm_w)
            z = logit_w + g
            wm_eff = 1.0 / (1.0 + np.exp(-z))

            # Mixture policy
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action for this state
            correct_a = int(correct_actions[s_mapped])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # Update gate
            target_g = win_boost if r > 0.5 else -loss_suppression
            g = (1.0 - kappa) * g + kappa * target_g

            # RL update
            delta = r - Q_s[a]
            q[s_mapped, a] += lr * delta

            # WM update: unified update/decay
            one_hot = np.zeros(nA)
            one_hot[a] = 1.0
            if r > 0.5:
                w[s_mapped, :] = (1.0 - kappa) * w[s_mapped, :] + kappa * one_hot
            else:
                w[s_mapped, :] = (1.0 - kappa) * w[s_mapped, :] + kappa * w_0[s_mapped, :]

            tr += 1

    return simulated_actions, simulated_rewards
