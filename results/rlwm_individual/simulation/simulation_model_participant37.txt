def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards for:
    RL with asymmetric learning rates and set-size-depressed precision + WM with set-size-driven interference.

    Inputs:
    - stimulus: array of states per trial (integers 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array indicating set size for each trial (constant within a block)
    - correct_answer: array of correct action (0..2) per trial
    - parameters: [alpha_pos, alpha_neg, softmax_beta, wm_weight, wm_interference, size_beta_drop]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    alpha_pos, alpha_neg, softmax_beta, wm_weight, wm_interference, size_beta_drop = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size effects
        beta_eff = softmax_beta * np.exp(-max(0, nS - 3) * size_beta_drop)
        gamma_eff = 1.0 - (1.0 - wm_interference) ** (1 + max(0, nS - 3))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute policy probabilities
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            p_rl = np.exp(beta_eff * (Q_s - np.max(Q_s)))
            p_rl = p_rl / np.sum(p_rl)

            # WM softmax (near-deterministic)
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm = p_wm / np.sum(p_wm)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Outcome
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update with asymmetric learning rate
            pe = r - Q_s[a]
            alpha_t = alpha_pos if pe >= 0.0 else alpha_neg
            q[s, a] += alpha_t * pe

            # WM interference toward uniform, then rewarded overwriting
            w[s, :] = (1.0 - gamma_eff) * w[s, :] + gamma_eff * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
