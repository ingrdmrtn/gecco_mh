import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM mixture with confidence-gated WM policy and reward-gated storage.
    Simulation counterpart of cognitive_model2.

    Inputs:
    - stimulus: array of state indices per trial (assumed 0..nS-1 within each block)
    - blocks: array of block index per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial (0..nA-1)
    - parameters: (lr, wm_weight, softmax_beta, lr_wm, decay, theta)

    Returns:
    - simulated_actions: array of chosen actions (0..nA-1)
    - simulated_rewards: array of rewards (0/1)
    """
    lr, wm_weight, softmax_beta, lr_wm, decay, theta = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    unique_blocks = np.unique(blocks)
    for b in unique_blocks:
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_idx)):
            s = block_states[t]
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax policy
            p_rl = np.exp(softmax_beta * Q_s)
            p_rl /= np.sum(p_rl)



            # WM confidence gating based on margin between top two W_s
            sorted_vals = np.sort(W_s)[::-1]
            if nA >= 2:
                margin = sorted_vals[0] - sorted_vals[1]
            else:
                margin = sorted_vals[0]

            if margin <= theta:
                gamma = 0.0
            else:
                gamma = min(1.0, (margin - theta) / max(eps, 1.0 - theta))

            beta_wm_eff = softmax_beta_wm * gamma
                        # WM softmax policy (very sharp)
            p_wm_soft = np.exp(beta_wm_eff * W_s)
            p_wm_soft /= np.sum(p_wm_soft)



            p_uniform = np.ones(nA) / nA
            p_wm = gamma * p_wm_soft + (1.0 - gamma) * p_uniform

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = p_total / np.maximum(p_total.sum(), eps)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_idx[t]] = a

            # Determine reward from correct answer on this trial
            correct_a = block_correct[t]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM global decay toward uniform
            w = (1.0 - decay) * w + decay * w_0

            # Reward-gated WM storage (only if rewarded)
            if r > 0.5:
                w[s, :] = (1.0 - lr_wm) * w[s, :]
                w[s, a] += lr_wm

            tr += 1

    return simulated_actions, simulated_rewards
