def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with capacity-limited working memory, WM decay, and WM learning rate.

    Parameters
    ----------
    - stimulus: array of state indices for each trial
    - blocks: array of block indices for each trial
    - set_sizes: array of set size (number of distinct states) for each trial
    - correct_answer: array of correct action per trial
    - parameters: tuple/list of 6 floats
        lr:            RL learning rate (0..1)
        wm_weight:     Base WM mixture weight (0..1)
        softmax_beta:  RL inverse temperature (scaled by 10 internally)
        wm_decay:      WM decay toward uniform per trial (0..1)
        wm_lr:         WM learning rate for rewarded associations (0..1)
        wm_capacity:   WM capacity in number of state-action pairs

    Returns
    -------
    - simulated_actions: array of simulated actions
    - simulated_rewards: array of simulated rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_lr, wm_capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        # Determine set size for this block (assumed constant within block)
        nA = 3
        nS = int(set_sizes[block_idx][0])

        # Map from state to correct action using first occurrence
        # Assumes states are labeled 0..nS-1
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of this state within the block
            where_s = np.where(block_states == s)[0]
            if len(where_s) == 0:
                # If a state label is not present (shouldn't happen), default to 0
                correct_actions_by_state[s] = 0
            else:
                correct_actions_by_state[s] = int(block_correct[where_s[0]])

        # Initialize RL and WM values to uniform
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight scaled by capacity relative to set size
        wm_eff_scale = min(1.0, float(wm_capacity) / float(nS))
        wm_eff = wm_weight * wm_eff_scale

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax
            Q_s_center = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_s_center)
            p_rl /= np.sum(p_rl)

            # WM softmax (very sharp)
            W_s_center = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_s_center)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Generate reward based on correct action for this state
            correct_a = int(correct_actions_by_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay for the current state toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update on rewarded trials toward one-hot for chosen action
            if r > 0:
                oh = np.zeros(nA)
                oh[a] = 1.0
                w[s, :] = (1.0 - wm_lr) * w[s, :] + wm_lr * oh

            tr += 1

    return simulated_actions, simulated_rewards
