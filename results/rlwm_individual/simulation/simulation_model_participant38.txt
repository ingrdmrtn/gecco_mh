import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the RL+WM model with prediction-error-gated arbitration
    and load-modulated gating, matching the provided fitting logic.

    Inputs:
    - stimulus: array of state identifiers per trial (assumed 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set-size values per trial (used to get nS per block)
    - correct_answer: array of correct actions per trial (0..nA-1)
    - parameters: [lr, wm_weight, softmax_beta, pe_gate, wm_decay, wm_learn]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    lr, wm_weight, softmax_beta, pe_gate, wm_decay, wm_learn = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]
        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute load term for gating
        load_term = np.clip((nS - 3.0) / 3.0, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])
            correct_a_t = int(block_correct[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Action-conditional gating via predicted PE for each candidate action
            # Reward is deterministic given correct action on this trial
            r_all = np.zeros(nA)
            r_all[correct_a_t] = 1.0

            pe_all = np.abs(r_all - Q_s)  # |r(a) - Q(s,a)|
            gate_all = pe_gate * (1.0 + load_term) * pe_all

            # Map baseline wm_weight through logit, subtract gate, then sigmoid back
            eps = 1e-8
            wm_w_clipped = np.clip(wm_weight, eps, 1.0 - eps)
            logit_w0 = np.log(wm_w_clipped) - np.log(1.0 - wm_w_clipped)
            wm_weight_t_all = 1.0 / (1.0 + np.exp(-(logit_w0 - gate_all)))

            # Mixture for each action, then normalize to get a proper distribution
            p_mix = wm_weight_t_all * p_wm + (1.0 - wm_weight_t_all) * p_rl
            p_mix = np.maximum(p_mix, 1e-12)
            p_mix /= np.sum(p_mix)

            # Sample action
            a = int(np.random.choice(nA, p=p_mix))
            simulated_actions[block_indices[t]] = a

            # Realized reward
            r = 1 if a == correct_a_t else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Keep WM row a valid distribution
            w[s, :] = np.clip(w[s, :], 1e-8, 1.0)
            w[s, :] /= np.sum(w[s, :])

            tr_global += 1

    return simulated_actions, simulated_rewards
