import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with recency-weighted WM and per-state perseveration.

    Parameters:
    - stimulus: sequence of stimuli (state indices 0..nS-1 within each block) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - parameters: [lr, wm_weight, softmax_beta, wm_recency_lambda, perseveration_kappa, wm_beta_scale]

    Returns:
    - simulated_actions: simulated action choices (array, int)
    - simulated_rewards: simulated binary rewards (array, int)
    """

    lr, wm_weight, softmax_beta, wm_recency_lambda, perseveration_kappa, wm_beta_scale = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    nA = 3

    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_actions = correct_answer[block_indices]
        nS = int(set_sizes[block_indices][0])

        # Map each state (0..nS-1) to its correct action within the block (assumed stationary)
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for st in range(nS):
            idx = np.where(block_states == st)[0]
            if len(idx) > 0:
                correct_actions_by_state[st] = block_correct_actions[idx[0]]
            else:
                # Fallback (should not happen if all states appear)
                correct_actions_by_state[st] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        last_seen = -1 * np.ones(nS, dtype=int)
        prev_action = -1 * np.ones(nS, dtype=int)

        beta_wm_eff = softmax_beta_wm * (max(1e-6, wm_beta_scale) ** max(0, nS - 1))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with perseveration
            Q_s = q[s, :].copy()
            if prev_action[s] >= 0:
                Q_s[prev_action[s]] += perseveration_kappa
            # Stable softmax for RL
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM effective trace with recency
            if last_seen[s] >= 0:
                dt = t - last_seen[s]
                rec = (wm_recency_lambda ** dt)
                W_s_eff = rec * w[s, :] + (1.0 - rec) * w_0[s, :]
            else:
                W_s_eff = w[s, :]

            wm_logits = beta_wm_eff * W_s_eff
            wm_logits -= np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Arbitration
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_indices[t]] = a

            # Outcome based on correct mapping for this state
            correct_a = int(correct_actions_by_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update: store strongly only when rewarded
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot

            # Bookkeeping for recency and perseveration
            last_seen[s] = t
            prev_action[s] = a

            tr_global += 1

    return simulated_actions, simulated_rewards
