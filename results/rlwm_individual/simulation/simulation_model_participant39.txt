def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM capacity-and-forgetting simulation (slot-like WM + model-free RL).

    Inputs:
    - stimulus: array of state indices for each trial
    - blocks: array of block id for each trial
    - set_sizes: array of set size (nS) for each trial (constant within a block)
    - correct_answer: array of correct action (0..nA-1) for each trial
    - parameters: [lr, wm_capacity, softmax_beta, wm_decay, lapse]

    Returns:
    - simulated_actions: array of sampled actions (0..nA-1)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_capacity, softmax_beta, wm_decay, lapse = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # not explicitly used (WM uses stored probs + lapse)
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3  # number of actions

    # Iterate block by block
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]

        # Determine set size (assumed constant within a block)
        nS = int(set_sizes[block_idx[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-scaled WM weight for the whole block
        cap_scale = min(1.0, max(0.0, wm_capacity) / max(1.0, nS))
        wm_weight_eff = cap_scale

        # Trial loop within block
        for local_t, tr in enumerate(block_idx):
            s = int(block_states[local_t])
            correct_a = int(block_correct[local_t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy: softmax over Q-values
            q_shift = Q_s - np.max(Q_s)  # for numerical stability
            p_rl = np.exp(softmax_beta * q_shift)
            p_rl = p_rl / np.clip(np.sum(p_rl), eps, None)

            # WM policy: use WM contents (probabilities) with lapse toward uniform
            p_wm = (1.0 - lapse) * W_s + lapse * (1.0 / nA)
            p_wm = p_wm / np.clip(np.sum(p_wm), eps, None)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, None)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Generate reward from ground-truth mapping
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update (delta rule)
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform for the queried state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update contingent on outcome
            if r > 0.5:
                # Store correct action deterministically (one-hot)
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.0 * w[s, :] + one_hot
            else:
                # Mild suppression of the chosen (incorrect) action and redistribute
                drop = 0.2
                moved = min(drop, max(0.0, w[s, a] - 1e-6))
                w[s, a] -= moved
                redistribute = moved / (nA - 1) if nA > 1 else 0.0
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += redistribute

            # Ensure valid probability vector
            w[s, :] = np.clip(w[s, :], eps, None)
            w[s, :] = w[s, :] / np.sum(w[s, :])

    return simulated_actions, simulated_rewards
