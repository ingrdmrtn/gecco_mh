def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the RL + WM arbitration by confidence with load-dependent down-weighting
    and anti-Hebbian error (matching cognitive_model2).

    Parameters:
    - stimulus: array of state indices per trial (ints starting at 0)
    - blocks: array of block indices per trial
    - set_sizes: array of block-dependent set sizes per trial
    - correct_answer: array of correct actions per trial (ints in [0, nA-1])
    - parameters: [lr, wm_weight, softmax_beta, lambda_load, wm_decay]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    lr, wm_weight, softmax_beta, lambda_load, wm_decay = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its (assumed fixed within block) correct action
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            # Take the first occurrence's correct action for this state
            mask = (block_states == s)
            if np.any(mask):
                state_correct[s] = int(block_correct[mask][0])
            else:
                state_correct[s] = 0  # fallback (shouldn't happen if states appear)

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load factor used in arbitration down-weighting (baseline set size = 3)
        load_factor = max(0.0, (nS - 3) / 3.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM softmax (very sharp/deterministic)
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Confidence from WM margin (top1 - top2), squashed
            sorted_W = np.sort(W_s)[::-1]
            top1 = sorted_W[0]
            top2 = sorted_W[1] if nA > 1 else 0.0
            margin = np.clip(top1 - top2, 0.0, 1.0)
            conf = 1.0 / (1.0 + np.exp(-10.0 * (margin - 0.1)))

            # Effective WM weight: base * load down-weighting * confidence
            wm_weight_eff = wm_weight * np.clip(1.0 - lambda_load * load_factor, 0.0, 1.0) * conf

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward from correct mapping
            r = 1 if a == state_correct[s] else 0
            simulated_rewards[block_idx[t]] = r

            # RL update (Rescorlaâ€“Wagner)
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform on the active state row
            decay = np.clip(wm_decay, 0.0, 1.0)
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # WM update: strengthen on correct, anti-Hebbian suppress on incorrect
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                eta_pos = 0.7
                w[s, :] = (1.0 - eta_pos) * w[s, :] + eta_pos * one_hot
            else:
                eta_neg = 0.3
                suppress = np.zeros(nA)
                suppress[a] = 1.0
                w[s, :] = w[s, :] * (1.0 - eta_neg * suppress)

                row_sum = np.sum(w[s, :])
                if row_sum > 0:
                    w[s, :] /= row_sum
                else:
                    w[s, :] = w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
