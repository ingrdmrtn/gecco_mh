import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate behavior from the RL+WM model with one-shot WM storage, decay, and
    set-size-dependent precision.

    Parameters
    ----------
    stimulus : array-like of int
        State indices on each trial (0..nS-1 within each block).
    blocks : array-like of int
        Block index per trial; states reset across blocks.
    set_sizes : array-like of int
        Set size (number of distinct states in block) per trial.
    correct_answer : array-like of int
        Correct action (0..nA-1) for each trial.
    parameters : iterable of 6 floats
        lr, wm_weight, softmax_beta, wm_decay, wm_store, wm_capacity_exp

    Returns
    -------
    simulated_actions : np.ndarray of int
    simulated_rewards : np.ndarray of int
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_store, wm_capacity_exp = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # high precision WM before set-size scaling

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    eps = 1e-12
    tr = 0

    # Iterate over blocks in order of appearance
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        # Number of states in this block (consistent with fitting code)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumes states are 0..nS-1)
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # Find first trial in block with this state to read its correct action
            mask_s = np.where(block_states == s)[0]
            if len(mask_s) == 0:
                # If a state index does not occur (shouldn't happen), default to 0
                correct_actions_by_state[s] = 0
            else:
                correct_actions_by_state[s] = int(block_correct[mask_s[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM precision for this block (depends on set size)
        beta_wm_eff = softmax_beta_wm * (3.0 / float(nS)) ** wm_capacity_exp

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            logits_rl = softmax_beta * Q_s
            logits_rl = logits_rl - np.max(logits_rl)  # stability
            pi_rl = np.exp(logits_rl)
            pi_rl = pi_rl / (np.sum(pi_rl) + eps)

            # WM policy
            logits_wm = beta_wm_eff * W_s
            logits_wm = logits_wm - np.max(logits_wm)  # stability
            pi_wm = np.exp(logits_wm)
            pi_wm = pi_wm / (np.sum(pi_wm) + eps)

            # Mixture policy
            p_total = wm_weight * pi_wm + (1.0 - wm_weight) * pi_rl
            p_total = p_total / (np.sum(p_total) + eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == correct_actions_by_state[s] else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # One-shot WM storage on reward
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_store) * w[s, :] + wm_store * one_hot

            # Renormalize WM row to a proper distribution
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum
            else:
                w[s, :] = w_0[s, :].copy()

            tr += 1

    return simulated_actions, simulated_rewards
