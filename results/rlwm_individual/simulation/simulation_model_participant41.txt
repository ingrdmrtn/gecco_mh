def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM with graded WM capacity and decay (simulation).
    - RL: single learning rate with softmax choice.
    - WM: fast delta-rule with decay and capacity-limited weight.
    - Mixture: choice probability is a convex combination of WM and RL policies,
      where the effective WM weight scales with set size via a capacity parameter.

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (assumed constant within block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, k_wm, wm_lr]

    Returns:
    - simulated_actions: array of simulated action choices
    - simulated_rewards: array of simulated binary rewards
    """
    lr, wm_weight, softmax_beta, wm_decay, k_wm, wm_lr = parameters

    softmax_beta = softmax_beta * 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])
            correct_a = int(block_correct[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            logits_rl = softmax_beta * Q_s
            max_rl = np.max(logits_rl)
            p_rl = np.exp(logits_rl - max_rl)
            p_rl /= np.sum(p_rl)

            # WM softmax
            logits_wm = softmax_beta_wm * W_s
            max_wm = np.max(logits_wm)
            p_wm = np.exp(logits_wm - max_wm)
            p_wm /= np.sum(p_wm)

            # Effective WM weight scales with current set size
            nS_current = float(block_set_sizes[t])
            cap_scale = min(1.0, max(0.0, k_wm / max(nS_current, 1.0)))
            wm_weight_eff = np.clip(wm_weight * cap_scale, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform prior
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM delta-rule update on chosen action
            delta_w = r - W_s[a]
            w[s, a] += wm_lr * delta_w

            # Keep WM row a proper probability distribution
            w[s, :] = np.clip(w[s, :], 1e-8, None)
            w[s, :] = w[s, :] / np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
