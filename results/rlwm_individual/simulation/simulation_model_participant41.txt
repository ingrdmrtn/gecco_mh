def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate actions and rewards for the hybrid RL + surprise-gated WM model with
    WM precision boost for small set sizes and choice perseveration.

    Inputs:
    - stimulus: array of states for each trial
    - blocks: array of block indices for each trial
    - set_sizes: array of set size for each trial (constant within a block)
    - correct_answer: array of correct action for each trial
    - parameters: [lr, base_beta, wm_chunk_adv, surprise_sens, wm_decay, choice_bias]

    Returns:
    - simulated_actions: array of sampled actions
    - simulated_rewards: array of binary rewards
    """
    lr, base_beta, wm_chunk_adv, surprise_sens, wm_decay, choice_bias = parameters
    softmax_beta = base_beta * 10.0
    softmax_beta_wm_base = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action using first occurrence in this block
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                state_to_correct[s] = block_correct[idx_s[0]]
            else:
                # Fallback if a state does not appear (shouldn't happen), set arbitrary valid action
                state_to_correct[s] = 0

        # Initialize RL and WM stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Surprise tracker (per state) and perseveration
        last_pe_abs = np.ones(nS)
        last_action = -1

        # Decay parameter
        lam = np.clip(wm_decay, 0.0, 1.0)

        # WM precision boost based on set size
        size_factor = 3.0 / max(3.0, float(nS))
        beta_wm_eff = softmax_beta_wm_base * (1.0 + wm_chunk_adv * (size_factor - 0.5))
        beta_wm_eff = max(1.0, beta_wm_eff)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Choice perseveration bias (applied as additive bias on the last chosen action)
            bias_vec = np.zeros(nA)
            if last_action >= 0 and last_action < nA:
                bias_vec[last_action] = choice_bias

            # RL policy
            logits_rl = softmax_beta * (Q_s + bias_vec)
            # stabilize
            logits_rl -= np.max(logits_rl)
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # WM policy
            logits_wm = beta_wm_eff * (W_s + bias_vec)
            logits_wm -= np.max(logits_wm)
            p_wm = np.exp(logits_wm)
            p_wm /= np.sum(p_wm)

            # Arbitration: WM availability gated by set size, surprise, and WM strength
            wm_strength = np.max(W_s) - (1.0 / nA)
            denom = max(1e-6, 1.0 - 1.0 / nA)
            strength_scale = np.clip(wm_strength / denom, 0.0, 1.0)
            pe_gate = 1.0 / (1.0 + surprise_sens * last_pe_abs[s])
            wm_avail = np.clip(size_factor * pe_gate * strength_scale, 0.0, 1.0)

            p_total = wm_avail * p_wm + (1.0 - wm_avail) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr + t] = a

            # Outcome
            r = 1 if a == state_to_correct[s] else 0
            simulated_rewards[tr + t] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta
            last_pe_abs[s] = abs(delta)

            # WM update: if rewarded, store one-hot; then decay toward baseline
            if r > 0.5:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0
                w[s, :] /= np.sum(w[s, :])

            # Global decay toward baseline
            w = (1.0 - lam) * w + lam * w_0

            # Update perseveration
            last_action = a

        tr += len(block_states)

    return simulated_actions, simulated_rewards
