def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using the RL + one-shot WM model with set-size-dependent decay and lapse.

    Inputs:
    - stimulus: array of state identifiers per trial (assumed 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct actions per trial
    - parameters: tuple/list of (lr, wm_weight, softmax_beta, wm_decay_base, wm_lapse_base)

    Returns:
    - simulated_actions: array of chosen actions per trial (ints in 0..2)
    - simulated_rewards: array of rewards per trial (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay_base, wm_lapse_base = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]
        # Determine set size for this block
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (take first occurrence)
        correct_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            first_occ = np.where(block_states == s)[0]
            if len(first_occ) == 0:
                # If a state does not occur, default to 0 (won't be used)
                correct_by_state[s] = 0
            else:
                correct_by_state[s] = block_correct[first_occ[0]]

        # Initialize RL and WM stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute set-size dependent WM decay and lapse (constant within block)
        # Replicate fitting code logic
        size_factor = (nS - 3) / max(3, nS - 0)
        size_factor = max(0.0, min(1.0, size_factor))
        wm_decay = wm_decay_base + (1.0 - wm_decay_base) * size_factor
        wm_lapse = wm_lapse_base + (1.0 - wm_lapse_base) * size_factor

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = np.exp(softmax_beta * Q_s)
            p_rl /= np.sum(p_rl)

            # WM policy: deterministic softmax over WM row, then lapse to uniform
            p_wm_det = np.exp(softmax_beta_wm * W_s)
            p_wm_det /= np.sum(p_wm_det)



            p_wm = (1.0 - wm_lapse) * p_wm_det + wm_lapse * (1.0 / nA)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome based on correct action for this state
            r = 1 if a == correct_by_state[s] else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for this state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, write one-shot memory for this state
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            tr_global += 1

    return simulated_actions, simulated_rewards
