def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards for the RL+WM mixture with capacity-limited WM and decay.

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size for the block on each trial (can be constant within block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, K, wm_decay]

    Returns:
    - simulated_actions: array of simulated actions per trial
    - simulated_rewards: array of simulated binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, K, wm_decay = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    eps = 1e-12

    trial_idx = 0
    for b in np.unique(blocks):
        block_ix = np.where(blocks == b)[0]
        block_states = stimulus[block_ix]
        block_correct = correct_answer[block_ix]
        block_set_sizes = set_sizes[block_ix]

        # In the fitting code, nS is taken from the first trial's set size in the block
        nS = int(block_set_sizes[0])

        # Build a mapping from state -> correct action within this block using the first occurrence
        # Assumes states are labeled 0..nS-1
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            first_occ = np.where(block_states == s)[0]
            if len(first_occ) > 0:
                state_to_correct[s] = int(block_correct[first_occ[0]])
            else:
                # If a state does not appear (edge case), default to 0
                state_to_correct[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-scaled WM weight (fixed within a block)
        cap_factor = min(1.0, max(0.0, K / max(1, nS)))
        wm_weight_eff = np.clip(wm_weight * cap_factor, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl = prl / (np.sum(prl) + eps)

            # WM softmax (near-deterministic)
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm = pwm / (np.sum(pwm) + eps)

            # Mixture policy
            p_total = wm_weight_eff * pwm + (1.0 - wm_weight_eff) * prl
            p_total = p_total / (np.sum(p_total) + eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[trial_idx] = a

            # Generate reward based on correctness for this state
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[trial_idx] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay towards baseline for current state only
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, boost the chosen action's WM value
            if r > 0.5:
                w[s, a] = w[s, a] + wm_decay * 1.0
                w[s, :] = np.maximum(w[s, :], 0.0)

            trial_idx += 1

    return simulated_actions, simulated_rewards
