def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with capacity scaling, WM decay, and undirected lapses that increase with set size.

    Parameters:
    - stimulus: sequence of stimuli for each trial (array-like of ints)
    - blocks: block index for each trial (array-like of ints)
    - set_sizes: set size on each trial (array-like of ints)
    - correct_answer: correct action on each trial (array-like of ints)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_capacity_K, lapse_beta]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_capacity_K, lapse_beta = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    n_trials = len(stimulus)
    nA = 3

    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    lapse_amp = 1.0 / (1.0 + np.exp(-lapse_beta))
    eps = 1e-12

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = np.array(stimulus)[block_idx]
        block_correct = np.array(correct_answer)[block_idx]
        block_set_sizes = np.array(set_sizes)[block_idx]

        # Map raw state labels in this block to 0..nS-1
        unique_states = np.unique(block_states_raw)
        state_to_idx = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_idx[s] for s in block_states_raw], dtype=int)

        # Determine set size for initializing tables (use provided set size for this block)
        nS = int(block_set_sizes[0])

        # Build per-state correct action map (assumes stable mapping within block)
        correct_map = np.zeros(nS, dtype=int)
        # Default fill with first encountered mapping for each state index
        seen = np.zeros(nS, dtype=bool)
        for t_local, s_idx in enumerate(block_states):
            if not seen[s_idx]:
                correct_map[s_idx] = int(block_correct[t_local])
                seen[s_idx] = True

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t_local, tr in enumerate(block_idx):
            s = int(block_states[t_local])
            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl = p_rl / np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm = p_wm / np.sum(p_wm)

            # Effective WM mixture weight with capacity scaling
            nS_curr = int(block_set_sizes[t_local])
            cap_scale = min(1.0, wm_capacity_K / max(1.0, float(nS_curr)))
            wm_weight_eff = np.clip(wm_weight * cap_scale, 0.0, 1.0)

            # Lapse increases with set size (0 at set size 3; scales up to lapse_amp at 6)
            lapse = lapse_amp * max(0.0, (nS_curr - 3.0) / 3.0)
            lapse = np.clip(lapse, 0.0, 1.0)

            # Mixture and final choice probabilities
            mix_prob = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * mix_prob + lapse * (1.0 / nA)
            p_total = np.clip(p_total, eps, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct mapping
            correct_a = int(correct_map[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform, then overwrite/adjust based on outcome
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r >= 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot
            else:
                w[s, a] = 0.5 * w[s, a] + 0.5 * w_0[s, a]

    return simulated_actions, simulated_rewards
