def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with eligibility traces and set-size interference.

    Inputs:
    - stimulus: sequence of stimuli (state indices) for each trial (array-like, int)
    - blocks: block index for each trial (array-like, int)
    - set_sizes: block-dependent set sizes on each trial (array-like, int)
    - correct_answer: correct action for each trial (array-like, int)
    - parameters: [lr, wm_weight, softmax_beta, trace, interference, enc_gain]

    Returns:
    - simulated_actions: simulated action choices (array, int)
    - simulated_rewards: simulated binary rewards (array, int)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, trace, interference, enc_gain = parameters
    # Apply constraints analogous to fitting code
    wm_weight = float(np.clip(wm_weight, 0.0, 1.0))
    lr = float(np.clip(lr, 0.0, 1.0))
    trace = float(np.clip(trace, 0.0, 1.0))
    interference = max(0.0, float(interference))
    enc_gain = float(np.clip(enc_gain, 0.0, 1.0))

    softmax_beta = float(softmax_beta) * 10.0
    softmax_beta_wm = 50.0

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes, dtype=int)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        # Trials in this block
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Per fitting code, nS is taken from the provided set size for the block
        nS = int(block_set_sizes[0])

        # Build a mapping from state -> correct action using first occurrence
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            mask_s = (block_states == s)
            if np.any(mask_s):
                correct_actions[s] = block_correct[mask_s][0]
            else:
                # If a state doesn't appear (edge case), default to action 0
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        # Eligibility traces
        e = np.zeros((nS, nA))

        # Set-size interference on WM mixture weight
        wm_weight_eff = wm_weight / (1.0 + interference * max(0, nS - 1))
        wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

        for t in range(len(block_states)):
            s = block_states[t]

            # RL policy
            Q_s = q[s, :]
            Q_center = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * Q_center)
            p_rl /= np.sum(p_rl)

            # WM policy
            W_s = w[s, :]
            W_center = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * W_center)
            p_wm /= np.sum(p_wm)

            # Arbitration
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[tr] = a

            # Reward from environment (correct mapping per state)
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # RL update with eligibility trace
            delta = r - Q_s[a]
            e *= trace
            e[s, a] += 1.0
            q += lr * delta * e

            # WM encoding on reward, with normalization/clipping
            if r > 0:
                w[s, :] = (1.0 - enc_gain) * w[s, :]
                w[s, a] += enc_gain

            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] = np.clip(w[s, :], 1e-9, 1.0)
                w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
