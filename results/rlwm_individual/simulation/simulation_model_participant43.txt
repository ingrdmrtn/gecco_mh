def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture simulation with capacity-limited WM and decay toward uniform.

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct action per trial
    - parameters: tuple/list (lr, wm_weight, softmax_beta, K, wm_decay)

    Returns:
    - simulated_actions: array of simulated choices
    - simulated_rewards: array of simulated binary rewards
    """
    import numpy as np

    lr, wm_weight, softmax_beta, K, wm_decay = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM when strong memory is present

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_size = int(set_sizes[block_indices][0])

        nA = 3
        nS = block_set_size  # number of distinct stimuli in the block

        # Build state -> correct action mapping (assumes consistent mapping within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            idx = np.where(block_states == s)[0]
            if len(idx) > 0:
                correct_actions[s] = int(block_correct[idx[0]])
            else:
                # Fallback if a state is not encountered (shouldn't happen in typical datasets)
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight given capacity and set size
        cap_factor = min(1.0, float(K) / float(nS))
        wm_w_eff = np.clip(wm_weight * cap_factor, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Apply WM decay toward uniform on the visited state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Compute RL and WM policies via softmax
            # RL
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(z_rl)
            p_rl /= np.sum(p_rl)

            # WM
            z_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(z_wm)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, None)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update
            if r > 0.5:
                # One-shot storage of correct action
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Eliminate unrewarded action; renormalize or reset to uniform if empty
                w[s, a] = 0.0
                if np.all(w[s, :] <= 1e-12):
                    w[s, :] = w_0[s, :].copy()
                else:
                    w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
