def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + WM with surprise-gated WM encoding and load-modulated forgetting.

    Parameters:
    - stimulus: sequence of stimuli (state indices) for each trial (array)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action on each trial (array)
    - parameters: [lr, wm_weight_base, softmax_beta, wm_forget, pe_sensitivity]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_weight_base, softmax_beta, wm_forget, pe_sensitivity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_indices = np.arange(n_trials)
    out_pos = 0  # running index across trials when filling outputs in block order

    for b in np.unique(blocks):
        block_idx = tr_indices[blocks == b]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Use provided set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])
        load = max(0.0, (nS - 3.0) / 3.0)  # 0 for 3, 1 for 6
        wm_forget_eff = np.clip(wm_forget * (1.0 + load), 0.0, 1.0)

        # Map each state to its correct action (assumed stable within block)
        # States are assumed to be in 0..nS-1 for the block
        state_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t in range(len(block_states)):
            s = int(block_states[t])
            if not seen[s]:
                state_correct[s] = int(block_correct[t])
                seen[s] = True

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM choice probabilities
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Surprise and attention
            surprise = 1.0 - np.max(Q_s)
            attention = 1.0 / (1.0 + np.exp(-pe_sensitivity * surprise))

            wm_weight_eff = np.clip(wm_weight_base * attention, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            correct_a = state_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward-gated encoding vs decay
            if r >= 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                encode_strength = np.clip(attention * wm_forget_eff, 0.0, 1.0)
                w[s, :] = (1.0 - encode_strength) * w[s, :] + encode_strength * target
            else:
                w[s, :] = (1.0 - wm_forget_eff) * w[s, :] + wm_forget_eff * w_0[s, :]

    return simulated_actions, simulated_rewards
