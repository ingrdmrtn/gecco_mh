import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the RL + gated retrieval model with age- and load-dependent gating and noisy retrieval.

    Inputs:
    - stimulus: array of state indices per trial (ints)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (should be constant within each block; values like 3 or 6)
    - correct_answer: array of correct action indices per trial (0..2)
    - age: participant age (float or int)
    - parameters: list/tuple of 5 parameters [alpha, beta, gate_base, retrieve_noise, age_sensitivity]

    Returns:
    - simulated_actions: array of chosen actions (ints)
    - simulated_rewards: array of rewards (0/1)
    """
    alpha, beta, gate_base, retrieve_noise, age_sensitivity = parameters

    # Constrain parameters consistent with fitting code
    alpha = min(max(alpha, 1e-6), 1.0)
    beta = max(beta, 1e-6) * 5.0
    retrieve_noise = min(max(retrieve_noise, 0.0), 1.0)
    age_sensitivity = max(0.0, age_sensitivity)

    # Age flag
    is_older = 1.0 if (np.asarray(age).item() >= 45) else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        idx = np.where(blocks == b)[0]
        b_states = stimulus[idx].astype(int)
        b_correct = correct_answer[idx].astype(int)

        # Block set size (assumed constant within block)
        b_setsize = int(set_sizes[idx][0]) if np.ndim(set_sizes) > 0 else int(set_sizes)
        nA = 3
        nS = b_setsize

        # Map each state id to its correct action (take first occurrence within the block)
        # Assumes state labels within block are compatible with 0..nS-1
        state_to_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t in range(len(b_states)):
            s = b_states[t]
            if 0 <= s < nS and not seen[s]:
                state_to_correct[s] = b_correct[t]
                seen[s] = True

        # Initialize RL and retrieval memories
        Q = np.zeros((nS, nA))                 # RL values
        M = np.ones((nS, nA)) / nA             # retrieval cache starts uniform

        # Load penalty as in fitting code
        load_penalty = 0.0 if b_setsize <= 3 else 1.0

        for t in range(len(b_states)):
            s = b_states[t]
            # Safety clamp for state index
            if s < 0 or s >= nS:
                # if stimulus indexing is out of expected range, wrap or clamp
                s = np.clip(s, 0, nS - 1)

            # RL policy
            q_s = Q[s, :]
            logits_rl = beta * (q_s - np.max(q_s))
            exp_rl = np.exp(logits_rl)
            p_rl = exp_rl / np.sum(exp_rl)

            # Retrieval policy with noise mixed with uniform
            m_s = M[s, :]
            logits_ret = beta * (m_s - np.max(m_s))
            exp_ret = np.exp(logits_ret)
            p_ret_clean = exp_ret / np.sum(exp_ret)

            noise = min(1.0, retrieve_noise * (1.0 + 0.5 * age_sensitivity * is_older))
            p_ret = (1.0 - noise) * p_ret_clean + noise * (1.0 / nA)

            # Gate between retrieval and RL, depends on load and age
            gate_logit = gate_base - (0.75 * load_penalty) - (age_sensitivity * is_older)
            gate = 1.0 / (1.0 + np.exp(-gate_logit))
            gate = min(max(gate, 0.0), 1.0)

            # Mixture policy
            p = gate * p_ret + (1.0 - gate) * p_rl
            p = np.maximum(p, 1e-12)
            p = p / np.sum(p)

            # Sample action
            a = np.random.choice(nA, p=p)
            simulated_actions[idx[t]] = a

            # Simulate reward from correct answer
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[idx[t]] = r

            # RL update
            pe = r - Q[s, a]
            Q[s, a] += alpha * pe

            # Retrieval dynamics: relax toward uniform each trial, stronger with load/age
            relax = 0.10 + 0.10 * load_penalty + 0.10 * is_older
            relax = min(max(relax, 0.0), 1.0)
            M[s, :] = (1.0 - relax) * M[s, :] + relax * (np.ones(nA) / nA)

            # Retrieval imprinting on rewarded trials
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                M[s, :] = 0.2 * M[s, :] + 0.8 * target

            tr += 1

    return simulated_actions, simulated_rewards
