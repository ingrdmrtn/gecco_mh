import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards from the RL+WM mixture model with capacity scaling and decay.

    Parameters:
    - stimulus: array of state indices per trial (assumed 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array giving the set size for each trial (constant within block)
    - correct_answer: array of correct action per trial (0..nA-1)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, capacity_k]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    lr, wm_weight, softmax_beta, wm_decay, capacity_k = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Follow fitting code: set nS from provided set_sizes (assumed constant in block)
        nS = int(block_set_sizes[0])

        # Build per-state correct action map using the first occurrence of each state
        # Assumes states are labeled 0..nS-1 within the block
        state_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of state s in this block
            where_s = np.where(block_states == s)[0]
            if len(where_s) > 0:
                state_correct[s] = block_correct[where_s[0]]
            else:
                # If state not encountered (shouldn't happen), default to 0
                state_correct[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity scaling for WM weight (fixed within block)
        cap_scale = min(1.0, capacity_k / max(1.0, float(nS)))
        wm_w_eff_block = wm_weight * cap_scale

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Global WM decay toward uniform each trial
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture policy with capacity-scaled WM weight
            p_total = wm_w_eff_block * p_wm + (1.0 - wm_w_eff_block) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[block_idx[t]] = a

            # Generate reward based on correct action for this state
            correct_a = state_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: store last rewarded response one-shot
            if r > 0.5:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
