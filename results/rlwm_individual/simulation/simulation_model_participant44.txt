def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + surprise-gated WM arbitration with age-adjusted learning speed (simulation).

    Parameters:
    - stimulus: state index for each trial (array of ints)
    - blocks: block index for each trial (array of ints)
    - set_sizes: block-dependent set size for each trial (array of ints; typically 3 or 6)
    - correct_answer: correct action for each trial (array of ints in [0, 2])
    - age: participant's age (float)
    - parameters: [lr, wm_weight_base, softmax_beta, surprise_sensitivity, age_rl_slowdown]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints: 0/1)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, surprise_sensitivity, age_rl_slowdown = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    older = 1.0 if age >= 45 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx].astype(int)

        nA = 3
        # Use provided set size per block (assumed consistent within block)
        nS = int(block_set_sizes[0])

        # Build mapping from state -> correct action using first occurrence per state
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first trial in block with this state
            inds = np.where(block_states == s)[0]
            if len(inds) == 0:
                # If state not present (shouldn't happen), default to 0
                correct_actions[s] = 0
            else:
                correct_actions[s] = int(block_correct[inds[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track surprise per state to gate WM at decision time (initialized to 0)
        last_surprise = np.zeros(nS, dtype=float)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Softmax policies
            def softmax(x, beta):
                z = x - np.max(x)
                p = np.exp(beta * z)
                return p / np.sum(p)

            p_rl = softmax(Q_s, softmax_beta)
            p_wm = softmax(W_s, softmax_beta_wm)

            # Arbitration weight
            load_scale = 3.0 / float(nS)  # 1.0 for set size 3; 0.5 for set size 6
            wm_mix = wm_weight_base * load_scale * (1.0 - 0.4 * older) * (1.0 + surprise_sensitivity * last_surprise[s])
            wm_mix = np.clip(wm_mix, 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # RL update
            eff_lr = lr * (1.0 - age_rl_slowdown * older)
            delta = r - Q_s[a]
            q[s, a] += eff_lr * delta

            # WM update
            if r > 0.5:
                # Rewarded: move toward a peaked distribution on chosen action
                surprise_now = abs(delta)  # use pre-update Q_s for delta
                alpha_w = np.clip(0.5 + 0.5 * surprise_now, 0.0, 1.0)
                peaked = (1.0 / nA) * np.ones(nA)
                peaked[a] = 1.0
                peaked /= np.sum(peaked)
                w[s, :] = (1.0 - alpha_w) * W_s + alpha_w * peaked
                last_surprise[s] = surprise_now
            else:
                # Unrewarded: decay toward uniform; decay depends on set size
                decay = 0.3 + 0.4 * (nS - 3) / 3.0  # 0.3 for set size 3; 0.7 for set size 6
                w[s, :] = (1.0 - decay) * W_s + decay * w_0[s, :]
                last_surprise[s] = abs(delta)

            tr += 1

    return simulated_actions, simulated_rewards
