def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + one-shot WM with set-size-dependent mixture and decay (simulation)

    Parameters:
    - stimulus: sequence of stimuli for each trial (array-like)
    - blocks: block index for each trial (array-like)
    - set_sizes: set size for each trial (array-like)
    - correct_answer: correct action for each trial (array-like)
    - parameters: list-like [lr, wm_weight_small, wm_weight_large, softmax_beta, wm_decay, wm_store]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    import numpy as np

    lr, wm_weight_small, wm_weight_large, softmax_beta, wm_decay, wm_store = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    stimulus = np.asarray(stimulus)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes)
    correct_answer = np.asarray(correct_answer)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    nA = 3

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_orig = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        # Remap states in this block to consecutive indices [0..nS-1]
        unique_states, inv_states = np.unique(block_states_orig, return_inverse=True)
        block_states = inv_states
        nS_block = len(unique_states)

        # Choose WM mixture weight based on provided set size (small vs large)
        nS_param = int(block_set_sizes[0])
        wm_weight_block = wm_weight_small if nS_param <= 3 else wm_weight_large

        # Build per-state correct action mapping using first occurrence
        correct_actions = np.zeros(nS_block, dtype=int)
        for s_idx in range(nS_block):
            first_occurrence = np.where(block_states == s_idx)[0][0]
            correct_actions[s_idx] = int(block_correct[first_occurrence])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS_block, nA))
        w = (1.0 / nA) * np.ones((nS_block, nA))
        w_0 = (1.0 / nA) * np.ones((nS_block, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            q_centered = softmax_beta * Q_s
            q_centered -= np.max(q_centered)
            p_rl = np.exp(q_centered)
            p_rl /= np.sum(p_rl)

            w_centered = softmax_beta_wm * W_s
            w_centered -= np.max(w_centered)
            p_wm = np.exp(w_centered)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Simulate reward based on correct action
            correct_a = correct_actions[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM passive decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM one-shot storage on reward
            if r > 0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - wm_store) * w[s, :] + wm_store * onehot

            tr_global += 1

    return simulated_actions, simulated_rewards
