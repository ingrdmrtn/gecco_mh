def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using:
    RL + Uncertainty-gated WM with load-sensitive arbitration.

    Inputs
    - stimulus: array of state identifiers per trial
    - blocks: array of block ids per trial
    - set_sizes: array of set sizes per trial (block-constant; used to get nS per block)
    - correct_answer: array of correct actions per trial (0..nA-1)
    - parameters: [lr, wm_weight, softmax_beta, kappa_meta, lambda_load]

    Outputs
    - simulated_actions: array of chosen actions (0..nA-1)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, kappa_meta, lambda_load = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # WM is near-deterministic

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    eps = 1e-12
    base_logit = np.log((wm_weight + eps) / (1.0 - wm_weight + eps))

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states_orig = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Map possibly non-consecutive state ids to 0..nS-1
        unique_states = np.unique(block_states_orig)
        nS = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else len(unique_states)
        # Ensure nS matches the number of unique states found; fall back to unique count if mismatch
        if nS != len(unique_states):
            nS = len(unique_states)
        state_to_local = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_local[s] for s in block_states_orig], dtype=int)

        # Per-state correct action (use first occurrence)
        correct_actions_per_state = np.zeros(nS, dtype=int)
        for s_orig, s_local in state_to_local.items():
            first_idx = np.where(block_states_orig == s_orig)[0][0]
            correct_actions_per_state[s_local] = int(block_correct[first_idx])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load-dependent WM decay (as in fitting code)
        decay_wm = np.clip(0.025 + 0.075 * lambda_load * max(0, nS - 3) / 3.0, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies (stable)
            # RL
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Arbitration based on confidence and load
            conf_rl = float(np.max(Q_s) - np.min(Q_s))
            conf_wm = float(np.max(W_s) - np.min(W_s))
            wm_signal = kappa_meta * (conf_wm - conf_rl) - lambda_load * max(0, nS - 3)

            wm_mix = 1.0 / (1.0 + np.exp(-(base_logit + wm_signal)))
            wm_mix = np.clip(wm_mix, 0.0, 1.0)

            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Generate reward from correct mapping
            correct_a = int(correct_actions_per_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w = (1.0 - decay_wm) * w + decay_wm * w_0

            # WM update contingent on outcome
            if r > 0.5:
                # Move row toward one-hot on chosen action
                w[s, :] = (1.0 - lr) * w[s, :]
                w[s, a] += lr
            else:
                # Soften erroneous association toward uniform on chosen action
                w[s, a] = (1.0 - lr) * w[s, a] + lr * (1.0 / nA)

            # Renormalize WM row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

            tr += 1

    return simulated_actions, simulated_rewards
