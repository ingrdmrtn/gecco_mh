def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + capacity-limited WM mixture with age-adjusted capacity and decay (simulation).
    
    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block ids per trial
    - set_sizes: array of set size per trial (constant within block)
    - correct_answer: array of correct action per trial (0,1,2), constant per state within block
    - age: participant age (float)
    
    Returns:
    - simulated_actions: array of chosen actions (0,1,2)
    - simulated_rewards: array of rewards (0 or 1)
    """

    alpha, beta, K_base, k_age, wm_decay, lapse = parameters

    lapse = np.clip(lapse, 0.0, 1.0)
    alpha = np.clip(alpha, 0.0, 1.0)
    wm_decay = np.clip(wm_decay, 0.0, 1.0)

    age_val = float(age)
    older = 1.0 if age_val >= 45.0 else 0.0

    n_trials = len(stimulus)
    nA = 3
    eps = 1e-12

    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    unique_blocks = np.unique(blocks)

    for b in unique_blocks:
        idx = np.where(blocks == b)[0]
        block_states = np.array(stimulus)[idx]
        block_correct = np.array(correct_answer)[idx]
        block_set_sizes = np.array(set_sizes)[idx]

        # Determine set size for the block
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block
        # Assumes consistent correct action per state in a block
        state_to_correct = {}
        for t in range(len(block_states)):
            s = int(block_states[t])
            if s not in state_to_correct:
                state_to_correct[s] = int(block_correct[t])

        # Initialize RL and WM
        Q = np.zeros((nS, nA))  # RL Q-values
        W = np.zeros((nS, nA))  # WM associative strengths

        # WM retrieval probability for this block (capacity-limited)
        K_eff = max(0.0, K_base - k_age * older)
        p_retrieve_block = min(1.0, K_eff / max(1.0, float(nS)))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            q_s = Q[s, :]
            prefs_rl = beta * (q_s - np.max(q_s))
            exp_rl = np.exp(prefs_rl)
            pi_rl = exp_rl / np.sum(exp_rl)

            # WM policy
            w_s = W[s, :]
            if np.sum(w_s) <= 0:
                pi_wm = np.ones(nA) / nA
            else:
                pi_wm = w_s / np.sum(w_s)

            # Arbitration and lapse
            w_wm = p_retrieve_block
            pi_mix = w_wm * pi_wm + (1.0 - w_wm) * pi_rl
            pi = (1.0 - lapse) * pi_mix + lapse * (1.0 / nA)
            pi = np.clip(pi, eps, 1.0)
            pi /= np.sum(pi)

            # Sample action
            a = int(np.random.choice(nA, p=pi))
            simulated_actions[idx[t]] = a

            # Outcome based on correct action mapping
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[idx[t]] = r

            # RL update
            delta = r - Q[s, a]
            Q[s, a] += alpha * delta

            # WM update with decay logic mirroring fitting code
            if r >= 0.5:
                # Decay other associations for this state, then store the correct pair strongly
                W[s, :] *= (1.0 - wm_decay)
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                W[s, :] = np.maximum(W[s, :], one_hot)
            else:
                # Global decay when feedback is incorrect
                W *= (1.0 - wm_decay)

            tr += 1

    return simulated_actions, simulated_rewards
