import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards from the RL+WM mixture model with WM learning, decay,
    and capacity-based set-size modulation.

    Inputs:
    - stimulus: array of state identities per trial (0..nS-1 within a block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct actions per trial (0..2)
    - parameters: [lr, wm_weight, softmax_beta, lr_wm, wm_decay, kappa]

    Returns:
    - simulated_actions: array of simulated choices per trial
    - simulated_rewards: array of simulated binary rewards per trial
    """

    lr, wm_weight, softmax_beta, lr_wm, wm_decay, kappa = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    eps = 1e-12

    tr = 0
    unique_blocks = np.unique(blocks)
    for b in unique_blocks:
        block_idx = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus)[block_idx]
        block_set_sizes = np.asarray(set_sizes)[block_idx]
        block_correct = np.asarray(correct_answer)[block_idx]

        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumes consistent within block)
        correct_actions_per_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            s_mask = (block_states == s)
            # fallback in case of missing state trials (shouldn't happen)
            if not np.any(s_mask):
                correct_actions_per_state[s] = 0
            else:
                first_idx_in_block = np.where(s_mask)[0][0]
                correct_actions_per_state[s] = int(block_correct[first_idx_in_block])

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight modulated by set size
        wm_wt_eff = wm_weight * (3.0 / float(nS))**kappa
        wm_wt_eff = float(np.clip(wm_wt_eff, 0.0, 1.0))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Softmax policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # Numerically stable softmax
            q_logits = softmax_beta * (Q_s - np.max(Q_s))
            w_logits = softmax_beta_wm * (W_s - np.max(W_s))

            p_rl = np.exp(q_logits)
            p_rl /= np.maximum(np.sum(p_rl), eps)

            p_wm = np.exp(w_logits)
            p_wm /= np.maximum(np.sum(p_wm), eps)

            p_total = wm_wt_eff * p_wm + (1.0 - wm_wt_eff) * p_rl
            p_total = np.maximum(p_total, 0)
            p_total /= np.maximum(np.sum(p_total), eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Generate reward based on correct action for this state
            r = 1 if a == correct_actions_per_state[s] else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay (on current state) then learning on chosen action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            w[s, a] += lr_wm * (r - w[s, a])

            tr += 1

    return simulated_actions, simulated_rewards
