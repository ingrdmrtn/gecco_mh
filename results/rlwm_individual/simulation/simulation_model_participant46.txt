def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM with slot-limited capacity and decay (mixture policy) â€” simulation.

    Inputs:
    - stimulus: array of stimulus (state) IDs per trial
    - blocks: array of block IDs per trial
    - set_sizes: array of set size for the block on each trial
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_learn, wm_capacity]

    Returns:
    - simulated_actions: int array of chosen actions per trial
    - simulated_rewards: int array of rewards (0/1) per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_learn, wm_capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM
    n_trials = len(stimulus)
    nA = 3
    eps = 1e-12

    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)


    trial_counter = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_size_declared = int(set_sizes[block_idx][0])

        # Remap states within block to 0..nS-1 to be safe
        unique_states = np.unique(block_states_raw)
        nS_unique = len(unique_states)
        state_to_local = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_local[s] for s in block_states_raw], dtype=int)

        # Build per-state correct action map (local indexing)
        correct_actions_local = np.zeros(max(nS_unique, block_set_size_declared), dtype=int)
        for s_raw in unique_states:
            local = state_to_local[s_raw]
            # find first occurrence of this state in the block to get its correct action
            first_idx = np.where(block_states_raw == s_raw)[0][0]
            correct_actions_local[local] = block_correct[first_idx]

        # Initialize RL and WM values (size by declared set size, as in fitting)
        nS = max(block_set_size_declared, nS_unique)
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity scaling of WM mixture weight (block-level)
        cap_scale = min(1.0, wm_capacity / max(1.0, float(block_set_size_declared)))
        wm_weight_block = np.clip(wm_weight * cap_scale, 0.0, 1.0)

        # Iterate trials within the block in order
        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]


            # RL and WM policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)


            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = np.clip(p_total, 0.0, 1.0)
            p_total = p_total / max(p_total.sum(), eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward based on correct action for this state
            correct_a = int(correct_actions_local[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform prior for the current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM one-shot learning if rewarded
            if r > 0:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_learn) * w[s, :] + wm_learn * target

            trial_counter += 1

    return simulated_actions, simulated_rewards
