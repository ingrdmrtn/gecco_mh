def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from a Gated WM-RL hybrid with entropy-based WM reliance,
    modulated by age and set size. Mirrors the logic of cognitive_model3.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct action per trial (0..2)
    - age: participant's age (float or 1-element array)

    Returns:
    - simulated_actions: array of simulated action choices per trial (0..2)
    - simulated_rewards: array of simulated binary rewards per trial (0/1)
    """


    alpha, beta, wm_gate_base, wm_reliance_gain, noise = parameters

    # Age handling
    a = age[0] if hasattr(age, "__len__") else age
    is_older = 1.0 if a >= 45 else 0.0

    # WM noise effects and WM softmax precision
    noise_eff = max(1e-3, noise * (1.2 * is_older + 0.8 * (1.0 - is_older)))
    beta_wm = 5.0 / noise_eff  # higher => sharper WM policy

    eps = 1e-12
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        # Trials in this block
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_setsizes = set_sizes[block_idx]
        nS = int(block_setsizes[0])  # set size constant within block

        # RL values
        Q = np.zeros((nS, nA))

        # WM distribution over actions per state (probabilities)
        WM = (1.0 / nA) * np.ones((nS, nA))

        # Confidence tracker (not directly used in policy; kept for parity with fitting code)
        R = np.zeros(nS)

        # Determine correct action per state (assumed stable within block)
        correct_per_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first occurrence's correct action for this state
            idx_s = np.where(block_states == s)[0]
            if len(idx_s) > 0:
                correct_per_state[s] = block_correct[idx_s[0]]
            else:
                # fallback to a default if state doesn't appear (shouldn't happen)
                correct_per_state[s] = 0

        # Set-size term used in the gate
        size_term = -0.6 * (nS - 3)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy
            logits_rl = beta * (Q[s, :] - np.max(Q[s, :]))
            p_rl = np.exp(logits_rl)
            p_rl = p_rl / np.sum(p_rl)

            # WM policy: center WM around uniform before softmax, as in fitting
            wm_centered = WM[s, :] - (1.0 / nA)
            logits_wm = beta_wm * (wm_centered - np.max(wm_centered))
            p_wm = np.exp(logits_wm)
            p_wm = p_wm / np.sum(p_wm)

            # Entropy of WM state distribution (normalized)
            wm_dist = WM[s, :]
            ent = -np.sum(wm_dist * np.log(wm_dist + eps)) / np.log(nA)

            # Gate between WM and RL
            age_term = -0.6 if is_older else 0.6
            gate_logit = wm_gate_base + wm_reliance_gain * (1.0 - ent) + age_term + size_term
            gate = 1.0 / (1.0 + np.exp(-gate_logit))
            gate = min(max(gate, 0.0), 1.0)

            # Mixture policy and action sampling
            p = gate * p_wm + (1.0 - gate) * p_rl
            p = np.maximum(p, eps)
            p = p / np.sum(p)
            a_ch = int(np.random.choice(nA, p=p))

            # Reward given the correct action for this state
            r = 1 if a_ch == int(correct_per_state[s]) else 0

            # Store simulation outputs
            simulated_actions[block_idx[t]] = a_ch
            simulated_rewards[block_idx[t]] = r

            # RL update
            pe = r - Q[s, a_ch]
            Q[s, a_ch] += alpha * pe

            # WM update with one-shot strengthening on reward, decay otherwise
            if r > 0.5:
                WM[s, :] = eps
                WM[s, a_ch] = 1.0 - (nA - 1) * eps
                R[s] = min(1.0, R[s] + 0.5)
            else:
                R[s] = max(0.0, R[s] - 0.2)
                WM[s, :] = 0.9 * WM[s, :] + 0.1 * (1.0 / nA)

            # WM noise-induced drift toward uniform
            WM[s, :] = (1.0 - 0.1 * noise_eff) * WM[s, :] + (0.1 * noise_eff) * (1.0 / nA)
            WM[s, :] = np.maximum(WM[s, :], eps)
            WM[s, :] = WM[s, :] / np.sum(WM[s, :])

    return simulated_actions, simulated_rewards
