def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards using Model 1: Dual-rate RL + WM mixture with set-sizeâ€“specific WM weight and WM leak.

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (used to determine WM weight per block)
    - correct_answer: array of correct action per trial
    - parameters: [lr_pos, lr_neg, softmax_beta, wm_w_small, wm_w_large, wm_decay]

    Returns:
    - simulated_actions: array of simulated actions per trial
    - simulated_rewards: array of simulated rewards (0/1) per trial
    """
    lr_pos, lr_neg, softmax_beta, wm_w_small, wm_w_large, wm_decay = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices]
        block_correct_actions = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size for the block (as in fitting code)
        nS = int(block_set_sizes[0])

        # Build mapping from state to correct action
        correct_actions = []
        for st in range(nS):
            mask_st = (block_states == st)
            # Assume each state appears at least once in the block
            correct_actions.append(block_correct_actions[mask_st][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set WM mixture weight based on set size
        if nS <= 3:
            wm_weight_eff = np.clip(wm_w_small, 0.0, 1.0)
        elif nS >= 6:
            wm_weight_eff = np.clip(wm_w_large, 0.0, 1.0)
        else:
            alpha = (nS - 3) / max(1.0, (6 - 3))
            wm_weight_eff = np.clip((1 - alpha) * wm_w_small + alpha * wm_w_large, 0.0, 1.0)

        store_strength = np.clip(2.0 * max(0.0, wm_decay), 0.0, 1.0)

        # Iterate through trials in block
        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM policy (more deterministic)
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action
            correct_action = int(correct_actions[s])
            r = 1 if a == correct_action else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            lr = lr_pos if delta >= 0 else lr_neg
            q[s, a] += lr * delta

            # WM decay toward uniform for this state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on reward
            if r > 0:
                w[s, :] = (1.0 - store_strength) * w[s, :]
                w[s, a] += store_strength
                # Normalize
                w[s, :] = np.clip(w[s, :], 1e-12, None)
                w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
