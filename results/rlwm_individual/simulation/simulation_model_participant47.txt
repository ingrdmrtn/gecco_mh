def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with one-shot WM encoding and decay, and set-size scaling of WM weight (simulation).

    Inputs:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (used to determine nS within each block)
    - correct_answer: array of correct action (0..2) per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_size_sensitivity]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_size_sensitivity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use provided set size for this block (as in fitting code)
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM mixture weight scales with set size
        wm_eff = wm_weight * (3.0 / float(nS)) ** wm_size_sensitivity
        wm_eff = np.clip(wm_eff, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            rl_logits = softmax_beta * Q_s
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM softmax (deterministic)
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward from correct answer on this trial
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update
            if r >= 1:
                # small stabilization then overwrite with one-hot for chosen action
                w[s, :] = (1.0 - wm_decay) * w[s, :]
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # decay toward uniform for this state
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
