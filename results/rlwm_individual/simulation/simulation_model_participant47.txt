def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate choices and rewards under RL + WM mixture with capacity-scaling and WM decay.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within block)
    - blocks: array of block index per trial
    - set_sizes: array with block set size (nS) per trial
    - correct_answer: array of correct action (0..2) per trial
    - parameters: [lr, wm_weight, softmax_beta, phi, eta_wm, K]

    Returns:
    - simulated_actions: array of simulated actions per trial (0..2)
    - simulated_rewards: array of simulated rewards per trial (0/1)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, phi, eta_wm, K = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    # Iterate blocks in order of appearance
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)

        nA = 3
        # Use provided set size for the block
        nS = int(set_sizes[block_idx[0]])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity scaling of WM weight
        cap_scale = min(1.0, max(0.0, K / float(nS)))
        wm_weight_eff = wm_weight_base * cap_scale
        wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute RL and WM policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax for RL
            q_exp = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl = q_exp / np.sum(q_exp)

            # Softmax for WM (very sharp)
            w_exp = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm = w_exp / np.sum(w_exp)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr + t] = a

            # Outcome
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[tr + t] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform baseline
            w[s, :] = (1.0 - phi) * w[s, :] + phi * w_0[s, :]

            # WM encoding on rewarded trials
            if r > 0:
                # Move row toward one-hot on chosen action with strength eta_wm
                w[s, :] = (1.0 - eta_wm) * w[s, :]
                w[s, a] += eta_wm

        tr += len(block_states)

    return simulated_actions, simulated_rewards
