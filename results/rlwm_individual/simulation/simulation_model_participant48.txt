def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + capacity-limited Working Memory (WM) mixture model simulation.

    Parameters:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array giving the set size for the block of each trial (constant within block)
    - correct_answer: array of correct actions per trial (0..nA-1)
    - parameters: [lr, wm_weight, softmax_beta, wm_lr, wm_decay, capacity_k]

    Returns:
    - simulated_actions: array of chosen actions per trial (0..nA-1)
    - simulated_rewards: array of rewards per trial (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_lr, wm_decay, capacity_k = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM
    nA = 3

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)


    # Iterate blocks
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        # Determine set size for this block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Initialize value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity-limited WM weight for this block
        wm_gate = 1.0 / (1.0 + np.exp(capacity_k * (nS - 3)))
        wm_weight_eff = wm_weight * wm_gate

        # Simulate each trial in block
        for local_t, tr in enumerate(block_idx):
            s = int(block_states[local_t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward from correct answer on this trial
            correct_a = int(block_correct[local_t])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward baseline, then WM update on chosen action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            w[s, a] += wm_lr * (r - w[s, a])

    return simulated_actions, simulated_rewards
