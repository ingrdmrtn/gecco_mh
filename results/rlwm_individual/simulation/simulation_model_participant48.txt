def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards under the RL+WM mixture with capacity limits,
    age-dependent WM weight, perseveration bias, and lapse, matching cognitive_model1 logic.

    Inputs:
    - stimulus: state index at each trial (array-like; can be arbitrary labels per block)
    - blocks: block index for each trial (array-like)
    - set_sizes: set size for each trial/block (array-like; typically constant within block)
    - correct_answer: correct action on each trial (array-like; assumed consistent per state within a block)
    - age: participant age (scalar)

    Returns:
    - simulated_actions: array of simulated action choices (ints in [0..2])
    - simulated_rewards: array of simulated binary rewards (0/1)
    """

    alpha, beta, wm_weight, wm_capacity, perseveration, lapse = parameters

    nA = 3
    age_val = age
    is_older = 1.0 if age_val >= 45 else 0.0
    wm_weight_eff_scale = 1.0 - 0.3 * is_older
    pers_eff_scale = 1.0 + 0.3 * is_older
    lapse = min(max(lapse, 0.0), 0.2)

    stimulus = np.asarray(stimulus)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes)
    correct_answer = np.asarray(correct_answer)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Go block by block (preserving trial order)
    unique_blocks = np.unique(blocks)
    for b in unique_blocks:
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_set_sizes = set_sizes[block_idx]
        block_correct = correct_answer[block_idx]

        # Map block-specific states to 0..nS-1 (robust to arbitrary labels)
        unique_states = np.unique(block_states_raw)
        state_to_local = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_local[s] for s in block_states_raw], dtype=int)

        # Determine set size as provided (consistent with fitting code)
        # If varying within block, use the current trial's value for capacity scaling.
        nS = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else len(unique_states)

        # Build per-state correct action (assume stable within block)
        per_state_correct = np.full(len(unique_states), -1, dtype=int)
        for i, s_local in enumerate(block_states):
            if per_state_correct[s_local] < 0:
                per_state_correct[s_local] = int(block_correct[i])
        # Fallback if any state missing: set to mode within that state
        for s_local in range(len(unique_states)):
            if per_state_correct[s_local] < 0:
                candidates = block_correct[block_states == s_local]
                per_state_correct[s_local] = int(np.bincount(candidates.astype(int), minlength=nA).argmax())

        # Initialize RL and WM
        Q = np.ones((nS, nA)) / nA  # Q defined by provided set size nS
        WM_store = -1 * np.ones(nS, dtype=int)
        last_action = -1

        # Step through trials in this block
        for t_in_block, tr in enumerate(block_idx):
            s = int(block_states[t_in_block])
            # Guard against states beyond provided set size
            if s < 0 or s >= nS:
                # Skip invalid state; choose uniformly and no learning
                a = np.random.randint(nA)
                simulated_actions[tr] = a
                simulated_rewards[tr] = 0
                last_action = a
                continue

            # RL policy with perseveration bias
            Q_s = Q[s].copy()
            bias = np.zeros(nA)
            if 0 <= last_action < nA:
                bias[last_action] = perseveration * pers_eff_scale
            logits = beta * Q_s + bias
            logits -= np.max(logits)  # numerical stability
            exp_logits = np.exp(logits)
            p_rl = exp_logits / np.sum(exp_logits)

            # WM policy
            this_set_size = int(block_set_sizes[t_in_block]) if t_in_block < len(block_set_sizes) else nS
            cap_ratio = min(1.0, max(0.0, wm_capacity / max(1.0, float(this_set_size))))
            wm_w = wm_weight * wm_weight_eff_scale * cap_ratio

            p_wm = np.ones(nA) / nA
            if 0 <= WM_store[s] < nA:
                stored_a = int(WM_store[s])
                high = 0.9
                low = (1.0 - high) / (nA - 1)
                p_wm = low * np.ones(nA)
                p_wm[stored_a] = high

            # Mixture and lapse
            p_mix = wm_w * p_wm + (1.0 - wm_w) * p_rl
            p_final = (1.0 - lapse) * p_mix + lapse * (np.ones(nA) / nA)
            p_final = np.clip(p_final, 1e-12, 1.0)
            p_final = p_final / np.sum(p_final)

            # Sample action
            a = int(np.random.choice(nA, p=p_final))
            simulated_actions[tr] = a

            # Outcome based on correct action for this state
            correct_a = int(per_state_correct[s]) if 0 <= per_state_correct[s] < nA else 0
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            pe = float(r) - Q[s, a]
            Q[s, a] += alpha * pe

            # WM update/store
            if r >= 0.5:
                WM_store[s] = a
            else:
                if WM_store[s] == a:
                    WM_store[s] = -1

            # Update perseveration trace
            last_action = a

    return simulated_actions, simulated_rewards
