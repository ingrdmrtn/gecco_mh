def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + capacity-limited working memory (WM) mixture model simulation.

    Inputs:
    - stimulus: array of states per trial (ints from 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within a block)
    - correct_answer: array of correct actions per trial (ints 0..2)
    - parameters: [lr, wm_weight, softmax_beta, wm_capacity, wm_decay, wm_load_sensitivity]

    Returns:
    - simulated_actions: array of sampled actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_capacity, wm_decay, wm_load_sensitivity = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3

    # Process each block independently
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        # Preserve original temporal order within the block
        # (assuming block_indices are already in order; if not, sort by index)
        block_indices = np.array(sorted(block_indices))

        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)

        # Determine set size for this block
        nS = int(set_sizes[block_indices][0])

        # Build a mapping from state -> correct action for this block
        # Assumes each state consistently maps to the same correct action within the block
        unique_states = np.unique(block_states)
        # Ensure nS equals the number of unique states if needed
        # but initialize according to declared nS to match fitting code
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute block-level WM mixture weight with capacity/load scaling
        cap_ratio = min(1.0, max(1e-8, wm_capacity) / float(nS))
        wm_weight_block = wm_weight * (cap_ratio ** max(0.0, wm_load_sensitivity))

        # For retrieving correct action per state
        # Use first occurrence of each state in the block to define its correct action
        state_to_correct = {}
        for s in unique_states:
            first_idx = np.where(block_states == s)[0][0]
            state_to_correct[int(s)] = int(block_correct[first_idx])

        # Iterate trials in the block
        for idx in block_indices:
            s = int(stimulus[idx])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl /= np.sum(prl)

            # WM policy
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm /= np.sum(pwm)

            # Mixture policy
            p_total = wm_weight_block * pwm + (1.0 - wm_weight_block) * prl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / p_total.sum()

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[idx] = a

            # Deliver reward
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[idx] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update (state-local decay toward uniform; reinforce correct action when rewarded)
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * one_hot
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

    return simulated_actions, simulated_rewards
