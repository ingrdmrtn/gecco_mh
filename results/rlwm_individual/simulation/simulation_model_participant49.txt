import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM mixture with WM lapse under high set size and WM decay.

    Parameters:
    - stimulus: array of state indices per trial
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within block)
    - correct_answer: array of correct action per trial
    - parameters: [lr, wm_weight, softmax_beta, wm_alpha, wm_decay, wm_lapse6]

    Returns:
    - simulated_actions: array of simulated action choices
    - simulated_rewards: array of simulated binary rewards
    """
    lr, wm_weight, softmax_beta, wm_alpha, wm_decay, wm_lapse6 = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    nA = 3
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nS = int(block_set_sizes[0])

        # Map each state to its correct action within this block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            first_occ = np.where(block_states == s)[0]
            if len(first_occ) == 0:
                # If a state never occurs (shouldn't happen), default to 0
                correct_actions[s] = 0
            else:
                correct_actions[s] = int(block_correct[first_occ[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w0 = (1.0 / nA) * np.ones((nS, nA))

        wm_lapse = 0.0 if nS <= 3 else float(np.clip(wm_lapse6, 0.0, 1.0))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax policy
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl = prl / np.clip(prl.sum(), eps, None)

            # WM softmax policy + lapse inside WM
            pwm_soft = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm_soft = pwm_soft / np.clip(pwm_soft.sum(), eps, None)
            pwm = (1.0 - wm_lapse) * pwm_soft + wm_lapse * (1.0 / nA)

            # Mixture policy
            p_total = wm_weight * pwm + (1.0 - wm_weight) * prl
            p_total = p_total / np.clip(p_total.sum(), eps, None)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == correct_actions[s] else 0
            simulated_rewards[tr] = r

            # RL update
            q[s, a] += lr * (r - q[s, a])

            # WM decay towards uniform for the visited state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w0[s, :]

            # WM encoding on rewarded trials
            if r > 0.5:
                w[s, :] = (1.0 - wm_alpha) * w[s, :]
                w[s, a] += wm_alpha

            tr += 1

    return simulated_actions, simulated_rewards
