def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL+WM with uncertainty-weighted arbitration and load-dependent WM decay.

    Parameters:
    - stimulus: sequence of stimuli (states) for each trial (array of ints)
    - blocks: block index for each trial (array)
    - set_sizes: block-dependent set sizes on each trial (array)
    - correct_answer: correct action for each trial (array of ints)
    - parameters: [lr, wm_bias, softmax_beta, entropy_gain, decay_load]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_bias, softmax_beta, entropy_gain, decay_load = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    H_ref = np.log(nA)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        block_set_sizes = set_sizes[block_indices]

        nS = int(block_set_sizes[0])
        load_scaled = max(0.0, (nS - 3) / 3.0)
        wm_decay_eff = min(1.0, max(0.0, decay_load * load_scaled))

        # Map each state to its correct action within the block
        correct_for_state = np.zeros(nS, dtype=int)
        for st in range(nS):
            idx = np.where(block_states == st)[0]
            if len(idx) > 0:
                correct_for_state[st] = int(block_correct[idx[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            rl_probs = np.exp(rl_logits)
            rl_probs /= np.sum(rl_probs)

            # WM policy
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            wm_probs = np.exp(wm_logits)
            wm_probs /= np.sum(wm_probs)

            # WM entropy and arbitration weight
            H_wm = -np.sum(wm_probs * np.log(np.clip(wm_probs, 1e-12, 1.0)))
            wm_logit = wm_bias + entropy_gain * (H_ref - H_wm) - load_scaled
            wm_weight_eff = 1.0 / (1.0 + np.exp(-wm_logit))

            # Mixture policy
            p_total = wm_weight_eff * wm_probs + (1.0 - wm_weight_eff) * rl_probs
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action for this state
            r = 1 if a == correct_for_state[s] else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay and update
            w[s, :] = (1.0 - wm_decay_eff) * w[s, :] + wm_decay_eff * w_0[s, :]
            if r > 0:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0

            tr += 1

    return simulated_actions, simulated_rewards
