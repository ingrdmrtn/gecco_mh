import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate actions and rewards using:
    - RL with choice stickiness
    - WM availability governed by capacity (sigmoid of set size)
    - WM softmax with high inverse temperature and write/decay dynamics

    Inputs:
    - stimulus: state index for each trial (array-like, ints)
    - blocks: block index for each trial (array-like, ints)
    - set_sizes: set size for each trial (array-like, ints) [used for WM availability per block]
    - correct_answer: correct action for each trial (array-like, ints in 0..2)
    - parameters: [lr, wm_weight, softmax_beta, capacity_k, capacity_slope, stickiness]

    Returns:
    - simulated_actions: array of chosen actions
    - simulated_rewards: array of binary rewards
    """
    lr, wm_weight, softmax_beta, capacity_k, capacity_slope, stickiness = parameters
    softmax_beta *= 10.0
    base_softmax_beta_wm = 50.0
    nA = 3

    stimulus = np.asarray(stimulus, dtype=int)
    blocks = np.asarray(blocks)
    set_sizes = np.asarray(set_sizes)
    correct_answer = np.asarray(correct_answer, dtype=int)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        # Capacity/availability uses provided set size for the block
        nS_capacity = int(set_sizes[block_idx][0])

        # State space for indexing Q/WM uses observed state labels
        max_state = int(block_states.max())
        nS_q = max_state + 1

        # Map each state to its correct action (assume stable within block)
        correct_actions = np.zeros(nS_q, dtype=int)
        for s in range(nS_q):
            s_trials = np.where(block_states == s)[0]
            if len(s_trials) > 0:
                correct_actions[s] = int(block_correct[s_trials[0]])
            else:
                correct_actions[s] = 0  # default (won't be used if state never occurs)

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS_q, nA))
        w = (1.0 / nA) * np.ones((nS_q, nA))
        w_0 = (1.0 / nA) * np.ones((nS_q, nA))
        last_action = -1 * np.ones(nS_q, dtype=int)

        # WM availability for the block
        avail = 1.0 / (1.0 + np.exp(capacity_slope * (float(nS_capacity) - capacity_k)))
        wm_weight_eff = wm_weight * avail

        for local_t, tr in enumerate(block_idx):
            s = int(block_states[local_t])

            # RL policy with choice stickiness
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += stickiness
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy
            W_s = w[s, :]
            p_wm = np.exp(base_softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture policy with availability-gated WM
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action for this state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM update: reward-gated write or decay toward uniform
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                write_rate = min(1.0, 0.5 + 0.5 * avail)
                w[s, :] = (1.0 - write_rate) * w[s, :] + write_rate * one_hot
            else:
                decay_rate = min(1.0, 0.5 + 0.5 * (1.0 - avail))
                w[s, :] = (1.0 - decay_rate) * w[s, :] + decay_rate * w_0[s, :]

            # Update stickiness memory
            last_action[s] = a

    return simulated_actions, simulated_rewards
