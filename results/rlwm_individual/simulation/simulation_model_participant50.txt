import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL + capacity-limited Working Memory (WM) mixture with WM decay.

    Parameters:
    - stimulus: array of stimuli (state identifiers) for each trial
    - blocks: array of block indices for each trial
    - set_sizes: array of (block-dependent) set sizes for each trial
    - correct_answer: array of correct actions for each trial (0..nA-1)
    - parameters: [lr, wm_weight, softmax_beta, wm_lr, wm_decay, capacity]

    Returns:
    - simulated_actions: array of simulated action choices (int)
    - simulated_rewards: array of simulated binary rewards (0/1)
    """
    lr, wm_weight, softmax_beta, wm_lr, wm_decay, capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3

    # Iterate block by block
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_size = int(set_sizes[block_idx][0])

        # Map states within this block to 0..nS-1
        unique_states = []
        state_map = {}
        for s in block_states_raw:
            if s not in state_map:
                state_map[s] = len(state_map)
                unique_states.append(s)
        nS = len(unique_states)  # number of distinct states in this block

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Use set size for capacity dilution as in the fitted model
        nS_for_capacity = int(block_set_size)

        # Simulate trials in this block
        for t_local, t_global in enumerate(block_idx):
            s_raw = block_states_raw[t_local]
            s = state_map[s_raw]

            # Policies
            Q_s = q[s, :]
            # RL softmax
            rl_logits = softmax_beta * Q_s
            # For numerical stability
            rl_logits -= np.max(rl_logits)
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM softmax
            W_s = w[s, :]
            wm_logits = softmax_beta_wm * W_s
            wm_logits -= np.max(wm_logits)
            p_soft_wm = np.exp(wm_logits)
            p_soft_wm /= np.sum(p_soft_wm)

            # Capacity dilution
            p_slot = min(1.0, float(capacity) / float(nS_for_capacity))
            p_wm = p_slot * p_soft_wm + (1.0 - p_slot) * (1.0 / nA)

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = np.random.choice(nA, p=p_total)
            simulated_actions[t_global] = a

            # Reward from correct answer
            correct_a = int(block_correct[t_local])
            r = 1 if a == correct_a else 0
            simulated_rewards[t_global] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM global decay toward uniform
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM state-specific update toward target vector (uniform except action=a set to r)
            t_vec = np.copy(w_0[s, :])
            t_vec[a] = r
            w[s, :] = (1.0 - wm_lr) * w[s, :] + wm_lr * t_vec

    return simulated_actions, simulated_rewards
