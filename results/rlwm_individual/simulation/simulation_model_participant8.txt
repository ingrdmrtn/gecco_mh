def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    Simulate actions and rewards for the RL + error-driven WM gating with set-size interference model.

    Inputs:
    - stimulus: array of state indices per trial (assumed 0..nS-1 within block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (used to get nS per block)
    - correct_answer: array of correct action per trial (0..2)
    - parameters: (lr, wm_weight, softmax_beta, wm_gate_err, interference_rate)

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_gate_err, interference_rate = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate blocks in order of appearance
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Use provided set size as in fitting code
        nS = int(block_set_sizes[0])

        # Map state -> correct action within this block (assumed stable per state)
        correct_per_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # first occurrence of this state in the block
            inds = np.where(block_states == s)[0]
            if len(inds) > 0:
                correct_per_state[s] = int(block_correct[inds[0]])
            else:
                # If state not present (edge case), default to 0
                correct_per_state[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Precompute forgetting rates dependent on set size
        forget = 1.0 - np.exp(-interference_rate * (float(nS) / 3.0))
        leak = 0.1 * forget if nS > 3 else 0.0
        ss_scale = 3.0 / max(3.0, float(nS))

        for local_t, tr in enumerate(block_idx):
            s = int(block_states[local_t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Softmax policies
            # RL policy
            x_rl = softmax_beta * Q_s
            x_rl -= np.max(x_rl)
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (near-deterministic)
            x_wm = softmax_beta_wm * W_s
            x_wm -= np.max(x_wm)
            p_wm = np.exp(x_wm)
            p_wm /= np.sum(p_wm)

            # Use current RL surprise to set arbitration after observing outcome;
            # but to act, we need a gating weight. We follow the fitting logic,
            # where gating depends on abs(r - Q[a]) on the current trial.
            # We therefore compute action probabilities using previous trial's gating proxy.
            # To match the likelihood logic closely, we act using mixture with gating that
            # will be computed after sampling and observing r on this same trial.
            # Implementation: act using base mixture (wm_weight scaled by set size, without error term),
            # then recompute the effective weight after observing r for learning updates.
            base_wm_weight = np.clip(wm_weight * ss_scale, 0.0, 1.0)
            p_total = base_wm_weight * p_wm + (1.0 - base_wm_weight) * p_rl

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Generate reward from environment
            r = 1 if a == int(correct_per_state[s]) else 0
            simulated_rewards[tr] = r

            # Now compute gating using surprise on this trial (for log-likelihood it affected choice prob,
            # here it will affect the WM vs RL mixture notionally; we apply it to nothing directly post-hoc,
            # but we maintain the same learning dynamics as in fitting.)
            delta_peek = r - Q_s[a]
            gate_err = 1.0 - np.exp(-wm_gate_err * abs(delta_peek))
            wm_weight_eff = np.clip(wm_weight * gate_err * ss_scale, 0.0, 1.0)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM forgetting/interference on current state
            w[s, :] = (1.0 - forget) * W_s + forget * w_0[s, :]

            # Reward-driven WM strengthening
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot

            # Cross-state leak for high set size
            if nS > 3:
                for s2 in range(nS):
                    if s2 == s:
                        continue
                    w[s2, :] = (1.0 - leak) * w[s2, :] + leak * w_0[s2, :]

            # Normalize WM row for current state
            w[s, :] = np.clip(w[s, :], 1e-12, None)
            w[s, :] /= np.sum(w[s, :])

    return simulated_actions, simulated_rewards
