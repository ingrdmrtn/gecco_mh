def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using Model 1: RL with state-local forgetting and directed exploration,
    modulated by set size and age.

    Inputs:
    - stimulus: array of state indices per trial (0..set_size-1 within block)
    - blocks: array of block index per trial
    - set_sizes: array of set size per trial (constant within block; e.g., 3 or 6)
    - correct_answer: array of correct action per trial (0,1,2). Assumed constant per state within a block.
    - age: participant age in years (float or array-like with first element used)
    - parameters: [alpha, beta, decay_base, age_effect, explore_bonus]

    Returns:
    - simulated_actions: array of simulated choices per trial (0,1,2)
    - simulated_rewards: array of simulated binary rewards per trial (0/1)
    """
    import numpy as np

    alpha, beta, decay_base, age_effect, explore_bonus = parameters
    beta = beta * 10.0

    a0 = age[0] if hasattr(age, "__len__") else age
    is_older = 1.0 if a0 >= 45 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx].astype(int)

        nA = 3
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumed constant within block)
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for st in range(nS):
            st_mask = (block_states == st)
            if np.any(st_mask):
                correct_actions_by_state[st] = block_correct[st_mask][0]
            else:
                # If a state doesn't appear (rare), default to 0
                correct_actions_by_state[st] = 0

        # Initialize Q-values and visit counts
        Q = np.zeros((nS, nA))
        N = np.zeros((nS, nA))

        base_decay = 1.0 / (1.0 + np.exp(-decay_base))

        for t in range(len(block_states)):
            s = int(block_states[t])
            set_size = float(block_set_sizes[t])

            # Effective decay increases with set size and for older adults
            eff_decay = base_decay * (set_size / 6.0) * (1.0 + age_effect * is_older)
            eff_decay = min(max(eff_decay, 0.0), 1.0)

            # Directed exploration bonus diminishes with older age via (1 - age_effect * is_older)
            bonus_scale = (set_size / 6.0) * (1.0 - age_effect * is_older)
            bonus_scale = max(0.0, bonus_scale)
            bonus = np.zeros(nA)
            for aa in range(nA):
                bonus[aa] = explore_bonus * bonus_scale / np.sqrt(1.0 + N[s, aa])

            # Softmax over Q + bonus
            prefs = Q[s, :] + bonus
            prefs_centered = prefs - np.max(prefs)
            expv = np.exp(beta * prefs_centered)
            p = expv / np.sum(expv)

            # Sample action
            a = int(np.random.choice(nA, p=p))
            simulated_actions[tr] = a

            # Simulate reward based on correct action for the state
            r = 1 if a == int(correct_actions_by_state[s]) else 0
            simulated_rewards[tr] = r

            # Update visit counts
            N[s, a] += 1.0

            # RL update
            pe = r - Q[s, a]
            Q[s, a] += alpha * pe

            # Forgetting on non-chosen actions in the visited state
            for aa in range(nA):
                if aa != a:
                    Q[s, aa] *= (1.0 - eff_decay)

            tr += 1

    return simulated_actions, simulated_rewards
