def simulate_model(stimulus, blocks, set_sizes, correct_answer, parameters):
    """
    RL with eligibility trace + WM mixture with set-size logistic arbitration (simulation).

    Inputs:
    - stimulus: array of state indices per trial (ints)
    - blocks: array of block indices per trial (ints)
    - set_sizes: array of set sizes per trial (ints) [constant within a block]
    - correct_answer: array of correct action per trial (ints in {0,1,2})
    - parameters: [lr, wm_weight0, softmax_beta, wm_decay, lambda_et, ns_alpha]

    Returns:
    - simulated_actions: array of sampled actions per trial (ints)
    - simulated_rewards: array of rewards per trial (0/1)
    """
    import numpy as np

    lr, wm_weight0, softmax_beta, wm_decay, lambda_et, ns_alpha = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    nA = 3
    eps = 1e-12

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate over blocks
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_indices.sort()  # preserve temporal order
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        # Use provided set size per block (assumed constant within block)
        nS = int(set_sizes[block_indices][0])

        # Map each state to its correct action (take first occurrence in block)
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            idxs = np.where(block_states == s)[0]
            if len(idxs) > 0:
                state_to_correct[s] = block_correct[idxs[0]]
            else:
                # If a state doesn't appear (edge case), default to 0
                state_to_correct[s] = 0

        # Initialize RL, WM, and eligibility trace
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        e = np.zeros((nS, nA))

        # Arbitration weight for WM as a function of set size
        wm_weight = wm_weight0 / (1.0 + np.exp(ns_alpha * (nS - 3.0)))
        wm_weight = float(np.clip(wm_weight, 0.0, 1.0))

        # Trial loop within block
        for local_t, tr in enumerate(block_indices):
            s = int(block_states[local_t])

            # Compute choice probabilities from RL and WM
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Softmax for RL
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl = prl / np.maximum(np.sum(prl), eps)

            # Softmax for WM (very deterministic)
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm = pwm / np.maximum(np.sum(pwm), eps)

            # Mixture
            p_total = wm_weight * pwm + (1.0 - wm_weight) * prl
            # Numerical safety: ensure non-negativity and normalization
            p_total = np.clip(p_total, 0.0, None)
            ssum = np.sum(p_total)
            if ssum <= 0:
                p_total = np.ones(nA) / nA
            else:
                p_total = p_total / ssum

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update with eligibility trace
            delta = r - Q_s[a]
            e *= lambda_et
            e[s, a] += 1.0
            q += lr * delta * e

            # WM decay toward baseline
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # If rewarded, reinforce the chosen association in WM
            if r > 0.5:
                w[s, :] = (1.0 - wm_decay) * w[s, :]
                w[s, a] += wm_decay

    return simulated_actions, simulated_rewards
