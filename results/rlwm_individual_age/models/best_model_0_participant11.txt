def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity/load- and age-adjusted WM weight and WM decay.

    Mixture policy: p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
    - RL system: tabular Q-learning.
    - WM system: fast, one-shot update for rewarded state-action, with decay toward uniform.
    - Arbitration: effective wm_weight decreases with set size and with older age.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - wm_weight_base: base WM mixture weight (0-1)
    - softmax_beta: RL inverse temperature (>0), internally scaled by 10
    - wm_decay: WM update/decay rate (0-1); governs overwrite strength on reward and decay toward uniform
    - load_sensitivity: scales how much larger set sizes reduce WM weight (>=0)
    - age_effect: multiplicative penalty on WM weight for older adults (>=0). Effective weight = wm_weight_base / (1+age_effect) if age>=45

    Inputs:
    - states, actions, rewards, blocks, set_sizes: 1D arrays (same length) for trials
    - age: array-like with one value (years)
    - Returns: negative log-likelihood of observed actions under the model
    """
    lr, wm_weight_base, softmax_beta, wm_decay, load_sensitivity, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM readout
    age = age[0]
    is_older = 1.0 if age >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]


            Q_center = Q_s - np.max(Q_s)
            exp_Q = np.exp(softmax_beta * Q_center)
            p_rl = exp_Q[a] / np.sum(exp_Q)

            W_center = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * W_center)
            p_wm = exp_W[a] / np.sum(exp_W)


            load_factor = 1.0 / (1.0 + load_sensitivity * max(0, nS - 3))
            age_factor = 1.0 / (1.0 + age_effect * is_older)
            wm_weight_eff = np.clip(wm_weight_base * load_factor * age_factor, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            delta = r - Q_s[a]
            q[s, a] += lr * delta


            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            if r > 0.0:

                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * one_hot

            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p
