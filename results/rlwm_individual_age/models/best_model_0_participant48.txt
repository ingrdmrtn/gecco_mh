def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity- and age-gated WM, plus global decay and lapse.
    
    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..set_size-1 within a block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of int
        Binary feedback (0/1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for each trial (3 or 6).
    age : array-like of float
        Participant age; age[0] used to determine older/younger group.
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse)
        - lr: RL learning rate (0..1)
        - wm_weight: mixture weight on WM policy (0..1)
        - softmax_beta: inverse temperature for RL (rescaled in-model)
        - wm_decay: global decay of WM traces toward uniform each trial (0..1)
        - wm_capacity: WM capacity K (in items) for younger adults; older adults assumed K-1
        - lapse: probability of random choice on each trial (0..1)
    
    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    
    Notes
    -----
    - WM is item-specific, one-shot storage of rewarded S-A pairs.
    - WM contribution is gated by set size vs effective capacity:
      Older adults (age>=45) have reduced effective K by 1 item.
    - WM traces decay globally toward uniform each trial with wm_decay.
    - Final choice prob mixes WM and RL and includes independent lapses.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse = model_parameters
    softmax_beta *= 10.0  # RL beta has a higher upper bound
    softmax_beta_wm = 50.0  # WM is very deterministic when available
    age_val = age[0]
    older = 1.0 if age_val >= 45 else 0.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))

        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        K_eff = max(1.0, wm_capacity - older*1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))


            nS_block = float(nS)
            wm_available = 1.0 if (nS_block <= K_eff) else 0.0

            if wm_available > 0.5:

                p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            else:

                p_wm = 1.0 / nA

            p_total = (1.0 - lapse) * (p_wm * wm_weight + (1.0 - wm_weight) * p_rl) + lapse * (1.0 / nA)
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            delta = r - Q_s[a]
            q[s, a] += lr * delta


            w = wm_decay * w + (1.0 - wm_decay) * w_0

            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:

                w[s, a] *= (1.0 - 0.5 * (1.0 - wm_decay))

        blocks_log_p += log_p

    return -blocks_log_p
