def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity/load- and age-adjusted WM weight and WM decay.

    Mixture policy: p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
    - RL system: tabular Q-learning.
    - WM system: fast, one-shot update for rewarded state-action, with decay toward uniform.
    - Arbitration: effective wm_weight decreases with set size and with older age.

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - wm_weight_base: base WM mixture weight (0-1)
    - softmax_beta: RL inverse temperature (>0), internally scaled by 10
    - wm_decay: WM update/decay rate (0-1); governs overwrite strength on reward and decay toward uniform
    - load_sensitivity: scales how much larger set sizes reduce WM weight (>=0)
    - age_effect: multiplicative penalty on WM weight for older adults (>=0). Effective weight = wm_weight_base / (1+age_effect) if age>=45

    Inputs:
    - states, actions, rewards, blocks, set_sizes: 1D arrays (same length) for trials
    - age: array-like with one value (years)
    - Returns: negative log-likelihood of observed actions under the model
    """
    lr, wm_weight_base, softmax_beta, wm_decay, load_sensitivity, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM readout
    age = age[0]
    is_older = 1.0 if age >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (softmax)
            # stable computation of p(action=a)
            Q_center = Q_s - np.max(Q_s)
            exp_Q = np.exp(softmax_beta * Q_center)
            p_rl = exp_Q[a] / np.sum(exp_Q)

            # Working Memory policy (softmax over W)
            W_center = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * W_center)
            p_wm = exp_W[a] / np.sum(exp_W)

            # Arbitration weight adjusted by load and age
            # Larger set sizes reduce WM contribution; older age reduces further
            load_factor = 1.0 / (1.0 + load_sensitivity * max(0, nS - 3))
            age_factor = 1.0 / (1.0 + age_effect * is_older)
            wm_weight_eff = np.clip(wm_weight_base * load_factor * age_factor, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform, then overwrite on rewarded actions
            # Decay current state's WM toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If reward, write a strong one-shot memory favoring chosen action
            if r > 0.0:
                # Overwrite by pushing probability mass toward chosen action
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * one_hot

            # Renormalize to maintain a distribution
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with slot-limited WM, RL decay, and perseveration bias.

    Mixture policy: p_total = wm_weight * p_wm + (1 - wm_weight) * p_rl
    - WM: When set size <= K slots (age-adjusted), WM dominates; otherwise weight scales by K/nS.
    - RL: Q-learning with value decay toward uniform baseline and perseveration bias on last action in state.
    - Age: Older adults effectively have one fewer WM slot (min 1).

    Parameters (model_parameters):
    - lr: RL learning rate (0-1)
    - wm_weight_base: base WM mixture weight (0-1)
    - softmax_beta: RL inverse temperature (>0), internally scaled by 10
    - rl_decay: RL value decay rate toward uniform (0-1)
    - perseveration: bias added to last chosen action in a state before softmax (>=0)
    - k_slots: nominal WM capacity in slots for younger group (>=1); older group uses max(1, k_slots-1)

    Inputs:
    - states, actions, rewards, blocks, set_sizes: 1D arrays (same length) for trials
    - age: array-like with one value (years)
    - Returns: negative log-likelihood of observed actions under the model
    """
    lr, wm_weight_base, softmax_beta, rl_decay, perseveration, k_slots = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = (age_val >= 45)

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize Q and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Last action per state for perseveration (-1 means none yet)
        last_action = -1 * np.ones(nS, dtype=int)

        # Effective WM capacity
        k_eff = int(max(1, int(round(k_slots)) - (1 if is_older else 0)))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]

            # RL decay toward uniform before using for policy
            q[s, :] = (1.0 - rl_decay) * q[s, :] + rl_decay * (1.0 / nA)

            # Perseveration bias: add to last chosen action for this state
            Q_hat = q[s, :].copy()
            if last_action[s] >= 0:
                Q_hat[last_action[s]] += perseveration

            # RL policy (softmax with biased Q_hat)
            Qc = Q_hat - np.max(Q_hat)
            exp_Q = np.exp(softmax_beta * Qc)
            p_rl = exp_Q[a] / np.sum(exp_Q)

            # WM policy
            W_s = w[s, :]
            Wc = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * Wc)
            p_wm = exp_W[a] / np.sum(exp_W)

            # Arbitration: when nS <= k_eff, use base weight; else scale by k_eff/nS
            if nS <= k_eff:
                wm_weight_eff = np.clip(wm_weight_base, 0.0, 1.0)
            else:
                wm_weight_eff = np.clip(wm_weight_base * (k_eff / float(nS)), 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL learning update (on unbiased q)
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: write on reward, otherwise keep as is (simple store)
            if r > 0.0:
                w[s, :] = (1.0 / (nA - 1)) * np.ones(nA)
                w[s, a] = 1.0  # give chosen action the peak; will be normalized below
                w[s, :] /= np.sum(w[s, :])

            # Update perseveration memory
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with uncertainty-based arbitration and asymmetric RL learning rates, plus lapse.

    Mixture policy with dynamic WM weight:
    - WM weight increases when WM policy is confident (low entropy) and decreases with larger set sizes.
    - RL: separate learning rates for positive and negative prediction errors.
    - Lapse rate: small probability of random responding; age increases lapse.

    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive PE (0-1)
    - lr_neg: RL learning rate for negative PE (0-1)
    - wm_weight_base: base WM mixture weight (0-1)
    - softmax_beta: RL inverse temperature (>0), internally scaled by 10
    - epsilon: base lapse rate (0-1)
    - load_sensitivity: scales how much larger set sizes reduce WM weight (>=0)

    Inputs:
    - states, actions, rewards, blocks, set_sizes: 1D arrays (same length) for trials
    - age: array-like with one value (years)
    - Returns: negative log-likelihood of observed actions under the model
    """
    lr_pos, lr_neg, wm_weight_base, softmax_beta, epsilon, load_sensitivity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            Qc = Q_s - np.max(Q_s)
            exp_Q = np.exp(softmax_beta * Qc)
            p_rl = exp_Q[a] / np.sum(exp_Q)

            # WM policy
            Wc = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * Wc)
            p_wm = exp_W[a] / np.sum(exp_W)

            # Uncertainty-based arbitration:
            # Compute WM entropy (base e), normalize to [0,1] by dividing by log(nA)
            p_wm_full = exp_W / np.sum(exp_W)
            entropy = -np.sum(p_wm_full * np.log(np.clip(p_wm_full, 1e-12, 1.0)))
            norm_entropy = entropy / np.log(nA)  # 0 (certain) to 1 (max uncertainty)

            # Load reduces WM weight; certainty increases it
            load_factor = 1.0 / (1.0 + load_sensitivity * max(0, nS - 3))
            certainty_factor = 1.0 - norm_entropy  # higher when WM more peaked
            wm_weight_eff = np.clip(wm_weight_base * load_factor * certainty_factor, 0.0, 1.0)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl

            # Lapse with age: older group has higher lapse (double)
            epsilon_eff = np.clip(epsilon * (1.0 + is_older), 0.0, 1.0)
            p_total = (1.0 - epsilon_eff) * p_total + epsilon_eff * (1.0 / nA)
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            eta = lr_pos if pe >= 0.0 else lr_neg
            q[s, a] += eta * pe

            # WM update: decay toward uniform and one-shot write on reward
            w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = 0.5 * w[s, :] + 0.5 * one_hot
            # Renormalize row
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p
