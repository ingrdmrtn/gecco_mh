def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with age- and capacity-dependent WM engagement and WM decay.

    Idea:
    - Choices are a mixture of a model-free RL policy and a working memory (WM) policy.
    - WM encodes one-shot rewarded S-A associations and decays toward uniform each trial.
    - WM engagement is limited by capacity (K), which depends on age group, and by set size (nS).
      Effective WM weight = wm_weight * min(1, K/nS).
    - RL updates via delta rule.

    Parameters (model_parameters):
    - lr: scalar in (0,1), RL learning rate.
    - wm_weight: base weight in [0,1] for WM contribution when fully engaged.
    - softmax_beta: inverse temperature for RL; internally scaled up by 10.
    - wm_decay: rate in [0,1] by which WM entries drift back toward uniform each trial.
    - K_young: WM capacity for younger group (e.g., 3-4).
    - K_old: WM capacity for older group (e.g., 2-3).

    Inputs:
    - states: array of state indices per trial (block-local indexing assumed 0..nS-1).
    - actions: array of chosen actions per trial (0..2).
    - rewards: array of rewards (0/1) per trial.
    - blocks: array of block indices per trial.
    - set_sizes: array of set size per trial (3 or 6).
    - age: array-like with a single element (participant age in years).
    - model_parameters: list/array of the six parameters above.

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, K_young, K_old = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy
    age_val = age[0]
    is_older = 1 if age_val >= 45 else 0
    K = K_old if is_older else K_young

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values and WM store
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM engagement given capacity and set size
        wm_engagement = min(1.0, max(0.0, K / float(nS)))
        wm_mix = wm_weight * wm_engagement

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Compute RL policy
            Q_s = q[s, :]
            # softmax trick: p(a) = 1 / sum exp(beta*(Q - Q[a]))
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1.0 / nA

            # WM decay toward uniform every trial (global drift)
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # Compute WM policy
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Mixture
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: one-shot store if rewarded
            if r >= 1.0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            # if not rewarded, we leave WM as is (decay already applied)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with set-size and age-based WM gating, separate RL learning rates for gains/losses,
    and a lapse process.

    Idea:
    - RL system uses separate learning rates for positive and negative prediction errors.
    - WM system stores recent rewarded action per state and softly avoids recently failed actions.
    - WM contribution is gated by set size via a logistic function; older adults have steeper
      set-size sensitivity (age-dependent gating).
    - A lapse parameter mixes in a uniform-random policy.

    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive PE (r - Q > 0).
    - lr_neg: RL learning rate for negative PE (r - Q <= 0).
    - wm_weight: base WM mixture weight.
    - softmax_beta: RL inverse temperature; internally scaled up by 10.
    - lapse: probability of a random choice (0..1).
    - gate_slope: slope of logistic gating on set size; larger = sharper drop with nS.

    Age usage:
    - Younger (age < 45): gating center K0 = 4
    - Older (age >= 45): gating center K0 = 3 and effective slope = 1.5 * gate_slope

    Inputs:
    - states, actions, rewards, blocks, set_sizes, age: as described in model1.
    - model_parameters: the six parameters above.

    Returns:
    - Negative log-likelihood.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, lapse, gate_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1 if age_val >= 45 else 0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL values and WM store
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age-dependent gating parameters
        K0 = 3.0 if is_older else 4.0
        slope = 1.5 * gate_slope if is_older else gate_slope

        # WM engagement via logistic gate of set size
        gate = 1.0 / (1.0 + np.exp(slope * (nS - K0)))
        wm_mix = wm_weight * gate

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1.0 / nA

            # WM: mild decay and update
            # Decay slightly toward uniform each step (fixed small drift)
            decay = 0.05
            w = (1.0 - decay) * w + decay * w_0

            # WM policy
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Mixture plus lapse
            p_mixture = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = (1.0 - lapse) * p_mixture + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            lr = lr_pos if pe > 0 else lr_neg
            q[s, a] += lr * pe

            # WM update:
            if r >= 1.0:
                # store the rewarded action (one-shot)
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # softly avoid the failed action by shifting mass to others
                shift = 0.2
                loss_mass = min(shift, w[s, a])
                w[s, a] -= loss_mass
                distribute = loss_mass / (nA - 1)
                for aa in range(nA):
                    if aa != a:
                        w[s, aa] += distribute

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with perseveration bias and age-modulated WM decay.

    Idea:
    - Choices come from a mixture of RL and WM.
    - RL policy includes a perseveration bias toward the last chosen action in the current state.
    - WM stores rewarded actions; WM decay rate depends on age through a multiplicative factor.
    - Age parameter modulates either increasing perseveration or increasing WM decay in older group.

    Parameters (model_parameters):
    - lr: RL learning rate.
    - wm_weight: WM mixture weight.
    - softmax_beta: RL inverse temperature; internally scaled by 10.
    - wm_decay: base WM decay rate toward uniform.
    - perseveration: bias weight added to the last chosen action in the state within RL policy.
    - age_bias: multiplicative factor applied to wm_decay if older, and to perseveration if older.
      Specifically: if older, wm_decay *= (1 + age_bias) and perseveration *= (1 + age_bias);
      if younger, wm_decay and perseveration are unchanged.

    Inputs:
    - states, actions, rewards, blocks, set_sizes, age, model_parameters as described above.

    Returns:
    - Negative log-likelihood.
    """
    lr, wm_weight, softmax_beta, wm_decay, perseveration, age_bias = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1 if age_val >= 45 else 0

    # Age modulation
    if is_older:
        wm_decay = wm_decay * (1.0 + age_bias)
        perseveration = perseveration * (1.0 + age_bias)

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track last chosen action per state for perseveration
        last_action = -np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy with perseveration bias
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += perseveration
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1.0 / nA

            # WM decay
            w = (1.0 - wm_decay) * w + wm_decay * w_0

            # WM policy
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Mixture
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - q[s, a]
            q[s, a] += lr * pe

            # WM update: store reward; if no reward, do nothing (decay handles forgetting)
            if r >= 1.0:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Update perseveration memory
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
