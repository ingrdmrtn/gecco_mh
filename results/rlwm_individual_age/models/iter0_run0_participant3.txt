def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Cognitive model 1: Capacity-limited WM + RL mixture.
    - RL: tabular Q-learning with softmax policy.
    - WM: state-specific one-shot memory that stores the last rewarded action with decay.
    - Mixture: WM contribution scales with set size relative to WM capacity, and with age group.
      Younger group (age < 45) has higher effective WM capacity than older.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1).
    - wm_weight: Base mixture weight of WM vs RL (0..1).
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally).
    - wm_decay: Decay rate of WM weights toward uniform (0..1).
    - wm_alpha: Strength of WM encoding on rewarded trials (0..1).
    - K_capacity: Nominal WM capacity (in number of state-action mappings; e.g., between 2..6).

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_alpha, K_capacity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]

    # Age-modulated effective capacity: younger preserve nominal capacity, older reduced by 1 (min 1)
    if age < 45:
        K_eff = max(1.0, K_capacity)
    else:
        K_eff = max(1.0, K_capacity - 1.0)

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action a)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over WM weights (prob of chosen action a)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Capacity-limited WM mixture weight (bounded in [0,1])
            # Scales WM contribution down when set size exceeds effective capacity
            cap_ratio = min(1.0, K_eff / float(nS))
            wmix = np.clip(wm_weight * cap_ratio, 0.0, 1.0)

            p_total = p_wm * wmix + (1.0 - wmix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # 1) decay toward uniform for the current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # 2) if rewarded, encode chosen action as a sharpened one-hot attractor
            if r > 0.0:
                # move probability mass toward the chosen action
                w[s, :] = (1.0 - wm_alpha) * w[s, :]
                w[s, a] += wm_alpha

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Cognitive model 2: Uncertainty-gated WM + RL.
    - RL: tabular Q-learning with softmax policy.
    - WM: decaying associative weights; encodes rewarded actions.
    - Mixture gating: WM contribution increases with RL policy uncertainty (entropy)
      and decreases with set size (load). Age group biases the gate (younger favor WM, older suppress WM).

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values (0..1).
    - wm_base: Base WM mixture weight (0..1).
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally).
    - wm_decay: Decay rate of WM weights toward uniform (0..1).
    - gate_temp: Sharpness of the uncertainty/load gating (positive).
    - age_bias: Age bias term for WM gating (positive => younger more WM, older less WM).

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    lr, wm_base, softmax_beta, wm_decay, gate_temp, age_bias = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]

    # Age term: younger bias WM up, older bias WM down
    if age < 45:
        age_term = +abs(age_bias)
    else:
        age_term = -abs(age_bias)

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy prob for chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # RL uncertainty via entropy of RL softmax distribution
            logits = softmax_beta * (Q_s - np.max(Q_s))
            exp_logits = np.exp(logits)
            pi = exp_logits / np.sum(exp_logits)
            entropy = -np.sum(pi * np.log(np.clip(pi, 1e-12, 1.0)))
            # Normalize by max entropy log(nA)
            H_norm = entropy / np.log(len(Q_s))

            # WM policy prob for chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Load signal: higher set size reduces WM weight; normalize to [0,1] with 3..6 mapping
            load_norm = (float(nS) - 3.0) / (6.0 - 3.0) if nS >= 3 else 0.0
            # Gate combines uncertainty (encourages WM), load (discourages WM), and age bias
            gate_input = H_norm - load_norm + age_term
            gate = 1.0 / (1.0 + np.exp(-gate_temp * gate_input))
            wmix = np.clip(wm_base * gate, 0.0, 1.0)

            p_total = p_wm * wmix + (1.0 - wmix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay each trial for this state; encode if reward
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.0:
                # encode with strength proportional to uncertainty (stronger WM when RL is uncertain)
                enc = np.clip(H_norm, 0.0, 1.0)
                w[s, :] = (1.0 - enc) * w[s, :]
                w[s, a] += enc

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Cognitive model 3: Asymmetric RL learning + perseveration + noisy WM retrieval.
    - RL: tabular Q-learning with separate learning rates for positive and negative prediction errors.
    - Policy: RL softmax includes a state-specific perseveration bonus on repeating the last choice.
    - WM: decaying weights encoding rewarded actions; retrieval noise increases with set size and age.
    - Mixture: WM weight decreases with load; age increases WM noise.

    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive prediction errors (0..1).
    - lr_neg: RL learning rate for negative prediction errors (0..1).
    - wm_weight: Base WM mixture weight (0..1).
    - softmax_beta: Inverse temperature for RL softmax (rescaled internally).
    - persev: Perseveration bonus added to the value of the previously chosen action in a state.
    - wm_noise: Scales WM retrieval noise as a function of set size and age.

    Returns:
    - Negative log-likelihood of the observed action sequence under the model.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, persev, wm_noise = model_parameters
    softmax_beta *= 10.0
    base_beta_wm = 50.0
    age = age[0]

    # Age factor for WM noise: 0 for younger, 1 for older
    age_factor = 0.0 if age < 45 else 1.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track previous choice per state for perseveration
        prev_choice = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()

            # Add perseveration bonus to the previously chosen action in this state
            if prev_choice[s] >= 0:
                Q_s[prev_choice[s]] += persev

            # RL policy prob for chosen action with perseveration bias
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy with retrieval noise that increases with set size and age
            # Higher wm_noise, larger set size, and older age => lower effective beta_wm
            noise_scale = 1.0 + wm_noise * (max(0, nS - 3)) * (1.0 + age_factor)
            beta_wm_eff = base_beta_wm / noise_scale
            W_s = w[s, :]
            p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture: WM weight decreases with set size (load)
            wmix = np.clip(wm_weight * (3.0 / float(nS)), 0.0, 1.0)

            p_total = p_wm * wmix + (1.0 - wmix) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - q[s, a]
            if pe >= 0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM update: decay, then encode on reward
            w[s, :] = 1.0 * ((w[s, :] * (1.0 - 0.5)) + (w_0[s, :] * 0.5))  # mild decay to uniform
            if r > 0.0:
                # Encode rewarded action strongly
                w[s, :] = 0.0 * w[s, :] + 0.0 * w_0[s, :]  # clear
                w[s, :] = (1.0 / nA) * np.ones(nA)  # start from uniform
                w[s, a] = 1.0  # store as deterministic target
                # renormalize softly to avoid degenerate probability under finite beta_wm
                w[s, :] = w[s, :] / np.sum(w[s, :])

            # Update perseveration memory
            prev_choice[s] = a

        blocks_log_p += log_p

    return -blocks_log_p
