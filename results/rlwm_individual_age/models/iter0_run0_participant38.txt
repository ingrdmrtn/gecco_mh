def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+Capacity-limited Working Memory (age-sensitive)
    Mixture of model-free RL and a capacity-limited working-memory (WM) component.
    WM stores the most recently rewarded action for each state and decays toward uniform.
    WM availability scales with set size and declines in the older group.

    Parameters
    - model_parameters: tuple/list of
        lr: RL learning rate (0..1)
        wm_weight: Mixture weight for WM vs RL in action policy (0..1)
        softmax_beta: Inverse temperature for RL (rescaled internally)
        wm_decay: Per-trial decay of WM toward uniform (0..1)
        wm_capacity: Effective WM capacity (in number of state-action pairs)
        old_penalty: Proportional reduction (0..1) of WM capacity if age >= 45
    Returns
    - Negative log-likelihood of the observed action sequence
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, old_penalty = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_old = 1.0 if age_val >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL Q and WM distributions
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age-modulated WM capacity => availability per trial
        K_eff = max(0.0, wm_capacity * (1.0 - old_penalty * is_old))
        # Availability of WM for a given set size
        avail = min(1.0, K_eff / max(1.0, nS))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy: softmax prob of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy:
            # If WM contains a confident item (W_s near one-hot), softmax_wm will be near-deterministic.
            p_wm_mem = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            # Effective WM policy blends availability with uniform when not available
            p_wm = avail * p_wm_mem + (1.0 - avail) * (1.0 / nA)

            # Mixture policy and log likelihood
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform, then store last rewarded action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + Short-term perseveration WM, with age- and load-modulated exploration
    WM encodes the last chosen action for each state (a short-term "sticky" policy).
    RL temperature is reduced (more exploration) for older adults under high load (set size 6).

    Parameters
    - model_parameters: tuple/list of
        lr: RL learning rate (0..1)
        wm_weight: Mixture weight for WM vs RL in action policy (0..1)
        softmax_beta: Baseline RL inverse temperature (rescaled internally)
        wm_decay: Decay of WM toward uniform on each trial (0..1)
        stickiness: Scales WM certainty (>=0; higher => stronger perseveration)
        old_explore: Fractional reduction (0..1) of RL beta if age>=45 and set size=6
    Returns
    - Negative log-likelihood of the observed action sequence
    """
    lr, wm_weight, softmax_beta, wm_decay, stickiness, old_explore = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_old = 1.0 if age_val >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Older adults under high load explore more: reduce RL beta
        beta_rl_block = softmax_beta
        if (nS >= 6) and (is_old > 0.5):
            beta_rl_block = softmax_beta * max(0.0, 1.0 - old_explore)

        # WM availability scales with load (simple 3/nS heuristic)
        avail = min(1.0, 3.0 / max(1.0, nS))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(beta_rl_block * (Q_s - Q_s[a])))

            # WM policy encodes short-term perseveration: last chosen action for this state
            # Update WM to reflect most recent choice signal (perseveration)
            # Implement decay first (to keep total mass normalized), then reinforce last chosen action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            w[s, :] = 0.0
            w[s, a] = 1.0

            # Convert WM map to near-deterministic policy with adjustable stickiness
            beta_wm_eff = softmax_beta_wm * max(0.0, stickiness)
            W_s = w[s, :]
            p_wm_mem = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))
            p_wm = avail * p_wm_mem + (1.0 - avail) * (1.0 / nA)

            # Mixture and log likelihood
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + Error-gated Working Memory with age-sensitive decay and capacity
    WM stores the most recently rewarded action per state. WM availability declines with:
      - larger set sizes,
      - recent error load (error trace),
      - older age (via higher decay and reduced capacity).
    The mixture between WM and RL is fixed, but WM policy reliability is gated by these factors.

    Parameters
    - model_parameters: tuple/list of
        lr: RL learning rate (0..1)
        wm_weight: Mixture weight for WM vs RL in policy (0..1)
        softmax_beta: RL inverse temperature (rescaled internally)
        wm_decay: Base WM decay toward uniform (0..1); increased in older adults
        wm_capacity: Base WM capacity (in items); reduced in older adults
        error_sensitivity: Strength (>=0) with which recent errors reduce WM availability
    Returns
    - Negative log-likelihood of the observed action sequence
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, error_sensitivity = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_old = 1.0 if age_val >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0

    for b in np.unique(blocks):
        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Error trace to gate WM availability
        err_trace = 0.0

        # Age-adjusted decay and capacity
        decay_eff = min(1.0, wm_decay * (1.0 + 0.5 * is_old))
        cap_eff_base = max(0.0, wm_capacity * (1.0 - 0.3 * is_old))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM availability depends on set size and current error load
            cap_eff = max(0.0, cap_eff_base - max(0.0, error_sensitivity) * err_trace)
            avail = min(1.0, cap_eff / max(1.0, nS))

            # WM policy (near-deterministic over stored action, otherwise uniform)
            p_wm_mem = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = avail * p_wm_mem + (1.0 - avail) * (1.0 / nA)

            # Mixture and log likelihood
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            log_p += np.log(max(p_total, eps))

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay then store only if rewarded
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

            # Update error trace (decays with the same timescale as WM; grows with errors)
            err_trace = (1.0 - decay_eff) * err_trace + (1.0 - r)

        blocks_log_p += log_p

    return -blocks_log_p
