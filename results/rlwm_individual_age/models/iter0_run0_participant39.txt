def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with capacity-limited WM and age-dependent capacity cost.

    Idea:
    - Choices are a mixture of a model-free RL system and a working-memory (WM) system.
    - WM has a limited effective capacity K that scales the WM contribution by K / set_size.
    - Older adults (age >= 45) have reduced effective WM capacity relative to younger adults.
    - WM representations decay toward uniform over time; WM stores last rewarded action for each state.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: Baseline mixture weight of WM contribution (0..1)
    - softmax_beta: Inverse temperature for RL policy (scaled up internally)
    - wm_decay: Decay rate of WM toward uniform per trial (0..1)
    - K_base: Baseline WM capacity (in "state slots")
    - age_capacity_drop: Amount of capacity reduction for older adults (>=0)

    Inputs:
    - states, actions, rewards: trial-wise arrays
    - blocks: block index per trial (states are block-specific)
    - set_sizes: set size per trial (3 or 6)
    - age: array with a single value (participant age)
    - Returns: negative log-likelihood of observed actions
    """
    lr, wm_weight, softmax_beta, wm_decay, K_base, age_capacity_drop = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM when available
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    # Effective capacity with age penalty
    K_eff_base = max(0.0, K_base - is_older * age_capacity_drop)

    eps = 1e-12
    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL Q-values and WM probabilities
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL policy for chosen action
            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            # WM policy for chosen action
            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Capacity-weighted WM mixture (set-size dependent, age-adjusted capacity)
            K_eff = K_eff_base
            capacity_scale = min(1.0, K_eff / float(nS)) if nS > 0 else 0.0
            wm_mix = wm_weight * capacity_scale

            # Total policy (mixture)
            p_total = wm_mix * p_wm + (1.0 - wm_mix) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM value updating: store last rewarded action deterministically
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with age-dependent WM weighting and lapse; set-size scaled WM reliability.

    Idea:
    - Choices are a mixture of RL and WM, with an additional lapse (uniform responding) component.
    - WM reliability (mixture weight) scales inversely with set size (3 vs 6) and is reduced for older adults.
    - WM stores last rewarded action; WM representation decays toward uniform.
    - Lapse accounts for occasional random responses, more likely under high load.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - wm_weight: Baseline WM weight (0..1)
    - softmax_beta: RL inverse temperature (scaled internally)
    - lapse: Mixture weight for uniform random response (0..0.5)
    - wm_decay: WM decay toward uniform (0..1)
    - age_wm_drop: Proportional reduction of WM weight for older adults (0..1)

    Inputs:
    - states, actions, rewards: trial-wise arrays
    - blocks: block index per trial
    - set_sizes: set size per trial
    - age: array with a single value
    - Returns: negative log-likelihood of observed actions
    """
    lr, wm_weight, softmax_beta, lapse, wm_decay, age_wm_drop = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    eps = 1e-12
    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Set-size scaling: WM more reliable at small set size
            ss = float(nS)
            ss_scale = 3.0 / ss  # =1 for 3, =0.5 for 6
            # Age reduction of WM weight for older adults
            wm_weight_eff = wm_weight * ss_scale * (1.0 - is_older * age_wm_drop)
            wm_weight_eff = np.clip(wm_weight_eff, 0.0, 1.0)

            # Lapse mixture with uniform random policy
            p_uniform = 1.0 / nA
            p_mix = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * p_uniform
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update: store last rewarded action
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with probabilistic WM storage gate modulated by set size and age.

    Idea:
    - WM storage is probabilistic and depends on set size (harder to store at size 6).
    - Older adults have lower probability to store and maintain WM contents.
    - WM contributes via mixture with RL; both the WM mixture weight and storage probability
      are downscaled at larger set sizes.

    Parameters (model_parameters):
    - lr: RL learning rate (0..1)
    - softmax_beta: RL inverse temperature (scaled internally)
    - wm_weight_base: Baseline WM mixture weight (0..1)
    - store_prob_base: Baseline probability to store WM on a rewarded trial (0..1)
    - wm_decay: WM decay toward uniform (0..1)
    - age_slope: Proportional reduction applied to both WM weight and storage for older adults (0..1)

    Inputs:
    - states, actions, rewards: trial-wise arrays
    - blocks: block index per trial
    - set_sizes: set size per trial
    - age: array with a single value
    - Returns: negative log-likelihood of observed actions
    """
    lr, softmax_beta, wm_weight_base, store_prob_base, wm_decay, age_slope = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    eps = 1e-12
    rng = np.random  # assumed available if simulation is needed; here we use expectations only

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / max(denom_rl, eps)

            W_s = w[s, :]
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / max(denom_wm, eps)

            # Set-size factor: WM weight and storage reduced at larger set size
            ss = float(nS)
            ss_factor = 3.0 / ss  # 1 at set size 3, 0.5 at set size 6

            # Age reductions apply multiplicatively
            age_factor = (1.0 - age_slope * is_older)

            wm_weight_eff = np.clip(wm_weight_base * ss_factor * age_factor, 0.0, 1.0)
            store_prob_eff = np.clip(store_prob_base * ss_factor * age_factor, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM probabilistic storage of rewarded action (in expectation form)
            # Rather than sampling, we blend toward the stored one-hot by store_prob_eff.
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - store_prob_eff) * w[s, :] + store_prob_eff * one_hot

        blocks_log_p += log_p

    return -blocks_log_p
