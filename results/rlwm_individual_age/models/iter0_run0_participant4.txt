def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity penalty and age-modulated WM reliance.

    Combines:
    - Model-free RL (Q-learning) with a single learning rate.
    - Working-memory (WM) policy that stores the last rewarded action per state with decay/refresh.
    - Mixture of RL and WM depends on: (a) a base WM weight, (b) set size penalty (capacity pressure),
      and (c) age group (younger rely more on WM).

    Parameters
    ----------
    states : array-like of shape (T,)
        State index on each trial (0..nS-1 per block).
    actions : array-like of shape (T,)
        Chosen action on each trial (0..2).
    rewards : array-like of shape (T,)
        Binary feedback on each trial (0/1).
    blocks : array-like of shape (T,)
        Block index per trial.
    set_sizes : array-like of shape (T,)
        The set size presented on each trial (3 or 6). Constant within a block.
    age : array-like of shape (1,)
        Participant age; used to set age group (<45 = younger, >=45 = older).
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_decay, wm_penalty)
        - lr: RL learning rate (0..1).
        - wm_weight: Base WM mixture weight (0..1).
        - softmax_beta: Inverse temperature for RL policy (scaled inside).
        - wm_decay: WM encoding/decay strength (0..1). Larger = stronger overwrite toward the last rewarded action, and faster decay toward uniform when unrewarded.
        - wm_penalty: Capacity penalty applied when set size is large (0..1). Effective WM weight multiplied by (1 - wm_penalty) at set size 6.

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_penalty = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]

    # Age modulation: younger rely more on WM
    is_younger = 1.0 if age < 45 else 0.0
    age_boost = 1.0 + 0.25 * is_younger  # up to +25% WM reliance for younger adults

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity penalty by set size
        size_penalty = 1.0
        if nS > 3:
            size_penalty = (1.0 - wm_penalty)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (deterministic softmax over the WM row)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Effective WM weight modulated by age and set size
            wm_eff = wm_weight * age_boost * size_penalty
            wm_eff = np.clip(wm_eff, 0.0, 1.0)

            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update:
            # - Always decay toward uniform a bit
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # - If rewarded, overwrite toward a one-hot memory for the chosen action
            if r > 0.0:
                w[s, :] = (1.0 - wm_decay) * w_0[s, :]
                w[s, a] += wm_decay

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with lapse, set-size-dependent WM capacity gating, and age effects on capacity.

    Features:
    - Model-free RL (Q-learning).
    - WM policy that encodes rewarded actions; decays otherwise.
    - Mixture weight scaled by an age-dependent capacity K (K=4 younger, K=2 older).
      Effective WM usage is proportional to min(1, K / set_size).
    - Lapse rate (action-independent noise).

    Parameters
    ----------
    states : array-like
        State index per trial.
    actions : array-like
        Chosen action per trial (0..2).
    rewards : array-like
        Binary reward per trial.
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (constant within block).
    age : array-like of shape (1,)
        Participant age; used to define age group.
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, lapse, wm_decay)
        - lr: RL learning rate (0..1).
        - wm_weight: Base WM mixture weight (0..1).
        - softmax_beta: RL inverse temperature (scaled inside).
        - lapse: Probability of random choice (0..0.2 typical). Mixes in a uniform policy.
        - wm_decay: WM decay/overwrite strength (0..1).

    Returns
    -------
    float
        Negative log-likelihood.
    """
    lr, wm_weight, softmax_beta, lapse, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]

    # Age-dependent WM capacity (theoretical K slots)
    K = 4.0 if age < 45 else 2.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity gating factor: proportion of set items that can be held
        cap_factor = min(1.0, K / float(nS))
        wm_eff = np.clip(wm_weight * cap_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with lapse
            p_mix = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform; on reward, store the chosen action
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.0:
                w[s, :] = (1.0 - wm_decay) * w_0[s, :]
                w[s, a] += wm_decay

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning rates + WM with interference, age-modulated interference.

    Features:
    - RL: separate learning rates for positive vs. negative prediction errors.
    - WM: encodes only when rewarded; between-trial interference blends WM toward the
      average of all state memories. Interference increases with set size and age.
    - Mixture of WM and RL as in RLWM.

    Parameters
    ----------
    states : array-like
        State index per trial.
    actions : array-like
        Chosen action per trial.
    rewards : array-like
        Binary reward per trial.
    blocks : array-like
        Block index per trial.
    set_sizes : array-like
        Set size per trial (3 or 6).
    age : array-like of shape (1,)
        Participant age; used for age group.
    model_parameters : tuple/list
        (lr_pos, lr_neg, wm_weight, softmax_beta, interference)
        - lr_pos: RL learning rate for positive PE (0..1).
        - lr_neg: RL learning rate for negative PE (0..1).
        - wm_weight: Base WM mixture weight (0..1).
        - softmax_beta: RL inverse temperature (scaled inside).
        - interference: Base WM interference strength (0..1); scaled by set size and age.

    Returns
    -------
    float
        Negative log-likelihood.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, interference = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age = age[0]

    # Age group
    is_older = 1.0 if age >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Interference scale: stronger for larger set size and in older age
        size_scale = 1.0 if nS <= 3 else 1.5
        age_scale = 1.25 if is_older > 0.5 else 1.0
        intf_eff = np.clip(interference * size_scale * age_scale, 0.0, 1.0)

        # WM effective weight also dampened by interference
        wm_eff = np.clip(wm_weight * (1.0 - 0.5 * intf_eff), 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            if pe >= 0.0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM update:
            # 1) Global interference: blend each row towards the block-average memory
            w_mean = w.mean(axis=0, keepdims=True)
            w = (1.0 - intf_eff) * w + intf_eff * w_mean

            # 2) Encode rewarded action for current state
            if r > 0.0:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # If not rewarded, slight decay toward uniform for that state
                w[s, :] = 0.9 * w[s, :] + 0.1 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
