Below are three standalone cognitive models that implement RL+WM mechanisms, include age-group effects, and return the negative log-likelihood of the observed choices.

Notes:
- Signature: (states, actions, rewards, blocks, set_sizes, age, model_parameters)
- No imports inside functions. Assume numpy as np is already imported.
- All parameters are used, at most 6 per model.
- The code follows the provided template structure and fills in the WM policy and WM updating where indicated.
- Numerical safety: probabilities are clamped with a small epsilon to avoid log(0).


def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with age- and set-size-dependent WM weight and WM decay.

    Parameters
    ----------
    states : array-like of int
        State identifiers for each trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen actions per trial (0..2).
    rewards : array-like of int
        Binary feedback per trial (0 or 1).
    blocks : array-like of int
        Block index per trial; states do not carry over across blocks.
    set_sizes : array-like of int
        Set size (3 or 6) per trial; constant within a block.
    age : array-like
        Participant age as a one-element array; used to determine age group.
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_capacity, age_scale, wm_decay)
        - lr: RL learning rate (0..1).
        - wm_weight: base WM mixture weight (0..1).
        - softmax_beta: inverse temperature for RL policy (scaled by 10 internally).
        - wm_capacity: WM capacity anchor controlling load-sensitivity of WM use.
        - age_scale: decrement to effective capacity for older adults (>=45).
        - wm_decay: WM decay rate controlling drift to baseline per trial and load.

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr, wm_weight, softmax_beta, wm_capacity, age_scale, wm_decay = model_parameters
    softmax_beta *= 10.0  # RL beta has higher upper bound
    softmax_beta_wm = 50.0  # WM is near-deterministic
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM mixture weight (age- and load-sensitive)
        # Higher set size reduces WM reliance; older age reduces effective capacity
        cap_eff = wm_capacity - age_scale * is_older
        # map capacity vs set size into [0,1] via sigmoid
        load_term = cap_eff - float(nS)
        wm_block_weight = 1.0 / (1.0 + np.exp(-2.0 * load_term))
        wm_block_weight *= wm_weight
        # WM decay increases with larger set sizes and in older adults
        decay_eff = 1.0 - np.exp(-wm_decay * (float(nS) / 3.0) * (1.0 + 0.5 * is_older))
        decay_eff = np.clip(decay_eff, 0.0, 1.0)

        # WM learning rates (tie to wm_weight to ensure parameter use)
        wm_alpha_pos = np.clip(0.5 + 0.5 * wm_weight, 0.0, 1.0)
        wm_alpha_neg = np.clip(0.2 + 0.3 * wm_weight, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability for the chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM choice probability using near-deterministic softmax over W_s
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture policy with block-level WM weight
            p_total = wm_block_weight * p_wm + (1.0 - wm_block_weight) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay WM towards baseline
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]
            # Associative WM update: move chosen action value toward observed outcome
            if r > 0.5:
                w[s][a] += wm_alpha_pos * (1.0 - w[s][a])
            else:
                w[s][a] += wm_alpha_neg * (0.0 - w[s][a])

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with perseveration bias, lapse, and win-stay WM. Age increases lapse and reduces WM under load.

    Parameters
    ----------
    states : array-like of int
        State identifiers per trial.
    actions : array-like of int
        Chosen actions per trial (0..2).
    rewards : array-like of int
        Binary feedback per trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial; constant within block.
    age : array-like
        Participant age as a one-element array; used to determine age group.
    model_parameters : tuple/list
        (lr_pos, lr_neg, wm_weight, softmax_beta, perseveration, lapse)
        - lr_pos: RL learning rate for rewarded trials.
        - lr_neg: RL learning rate for unrewarded trials.
        - wm_weight: base WM mixture weight.
        - softmax_beta: inverse temperature for RL policy (scaled by 10 internally).
        - perseveration: bias added to repeating last action in a state.
        - lapse: base lapse rate; final lapse increases in older adults.

    Returns
    -------
    float
        Negative log-likelihood of the observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, perseveration, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0
    blocks_log_p = 0.0
    eps = 1e-12

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # State-specific memory of last rewarded action and its strength
        mem_action = -1 * np.ones(nS, dtype=int)
        mem_strength = np.zeros(nS)

        # Perseveration bias scales with age
        pers_bias = perseveration * (1.0 + 0.5 * is_older)
        # Lapse increases with age via sigmoid transform
        lapse_rate = 1.0 / (1.0 + np.exp(-(lapse + 0.5 * is_older)))
        lapse_rate = np.clip(lapse_rate, 0.0, 1.0)

        # WM weight reduced by set size and age
        wm_block_weight = wm_weight * (3.0 / float(nS)) * (1.0 / (1.0 + 0.5 * is_older))
        wm_block_weight = np.clip(wm_block_weight, 0.0, 1.0)

        # Memory decay per trial (faster decay with larger set and in older adults)
        gamma = 1.0 - np.exp(-0.6 * (3.0 / float(nS)) / (1.0 + 0.5 * is_older))
        gamma = np.clip(gamma, 0.0, 1.0)

        # Track last action per state for perseveration
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL values with perseveration bonus to the previously chosen action in this state
            Q_s = q[s, :].copy()
            if last_action[s] >= 0:
                Q_s[last_action[s]] += pers_bias

            # RL probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM: win-stay memory represented as a strong preference for last rewarded action if present
            W_s = np.zeros(3)
            if mem_action[s] >= 0:
                W_s[mem_action[s]] = mem_strength[s]
            # Softmax to compute probability of chosen action under WM
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture of WM and RL with lapse to uniform
            p_mix = wm_block_weight * p_wm + (1.0 - wm_block_weight) * p_rl
            p_total = (1.0 - lapse_rate) * p_mix + lapse_rate * (1.0 / 3.0)
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            delta = r - q[s][a]
            alpha = lr_pos if r > 0.5 else lr_neg
            q[s][a] += alpha * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay memory strength
            mem_strength[s] = (1.0 - gamma) * mem_strength[s]
            # Update memory on reward: store action with full strength; otherwise decay only
            if r > 0.5:
                mem_action[s] = a
                mem_strength[s] = 1.0

            # Keep a lightweight W buffer consistent with memory for WM softmax
            w[s, :] = w_0[s, :]
            if mem_action[s] >= 0:
                w[s, mem_action[s]] = w[s, mem_action[s]] + mem_strength[s]

            # Update last action for perseveration
            last_action[s] = a

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with meta-control: WM weight adapts to recent surprise (PE) and load; age increases WM noise/decay.

    Parameters
    ----------
    states : array-like of int
        State identifiers per trial.
    actions : array-like of int
        Chosen actions per trial (0..2).
    rewards : array-like of int
        Binary feedback per trial (0 or 1).
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial; constant within a block.
    age : array-like
        Participant age as a one-element array; used to determine age group.
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_decay, age_noise, pe_temp)
        - lr: RL learning rate.
        - wm_weight: base WM mixture weight.
        - softmax_beta: inverse temperature for RL policy (scaled by 10 internally).
        - wm_decay: base WM decay rate per trial.
        - age_noise: increases WM decay/noise in older adults (>=45).
        - pe_temp: temperature for converting PE into reliability (higher reduces PE weighting).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, age_noise, pe_temp = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0
    blocks_log_p = 0.0
    eps = 1e-12

    # PE smoothing parameter from pe_temp
    k_pe = 1.0 / (1.0 + np.exp(-pe_temp))  # in (0,1)

    for b in np.unique(blocks):
        mask = (blocks == b)
        block_actions = actions[mask]
        block_rewards = rewards[mask]
        block_states = states[mask]
        block_set_sizes = set_sizes[mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Track running absolute PE per state to estimate surprise
        pe_run = np.zeros(nS)

        # Effective WM decay increases with set size and age_noise in older adults
        decay_eff = 1.0 - np.exp(-wm_decay * (float(nS) / 3.0) * (1.0 + age_noise * is_older))
        decay_eff = np.clip(decay_eff, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute current PE and update running surprise for gating
            pe = abs(r - Q_s[a])
            pe_run[s] = (1.0 - k_pe) * pe_run[s] + k_pe * pe
            reliability = 1.0 - pe_run[s]  # lower with surprising outcomes
            reliability = np.clip(reliability, 0.0, 1.0)

            # WM mixture weight adapts to load, age, and reliability
            load_penalty = (float(nS) - 3.0) / 3.0  # 0 for nS=3, 1 for nS=6
            base_gate = wm_weight - load_penalty - (age_noise * is_older)
            wm_gate = 1.0 / (1.0 + np.exp(-2.0 * base_gate))
            wm_gate *= reliability
            wm_gate = np.clip(wm_gate, 0.0, 1.0)

            p_total = wm_gate * p_wm + (1.0 - wm_gate) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # WM decays towards baseline then updates chosen action toward outcome
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]
            # Associative WM update: move toward 1 if rewarded, else toward 0
            alpha_wm = 1.0 - (1.0 - decay_eff)  # align with decay scale
            if r > 0.5:
                w[s][a] += alpha_wm * (1.0 - w[s][a])
            else:
                w[s][a] += alpha_wm * (0.0 - w[s][a])

        blocks_log_p += log_p

    return -blocks_log_p
