def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with WM capacity, decay, and age-dependent WM engagement.

    The model blends a standard RL softmax policy with a working-memory (WM) policy.
    WM maintains a fast, one-shot mapping from state to action that decays over trials.
    WM engagement scales with effective set-size (capacity limit) and age group.

    Negative log-likelihood of observed choices is returned.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback per trial.
    blocks : array-like of int
        Block index per trial. States do not carry over across blocks.
    set_sizes : array-like of int
        The set size of the current block (3 or 6), repeated per trial in that block.
    age : array-like of float
        Participant age (single value array). Used to modulate WM engagement; older adults (>45)
        rely less on WM.
    model_parameters : sequence of 6 floats
        lr : RL learning rate (0..1).
        wm_weight : Baseline WM mixture weight (0..1).
        softmax_beta : Inverse temperature for RL policy (scaled up internally).
        phi : WM decay rate toward uniform (0..1).
        K : WM capacity (in number of state-action mappings effectively supported).
        age_sensitivity : Controls how age shifts WM engagement (positive => older uses less WM).

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, phi, K, age_sensitivity = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # deterministic WM policy
    age = age[0]

    # Age factor: older adults (>45) engage WM less (logistic transform)
    # age_factor in (0,1); if age > 45, factor < 0.5 for positive age_sensitivity
    age_center = 45.0
    age_factor = 1.0 / (1.0 + np.exp(age_sensitivity * (age - age_center)))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM stores
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity/Load factor: how much WM can be used at this set-size
        cap_factor = min(1.0, max(0.0, K / float(nS)))  # in [0,1]

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability of the chosen action (softmax)
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1.0 / nA

            # WM choice probability of the chosen action (softmax over WM store)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Effective WM weight: baseline scaled by capacity and age
            wm_weight_eff = wm_weight * cap_factor * age_factor
            wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w = (1.0 - phi) * w + phi * w_0

            # WM fast update based on outcome
            if r >= 0.5:
                # Store the rewarded mapping deterministically
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # If unrewarded, suppress chosen action and spread mass over others
                w[s, a] = 0.0
                others = [aa for aa in range(nA) if aa != a]
                w[s, others] = 1.0 / (nA - 1)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning rates + WM one-shot learning; age and load gate WM usage.

    The model mixes:
    - RL with separate learning rates for positive and negative outcomes.
    - WM that encodes most recent rewarded mapping per state with decay.
    WM engagement is modulated by set size (harder at 6) and age group.

    Negative log-likelihood of observed choices is returned.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback per trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial.
    age : array-like of float
        Participant age; older adults (>45) are modeled to rely less on WM and more on RL.
    model_parameters : sequence of 6 floats
        lr_pos : RL learning rate for rewards (0..1).
        lr_neg : RL learning rate for non-rewards (0..1).
        wm_weight : Baseline WM weight (0..1).
        softmax_beta : RL inverse temperature (scaled internally).
        phi : WM decay rate toward uniform (0..1).
        age_sensitivity : Controls WM reduction with age and load.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, phi, age_sensitivity = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]

    age_center = 45.0
    # Base age factor (older -> less WM)
    age_factor = 1.0 / (1.0 + np.exp(age_sensitivity * (age_val - age_center)))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load gating: logistic transform of set size (3 easier -> more WM; 6 harder -> less WM)
        # Map nS=3 -> ~high; nS=6 -> ~low through a simple inverse relation then squash
        load_raw = (3.0 / float(nS))  # 1.0 at 3, 0.5 at 6
        load_factor = 1.0 / (1.0 + np.exp(-5.0 * (load_raw - 0.75)))  # ~high for 3, low for 6

        # Combine baseline, age, and load effects
        wm_weight_block = wm_weight * age_factor * load_factor
        wm_weight_block = min(max(wm_weight_block, 0.0), 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1.0 / nA

            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            if r >= 0.5:
                q[s, a] += lr_pos * (r - Q_s[a])
            else:
                q[s, a] += lr_neg * (r - Q_s[a])

            # WM decay toward uniform baseline
            w = (1.0 - phi) * w + phi * w_0

            # WM one-shot update
            if r >= 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Penalize chosen action in WM and spread probability
                w[s, a] = 0.0
                others = [aa for aa in range(nA) if aa != a]
                w[s, others] = 1.0 / (nA - 1)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM with age- and load-dependent lapse rate and age-dependent WM decay.

    The model mixes RL and WM policies, then adds a lapse (random choice) component
    that increases with cognitive load (set size) and age in older adults.
    WM decays toward uniform with a decay rate that also increases with age.

    Negative log-likelihood of observed choices is returned.

    Parameters
    ----------
    states : array-like of int
        State index per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action per trial (0..2).
    rewards : array-like of {0,1}
        Binary feedback per trial.
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial.
    age : array-like of float
        Participant age; older adults (>45) show higher lapse and WM decay here.
    model_parameters : sequence of 6 floats
        lr : RL learning rate (0..1).
        wm_weight : Baseline WM weight (0..1).
        softmax_beta : RL inverse temperature (scaled internally).
        phi_base : Baseline WM decay (0..1) before age/load modulation.
        lapse_base : Baseline lapse probability (0..1).
        age_sensitivity : Controls how age shifts both decay and lapse (positive => older more lapse/decay).

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, phi_base, lapse_base, age_sensitivity = model_parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    age_center = 45.0
    # Age effect scaler in (0,1), older -> lower value when age_sensitivity>0
    age_factor = 1.0 / (1.0 + np.exp(age_sensitivity * (age_val - age_center)))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Load factor: larger set size -> more lapse, less WM
        load_raw = (3.0 / float(nS))  # 1.0 for 3, 0.5 for 6
        # Lapse increases as load_raw decreases; WM decreases similarly
        load_for_lapse = 1.0 - load_raw  # 0 for 3, 0.5 for 6

        # Age increases decay and lapse via logistic transforms
        phi = np.clip(phi_base + load_for_lapse * (1.0 - age_factor), 0.0, 1.0)
        lapse = np.clip(lapse_base + load_for_lapse * (1.0 - age_factor), 0.0, 1.0)

        # WM weight reduced by load and age
        wm_weight_block = wm_weight * age_factor * load_raw
        wm_weight_block = min(max(wm_weight_block, 0.0), 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            p_rl = 1.0 / denom_rl if denom_rl > 0 else 1.0 / nA

            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm = 1.0 / denom_wm if denom_wm > 0 else 1.0 / nA

            # Mixture then lapse
            p_mix = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w = (1.0 - phi) * w + phi * w_0

            # WM update driven by outcome
            if r >= 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                w[s, a] = 0.0
                others = [aa for aa in range(nA) if aa != a]
                w[s, others] = 1.0 / (nA - 1)

        blocks_log_p += log_p

    return -blocks_log_p
