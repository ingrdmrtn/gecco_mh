def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity- and age-gated WM, plus global decay and lapse.
    
    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..set_size-1 within a block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of int
        Binary feedback (0/1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for each trial (3 or 6).
    age : array-like of float
        Participant age; age[0] used to determine older/younger group.
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse)
        - lr: RL learning rate (0..1)
        - wm_weight: mixture weight on WM policy (0..1)
        - softmax_beta: inverse temperature for RL (rescaled in-model)
        - wm_decay: global decay of WM traces toward uniform each trial (0..1)
        - wm_capacity: WM capacity K (in items) for younger adults; older adults assumed K-1
        - lapse: probability of random choice on each trial (0..1)
    
    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    
    Notes
    -----
    - WM is item-specific, one-shot storage of rewarded S-A pairs.
    - WM contribution is gated by set size vs effective capacity:
      Older adults (age>=45) have reduced effective K by 1 item.
    - WM traces decay globally toward uniform each trial with wm_decay.
    - Final choice prob mixes WM and RL and includes independent lapses.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse = model_parameters
    softmax_beta *= 10.0  # RL beta has a higher upper bound
    softmax_beta_wm = 50.0  # WM is very deterministic when available
    age_val = age[0]
    older = 1.0 if age_val >= 45 else 0.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL Q-values
        q = (1.0 / nA) * np.ones((nS, nA))

        # WM traces and baseline
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective capacity: older adults have K reduced by 1 item (min 1)
        K_eff = max(1.0, wm_capacity - older*1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL choice probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM availability gate based on set size vs capacity
            nS_block = float(nS)
            wm_available = 1.0 if (nS_block <= K_eff) else 0.0

            if wm_available > 0.5:
                # Deterministic WM softmax over WM traces for this state
                p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            else:
                # If beyond capacity, WM is effectively unavailable for this set
                p_wm = 1.0 / nA

            # Mixture of WM and RL, plus an independent lapse to uniform
            p_total = (1.0 - lapse) * (p_wm * wm_weight + (1.0 - wm_weight) * p_rl) + lapse * (1.0 / nA)
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Global decay toward uniform
            w = wm_decay * w + (1.0 - wm_decay) * w_0
            # One-shot encoding if rewarded; mild suppression if not rewarded
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Reduce confidence in the chosen action for this state
                w[s, a] *= (1.0 - 0.5 * (1.0 - wm_decay))

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with confidence-gated WM usage and age-/load-adjusted gate, plus WM decay.
    
    Parameters
    ----------
    states : array-like of int
        State index on each trial.
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of int
        Binary feedback (0/1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for each trial (3 or 6).
    age : array-like of float
        Participant age; age[0] used to determine older/younger group.
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_threshold, wm_decay)
        - lr: RL learning rate (0..1)
        - wm_weight: mixture weight on WM policy (0..1)
        - softmax_beta: inverse temperature for RL (rescaled in-model)
        - wm_threshold: confidence threshold for WM use (margin criterion)
        - wm_decay: global decay of WM traces toward uniform each trial (0..1)
    
    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    
    Notes
    -----
    - WM softmax is used only if its confidence (max-min of W_s) exceeds a gate.
    - The gate is stricter for older adults and larger set sizes
      (threshold scaled by (1 + older) * (set_size/3)).
    - WM traces are one-shot for rewarded pairs and decay globally.
    """
    lr, wm_weight, softmax_beta, wm_threshold, wm_decay = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    older = 1.0 if age_val >= 45 else 0.0
    eps = 1e-12

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Confidence (margin) in WM for this state
            wm_margin = np.max(W_s) - np.min(W_s)
            # Threshold scales up with age group and set size
            thresh = wm_threshold * (1.0 + older) * (float(nS) / 3.0)

            if wm_margin >= thresh:
                p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            else:
                p_wm = 1.0 / nA

            p_total = p_wm * wm_weight + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Global decay toward uniform
            w = wm_decay * w + (1.0 - wm_decay) * w_0
            # One-shot encoding of rewarded S-A; mild unlearning on errors
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Push chosen action for this state toward baseline
                w[s, a] = w[s, a] * wm_decay + (1.0 - wm_decay) * (1.0 / nA)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with valence-asymmetric learning rates + probabilistic WM encoding that depends on age and load.
    
    Parameters
    ----------
    states : array-like of int
        State index on each trial.
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of int
        Binary feedback (0/1).
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size for each trial (3 or 6).
    age : array-like of float
        Participant age; age[0] used to determine older/younger group.
    model_parameters : tuple/list
        (lr_pos, lr_neg, wm_weight, softmax_beta, wm_encoding, lapse)
        - lr_pos: RL learning rate for positive prediction errors
        - lr_neg: RL learning rate for negative prediction errors
        - wm_weight: mixture weight on WM policy (0..1)
        - softmax_beta: inverse temperature for RL (rescaled in-model)
        - wm_encoding: baseline WM encoding strength; higher -> stronger WM
        - lapse: probability of random choice on each trial (0..1)
    
    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    
    Notes
    -----
    - WM "encoding strength" is modeled deterministically as a scale factor on WM
      determinism: beta_wm_eff = beta_wm * sigmoid(wm_encoding - load - age_group),
      where load = (set_size - 3), age_group = 1 for older, 0 for younger.
    - Larger set sizes and older age reduce effective WM determinism.
    - RL learns with separate learning rates for gains and losses.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, wm_encoding, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    older = 1.0 if age_val >= 45 else 0.0
    eps = 1e-12

    # Sigmoid helper (no imports)
    def sigmoid(x):
        return 1.0 / (1.0 + np.exp(-x))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # Effective WM determinism decreases with load and older age
            load = float(nS - 3)
            beta_wm_eff = softmax_beta_wm * sigmoid(wm_encoding - load - older)
            # If beta_wm_eff is very small, WM policy approximates uniform
            if beta_wm_eff <= 1e-6:
                p_wm = 1.0 / nA
            else:
                p_wm = 1.0 / np.sum(np.exp(beta_wm_eff * (W_s - W_s[a])))

            # Mixture + lapse
            p_total = (1.0 - lapse) * (p_wm * wm_weight + (1.0 - wm_weight) * p_rl) + lapse * (1.0 / nA)
            p_total = max(p_total, eps)
            log_p += np.log(p_total)

            # RL update with valence asymmetry
            pe = r - Q_s[a]
            if pe >= 0.0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # "Encoding strength" shapes how sharply WM stores rewarded associations.
            # Use beta_wm_eff-derived weight to set one-shot storage sharpness.
            if r > 0.5:
                # Move W_s toward a one-hot at action a, with strength depending on beta_wm_eff
                sharpen = sigmoid((beta_wm_eff / softmax_beta_wm) * 5.0)  # in (0,1)
                w[s, :] = (1.0 - sharpen) * w_0[s, :] + sharpen * 0.0
                w[s, a] = (1.0 - sharpen) * w_0[s, a] + sharpen * 1.0
            else:
                # Gentle drift toward uniform when unrewarded
                drift = 0.2
                w[s, :] = (1.0 - drift) * w[s, :] + drift * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
