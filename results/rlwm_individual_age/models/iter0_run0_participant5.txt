def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity- and age-modulated WM access and decay.

    Idea:
    - Choices are a mixture of model-free RL and a capacity-limited working memory (WM).
    - WM is more reliable in small set sizes and less so in large set sizes.
    - Age group modulates WM access: older adults have reduced WM access (scaled by age_effect),
      while younger adults use the base parameter.
    - WM stores the last rewarded action per state (one-shot encoding) with decay toward a uniform prior.

    Parameters (model_parameters):
    - lr: RL learning rate for Q-values.
    - wm_weight: Base mixture weight for WM contribution (will be modulated by set size and age).
    - softmax_beta: Inverse temperature for RL choice (internally scaled by 10 as per template).
    - wm_decay: Decay rate of WM toward uniform baseline each trial (0=no decay, 1=full reset).
    - capacity_k: Set-size threshold controlling WM access; higher values allow WM to operate at larger set sizes.
    - age_effect: Multiplicative reduction of WM access for older adults (>45). For younger, factor=1.

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, wm_weight, softmax_beta, wm_decay, capacity_k, age_effect = model_parameters
    softmax_beta *= 10  # beta has a higher upper bound
    softmax_beta_wm = 50  # very deterministic
    age = age[0]
    is_older = age > 45

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = block_set_sizes[0]

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Compute block-specific WM mixture weight based on set size and age
        # Gate WM access with a smooth capacity function: higher when nS < capacity_k
        # gate in (0,1): sigmoid(capacity_k - nS)
        gate = 1.0 / (1.0 + np.exp(-(capacity_k - nS)))
        age_factor = (age_effect if is_older else 1.0)
        wm_weight_block = np.clip(wm_weight * gate * age_factor, 0.0, 1.0)

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (probability of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM policy: softmax over WM weights; with high beta it approximates an argmax.
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with block- and age-adjusted WM weight
            p_total = p_wm * wm_weight_block + (1 - wm_weight_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay WM toward uniform baseline
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # One-shot encoding on rewarded trials: move strongly toward one-hot on chosen action
            if r > 0:
                # Push the chosen action toward 1 and others toward 0 while preserving probability mass
                w[s, :] = 0.0 * w[s, :] + 0.0  # clear row (we reconstruct one-hot below)
                w[s, a] = 1.0
            else:
                # On negative feedback, suppress the chosen action slightly to reflect avoidance memory
                # while leaving others relatively higher
                suppress = 0.2
                w[s, a] = max(w[s, a] - suppress, 0.0)
                # Renormalize to a distribution to keep softmax scale stable
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :]  # fallback to uniform if fully suppressed
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with valence-asymmetric learning + WM with interference and recency.

    Idea:
    - RL uses separate learning rates for positive vs. negative prediction errors (alpha+ and alpha-).
    - WM maintains a recency-weighted memory trace of rewarded actions per state, but suffers from
      set-size-dependent interference. Larger sets dilute WM via an interference parameter.
    - Age group modulates interference: older adults experience stronger interference.

    Parameters (model_parameters):
    - lr_pos: RL learning rate for positive prediction errors.
    - lr_neg: RL learning rate for negative prediction errors.
    - wm_weight: Base mixture weight for WM contribution.
    - softmax_beta: Inverse temperature for RL choice (internally scaled by 10 as per template).
    - wm_decay: Recency decay of WM toward uniform per trial.
    - interference: Strength of WM interference per additional item beyond 1 (scaled by set size and age).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, wm_decay, interference = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    age = age[0]
    is_older = age > 45

    # Age modulation of interference: amplify for older group
    age_interference_factor = 1.5 if is_older else 1.0

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = block_set_sizes[0]

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        # Compute WM mixture weight reduced by set-size interference
        # Effective interference grows with (nS - 1)
        inter = interference * (nS - 1) * age_interference_factor
        wm_weight_block = np.clip(wm_weight * (1.0 / (1.0 + inter)), 0.0, 1.0)

        # Maintain a simple recency signal: we will boost WM encoding on first successful encounter
        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # [FILL IN THE POLICY FOR THE WORKING MEMORY]
            # WM softmax policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = p_wm * wm_weight_block + (1 - wm_weight_block) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with valence asymmetry
            pe = r - Q_s[a]
            if pe >= 0:
                q[s][a] += lr_pos * pe
            else:
                q[s][a] += lr_neg * pe

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay toward uniform prior (recency)
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Interference: bleed some mass from the current state's WM into uniform when set size is large
            if nS > 1:
                bleed = np.clip(inter / (inter + 1.0), 0.0, 0.9)
                w[s, :] = (1.0 - bleed) * w[s, :] + bleed * w_0[s, :]

            # Encode after outcome: reward strengthens chosen action; non-reward weakly suppresses it
            if r > 0:
                # Move distribution toward one-hot on chosen action, but not fully (to allow interference effects)
                strengthen = 0.8
                w[s, :] = (1.0 - strengthen) * w[s, :]  # shrink others
                w[s, a] += strengthen
                # Renormalize
                w[s, :] /= np.sum(w[s, :])
            else:
                suppress = 0.1
                w[s, a] = max(0.0, w[s, a] - suppress)
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RLâ€“WM arbitration model based on confidence/uncertainty with age modulation.

    Idea:
    - Choices are mixed between RL and WM, but the mixture weight is determined online by
      an arbitration signal: when WM is confident and RL is uncertain, rely more on WM.
    - RL uncertainty is proxied by the entropy of the RL softmax distribution at the current state.
    - WM confidence is proxied by the margin (max minus second max) of the WM action weights.
    - Age modulates arbitration baseline: older adults shift weight away from WM (k_age parameter).

    Parameters (model_parameters):
    - lr: RL learning rate.
    - base_wm_weight: Baseline WM mixture weight before arbitration.
    - softmax_beta: Inverse temperature for RL (internally scaled by 10).
    - arb_gain: Sensitivity of the arbitration to (WM_confidence - RL_uncertainty).
    - wm_decay: Decay rate of WM toward uniform prior.
    - k_age: Additive age bias on WM weight (negative values reduce WM use in older adults).

    Returns:
    - Negative log-likelihood of observed choices.
    """
    lr, base_wm_weight, softmax_beta, arb_gain, wm_decay, k_age = model_parameters
    softmax_beta *= 10
    softmax_beta_wm = 50
    age = age[0]
    is_older = age > 45

    blocks_log_p = 0
    for b in np.unique(blocks):

        block_actions = actions[blocks == b]
        block_rewards = rewards[blocks == b]
        block_states = states[blocks == b]
        block_set_sizes = set_sizes[blocks == b]

        nA = 3
        nS = block_set_sizes[0]

        q = (1 / nA) * np.ones((nS, nA))
        w = (1 / nA) * np.ones((nS, nA))
        w_0 = (1 / nA) * np.ones((nS, nA))

        log_p = 0
        for t in range(len(block_states)):

            a = block_actions[t]
            s = block_states[t]
            r = block_rewards[t]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy and its entropy (uncertainty)
            denom_rl = np.sum(np.exp(softmax_beta * (Q_s - np.max(Q_s))))
            pi_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s))) / denom_rl
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            rl_entropy = -np.sum(pi_rl * np.log(np.clip(pi_rl, 1e-12, 1.0)))

            # WM policy and its confidence (margin between best and second-best)
            denom_wm = np.sum(np.exp(softmax_beta_wm * (W_s - np.max(W_s))))
            pi_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s))) / denom_wm
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            sorted_W = np.sort(W_s)
            wm_conf = sorted_W[-1] - sorted_W[-2]  # larger margin => more confident

            # Arbitration: increase WM weight when WM is confident and RL is uncertain
            # base_wm_weight in (0,1); arb term passed through sigmoid-like transform by clipping
            arb_signal = wm_conf - rl_entropy  # higher => more WM
            wm_weight = base_wm_weight + arb_gain * arb_signal

            # Age modulation: older adults shift down by k_age; younger shift up by -k_age if negative
            if is_older:
                wm_weight += k_age  # typically negative to reduce WM use in older adults
            else:
                wm_weight -= min(0.0, k_age)  # if k_age is negative, give younger a slight boost

            wm_weight = np.clip(wm_weight, 0.0, 1.0)

            # Mixture likelihood
            p_total = p_wm * wm_weight + (1 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s][a] += lr * delta

            # [FILL IN THE VALUE UPDATING FOR THE WORKING MEMORY]
            # Decay WM toward uniform prior
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Outcome-driven WM encoding:
            # - Reward: push toward one-hot on chosen action
            # - No reward: mild suppression of chosen action (avoidance trace)
            if r > 0:
                push = 0.9
                w[s, :] = (1.0 - push) * w[s, :]
                w[s, a] += push
                w[s, :] /= np.sum(w[s, :])
            else:
                suppress = 0.15
                w[s, a] = max(0.0, w[s, a] - suppress)
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= row_sum

        blocks_log_p += log_p

    return -blocks_log_p
