def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity- and age-modulated WM weight, WM decay, and softmax decision.

    Parameters
    ----------
    states : array-like of int
        State indices per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action indices per trial (0..2).
    rewards : array-like of int
        Binary feedback per trial (0 or 1).
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial.
    age : array-like of float
        Age of the participant; age[0] is used. Older group: age > 45.
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, wm_decay, capacity_C, age_effect)
        - lr: RL learning rate (0..1).
        - wm_weight: Base mixture weight on WM (0..1).
        - softmax_beta: Inverse temperature for RL policy; internally scaled up (x10).
        - wm_decay: Per-trial decay of WM toward uniform (0..1).
        - capacity_C: WM capacity proxy (1..6); down-weights WM when set size exceeds capacity.
        - age_effect: Scales how much being older reduces WM contribution (positive reduces WM with age).

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, capacity_C, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # deterministic WM policy
    participant_age = age[0]
    is_older = 1.0 if participant_age > 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy for chosen action a
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # Working memory policy: softmax over WM weights (deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Effective WM mixture weight: capacity- and age-modulated
            capacity_factor = 1.0 if nS <= capacity_C else max(0.0, capacity_C / float(nS))
            # Older participants down-weight WM as age_effect increases
            age_mod = 1.0 / (1.0 + np.exp(age_effect * (participant_age - 45.0)))
            wm_weight_eff = np.clip(wm_weight * capacity_factor * age_mod, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform baseline
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update: on rewarded trials, write the correct mapping (one-hot toward chosen action)
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                # Overwrite strongly to emulate episodic WM encoding
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with RL perseveration bias and WM one-shot store, both modulated by set size and age.

    Parameters
    ----------
    states : array-like of int
        State indices per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action indices per trial (0..2).
    rewards : array-like of int
        Binary feedback per trial (0 or 1).
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial.
    age : array-like of float
        Age of the participant; age[0] is used. Older group: age > 45.
    model_parameters : tuple
        (lr, wm_weight, softmax_beta, wm_forget, perseveration, age_effect)
        - lr: RL learning rate (0..1).
        - wm_weight: Base mixture weight on WM (0..1).
        - softmax_beta: Inverse temperature for RL policy; internally scaled up (x10).
        - wm_forget: Per-trial WM forgetting toward uniform (0..1).
        - perseveration: Bias to repeat last action in state s in the RL policy (>=0).
        - age_effect: Scales how much age reduces WM contribution (positive reduces WM with age).

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_forget, perseveration, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    participant_age = age[0]

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # Track last action per state for perseveration
        last_action = -1 * np.ones(nS, dtype=int)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Add perseveration bias to RL values for repeating last action in this state
            if last_action[s] >= 0:
                bias = np.zeros(nA)
                bias[last_action[s]] = perseveration
                Q_s = Q_s + bias

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: if WM has a strong memory (peaked), it's near-deterministic; otherwise near-uniform
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WM weight modulated by set size and age (older -> lower WM weight)
            capacity_factor = 1.0 if nS <= 3 else 0.5  # simple heuristic: larger set halves WM influence
            age_mod = 1.0 / (1.0 + np.exp(age_effect * (participant_age - 45.0)))
            wm_weight_eff = np.clip(wm_weight * capacity_factor * age_mod, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - (q[s, a])
            q[s, a] += lr * delta

            # WM forgetting toward uniform
            w[s, :] = (1.0 - wm_forget) * w[s, :] + wm_forget * w_0[s, :]

            # WM one-shot store on reward
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

            # Update perseveration tracker
            last_action[s] = a

        blocks_log_p += log_p

    return -float(blocks_log_p)


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with trial-wise WM gating by surprise (prediction error), set size, and age.

    Parameters
    ----------
    states : array-like of int
        State indices per trial (0..nS-1 within a block).
    actions : array-like of int
        Chosen action indices per trial (0..2).
    rewards : array-like of int
        Binary feedback per trial (0 or 1).
    blocks : array-like of int
        Block index per trial.
    set_sizes : array-like of int
        Set size (3 or 6) per trial.
    age : array-like of float
        Age of the participant; age[0] is used. Older group: age > 45.
    model_parameters : tuple
        (lr, wm_weight_base, softmax_beta, wm_decay, gating_sensitivity, age_effect)
        - lr: RL learning rate (0..1).
        - wm_weight_base: Baseline WM mixture weight before gating (0..1).
        - softmax_beta: Inverse temperature for RL policy; internally scaled up (x10).
        - wm_decay: Per-trial WM decay toward uniform (0..1).
        - gating_sensitivity: Controls how strongly surprise and set size gate WM weight.
        - age_effect: Scales how much age reduces WM contribution (positive reduces WM with age).

    Returns
    -------
    float
        Negative log-likelihood of observed choices under the model.
    """
    lr, wm_weight_base, softmax_beta, wm_decay, gating_sensitivity, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    participant_age = age[0]

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute surprise signal from current policy values (before update)
            pe_like = abs(r - Q_s[a])  # |prediction error|
            # Gating: higher WM use when surprise is low (stable mapping), lower when high or set size is large
            set_size_term = 1.0 - (nS - 3.0) / 3.0  # 1 for nS=3, 0 for nS=6
            age_mod = 1.0 / (1.0 + np.exp(age_effect * (participant_age - 45.0)))

            gate_input = wm_weight_base + gating_sensitivity * (set_size_term - pe_like)
            wm_weight_eff = np.clip(1.0 / (1.0 + np.exp(-gate_input)) * age_mod, 0.0, 1.0)

            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update: strengthen chosen action on reward; on no-reward, do nothing beyond decay
            if r > 0.0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

        blocks_log_p += log_p

    return -float(blocks_log_p)
