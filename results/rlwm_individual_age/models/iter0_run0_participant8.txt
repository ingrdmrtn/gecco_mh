def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL + capacity-limited WM mixture with age- and set-size-modulated WM weighting.
    
    Parameters
    ----------
    states : array-like (int)
        State index on each trial.
    actions : array-like (int)
        Chosen action (0..2) on each trial.
    rewards : array-like (0/1)
        Binary feedback on each trial.
    blocks : array-like (int)
        Block index for each trial.
    set_sizes : array-like (int)
        Set size for each trial (3 or 6).
    age : array-like (float)
        Age in years (single-element array); used to determine age group (>=45 older).
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, wm_decay, wm_capacity, age_effect)
        - lr: RL learning rate (0..1).
        - wm_weight: baseline contribution of WM to choice (0..1).
        - softmax_beta: inverse temperature for RL policy; internally scaled by 10.
        - wm_decay: decay toward uniform in WM (0..1) per trial.
        - wm_capacity: capacity pivot controlling set-size attenuation of WM (in [3,6] ideally).
        - age_effect: multiplicative reduction of WM weight for older adults (>=45).
    
    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM policy
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM weight: reduced for larger set sizes and older age group
        # Set-size attenuation via logistic around wm_capacity pivot
        # For nS < wm_capacity -> near 1, for nS > wm_capacity -> reduced
        ss_att = 1.0 / (1.0 + np.exp(nS - wm_capacity))
        wm_weight_age = wm_weight * (1.0 - age_effect * is_older)
        wm_weight_eff = np.clip(wm_weight_age * ss_att, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (softmax, implemented as likelihood of chosen action)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WORKING MEMORY POLICY: near-deterministic softmax over WM weights
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform, then store rewarded action
            # 1) decay toward prior
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            # 2) if rewarded, encode strong association for (s,a)
            if r > 0.5:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0
            else:
                # mild suppression of the chosen action on errors
                w[s, a] = (1.0 - 0.5 * wm_decay) * w[s, a]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with separate gain/loss learning rates + WM with failure probability
    that increases with set size and age.
    
    Parameters
    ----------
    states : array-like (int)
        State index on each trial.
    actions : array-like (int)
        Chosen action (0..2) on each trial.
    rewards : array-like (0/1)
        Binary feedback on each trial.
    blocks : array-like (int)
        Block index for each trial.
    set_sizes : array-like (int)
        Set size for each trial (3 or 6).
    age : array-like (float)
        Age in years (single-element array); used to determine age group (>=45 older).
    model_parameters : tuple/list
        (lr_pos, wm_weight, softmax_beta, lr_neg, wm_fail_base, wm_fail_slope, age_effect)
        - lr_pos: RL learning rate for rewards.
        - wm_weight: baseline WM contribution.
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - lr_neg: RL learning rate for non-rewards.
        - wm_fail_base: base probability of WM failure at set size 3.
        - wm_fail_slope: increase in WM failure from set size 3 to 6.
        - age_effect: additive increase in WM failure for older adults (>=45).
    
    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    lr_pos, wm_weight, softmax_beta, lr_neg, wm_fail_base, wm_fail_slope, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM failure probability scales with set size and age
        # At set size 3 => wm_fail_base; at 6 => wm_fail_base + wm_fail_slope
        pfail = wm_fail_base + wm_fail_slope * max(0, nS - 3) + age_effect * is_older
        pfail = float(np.clip(pfail, 0.0, 1.0))

        # Effective WM weight: WM contributes only when it does not fail
        wm_weight_eff = wm_weight * (1.0 - pfail)
        wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax over WM; on failure, uniform choice
            p_wm_soft = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            p_wm_fail = 1.0 / nA
            p_wm = (1.0 - pfail) * p_wm_soft + pfail * p_wm_fail

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            if r > 0.5:
                delta = r - Q_s[a]
                q[s, a] += lr_pos * delta
            else:
                delta = r - Q_s[a]
                q[s, a] += lr_neg * delta

            # WM update: decay slightly, then store if rewarded
            # Light decay toward uniform (tied to failure base as a proxy for maintenance limits)
            decay = np.clip(0.25 * (wm_fail_base + 0.5 * wm_fail_slope), 0.0, 0.5)
            w = (1.0 - decay) * w + decay * w_0
            if r > 0.5:
                # Store the rewarded mapping strongly
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0
            else:
                # On errors, reduce confidence in the chosen action
                w[s, a] *= (1.0 - 0.5 * decay)

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with forgetting + WM capacity gradient and age-modulated RL forgetting.
    
    Parameters
    ----------
    states : array-like (int)
        State index on each trial.
    actions : array-like (int)
        Chosen action (0..2) on each trial.
    rewards : array-like (0/1)
        Binary feedback on each trial.
    blocks : array-like (int)
        Block index for each trial.
    set_sizes : array-like (int)
        Set size for each trial (3 or 6).
    age : array-like (float)
        Age in years (single-element array); used to determine age group (>=45 older).
    model_parameters : tuple/list
        (lr, wm_weight, softmax_beta, rl_forget, cap_slope, age_effect)
        - lr: RL learning rate.
        - wm_weight: baseline WM contribution.
        - softmax_beta: RL inverse temperature; internally scaled by 10.
        - rl_forget: RL forgetting rate toward uniform (0..1) each trial.
        - cap_slope: controls how fast WM weight decays with set size (exp decay).
        - age_effect: multiplicative increase of RL forgetting for older adults (>=45).
    
    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, rl_forget, cap_slope, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    is_older = 1.0 if age_val >= 45 else 0.0

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age-modulated RL forgetting
        rl_forget_eff = np.clip(rl_forget * (1.0 + age_effect * is_older), 0.0, 1.0)

        # WM weight decays exponentially with set size above 3
        wm_weight_eff = wm_weight * np.exp(-cap_slope * max(0, nS - 3))
        wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            log_p += np.log(p_total)

            # RL update with forgetting toward uniform prior
            delta = r - Q_s[a]
            q[s, a] += lr * delta
            # apply global forgetting toward uniform each trial (all states)
            q = (1.0 - rl_forget_eff) * q + rl_forget_eff * (1.0 / nA)

            # WM update: fast encoding, mild passive decay toward uniform with set-size dependence
            wm_decay = np.clip(0.1 + 0.15 * max(0, nS - 3), 0.0, 0.5)
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0.5:
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0
            else:
                # On errors, do not strengthen memory; small suppression of chosen action
                w[s, a] *= (1.0 - 0.3 * wm_decay)

        blocks_log_p += log_p

    return -blocks_log_p
