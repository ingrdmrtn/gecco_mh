def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with one-shot WM learning and age- and set-size-dependent WM reliance.

    Idea:
    - Choices are driven by a mixture of incremental RL and a fast-working-memory (WM) store.
    - WM learns quickly from rewarded outcomes, and reverts toward uniform on non-reward.
    - The arbitration weight on WM decreases with higher set size (greater load) and differs by age group.
      Younger adults (age < 45) rely more on WM than older adults.

    Parameters
    ----------
    model_parameters : array-like of length 6
        lr : RL learning rate (0..1)
        wm_bias : baseline logit for WM reliance (higher => more WM)
        softmax_beta : inverse temperature for RL policy (scaled internally)
        eta_wm : WM learning rate toward one-shot storage (0..1)
        setsize_slope : slope on (set_size - 3) for WM reliance (negative values reduce WM weight as set size increases)
        age_effect : additional logit shift if older (age >= 45). Positive values reduce WM in older if negative sign is used in formula below.

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, wm_bias, softmax_beta, eta_wm, setsize_slope, age_effect = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM
    age_val = age[0] if hasattr(age, "__len__") else age
    age_group = 1 if age_val >= 45 else 0  # 0=younger,1=older

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy probability of chosen action (stable softmax form)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy: softmax on WM table (nearly deterministic)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Arbitration: WM weight depends on set size and age group (older => lower WM if age_effect < 0)
            # logit(wm_weight) = wm_bias + setsize_slope*(nS-3) + age_effect*age_group
            wm_logit = wm_bias + setsize_slope * (nS - 3) + age_effect * age_group
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: one-shot update toward chosen action on reward; revert toward uniform otherwise
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * onehot
            else:
                # push WM toward uncertainty when outcome is not informative or incorrect
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning rates + capacity-limited WM with decay.
    WM is relied upon primarily when set size is within capacity; otherwise RL dominates.
    Age group reduces WM capacity/reliance and accelerates WM decay in older adults.

    Idea:
    - RL uses separate learning rates for positive vs. negative prediction errors.
    - WM stores recent correct action for each state, but decays toward uniform.
    - Arbitration weight emphasizes WM when set size <= effective capacity; older adults have lower effective WM reliance.

    Parameters
    ----------
    model_parameters : array-like of length 5
        lr_pos : RL learning rate for positive PE
        lr_neg : RL learning rate for negative PE
        wm_weight_base : baseline logit for WM reliance when within capacity
        decay_base : baseline logit for WM decay toward uniform (higher => more decay)
        softmax_beta : inverse temperature for RL policy (scaled internally)

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr_pos, lr_neg, wm_weight_base, decay_base, softmax_beta = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0] if hasattr(age, "__len__") else age
    age_group = 1 if age_val >= 45 else 0  # 0=younger,1=older

    # Effective capacity K is not an explicit parameter to save degrees of freedom.
    # We approximate capacity via a soft switch using set size in the WM weight and decay.
    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WM reliance increases when set size is small; older adults have lower WM reliance
            # logit(wm_weight) = wm_weight_base - 0.8*(nS-3) - 0.6*age_group
            wm_logit = wm_weight_base - 0.8 * (nS - 3) - 0.6 * age_group
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update with asymmetric learning rates
            pe = r - Q_s[a]
            lr_t = lr_pos if pe >= 0 else lr_neg
            q[s, a] += lr_t * pe

            # WM decay depends on set size and age: larger sets and older => faster decay
            # decay_logit -> decay in [0,1]
            decay_logit = decay_base + 0.6 * (nS - 3) + 0.5 * age_group
            decay = 1.0 / (1.0 + np.exp(-decay_logit))
            # Apply decay for the visited state
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # WM learning: if rewarded, move toward one-hot chosen; else slight push toward uniform
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                # fast consolidation when correct (fixed strong step to reflect capacity-limited, slot-like WM)
                w[s, :] = 0.3 * w[s, :] + 0.7 * onehot
            else:
                w[s, :] = 0.8 * w[s, :] + 0.2 * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    Uncertainty-based arbitration between RL and WM with WM decay and error-driven consolidation.
    WM reliance increases when WM is confident relative to RL; younger adults bias toward WM.

    Idea:
    - Compute RL confidence from RL softmax distribution concentration.
    - Compute WM confidence from the entropy of WM action weights.
    - Arbitration weight is a sigmoid of (WM_conf - RL_conf), with age and set-size terms.
    - WM decays toward uniform, more so for larger set sizes; learning consolidates on rewards.

    Parameters
    ----------
    model_parameters : array-like of length 6
        lr : RL learning rate
        softmax_beta : inverse temperature for RL (scaled internally)
        wm_eta : WM consolidation rate toward one-hot after reward (0..1)
        wm_decay : base logit for WM decay toward uniform (converted to [0,1])
        wm_bias : baseline logit for WM reliance
        age_weight : weight that increases WM reliance in younger adults (age < 45)

    Returns
    -------
    float
        Negative log-likelihood of observed choices.
    """
    lr, softmax_beta, wm_eta, wm_decay, wm_bias, age_weight = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0] if hasattr(age, "__len__") else age
    age_group = 1 if age_val >= 45 else 0  # 0=younger,1=older

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy (and full distribution for confidence)
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))
            # Full softmax distribution for RL confidence
            q_shift = Q_s - np.max(Q_s)
            prl_vec = np.exp(softmax_beta * q_shift)
            prl_vec = prl_vec / np.sum(prl_vec)
            rl_conf = np.max(prl_vec)  # concentration as confidence

            # WM policy and confidence via entropy
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))
            w_shift = W_s - np.max(W_s)
            pwm_vec = np.exp(softmax_beta_wm * w_shift)
            pwm_vec = pwm_vec / np.sum(pwm_vec)
            # entropy normalized to [0,1]
            entropy = -np.sum(pwm_vec * np.log(np.clip(pwm_vec, 1e-12, 1.0))) / np.log(nA)
            wm_conf = 1.0 - entropy

            # Arbitration weight: higher when WM is more confident than RL, boosted in younger, reduced for larger sets
            wm_logit = wm_bias + 2.0 * (wm_conf - rl_conf) + age_weight * (1 - age_group) - 0.3 * (nS - 3)
            wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = max(p_total, 1e-12)
            log_p += np.log(p_total)

            # RL update
            pe = r - Q_s[a]
            q[s, a] += lr * pe

            # WM decay increases with set size; base decay from wm_decay logit
            decay_base = 1.0 / (1.0 + np.exp(-wm_decay))
            decay_eff = decay_base + 0.05 * (nS - 3)
            decay_eff = np.clip(decay_eff, 0.0, 1.0)
            w[s, :] = (1.0 - decay_eff) * w[s, :] + decay_eff * w_0[s, :]

            # WM learning: consolidate on reward toward one-hot; otherwise drift to uniform
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - wm_eta) * w[s, :] + wm_eta * onehot
            else:
                w[s, :] = (1.0 - wm_eta) * w[s, :] + wm_eta * w_0[s, :]

        blocks_log_p += log_p

    return -blocks_log_p
