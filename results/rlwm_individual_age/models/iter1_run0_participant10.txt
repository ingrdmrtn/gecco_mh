Below are three standalone cognitive models that implement RL+WM accounts for this RLWM task. Each function follows the requested template, uses all parameters, returns the negative log-likelihood of the observed choices, and incorporates age group effects as specified.

Note: Assumes numpy as np is already imported.

def cognitive_model1(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM mixture with capacity- and age-gated WM, plus lapse.
    - The WM policy is near-deterministic over a working memory table that stores
      the last rewarded action for each state (with decay and corrective updates after errors).
    - The mixture weight assigned to WM is scaled down when set size exceeds capacity
      and for older adults.
    - RL uses a single learning rate and softmax choice rule.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Reward feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size per trial (e.g., 3 or 6), constant within a block.
    age : array-like (length 1)
        Participant age in years.
    model_parameters : list or array
        [lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse]
        - lr: RL learning rate (0..1)
        - wm_weight: baseline WM mixture weight (0..1)
        - softmax_beta: inverse temperature for RL (scaled by 10 internally)
        - wm_decay: per-trial decay of WM weights toward uniform (0..1)
        - wm_capacity: WM capacity parameter controlling set-size gating
        - lapse: lapse rate to uniform responding (0..1)

    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy
    age_val = age[0]
    older = age_val >= 45

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age and capacity-adjusted WM weight: large for small set sizes, reduced for older adults
        # cap_effect ~ 1 when nS << wm_capacity, ~ 0 when nS >> wm_capacity
        cap_effect = 1.0 / (1.0 + np.exp((nS - wm_capacity)))
        age_factor = 0.75 if older else 1.0
        wm_mix_base = np.clip(wm_weight * cap_effect * age_factor, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # Current state values
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM policy (softmax over WM weights)
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Mixture with lapse
            p_choice = (1.0 - lapse) * (wm_mix_base * p_wm + (1.0 - wm_mix_base) * p_rl) + lapse * (1.0 / nA)
            p_choice = np.clip(p_choice, 1e-12, 1.0 - 1e-12)
            log_p += np.log(p_choice)

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay and update:
            # - Decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            # - If rewarded, store one-hot for chosen action
            if r == 1.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot
            else:
                # If error, suppress the chosen action in WM and renormalize (guard against zero-sum)
                w[s, a] = 0.0
                ssum = np.sum(w[s, :])
                if ssum <= 1e-8:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= ssum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model2(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL with asymmetric learning rates + WM with binding-noise and set-size/age-dependent reliability.
    - RL uses separate learning rates for positive and negative outcomes.
    - WM stores last rewarded action per state (like a one-shot learner) with decay.
    - WM retrieval is noisy: a binding-noise parameter increases with set size and more strongly for older adults.
      The resulting WM policy is a mixture of a near-deterministic WM softmax and uniform responding.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Reward feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size per trial (e.g., 3 or 6), constant within a block.
    age : array-like (length 1)
        Participant age in years.
    model_parameters : list or array
        [alpha_pos, alpha_neg, wm_weight, softmax_beta, bind_noise, age_mod]
        - alpha_pos: RL learning rate for reward=1
        - alpha_neg: RL learning rate for reward=0
        - wm_weight: baseline WM weight in the mixture (0..1)
        - softmax_beta: RL inverse temperature (scaled by 10 internally)
        - bind_noise: scales WM binding noise with set size (>=0)
        - age_mod: additional WM noise for older adults (>=0)

    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    alpha_pos, alpha_neg, wm_weight, softmax_beta, bind_noise, age_mod = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    older = age_val >= 45

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Set-size and age-dependent WM binding noise:
        # epsilon increases with set size and is higher for older adults.
        base_eps = 1.0 - np.exp(-bind_noise * max(1, nS))
        age_eps = age_mod if older else 0.0
        epsilon_wm = np.clip(base_eps + age_eps, 0.0, 1.0)

        # Use binding noise also as a decay driver (more items -> more interference)
        wm_decay = np.clip(base_eps, 0.0, 1.0)

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM "clean" policy (deterministic softmax on WM weights)
            p_wm_clean = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # WM retrieval with binding noise -> mixture with uniform
            p_wm = (1.0 - epsilon_wm) * p_wm_clean + epsilon_wm * (1.0 / nA)

            # Mixture weight (constant across trials in block in this model)
            mix = np.clip(wm_weight, 0.0, 1.0)
            p_choice = np.clip(mix * p_wm + (1.0 - mix) * p_rl, 1e-12, 1.0 - 1e-12)
            log_p += np.log(p_choice)

            # RL update with asymmetric learning rates
            delta = r - Q_s[a]
            alpha = alpha_pos if r > 0.0 else alpha_neg
            q[s, a] += alpha * delta

            # WM decay and update
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r == 1.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot
            else:
                # error-driven suppression and renormalization
                w[s, a] = 0.0
                ssum = np.sum(w[s, :])
                if ssum <= 1e-8:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= ssum

        blocks_log_p += log_p

    return -blocks_log_p


def cognitive_model3(states, actions, rewards, blocks, set_sizes, age, model_parameters):
    """
    RL+WM arbitration with uncertainty-based gating and RL stickiness.
    - RL uses a single learning rate and includes a state-specific choice stickiness bias.
    - WM stores last rewarded action (with decay). The WM contribution is scaled by:
        (i) lower weight with larger set sizes,
        (ii) lower weight for older adults,
        (iii) higher weight when WM is more reliable (sharper WM distribution) than RL (higher RL uncertainty).
    - Arbitration weight is computed online at each trial from WM/RL uncertainty.

    Parameters
    ----------
    states : array-like of int
        State index on each trial (0..nS-1 within block).
    actions : array-like of int
        Chosen action on each trial (0..2).
    rewards : array-like of {0,1}
        Reward feedback on each trial.
    blocks : array-like of int
        Block index for each trial.
    set_sizes : array-like of int
        Set size per trial (e.g., 3 or 6), constant within a block.
    age : array-like (length 1)
        Participant age in years.
    model_parameters : list or array
        [lr, wm_base, softmax_beta, wm_decay, stickiness, age_slope]
        - lr: RL learning rate (0..1)
        - wm_base: baseline WM bias in arbitration (logit space)
        - softmax_beta: RL inverse temperature (scaled by 10 internally)
        - wm_decay: WM decay toward uniform (0..1)
        - stickiness: strength of within-state choice stickiness (>0 increases repetition)
        - age_slope: negative impact of age-group (older) on WM arbitration (<=0 reduces WM for older)

    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    lr, wm_base, softmax_beta, wm_decay, stickiness, age_slope = model_parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    age_val = age[0]
    older = age_val >= 45

    # helper for entropy
    def softmax_probs(x, beta):
        z = x - np.max(x)
        ez = np.exp(beta * z)
        return ez / np.sum(ez)

    def entropy(p):
        p = np.clip(p, 1e-12, 1.0)
        return -np.sum(p * np.log(p))

    blocks_log_p = 0.0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_actions = actions[block_mask]
        block_rewards = rewards[block_mask]
        block_states = states[block_mask]
        block_set_sizes = set_sizes[block_mask]

        nA = 3
        nS = int(block_set_sizes[0])

        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        # stickiness bias per state-action (decays implicitly by not reinforcing)
        bias = np.zeros((nS, nA))

        # Set-size penalty for WM; more items -> lower WM arbitration
        size_penalty = (nS - 3)  # 0 at set size 3, positive at 6
        age_term = age_slope if older else 0.0  # reduces WM if age_slope < 0

        log_p = 0.0
        for t in range(len(block_states)):
            a = int(block_actions[t])
            s = int(block_states[t])
            r = float(block_rewards[t])

            # RL effective values include stickiness bias for this state
            Q_s = q[s, :] + stickiness * bias[s, :]
            W_s = w[s, :]

            # RL probability of chosen action
            p_rl = 1.0 / np.sum(np.exp(softmax_beta * (Q_s - Q_s[a])))

            # WM probability of chosen action
            p_wm = 1.0 / np.sum(np.exp(softmax_beta_wm * (W_s - W_s[a])))

            # Compute uncertainty terms
            p_rl_vec = softmax_probs(Q_s, softmax_beta)
            p_wm_vec = softmax_probs(W_s, softmax_beta_wm)
            H_rl = entropy(p_rl_vec)
            H_wm = entropy(p_wm_vec)
            wm_reliability = (np.max(W_s) - np.partition(W_s, -2)[-2])  # separation between top two
            rl_uncertainty = H_rl  # higher entropy -> more uncertain RL

            # Arbitration weight in logit space; constants scale terms
            # Larger set size and older age reduce WM weight; higher WM reliability and higher RL uncertainty increase it.
            logit_wm = (
                wm_base
                - 0.8 * size_penalty
                + 1.0 * (wm_reliability - H_wm)  # sharper WM -> larger positive
                + 0.7 * rl_uncertainty
                + age_term
            )
            wm_mix = 1.0 / (1.0 + np.exp(-logit_wm))
            wm_mix = np.clip(wm_mix, 0.0, 1.0)

            p_choice = np.clip(wm_mix * p_wm + (1.0 - wm_mix) * p_rl, 1e-12, 1.0 - 1e-12)
            log_p += np.log(p_choice)

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # Update stickiness bias (increase bias for chosen action in this state; decay others slightly)
            bias[s, :] *= 0.9
            bias[s, a] += 1.0

            # WM decay and update
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r == 1.0:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot
            else:
                w[s, a] = 0.0
                ssum = np.sum(w[s, :])
                if ssum <= 1e-8:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= ssum

        blocks_log_p += log_p

    return -blocks_log_p
