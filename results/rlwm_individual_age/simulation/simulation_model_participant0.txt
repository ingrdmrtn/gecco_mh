import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with one-shot WM learning and age- and set-size-dependent WM reliance.

    Inputs:
    - stimulus: array of states per trial (ints, 0..nS-1 within a block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size (nS) per trial (constant within block)
    - correct_answer: array of correct action (0..nA-1) per trial
    - age: participant's age (float or int)
    - parameters: [lr, wm_bias, softmax_beta, eta_wm, setsize_slope, age_effect]

    Returns:
    - simulated_actions: array of chosen actions per trial (ints 0..2)
    - simulated_rewards: array of rewards per trial (0/1)
    """
    lr, wm_bias, softmax_beta, eta_wm, setsize_slope, age_effect = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Age group: 0 = younger (<45), 1 = older (>=45)
    age_group = 1 if (age if not hasattr(age, "__len__") else age[0]) >= 45 else 0

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        # Set size for this block (assumed constant within block)
        nS = int(set_sizes[block_idx[0]])
        nA = 3

        # Build a state -> correct action mapping using first occurrence per state
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find the first trial in the block for this state
            mask_s = np.where(block_states == s)[0]
            if len(mask_s) > 0:
                correct_actions[s] = block_correct[mask_s[0]]
            else:
                # Fallback if state not present (shouldn't happen if set_sizes is consistent)
                correct_actions[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Arbitration weight for this block (depends on set size and age)
        wm_logit = wm_bias + setsize_slope * (nS - 3) + age_effect * age_group
        wm_weight = 1.0 / (1.0 + np.exp(-wm_logit))

        for t in range(len(block_states)):
            s = int(block_states[t])
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action for this state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: one-shot toward one-hot on reward; revert toward uniform on no-reward
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * onehot
            else:
                w[s, :] = (1.0 - eta_wm) * w[s, :] + eta_wm * w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
