def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with capacity- and age-gated WM, plus lapse.
    Simulates choices and rewards following the logic of cognitive_model1.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant's age in years (float)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse]

    Returns:
    - simulated_actions: array of chosen actions (0..2)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM policy

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    age_val = float(age)
    older = age_val >= 45

    tr_global = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        nS = int(block_set_sizes[0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Mixture gating by capacity and age (constant within block)
        cap_effect = 1.0 / (1.0 + np.exp((nS - wm_capacity)))
        age_factor = 0.75 if older else 1.0
        wm_mix_base = np.clip(wm_weight * cap_effect * age_factor, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])
            correct_a = int(block_correct[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax policy
            q_shift = Q_s - np.max(Q_s)
            p_rl = np.exp(softmax_beta * q_shift)
            p_rl /= np.sum(p_rl)

            # WM softmax policy (near-deterministic over WM table)
            w_shift = W_s - np.max(W_s)
            p_wm = np.exp(softmax_beta_wm * w_shift)
            p_wm /= np.sum(p_wm)

            # Mixture with lapse
            p_total = (1.0 - lapse) * (wm_mix_base * p_wm + (1.0 - wm_mix_base) * p_rl) + lapse * (1.0 / nA)
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update (delta rule)
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update based on feedback
            if r == 1:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = onehot
            else:
                # zero out the chosen action; renormalize or reset if empty
                w[s, a] = 0.0
                ssum = np.sum(w[s, :])
                if ssum <= 1e-8:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= ssum

            tr_global += 1

    return simulated_actions, simulated_rewards
