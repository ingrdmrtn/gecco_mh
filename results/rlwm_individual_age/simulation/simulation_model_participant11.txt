def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using RL+WM mixture with capacity/load- and age-adjusted WM weight and WM decay.

    Parameters:
    - stimulus: array of state IDs per trial (assumed 0..nS-1 within a block)
    - blocks: array of block IDs per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action per trial (assumed fixed per state within a block)
    - age: participant's age (float or array-like with one float)
    - parameters: [lr, wm_weight_base, softmax_beta, wm_decay, load_sensitivity, age_effect]

    Returns:
    - simulated_actions: array of sampled actions per trial (ints)
    - simulated_rewards: array of binary rewards per trial (ints)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, wm_decay, load_sensitivity, age_effect = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # highly deterministic WM readout

    # Handle age input (float or array-like)
    if isinstance(age, (list, tuple, np.ndarray)):
        age_val = float(age[0])
    else:
        age_val = float(age)
    is_older = 1.0 if age_val >= 45.0 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        block_states = stimulus[block_indices].astype(int)
        block_correct = correct_answer[block_indices].astype(int)
        # Use provided set size for this block (assumed constant within block)
        nS = int(set_sizes[block_indices][0])
        nA = 3

        # Build state -> correct action mapping (assumed stationary within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # Find first occurrence of this state in the block and take its correct action
            idxs = np.where(block_states == s)[0]
            if len(idxs) > 0:
                correct_actions[s] = block_correct[idxs[0]]
            else:
                # If a state index is not present (shouldn't happen if nS matches),
                # default to a random action to avoid index errors.
                correct_actions[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute arbitration factors that depend only on set size and age (constant within block)
        load_factor = 1.0 / (1.0 + load_sensitivity * max(0, nS - 3))
        age_factor = 1.0 / (1.0 + age_effect * is_older)
        wm_weight_eff = float(np.clip(wm_weight_base * load_factor * age_factor, 0.0, 1.0))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Get current values
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax
            Q_center = Q_s - np.max(Q_s)
            exp_Q = np.exp(softmax_beta * Q_center)
            p_rl = exp_Q / np.sum(exp_Q)

            # WM softmax (sharper)
            W_center = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * W_center)
            p_wm = exp_W / np.sum(exp_W)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Deliver reward based on correct action for this state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform for the current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM one-shot overwrite on rewarded trials
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * one_hot

            # Normalize WM row to sum to 1 (safety)
            row_sum = np.sum(w[s, :])
            if row_sum > 0:
                w[s, :] /= row_sum

            tr += 1

    return simulated_actions, simulated_rewards
