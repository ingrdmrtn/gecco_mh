def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL+WM mixture model with capacity-gated WM access
    and decay-based WM maintenance.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (constant within block; e.g., 3 or 6)
    - correct_answer: array of correct actions per trial (0..2)
    - age: scalar age in years (or length-1 array)
    - parameters: [lr, wm_weight, softmax_beta, phi_decay, K_capacity, age_bias]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, phi_decay, K_capacity, age_bias = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    # Handle age scalar/array and age group
    age_val = age
    is_older = age_val >= 45

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_set_sizes = set_sizes[block_idx]
        block_correct = correct_answer[block_idx]

        nA = 3
        # Set size for this block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Build mapping from state to correct action in this block
        # Assumes states are 0..nS-1 within block
        correct_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            s_mask = (block_states == s)
            # Use the first occurrence's correct action
            correct_by_state[s] = int(block_correct[s_mask][0])

        # Initialize RL and WM tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity gate and effective WM weight (constant within block)
        K_eff = K_capacity - age_bias if is_older else K_capacity + age_bias
        K_eff = np.clip(K_eff, 1.0, 6.0)
        SS = float(nS)
        gate = 1.0 / (1.0 + np.exp(SS - K_eff))
        wm_weight_eff = float(np.clip(wm_weight * gate, 0.0, 1.0))

        # Iterate through trials within the block
        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Compute RL and WM choice probabilities

            exp_Q = np.exp(softmax_beta * Q_s)
            p_rl = exp_Q / np.sum(exp_Q)

            exp_W = np.exp(softmax_beta_wm * W_s)
            p_wm = exp_W / np.sum(exp_W)

            # Mixture policy
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Generate reward based on correct action for the state
            correct_a = correct_by_state[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM maintenance/encoding with decay
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - phi_decay) * w[s, :] + phi_decay * one_hot
            else:
                w[s, :] = (1.0 - phi_decay) * w[s, :] + phi_decay * w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
