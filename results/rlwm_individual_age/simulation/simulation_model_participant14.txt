def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM with capacity limit (simulation)

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial's block (3 or 6)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant age (float)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, K_param]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, K_param = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    is_older = 1.0 if age >= 45 else 0.0

    # Map K_param to [1,6] and adjust for age
    K_cont = 1.0 + 5.0 * (1.0 / (1.0 + np.exp(-K_param)))  # in [1,6]
    K_eff = float(np.clip(K_cont - 1.0 * is_older, 1.0, 6.0))
    K_cap = int(np.floor(K_eff))

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    tr = 0

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        nS = int(set_sizes[block_idx[0]])

        # Determine WM eligibility based on first-appearance order up to capacity
        wm_eligible = np.zeros(nS, dtype=bool)
        seen_order = []
        for t in range(len(block_states)):
            s_all = int(block_states[t])
            if s_all not in seen_order:
                seen_order.append(s_all)
                if np.sum(wm_eligible) < K_cap:
                    wm_eligible[s_all] = True

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Trial loop within block
        for t in range(len(block_states)):
            s = int(block_states[t])

            # Compute RL policy
            Q_s = q[s, :]
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # Compute WM policy
            if wm_eligible[s]:
                W_s = w[s, :]
                p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
                p_wm /= np.sum(p_wm)
            else:
                p_wm = np.ones(nA) / nA

            # Mixture policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay (for all states, but we only learn on eligible states)
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM learning on rewarded trials for eligible states
            if wm_eligible[s] and r > 0.5:
                learn = 0.7
                w[s, :] = (1.0 - learn) * w[s, :]
                w[s, a] += learn

            tr += 1

    return simulated_actions, simulated_rewards
