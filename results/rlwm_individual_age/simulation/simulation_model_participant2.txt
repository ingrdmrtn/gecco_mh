def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from the RL+WM model with reward-gated WM encoding
    and set size/age-dependent WM engagement (matching cognitive_model2).

    Inputs:
    - stimulus: array of state/stimulus identifiers per trial (can be arbitrary labels per block)
    - blocks: array of block id per trial
    - set_sizes: array of set size per trial (used to scale WM precision/engagement)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant age (float or length-1 array)
    - parameters: [lr, wm_weight, softmax_beta, gate, beta_wm_base, decay]

    Returns:
    - simulated_actions: array of sampled actions (0..2) per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, gate, beta_wm_base, decay = parameters
    softmax_beta *= 10.0

    # Age handling
    try:
        age_val = float(age[0]) if hasattr(age, "__len__") else float(age)
    except Exception:
        age_val = float(age)
    younger = 1 if age_val < 45 else 0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        # Localize state ids to 0..nS_unique-1 for indexing
        unique_states = np.unique(block_states_raw)
        state_to_local = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_local[s] for s in block_states_raw], dtype=int)

        nA = 3
        nS_unique = len(unique_states)
        # Use provided set size to scale WM engagement/precision (as in fitting)
        nS_scale = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else nS_unique

        # Build per-state correct action (assumed stable within a block)
        correct_actions = np.zeros(nS_unique, dtype=int)
        for s_local in range(nS_unique):
            mask = (block_states == s_local)
            # Take the first occurrence's correct action
            correct_actions[s_local] = int(block_correct[mask][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS_unique, nA))
        w = (1.0 / nA) * np.ones((nS_unique, nA))
        w_0 = (1.0 / nA) * np.ones((nS_unique, nA))

        for t in range(len(block_states)):
            s = block_states[t]
            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            logits_rl = softmax_beta * Q_s
            max_lr = np.max(logits_rl)
            p_rl = np.exp(logits_rl - max_lr)
            p_rl /= np.sum(p_rl)

            # WM precision and engagement scaling
            size_scale = 3.0 / max(1.0, float(nS_scale))
            age_scale_beta = 1.0 if younger == 1 else 0.7
            beta_wm = beta_wm_base * size_scale * age_scale_beta

            logits_wm = beta_wm * W_s
            max_wm = np.max(logits_wm)
            p_wm = np.exp(logits_wm - max_wm)
            p_wm /= np.sum(p_wm)

            gate_eff = np.clip(wm_weight * gate * size_scale * (1.0 if younger == 1 else 0.85), 0.0, 1.0)

            p_total = gate_eff * p_wm + (1.0 - gate_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward from correct mapping
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform each trial
            w[s, :] = (1.0 - decay) * w[s, :] + decay * w_0[s, :]

            # Reward-gated WM encoding: only after rewarded trials, attract further to one-hot
            if r > 0.5:
                onehot = np.zeros(nA)
                onehot[a] = 1.0
                w[s, :] = (1.0 - decay) * w[s, :] + decay * onehot

            tr += 1

    return simulated_actions, simulated_rewards
