def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using cognitive_model2:
    Uncertainty-gated WM + RL.

    Inputs:
    - stimulus: array of state IDs per trial
    - blocks: array of block IDs per trial
    - set_sizes: array of set sizes per trial (not strictly required; inferred from unique states per block)
    - correct_answer: array of correct actions per trial
    - age: participant age (float or length-1 array)
    - parameters: [lr, wm_base, softmax_beta, wm_decay, gate_temp, age_bias]

    Returns:
    - simulated_actions: array of sampled actions
    - simulated_rewards: array of binary rewards
    """
    import numpy as np

    lr, wm_base, softmax_beta, wm_decay, gate_temp, age_bias = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Handle age input
    if isinstance(age, (list, tuple, np.ndarray)):
        age_val = float(age[0])
    else:
        age_val = float(age)

    if age_val < 45:
        age_term = +abs(age_bias)
    else:
        age_term = -abs(age_bias)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = np.array(stimulus[block_idx], dtype=int)
        block_correct = np.array(correct_answer[block_idx], dtype=int)

        # Remap states in this block to 0..nS-1 to index arrays safely
        unique_states = np.unique(block_states_raw)
        nS = len(unique_states)
        state_map = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_map[s] for s in block_states_raw], dtype=int)

        # Determine correct action per state (assume stationary mapping within block)
        nA = 3
        correct_per_state = np.zeros(nS, dtype=int)
        for s_raw in unique_states:
            s_idx = state_map[s_raw]
            first_occ = np.where(block_states_raw == s_raw)[0][0]
            correct_per_state[s_idx] = int(block_correct[first_occ])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = block_states[t]
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            logits_rl = softmax_beta * (Q_s - np.max(Q_s))
            exp_rl = np.exp(logits_rl)
            p_rl = exp_rl / np.sum(exp_rl)

            # RL entropy and normalized entropy
            entropy = -np.sum(p_rl * np.log(np.clip(p_rl, 1e-12, 1.0)))
            H_norm = entropy / np.log(nA)

            # WM policy
            logits_wm = softmax_beta_wm * (W_s - np.max(W_s))
            exp_wm = np.exp(logits_wm)
            p_wm = exp_wm / np.sum(exp_wm)

            # Load normalization (as in fitting code)
            load_norm = (float(nS) - 3.0) / (6.0 - 3.0) if nS >= 3 else 0.0

            # Gate and mixture weight
            gate_input = H_norm - load_norm + age_term
            gate = 1.0 / (1.0 + np.exp(-gate_temp * gate_input))
            wmix = np.clip(wm_base * gate, 0.0, 1.0)

            # Mixture policy
            p_total = wmix * p_wm + (1.0 - wmix) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action mapping
            r = 1 if a == int(correct_per_state[s]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform and encoding on rewarded trials
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0:
                enc = np.clip(H_norm, 0.0, 1.0)
                w[s, :] = (1.0 - enc) * w[s, :]
                w[s, a] += enc

            tr += 1

    return simulated_actions, simulated_rewards
