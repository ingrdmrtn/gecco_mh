def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using:
    RL with forgetting + reward-gated WM and uncertainty-based arbitration, with age-dependent noise.

    Inputs:
    - stimulus: array of state indices per trial within each block (0..nS-1)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (e.g., 3 or 6); constant within each block
    - correct_answer: array of correct action indices per trial (0..2)
    - age: participant's age (float or int)
    - parameters: [lr, wm_weight, softmax_beta, rl_forget, wm_decay, lapse]

    Returns:
    - simulated_actions: array of simulated actions (0..2) per trial
    - simulated_rewards: array of simulated binary rewards (0/1) per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, rl_forget, wm_decay, lapse = parameters

    age_val = float(age)
    is_old = age_val >= 45
    beta_scale = 0.7 if is_old else 1.0
    lapse_scale = 1.5 if is_old else 1.0

    softmax_beta = (softmax_beta * 10.0) * beta_scale
    softmax_beta_wm = 50.0  # near-deterministic WM policy
    lapse_eff_base = np.clip(lapse * lapse_scale, 0.0, 1.0)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3
    eps = 1e-12
    tr = 0

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        nS = int(set_sizes[block_idx][0])  # set size is constant within block

        # Build correct action lookup per state (assume stationary mapping within block)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            state_mask = (block_states == s)
            # Fallback if state not present yet (should not happen in typical designs)
            if np.any(state_mask):
                correct_actions[s] = block_correct[state_mask][0]
            else:
                # default to 0 if unseen (won't influence if state never appears)
                correct_actions[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Arbitration size factor: 1 at nS=3, 0 at nS=6
        size_factor = 1.0 - max(0.0, (nS - 3) / 3.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL global forgetting toward uniform
            q = (1.0 - rl_forget) * q + rl_forget * (1.0 / nA)

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL policy
            x_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (near-deterministic)
            x_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(x_wm)
            p_wm /= np.sum(p_wm)

            # Arbitration weight
            wm_certainty = float(np.max(W_s))  # in [~1/nA, 1]
            wm_w_eff = np.clip(wm_weight * size_factor * wm_certainty, 0.0, 1.0)

            # Mix policies and apply lapse
            p_mix = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_mix = np.clip(p_mix, eps, 1.0)
            p_mix /= np.sum(p_mix)

            lapse_eff = lapse_eff_base
            p_total = (1.0 - lapse_eff) * p_mix + lapse_eff * (1.0 / nA)
            p_total = np.clip(p_total, eps, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome based on correct action mapping
            correct_a = int(correct_actions[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: reward-gated, otherwise decay toward uniform
            if r > 0.5:
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * target
            else:
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            tr += 1

    return simulated_actions, simulated_rewards
