def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using:
    Exponential-decay Working Memory with load- and age-modulated arbitration.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block ids per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant age (float or length-1 array)
    - parameters: (lr, wm_weight, softmax_beta, wm_decay_tau, gamma)

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay_tau, gamma = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Age scaling (older adults have reduced WM influence)
    age_val = age[0] if isinstance(age, (list, tuple, np.ndarray)) else age
    age_scale = 0.75 if age_val >= 45 else 1.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3

    # Iterate block-wise to reinitialize latent states per block
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        nS = int(set_sizes[block_idx][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Time since last successful WM storage for each state
        time_since_update = np.full(nS, np.inf)

        # Arbitration parameters depending on set size and age
        load_scale = (3.0 / float(nS)) ** gamma
        eta = wm_weight * load_scale * age_scale
        eta = min(max(eta, 0.0), 1.0)

        for i, tr in enumerate(block_idx):
            s = int(block_states[i])

            # Apply WM decay for this state based on time since last successful store
            dt = time_since_update[s]
            if np.isfinite(dt):
                d = 1.0 - np.exp(-max(dt, 0.0) / max(wm_decay_tau, 1e-6))
                w[s, :] = (1.0 - d) * w[s, :] + d * w_0[s, :]

            # Compute RL and WM policies
            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax for RL
            x_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(x_rl)
            p_rl /= np.sum(p_rl)

            # Softmax for WM (very sharp)
            x_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(x_wm)
            p_wm /= np.sum(p_wm)

            # Mixture policy
            p_total = eta * p_wm + (1.0 - eta) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct answer for this trial
            r = 1 if a == int(block_correct[i]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update and time tracking
            if r > 0:
                # Store deterministic association
                w[s, :] = 0.0
                w[s, a] = 1.0
                time_since_update[s] = 0.0
            else:
                # Increment counter for this state (only when visited and not rewarded)
                if np.isfinite(time_since_update[s]):
                    time_since_update[s] += 1.0
                else:
                    time_since_update[s] = 1.0

    return simulated_actions, simulated_rewards
