import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + Capacity-limited Working Memory (age-sensitive) simulation

    Parameters:
    - stimulus: array of state indices for each trial
    - blocks: array of block ids for each trial
    - set_sizes: array giving the set size for the block on each trial
    - correct_answer: array of correct action (0..2) for each trial
    - age: participant's age (float or array-like)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_capacity, old_penalty]

    Returns:
    - simulated_actions: simulated action choices (array of ints)
    - simulated_rewards: simulated binary rewards (array of ints)
    """
    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, old_penalty = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    nA = 3

    # Handle age scalar/array
    age_val = age[0] if hasattr(age, "__len__") and len(np.atleast_1d(age)) > 0 else age
    is_old = 1.0 if age_val >= 45 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate blocks
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        # Determine set size for block (assumed constant within block)
        nS = int(set_sizes[block_idx][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # WM availability based on capacity and age
        K_eff = max(0.0, wm_capacity * (1.0 - old_penalty * is_old))
        avail = min(1.0, K_eff / max(1.0, nS))
        uniform = np.ones(nA) / nA

        for t_local, tr in enumerate(block_idx):
            s = int(block_states[t_local])
            # Clip state index to avoid out-of-bounds if inputs are unexpected
            s = int(np.clip(s, 0, nS - 1))

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            logits_rl = softmax_beta * Q_s
            logits_rl -= np.max(logits_rl)
            p_rl = np.exp(logits_rl)
            p_rl /= np.sum(p_rl)

            # WM policy (memory-based, high inverse temperature)
            logits_wm = softmax_beta_wm * W_s
            logits_wm -= np.max(logits_wm)
            p_wm_mem = np.exp(logits_wm)
            p_wm_mem /= np.sum(p_wm_mem)

            # Availability-weighted WM policy (mixture with uniform when unavailable)
            p_wm = avail * p_wm_mem + (1.0 - avail) * uniform

            # Final choice policy
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action for current state
            correct_a = int(block_correct[t_local])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform, then store if rewarded
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = 0.0
                w[s, a] = 1.0

    return simulated_actions, simulated_rewards
