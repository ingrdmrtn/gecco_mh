def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate actions and rewards for the RL+WM model with probabilistic WM storage
    modulated by set size and age (matching cognitive_model3 logic).

    Inputs:
    - stimulus: array of state indices per trial (ints, 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size for each trial (constant within block)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant age (float or array-like with single value)
    - parameters: [lr, softmax_beta, wm_weight_base, store_prob_base, wm_decay, age_slope]

    Returns:
    - simulated_actions: array of chosen actions (0..2) per trial
    - simulated_rewards: array of binary rewards (0/1) per trial
    """
    import numpy as np

    lr, softmax_beta, wm_weight_base, store_prob_base, wm_decay, age_slope = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Handle age being float or array
    age_val = float(age[0]) if hasattr(age, "__len__") and len(np.atleast_1d(age)) > 0 else float(age)
    is_older = 1.0 if age_val >= 45 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    eps = 1e-12
    tr = 0

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx].astype(int)

        nA = 3
        # Set size is constant within a block
        nS = int(np.unique(block_set_sizes)[0])

        # Map each state to its correct action within the block
        # Assumes consistency: each state has a single correct action
        correct_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of this state within the block
            mask_s = (block_states == s)
            if np.any(mask_s):
                correct_by_state[s] = int(block_correct[mask_s][0])
            else:
                # If state not encountered, default to 0 (won't be used)
                correct_by_state[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            # Effective mixture weight and storage prob depend on set size and age
            ss = float(nS)
            ss_factor = 3.0 / ss  # 1 at set size 3, 0.5 at set size 6
            age_factor = (1.0 - age_slope * is_older)

            wm_weight_eff = np.clip(wm_weight_base * ss_factor * age_factor, 0.0, 1.0)
            store_prob_eff = np.clip(store_prob_base * ss_factor * age_factor, 0.0, 1.0)

            # Action probabilities from RL and WM
            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax for RL
            x_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(x_rl)
            p_rl /= np.maximum(np.sum(p_rl), eps)

            # Softmax for WM (very deterministic)
            x_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(x_wm)
            p_wm /= np.maximum(np.sum(p_wm), eps)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, eps, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome based on correct action for current state
            r = 1 if a == int(correct_by_state[s]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # Probabilistic WM storage on rewarded trials
            if r > 0:
                if np.random.rand() < store_prob_eff:
                    one_hot = np.zeros(nA)
                    one_hot[a] = 1.0
                    w[s, :] = one_hot
                # else: keep decayed w[s,:] (no additional store)

            tr += 1

    return simulated_actions, simulated_rewards
