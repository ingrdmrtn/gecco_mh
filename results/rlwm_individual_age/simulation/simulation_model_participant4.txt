def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with capacity penalty and age-modulated WM reliance (simulation).

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within a block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block; typically 3 or 6)
    - correct_answer: array of correct action per trial (0..2)
    - age: participant age (float or length-1 array); younger if <45
    - parameters: (lr, wm_weight, softmax_beta, wm_decay, wm_penalty)

    Returns:
    - simulated_actions: array of sampled actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_penalty = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Allow age to be float or array-like with single value
    if np.ndim(age) > 0:
        age_val = float(np.ravel(age)[0])
    else:
        age_val = float(age)

    is_younger = 1.0 if age_val < 45.0 else 0.0
    age_boost = 1.0 + 0.25 * is_younger  # up to +25% WM reliance for younger adults

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Determine number of unique states in block (or use provided set size)
        # Use provided set size to be consistent with capacity manipulation
        nS = int(block_set_sizes[0])

        # Map each state to its correct action (assumes consistent mapping within block)
        unique_states = np.unique(block_states)
        # If states are 0..nS-1 this is straightforward; create a lookup of length nS
        correct_actions_per_state = np.zeros(nS, dtype=int)
        for s in unique_states:
            # take first occurrence's correct action
            s_mask = (block_states == s)
            correct_actions_per_state[s] = int(block_correct[np.where(s_mask)[0][0]])

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity penalty for large set size
        size_penalty = 1.0
        if nS > 3:
            size_penalty = (1.0 - wm_penalty)

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy (nearly deterministic)
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Effective WM weight with age boost and capacity penalty
            wm_eff = wm_weight * age_boost * size_penalty
            wm_eff = float(np.clip(wm_eff, 0.0, 1.0))

            # Mixture policy
            p_total = wm_eff * p_wm + (1.0 - wm_eff) * p_rl
            # Numerical guard
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome based on correct action for this state
            correct_a = int(correct_actions_per_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, strengthen memory for chosen action (overwrite-like step)
            if r > 0:
                w[s, :] = (1.0 - wm_decay) * w_0[s, :]
                w[s, a] += wm_decay

            tr += 1

    return simulated_actions, simulated_rewards
