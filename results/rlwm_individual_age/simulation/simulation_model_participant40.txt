def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the RL+WM mixture with capacity- and age-adjusted WM, and WM decay.

    Inputs:
    - stimulus: array of state identities per trial (integers starting at 0 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (used to determine number of states in each block)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant's age (float)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, capacity_K, age_bias]

    Returns:
    - simulated_actions: array of chosen actions per trial (0..2)
    - simulated_rewards: array of rewards per trial (0 or 1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, capacity_K, age_bias = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    is_older = age >= 45

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Follow fitting code: use provided set size for the block
        nS = int(block_set_sizes[0])

        # Map each state in this block to its correct action
        # Assume states are labeled 0..nS-1 within block
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            inds = np.where(block_states == s)[0]
            if len(inds) > 0:
                state_to_correct[s] = block_correct[inds[0]]
            else:
                # If a state does not appear (edge case), default to 0
                state_to_correct[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age modulation
        if is_older:
            wm_weight_age = wm_weight * (1.0 - age_bias)
        else:
            wm_weight_age = wm_weight * (1.0 + age_bias)
        wm_weight_age = np.clip(wm_weight_age, 0.0, 1.0)

        # Capacity modulation: down-weight WM when set size exceeds capacity
        capacity_factor = min(1.0, max(0.0, capacity_K) / max(1.0, nS))
        wm_weight_eff_block = wm_weight_age * capacity_factor

        # Simulate trials within the block in temporal order
        for t_local, tr in enumerate(block_idx):
            s = int(block_states[t_local])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM policies (softmax)
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture policy with effective WM weight
            p_total = wm_weight_eff_block * p_wm + (1.0 - wm_weight_eff_block) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Determine reward from correct mapping
            correct_a = state_to_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for the current stimulus row
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM storage on rewarded trials: reset row to uniform then set chosen to 1
            if r > 0.5:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0

    return simulated_actions, simulated_rewards
