def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using the RL+WM mixture model with capacity limit,
    age-adjusted WM efficacy, WM decay/forgetting, and lapses.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within block)
    - blocks: array of block index per trial (states reset each block)
    - set_sizes: array of set size per trial (e.g., 3 or 6), constant within block
    - correct_answer: array of correct action per trial (0,1,2)
    - age: participant's age (float)
    - parameters: [lr, wm_weight, softmax_beta, K, wm_decay, lapse]

    Returns:
    - simulated_actions: array of chosen actions (0,1,2)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, K, wm_decay, lapse = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM readout

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Age adjustment of effective capacity
    age_norm = np.clip((age - 45.0) / 45.0, 0.0, 1.0)
    K_eff_factor = 1.0 - 0.5 * age_norm  # 1.0 for <=45y, down to 0.5 for >=90y


    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_set_sizes = set_sizes[block_idx]
        block_correct = correct_answer[block_idx]

        nA = 3
        # Use provided set size per block (matching the fitting code's assumption)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action in this block
        # Assumes one correct action per state in the block
        state_correct = np.zeros(nS, dtype=int)
        for st in range(nS):
            inds = np.where(block_states == st)[0]
            if len(inds) > 0:
                state_correct[st] = block_correct[inds[0]]
            else:
                # Fallback if a state is missing by chance (shouldn't happen)
                state_correct[st] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity- and age-adjusted WM mixture weight for this block
        cap_ratio = np.clip((K * K_eff_factor) / max(nS, 1), 0.0, 1.0)
        wm_mix_block = np.clip(wm_weight * cap_ratio, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL and WM choice probabilities
            Q_s = q[s, :]
            W_s = w[s, :]


            exp_Q = np.exp(softmax_beta * Q_s)
            p_rl = exp_Q / np.sum(exp_Q)

            W_centered = W_s - np.max(W_s)
            exp_W = np.exp(softmax_beta_wm * W_centered)
            p_wm = exp_W / np.sum(exp_W)

            # Mixture and lapse
            p_mix = wm_mix_block * p_wm + (1.0 - wm_mix_block) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (1.0 / nA)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Reward from environment
            correct_a = state_correct[s]
            r = 1 if a == correct_a else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform prior (light decay every trial for visited state)
            w[s, :] = (1.0 - 0.25 * wm_decay) * w[s, :] + (0.25 * wm_decay) * w_0[s, :]

            # WM update based on feedback
            if r > 0.5:
                # Push towards one-hot at chosen action
                target = np.zeros(nA)
                target[a] = 1.0
                w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * target
            else:
                # Mild push away from incorrect action toward other actions equally
                err_push = np.full(nA, 1.0 / (nA - 1))
                err_push[a] = 0.0
                w[s, :] = (1.0 - 0.25 * wm_decay) * w[s, :] + (0.25 * wm_decay) * err_push

            tr += 1

    return simulated_actions, simulated_rewards
