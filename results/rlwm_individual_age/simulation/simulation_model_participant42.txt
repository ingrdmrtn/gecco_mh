def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL + WM with decay, limited capacity, and state-specific perseveration (stickiness) bias.

    Parameters:
    - stimulus: state identifiers per trial (array of int; assumed 0..nS-1 within a block)
    - blocks: block index for each trial (array)
    - set_sizes: set size for each trial (array; typically constant within block)
    - correct_answer: correct action per trial (array of int in 0..2)
    - age: participant's age (float or array-like with first element used)
    - parameters: [lr, wm_weight, softmax_beta, phi, k_base, stickiness]

    Returns:
    - simulated_actions: simulated action choices (array of int)
    - simulated_rewards: simulated binary rewards (array of int)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, phi, k_base, stickiness = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    age_val = age[0] if isinstance(age, (list, tuple, np.ndarray)) else age
    is_older = age_val >= 45
    capacity_scale = 0.6 if is_older else 1.0
    stickiness_eff = stickiness * (1.3 if is_older else 1.0)

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    for b in np.unique(blocks):
        block_indices = np.where(blocks == b)[0]
        # Maintain original trial order within the block
        block_indices.sort()
        block_states = stimulus[block_indices]
        block_correct = correct_answer[block_indices]
        block_set_sizes = set_sizes[block_indices]

        nA = 3
        # Use declared set size for the block (assumed consistent within block)
        nS = int(block_set_sizes[0]) if len(block_set_sizes) > 0 else int(np.max(block_states) + 1)

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))
        last_action = -1 * np.ones(nS, dtype=int)

        for idx in block_indices:
            s = int(stimulus[idx])
            r_correct = int(correct_answer[idx])
            nS_current = int(set_sizes[idx])

            # RL policy
            Q_s = q[s, :]
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)



            # WM policy with stickiness bias in logits
            wm_logits = w[s, :].copy()
            if last_action[s] >= 0:
                wm_logits[last_action[s]] += stickiness_eff

            p_wm_soft = np.exp(softmax_beta_wm * (wm_logits - np.max(wm_logits)))
            p_wm_soft /= np.sum(p_wm_soft)



            # Capacity-limited WM mixture with uniform
            k_eff = max(0.0, capacity_scale * k_base)
            if k_eff >= nS_current:
                cap_factor = 1.0
            else:
                cap_factor = k_eff / max(1.0, nS_current)
            p_wm = cap_factor * p_wm_soft + (1.0 - cap_factor) * (1.0 / nA)

            # Mixture of WM and RL
            p_total = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total = p_total / p_total.sum()

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[idx] = a

            # Outcome
            r = 1 if a == r_correct else 0
            simulated_rewards[idx] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay
            w = (1.0 - phi) * w + phi * w_0
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot
            else:
                w[s, a] = max(0.0, w[s, a] - 0.5)
                row_sum = max(w[s, :].sum(), 1e-12)
                w[s, :] = w[s, :] / row_sum
                w[s, :] = 0.5 * w[s, :] + 0.5 * w_0[s, :]

            # Update perseveration memory
            last_action[s] = a

            tr_global += 1

    return simulated_actions, simulated_rewards
