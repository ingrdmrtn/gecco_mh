def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards from the RL+WM mixture with capacity- and age-modulated WM arbitration
    and one-shot WM learning (matching cognitive_model1).

    Inputs:
    - stimulus: array of state indices per trial (integers, 0..set_size-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within block)
    - correct_answer: array of correct action per trial (0,1,2), consistent within state and block
    - age: participant age (float)
    - parameters: list [lr, wm_weight, softmax_beta, wm_alpha, K, age_old_factor]

    Returns:
    - simulated_actions: array of sampled actions per trial (0,1,2)
    - simulated_rewards: array of rewards per trial (0 or 1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_alpha, K, age_old_factor = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    age_val = float(age)
    is_old = 1.0 if age_val >= 45.0 else 0.0

    tr_global = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Determine number of states in this block
        if len(block_set_sizes) > 0:
            nS = int(block_set_sizes[0])
        else:
            nS = int(len(np.unique(block_states)))

        # Build mapping from state -> correct action (assume consistent within block)
        state_to_correct = np.zeros(nS, dtype=int)
        for s in range(nS):
            # Find trials within the block for this state
            mask_s = (block_states == s)
            # Fallback if state not present (should not happen): choose 0
            if not np.any(mask_s):
                state_to_correct[s] = 0
            else:
                state_to_correct[s] = int(block_correct[mask_s][0])

        # Initialize value tables
        q = (1.0 / nA) * np.ones((nS, nA))   # RL values
        w = (1.0 / nA) * np.ones((nS, nA))   # WM values
        w_0 = (1.0 / nA) * np.ones((nS, nA)) # WM prior (uniform)

        for t in range(len(block_states)):
            tr = block_idx[t]
            s = int(block_states[t])
            r_correct = state_to_correct[s]
            nS_now = float(block_set_sizes[t]) if len(block_set_sizes) > 0 else float(nS)

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL and WM choice distributions

            exp_Q = np.exp(softmax_beta * Q_s)
            p_rl = exp_Q / np.sum(exp_Q)

            exp_W = np.exp(softmax_beta_wm * W_s)
            p_wm = exp_W / np.sum(exp_W)

            # Arbitration: capacity and age modulation
            set_factor = min(1.0, max(0.0, K / max(nS_now, 1.0)))
            age_factor = (1.0 - is_old) + is_old * max(0.0, min(1.0, age_old_factor))
            wm_w_eff = max(0.0, min(1.0, wm_weight * set_factor * age_factor))

            p_total = wm_w_eff * p_wm + (1.0 - wm_w_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome based on correctness
            r = 1 if a == r_correct else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM update: decay toward uniform prior, then one-shot toward observed outcome
            w[s, :] = (1.0 - wm_alpha) * w[s, :] + wm_alpha * w_0[s, :]
            w[s, a] = (1.0 - wm_alpha) * w[s, a] + wm_alpha * r

            tr_global += 1

    return simulated_actions, simulated_rewards
