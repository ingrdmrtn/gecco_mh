def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with WM capacity, decay, and age-dependent WM engagement (simulation).

    Parameters:
    - stimulus: sequence of stimuli for each trial (array-like). Within a block, stimuli map to states.
    - blocks: block index for each trial (array-like)
    - set_sizes: block-dependent set sizes on each trial (array-like)
    - correct_answer: correct action on each trial (array-like)
    - age: participant's age (float)
    - parameters: list/tuple of [lr, wm_weight, softmax_beta, phi, K, age_sensitivity]

    Returns:
    - simulated_actions: simulated action choices (array of int)
    - simulated_rewards: simulated binary rewards (array of int)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, phi, K, age_sensitivity = parameters

    # Match fitting code scalings
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    age_center = 45.0
    age_factor = 1.0 / (1.0 + np.exp(age_sensitivity * (age - age_center)))

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3

    # Iterate blocks in order of appearance
    unique_blocks = np.unique(blocks)
    for b in unique_blocks:
        block_idx = np.where(blocks == b)[0]
        block_stim = np.array(stimulus)[block_idx]
        block_correct = np.array(correct_answer)[block_idx]
        block_set_sizes = np.array(set_sizes)[block_idx]

        # Determine number of states from block set size (as in fitting code)
        nS = int(block_set_sizes[0])

        # Map stimuli within block to state indices 0..nS-1 (order of first appearance)
        stim_to_state = {}
        state_counter = 0
        for st in block_stim:
            if st not in stim_to_state:
                stim_to_state[st] = state_counter
                state_counter += 1
                if state_counter >= nS:
                    # Stop once capacity reached per set size (robust if extras present)
                    break

        # Ensure all stimuli in the block have an assigned state (fallback if nS underestimated)
        for st in block_stim:
            if st not in stim_to_state:
                stim_to_state[st] = min(len(stim_to_state), nS - 1)

        # Build correct action per state using first occurrence for each stimulus/state
        correct_actions_by_state = np.zeros(nS, dtype=int)
        seen_state = np.zeros(nS, dtype=bool)
        for t_local, st in enumerate(block_stim):
            s = stim_to_state[st]
            if not seen_state[s]:
                correct_actions_by_state[s] = int(block_correct[t_local])
                seen_state[s] = True
            if np.all(seen_state):
                break

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity factor for this block
        cap_factor = min(1.0, max(0.0, K / float(nS)))
        wm_weight_eff = wm_weight * cap_factor * age_factor
        wm_weight_eff = min(max(wm_weight_eff, 0.0), 1.0)

        # Simulate trials within the block in original order
        for idx_in_block, tr in enumerate(block_idx):
            st = stimulus[tr]
            s = stim_to_state[st]

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            z_rl = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(z_rl)
            p_rl /= np.sum(p_rl)

            # WM softmax (near-deterministic)
            z_wm = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(z_wm)
            p_wm /= np.sum(p_wm)

            # Mixture policy with effective WM weight
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Determine reward from correct action
            correct_a = int(correct_actions_by_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform
            w = (1.0 - phi) * w + phi * w_0

            # WM update based on feedback
            if r >= 0.5:
                # Encode correct mapping strongly
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Suppress chosen incorrect action; distribute mass over others
                w[s, a] = 0.0
                others = [aa for aa in range(nA)]
                # Correct list of other actions
                others = [aa for aa in range(nA) if aa != a]
                for aa in others:
                    w[s, aa] = 1.0 / (nA - 1)

    return simulated_actions, simulated_rewards
