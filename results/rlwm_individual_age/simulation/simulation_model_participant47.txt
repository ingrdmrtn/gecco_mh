def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with age- and capacity-modulated WM contribution (simulation).

    Inputs:
    - stimulus: array of trial-wise state indices (can be arbitrary labels within a block)
    - blocks: array of trial-wise block indices
    - set_sizes: array of trial-wise set size (constant within a block)
    - correct_answer: array of trial-wise correct action (0..2)
    - age: participant age (float or length-1 array-like)
    - parameters: [lr, wm_weight_base, softmax_beta, wm_decay, k_capacity]

    Returns:
    - simulated_actions: array of sampled actions (0..2)
    - simulated_rewards: array of binary rewards (0/1)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, wm_decay, k_capacity = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM readout

    # Handle age input
    try:
        age_val = float(age[0]) if hasattr(age, "__len__") else float(age)
    except Exception:
        age_val = float(age)
    is_older = age_val >= 45

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states_raw = stimulus[block_idx]
        block_set_size = int(set_sizes[block_idx][0])  # constant within block
        block_correct = correct_answer[block_idx]

        # Build a mapping from raw stimulus labels to 0..nS-1
        unique_states = np.unique(block_states_raw)
        # If provided set size disagrees with unique states, trust unique states for mapping, but use set size for WM weight
        state_to_int = {s: i for i, s in enumerate(unique_states)}
        nS_internal = len(unique_states)
        nA = 3

        # Extract per-state correct action (assume stationary mapping within block)
        correct_by_state = np.zeros(nS_internal, dtype=int)
        for s_raw in unique_states:
            s_mask = (block_states_raw == s_raw)
            # take the first occurrence's correct action
            correct_by_state[state_to_int[s_raw]] = int(block_correct[s_mask][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS_internal, nA))
        w = (1.0 / nA) * np.ones((nS_internal, nA))
        w_0 = (1.0 / nA) * np.ones((nS_internal, nA))

        # Precompute mixture weight factors that are constant within block
        cap_factor = min(1.0, max(0.0, k_capacity) / max(1.0, float(block_set_size)))
        age_factor = 0.65 if is_older else 1.0
        wm_weight_eff = np.clip(wm_weight_base * age_factor * cap_factor, 0.0, 1.0)

        for t, global_t in enumerate(block_idx):
            s_raw = block_states_raw[t]
            s = state_to_int[s_raw]

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # RL softmax
            prl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            prl /= np.sum(prl)

            # WM softmax
            pwm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            pwm /= np.sum(pwm)

            # Mixture
            p_total = wm_weight_eff * pwm + (1.0 - wm_weight_eff) * prl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[global_t] = a

            # Determine reward from ground-truth rule
            correct_a = int(correct_by_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[global_t] = r

            # RL update (delta rule)
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for the current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM update on rewarded trials, mirroring fitting logic
            if r > 0.5:
                # Additional attraction to chosen action and renormalize
                w[s, :] = (1.0 - wm_decay) * w[s, :]
                w[s, a] += wm_decay
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])
            else:
                # Just renormalize after decay
                w[s, :] = np.maximum(w[s, :], 1e-12)
                w[s, :] /= np.sum(w[s, :])

            tr += 1

    return simulated_actions, simulated_rewards
