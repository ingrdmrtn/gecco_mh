def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with capacity- and age-gated WM, plus global decay and lapse.

    Inputs:
    - stimulus: array of state indices per trial (0..set_size-1 within a block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size for each trial (consistent within block)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant age (float or array-like)
    - parameters: (lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse)

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of binary rewards per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_capacity, lapse = parameters
    softmax_beta *= 10.0  # RL beta has a higher upper bound
    softmax_beta_wm = 50.0  # WM is very deterministic when available

    # Handle age input (float or array-like)
    try:
        age_val = float(age[0])
    except (TypeError, IndexError):
        age_val = float(age)
    older = 1.0 if age_val >= 45 else 0.0

    n_trials = len(stimulus)
    nA = 3

    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        # Determine set size for this block
        nS = int(block_set_sizes[0])

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective WM capacity with age penalty
        K_eff = max(1.0, wm_capacity - older * 1.0)

        for t in range(len(block_states)):
            idx = block_idx[t]
            s = int(block_states[t])
            correct_a = int(block_correct[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl = p_rl / np.sum(p_rl)

            # WM availability gating by set size vs capacity
            wm_available = 1.0 if float(nS) <= K_eff else 0.0

            if wm_available > 0.5:
                p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
                p_wm = p_wm / np.sum(p_wm)
            else:
                p_wm = np.ones(nA) / nA

            # Mixture and lapse
            p_mix = wm_weight * p_wm + (1.0 - wm_weight) * p_rl
            p_total = (1.0 - lapse) * p_mix + lapse * (np.ones(nA) / nA)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[idx] = a

            # Simulate reward from correct action
            r = 1 if a == correct_a else 0
            simulated_rewards[idx] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # Global WM decay toward uniform
            w = wm_decay * w + (1.0 - wm_decay) * w_0

            # WM item-specific update
            if r > 0.5:
                # Store rewarded S-A pair as one-hot
                w[s, :] = 0.0
                w[s, a] = 1.0
            else:
                # Penalize chosen action's WM trace (as in fitting code)
                w[s, a] *= (1.0 - 0.5 * (1.0 - wm_decay))

            tr += 1

    return simulated_actions, simulated_rewards
