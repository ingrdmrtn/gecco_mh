def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RLâ€“WM mixture with capacity-limited WM and age-dependent capacity reduction.

    Inputs:
    - stimulus: array of state indices per trial (ints starting at 0)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (used to determine nS within each block)
    - correct_answer: array of correct action per trial (0..nA-1)
    - age: participant age (float or length-1 array)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, K_base, K_old_drop]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, K_base, K_old_drop = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # very deterministic WM

    # Determine age group
    try:
        is_old = 1.0 if float(np.array(age).ravel()[0]) >= 45 else 0.0
    except Exception:
        is_old = 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    nA = 3  # number of actions

    # Iterate over blocks
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.array(stimulus[block_idx], dtype=int)
        block_set_sizes = np.array(set_sizes[block_idx], dtype=int)
        block_correct = np.array(correct_answer[block_idx], dtype=int)

        # Determine number of states in this block (as used in fitting)
        nS = int(block_set_sizes[0])

        # Map each state (0..nS-1) to its correct action using first occurrence
        state_to_correct = np.zeros(nS, dtype=int)
        seen = np.zeros(nS, dtype=bool)
        for t_local, s in enumerate(block_states):
            if 0 <= s < nS and not seen[s]:
                state_to_correct[s] = block_correct[t_local]
                seen[s] = True
        # For any unseen states (edge cases), default to first action
        if not np.all(seen):
            missing_states = np.where(~seen)[0]
            for s in missing_states:
                state_to_correct[s] = 0

        # Initialize RL and WM values
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Age-adjusted WM gate based on effective capacity vs set size
        K_eff = max(0.0, K_base - is_old * K_old_drop)
        cap_gate = min(1.0, K_eff / max(1.0, nS))
        gate = float(np.clip(wm_weight * cap_gate, 0.0, 1.0))

        # Simulate trials in this block
        for t_local, tr in enumerate(block_idx):
            s = int(block_states[t_local])

            # Compute policy from RL and WM
            Q_s = q[s, :]
            W_s = w[s, :]

            exp_Q = np.exp(softmax_beta * Q_s)
            p_rl = exp_Q / np.sum(exp_Q)

            exp_W = np.exp(softmax_beta_wm * W_s)
            p_wm = exp_W / np.sum(exp_W)


            p_total = gate * p_wm + (1.0 - gate) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            correct_a = int(state_to_correct[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for this state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # If rewarded, store perfect memory for this item-action
            if r > 0.5:
                w[s, :] = w_0[s, :]
                w[s, a] = 1.0

    return simulated_actions, simulated_rewards
