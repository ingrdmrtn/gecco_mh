import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using:
    - RL with valence-asymmetric learning (separate lr for pos/neg PEs)
    - WM with interference (set-size and age dependent) and recency/bleed dynamics

    Inputs:
    - stimulus: array of state indices per trial (assumed 0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set size per trial (constant within a block)
    - correct_answer: array of correct action (0..2) per trial
    - age: participant age (float or array-like with one value)
    - parameters: [lr_pos, lr_neg, wm_weight, softmax_beta, wm_decay, interference]

    Returns:
    - simulated_actions: array of chosen actions per trial
    - simulated_rewards: array of rewards (0/1) per trial
    """
    lr_pos, lr_neg, wm_weight, softmax_beta, wm_decay, interference = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Resolve age and age factor
    if isinstance(age, (list, tuple, np.ndarray)):
        age_val = float(age[0])
    else:
        age_val = float(age)
    age_interference_factor = 1.5 if age_val > 45 else 1.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    # Iterate over blocks
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx]
        block_correct = correct_answer[block_idx]
        block_set_sizes = set_sizes[block_idx]

        nA = 3
        # Assume states are indexed 0..nS-1 within block
        nS = int(block_set_sizes[0])

        # Build per-state correct action mapping (assumes consistency within state)
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of this state in block
            sel = np.where(block_states == s)[0]
            if len(sel) == 0:
                # if state not present (edge case), default to 0
                correct_actions[s] = 0
            else:
                correct_actions[s] = int(block_correct[sel[0]])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Compute block-level WM mixture weight with interference and age
        inter = interference * (nS - 1) * age_interference_factor
        wm_weight_block = np.clip(wm_weight * (1.0 / (1.0 + inter)), 0.0, 1.0)

        # Step through trials of this block in temporal order
        for local_t, tr in enumerate(block_idx):
            s = int(block_states[local_t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # RL policy
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl /= np.sum(p_rl)

            # WM policy
            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight_block * p_wm + (1.0 - wm_weight_block) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Reward based on correct action for the state
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[tr] = r

            # RL update (valence-asymmetric)
            pe = r - Q_s[a]
            if pe >= 0:
                q[s, a] += lr_pos * pe
            else:
                q[s, a] += lr_neg * pe

            # WM recency decay toward uniform
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM interference bleed for larger set sizes
            if nS > 1:
                bleed = np.clip(inter / (inter + 1.0), 0.0, 0.9)
                w[s, :] = (1.0 - bleed) * w[s, :] + bleed * w_0[s, :]

            # WM reinforcement/suppression based on outcome
            if r > 0:
                strengthen = 0.8
                w[s, :] = (1.0 - strengthen) * w[s, :]
                w[s, a] += strengthen
                w[s, :] /= np.sum(w[s, :])
            else:
                suppress = 0.1
                w[s, a] = max(0.0, w[s, a] - suppress)
                row_sum = np.sum(w[s, :])
                if row_sum <= 0:
                    w[s, :] = w_0[s, :]
                else:
                    w[s, :] /= row_sum

    return simulated_actions, simulated_rewards
