import numpy as np

def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL+WM mixture model with capacity- and age-modulated WM,
    WM decay, and softmax decision.

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within a block)
    - blocks: array of block index per trial
    - set_sizes: array of set size per trial (should be constant within block)
    - correct_answer: array of correct action index per trial (0..2)
    - age: participant's age (float or array-like; if array, age[0] is used)
    - parameters: (lr, wm_weight, softmax_beta, wm_decay, capacity_C, age_effect)

    Returns:
    - simulated_actions: array of simulated action choices
    - simulated_rewards: array of simulated binary rewards
    """
    lr, wm_weight, softmax_beta, wm_decay, capacity_C, age_effect = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0  # near-deterministic WM policy

    # Handle age input being scalar or array-like
    participant_age = float(age[0]) if hasattr(age, "__len__") else float(age)
    # Age modulation (older reduces WM contribution as age_effect > 0)
    age_mod = 1.0 / (1.0 + np.exp(age_effect * (participant_age - 45.0)))

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)


    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        nA = 3
        # Set size is constant within block
        nS = int(set_sizes[block_idx][0])

        # Capacity factor (down-weight WM when set size exceeds capacity)
        capacity_factor = 1.0 if nS <= capacity_C else max(0.0, float(capacity_C) / float(nS))
        wm_weight_eff = np.clip(wm_weight * capacity_factor * age_mod, 0.0, 1.0)

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        for t in range(len(block_states)):
            s = int(block_states[t])

            Q_s = q[s, :].copy()
            W_s = w[s, :].copy()

            # Policies
            exp_Q = np.exp(softmax_beta * Q_s)
            p_rl = exp_Q / np.sum(exp_Q)

            exp_W = np.exp(softmax_beta_wm * W_s)
            p_wm = exp_W / np.sum(exp_W)


            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = np.clip(p_total, 1e-12, 1.0)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr] = a

            # Outcome
            r = 1 if a == int(block_correct[t]) else 0
            simulated_rewards[tr] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform, then overwrite on rewarded trials
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = one_hot

            tr += 1

    return simulated_actions, simulated_rewards
