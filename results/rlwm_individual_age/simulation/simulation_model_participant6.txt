def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RLâ€“WM arbitration by uncertainty with age bias and action perseveration (simulation).

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block index per trial
    - set_sizes: array of set size per trial (e.g., 3 or 6)
    - correct_answer: array of correct action per trial (0..2)
    - age: participant age (float)
    - parameters: [lr, wm_weight, softmax_beta, age_bias_wm, perseveration, wm_noise]

    Returns:
    - simulated_actions: array of simulated actions per trial
    - simulated_rewards: array of simulated binary rewards per trial
    """
    import numpy as np

    lr, wm_weight_param, softmax_beta, age_bias_wm, perseveration, wm_noise = parameters

    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    is_older = 1 if age >= 45 else 0
    eps = 1e-12
    decay = 0.05  # WM decay toward uniform each trial

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr_global = 0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        idx = np.where(block_mask)[0]

        block_states = np.array(stimulus)[idx].astype(int)
        block_correct = np.array(correct_answer)[idx].astype(int)
        block_set_sizes = np.array(set_sizes)[idx]

        nA = 3
        # Use the provided set size for the block (assumed constant within block)
        nS = int(block_set_sizes[0])

        # Map each state to its correct action within the block (assumed stable)
        correct_actions_by_state = np.zeros(nS, dtype=int)
        for s in range(nS):
            # find first occurrence of state s in the block
            inds = np.where(block_states == s)[0]
            if len(inds) > 0:
                correct_actions_by_state[s] = block_correct[inds[0]]
            else:
                # Fallback (should not happen): default to 0
                correct_actions_by_state[s] = 0

        # Initialize RL and WM value tables
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        prev_action = -1 * np.ones(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with perseveration on last action in this state
            Q_s = q[s, :].copy()
            if prev_action[s] >= 0:
                Q_s[prev_action[s]] += perseveration

            Qs_shift = Q_s - np.max(Q_s)
            exp_rl = np.exp(softmax_beta * Qs_shift)
            probs_rl = exp_rl / np.sum(exp_rl)
            probs_rl = np.clip(probs_rl, eps, 1.0)
            probs_rl /= np.sum(probs_rl)

            # WM decay toward uniform baseline
            w = (1.0 - decay) * w + decay * w_0

            # WM policy with partial perseveration and noise-controlled inverse temperature
            W_s = w[s, :].copy()
            if prev_action[s] >= 0:
                W_s[prev_action[s]] += 0.25 * perseveration

            W_shift = W_s - np.max(W_s)
            beta_wm_eff = softmax_beta_wm / (1.0 + wm_noise)
            exp_wm = np.exp(beta_wm_eff * W_shift)
            probs_wm = exp_wm / np.sum(exp_wm)
            probs_wm = np.clip(probs_wm, eps, 1.0)
            probs_wm /= np.sum(probs_wm)

            # Arbitration by relative uncertainty (entropy), with age bias and intercept
            H_rl = -np.sum(probs_rl * np.log(np.clip(probs_rl, eps, 1.0)))
            H_wm = -np.sum(probs_wm * np.log(np.clip(probs_wm, eps, 1.0)))

            age_term = (1.0 - is_older) - is_older  # +1 if younger, -1 if older
            arb_input = wm_weight_param + (H_rl - H_wm) + age_bias_wm * age_term
            wm_weight_eff = 1.0 / (1.0 + np.exp(-arb_input))
            wm_weight_eff = float(np.clip(wm_weight_eff, 0.0, 1.0))

            # Mixture policy and action sampling
            p_total = wm_weight_eff * probs_wm + (1.0 - wm_weight_eff) * probs_rl
            p_total = np.clip(p_total, eps, 1.0)
            p_total /= np.sum(p_total)

            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[idx[t]] = a

            # Outcome based on correct mapping
            correct_action = int(correct_actions_by_state[s])
            r = 1 if a == correct_action else 0
            simulated_rewards[idx[t]] = r

            # RL update with asymmetric learning rates (lr for positive, 1-lr for negative)
            delta = r - q[s, a]
            if delta >= 0:
                q[s, a] += lr * delta
            else:
                q[s, a] += (1.0 - lr) * delta

            # WM update: strengthen on correct, otherwise drift toward uniform for chosen action
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                w[s, :] = (1.0 - lr) * w[s, :] + lr * one_hot
            else:
                w[s, a] = (1.0 - lr) * w[s, a] + lr * (1.0 / nA)

            # Update perseveration memory
            prev_action[s] = a

            tr_global += 1

    return simulated_actions, simulated_rewards
