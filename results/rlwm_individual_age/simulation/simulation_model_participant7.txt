def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    RL+WM mixture with capacity-limited WM and age-sensitive gating (simulation).

    Inputs:
    - stimulus: array of state indices per trial (0..nS-1 within each block)
    - blocks: array of block indices per trial
    - set_sizes: array of set sizes per trial (3 or 6)
    - correct_answer: array of correct action per trial (0..2)
    - age: participant's age (float or length-1 array-like)
    - parameters: [lr, wm_weight, softmax_beta, wm_decay, wm_noise]

    Returns:
    - simulated_actions: array of chosen actions (int)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, wm_decay, wm_noise = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Handle age input shape
    age_val = age[0] if hasattr(age, "__len__") and len(np.atleast_1d(age)) > 0 else float(age)
    is_older = 1 if age_val >= 45 else 0
    K_young, K_old = 4.0, 2.5

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = stimulus[block_idx].astype(int)
        block_correct = correct_answer[block_idx].astype(int)
        nA = 3

        # Determine nS for this block from provided set_sizes (consistent with fitting)
        nS = int(np.unique(set_sizes[block_idx])[0])

        # Build state -> correct action mapping within block
        # Assumes each state in the block has a consistent correct action
        correct_map = np.zeros(nS, dtype=int)
        for s in range(nS):
            # take the first occurrence's correct action
            mask_s = (block_states == s)
            if not np.any(mask_s):
                # if a state index doesn't appear (edge case), default to 0
                correct_map[s] = 0
            else:
                correct_map[s] = int(block_correct[mask_s][0])

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Capacity scaling by age and set size
        K = K_old if is_older else K_young
        cap_scale = min(1.0, K / float(nS))
        wm_weight_eff_block = np.clip(wm_weight * cap_scale, 0.0, 1.0)

        for t in range(len(block_states)):
            s = int(block_states[t])
            Q_s = q[s, :]
            W_s = w[s, :]

            # RL softmax
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            p_rl = np.exp(rl_logits)
            p_rl /= np.sum(p_rl)

            # WM softmax (near-deterministic)
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            p_wm = np.exp(wm_logits)
            p_wm /= np.sum(p_wm)

            # Mixture
            p_total = wm_weight_eff_block * p_wm + (1.0 - wm_weight_eff_block) * p_rl
            p_total = np.maximum(p_total, 1e-12)
            p_total /= np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[tr + t] = a

            # Reward based on correct mapping
            r = 1 if a == correct_map[s] else 0
            simulated_rewards[tr + t] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # WM decay toward uniform for current state
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]

            # WM encoding on reward with noise
            if r > 0.5:
                one_hot = np.zeros(nA)
                one_hot[a] = 1.0
                encoded = (1.0 - wm_noise) * one_hot + wm_noise * (1.0 / nA)
                w[s, :] = encoded

        tr += len(block_states)

    return simulated_actions, simulated_rewards
