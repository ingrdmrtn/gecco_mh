def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards using RL with forgetting + WM capacity gradient
    and age-modulated RL forgetting (matching cognitive_model3).

    Inputs:
    - stimulus: array of state indices (can be any integers; remapped per block)
    - blocks: array of block indices for each trial
    - set_sizes: array of set sizes for each trial (per block; first in block used)
    - correct_answer: array of correct action (0..2) for each trial
    - age: float age in years (or array-like with one element)
    - parameters: (lr, wm_weight, softmax_beta, rl_forget, cap_slope, age_effect)

    Returns:
    - simulated_actions: array of chosen actions (0..2)
    - simulated_rewards: array of rewards (0/1)
    """
    import numpy as np

    lr, wm_weight, softmax_beta, rl_forget, cap_slope, age_effect = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0

    # Handle age input being float or array-like
    try:
        age_val = float(age[0]) if hasattr(age, "__len__") else float(age)
    except Exception:
        age_val = float(age)
    is_older = 1.0 if age_val >= 45 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    tr = 0
    for b in np.unique(blocks):
        block_mask = (blocks == b)
        block_indices = np.where(block_mask)[0]
        block_states_raw = np.array(stimulus)[block_mask]
        block_correct = np.array(correct_answer)[block_mask]

        # Map states within the block to 0..nS-1 to index matrices safely
        unique_states = np.unique(block_states_raw)
        state_to_idx = {s: i for i, s in enumerate(unique_states)}
        block_states = np.array([state_to_idx[s] for s in block_states_raw])
        nS = int(set_sizes[block_mask][0]) if np.any(block_mask) else len(unique_states)
        nS = max(nS, len(unique_states))  # ensure capacity for all observed states

        # Determine per-state correct action (use first occurrence)
        nA = 3
        correct_per_state = np.zeros(nS, dtype=int)
        # Initialize with a default (0), then fill for observed states
        for s_raw, s_idx, ca in zip(block_states_raw, block_states, block_correct):
            if s_idx not in []:  # just iterate; first assignment kept by checking fill status
                correct_per_state[s_idx] = ca
        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))
        w = (1.0 / nA) * np.ones((nS, nA))
        w_0 = (1.0 / nA) * np.ones((nS, nA))

        # Effective parameters for this block
        rl_forget_eff = float(np.clip(rl_forget * (1.0 + age_effect * is_older), 0.0, 1.0))
        wm_weight_eff = float(np.clip(wm_weight * np.exp(-cap_slope * max(0, nS - 3)), 0.0, 1.0))
        wm_decay = float(np.clip(0.1 + 0.15 * max(0, nS - 3), 0.0, 0.5))

        for t in range(len(block_indices)):
            s = int(block_states[t])

            Q_s = q[s, :]
            W_s = w[s, :]

            # Softmax policies
            p_rl = np.exp(softmax_beta * (Q_s - np.max(Q_s)))
            p_rl = p_rl / np.sum(p_rl)

            p_wm = np.exp(softmax_beta_wm * (W_s - np.max(W_s)))
            p_wm = p_wm / np.sum(p_wm)

            # Mixture
            p_total = wm_weight_eff * p_wm + (1.0 - wm_weight_eff) * p_rl
            p_total = p_total / np.sum(p_total)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_indices[t]] = a

            # Reward based on correct action
            correct_a = int(correct_per_state[s])
            r = 1 if a == correct_a else 0
            simulated_rewards[block_indices[t]] = r

            # RL update
            delta = r - Q_s[a]
            q[s, a] += lr * delta

            # RL forgetting toward uniform
            q = (1.0 - rl_forget_eff) * q + rl_forget_eff * (1.0 / nA)

            # WM decay and update
            w = (1.0 - wm_decay) * w + wm_decay * w_0
            if r > 0.5:
                # Store perfect memory for rewarded pair
                w[s, :] = w_0[s, :].copy()
                w[s, a] = 1.0
            else:
                # Slight suppression for the chosen, unrewarded action
                w[s, a] *= (1.0 - 0.3 * wm_decay)

            tr += 1

    return simulated_actions, simulated_rewards
