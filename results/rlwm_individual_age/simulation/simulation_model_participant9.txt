def simulate_model(stimulus, blocks, set_sizes, correct_answer, age, parameters):
    """
    Simulate choices and rewards for the RL+WM arbitration-by-uncertainty model
    with age bias and stickiness.

    Inputs:
    - stimulus: array of states per trial (ints)
    - blocks: array of block indices per trial (ints)
    - set_sizes: array of set sizes per trial (ints) [block-constant]
    - correct_answer: array of correct action per trial (ints in {0,1,2})
    - age: participant's age (float)
    - parameters: [lr, wm_weight_base, softmax_beta, wm_decay, age_bias, stickiness]

    Returns:
    - simulated_actions: array of chosen actions per trial (ints)
    - simulated_rewards: array of rewards per trial (0/1)
    """
    import numpy as np

    lr, wm_weight_base, softmax_beta, wm_decay, age_bias, stickiness = parameters
    softmax_beta *= 10.0
    softmax_beta_wm = 50.0
    is_older = 1.0 if age >= 45 else 0.0

    n_trials = len(stimulus)
    simulated_actions = np.zeros(n_trials, dtype=int)
    simulated_rewards = np.zeros(n_trials, dtype=int)

    eps = 1e-12
    tr = 0

    for b in np.unique(blocks):
        block_idx = np.where(blocks == b)[0]
        block_states = np.asarray(stimulus)[block_idx].astype(int)
        block_correct = np.asarray(correct_answer)[block_idx].astype(int)
        block_set_sizes = np.asarray(set_sizes)[block_idx].astype(int)

        nA = 3
        # Follow fitting code: nS is the block set size (not necessarily number of unique observed states)
        nS = int(block_set_sizes[0])

        # Build a mapping from state id (0..nS-1) to its correct action in this block
        # Assumes states are labeled 0..nS-1 and have a stable correct action within a block
        correct_actions = np.zeros(nS, dtype=int)
        for s in range(nS):
            inds = np.where(block_states == s)[0]
            if len(inds) > 0:
                correct_actions[s] = int(block_correct[inds[0]])
            else:
                # Fallback in case a state is not presented (rare); assign a default
                correct_actions[s] = 0

        # Initialize RL and WM
        q = (1.0 / nA) * np.ones((nS, nA))   # RL Q-values
        w = (1.0 / nA) * np.ones((nS, nA))   # WM weights
        w_0 = (1.0 / nA) * np.ones((nS, nA)) # WM baseline
        prev_choice = -1 * np.ones(nS, dtype=int)

        for t in range(len(block_states)):
            s = int(block_states[t])

            # RL policy with stickiness
            Q_s = q[s, :].copy()
            if prev_choice[s] >= 0:
                Q_s[prev_choice[s]] += stickiness
            rl_logits = softmax_beta * (Q_s - np.max(Q_s))
            prl_all = np.exp(rl_logits)
            prl_all = prl_all / np.sum(prl_all)

            # WM policy
            W_s = w[s, :].copy()
            wm_logits = softmax_beta_wm * (W_s - np.max(W_s))
            pwm_all = np.exp(wm_logits)
            pwm_all = pwm_all / np.sum(pwm_all)

            # Entropy-based arbitration
            H_rl = -np.sum(prl_all * (np.log(prl_all + eps))) / np.log(nA)
            H_wm = -np.sum(pwm_all * (np.log(pwm_all + eps))) / np.log(nA)

            base = wm_weight_base * (3.0 / float(nS)) * (1.0 - is_older * np.clip(age_bias, 0.0, 1.0))
            base = np.clip(base, 0.0, 1.0)

            inv_unc_rl = (1.0 - H_rl)
            inv_unc_wm = (1.0 - H_wm)
            denom = inv_unc_rl + inv_unc_wm + eps
            arb = (inv_unc_wm / denom)
            wm_weight_eff = np.clip(base * arb + (1.0 - base) * 0.0 + base * 0.0, 0.0, 1.0)

            # Mixture policy
            p_total = wm_weight_eff * pwm_all + (1.0 - wm_weight_eff) * prl_all
            p_total = p_total / np.sum(p_total)
            p_total = np.clip(p_total, 0.0, 1.0)
            p_total = p_total / (np.sum(p_total) + eps)

            # Sample action
            a = int(np.random.choice(nA, p=p_total))
            simulated_actions[block_idx[t]] = a

            # Outcome
            r = 1 if a == int(correct_actions[s]) else 0
            simulated_rewards[block_idx[t]] = r

            # RL update
            delta = r - q[s, a]
            q[s, a] += lr * delta

            # WM update (decay towards baseline; then strengthen chosen if rewarded)
            w[s, :] = (1.0 - wm_decay) * w[s, :] + wm_decay * w_0[s, :]
            if r > 0.5:
                w[s, :] = (1.0 - wm_decay) * w[s, :]
                w[s, a] += wm_decay

            # Update stickiness memory
            prev_choice[s] = a

            tr += 1

    return simulated_actions, simulated_rewards
