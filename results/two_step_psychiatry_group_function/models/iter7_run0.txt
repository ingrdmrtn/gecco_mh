Here are three new cognitive models for the two-step decision task.

### Model 1: Hybrid Learner with Eligibility Traces
This model implements the full "Daw et al. (2011)" architecture by adding an eligibility trace parameter (`lambda_coeff`) to the Hybrid learner. This allows the Model-Free system to update Stage 1 values based on the final reward prediction error, bridging the gap between Stage 1 choice and Stage 2 outcome.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner (MB + MF) with Eligibility Traces (TD-lambda).
    
    Combines Model-Based planning with a Model-Free system that uses eligibility traces 
    to update Stage 1 values based on Stage 2 outcomes.

    Parameters:
    learning_rate: [0, 1] - Learning rate for value updates.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weight parameter (0 = pure MF, 1 = pure MB).
    lambda_coeff: [0, 1] - Eligibility trace decay parameter. 
                           Allows Stage 2 reward prediction error to update Stage 1 values.
    """
    learning_rate, beta, w, lambda_coeff = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Q-value integration
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # MF Updates
        # 1. Prediction error at stage 1 transition (TD(0) update)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 2. Prediction error at stage 2 outcome
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # 3. Eligibility trace update: Stage 1 value is updated by Stage 2 error scaled by lambda
        q_stage1_mf[action_1[trial]] += learning_rate * lambda_coeff * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Learner with Passive Decay
This model hypothesizes that participants do not hold values perfectly in memory. Instead, the Q-values of unchosen actions (both spaceships and aliens) decay towards zero over time. This "forgetting" mechanism competes with active learning.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner (MB + MF) with Passive Decay.
    
    Q-values for unchosen actions decay towards zero on each trial, 
    simulating forgetting or memory interference.

    Parameters:
    learning_rate: [0, 1] - Learning rate for chosen actions.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weight parameter (0 = pure MF, 1 = pure MB).
    decay_rate: [0, 1] - Rate at which unchosen Q-values decay (0=no decay, 1=instant forget).
    """
    learning_rate, beta, w, decay_rate = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Decay unchosen Stage 1 action
        curr_a1 = action_1[trial]
        q_stage1_mf[1 - curr_a1] *= (1 - decay_rate)
        
        # Decay unchosen Stage 2 action (in the current state only)
        curr_a2 = action_2[trial]
        q_stage2_mf[state_idx, 1 - curr_a2] *= (1 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with Win-Stay Lose-Shift (WSLS) Bias
This model augments the Hybrid learner with a primitive "Win-Stay, Lose-Shift" heuristic. Unlike simple stickiness (which repeats the previous action regardless of outcome), this bias promotes repeating the previous Stage 1 choice only if it was rewarded, and promotes switching if it was not.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner (MB + MF) with Primitive Win-Stay Lose-Shift (WSLS) Bias.
    
    Adds a heuristic bias to the Stage 1 choice based on the previous trial's outcome.
    If the previous trial was rewarded, the bias favors repeating the action.
    If unrewarded, the bias favors switching actions.

    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Weight parameter.
    wsls_bias: [0, 5] - Magnitude of the WSLS bias added to Q-values.
    """
    learning_rate, beta, w, wsls_bias = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply WSLS Bias to the net Q-values
        if last_action_1 != -1:
            if last_reward == 1:
                # Win-Stay: Add bias to previous action
                q_net[last_action_1] += wsls_bias
            else:
                # Lose-Shift: Add bias to the OTHER action to promote switching
                q_net[1 - last_action_1] += wsls_bias

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
