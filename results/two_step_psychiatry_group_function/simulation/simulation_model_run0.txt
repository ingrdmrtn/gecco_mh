import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Learner (MB + MF) with Choice Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, stickiness = parameters
    
    # Define transition matrix (same as in fitting code)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the first stage (Model-Free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the second stage (Model-Free)
    # Shape (2, 2) for 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2))

    # Initialize arrays to store simulation data
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 # Initialize with no previous action

    for t in range(n_trials):
        # Organize current trial reward probabilities
        # State 0 (drift1, drift2), State 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness (perseverance)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        # transition_matrix[a1] gives [p(s2=0|a1), p(s2=1|a1)]
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2 (pure MF based on current state Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning / Updates ---

        # 1. Update Stage 1 MF Q-values (TD(0) style, based on Stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state_idx, action_2] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 2. Update Stage 2 MF Q-values based on received reward
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
