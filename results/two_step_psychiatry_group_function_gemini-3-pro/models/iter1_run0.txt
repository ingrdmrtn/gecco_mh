Here are three cognitive models based on the two-step task literature, exploring different mechanisms than the pure Model-Free learner provided in the feedback.

### Model 1: Hybrid Model (Model-Based + Model-Free)
This model implements the classic "Hybrid" architecture often used to analyze this task. It assumes participants make first-stage choices by combining a Model-Based (MB) value (calculated using the transition matrix) and a Model-Free (MF) value (learned via TD errors). A weighting parameter `w` determines the balance between these two systems.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner (Model-Based + Model-Free).
    
    This model computes Q-values for the first stage as a weighted mixture of 
    Model-Based (planning via transition matrix) and Model-Free (TD learning) values.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature for softmax choice.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix: 
    # Row 0 (Spaceship 0) -> [0.7 to Planet 0, 0.3 to Planet 1]
    # Row 1 (Spaceship 1) -> [0.3 to Planet 0, 0.7 to Planet 1]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize values
    q_stage1_mf = np.ones(2) * 0.5      # MF values for Spaceships
    q_stage2_mf = np.ones((2, 2)) * 0.5 # MF values for Aliens (shared by MB system)

    for trial in range(n_trials):
        # --- Stage 1 Policy (Hybrid) ---
        # 1. Calculate Model-Based values
        # Max value available at each planet
        max_q_stage2 = np.max(q_stage2_mf, axis=1) 
        # Bellman equation: Transition * max_next_state
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Softmax choice
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # Planet reached

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Updates ---
        # Stage 2 Prediction Error
        pe_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 2 values (Aliens)
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_stage2
        
        # Stage 1 Prediction Error (SARSA-style for MF component)
        # Using the value of the state actually reached (q_stage2_mf of chosen option)
        v_state_2 = q_stage2_mf[state_idx, action_2[trial]]
        pe_stage1 = v_state_2 - q_stage1_mf[action_1[trial]]
        
        # Update Stage 1 MF values
        # Note: Eligibility trace is implicitly 0 here to distinguish from Model 2 feedback
        q_stage1_mf[action_1[trial]] += learning_rate * pe_stage1 

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseveration Model (Pure MF + Choice Stickiness)
This model modifies the pure Model-Free learner (similar to the one in the feedback) but adds a "perseveration" parameter `stickiness`. This accounts for the tendency of participants to simply repeat their previous motor action (choice of spaceship) regardless of reward history.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Choice Perseveration (Stickiness).
    
    Adds a bias to repeat the previous Stage 1 choice, independent of value learning.
    
    Parameters:
    learning_rate: [0,1] - Rate at which values are updated.
    beta: [0,10] - Inverse temperature for softmax choice.
    stickiness: [0,5] - Bonus added to the Q-value of the previously chosen spaceship.
    """
    learning_rate, beta, stickiness = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    # Track previous choice (initialize arbitrarily, e.g., -1 so it doesn't trigger on trial 0)
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Calculate logits: Value + Stickiness Bonus
        logits_1 = beta * q_stage1_mf.copy()
        
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Updates (TD-1 / Eligibility Trace) ---
        # We use eligibility trace = 1.0 (full propagation) here implicitly
        # by updating stage 1 with the final reward error directly
        
        # Stage 2 update
        pe_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_stage2
        
        # Stage 1 update (Direct reinforcement from reward)
        # This is equivalent to TD(1) or direct reinforcement of the choice chain
        q_stage1_mf[action_1[trial]] += learning_rate * pe_stage2
        
        # Store action for next trial's stickiness
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Dual-Learning Rate Model (Positive/Negative Split)
This model posits that participants might learn differently from wins (rewards) versus losses (omissions). It splits the learning rate into `alpha_pos` (for positive prediction errors) and `alpha_neg` (for negative prediction errors). This asymmetry is often observed in psychiatric populations or reward sensitivity studies.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Learning Rate Model.
    
    Uses separate learning rates for positive prediction errors (better than expected)
    and negative prediction errors (worse than expected).
    
    Parameters:
    alpha_pos: [0,1] - Learning rate for positive prediction errors.
    alpha_neg: [0,1] - Learning rate for negative prediction errors.
    beta: [0,10] - Inverse temperature.
    eligibility: [0,1] - Eligibility trace for Stage 1 updates.
    """
    alpha_pos, alpha_neg, beta, eligibility = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Updates ---
        
        # 1. Stage 1 TD error (based on transition to state 2)
        v_state_2 = q_stage2_mf[state_idx, action_2[trial]]
        pe_stage1 = v_state_2 - q_stage1_mf[action_1[trial]]
        
        # Apply asymmetric learning rate for Stage 1
        lr_1 = alpha_pos if pe_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += lr_1 * pe_stage1

        # 2. Stage 2 TD error (based on reward)
        pe_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Apply asymmetric learning rate for Stage 2
        lr_2 = alpha_pos if pe_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * pe_stage2
        
        # 3. Eligibility Trace Update for Stage 1
        # Using the same asymmetric logic based on the PE sign
        q_stage1_mf[action_1[trial]] += lr_2 * eligibility * pe_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
