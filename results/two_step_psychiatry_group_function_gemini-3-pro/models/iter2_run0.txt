Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Model with Mixing Parameter (`w`)
This model combines Model-Based (MB) and Model-Free (MF) reinforcement learning strategies. It explicitly calculates values using the transition matrix for the MB component and standard TD learning for the MF component. A mixing parameter `w` determines the relative weight of each strategy in the first-stage decision.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning.
    
    This model computes Q-values for the first stage using a weighted mixture of 
    Model-Free (TD-learning) and Model-Based (planning via transition matrix) values.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature for softmax choice.
    w: [0,1] - Mixing weight (0 = Pure Model-Free, 1 = Pure Model-Based).
    lambd: [0,1] - Eligibility trace parameter linking stage 2 reward to stage 1 choice.
    """
    learning_rate, beta, w, lambd = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as described in the task (70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize Q-values
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value: V_MB(s1) = T(s1, s2) * max(Q_MF(s2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # 0 or 1 (Planet X or Y)

        # --- Stage 2 Decision ---
        # Standard Model-Free Q-learning for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 TD Error (SARSA style using the value of the state reached)
        # Note: We use the value of the chosen action in stage 2 as the proxy for state value
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 TD Error
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace: Propagate stage 2 error back to stage 1 choice
        q_stage1_mf[action_1[trial]] += learning_rate * lambd * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseverance (Choice Stickiness) Model
This model adds a "stickiness" parameter to the standard Model-Free learning. It captures the tendency of participants to repeat their previous choice regardless of reward history (perseverance) or switch frequently (alternation), independent of value-based learning.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Reinforcement Learning with Choice Perseverance.
    
    Includes a 'stickiness' parameter that biases the agent towards 
    repeating the previous Stage 1 action, independent of reward.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature.
    stickiness: [-5, 5] - Bias for repeating the last choice (positive) or switching (negative).
        Note: While bounds are usually [0,1], stickiness is an additive bias term 
        inside the softmax, so it can be larger/negative. We assume a normalized input range here.
    """
    learning_rate, beta, stickiness = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1 # No previous action on trial 0

    for trial in range(n_trials):

        # --- Stage 1 Decision ---
        # Calculate effective Q-values including stickiness bonus
        q_effective = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_effective[last_action_1] += stickiness

        exp_q1 = np.exp(beta * q_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Standard TD(1) / SARSA updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Direct reinforcement of Stage 1 choice by Stage 2 reward (TD(1))
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Separate Learning Rates for Stages
This model hypothesizes that learning dynamics differ between the abstract choice of a spaceship (Stage 1) and the concrete choice of an alien (Stage 2). It assigns a distinct learning rate to each stage, allowing the model to capture if participants update their beliefs about spaceships faster or slower than their beliefs about aliens.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dual Learning Rate Model (Stage-Specific).
    
    Uses separate learning rates for Stage 1 (Spaceship choice) and 
    Stage 2 (Alien choice). This captures potential differences in 
    plasticity between the higher-level planning stage and the terminal reward stage.
    
    Parameters:
    lr_stage1: [0,1] - Learning rate for the first stage (Spaceships).
    lr_stage2: [0,1] - Learning rate for the second stage (Aliens).
    beta: [0,10] - Inverse temperature.
    eligibility: [0,1] - Weight of the direct reinforcement from reward to Stage 1.
    """
    lr_stage1, lr_stage2, beta, eligibility = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        
        # Update Stage 1 based on prediction of Stage 2 value
        # Using lr_stage1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_stage1 * delta_stage1
        
        # Update Stage 2 based on reward
        # Using lr_stage2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_stage2 * delta_stage2
        
        # Eligibility trace update for Stage 1 based on final reward
        # Often modeled as a function of the stage 1 learning rate or a mix, 
        # here we scale it by lr_stage1 to keep consistencies in update magnitude for stage 1.
        q_stage1_mf[action_1[trial]] += lr_stage1 * eligibility * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
