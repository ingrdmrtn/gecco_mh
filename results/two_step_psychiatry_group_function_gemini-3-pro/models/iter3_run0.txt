Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Model-Based/Model-Free with Mixing Parameter `w`
This model implements the classic "Daw et al. (2011)" hybrid architecture. It maintains both a Model-Based (planning-based) valuation of the first stage and a Model-Free (experience-based) valuation. A mixing parameter `w` determines the balance between these two systems during the first-stage choice. The second stage is purely model-free.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL.
    
    Combines a Model-Free learner (TD-SARSA) with a Model-Based learner 
    (planning using the transition matrix). A mixing weight 'w' controls 
    the contribution of each system to the Stage 1 decision.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1] (0 = Pure Model-Free, 1 = Pure Model-Based)
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition probabilities as per task description
    # Row 0: Spaceship A -> [Planet X (0), Planet Y (1)]
    # Row 1: Spaceship U -> [Planet X (0), Planet Y (1)]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values initialization (0.5 assumes neutral prior for binary 0/1 rewards)
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # Model-Based Value: V_MB(s1) = Sum(P(s2|s1) * max(Q_MF(s2, a2)))
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Max value of next states
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # 0 for Planet X, 1 for Planet Y

        # --- Stage 2 Policy ---
        # Pure Model-Free choice at the second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updating ---
        # 1. Update Stage 1 MF Q-value (TD(0) style, using Stage 2 Q-value as proxy for V(s'))
        # Note: Standard SARSA would use Q(s', a'), Q-learning uses max Q(s', :).
        # Here we use the chosen action value q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 2. Update Stage 2 MF Q-value based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: In this simple version, we do not apply eligibility traces (lambda=0 implicitly)
        # to carry the stage 2 RPE back to stage 1.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free RL with Asymmetric Learning Rates
This model hypothesizes that participants learn differently from positive outcomes (rewards) versus negative outcomes (omission of reward). It splits the learning rate into `alpha_pos` and `alpha_neg`. This allows the model to capture biases like optimism (ignoring bad news) or pessimism (overweighting bad news), combined with a standard eligibility trace `lambda` to credit the first stage choice for the final outcome.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Asymmetric Learning Rates and Eligibility Trace.
    
    Uses separate learning rates for positive prediction errors (alpha_pos)
    and negative prediction errors (alpha_neg).
    
    Bounds:
    alpha_pos: [0,1]
    alpha_neg: [0,1]
    beta: [0,10]
    lambd: [0,1] (Eligibility trace decay)
    """
    alpha_pos, alpha_neg, beta, lambd = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]]) # Unused in pure MF, but kept for structure
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updating ---
        # Stage 1 Prediction Error
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Apply asymmetric learning rate for Stage 1 update
        lr_s1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += lr_s1 * delta_stage1
        
        # Stage 2 Prediction Error
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Apply asymmetric learning rate for Stage 2 update
        lr_s2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_s2 * delta_stage2
        
        # Eligibility Trace: Update Stage 1 again based on Stage 2 outcome
        # The trace allows the reward at stage 2 to influence stage 1 value immediately
        q_stage1_mf[action_1[trial]] += lr_s2 * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Choice Stickiness and Eligibility
This is a comprehensive model that combines Model-Based and Model-Free learning (via `w`) but adds `stickiness` to capture simple motor/choice perseveration (repeating the same key press regardless of value) and `lambd` (eligibility trace) to allow the Model-Free component to learn from the second stage outcome effectively. This configuration tests if participants use a mixture of planning, habit, and simple repetition.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model (MB/MF) with Choice Stickiness and Eligibility Trace.
    
    Combines:
    1. Weighted MB and MF valuation (w).
    2. Eligibility trace (lambda) for the MF component to link S2 reward to S1 choice.
    3. Choice stickiness bias on the first stage.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1]
    lambd: [0,1]
    stickiness: [-5, 5]
    """
    learning_rate, beta, w, lambd, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # 1. Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Add Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updating ---
        # Stage 1 MF Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 MF Update (Reward)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility Trace Update (TD(1)) for Stage 1
        # Propagate the stage 2 RPE back to stage 1
        q_stage1_mf[action_1[trial]] += learning_rate * lambd * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
