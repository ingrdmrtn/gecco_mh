Here are three new cognitive models that explore alternative parameter combinations and mechanisms (asymmetric learning rates, stage-specific exploration, and dynamic transition learning) as requested.

import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Hybrid Model (MB/MF) with Valence-Dependent Learning.
    
    Distinguishes between learning from positive prediction errors (alpha_pos)
    and negative prediction errors (alpha_neg) within a Hybrid MB/MF framework.
    This captures optimism/pessimism biases in value updating.

    Bounds:
    alpha_pos: [0,1]
    alpha_neg: [0,1]
    beta: [0,10]
    w: [0,1]
    """
    alpha_pos, alpha_neg, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        learning_rate_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += learning_rate_1 * delta_stage1
        
        # Update Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        learning_rate_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Model with Dual Betas and Eligibility Trace.
    
    Uses separate inverse temperature parameters (beta_1, beta_2) for the two stages,
    acknowledging that the exploration-exploitation trade-off may differ between 
    abstract spaceship choices and concrete alien choices. Includes eligibility 
    trace (lambd) to connect stage 2 outcomes to stage 1.

    Bounds:
    learning_rate: [0,1]
    beta_1: [0,10]
    beta_2: [0,10]
    lambd: [0,1]
    """
    learning_rate, beta_1, beta_2, lambd = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]]) # Unused in pure MF, but kept for template consistency
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # policy for the first choice
        exp_q1 = np.exp(beta_1 * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility Trace Update (connect reward to stage 1 choice)
        q_stage1_mf[action_1[trial]] += learning_rate * lambd * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Hybrid Model with Learned Transitions.
    
    Instead of assuming a fixed 70/30 transition matrix, this agent learns the 
    transition probabilities (P(State|Action)) trial-by-trial using a delta rule
    (lr_trans). This allows the Model-Based component to adapt if the environment 
    structure is perceived to change.

    Bounds:
    lr_reward: [0,1]
    lr_trans: [0,1]
    beta: [0,10]
    w: [0,1]
    """
    lr_reward, lr_trans, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Initialize estimated transition matrix (start with uniform prior or standard expectation)
    # Rows: Action (0,1), Cols: State (0,1)
    est_transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]]) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Use the *learned* transition matrix for MB calculation
        q_stage1_mb = est_transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_action_1 = action_1[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1 (MF)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[chosen_action_1]
        q_stage1_mf[chosen_action_1] += lr_reward * delta_stage1
        
        # Update Stage 2 (MF)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_reward * delta_stage2
        
        # Update Transition Matrix (Dynamic MB)
        # 1. Update the prob of the state we actually landed on towards 1
        curr_prob = est_transition_matrix[chosen_action_1, state_idx]
        est_transition_matrix[chosen_action_1, state_idx] += lr_trans * (1 - curr_prob)
        
        # 2. Update the prob of the other state towards 0
        other_state = 1 - state_idx
        est_transition_matrix[chosen_action_1, other_state] = 1.0 - est_transition_matrix[chosen_action_1, state_idx]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
