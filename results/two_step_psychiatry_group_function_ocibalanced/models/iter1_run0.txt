Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Model-Based / Model-Free (Daw et al. 2011 style)
This is the classic "Two-Step" model that mixes Model-Based (planning using the transition matrix) and Model-Free (TD learning) strategies. The parameter `w` controls the balance: `w=1` is purely Model-Based, `w=0` is purely Model-Free.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning.
    
    Combines a Model-Based controller (which uses the transition matrix to plan)
    and a Model-Free controller (which uses TD errors). A mixing weight 'w' 
    arbitrates between them.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1] (Mixing weight: 0=MF, 1=MB)
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as per task description
    # A (0) -> X (0) common (0.7), U (1) -> Y (1) common (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)      # MF values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # MF values for Stage 2 (also used by MB for leaf nodes)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Q_MB(s1, a1) = sum(P(s2|s1,a1) * max(Q_MF(s2, a2)))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # Planet arrived at

        # --- Stage 2 Choice ---
        # Standard Softmax on Stage 2 values
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # SARSA / TD(0) updates
        
        # Prediction error at Stage 2 (Reward Prediction Error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Prediction error at Stage 1 (State Prediction Error for MF)
        # Note: In standard hybrid models, MF updates via TD(1) or TD(0). 
        # Here we use a simple TD(1) approximation logic often used: 
        # The Stage 1 MF value is updated towards the Q-value of the chosen stage 2 option.
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]

        # Update Stage 2 values
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Update Stage 1 MF values (using eligibility trace logic effectively lambda=1)
        # Standard implementation updates Q1 based on reward directly or via the 2nd stage Q
        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseverance / Choice Autocorrelation
This model is purely Model-Free (like the one in your feedback) but adds a **perseverance parameter** (`stickiness`). Humans often repeat their last choice regardless of reward ("stickiness"). This parameter captures that tendency, adding a bonus to the previously chosen action at Stage 1.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Choice Perseverance (Stickiness).
    
    Adds a 'stickiness' bonus to the action chosen on the previous trial,
    capturing the tendency to repeat choices regardless of outcome.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    stickiness: [0,5] (Bonus added to logits of repeated action)
    """
    learning_rate, beta, stickiness = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with an impossible action

    for trial in range(n_trials):

        # --- Stage 1 Choice with Stickiness ---
        # Calculate logits
        logits_1 = beta * q_stage1_mf
        
        # Add stickiness bonus if it's not the first trial
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store action for next trial's stickiness
        last_action_1 = action_1[trial]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning (TD-Lambda with lambda=1 for simplicity) ---
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]

        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning Rates (Positive vs Negative Prediction Errors)
This model investigates if participants learn differently from positive surprises (getting a coin when not expecting one) versus negative surprises (getting nothing when expecting a coin). It uses two separate learning rates: `alpha_pos` for positive prediction errors and `alpha_neg` for negative ones.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Asymmetric Learning Rates.
    
    Uses different learning rates for positive and negative prediction errors,
    allowing the agent to be more sensitive to rewards or punishments (omissions).
    
    Bounds:
    alpha_pos: [0,1] (Learning rate for RPE > 0)
    alpha_neg: [0,1] (Learning rate for RPE < 0)
    beta: [0,10]
    """
    alpha_pos, alpha_neg, beta = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Calculate RPE at second stage
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Determine which learning rate to use for Stage 2 update
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2
        
        # Calculate RPE at first stage (driving the value of the spaceship)
        # We use the updated stage 2 value to drive stage 1 (SARSA-like)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Determine which learning rate to use for Stage 1 update
        lr_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += lr_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
