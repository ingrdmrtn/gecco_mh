Here is the simulation code for the Model-Free RL with Choice Perseverance (Stickiness) model.

```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free RL with Choice Perseverance (Stickiness).
    
    This model adds a 'stickiness' bonus to the action chosen on the previous trial
    at the first stage, capturing the tendency to repeat choices regardless of outcome.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q-values for 2 states * 2 actions in stage 2
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed for the standard 2-step task)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    rng = np.random.default_rng()
    last_action_1 = -1 # Initialize with an impossible action for the first trial

    for t in range(n_trials):
        # 1. Stage 1 Choice
        logits_1 = beta * q_stage1_mf.copy()
        
        # Apply stickiness bonus if a previous action exists
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        prob_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=prob_1)
        last_action_1 = a1
        
        # 2. Transition to Stage 2
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        qs_current_state = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * qs_current_state)
        prob_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=prob_2)
        
        # 4. Reward Outcome
        # Organize drifts into a structure matching state/action indices
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Learning (Model-Free Updates)
        # Calculate prediction errors
        # Note: The fitting code calculates delta_stage1 based on the Q-value of the 
        # *chosen* second stage action, which is standard SARSA-like logic often used here.
        
        delta_stage2 = r - q_stage2_mf[s2, a2]
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Update Q-values
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # The fitting code updates stage 1 Q-value using the sum of both prediction errors
        # (effectively TD(1) or direct reinforcement from the final reward).
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
