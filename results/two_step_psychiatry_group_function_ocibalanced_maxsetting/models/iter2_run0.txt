Here are three new cognitive models based on the participant data and the two-step task structure.

### Model 1: Hybrid Model with Separate Stage Learning Rates
This model hypothesizes that participants update their expectations about the stable transition structure (Stage 1) at a different rate than they update their expectations about the fluctuating alien rewards (Stage 2). Since Stage 1 is structurally stable (70/30 transitions) while Stage 2 is a drifting bandit task, using a single learning rate might be suboptimal.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Learning Rates for Stage 1 and Stage 2.
    
    Allows the model to learn the value of the spaceship choice (Stage 1) 
    and the alien choice (Stage 2) at different speeds.
    
    Parameters:
    lr_1: [0, 1] - Learning rate for Stage 1 (spaceships).
    lr_2: [0, 1] - Learning rate for Stage 2 (aliens).
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting of Model-Based values (0=Pure MF, 1=Pure MB).
    """
    lr_1, lr_2, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        # Update Stage 1 using lr_1
        q_stage1_mf[a1] += lr_1 * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        # Update Stage 2 using lr_2
        q_stage2_mf[s_idx, a2] += lr_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Passive Decay
The task description notes that alien reward probabilities "changed slowly over trials." In such non-stationary environments, memory of old reward values can become obsolete. This model implements a decay mechanism where unchosen options in Stage 2 slowly drift back towards a neutral value (0.5), representing increased uncertainty or forgetting over time.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Passive Decay for Unchosen Options.
    
    Unchosen aliens in Stage 2 decay towards a neutral value (0.5) to account 
    for the drifting probabilities in the environment (forgetting/uncertainty).
    
    Parameters:
    learning_rate: [0, 1] - Update rate for chosen actions.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Model-Based weight.
    decay_rate: [0, 1] - Rate at which unchosen values drift to 0.5.
    """
    learning_rate, beta, w, decay_rate = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    # Initialize stage 2 values at 0.5 (neutral expectation)
    q_stage2_mf = np.full((2, 2), 0.5)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Decay the unchosen option in the current state towards 0.5
        unchosen_a2 = 1 - a2
        q_stage2_mf[s_idx, unchosen_a2] += decay_rate * (0.5 - q_stage2_mf[s_idx, unchosen_a2])
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Full Hybrid Model (Eligibility Trace + Stickiness)
This model combines the Model-Based/Model-Free hybrid approach with two critical mechanisms: eligibility traces ($\lambda$) and choice stickiness. The eligibility trace allows the final reward to directly reinforce the Stage 1 choice (a hallmark of complex temporal difference learning), while stickiness captures the strong motor perseverance observed in the participant data.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Full Hybrid Model with Eligibility Traces and Stickiness.
    
    Combines Model-Based planning with a Model-Free system that uses eligibility 
    traces (Lambda) to credit Stage 1 choices with Stage 2 outcomes directly.
    Also includes stickiness to account for choice perseverance.
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Model-Based weight.
    lambd: [0, 1] - Eligibility trace parameter (0 = TD(0), 1 = Monte Carlo).
    stickiness: [0, 5] - Perseverance bonus for repeated Stage 1 choices.
    """
    learning_rate, beta, w, lambd, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness to logits
        logits = beta * q_net
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 1: Includes eligibility trace from Stage 2 prediction error
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambd * delta_stage2
        
        # Update Stage 2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
