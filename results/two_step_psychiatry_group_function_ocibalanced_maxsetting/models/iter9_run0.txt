Here are three new cognitive models for the two-step task.

### Model 1: Outcome-Dependent Model-Based Weighting
This model hypothesizes that the participant's reliance on Model-Based (planning) vs. Model-Free (habit) strategies is not static but fluctuates based on recent success. A reward might reinforce the current strategy (or trigger "exploitation" of the model), while a lack of reward might cause a shift in strategy.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Outcome-Dependent Model-Based Weighting (Strategy Switching).
    
    The weight 'w' controlling the balance between Model-Based (MB) and 
    Model-Free (MF) control depends on the outcome of the previous trial.
    Allows the agent to adjust their planning depth or reliance on habits
    dynamically in response to wins (w_win) versus losses (w_lose).
    
    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta: [0, 10] - Inverse temperature (exploration/exploitation).
    w_win: [0, 1] - Weight for MB system after a rewarded trial.
    w_lose: [0, 1] - Weight for MB system after an unrewarded trial.
    stickiness: [0, 5] - Choice perseverance bonus.
    """
    learning_rate, beta, w_win, w_lose, stickiness = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0 # Default context for first trial
    
    for trial in range(n_trials):
        # Determine w based on previous outcome
        w = w_win if last_reward == 1 else w_lose
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])
        
        # Stage 1 MF Update (driven by Stage 2 value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update (driven by Reward)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Counterfactual Updating
This model introduces a mechanism where the agent updates not only the chosen option but also the *unchosen* option in the first stage. The update is coupled: if the chosen option yields a positive prediction error, the unchosen option is pushed down (and vice versa), simulating a belief in competition between options.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Counterfactual Updating (Symmetric Coupling).
    
    When the chosen Stage 1 action is updated based on prediction error,
    the unchosen action is also updated in the opposite direction, scaled by
    a coupling parameter. This simulates 'competition' or counterfactual reasoning
    (e.g., "If A was good, B must be bad").
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weight.
    stickiness: [0, 5] - Perseverance bonus.
    cf_coupling: [0, 1] - Strength of counterfactual update for the unchosen option.
    """
    learning_rate, beta, w, stickiness, cf_coupling = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Counterfactual update for unchosen action
        # If delta is positive (good surprise), unchosen value decreases
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] -= learning_rate * cf_coupling * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Uncertainty-Driven Exploration
This model adds a "Novelty Bonus" to the first-stage decision. It tracks how long it has been since each spaceship was chosen and adds a bonus to the value of neglected options, encouraging the agent to explore parts of the state space they haven't visited recently.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Uncertainty-Driven Exploration (Novelty Bonus).
    
    Adds a bonus to the action values in Stage 1 proportional to the time 
    since that action was last chosen. This encourages exploration of 
    neglected options (uncertainty reduction).
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weight.
    stickiness: [0, 5] - Perseverance bonus.
    novelty_bonus: [0, 1] - Bonus weight for trials since last selection.
    """
    learning_rate, beta, w, stickiness, novelty_bonus = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    time_since_chosen = np.zeros(2) # Tracks trials since each S1 action was picked
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        # Add Stickiness
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Add Novelty Bonus (Exploration)
        logits_1 += novelty_bonus * time_since_chosen
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])
        
        # Update time since chosen
        time_since_chosen += 1
        time_since_chosen[a1] = 0
        
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
