import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Hybrid Model with Outcome-Dependent Stickiness.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_win, stick_lose]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_win, stick_lose = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows), 2 actions (cols)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])
    
    # History variables for stickiness
    last_action_1 = -1
    last_reward = 0
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        # Value of stage 2 states is the max Q-value available in that state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation for MB: Transition * Value of next states
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits = beta * q_net
        
        # Apply Outcome-Dependent Stickiness
        if last_action_1 != -1:
            if last_reward > 0:
                logits[last_action_1] += stick_win
            else:
                logits[last_action_1] += stick_lose
        
        # Softmax for Stage 1
        exp_q1 = np.exp(logits)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 Choice
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # --- Transition ---
        # Determine Stage 2 State based on transition matrix and a1
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard Softmax on Stage 2 MF values for the specific state arrived at
        q_s2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * q_s2)
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 Choice
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # --- Reward ---
        # Get probability based on state and action from drifts
        if s_idx == 0:
            p_r = drift1[t] if a2 == 0 else drift2[t]
        else: # s_idx == 1
            p_r = drift3[t] if a2 == 0 else drift4[t]
        
        r = int(rng.random() < p_r)
        
        # --- Updates ---
        # Update Stage 1 MF Q-value
        # Uses the Q-value of the *chosen* action in stage 2 (SARSA-like logic from fit code)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1
        last_reward = r
        
    return stage1_choice, state2, stage2_choice, reward
