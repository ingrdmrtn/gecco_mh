Here are three new cognitive models based on the two-step task structure, introducing mechanisms for counterfactual updating, outcome-dependent perseverance, and independent system scaling.

### Model 1: Counterfactual Stage 2 Learning
This model hypothesizes that participants learn not just from the outcome of the alien they chose, but also infer the value of the *unchosen* alien. If Alien A gives 0 coins, the participant assumes Alien B (on the same planet) would likely have given 1 coin (and vice versa), updating the unchosen option with a separate learning rate.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Counterfactual (Fictitious) Stage 2 Updating.

    In addition to standard TD learning for the chosen option, this model 
    updates the unchosen option in Stage 2 assuming a reciprocal reward structure 
    (if chosen=0, unchosen=1). This captures 'fictitious play' or counterfactual reasoning.

    Parameters:
    - learning_rate: [0, 1] Update rate for chosen options.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting (1 = Model-Based, 0 = Model-Free).
    - alpha_cf: [0, 1] Counterfactual learning rate for the unchosen Stage 2 option.
    """
    learning_rate, beta, w, alpha_cf = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 planets, 2 aliens each

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # Stage 1 MF Update (SARSA)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # Stage 2 MF Update (Standard)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Stage 2 Counterfactual Update (Unchosen Option)
        # Assume reciprocal reward: if reward=1, cf_reward=0; if reward=0, cf_reward=1
        unchosen_action_2 = 1 - action_2[trial]
        cf_reward = 1.0 - reward[trial]
        delta_cf = cf_reward - q_stage2_mf[state_idx, unchosen_action_2]
        q_stage2_mf[state_idx, unchosen_action_2] += alpha_cf * delta_cf

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Outcome-Specific Stickiness (Win-Stay/Lose-Stay)
Previous models used a single `stickiness` parameter. This model splits perseverance into `stick_win` and `stick_loss`. This allows the model to differentiate between "Win-Stay" (repeating a choice because it was rewarded, beyond value updating) and "Lose-Stay" (repeating a choice despite failure, or conversely, "Lose-Shift" if the parameter is low/negative).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Outcome-Specific Stickiness.

    Differentiates the tendency to repeat the previous Stage 1 choice based on 
    the previous trial's outcome. 
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting.
    - stick_win: [0, 5] Bonus added to previous choice if it resulted in a reward.
    - stick_loss: [0, 5] Bonus added to previous choice if it resulted in no reward.
    """
    learning_rate, beta, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply outcome-specific stickiness
        q_net_choice = q_net.copy()
        if last_action_1 != -1:
            if last_reward == 1:
                q_net_choice[last_action_1] += stick_win
            else:
                q_net_choice[last_action_1] += stick_loss

        exp_q1 = np.exp(beta * q_net_choice)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # Track history
        last_action_1 = action_1[trial]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Independent System Betas
Instead of using a mixing weight `w` (which forces the sum of weights to be 1) and a single `beta`, this model assigns independent inverse temperatures to the Model-Based and Model-Free systems (`beta_mb` and `beta_mf`). This allows the "confidence" or "influence" of each system to vary independently. It also includes general stickiness to remain competitive.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Independent System Betas (Dual Temperature Model).

    Instead of a weighting parameter 'w', this model uses separate inverse 
    temperatures for the Model-Based and Model-Free systems. This allows 
    the distinct systems to have different noise levels and contribution magnitudes.

    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta_mb: [0, 10] Inverse temperature for the Model-Based component.
    - beta_mf: [0, 10] Inverse temperature for the Model-Free component.
    - stickiness: [0, 5] General choice perseverance bonus.
    """
    learning_rate, beta_mb, beta_mf, stickiness = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine logits directly using separate betas
        # Note: We do not multiply by a shared beta later; these ARE the weighted values.
        logits = (beta_mb * q_stage1_mb) + (beta_mf * q_stage1_mf)
        
        if last_action_1 != -1:
            logits[last_action_1] += stickiness

        # Softmax
        # Subtract max for numerical stability (standard practice in softmax implementation)
        logits = logits - np.max(logits)
        exp_logits = np.exp(logits)
        probs_1 = exp_logits / np.sum(exp_logits)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        # Use beta_mf for stage 2 as it is purely model-free
        exp_q2 = np.exp(beta_mf * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
