Here are the 3 new cognitive models based on the two-step task structure.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dual-Beta Hybrid Model.
    
    Instead of using a weighting parameter 'w' to mix Model-Based (MB) and 
    Model-Free (MF) values, this model assigns separate inverse temperature 
    parameters (betas) to the MB and MF components in the Stage 1 decision. 
    This allows the model to capture independent fluctuations in the reliability 
    or influence of the two systems, rather than forcing a trade-off.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for MF values (Stage 1 and 2).
    - beta_mb: [0, 10] Inverse temperature for the Model-Based component at Stage 1.
    - beta_mf: [0, 10] Inverse temperature for the Model-Free component (Stage 1 and 2).
    - stickiness: [0, 5] Choice perseverance bonus for Stage 1.
    """
    learning_rate, beta_mb, beta_mf, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Stage 2 (2 states, 2 actions)

    last_action_1 = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation: V_MB(S1) = T * max(Q_MF(S2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine components with separate betas
        # Logits = beta_mb * Q_MB + beta_mf * Q_MF + stickiness
        logits_1 = (beta_mb * q_stage1_mb) + (beta_mf * q_stage1_mf)
        
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Softmax for Stage 1
        # Subtract max for numerical stability
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        # Standard Softmax using beta_mf
        logits_2 = beta_mf * q_stage2_mf[state_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Value Updates ---
        # 1. Update Stage 1 MF value using Stage 2 value (TD Learning)
        # Note: We use the value of the chosen state/action in Stage 2 as the target
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 2. Update Stage 2 MF value using Reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Direct Reinforcement Hybrid Model.
    
    This model adds a 'Direct Reinforcement' mechanism to the standard hybrid learner.
    While the standard Model-Free update (TD) chains values from Stage 2 back to Stage 1,
    the Direct update reinforces the Stage 1 choice based immediately on the final reward,
    ignoring the intermediate state. This captures a heuristic where participants 
    credit the spaceship directly for the coin.
    
    Parameters:
    - lr_td: [0, 1] Learning rate for standard Temporal Difference updates (Stage 1 & 2).
    - lr_direct: [0, 1] Learning rate for direct reinforcement of Stage 1 choice by Reward.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting (1 = Model-Based, 0 = Model-Free).
    """
    lr_td, lr_direct, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted mixture of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Value Updates ---
        
        # 1. Standard TD Update for Stage 1 MF
        delta_td = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_td * delta_td
        
        # 2. Direct Reinforcement Update for Stage 1 MF
        # This updates the spaceship value directly based on the coin received
        delta_direct = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_direct * delta_direct

        # 3. Standard Update for Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_td * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Accumulated Habit Hybrid Model.
    
    Replaces simple one-trial stickiness with an accumulated 'habit' trace.
    The habit trace increases for the chosen action and decays on every trial.
    This models habit formation as a continuous strengthening process rather 
    than a binary repetition bias.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting (1 = Model-Based, 0 = Model-Free).
    - habit_decay: [0, 1] Rate at which the habit trace decays (0 = instant forgetting, 1 = no decay).
    - habit_weight: [0, 5] Strength of the influence of the habit trace on choice.
    """
    learning_rate, beta, w, habit_decay, habit_weight = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Habit trace vector for Stage 1 choices (Spaceship A vs U)
    habit_trace = np.zeros(2)

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add accumulated habit trace to the logits
        logits_1 = beta * q_net + habit_weight * habit_trace
        
        logits_1 = logits_1 - np.max(logits_1) # Stability
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Habit Trace
        # Decay existing trace
        habit_trace *= habit_decay
        # Strengthen trace for chosen action
        habit_trace[action_1[trial]] += 1.0
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
