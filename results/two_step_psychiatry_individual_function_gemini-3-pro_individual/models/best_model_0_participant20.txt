def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free TD(lambda) Learner with Perseveration.
    
    A pure model-free agent that updates Stage 1 values based on both the
    Stage 2 state value (TD(0)) and the final reward (TD(1) via eligibility trace).
    Includes a perseveration bonus for choice stickiness.

    Parameters:
    learning_rate: [0,1] - Update rate.
    beta: [0,10] - Inverse temperature.
    lam: [0,1] - Eligibility trace parameter (0 = update Stage 1 only on State 2 value; 1 = update Stage 1 fully on Reward).
    perseveration: [0,1] - Bonus added to the previously chosen Stage 1 action.
    """
    learning_rate, beta, lam, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]]) # Unused in pure MF logic but kept for structure
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):


        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = perseveration
            
        exp_q1 = np.exp(beta * (q_stage1_mf + pers_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]
        last_action_1 = action_1[trial]

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]


        temp_delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1 + learning_rate * lam * temp_delta_stage2
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
