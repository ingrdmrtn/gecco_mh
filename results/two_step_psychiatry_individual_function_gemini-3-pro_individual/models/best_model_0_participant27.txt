def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF with Distinct Stage-1 MF Learning Rate.
    
    This model combines Model-Based (MB) and Model-Free (MF) values at stage 1.
    Unlike standard hybrid models that use a mixing weight 'w', this model simply sums
    the Q-values from both systems. Crucially, it allows the stage-1 MF values to be 
    updated with a specific learning rate (lr_mf1), separate from the rate used for 
    stage-2 values (lr_2).
    
    Parameters:
    - lr_mf1: [0, 1] Learning rate for stage 1 Model-Free Q-values.
    - lr_2: [0, 1] Learning rate for stage 2 Q-values (used by both MB and MF).
    - beta: [0, 10] Inverse temperature for softmax.
    """
    lr_mf1, lr_2, beta = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)      # MF value of spaceships
    q_stage2_mf = np.zeros((2, 2)) # Value of aliens (shared/base for MB)

    for trial in range(n_trials):


        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = q_stage1_mf + q_stage1_mb
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # 0 or 1 (Planets)

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]




        delta_stage1_mf = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_mf1 * delta_stage1_mf

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
