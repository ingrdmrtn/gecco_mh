def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    TD(lambda) Learner.
    
    Uses eligibility traces to update Stage 1 values based on Stage 2 outcomes directly.
    This creates a form of model-free learning that bridges the delay between 
    Stage 1 choice and final reward more effectively than simple TD(0).
    
    Parameters:
    learning_rate: [0,1] Rate for updating Q-values.
    beta: [0,10] Inverse temperature for softmax.
    lam: [0,1] Decay rate for eligibility traces (0=TD(0), 1=Monte Carlo).
    """
    learning_rate, beta, lam = model_parameters
    n_trials = len(action_1)

    q_stage1 = np.zeros(2) + 0.5
    q_stage2 = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        a1 = action_1[trial]
        state_idx = state[trial]

        exp_q2 = np.exp(beta * q_stage2[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]


        delta_1 = q_stage2[state_idx, a2] - q_stage1[a1]

        delta_2 = r - q_stage2[state_idx, a2]


        q_stage1[a1] += learning_rate * delta_1 + learning_rate * lam * delta_2

        q_stage2[state_idx, a2] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
