Here are three cognitive models based on the provided template, ranging from a pure Model-Based approach to a hybrid Model-Based/Model-Free reinforcement learning model, and finally a model incorporating perseveration.

### Model 1: Pure Model-Based Learning
This model assumes the participant builds a mental model of the task structure (the transition probabilities) and plans their first-stage actions by looking ahead to the expected value of the second stage. It ignores direct reinforcement of the first-stage choice (Model-Free learning) and relies entirely on the computed values of the planets.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Reinforcement Learning Model.
    
    This model assumes the agent calculates the value of stage 1 actions solely by 
    combining the known transition matrix with the learned values of the second-stage options.
    It does not maintain a separate Model-Free Q-value for the first stage.

    Parameters:
    learning_rate: [0, 1] - Rate at which second-stage Q-values are updated.
    beta: [0, 10] - Inverse temperature for softmax choice (exploration/exploitation).
    """
    learning_rate, beta = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix: 
    # Row 0 (Spaceship 0) -> 0.7 prob Planet 0, 0.3 prob Planet 1
    # Row 1 (Spaceship 1) -> 0.3 prob Planet 0, 0.7 prob Planet 1
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values for Stage 2 (2 states/planets, 2 actions/aliens)
    q_stage2_mf = np.zeros((2, 2)) # Initialized at 0.5 (neutral expectation) or 0

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE (Model-Based) ---
        # Calculate the max value available at each planet (state)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Calculate Model-Based values for Stage 1 actions
        # V(Action1) = P(State0|Action1)*MaxQ(State0) + P(State1|Action1)*MaxQ(State1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Softmax policy for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # The planet actually reached

        # --- STAGE 2 CHOICE ---
        # Standard Q-learning choice based on values of aliens at the current planet
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATING ---
        # Update Stage 2 Q-values based on reward received
        # Q(s, a) = Q(s, a) + alpha * (reward - Q(s, a))
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: In a pure MB model, we do not update a separate Stage 1 Q-value directly 
        # from reward. The Stage 1 value changes automatically because q_stage2_mf changed.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model-Based / Model-Free Learning
This is the classic "Daw et al. (2011)" style two-step model. It assumes the participant uses a weighted combination of Model-Based planning (using the transition matrix) and Model-Free learning (reinforcing the first choice based on the second choice's value). A mixing parameter `w` determines the balance between these two systems.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning Model.
    
    This model combines a habitual (Model-Free) system that repeats rewarded actions
    with a goal-directed (Model-Based) system that plans using the transition structure.
    
    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting parameter (0 = Pure Model-Free, 1 = Pure Model-Based).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Model-Free Q-values for Stage 1 (2 actions)
    q_stage1_mf = np.zeros(2)
    # Model-Free Q-values for Stage 2 (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- STAGE 1 CHOICE ---
        # 1. Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Integrate MB and MF values using weight w
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- STAGE 2 CHOICE ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATING ---
        # 1. Update Stage 1 MF values using SARSA-style update (TD(1) logic often used here, 
        # but here we use the value of the chosen stage 2 option as the target)
        # Target is Q_stage2(state, action2)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 2. Update Stage 2 MF values based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: Often a second update is applied to Stage 1 based on the reward (eligibility trace),
        # but standard TD(0) updates are implemented here for simplicity within the template constraints.
        # To make it strictly TD(1) or Lambda, we would add learning_rate * lambda * delta_stage2 to stage 1.
        # Here we stick to the template's implication of direct updates.
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2 # simple eligibility trace accumulation

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Choice Perseveration
This model extends the Hybrid approach by adding a "stickiness" or perseveration parameter. Participants often have a tendency to repeat their previous choice regardless of reward (motor perseveration). This model captures that tendency, which often improves model fit significantly in this task.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Model with Choice Perseveration.
    
    Adds a 'stickiness' parameter to the first stage choice. If the participant 
    chose Spaceship A on the last trial, they receive a "bonus" to the value of Spaceship A
    on the current trial, independent of learning.
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weighting.
    perseveration: [0, 5] - Bonus added to the previously chosen action (stickiness).
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):

        # --- STAGE 1 CHOICE ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Perseveration Bonus
        # We create a temporary Q-value vector for decision making
        q_decision = q_net.copy()
        if last_action_1 != -1:
            q_decision[last_action_1] += perseveration

        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- STAGE 2 CHOICE ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATING ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace update for stage 1
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2
        
        # Store action for next trial's perseveration
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
