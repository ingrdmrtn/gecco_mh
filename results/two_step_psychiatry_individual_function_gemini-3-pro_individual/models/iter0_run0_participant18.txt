Here are three cognitive models proposing different mechanisms for how the participant makes decisions in the two-step task.

### Model 1: Hybrid Model-Based/Model-Free Reinforcement Learning
This is the standard "Daw et al. (2011)" model structure, combining a Model-Free (MF) system that learns from direct experience and a Model-Based (MB) system that uses knowledge of the transition structure. This model assumes the participant weighs these two strategies using a mixing parameter `w`.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Standard Hybrid Model-Based / Model-Free RL model.
    
    This model assumes the agent computes Q-values for the first stage using a weighted 
    average of Model-Based (planning using transition matrix) and Model-Free (TD learning) values.
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated (alpha).
    - beta: [0, 10] Inverse temperature for softmax choice (exploration/exploitation).
    - w: [0, 1] Mixing weight. 1 = Pure Model-Based, 0 = Pure Model-Free.
    - lam: [0, 1] Eligibility trace parameter (lambda) connecting stage 2 to stage 1 updates.
    """
    learning_rate, beta, w, lam = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix: spaceship 0 -> planet 0 (0.7), spaceship 1 -> planet 1 (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1 (spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (aliens)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based value calculation: V(state) = max_a Q(state, a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q-value: w * MB + (1-w) * MF
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax policy for stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Observe state (planet)
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Choice ---
        # Standard Q-learning for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]

        # --- Updates ---
        # Stage 1 prediction error (SARSA-style, using Q-value of chosen stage 2 action)
        # Note: Some variants use max Q, here we stick to the chosen action for TD(lambda)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Stage 2 prediction error
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Update Stage 1 MF value: alpha * (delta1 + lambda * delta2)
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lam * delta_stage2)
        
        # Update Stage 2 MF value
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free with Perseveration
This model hypothesizes that the participant does not use the transition structure (Model-Based) at all. Instead, they rely purely on Model-Free reinforcement learning but are heavily influenced by "perseveration"â€”a tendency to repeat the previous choice regardless of reward. This is often observed in habit-driven behaviors.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Q-learning with Choice Perseveration.
    
    This model assumes the agent ignores the transition probabilities (w=0).
    It adds a 'perseveration' bonus to the previously chosen action at Stage 1,
    capturing stickiness or habit not explained by reward history.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - perseveration: [-5, 5] Bonus added to the Q-value of the previously chosen action. 
      (Positive = sticky, Negative = alternation).
    """
    learning_rate, beta, perseveration = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Indicator for previous action

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        # Calculate logits: Q-value + Perseveration Bonus
        logits_stage1 = q_stage1_mf.copy()
        if last_action_1 != -1:
            logits_stage1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * logits_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]
  
        # --- Updates ---
        # Pure TD(0) updates (no eligibility trace for simplicity in this variant)
        # The agent updates stage 1 based on the value of the state they arrived at.
        # Using max_q_stage2 for Stage 1 update (Q-learning)
        max_q_next = np.max(q_stage2_mf[state_idx])
        
        delta_stage1 = max_q_next - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Store action for next trial's perseveration
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning Rates (Positive vs Negative Prediction Errors)
This model investigates if the participant learns differently from success versus failure. It splits the learning rate into `alpha_pos` (for positive prediction errors) and `alpha_neg` (for negative prediction errors). It assumes a pure Model-Free framework to isolate the effect of asymmetric updating.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Asymmetric Learning Rates.
    
    This model separates the learning rate for positive prediction errors (better than expected)
    and negative prediction errors (worse than expected). This can capture optimism/pessimism bias.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors (delta > 0).
    - alpha_neg: [0, 1] Learning rate for negative prediction errors (delta < 0).
    - beta: [0, 10] Inverse temperature.
    """
    alpha_pos, alpha_neg, beta = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]
  
        # --- Updates ---
        # We use simple TD(0) chaining logic
        # Stage 1 Update
        # Value of the state reached (V(s') = max Q(s', a'))
        v_next_state = np.max(q_stage2_mf[state_idx])
        delta_stage1 = v_next_state - q_stage1_mf[a1]
        
        if delta_stage1 > 0:
            q_stage1_mf[a1] += alpha_pos * delta_stage1
        else:
            q_stage1_mf[a1] += alpha_neg * delta_stage1
            
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        if delta_stage2 > 0:
            q_stage2_mf[state_idx, a2] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[state_idx, a2] += alpha_neg * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
