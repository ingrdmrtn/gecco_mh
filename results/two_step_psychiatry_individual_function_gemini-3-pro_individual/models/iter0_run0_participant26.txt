Here are three cognitive models based on the two-step decision task, implementing different hypotheses about how the participant learns.

### Model 1: Pure Model-Based Reinforcement Learning
This model assumes the participant builds an internal model of the task structure (the transition matrix) and uses it to plan their first-stage choices. They calculate the expected value of each spaceship by combining the transition probabilities with the learned values of the aliens. It ignores model-free learning at the first stage.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Reinforcement Learning.
    
    Hypothesis: The participant relies entirely on the known transition structure 
    to calculate expected values for the first stage, rather than learning 
    first-stage values directly from experience.

    Parameters:
    learning_rate: [0, 1] - Rate at which second-stage (alien) values are updated.
    beta_1: [0, 10] - Inverse temperature for stage 1 choice (exploration/exploitation).
    beta_2: [0, 10] - Inverse temperature for stage 2 choice.
    """
    learning_rate, beta_1, beta_2 = model_parameters
    n_trials = len(action_1)

    # Fixed transition matrix as per task description (A->X mostly, U->Y mostly)
    # Rows: Spaceship A (0), Spaceship U (1)
    # Cols: Planet X (0), Planet Y (1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values for the second stage (aliens): 2 states (planets) x 2 actions (aliens)
    q_stage2 = np.zeros((2, 2)) 

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Calculate Model-Based values: Q_MB(s1, a1) = Sum(P(s2|s1,a1) * max(Q(s2, a2)))
        max_q_stage2 = np.max(q_stage2, axis=1) # Best value available on each planet
        q_stage1_mb = transition_matrix @ max_q_stage2 # Expected value of each spaceship

        # Softmax for Stage 1
        exp_q1 = np.exp(beta_1 * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Decision ---
        state_idx = state[trial] # Current planet
        
        # Softmax for Stage 2 (based on Q-values of aliens on this planet)
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Only update Stage 2 values based on reward (Model-Free update)
        pe_stage2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * pe_stage2
        
        # Note: No direct update for Stage 1 Q-values; they are re-calculated 
        # via the model (MB) at the start of the next trial.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free Reinforcement Learning (TD(1))
This model assumes the participant does *not* use the transition structure. Instead, they learn the value of spaceships directly from the rewards obtained at the very end of the trial. This is equivalent to a TD(1) or Monte Carlo approach where the final reward reinforces the first-stage choice directly, regardless of the transition type (common or rare).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Reinforcement Learning (Direct Reinforcement).
    
    Hypothesis: The participant ignores the transition structure and learns
    first-stage values solely based on the final reward received, treating
    the task as a simple chain of actions.

    Parameters:
    learning_rate: [0, 1] - Rate at which Q-values are updated.
    beta: [0, 10] - Inverse temperature for both stages (shared noise parameter).
    eligibility_lambda: [0, 1] - How much credit the first stage gets for the second stage reward.
    """
    learning_rate, beta, eligibility_lambda = model_parameters
    n_trials = len(action_1)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values for stage 1 (spaceships) and stage 2 (aliens)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Prediction error at stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 2 Q-values
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values
        # In this simple MF model, we can treat the Stage 1 update as driven 
        # by the Stage 2 value (TD(0)) plus the eligibility trace of the final reward (TD(1))
        # Here we implement a simplified update where Stage 1 is updated by the final reward directly
        # scaled by lambda.
        
        # Standard TD(0) error for stage 1: (Value of state 2) - (Value of state 1)
        # Note: Reward at stage 1 is 0.
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Update Stage 1 using both the immediate transition error and the final outcome error
        # This combines TD(0) and TD(1) logic via eligibility traces conceptually.
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1 + \
                                        learning_rate * eligibility_lambda * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model-Based / Model-Free (Dyna-style)
This model represents the classic "Two-Step" analysis hypothesis. The participant computes values for the first stage using a weighted combination of a Model-Based strategy (planning using the transition matrix) and a Model-Free strategy (learning from direct experience). A mixing parameter `w` determines the balance between these two systems.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Learning.
    
    Hypothesis: Decision making is a mixture of two systems: a Model-Based system
    that uses the transition matrix and a Model-Free system that uses TD errors.
    A weighting parameter `w` controls the dominance of the MB system.

    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta: [0, 10] - Inverse temperature for choices.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # 1. Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value Calculation
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        # Stage 2 is purely model-free (finding the best alien)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Learning ---
        # 1. Calculate TD Errors
        # Stage 1 error: (Value of State 2) - (Value of State 1)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Stage 2 error: (Reward) - (Value of State 2)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # 2. Update Values
        # Update Stage 1 MF value (SARSA-style update)
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 MF value
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
