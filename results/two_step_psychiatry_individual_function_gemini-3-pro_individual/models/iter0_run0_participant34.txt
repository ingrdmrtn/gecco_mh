Here are three cognitive models based on the provided template, ranging from pure model-based learning to a hybrid model-based/model-free reinforcement learning approach.

### Model 1: Pure Model-Based Learner with Decay
This model assumes the participant builds an internal model of the transition structure (Spaceship -> Planet) and values the first-stage choices purely by planning forward from the second-stage values. It includes a decay parameter for unchosen second-stage options, simulating forgetting.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner with Value Decay.
    
    This model assumes the agent calculates Stage 1 values entirely by combining 
    the transition matrix with the max values of Stage 2. It does not use a 
    Model-Free update for Stage 1 (TD-learning). It includes a decay rate for 
    unchosen Stage 2 actions.

    Parameters:
    learning_rate: [0, 1] - Rate at which Stage 2 Q-values are updated from rewards.
    decay_rate: [0, 1] - Rate at which unchosen Stage 2 Q-values decay toward 0.
    beta: [0, 10] - Inverse temperature for softmax choice (exploration/exploitation).
    """
    learning_rate, decay_rate, beta = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as per task description
    # Row 0: Spaceship A -> 70% Planet X (0), 30% Planet Y (1)
    # Row 1: Spaceship U -> 30% Planet X (0), 70% Planet Y (1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values for Stage 2 (2 states/planets, 2 actions/aliens)
    q_stage2_mf = np.zeros((2, 2)) # Initialized at 0.5 (neutral)

    for trial in range(n_trials):
        # --- Policy for the first choice (Pure Model-Based) ---
        # Calculate expected value of each planet (state)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Plan: Probability of reaching state * Value of state
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # The planet actually reached

        # --- Policy for the second choice ---
        # Standard value-based selection given the current state
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # No direct Stage 1 Q-value update (Pure MB uses calculated values)
        
        # Update Stage 2 Q-values based on reward (Model-Free)
        pe = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe
        
        # Decay unchosen actions in Stage 2
        # This simulates forgetting the value of aliens not visited
        chosen_alien = action_2[trial]
        unchosen_alien = 1 - chosen_alien
        q_stage2_mf[state_idx, unchosen_alien] *= (1 - decay_rate)
        
        # Also decay values for the unvisited planet
        unvisited_planet = 1 - state_idx
        q_stage2_mf[unvisited_planet, :] *= (1 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model-Based/Model-Free Learner (Standard "Daw" Model)
This model implements the classic hybrid theory where the agent's Step 1 decision is a weighted combination of a Model-Based strategy (planning using the transition matrix) and a Model-Free strategy (simple TD-learning from previous rewards).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Learner.
    
    The agent computes Stage 1 values using both a planning approach (MB) and 
    a caching approach (MF). The final choice is a weighted mixture of both.
    
    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Model-Free Q-values
    q_stage1_mf = np.zeros(2)      # Values for Spaceships A/U
    q_stage2_mf = np.zeros((2, 2)) # Values for Aliens (Planet 0, Planet 1)

    for trial in range(n_trials):

        # --- Policy for the first choice ---
        # 1. Model-Based Value: Plan using transition matrix + Max Stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value: Weighted sum of MB and MF values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Policy for the second choice ---
        # Standard Softmax on Stage 2 values
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # 1. Update Stage 1 MF value (TD(0) - driven by value of state reached)
        # Note: In full SARSA(lambda) this might use the Q-value of the next action, 
        # but here we use the max Q of the state reached (Q-learning style) or just 0 initialization.
        # We calculate prediction error based on the value of the state we landed in.
        value_of_state_reached = np.max(q_stage2_mf[state_idx])
        delta_stage1 = value_of_state_reached - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # 2. Update Stage 2 MF value based on Reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # 3. Eligibility Trace / Second Stage Update to Stage 1 (TD(1))
        # This allows the reward at the very end to update the spaceship choice directly
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Perseveration Model (Habitual Learner)
This model assumes the participant is purely Model-Free but exhibits "perseveration" (or stickiness). They are more likely to repeat their previous Stage 1 choice regardless of the reward, captured by a `perseveration` parameter. This captures motor habits or cognitive inertia often seen in these tasks.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Perseveration (Stickiness).
    
    This model relies purely on past rewards to value Step 1 (no transition matrix planning),
    but adds a 'stickiness' bonus to the previously chosen Step 1 action.
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    perseveration: [0, 5] - Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, perseveration = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix is unused in pure MF logic, but defined for consistency
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize as 'none'

    for trial in range(n_trials):

        # --- Policy for the first choice ---
        # Base values come from Model-Free learning
        q_net = q_stage1_mf.copy()
        
        # Add perseveration bonus if a previous action exists
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Record action for next trial's perseveration
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Policy for the second choice ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Update Stage 1 based on the value of the state reached (TD(0))
        # We use the max value of the next state as the proxy for state value
        v_next_state = np.max(q_stage2_mf[state_idx])
        delta_stage1 = v_next_state - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Direct reinforcement of Stage 1 from reward (TD(1) logic)
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
