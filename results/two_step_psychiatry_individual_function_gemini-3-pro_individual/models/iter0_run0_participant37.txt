Here are three cognitive models representing different strategies for the two-step task.

### Model 1: Pure Model-Free Learner (TD-Learning)
This model assumes the participant does not build a model of the transition structure (i.e., they ignore that Spaceship A usually goes to Planet X). Instead, they learn purely from experience using Temporal Difference (TD) learning. They update the value of the first-stage choice based on the value of the state they landed in, and update the second-stage choice based on the reward received.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Learner using SARSA(0) / TD-learning.
    The agent updates stage 1 values based on the value of the chosen stage 2 option,
    without using the transition matrix (Model-Based system is absent).
    
    Parameters:
    learning_rate: [0,1] Rate at which prediction errors update values.
    beta: [0,10] Inverse temperature for softmax choice rule (exploration vs exploitation).
    """
    learning_rate, beta = model_parameters
    n_trials = len(action_1)
  
    # Q-values for Stage 1 (2 actions) and Stage 2 (2 states x 2 actions)
    q_stage1_mf = np.zeros(2) + 0.5  # Initialize with neutral expectation
    q_stage2_mf = np.zeros((2, 2)) + 0.5

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # --- Stage 1 Decision ---
        # Standard softmax on Model-Free Q-values
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Decision ---
        # Softmax on Stage 2 Q-values specific to the current state (planet)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]

        # --- Learning ---
        
        # Update Stage 1: Driven by the value of the action taken in Stage 2 (SARSA style)
        # Prediction Error 1 = Q(s2, a2) - Q(s1, a1)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Update Stage 2: Driven by the actual reward received
        # Prediction Error 2 = Reward - Q(s2, a2)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] = q_stage2_mf[state_idx, a2] + learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Based Learner
This model assumes the participant fully utilizes the transition structure. They do not maintain a separate "cache" of values for the spaceships (Stage 1). Instead, on every trial, they calculate the value of a spaceship by looking at the best options available on the planets it leads to, weighted by the transition probabilities (0.7 vs 0.3).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner.
    Stage 1 values are computed on-the-fly by projecting Stage 2 values 
    through the transition matrix. Stage 2 values are learned via Rescorla-Wagner.
    
    Parameters:
    learning_rate: [0,1] Rate at which reward updates Stage 2 values.
    beta: [0,10] Inverse temperature for softmax.
    """
    learning_rate, beta = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix: Row 0 is Spaceship A -> [70% Planet X, 30% Planet Y]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Only Stage 2 values are stored directly. Stage 1 is derived.
    q_stage2_mf = np.zeros((2, 2)) + 0.5 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # --- Stage 1 Decision ---
        # 1. Find max value available on each planet
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # 2. Compute Model-Based value for spaceships: T * V(states)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 3. Softmax choice
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        # Standard softmax based on learned alien values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Update Stage 2 values based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: No separate Stage 1 update needed because q_stage1_mb is re-calculated 
        # fresh every trial based on the updated q_stage2_mf.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model (Model-Based + Model-Free + Perseveration)
This is the classic "Daw" model. It assumes the agent computes both a Model-Based value (planning) and a Model-Free value (habit) for the first stage and combines them using a weighting parameter `w`. It also includes a "perseveration" parameter, which captures the tendency to simply repeat the last choice regardless of reward.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model (MB/MF) with Choice Perseveration.
    Combines Model-Based planning and Model-Free learning for Stage 1 choices.
    Includes a 'stickiness' parameter for repeating the previous Stage 1 action.
    
    Parameters:
    learning_rate: [0,1] Update rate for Q-values.
    beta: [0,10] Inverse temperature.
    w: [0,1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    perseveration: [0,5] Bonus added to the previously chosen Stage 1 action (stickiness).
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    last_action_1 = -1 # No previous action for the first trial

    for trial in range(n_trials):

        # --- Stage 1 Decision ---
        
        # 1. Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value calculation: weighted sum of MB and MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # 3. Add Perseveration Bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # 4. Softmax
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]
  
        # --- Learning ---
        
        # MF Update Stage 1 (SARSA)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # MF Update Stage 2 (Rescorla-Wagner)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Track last action for perseveration
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
