Here are three new cognitive models that explore different mechanisms for solving the two-step task. These models introduce variations on how the first-stage values are calculated (hybrid learning) and how second-stage values are updated (counterfactual learning).

### Model 1: Hybrid Model (Model-Based + Model-Free)
This model combines model-based (planning using the transition matrix) and model-free (TD learning) strategies. The parameter `w` controls the balance between these two systems for the first-stage choice. It assumes the participant uses both a "habitual" system (MF) and a "goal-directed" system (MB).

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning.
    
    This model assumes the agent uses a weighted combination of model-based (planning)
    and model-free (habitual) values to make the first-stage choice.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weight parameter. 1 = Pure Model-Based, 0 = Pure Model-Free.
    - lambda_param: [0, 1] Eligibility trace parameter (controls stage 1 update from stage 2).
    """
    learning_rate, beta, w, lambda_param = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as per task description
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 states (planets), 2 actions (aliens)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: V(state) = max(Q(state, action))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_mb + (1-w) * Q_mf
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # Actual planet arrived at

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Learning ---
        # 1. Prediction error at stage 2 (Reward - Q_stage2)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # 2. Prediction error at stage 1 (Q_stage2 - Q_stage1)
        # Note: We use the value of the chosen stage 2 action as the target
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 3. Eligibility trace update (TD-lambda logic)
        # Stage 1 value is also nudged by the final reward prediction error
        q_stage1_mf[action_1[trial]] += learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseveration Model
This model is a pure Model-Free learner (TD(0)) but adds a `perseveration` parameter. This parameter captures the tendency to repeat (or switch from) the previous first-stage action regardless of the reward, a common behavior in sequential decision-making tasks.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free TD(0) Learning with Choice Perseveration.
    
    This model relies on temporal difference learning but includes a 'stickiness'
    parameter that biases the agent to repeat their previous stage-1 choice.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - perseveration: [0, 5] Bonus added to the previously chosen action (positive = sticky).
    """
    learning_rate, beta, perseveration = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with an impossible action

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        # Calculate logits: Q-value + perseveration bonus
        logits_1 = beta * q_stage1_mf.copy()
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration

        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Learning (TD-0) ---
        # Stage 1 update based on state transition (value of stage 2 state)
        # We use max(Q_stage2) as the proxy for the value of the state arrived at
        v_stage2_state = np.max(q_stage2_mf[state_idx])
        delta_stage1 = v_stage2_state - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 update based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Counterfactual Update Model
This model introduces a `counterfactual_alpha` parameter. In the second stage, when the participant chooses one alien and observes the reward, this model also updates the value of the *unchosen* alien, assuming a symmetrical structure (if chosen was good, unchosen might be bad, or simply decaying unchosen values). Here, we implement a decay/forgetting mechanism for the unchosen option.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learning with Counterfactual Updating (Forgetting).
    
    In stage 2, the chosen action is updated with the standard learning rate.
    The unchosen action in the same state decays towards 0 (or is updated counterfactually)
    controlled by a separate parameter.
    
    Parameters:
    - learning_rate: [0, 1] Standard update rate for chosen actions.
    - beta: [0, 10] Inverse temperature.
    - alpha_unchosen: [0, 1] Decay rate for the unchosen action in stage 2.
    """
    learning_rate, beta, alpha_unchosen = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Learning ---
        
        # Update Chosen Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Update Unchosen Stage 2 (Counterfactual / Forgetting)
        # The unchosen option decays toward a neutral value (e.g., 0.5 or 0)
        # Here we decay toward 0.5 (assuming uncertain probability) or just 0.
        # Let's implement decay toward 0.5 (chance level)
        unchosen_action = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action] += alpha_unchosen * (0.5 - q_stage2_mf[state_idx, unchosen_action])

        # Update Stage 1 (TD-0 style)
        # Using the value of the state reached (max Q)
        v_next = np.max(q_stage2_mf[state_idx])
        delta_stage1 = v_next - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
