Here are three new cognitive models for the two-step task, exploring different mechanisms like separate learning rates for positive/negative outcomes, pure model-free learning with eligibility traces (SARSA-like), and a simpler hybrid model without stickiness but with a dynamic learning rate.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Learning Rate Model (Model-Free).
    
    This model assumes the participant is purely Model-Free (habit-based) but learns 
    differently from positive (reward = 1) versus negative (reward = 0) outcomes.
    This captures potential optimism or pessimism biases.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for rewarded trials.
    - alpha_neg: [0, 1] Learning rate for unrewarded trials.
    - beta: [0, 10] Inverse temperature for choice stochasticity.
    - eligibility: [0, 1] Eligibility trace decay (lambda). How much the Stage 1 choice 
                   is updated by the Stage 2 prediction error.
    """
    alpha_pos, alpha_neg, beta, eligibility = model_parameters
    n_trials = len(action_1)
  
    # Q-values for Stage 1 (2 actions) and Stage 2 (2 states x 2 actions)
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) # State 0 (Planet X), State 1 (Planet Y)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]

        # --- Learning ---
        # Select learning rate based on reward outcome
        lr = alpha_pos if r == 1 else alpha_neg

        # Stage 2 Prediction Error (RPE)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Stage 1 Prediction Error (using Stage 2 Q-value as proxy for value of state)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]

        # Update Stage 2 Q-values
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Update Stage 1 Q-values: 
        # Direct update from transition (TD(0)) + eligibility trace update from Stage 2 RPE (TD(1)-ish)
        q_stage1_mf[a1] += lr * delta_stage1 + (lr * eligibility * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner with Forgetting.
    
    This model assumes the participant builds a model of the task structure to calculate
    values (Model-Based) but forgets the value of unchosen options or past rewards 
    over time. This reflects limited working memory capacity.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for observed rewards.
    - beta: [0, 10] Inverse temperature.
    - forgetting_rate: [0, 1] Decay rate for Q-values of all options on every trial.
    """
    learning_rate, beta, forgetting_rate = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values for Stage 2 aliens (the 'model' of reward probabilities)
    # Initialized to 0.5 (neutral expectation)
    q_stage2_mb = np.ones((2, 2)) * 0.5 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # --- Stage 1 Choice (Planning) ---
        # Calculate max value available in each state (planet)
        max_val_stage2 = np.max(q_stage2_mb, axis=1)
        
        # Plan: Expected value of action 1 = P(State|Action) * Value(State)
        q_stage1_mb = transition_matrix @ max_val_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a2 = action_2[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mb[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning & Forgetting ---
        
        # 1. Decay/Forget ALL Q-values slightly towards 0 (or neutral)
        q_stage2_mb = (1 - forgetting_rate) * q_stage2_mb
        
        # 2. Update the specific chosen alien's value based on reward
        # Standard Rescorla-Wagner update
        delta = reward[trial] - q_stage2_mb[state_idx, a2]
        q_stage2_mb[state_idx, a2] += learning_rate * delta
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Stage-Specific Betas.
    
    A standard hybrid (Model-Based + Model-Free) learner, but with distinct 
    exploration/exploitation parameters (betas) for the first stage (spaceship choice)
    and the second stage (alien choice). This tests if decision noise differs between 
    the high-level planning stage and the low-level harvesting stage.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta_stage1: [0, 10] Inverse temperature for Stage 1.
    - beta_stage2: [0, 10] Inverse temperature for Stage 2.
    - omega: [0, 1] Weighting: 1 = Pure Model-Based, 0 = Pure Model-Free.
    """
    learning_rate, beta_stage1, beta_stage2, omega = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Hybrid Value
        q_net = (omega * q_stage1_mb) + ((1 - omega) * q_stage1_mf)

        # Softmax with Stage 1 specific beta
        exp_q1 = np.exp(beta_stage1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]

        # --- Stage 2 Choice ---
        # Softmax with Stage 2 specific beta
        exp_q2 = np.exp(beta_stage2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # TD(0) update for Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # TD(0) update for Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Note: In this simple hybrid formulation, we often assume the MB system 
        # uses the MF Stage 2 values as its estimate of state value, so we don't
        # maintain a separate table for MB rewards, just reuse q_stage2_mf.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
