Here are three new cognitive models exploring different mechanisms for the two-step task, distinct from the pure model-based approach previously provided.

### Model 1: Pure Model-Free Learner (TD-Learning)
This model assumes the participant does not use the transition structure of the task at all. Instead, it relies entirely on temporal difference (TD) learning (SARSA/Q-learning logic). It updates stage 1 values based on stage 2 values, and stage 2 values based on rewards. This is a classic baseline in reinforcement learning.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Learner (TD-Learning).
    Updates values based solely on experienced reward prediction errors, ignoring
    the transition structure (probabilities of planets given spaceships).
    
    Bounds:
    learning_rate: [0,1] - Update rate for both stages.
    beta: [0,10] - Inverse temperature for both stages (shared noise).
    eligibility_trace: [0,1] - How much stage 2 reward updates stage 1 choice directly (lambda).
    """
    learning_rate, beta, eligibility_trace = model_parameters
    n_trials = len(action_1)
  
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # Values for spaceships
    q_stage2_mf = np.zeros((2, 2)) # Values for aliens (2 planets x 2 aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Current state (planet)
        state_idx = state[trial]
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # 1. Prediction error at stage 2 (Reward - Q_stage2)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # 2. Prediction error at stage 1 (Q_stage2 - Q_stage1)
        # Note: Using the value of the CHOSEN stage 2 action (SARSA-like)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Update Stage 1 Value
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 1 Value again based on Stage 2 RPE (Eligibility Trace)
        q_stage1_mf[action_1[trial]] += learning_rate * eligibility_trace * delta_stage2
        
        # Update Stage 2 Value
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Learner (Mixture of MB and MF)
This is the standard analysis model for this task (often called the Daw 2011 model). It assumes the participant computes both a Model-Based value (using the transition matrix) and a Model-Free value (using TD errors) for the first stage choice, and combines them using a mixing weight parameter (`w`).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Learner.
    Combines Q_MB and Q_MF for the first stage decision using a mixing weight 'w'.
    
    Bounds:
    learning_rate: [0,1] - Learning rate for Q-value updates.
    beta: [0,10] - Inverse temperature for softmax.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Calculation ---
        # Model-Based Value: Transition * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value: weighted sum of MB and MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Stage 2 RPE
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Stage 1 RPE (SARSA style)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Update MF values
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        # Note: Typically in Hybrid models, stage 1 also gets an eligibility trace update from stage 2
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2 
        
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Choice Kernel / Perseveration Model
This model hypothesizes that the participant is driven less by value learning and more by simple repetition or "stickiness." It adds a "choice kernel" (perseveration) parameter. The model tracks which action was taken previously and adds a bonus to that action's value in the next trial, decaying over time. This is crucial for explaining the participant's data, which shows repetitive choices.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Choice Perseveration (Stickiness).
    Includes a 'perseveration' parameter that biases the agent to repeat 
    the previously chosen action, regardless of reward.
    
    Bounds:
    learning_rate: [0,1] - Update rate for values.
    beta: [0,10] - Inverse temperature.
    perseveration: [0,5] - Bonus added to the logits of the previously chosen action.
    """
    learning_rate, beta, perseveration = model_parameters
    n_trials = len(action_1)
  
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choices (initialized to -1 meaning none)
    prev_action_1 = -1
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        # Base logits from value
        logits_1 = beta * q_stage1_mf
        
        # Add perseveration bonus if previous action exists
        if prev_action_1 != -1:
            logits_1[prev_action_1] += perseveration
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store choice for next trial
        prev_action_1 = action_1[trial]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        # (Standard softmax on Q-values for stage 2)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Standard TD updating
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: Eligibility trace omitted here to isolate the effect of perseveration

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
