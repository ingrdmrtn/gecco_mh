Here are 3 new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Model-Based / Model-Free RL (Classic "Daw" Model)
This is the canonical model for this task (often called the Daw et al. 2011 model). It assumes the agent uses a weighted combination of Model-Based (planning) and Model-Free (habitual) values to make the first-stage choice. The parameter `w` controls the balance: `w=1` is pure model-based, `w=0` is pure model-free.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning.
    
    This model computes stage 1 values as a weighted sum of Model-Based (MB) values
    (calculated via the transition matrix) and Model-Free (MF) values (cached from experience).
    Stage 2 is purely Model-Free.

    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature for choice stochasticity.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Q-values
    q_stage1_mf = np.zeros(2)       # Model-free values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2))  # Model-free values for stage 2 (Aliens on Planets)

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # 1. Calculate Model-Based values: V(state) = max(Q(state, action))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2 
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # The planet actually reached

        # --- Stage 2 Policy ---
        q_values_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_values_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updating ---
        # SARSA(0) / TD(0) update for Stage 1 MF
        # The reward for stage 1 is the value of the state reached in stage 2
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Prediction error for Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: In the full Daw model, there is often an eligibility trace connecting 
        # the stage 2 reward back to stage 1. This version simplifies to TD(0) at stage 1 
        # to fit the 3-parameter constraint requested alongside the hybrid logic.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free with Eligibility Traces (TD(lambda))
This model abandons the transition matrix knowledge entirely. Instead, it relies on a mechanism to credit the first action (choosing a spaceship) with the ultimate outcome (gold) via an eligibility trace parameter `lambda`. This allows "direct" learning from outcome to the first choice without explicit planning.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free RL with Eligibility Traces (TD-lambda).
    
    This model does not use the transition matrix. It learns stage 1 values solely through
    experience. The 'lambda' parameter controls how much the stage 2 reward directly 
    updates the stage 1 choice (connecting the outcome back to the start).

    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature.
    lambda_eligibility: [0,1] - Decay rate of eligibility trace (0 = TD(0), 1 = Monte Carlo).
    """
    learning_rate, beta, lambda_eligibility = model_parameters
    n_trials = len(action_1)
  
    # transition_matrix is NOT used here.
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # Planet reached

        # --- Stage 2 Policy ---
        q_values_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_values_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updating ---
        
        # 1. Update Stage 1 based on Stage 2 value (TD error at transition)
        # Prediction: Q(s1, a1), Target: Q(s2, a2)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 2. Update Stage 2 based on Reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # 3. Eligibility Trace Update: Pass the stage 2 prediction error back to stage 1
        # This allows the reward to influence the spaceship choice immediately
        q_stage1_mf[action_1[trial]] += learning_rate * lambda_eligibility * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Choice Perseveration (stickiness) + Pure Model-Based
This model adds a "perseveration" or "stickiness" parameter to the Pure Model-Based learner. This parameter captures the tendency to repeat the previous action regardless of reward history or model structure, a common phenomenon in behavioral data.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based RL with Choice Perseveration (Stickiness).
    
    This model assumes the agent plans using the transition matrix, but also has a 
    heuristic bias to repeat the last chosen action at Stage 1.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature.
    perseveration: [0,5] - Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Only need Stage 2 Q-values for MB calculation
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with an invalid action

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add perseveration bonus
        # We create a temporary Q vector for decision making
        q_decision = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_decision[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update tracker
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        q_values_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_values_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updating ---
        # Standard Q-learning for the second stage
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
