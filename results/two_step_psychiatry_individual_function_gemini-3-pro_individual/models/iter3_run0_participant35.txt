Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task. These models introduce variations in learning rates, perseveration, and noise to capture different cognitive strategies.

### Model 1: Separate Learning Rates for Positive and Negative Outcomes
This model extends the standard hybrid learner by allowing different learning rates for rewarded (`learning_rate_pos`) and unrewarded (`learning_rate_neg`) trials. This captures the psychological phenomenon where individuals may learn differently from success versus failure.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Asymmetric Learning Rates.
    
    This model distinguishes between learning from positive outcomes (wins) and 
    negative outcomes (losses/omissions). It allows for a bias in how strongly 
    good vs. bad news updates value expectations.
    
    Parameters:
    learning_rate_pos: [0, 1] Update rate for rewarded trials.
    learning_rate_neg: [0, 1] Update rate for unrewarded trials.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    """
    learning_rate_pos, learning_rate_neg, beta, w = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)       
    q_stage2_mf = np.zeros((2, 2))  

    for trial in range(n_trials):
        # 1. Stage 1 Choice Policy (Hybrid)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # 2. Stage 2 Choice Policy (Standard Softmax)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # 3. Learning
        # Determine which learning rate to use based on the reward received
        if reward[trial] > 0:
            lr = learning_rate_pos
        else:
            lr = learning_rate_neg

        # TD(1) update logic
        # Stage 1 TD error (driven by Stage 2 value prediction)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1

        # Stage 2 TD error (driven by actual reward)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2

        # Eligibility trace: Stage 1 value is also updated by the Stage 2 prediction error
        q_stage1_mf[action_1[trial]] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Choice Perseveration at Both Stages
This model introduces a "stickiness" or perseveration parameter (`perseveration`) that applies to *both* the first-stage spaceship choice and the second-stage alien choice. Unlike previous models that might only apply perseveration to the first stage, this assumes a general motor or cognitive inertia that biases the participant to repeat their immediately preceding action at every decision point.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Multi-Stage Perseveration.
    
    Adds a 'stickiness' bonus to the choice made on the previous trial, regardless 
    of reward. This perseveration applies to both the Stage 1 (spaceship) and 
    Stage 2 (alien) decisions.
    
    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    perseveration: [0, 5] Bonus added to the logits of the previously chosen action.
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)       
    q_stage2_mf = np.zeros((2, 2))  
    
    # Track previous choices (initially -1 to indicate no previous choice)
    prev_action_1 = -1
    prev_action_2 = np.full(2, -1) # Track prev choice for each state separately

    for trial in range(n_trials):
        # 1. Stage 1 Choice Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration bonus to logits before softmax
        logits_1 = beta * q_net
        if prev_action_1 != -1:
            logits_1[prev_action_1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # 2. Stage 2 Choice Policy
        logits_2 = beta * q_stage2_mf[state_idx]
        
        # Check if we have a previous action for this specific state (planet)
        if prev_action_2[state_idx] != -1:
            logits_2[prev_action_2[state_idx]] += perseveration

        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Update history
        prev_action_1 = action_1[trial]
        prev_action_2[state_idx] = action_2[trial]

        # 3. Learning
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace update
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Random Exploration (Epsilon-Greedy) + Softmax
Instead of relying solely on softmax temperature (`beta`) to handle exploration, this model includes an explicit "lapse rate" or epsilon parameter (`epsilon`). This accounts for random slips of attention or directed random exploration that occurs independently of value differences. It effectively puts a floor on the probability of choosing any action.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Epsilon-Greedy Mixture (Lapse Rate).
    
    Combines softmax decision making with a baseline level of random responding.
    With probability (1-epsilon), the agent chooses via Softmax. With probability 
    epsilon, the agent chooses uniformly at random. This captures "lapses" or 
    pure exploration.
    
    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    epsilon: [0, 1] Probability of choosing randomly (lapse rate).
    """
    learning_rate, beta, w, epsilon = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)       
    q_stage2_mf = np.zeros((2, 2))  

    for trial in range(n_trials):
        # 1. Stage 1 Choice Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Softmax probabilities
        exp_q1 = np.exp(beta * q_net)
        softmax_probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Mix with uniform random distribution (0.5 for binary choice)
        mixed_probs_1 = (1 - epsilon) * softmax_probs_1 + epsilon * 0.5
        p_choice_1[trial] = mixed_probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # 2. Stage 2 Choice Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        softmax_probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Mix with uniform random
        mixed_probs_2 = (1 - epsilon) * softmax_probs_2 + epsilon * 0.5
        p_choice_2[trial] = mixed_probs_2[action_2[trial]]

        # 3. Learning
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace update
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
