Here are three new cognitive models based on the two-step task structure.

### Model 1: Hybrid Model-Based / Model-Free Learner
This model hypothesizes that the participant uses a combination of two systems: a Model-Based system (planning using the transition matrix) and a Model-Free system (learning from direct experience). A mixing weight parameter `w` determines the balance between these two strategies.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner (MB + MF).
    Combines Model-Based (planning) and Model-Free (trial-and-error) values.
    
    Parameters:
    learning_rate: [0,1] Update rate for MF values (Stage 1 and 2).
    beta: [0,10] Inverse temperature for softmax choice.
    w: [0,1] Mixing weight. 1 = Pure MB, 0 = Pure MF.
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for spaceships
    q_stage2_mf = np.zeros((2, 2)) # Values for aliens (used by both systems)

    for trial in range(n_trials):

        # --- Policy for First Choice (Stage 1) ---
        # 1. Model-Based Value: Plan using transition matrix * max stage 2 value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value: Weighted sum of MB and MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Policy for Second Choice (Stage 2) ---
        # Standard softmax on Stage 2 values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Update Stage 1 MF value using Stage 2 value (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 value using Reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free Learner with Choice Perseveration
This model ignores the transition matrix entirely (Pure Model-Free). However, it adds a "stickiness" or perseveration parameter. This captures the tendency of participants to repeat their previous Stage 1 choice regardless of whether it was rewarded, a common heuristic in behavioral data.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Choice Perseveration (Stickiness).
    Ignores transition structure. Adds a bonus to the previously chosen action.
    
    Parameters:
    learning_rate: [0,1] Update rate for Q-values.
    beta: [0,10] Inverse temperature.
    stickiness: [0,5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, stickiness = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with impossible action

    for trial in range(n_trials):

        # --- Policy for First Choice (Stage 1) ---
        # Calculate effective values (Q-value + Stickiness)
        q_effective = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_effective[last_action_1] += stickiness

        exp_q1 = np.exp(beta * q_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store choice for next trial's stickiness
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Policy for Second Choice (Stage 2) ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Standard TD prediction errors
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Based Learner with Asymmetric Learning Rates
This model is a planner (Model-Based) like the template, but it updates its knowledge about the aliens (Stage 2) differently depending on the outcome. It uses separate learning rates for positive prediction errors (receiving a coin when unexpected) and negative prediction errors (not receiving a coin when expected). This captures optimism or pessimism bias.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Based Learner with Asymmetric Update Rates.
    Plans using the transition matrix, but updates Stage 2 values 
    differently for better-than-expected vs worse-than-expected outcomes.
    
    Parameters:
    alpha_pos: [0,1] Learning rate for positive prediction errors (gains).
    alpha_neg: [0,1] Learning rate for negative prediction errors (losses/omissions).
    beta: [0,10] Inverse temperature.
    """
    alpha_pos, alpha_neg, beta = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage2_mf = np.zeros((2, 2)) 

    for trial in range(n_trials):

        # --- Policy for First Choice (Stage 1) ---
        # Pure Model-Based Planning
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Policy for Second Choice (Stage 2) ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Only updating Stage 2 values (Stage 1 is calculated via planning)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Asymmetric update
        if delta_stage2 >= 0:
            q_stage2_mf[state_idx, action_2[trial]] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[state_idx, action_2[trial]] += alpha_neg * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
