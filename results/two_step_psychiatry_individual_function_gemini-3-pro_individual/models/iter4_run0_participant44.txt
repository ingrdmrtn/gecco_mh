Here are 3 new cognitive models based on the two-step decision task, exploring different parameter combinations and mechanisms as requested.

### Model 1: Hybrid Model-Based/Model-Free with Separate Learning Rates for Stages
This model hypothesizes that the participant might learn the value of the spaceships (Stage 1) at a different rate than they learn the value of the aliens (Stage 2). It combines Model-Based (planning) and Model-Free (trial-and-error) valuation using a mixing weight `w`.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF learner with separate learning rates for Stage 1 and Stage 2.
    
    Bounds:
    lr_stage1: [0,1] Learning rate for the first stage (spaceships).
    lr_stage2: [0,1] Learning rate for the second stage (aliens).
    beta: [0,10] Inverse temperature for softmax.
    w: [0,1] Mixing weight (0 = Pure MF, 1 = Pure MB).
    """
    lr_stage1, lr_stage2, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as per task description
    # A -> X (0.7), U -> Y (0.7) usually implies:
    # Row 0 (Space A): [0.7 to Planet X, 0.3 to Planet Y]
    # Row 1 (Space U): [0.3 to Planet X, 0.7 to Planet Y]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (State x Action)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        q_net_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_net_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Update Stage 1 MF value using Stage 1 specific learning rate
        # Using SARSA-style TD(0) for stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_stage1 * delta_stage1
        
        # Update Stage 2 MF value using Stage 2 specific learning rate
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_stage2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free Learner with Outcome-Specific Perseverance
This model introduces a nuanced form of "stickiness." Instead of just repeating the last action regardless of the outcome, this agent has a specific tendency to repeat actions that were *rewarded* (win-stay) versus a general tendency to repeat actions (perseveration).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free learner with separate perseveration parameters for 
    rewarded trials (win-stay) and general repetition (stickiness).
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    pers_gen: [0,5] General perseveration bonus (added to last choice).
    pers_rew: [0,5] Additional bonus added if the last choice was rewarded.
    """
    learning_rate, beta, pers_gen, pers_rew = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_a1 = -1
    last_a2 = -1 # We will track last action taken in specific states
    last_reward = 0

    # To track last action at stage 2 specifically per state is complex, 
    # usually simple stickiness is just "last button pressed". 
    # Here we track the last action taken in the *current* state.
    last_a2_per_state = np.full(2, -1) 
    last_rew_per_state = np.zeros(2)

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        q_eff_1 = q_stage1.copy()
        if last_a1 != -1:
            q_eff_1[last_a1] += pers_gen
            if last_reward > 0:
                q_eff_1[last_a1] += pers_rew

        exp_q1 = np.exp(beta * q_eff_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        q_eff_2 = q_stage2[state_idx].copy()
        prev_a2 = last_a2_per_state[state_idx]
        prev_r2 = last_rew_per_state[state_idx]
        
        if prev_a2 != -1:
            q_eff_2[prev_a2] += pers_gen
            if prev_r2 > 0:
                q_eff_2[prev_a2] += pers_rew
        
        exp_q2 = np.exp(beta * q_eff_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        q_stage1[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Store history
        last_a1 = action_1[trial]
        last_reward = reward[trial]
        
        last_a2_per_state[state_idx] = action_2[trial]
        last_rew_per_state[state_idx] = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Free Learner with Forgetting (Decay)
This model implements a standard Model-Free learner but adds a "forgetting" parameter. Unchosen actions slowly decay back to a neutral value (0 or 0.5), simulating memory loss or the assumption that the environment is volatile and old information becomes stale.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free learner where Q-values of unchosen actions decay toward 0.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    decay_rate: [0,1] Rate at which unchosen Q-values decay (1=no decay, 0=instant forget).
    """
    learning_rate, beta, decay_rate = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Standard TD update for chosen actions
        delta_stage1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        q_stage1[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # --- Decay for Unchosen Actions ---
        # Stage 1 unchosen
        unchosen_a1 = 1 - action_1[trial]
        q_stage1[unchosen_a1] *= decay_rate
        
        # Stage 2 unchosen (only in the visited state)
        unchosen_a2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_a2] *= decay_rate
        
        # Note: We do not decay the unvisited state in stage 2 in this specific implementation
        # to keep parameter count low and logic simple.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
