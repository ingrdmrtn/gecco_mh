Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Model-Based / Model-Free (Standard Daw et al. 2011 Model)
This is the canonical model for this task. It assumes the agent computes values in two ways:
1.  **Model-Based (MB):** Uses the transition matrix to plan forward from Stage 1.
2.  **Model-Free (MF):** Uses temporal difference learning (TD(1)) to update values based on experienced rewards.
The final choice at Stage 1 is a weighted combination of these two values, controlled by a mixing parameter `w`.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning.
    
    Combines a model-based controller (using the transition matrix) and a model-free
    controller (using direct reinforcement). The parameter 'w' controls the balance.
    
    Bounds:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weight for Model-Based control (1 = pure MB, 0 = pure MF).
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as per task description (A->X 0.7, U->Y 0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)       # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2))  # Values for Stage 2 (aliens)

    for trial in range(n_trials):

        # --- POLICY FOR FIRST CHOICE ---
        # 1. Calculate Model-Based values
        # Max value of the next stage states
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using the transition model
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- POLICY FOR SECOND CHOICE ---
        # Standard softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATING ---
        # 1. Second stage update (standard Q-learning)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # 2. First stage update (TD(1) / Direct Reinforcement)
        # We update stage 1 MF values based on the final reward
        delta_stage1 = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseveration Model (Habitual Repetition)
This model extends the basic learning framework by adding a **perseveration** parameter. It tests the hypothesis that the participant has a tendency to simply repeat their previous Stage 1 choice, regardless of the reward outcome (a "motor habit" or "stickiness"), in addition to learning from rewards.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Choice Perseveration.
    
    Includes a 'stickiness' parameter that biases the agent to repeat the 
    previous stage 1 choice, independent of reward history.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    perseveration: [0, 5] - Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, perseveration = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice (initialized to -1 or handled in loop)
    last_action_1 = -1

    for trial in range(n_trials):

        # --- POLICY FOR FIRST CHOICE ---
        # Calculate effective Q-values including perseveration bonus
        q_net = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        
        # Update tracker for next trial
        last_action_1 = action_1[trial]

        # --- POLICY FOR SECOND CHOICE ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATING ---
        # TD(1) Logic: Update both stages based on final reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        delta_stage1 = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Separate Learning Rates for Positive/Negative Prediction Errors
This model investigates **valence-dependent learning**. It hypothesizes that the participant might learn differently from positive surprises (getting a coin when not expecting one) versus negative surprises (getting nothing when expecting a coin). It uses two separate learning rates: `alpha_pos` and `alpha_neg`.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Learning Rates (Pos/Neg Prediction Errors).
    
    Differentiates between learning from positive prediction errors (better than expected)
    and negative prediction errors (worse than expected).
    
    Bounds:
    alpha_pos: [0, 1] - Learning rate for positive PE (delta > 0).
    alpha_neg: [0, 1] - Learning rate for negative PE (delta < 0).
    beta: [0, 10]
    """
    alpha_pos, alpha_neg, beta = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- POLICY FOR FIRST CHOICE ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- POLICY FOR SECOND CHOICE ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATING ---
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2
        
        # Stage 1 Update (TD(1))
        # Note: We use the reward directly for TD(1) here
        delta_stage1 = reward[trial] - q_stage1_mf[action_1[trial]]
        lr_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += lr_1 * delta_stage1
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
