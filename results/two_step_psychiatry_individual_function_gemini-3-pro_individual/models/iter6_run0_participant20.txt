Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Model with Separate Learning Rates for Transition Types (Common vs. Rare)
This model hypothesizes that participants might update their model-free values differently depending on whether the transition they experienced was expected (common) or unexpected (rare). This captures the idea that "surprise" might modulate learning.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Transition-Dependent Learning Rates.
    
    This model separates the stage 1 learning rate into two parameters based on 
    whether the transition was common or rare. This tests if "surprise" (rare transitions)
    drives learning differently than expected outcomes.

    Parameters:
    lr_common: [0, 1] Learning rate for Stage 1 updates after common transitions.
    lr_rare: [0, 1] Learning rate for Stage 1 updates after rare transitions.
    lr_stage2: [0, 1] Learning rate for Stage 2 updates.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Weight for Model-Based control (1 = Pure MB, 0 = Pure MF).
    """
    lr_common, lr_rare, lr_stage2, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix: Row 0 is Action 0 (A), Row 1 is Action 1 (U)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        
        # Determine if transition was common or rare
        chosen_action = action_1[trial]
        is_common = False
        if chosen_action == 0 and state_idx == 0: is_common = True
        elif chosen_action == 1 and state_idx == 1: is_common = True
        
        # Select learning rate based on transition type
        current_lr_s1 = lr_common if is_common else lr_rare

        # Stage 1 Update (TD(0) style for MF)
        # Note: In standard hybrid models, MF Q(s1) is often updated by Q(s2) (TD) or Reward (Monte Carlo)
        # Here we use the TD error driven by the value of the state arrived at.
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += current_lr_s1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_stage2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Based with Forgetting
This model assumes the participant relies entirely on a Model-Based strategy (calculating values based on the transition structure) but their memory of the second-stage reward values decays over time. This tests if sub-optimal performance is due to memory limitations rather than model-free habits.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based with Decay/Forgetting.
    
    This model assumes the agent is purely Model-Based (w=1 implicitly) but subject 
    to forgetting of the specific alien reward values (Stage 2 Q-values).
    Unchosen options decay toward a neutral value (0.5).

    Parameters:
    learning_rate: [0, 1] Update rate for Q-values based on reward.
    decay_rate: [0, 1] Rate at which unvisited state-action values decay to 0.5.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2.
    """
    learning_rate, decay_rate, beta_1, beta_2 = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize Q-values at 0.5 (neutral expectation for binary 0/1 rewards)
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):

        # --- Stage 1 Choice (Pure MB) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta_1 * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning & Decay ---
        
        # 1. Update chosen option
        chosen_a2 = action_2[trial]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # 2. Decay unchosen options
        # Iterate over all 4 possible stage 2 actions
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == chosen_a2):
                    q_stage2_mf[s, a] = (1 - decay_rate) * q_stage2_mf[s, a] + decay_rate * 0.5

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Reward-Dependent "Win-Stay, Lose-Shift" (WSLS) Heuristic
Instead of simple perseveration, this model adds a heuristic bias: if the previous trial was rewarded, stay with the same Stage 1 choice; if unrewarded, switch. This simple heuristic is often observed alongside reinforcement learning.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Reward-Dependent Repetition (WSLS).
    
    Adds a 'WSLS' parameter. If the previous trial was rewarded, the agent gets a 
    bonus to repeat the Stage 1 action. If unrewarded, they may get a penalty (or 
    reduced bonus) for repeating it. This captures primitive Win-Stay Lose-Shift 
    behavior distinct from value-based learning.

    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based control.
    wsls_strength: [0, 5] Magnitude of the bias to repeat a rewarded action (Win-Stay).
    """
    learning_rate, beta, w, wsls_strength = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits_1 = beta * q_net
        
        # Apply WSLS heuristic
        if last_action_1 != -1:
            if last_reward == 1:
                # Win-Stay: Boost the previously chosen action
                logits_1[last_action_1] += wsls_strength
            # Note: Lose-Shift is implicit here; by not adding the bonus (or adding 0), 
            # the relative probability of the other action is higher compared to the Win case.

        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
