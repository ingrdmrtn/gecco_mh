Here are 3 new cognitive models exploring different mechanisms for learning and decision making in the two-step task.

### Model 1: Hybrid Model with Separate Learning Rates for Stages
This model extends the classic hybrid approach by acknowledging that learning dynamics might differ between the abstract first-stage choice (choosing a spaceship) and the concrete second-stage choice (choosing an alien). It separates the learning rate into `lr_stage1` and `lr_stage2`, while maintaining the weighting parameter `w` to arbitrate between model-based and model-free control.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL with Stage-Specific Learning Rates.

    This model separates the learning process for the two stages. It hypothesizes that
    updating the value of a spaceship (Stage 1) might occur at a different rate than
    updating the value of a specific alien (Stage 2), perhaps due to the different
    nature of the prediction errors or the abstraction level of the choice.

    Parameters:
    lr_stage1: [0, 1] - Learning rate for Stage 1 (Model-Free Q-values).
    lr_stage2: [0, 1] - Learning rate for Stage 2 (Q-values).
    beta: [0, 10] - Inverse temperature for softmax choice.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    lr_stage1, lr_stage2, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MF and MB values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Calculate prediction errors
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Update Stage 1 MF values using specific learning rate
        q_stage1_mf[action_1[trial]] += lr_stage1 * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 2 values using specific learning rate
        q_stage2_mf[state_idx, action_2[trial]] += lr_stage2 * delta_stage2
        
        # Note: No eligibility trace here to isolate the effect of split learning rates.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Choice Perseveration
This model adds a "perseveration" parameter to the standard hybrid framework. Humans often have a tendency to repeat their previous choice regardless of reward (stickiness). This model captures that bias by adding a bonus to the Q-value of the previously chosen action in Stage 1.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL with Choice Perseveration (Stickiness).

    This model includes a perseveration parameter 'p' which accounts for the tendency
    to repeat the previous action at the first stage, independent of reward history.
    This is a common nuisance parameter in behavioral modeling that captures motor
    repetition or cognitive inertia.

    Parameters:
    learning_rate: [0, 1] - Rate at which Q-values are updated.
    beta: [0, 10] - Inverse temperature for softmax choice.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    perseveration: [0, 1] - Bonus added to the previously chosen action (0 = no bias).
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with an impossible action

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration bonus to the net Q-values before softmax
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update tracker for next trial
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Update Stage 1 MF
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 2
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Pure Model-Free with Asymmetric Learning Rates (Positive/Negative)
This model drops the model-based component entirely to test a different hypothesis: that the participant learns differently from positive prediction errors (better than expected) versus negative prediction errors (worse than expected). This asymmetry is often linked to dopaminergic signaling differences. It replaces `w` and `beta` complexity with `lr_pos` and `lr_neg`.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free RL with Asymmetric Learning Rates for Positive/Negative Errors.

    This model assumes no model-based planning (w=0). Instead, it focuses on how
    values are updated. It allows for different learning rates depending on whether
    the prediction error is positive (reward > expectation) or negative (reward < expectation).
    This can capture optimism or pessimism bias in learning.

    Parameters:
    lr_pos: [0, 1] - Learning rate for positive prediction errors (delta > 0).
    lr_neg: [0, 1] - Learning rate for negative prediction errors (delta < 0).
    beta: [0, 10] - Inverse temperature for softmax choice.
    lambda_eligibility: [0, 1] - Eligibility trace decay for Stage 1 updates.
    """
    lr_pos, lr_neg, beta, lambda_eligibility = model_parameters
    n_trials = len(action_1)
  
    # No transition matrix needed for pure MF
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice (Pure MF)
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        if delta_stage1 > 0:
            q_stage1_mf[action_1[trial]] += lr_pos * delta_stage1
        else:
            q_stage1_mf[action_1[trial]] += lr_neg * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        if delta_stage2 > 0:
            q_stage2_mf[state_idx, action_2[trial]] += lr_pos * delta_stage2
            # Eligibility trace update for Stage 1 (using pos rate for the propagated error)
            q_stage1_mf[action_1[trial]] += lr_pos * lambda_eligibility * delta_stage2
        else:
            q_stage2_mf[state_idx, action_2[trial]] += lr_neg * delta_stage2
            # Eligibility trace update for Stage 1 (using neg rate for the propagated error)
            q_stage1_mf[action_1[trial]] += lr_neg * lambda_eligibility * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
