Here are three new cognitive models based on the two-step task structure, exploring mechanisms of stage-specific exploration, outcome-dependent heuristics, and state confusion.

### Model 1: Model-Free with Stage-Specific Inverse Temperatures
This model assumes the participant is a pure Model-Free learner (no planning) but exhibits different levels of exploration/randomness in the two stages. For example, they might be very careful in choosing the spaceship (Stage 1) but more exploratory or noisy when choosing the alien (Stage 2).

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Stage-Specific Inverse Temperatures.
    
    Assumes the participant is a pure model-free learner but has different 
    exploration rates (beta) for the first stage (spaceship choice) and 
    the second stage (alien choice).
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta_stage1: [0, 10] Inverse temperature for Stage 1 (Spaceship) choices.
    - beta_stage2: [0, 10] Inverse temperature for Stage 2 (Alien) choices.
    """
    learning_rate, beta_stage1, beta_stage2 = model_parameters
    n_trials = len(action_1)

    # Transition matrix is not used in pure MF, but required by template structure
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)      # Values for Spaceship A vs U
    q_stage2_mf = np.zeros((2, 2)) # Values for Aliens (Planet X: W, S; Planet Y: P, H)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # policy for the first choice (Uses beta_stage1)
        # Note: In pure MF, we ignore MB values. We use q_stage1_mf directly.
        exp_q1 = np.exp(beta_stage1 * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice (Uses beta_stage2)
        exp_q2 = np.exp(beta_stage2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1 (TD-learning)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 (Reward learning)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free with Win-Stay Lose-Shift (WSLS) Bias
This model combines standard reinforcement learning with a heuristic "Win-Stay, Lose-Shift" strategy on the first stage. Unlike simple perseveration (which just repeats the last choice), this bias depends explicitly on whether the *previous* trial resulted in a reward.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Win-Stay Lose-Shift (WSLS) Bias.
    
    The participant biases their Stage 1 choice based on the outcome of the 
    previous trial. If they won, they get a 'stay' bonus for the same spaceship. 
    If they lost, they get a 'shift' incentive (negative bonus) for that spaceship.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax choices.
    - win_stay: [0, 5] Bonus added to previous choice Q-value if previous reward was 1.
    - lose_shift: [0, 5] Penalty subtracted from previous choice Q-value if previous reward was 0.
    """
    learning_rate, beta, win_stay, lose_shift = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_action_1 = -1
    prev_reward = -1

    for trial in range(n_trials):

        # policy for the first choice
        q_stage1_biased = q_stage1_mf.copy()
        
        # Apply WSLS bias
        if prev_action_1 != -1:
            if prev_reward == 1:
                q_stage1_biased[prev_action_1] += win_stay
            elif prev_reward == 0:
                q_stage1_biased[prev_action_1] -= lose_shift

        exp_q1 = np.exp(beta * q_stage1_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_action_1 = action_1[trial]
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Free with Spatial Generalization
This model hypothesizes that the participant "leaks" learning between the two planets. When they learn about an alien on Planet X (e.g., Alien 0), they also slightly update their estimate for the corresponding alien on Planet Y. This reflects a cognitive confusion or generalization between the two states.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Spatial Generalization.
    
    When the participant updates the value of an alien on the current planet,
    they also apply a fraction of that update to the same alien index on the 
    other planet. This represents state confusion or generalization.
    
    Parameters:
    - learning_rate: [0, 1] Primary update rate.
    - beta: [0, 10] Inverse temperature.
    - generalization_factor: [0, 1] Fraction of the update applied to the other planet.
                              0 = No generalization (Standard MF), 1 = Full confusion.
    """
    learning_rate, beta, generalization_factor = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # policy for the first choice
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        other_state_idx = 1 - state_idx

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 (With Generalization)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Primary update
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Generalization update (leakage to the other planet)
        q_stage2_mf[other_state_idx, action_2[trial]] += (learning_rate * generalization_factor) * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
