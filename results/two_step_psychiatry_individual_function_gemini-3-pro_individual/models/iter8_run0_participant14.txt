Here are three new cognitive models based on the two-step task structure. These models introduce mechanisms like choice perseveration (stickiness), eligibility traces (direct credit assignment), and stage-specific decision noise.

### Model 1: Hybrid Model with Choice Perseveration (Stickiness)
This model extends the standard Hybrid (Model-Based/Model-Free) approach by adding a "stickiness" parameter. This accounts for the tendency of participants to repeat their previous Stage 1 choice regardless of the reward or transition, a common behavioral artifact in this task.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model (MB/MF) with Choice Stickiness.
    
    Combines Model-Based (planning) and Model-Free (habit) values, plus a 
    perseveration bonus for repeating the last chosen spaceship.
    
    Parameters:
    learning_rate: [0, 1] Update rate for MF Q-values.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    stickiness: [0, 5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # MF Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1 # Initialize with impossible action

    for trial in range(n_trials):

        # --- Policy for Stage 1 ---
        # 1. Calculate Model-Based values: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Calculate Net Value: w * MB + (1-w) * MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # 3. Add Stickiness (Perseveration)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Policy for Stage 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Update Stage 1 MF value (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 MF value
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Store action for next trial's stickiness
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free with Eligibility Traces (TD-Lambda)
This model is a pure reinforcement learner (no model-based planning) but includes an eligibility trace parameter (`lambda`). This allows the outcome at the end of the trial (the reward) to "flow back" and directly update the value of the first-stage choice (spaceship), rather than waiting for the value to propagate back one step at a time over multiple trials.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free TD(Lambda) Learner.
    
    Uses eligibility traces to allow the Stage 2 reward to directly update 
    the Stage 1 choice value in a single trial.
    
    Parameters:
    learning_rate: [0, 1] Update rate.
    beta: [0, 10] Inverse temperature.
    lambda_param: [0, 1] Eligibility trace decay. 0 = TD(0) (standard), 1 = Monte Carlo.
    """
    learning_rate, beta, lambda_param = model_parameters
    n_trials = len(action_1)
  
    # No transition matrix usage (Pure MF)
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Policy for Stage 1 ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Policy for Stage 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Prediction Error 1: Value of chosen state 2 minus value of chosen spaceship
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Prediction Error 2: Reward minus value of chosen alien
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1: Includes a portion of the Stage 2 error (lambda * delta2)
        # This bridges the gap between choice 1 and reward.
        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + lambda_param * delta_stage2)
        
        # Update Stage 2: Standard TD update
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Separate Stage Temperatures
This model assumes the participant uses a hybrid strategy but exhibits different levels of randomness (exploration/exploitation) at the two stages. For example, they might be very precise when choosing spaceships (Stage 1) but more exploratory when choosing aliens (Stage 2).

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Dual Inverse Temperatures (Dual Beta).
    
    Allows for different levels of decision noise (exploration) at Stage 1 
    vs Stage 2.
    
    Parameters:
    learning_rate: [0, 1] Update rate.
    beta_1: [0, 10] Inverse temperature for Stage 1 (Spaceships).
    beta_2: [0, 10] Inverse temperature for Stage 2 (Aliens).
    w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta_1, beta_2, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Policy for Stage 1 ---
        # Model-Based Component
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Choice using beta_1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Policy for Stage 2 ---
        # Choice using beta_2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
