Here are three new cognitive models based on the provided data and constraints.

### Model 1: Hybrid Learner with Epsilon-Lapse (Noise)
This model assumes the participant uses a standard hybrid strategy (mixing Model-Based and Model-Free values) but is subject to "lapses" in attention. On a certain proportion of trials (`epsilon`), the participant ignores values entirely and chooses randomly. This helps account for the "0 coins" trials where the participant might have acted noisily despite learned values.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Epsilon-Lapse.
    
    Combines Model-Based (MB) and Model-Free (MF) learning, but includes a "lapse" 
    parameter (epsilon) representing the probability of making a random choice 
    regardless of value, capturing noise or inattention.

    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax (value sensitivity).
    - w: [0, 1] Weighting parameter. 1 = Pure MB, 0 = Pure MF.
    - epsilon: [0, 1] Probability of choosing randomly (uniform noise).
    """
    learning_rate, beta, w, epsilon = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix for MB calculation (A->X, U->Y mostly)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)      # MF values for Stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # MF values for Stage 2 (Aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # 1. Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Integrate MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Softmax probability
        exp_q1 = np.exp(beta * q_net)
        probs_1_softmax = exp_q1 / np.sum(exp_q1)
        
        # 4. Apply Epsilon-Greedy Mixture (Lapse)
        # P(a) = (1-eps) * Softmax(a) + eps * 0.5
        probs_1 = (1 - epsilon) * probs_1_softmax + epsilon * 0.5
        
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 MF Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free Learner with Rare-Transition Dampening
This model hypothesizes that the participant is primarily a Model-Free learner but uses a heuristic to handle the task structure. When a "Rare" transition occurs (e.g., Spaceship A goes to Planet Y), the participant suppresses learning for the first stage choice, reasoning that the outcome is not representative of the spaceship's true value.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Rare-Transition Dampening.
    
    A Model-Free learner that modulates its learning rate based on the transition type.
    If a transition is 'Rare' (unexpected), the learning rate for the Stage 1 update
    is multiplied by a dampening factor.
    
    Parameters:
    - learning_rate: [0, 1] Base update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax.
    - rare_dampening: [0, 1] Multiplier for learning rate on rare transitions. 
                             0 = no learning from rare events; 1 = standard Q-learning.
    """
    learning_rate, beta, rare_dampening = model_parameters
    n_trials = len(action_1)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy (Pure MF) ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # 0=Planet X, 1=Planet Y

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Check Transition Type ---
        # Common: A(0)->X(0) or U(1)->Y(1). Rare: A(0)->Y(1) or U(1)->X(0).
        is_rare = (action_1[trial] == 0 and state_idx == 1) or \
                  (action_1[trial] == 1 and state_idx == 0)
        
        # Adjust effective alpha
        current_alpha = learning_rate * rare_dampening if is_rare else learning_rate

        # --- Updates ---
        # Stage 1 Update (Dampened)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += current_alpha * delta_stage1

        # Stage 2 Update (Standard, driven by reward)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Q-Learning with Mean Reversion (Forgetting)
This model assumes that learned values are not stable indefinitely. Instead of just decaying, Q-values actively revert toward a neutral prior (0.5) over time. This captures a participant who "forgets" specific values if they haven't been reinforced recently, effectively pulling their strategy back towards exploration or indifference.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Q-Learning with Mean Reversion.
    
    A Model-Free learner where Q-values decay towards a neutral mean (0.5) 
    on every trial. This simulates forgetting or uncertainty growth, pulling 
    estimates back to the center.
    
    Parameters:
    - learning_rate: [0, 1] Update rate from prediction error.
    - beta: [0, 10] Inverse temperature for softmax.
    - reversion_rate: [0, 1] Rate at which Q-values revert to 0.5.
    """
    learning_rate, beta, reversion_rate = model_parameters
    n_trials = len(action_1)

    q_stage1_mf = np.zeros(2) + 0.5      # Initialize to neutral 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5 # Initialize to neutral 0.5

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Updates ---
        
        # 1. Standard TD Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # 2. Mean Reversion (Forgetting) applied to ALL values
        # Moves Q towards 0.5: Q = Q + lambda * (0.5 - Q)
        q_stage1_mf += reversion_rate * (0.5 - q_stage1_mf)
        q_stage2_mf += reversion_rate * (0.5 - q_stage2_mf)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
