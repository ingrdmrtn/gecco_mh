Here are three new cognitive models for the two-step task.

### Model 1: Dynamic Transition Learning (Adaptive MB)
This model relaxes the assumption that the participant has a fixed, perfect understanding of the transition probabilities (e.g., 70/30). Instead, the participant continuously updates their internal model of the spaceship-planet transitions based on experience. This captures potential uncertainty or the belief that the "common" transitions might shift over time.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Transition Learning Model:
    The agent learns the transition matrix (Model-Based component) online 
    using a Rescorla-Wagner update rule, rather than assuming fixed probabilities.
    
    Parameters:
    learning_rate: [0, 1] Learning rate for value updates (MF).
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Weight mixing MB (1) and MF (0).
    trans_lr: [0, 1] Learning rate for updating transition probabilities.
    pers: [0, 5] Perseverance bonus for repeating the last Stage 1 choice.
    """
    learning_rate, beta, w, trans_lr, pers = model_parameters
    n_trials = len(action_1)
  
    # Initialize subjective transition matrix (uniform prior = high uncertainty)
    # Row 0: Spaceship 0 -> [Prob Planet 0, Prob Planet 1]
    # Row 1: Spaceship 1 -> [Prob Planet 0, Prob Planet 1]
    subj_transition_matrix = np.array([[0.5, 0.5], [0.5, 0.5]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_choice = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Use current subjective transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = subj_transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Perseverance
        if last_choice != -1:
            q_net[last_choice] += pers

        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        current_a1 = action_1[trial]
        current_s = state[trial]
        last_choice = current_a1

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[current_s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        
        # 1. Update Subjective Transition Matrix
        # Update P(observed_state | action) towards 1
        curr_prob = subj_transition_matrix[current_a1, current_s]
        subj_transition_matrix[current_a1, current_s] += trans_lr * (1 - curr_prob)
        # Ensure probabilities sum to 1
        subj_transition_matrix[current_a1, 1 - current_s] = 1.0 - subj_transition_matrix[current_a1, current_s]

        # 2. Update Action Values (TD)
        delta_stage1 = q_stage2_mf[current_s, action_2[trial]] - q_stage1_mf[current_a1]
        q_stage1_mf[current_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[current_s, action_2[trial]]
        q_stage2_mf[current_s, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Outcome-Dependent Temperature (Confidence)
This model hypothesizes that the participant's decision noise (exploration vs. exploitation) varies based on the previous outcome. A win might increase confidence (higher beta, more deterministic), while a loss might induce exploration or confusion (lower beta, more random).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Outcome-Dependent Temperature Model:
    Uses distinct inverse temperature (beta) parameters depending on whether 
    the previous trial resulted in a reward or not.
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    w: [0, 1] Weight mixing MB (1) and MF (0).
    beta_win: [0, 10] Inverse temperature after a rewarded trial.
    beta_loss: [0, 10] Inverse temperature after an unrewarded trial.
    pers: [0, 5] Perseverance bonus.
    """
    learning_rate, w, beta_win, beta_loss, pers = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_choice = -1
    last_reward_val = 0 
    # Assume neutral/loss start or use mean beta. 
    # Here we default to beta_loss for the first trial as no win has occurred.

    for trial in range(n_trials):
        
        # Determine current beta based on previous outcome
        current_beta = beta_win if last_reward_val > 0 else beta_loss

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        if last_choice != -1:
            q_net[last_choice] += pers

        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_choice = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_reward_val = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Outcome-Dependent Model-Based Weight
This model proposes that the reliance on the Model-Based strategy ($w$) is dynamic. The participant may rely more on the cognitive map ($w_{win}$) after a success, but fall back to habitual/model-free responding ($w_{loss}$) after a failure (or vice-versa).

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Outcome-Dependent Model-Based Weight:
    The weighting parameter 'w' changes depending on the previous trial's outcome,
    reflecting a shift in strategy reliability between Model-Based and Model-Free.
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w_win: [0, 1] MB weight after a rewarded trial.
    w_loss: [0, 1] MB weight after an unrewarded trial.
    pers: [0, 5] Perseverance bonus.
    """
    learning_rate, beta, w_win, w_loss, pers = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_choice = -1
    last_reward_val = 0

    for trial in range(n_trials):
        
        # Determine current w based on previous outcome
        current_w = w_win if last_reward_val > 0 else w_loss

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = current_w * q_stage1_mb + (1 - current_w) * q_stage1_mf

        if last_choice != -1:
            q_net[last_choice] += pers

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_choice = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_reward_val = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
