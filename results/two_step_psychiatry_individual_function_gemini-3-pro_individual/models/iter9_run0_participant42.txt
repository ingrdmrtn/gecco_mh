Here are three new cognitive models for the two-step task.

### Model 1: Subjective Transition Hybrid Learner
This model hypothesizes that the participant does not use the objective transition probabilities (0.7/0.3) for their Model-Based calculation, but instead learns or assumes a subjective probability of the "common" transition. This captures potential misconceptions about the task structure or over/under-confidence in the spaceship reliability.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner where the participant's internal model of transition probabilities 
    is a free parameter, rather than fixed at 0.7.

    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] (Weight of Model-Based system)
    transition_prob: [0, 1] (Subjective probability of common transition)
    """
    learning_rate, beta, w, transition_prob = model_parameters
    n_trials = len(action_1)

    # Subjective transition matrix based on parameter
    # 0->0 and 1->1 are common (prob), 0->1 and 1->0 are rare (1-prob)
    transition_matrix = np.array([[transition_prob, 1 - transition_prob], 
                                  [1 - transition_prob, transition_prob]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        state_idx = state[trial]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updating
        # Stage 1 Update (MF)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update (MF)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Learner with Outcome-Dependent Perseveration
This model extends the concept of choice perseveration (stickiness) by allowing different stickiness values depending on whether the previous trial was a win or a loss. This explicitly models a "Win-Stay, Lose-Shift" heuristic bias layered on top of the reinforcement learning process.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner with outcome-dependent choice stickiness (Win-Stay, Lose-Shift bias).
    Stickiness is applied differently depending on whether the previous trial was rewarded.

    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    stick_win: [0, 5] (Bonus added to previous choice if rewarded)
    stick_loss: [-5, 5] (Bonus added to previous choice if unrewarded - negative implies switching)
    """
    learning_rate, beta, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply outcome-dependent stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                q_net[last_action_1] += stick_win
            else:
                q_net[last_action_1] += stick_loss

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        state_idx = state[trial]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Track history
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

        # Updating
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with Rare Transition Dampening
This model assumes that participants might discount or dampen the Model-Free learning signal when it comes from a "rare" transition (e.g., choosing Spaceship A but ending up at Planet Y). This acts as a filter to prevent noise from rare events from corrupting the cached Model-Free values.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner that modulates the Model-Free update rate based on transition type.
    Updates to Stage 1 MF values are dampened if the transition was rare, protecting
    cached values from misleading "rare" outcomes.

    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    rare_dampening: [0, 1] (Multiplier for learning rate on rare transitions)
    """
    learning_rate, beta, w, rare_dampening = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        state_idx = state[trial]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updating
        # Check if transition was common
        # 0->0 and 1->1 are common. 
        is_common = (action_1[trial] == state_idx)
        
        current_lr = learning_rate
        if not is_common:
            current_lr *= rare_dampening

        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += current_lr * delta_stage1
        
        # Stage 2 update is always standard
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
