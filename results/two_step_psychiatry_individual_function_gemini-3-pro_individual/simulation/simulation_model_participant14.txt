import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free TD(0) Learning 
    with Choice Perseveration model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q-values for the two states at stage 2, each having 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix for the 2-step task (fixed structure)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # Initialize with an impossible action for the first trial logic
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Determine Stage 1 Choice
        logits_1 = beta * q_stage1_mf.copy()
        
        # Apply perseveration bonus if a previous action exists
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        state_idx = s2 # 0 or 1 directly maps to the indices of q_stage2_mf
        
        # 3. Determine Stage 2 Choice
        # Standard softmax on stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Determine Reward
        # Get current probabilities for the specific state arrived at
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[state_idx][a2])
        
        # 5. Update Q-values (TD(0) Logic)
        
        # Stage 1 Update: Based on the value of the state arrived at (max Q of next state)
        # Note: The original fitting code used `v_stage2_state = np.max(q_stage2_mf[state_idx])`
        v_stage2_state = np.max(q_stage2_mf[state_idx])
        delta_stage1 = v_stage2_state - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update: Based on received reward
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update tracker for perseveration
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
