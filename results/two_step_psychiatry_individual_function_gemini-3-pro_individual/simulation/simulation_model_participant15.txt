import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid MF/MB Model with Choice Perseveration.
    
    This simulation matches the logic of 'cognitive_model1'. It blends Model-Based 
    (planning using the transition matrix) and Model-Free (TD learning) values, 
    adding a 'perseveration' bonus to the previously chosen first-stage action.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, perseveration = parameters
    
    # Fixed transition matrix (common in 2-step tasks)
    # Row 0: State 0 transition probs to State 1 (70%) and State 2 (30%)
    # Row 1: State 0 transition probs to State 1 (30%) and State 2 (70%)
    # Note: The fitting code uses states 0 and 1 for the second stage.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for stage 2 states and actions (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Net Q-values for Stage 1
        # Model-Based value calculation: Transition Matrix * Max Q(Stage 2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply perseveration bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # 2. Stage 1 Selection (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # transition_matrix[a1] gives [prob_state_0, prob_state_1]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Selection (Softmax)
        qs_current_state = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Outcome
        # Map drifts to the specific state/action pair
        # s2=0 corresponds to drift1/drift2, s2=1 corresponds to drift3/drift4
        if s2 == 0:
            current_drift = [drift1[t], drift2[t]]
        else:
            current_drift = [drift3[t], drift4[t]]
            
        r = int(rng.random() < current_drift[a2])
        
        # 6. Learning (TD Updates)
        # Note: The fitting code updates Stage 1 MF Q-value based on the value of the CHOSEN state/action in Stage 2
        # (This is a TD(1) or SARSA-like update logic implied by the fitting code: 
        # delta_stage1 = q_stage2_mf[state_idx, action_2] - q_stage1_mf[action_1])
        
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
