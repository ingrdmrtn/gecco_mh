import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Reinforcement Learning 
    model with Choice Perseveration (matching cognitive_model3).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, perseveration = parameters
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows), 2 actions (cols)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Transition matrix for the 2-step task
    # Row 0 (Action 0): 0.7 prob to State 0, 0.3 prob to State 1
    # Row 1 (Action 1): 0.3 prob to State 0, 0.7 prob to State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    last_action_1 = -1 
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        q_net = q_stage1_mf.copy()
        
        # Apply perseveration bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax Stage 2 (based on current state s2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Get current trial reward probabilities
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        current_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r_prob = current_probs[s2][a2]
        r = int(rng.random() < r_prob)

        # --- Learning Updates ---
        
        # Update Stage 1 Q-values
        # Using the max value of the next state (TD-0 style update)
        v_next_state = np.max(q_stage2_mf[s2])
        delta_stage1 = v_next_state - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 Q-values
        # Using the received reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update history
        last_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
