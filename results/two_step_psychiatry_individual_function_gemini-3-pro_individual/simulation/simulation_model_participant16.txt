import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Reinforcement Learning (TD(1)) model.
    
    This model assumes the agent ignores the transition structure (MB) and relies
    solely on direct reinforcement. The first-stage values are updated based on the 
    eventual reward received at the second stage.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta = parameters
    rng = np.random.default_rng()
    
    # Transition matrix (Common/Rare structure usually fixed in this task)
    # 70% chance to go to state matching action (0->0, 1->1), 30% otherwise
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)       # Values for Spaceships A and B
    q_stage2_mf = np.zeros((2, 2))  # Values for Aliens on Planet X and Y

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for trial in range(n_trials):
        # Current reward probabilities for the 4 terminal states
        reward_probs = [[drift1[trial], drift2[trial]], 
                        [drift3[trial], drift4[trial]]]

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition to Stage 2 ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        # Note: The fitting code uses TD(1)-like logic where stage 1 is updated 
        # directly by the final reward, ignoring the stage 2 Q-value prediction error.
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values directly from reward
        delta_stage1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[trial] = a1
        state2[trial] = s2
        stage2_choice[trial] = a2
        reward[trial] = r

    return stage1_choice, state2, stage2_choice, reward
