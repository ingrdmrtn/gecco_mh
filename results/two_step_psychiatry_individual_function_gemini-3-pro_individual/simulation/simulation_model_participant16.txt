import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Based Learner with Forgetting.
    
    This model assumes the participant builds a model of the task structure to calculate
    values (Model-Based) but forgets the value of options over time.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, forgetting_rate]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, forgetting_rate = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    # Transition matrix (fixed structure of the 2-step task)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values for the second stage (Model-Based values)
    # The fitting code initializes this to 0.5
    q_stage2_mb = np.ones((2, 2)) * 0.5 
    
    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # Current trial reward probabilities for the second stage states
        # drift1/2 correspond to state 0 (left/right), drift3/4 correspond to state 1 (left/right)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 Decision ---
        # Calculate max value available in stage 2 for each state
        max_val_stage2 = np.max(q_stage2_mb, axis=1)
        
        # Bellman equation: Q_MB(s1, a1) = sum(P(s2|s1,a1) * max(Q(s2, :)))
        q_stage1_mb = transition_matrix @ max_val_stage2
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 based on current state (s2)
        exp_q2 = np.exp(beta * q_stage2_mb[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning and Forgetting ---
        # 1. Apply forgetting to ALL stage 2 Q-values before updating the chosen one
        # This matches the logic: q_stage2_mb = (1 - forgetting_rate) * q_stage2_mb
        q_stage2_mb = (1 - forgetting_rate) * q_stage2_mb
        
        # 2. Update the specific state-action pair experienced
        # Note: The fitting code calculates delta using the *decayed* value from step 1 above.
        delta = r - q_stage2_mb[s2, a2]
        q_stage2_mb[s2, a2] += learning_rate * delta
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
