import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model (MB/MF) with Choice Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Model Variables
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # -----------------------
        # Stage 1 Decision
        # -----------------------
        
        # 1. Model-Based Value Calculation
        # V(s') = max(Q(s', a'))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Q_MB = T * V(s')
        q_stage1_mb = transition_matrix @ max_q_stage2

        # 2. Net Value Calculation (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # 3. Apply Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        # 4. Softmax Choice Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # -----------------------
        # Transition
        # -----------------------
        
        # Determine state 2 based on transition matrix probabilities for chosen a1
        # transition_matrix[0] = [0.7, 0.3], transition_matrix[1] = [0.3, 0.7]
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # -----------------------
        # Stage 2 Decision
        # -----------------------
        
        # Softmax Choice Stage 2 (using same beta)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # -----------------------
        # Reward Outcome
        # -----------------------
        
        # Map drifts to current state/action
        # State 0: Actions 0 (drift1), 1 (drift2)
        # State 1: Actions 0 (drift3), 1 (drift4)
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        prob_reward = current_reward_probs[state_idx][a2]
        r = int(rng.random() < prob_reward)
        
        # -----------------------
        # Learning Updates
        # -----------------------
        
        # Update Stage 1 MF Q-value (TD(0) / SARSA-like update based on Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value based on reward
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # -----------------------
        # Storage and cleanup
        # -----------------------
        
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
