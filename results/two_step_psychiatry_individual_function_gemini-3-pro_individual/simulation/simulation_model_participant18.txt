import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Hybrid Model-Based / Model-Free RL Model
    with Choice Perseverance.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseverance]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, perseverance = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf represents Q(s1, a)
    q_stage1_mf = np.zeros(2) + 0.5
    # q_stage2_mf represents Q(s2, a), shape (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    # Transition matrix (fixed)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Track the last action for perseverance bonus
    last_action_1 = -1 
    
    # Initialize random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Stage 1 Q-values
        
        # Model-Based Value Calculation: Q_MB = Transition * max(Q_Stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Perseverance Bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseverance
            
        # 2. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # Determine state based on transition probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice (Softmax)
        # Uses only Model-Free Q-values for the specific state arrived at
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Generation
        # Map drift inputs to the specific state/action pair
        # State 0 corresponds to drift1 (action 0) and drift2 (action 1)
        # State 1 corresponds to drift3 (action 0) and drift4 (action 1)
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 6. Learning / Updates
        
        # TD Update for Stage 1 (SARSA-like update using Q_Stage2)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Update for Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
