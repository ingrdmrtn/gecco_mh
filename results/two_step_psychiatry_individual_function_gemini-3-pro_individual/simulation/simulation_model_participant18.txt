import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Q-learning model with Choice Perseveration.
    
    This model assumes the agent ignores the transition probabilities (w=0).
    It adds a 'perseveration' bonus to the previously chosen action at Stage 1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, perseveration = parameters
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the two states at stage 2, each having 2 actions
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed for the task structure usually)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # Indicator for previous action
    rng = np.random.default_rng() # Initialize random number generator

    for t in range(n_trials):
        # 1. Setup Reward Probabilities for the current trial
        # State 0 has options associated with drift1 and drift2
        # State 1 has options associated with drift3 and drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # 2. Stage 1 Decision
        logits_stage1 = q_stage1_mf.copy()
        
        # Apply perseveration bonus if it's not the first trial
        if last_action_1 != -1:
            logits_stage1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * logits_stage1)
        prob_a1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=prob_a1)
        
        # 3. Transition to Stage 2
        # Determine next state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Decision
        # In the fitting code: state_idx = state[trial], exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        # Note: The fitting code uses the same beta for both stages.
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        prob_a2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=prob_a2)
        
        # 5. Outcome
        r = int(rng.random() < reward_probs[s2][a2])
        
        # 6. Learning / Updates
        # The fitting code uses "SARSA-like" TD(0) update for Stage 1 using max_q_next
        # delta_stage1 = max_q_next - q_stage1_mf[a1]
        max_q_next = np.max(q_stage2_mf[s2])
        delta_stage1 = max_q_next - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Standard Q-learning update for Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store history
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update last action for perseveration on next trial
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
