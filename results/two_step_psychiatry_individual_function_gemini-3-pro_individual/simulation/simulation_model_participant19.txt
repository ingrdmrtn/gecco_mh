import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model-Based / Model-Free RL 
    with Choice Perseveration model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, p_pers]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, p_pers = parameters

    # Fixed transition matrix (Common in 2-step tasks: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 Q-values (2 states x 2 actions)
    q_stage2 = np.zeros((2, 2))

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_choice = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        # state 0 corresponds to drifts 1 and 2
        # state 1 corresponds to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based value
        # Bellman equation: Transition Matrix * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-value (Weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Perseveration Bonus
        q_net_pers = q_net.copy()
        if prev_choice != -1:
            q_net_pers[prev_choice] += p_pers

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net_pers)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard Q-learning choice at second stage
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Stochastic reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])

        # --- Learning ---
        
        # Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 MF Q-values
        # Note: The fitting code uses q_stage2[s2, a2] as the target for stage 1 update
        # (SARSA-like update logic implied by the fitting code line:
        # delta_stage1 = q_stage2[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]])
        # Crucially, it uses the *updated* q_stage2 value if following the fitting code flow strictly.
        delta_stage1 = q_stage2[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous choice for next trial's perseveration
        prev_choice = a1

    return stage1_choice, state2, stage2_choice, reward
