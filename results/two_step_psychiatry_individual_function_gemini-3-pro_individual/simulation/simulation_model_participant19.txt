import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using Q-Learning with Passive Decay.
    
    Standard Q-learning where unchosen actions decay toward zero.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, decay_rate]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, decay_rate = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two options in the first stage
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the two options in each of the two second-stage states
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix for the task structure
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Create a random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. First Stage Choice
        # Calculate probabilities using softmax on Stage 1 Q-values
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Determine transition to second stage state (0 or 1)
        # If a1=0, probabilities are [0.7, 0.3]. If a1=1, probabilities are [0.3, 0.7].
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # 2. Second Stage Choice
        # Calculate probabilities using softmax on Stage 2 Q-values for the specific state s2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 3. Outcome / Reward
        # Get the reward probability for the current state (s2) and action (a2)
        # Mapping drifts to states/actions:
        # State 0 (Blue): Action 0 -> drift1, Action 1 -> drift2
        # State 1 (Pink): Action 0 -> drift3, Action 1 -> drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        p_reward = current_reward_probs[s2][a2]
        
        r = int(rng.random() < p_reward)
        
        # 4. Learning Updates
        
        # --- Stage 1 Update ---
        chosen_s1 = a1
        unchosen_s1 = 1 - chosen_s1
        
        # SARSA-style update for Stage 1: Q(s1, a1) <- Q(s1, a1) + lr * (Q(s2, a2) - Q(s1, a1))
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += learning_rate * delta_stage1
        
        # Passive Decay for unchosen Stage 1 action
        q_stage1_mf[unchosen_s1] *= (1 - decay_rate)
        
        # --- Stage 2 Update ---
        chosen_s2 = a2
        unchosen_s2 = 1 - chosen_s2
        
        # Standard Q-learning update for Stage 2: Q(s2, a2) <- Q(s2, a2) + lr * (reward - Q(s2, a2))
        delta_stage2 = r - q_stage2_mf[s2, chosen_s2]
        q_stage2_mf[s2, chosen_s2] += learning_rate * delta_stage2
        
        # Passive Decay for unchosen Stage 2 action (in the current state)
        q_stage2_mf[s2, unchosen_s2] *= (1 - decay_rate)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
