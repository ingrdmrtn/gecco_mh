import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model-Based / Model-Free RL 
    with Choice Perseveration, matching the logic of cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, perseveration = parameters

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Model Variables
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)       
    q_stage2_mf = np.zeros((2, 2))  
    
    last_action_1 = -1 # Initialize with an impossible action

    for t in range(n_trials):
        
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based component
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Calculate Net Q-values (Weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Calculate Logits
        logits_1 = beta * q_net
        
        # Apply Perseveration Bonus
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration

        # Softmax for Stage 1
        # Subtract max for numerical stability
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine State 2 based on transition matrix
        state_probs = transition_matrix[a1]
        s2 = np.random.choice([0, 1], p=state_probs)
        
        # --- Stage 2 Decision ---
        
        # Calculate Stage 2 Logits
        logits_2 = beta * q_stage2_mf[s2]
        
        # Softmax for Stage 2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        
        # Map state/action to specific drift parameter
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s2 == 1
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < p_reward else 0

        # --- Learning Updates ---
        
        # Stage 1 Update (TD)
        # Note: Uses q_stage2_mf value of chosen action, not max, per fitting code logic
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 Update (TD)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
