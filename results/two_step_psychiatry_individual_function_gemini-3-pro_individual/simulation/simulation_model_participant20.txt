import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free TD(lambda) Learner with Perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lam, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, lam, perseveration = parameters
    
    # Setup storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize values
    # q_stage1_mf corresponds to Q-values for the first stage actions (size 2)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q-values for the second stage states and actions (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed as per standard 2-step task logic)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = perseveration
            
        # Calculate Stage 1 probabilities (Softmax)
        exp_q1 = np.exp(beta * (q_stage1_mf + pers_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Calculate Stage 2 probabilities (Softmax) based on current state s2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get reward probability for the specific state and action from drifts
        # Map drifts: state 0 -> (drift1, drift2), state 1 -> (drift3, drift4)
        if s2 == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)
        
        # 5. Learning / Value Updates
        
        # Stage 1 TD error (prediction error based on Stage 2 Q-value)
        # Note: In the source code, this uses the Q-value of the *chosen* stage 2 action
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Stage 2 TD error (prediction error based on Reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # Update Stage 1 Q-values
        # Combines standard TD(0) update (learning_rate * delta_stage1)
        # with eligibility trace update from reward (learning_rate * lam * delta_stage2)
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lam * delta_stage2
        
        # Update Stage 2 Q-values
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
