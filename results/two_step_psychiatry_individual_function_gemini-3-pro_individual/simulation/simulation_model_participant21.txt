```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Q-Learning with Choice Stickiness model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    rng = np.random.default_rng() # Initialize random number generator

    learning_rate, beta, stickiness = parameters
    
    # Transition matrix for the two-step task (fixed structure)
    # Action 0 -> 70% chance State 0, 30% chance State 1
    # Action 1 -> 30% chance State 0, 70% chance State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions at stage 1
    # q_stage2_mf: Q-values for the two actions at stage 2 (for 2 states)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1 # No previous action for the first trial

    for t in range(n_trials):
        # Current trial reward probabilities for [State 0][Act 0, Act 1] and [State 1][Act 0, Act 1]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_net_stage1 = q_stage1_mf.copy()
        
        # Apply stickiness bonus if there was a previous action
        if prev_action_1 != -1:
            q_net_stage1[prev_action_1] += stickiness
            
        # Softmax choice rule for Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # Determine transition to Stage 2 state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Note: The fitting code uses one beta for both stages, so we use beta here too.
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates (Model-Free TD) ---
        
        # Stage 1 update: Driven by the Q-value of the state/action chosen in Stage 2
        # Note: Fitting code uses pure TD(0), not SARSA(lambda) or eligibility traces.
        # It updates stage 1 based on the value of the CHOSEN action in stage 2.
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 update: Driven by the actual reward received
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update history for stickiness next trial
        prev_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
