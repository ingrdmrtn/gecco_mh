import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Perseverance + Model-Free Learner (TD(1)-style logic).
    
    This model includes:
    1. A single learning rate for both stages.
    2. A single beta (inverse temperature) for both stages.
    3. A perseverance bonus applied to the previously chosen Stage 1 action.
    4. Simple TD learning where Stage 1 Q-values are updated directly by the final reward (ignoring the transition structure).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseverance]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, perseverance = parameters
    
    # Initialize Q-values
    # q_stage1 corresponds to the two choices in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two choices in each of the two second-stage states
    # Shape: (2 states, 2 actions)
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (standard 70/30 structure for the 2-step task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    
    # Initialize random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the 4 terminal states
        # State 0 (index 0) leads to drift1/drift2
        # State 1 (index 1) leads to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_net_stage1 = q_stage1.copy()
        
        # Apply perseverance bonus if it's not the first trial
        if last_action_1 != -1:
            q_net_stage1[last_action_1] += perseverance
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine the second state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (based on the specific state s2 arrived at)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward outcome based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning Updates ---
        # As per cognitive_model1, Stage 1 Q-values are updated directly by the final reward
        # (This implies a TD(1) or direct reinforcement approach rather than TD(0))
        delta_stage1 = r - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

        # Stage 2 Q-values updated by final reward
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update tracking variable
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
