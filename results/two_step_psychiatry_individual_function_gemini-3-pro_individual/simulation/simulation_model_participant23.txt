import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Learner with Outcome-Specific Learning Rates.
    
    This model differentiates learning from positive prediction errors (better than expected)
    vs negative prediction errors (worse than expected).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    alpha_pos, alpha_neg, beta = parameters
    
    # Initialize Q-values
    # q_stage1: values for the two actions in the first stage (state 0)
    q_stage1 = np.zeros(2)
    # q_stage2: values for the two actions in the two second-stage states (states 1 and 2)
    # Shape (2, 2) corresponds to [state_index][action_index]
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (standard for 2-step task)
    # Action 0 -> 70% State 1 (idx 0), 30% State 2 (idx 1)
    # Action 1 -> 30% State 1 (idx 0), 70% State 2 (idx 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Pre-allocate arrays for results
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # state2=0 (drift1, drift2), state2=1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # s2 is the index for q_stage2 (0 or 1)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning (Update Q-values) ---
        
        # 1. Update Stage 2 Q-values
        # Calculate Prediction Error 2
        pe2 = r - q_stage2[s2, a2]
        # Determine learning rate based on sign of PE
        lr2 = alpha_pos if pe2 > 0 else alpha_neg
        # Update
        q_stage2[s2, a2] += lr2 * pe2

        # 2. Update Stage 1 Q-values
        # Note: In the fitting code, the Stage 1 update uses q_stage2[s2, a2] 
        # as the target (TD-0 / SARSA logic typically found in these tasks).
        # We use the updated q_stage2 value if following the exact logic of the provided fitting code
        # where the updates happen sequentially.
        pe1 = q_stage2[s2, a2] - q_stage1[a1]
        
        # Determine learning rate based on sign of PE
        lr1 = alpha_pos if pe1 > 0 else alpha_neg
        # Update
        q_stage1[a1] += lr1 * pe1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
