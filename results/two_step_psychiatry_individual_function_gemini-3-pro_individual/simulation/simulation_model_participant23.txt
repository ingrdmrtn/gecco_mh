import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates the Model-Free Learner with Choice Perseveration (Stickiness).
    
    This simulation mirrors the logic of cognitive_model1, which includes:
    - Pure model-free TD learning (SARSA-like updates).
    - A 'stickiness' parameter (phi) that biases the stage 1 choice towards the previous action.
    - A single learning rate and beta used for both stages.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, phi]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.
                                                     drift1/2 correspond to state 0 (actions 0/1),
                                                     drift3/4 correspond to state 1 (actions 0/1).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, phi = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2) 
    # Stage 2: 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix for the task (standard 2-step task structure)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        q_stage1_modified = q_stage1_mf.copy()
        
        # Apply stickiness bonus if there was a previous action
        if last_action_1 != -1:
            q_stage1_modified[last_action_1] += phi

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modified)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state2 based on transition matrix probabilities for chosen a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Softmax selection for Stage 2 (using same beta as stage 1)
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get reward probability for the specific terminal state (s2, a2)
        # s2=0 uses drift1/drift2, s2=1 uses drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r_prob = current_reward_probs[s2][a2]
        r = int(rng.random() < r_prob)
        
        # 5. Learning Updates
        # Note: The fitting code updates q_stage1 based on the q_value of the *chosen* stage 2 action (SARSA style),
        # not the max (Q-learning style), and it propagates the stage 2 RPE back to stage 1.
        
        # Stage 1 TD error: (Q_stage2[s2, a2] - Q_stage1[a1])
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 TD error: (Reward - Q_stage2[s2, a2])
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Direct update to stage 1 Q-value based on stage 2 RPE (eligibility trace logic with lambda=1 implicitly)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
