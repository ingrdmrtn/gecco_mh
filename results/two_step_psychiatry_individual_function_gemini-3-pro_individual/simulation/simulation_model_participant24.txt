Here is the simulation code for the Asymmetric Learning Rate Model (Model-Free).

```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Asymmetric Learning Rate Model (Model-Free).
    
    This model assumes the agent updates values differently depending on whether 
    the prediction error is positive (better than expected) or negative (worse than expected).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    alpha_pos, alpha_neg, beta = parameters
    
    # Initialize arrays for storing data
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions in stage 1
    # q_stage2_mf: Q-values for the two actions in each of the two states in stage 2
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed for the 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning Updates ---
        
        # 1. Stage 1 TD Error (SARSA-style update using Q-value of chosen stage 2 action)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Asymmetric update for Stage 1
        if delta_stage1 > 0:
            q_stage1_mf[a1] += alpha_pos * delta_stage1
        else:
            q_stage1_mf[a1] += alpha_neg * delta_stage1
            
        # 2. Stage 2 TD Error
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # Asymmetric update for Stage 2
        if delta_stage2 > 0:
            q_stage2_mf[s2, a2] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[s2, a2] += alpha_neg * delta_stage2
            
        # 3. Direct update to Stage 1 based on Stage 2 RPE (Eligibility trace / TD(1))
        # Note: In the fitting code provided, this update happens directly to q_stage1_mf[a1]
        if delta_stage2 > 0:
            q_stage1_mf[a1] += alpha_pos * delta_stage2
        else:
            q_stage1_mf[a1] += alpha_neg * delta_stage2
            
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
