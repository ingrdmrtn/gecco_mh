import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using Reinforcement Learning with Choice Kernel (Perseveration).
    Tracks value (RL) and a choice trace (CK) to bias repetition.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, ck_learning_rate, beta_k]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, ck_learning_rate, beta_k = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2)) # 2 states (indexed 0, 1), 2 actions
    
    # Initialize Choice Kernels
    ck_stage1 = np.zeros(2)
    ck_stage2 = np.zeros((2, 2))
    
    # Transition matrix (standard 2-step task structure)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        logits_1 = (beta * q_stage1) + (beta_k * ck_stage1)
        exp_1 = np.exp(logits_1 - np.max(logits_1)) # Softmax stability
        probs_1 = exp_1 / np.sum(exp_1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on action and transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        logits_2 = (beta * q_stage2[s2]) + (beta_k * ck_stage2[s2])
        exp_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_2 / np.sum(exp_2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get current reward probabilities for the specific state
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_drifts[s2][a2])
        
        # 5. Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # 6. Update Choice Kernels (Decay then Increment)
        ck_stage1 *= (1 - ck_learning_rate)
        ck_stage2[s2] *= (1 - ck_learning_rate)
        
        ck_stage1[a1] += 1 * ck_learning_rate
        ck_stage2[s2, a2] += 1 * ck_learning_rate
        
        # 7. Update Q-values (SARSA / TD)
        # TD error for stage 1 (using Q-value of chosen stage 2 action as estimate)
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # TD error for stage 2
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # Direct update to stage 1 based on stage 2 RPE (TD(1) logic effectively)
        q_stage1[a1] += learning_rate * delta_stage2

    return stage1_choice, state2, stage2_choice, reward
