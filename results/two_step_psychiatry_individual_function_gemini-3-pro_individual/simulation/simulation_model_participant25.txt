import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Based Reinforcement Learning 
    with Perseveration (Stickiness) model.

    This model assumes the agent is purely model-based (planning using the transition matrix)
    but has a bias to repeat the last chosen action at stage 1 (perseveration).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, perseveration = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Fixed transition matrix as defined in the fitting code
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Storage for simulation results
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values for stage 2 (model-free)
    # The fitting code uses q_stage2_mf of shape (2, 2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for t in range(n_trials):
        # Current reward probabilities for the two states (0 and 1) and two actions
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        # Note: The fitting code uses `state` indices. Usually in these tasks:
        # state 0 -> drifts 1 & 2
        # state 1 -> drifts 3 & 4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate max Q-value for the second stage to plan backwards
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Model-based calculation: Expected value of stage 2 based on transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Apply perseveration bonus
        q_net = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration

        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix and chosen action a1
        # State 0 or 1 (mapped to indices 0 or 1 for q_stage2_mf access)
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 based on model-free Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning ---
        
        # Update Stage 2 Q-values (Model-Free update)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx 
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
