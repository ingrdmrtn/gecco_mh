import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Asymmetric Learning Rates model.
    This model differentiates between learning from positive prediction errors 
    (better than expected) and negative prediction errors (worse than expected).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, beta, perseverance]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    alpha_pos, alpha_neg, beta, perseverance = parameters

    # Transition matrix (fixed 70/30 structure standard for this task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two states (rows) and two actions (cols) in stage 2
    q_stage2_mf = np.zeros((2, 2))

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1

    for t in range(n_trials):
        # Current reward probabilities for the 4 terminal states
        # State 0 (drift1, drift2), State 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_effective = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_effective[last_action_1] += perseverance
            
        exp_q1 = np.exp(beta * q_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Note: In the fitting code, state was indexed directly. Here s2 is 0 or 1.
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning ---
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[s2, a2] += lr_2 * delta_stage2

        # Stage 1 Update (TD(0) logic from fitting code)
        # The fitting code updates Stage 1 Q-value based on the *updated* Stage 2 Q-value
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        lr_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[a1] += lr_1 * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
