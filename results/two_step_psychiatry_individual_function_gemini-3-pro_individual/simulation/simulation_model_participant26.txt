import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Model-Free TD(Lambda) with Perseveration model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lambda_eligibility, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, lambda_eligibility, perseveration = parameters
    
    # Initialize arrays for storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Values for the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Values for the two states (rows) and two actions (cols) at stage 2
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix: [Action 0 probs, Action 1 probs]
    # Standard 2-step task: Action 0 -> 70% State 0, Action 1 -> 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        q_net = q_stage1_mf.copy()

        # Apply perseveration bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on action 1 and transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using the specific state Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward probability based on state and action
        # State 0 uses drift1 (action 0) and drift2 (action 1)
        # State 1 uses drift3 (action 0) and drift4 (action 1)
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # --- Learning ---
        
        # 1. Update Stage 1 Q-value based on Stage 2 Q-value (TD(0) part)
        delta_1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1

        # 2. Update Stage 2 Q-value based on received reward
        delta_2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_2

        # 3. Eligibility Trace: Update Stage 1 Q-value based on Stage 2 prediction error
        q_stage1_mf[a1] += learning_rate * lambda_eligibility * delta_2
        
        # --- Housekeeping ---
        last_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
