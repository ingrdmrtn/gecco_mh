import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using a Model-Free learner with Choice Perseveration (Stickiness).
    
    This model updates Q-values based on reward prediction errors and adds a "stickiness"
    bonus to the previously chosen first-stage action.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1 corresponds to the two options in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two options in each of the two second-stage states
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed for the standard 2-step task)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    last_action_1 = -1  # Initialize with an impossible action
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for easier access: reward_probs[state][action]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 Decision ---
        q_stage1_choice = q_stage1.copy()
        
        # Apply perseveration bonus if a previous action exists
        if last_action_1 != -1:
            q_stage1_choice[last_action_1] += perseveration
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_choice)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 based on the current state (s2)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning Updates ---
        # Update Stage 2 Q-value (TD error based on reward)
        # Note: In the fitting code, `state_idx` was passed in `state`. Here `s2` is the index 0 or 1.
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-value (TD error based on Stage 2 Q-value)
        # This is a standard SARSA-like / TD(0) update propagating value back
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
