import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid MB/MF with Distinct Stage-1 MF Learning Rate model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_mf1, lr_2, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr_mf1, lr_2, beta = parameters
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # MF value of spaceships (Stage 1)
    q_stage2_mf = np.zeros((2, 2)) # Value of aliens (Stage 2)

    # Transition matrix
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values (Simple sum as per fitting code)
        q_net = q_stage1_mf + q_stage1_mb
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = rng.choice([0, 1], p=p_stage1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using just the Q-values for the current state)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        p_stage2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = rng.choice([0, 1], p=p_stage2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning Updates ---
        
        # Update Stage 1 MF values
        # Note: In this specific model, Stage 1 MF updates directly from the reward, 
        # not via a TD-error from Stage 2 values (unlike standard TD models).
        delta_stage1_mf = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_mf1 * delta_stage1_mf

        # Update Stage 2 values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr_2 * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
