```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Learner with Eligibility Traces (TD-Lambda).
    
    This model allows the reward at the second stage to directly influence 
    the value of the first stage choice, mediated by a decay parameter lambda.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lam]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    # Note: Unlike the example, the provided fitting code only has 3 parameters
    learning_rate, beta, lam = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the values of the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to values of actions at stage 2 (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Define transition matrix (standard for 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine the second stage state based on transition probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome/Reward
        # Get current reward probabilities for the specific state and action
        # State 0 corresponds to drifts 1 and 2, State 1 corresponds to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Learning / Updates
        
        # TD Error at Stage 2 (Reward Prediction Error)
        # delta_stage2 = reward - Q_stage2(s2, a2)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # Update Stage 2 Q-value
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # TD Error at Stage 1
        # In this specific model implementation: 
        # delta_stage1 = Q_stage2(s2, a2) - Q_stage1(a1)
        # Note: It uses the Q-value of the *chosen* action in stage 2, not max Q
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 Q-value
        # Incorporates both the immediate transition error (delta_stage1)
        # and the eligibility trace from the second stage reward (lam * delta_stage2)
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lam * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
