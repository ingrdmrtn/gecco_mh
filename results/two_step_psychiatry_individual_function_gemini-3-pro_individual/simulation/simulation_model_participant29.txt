import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with a perseveration 
    (stickiness) bonus for the first-stage choice.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters based on the fitting function's signature
    learning_rate, beta, w, perseveration = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Fixed transition matrix as defined in the model
    # Action 0 -> 70% State 0 (index 0), 30% State 1 (index 1)
    # Action 1 -> 30% State 0 (index 0), 70% State 1 (index 1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)       
    # Stage 2 MF Q-values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))  

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 corresponds to drifts 1 and 2
        # State 1 corresponds to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based value
        # MB value is transition matrix * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply perseveration bonus if a previous action exists
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        # transition_matrix[a1] gives [prob_state_0, prob_state_1]
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 using only MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        
        # Determine reward outcome based on drifts
        r = int(rng.random() < current_reward_probs[s2_idx][a2])

        # --- Learning / Updates ---

        # Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1 MF Q-values (TD Error using updated Stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state_idx, action_2] as the target
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1 

        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
