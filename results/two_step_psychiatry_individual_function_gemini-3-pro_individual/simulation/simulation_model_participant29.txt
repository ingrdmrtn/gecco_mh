import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Based Learner with Choice Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, perseveration = parameters

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values for Stage 2 (Model-Free)
    # Shape: (2, 2) corresponding to [state_idx, action]
    # Note: Fitting code uses state indices 0 and 1 for the second stage states.
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        # Mapping: state 0 -> [drift1, drift2], state 1 -> [drift3, drift4]
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values for Stage 1
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation: Transition * max_future_value
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Apply perseveration bonus
        q_net_stage1 = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_net_stage1[last_action_1] += perseveration
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix probabilities
        # transition_matrix[a1] gives [prob_state_0, prob_state_1] given action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 based on MF Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Determine reward outcome based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning Update ---
        
        # Update Stage 2 Q-values (Model-Free update)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for perseveration
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
