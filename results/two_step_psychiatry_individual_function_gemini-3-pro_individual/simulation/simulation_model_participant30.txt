import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid Model-Based / Model-Free RL model
    with Perseveration, based on the provided fitting code.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters matching the source model
    learning_rate, beta, w, perseveration = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Model state initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)       # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2))  # Values for Stage 2 (2 states x 2 actions)
    
    last_action_1 = -1 # Initialize with an impossible action
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Organize current trial reward probabilities
        # state 0 corresponds to drift1/drift2, state 1 corresponds to drift3/drift4
        current_reward_probs = np.array([
            [drift1[t], drift2[t]], 
            [drift3[t], drift4[t]]
        ])

        # 2. Stage 1 Choice
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply perseveration bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # Softmax selection Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # Determine next state based on transition matrix probabilities
        # state_idx 0 is "Planet 1", state_idx 1 is "Planet 2"
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        q_values_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_values_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Outcome
        r = int(rng.random() < current_reward_probs[state_idx, a2])
        
        # 6. Learning Updates
        # Note: The fitting code updates Stage 1 MF values based on the Q-value 
        # of the chosen Stage 2 action *before* the Stage 2 update occurs.
        
        # Update Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # 7. Record data for next trial
        last_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
