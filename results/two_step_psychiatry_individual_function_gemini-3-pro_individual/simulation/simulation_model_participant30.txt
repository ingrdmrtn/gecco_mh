import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free TD Learning (SARSA) model.
    
    This model assumes the participant learns values purely through experience 
    without using the transition structure. It updates Stage 1 values based on 
    the value of the state reached in Stage 2 (TD(0)).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta = parameters
    
    # Fixed transition matrix for the 2-step task (common structure)
    # Action 0 -> 70% State 0 (index 0), 30% State 1 (index 1)
    # Action 1 -> 30% State 0 (index 0), 70% State 1 (index 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: values for the two options in the first stage
    q_stage1 = np.zeros(2) 
    # q_stage2: values for the two options in each of the two second-stage states
    q_stage2 = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities organized by state [state][action]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates (SARSA logic) ---
        
        # 1. Update Stage 1 Q-value based on the value of the chosen action in Stage 2
        # TD Error: Value of next state-action pair - Value of current action
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 Q-value based on received reward
        # TD Error: Reward - Value of current action
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
