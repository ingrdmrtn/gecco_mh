import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Q-Learning with Choice Perseveration model.
    Matches the logic of cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration_strength]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, perseveration_strength = parameters
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # Values for Spaceship A vs U
    q_stage2_mf = np.zeros((2, 2)) # Values for Aliens (Planet X: W, S; Planet Y: P, H)
    
    # Transition matrix (Standard 2-step task configuration)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    prev_action_1 = -1 # No previous action on trial 0
    
    # Storage for simulation results
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize reward probabilities for the current trial
        # State 0 (Planet X) has options corresponding to drift1, drift2
        # State 1 (Planet Y) has options corresponding to drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_biased = q_stage1_mf.copy()
        
        # Apply perseveration bonus if a previous action exists
        if prev_action_1 != -1:
            q_stage1_biased[prev_action_1] += perseveration_strength
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_stage1_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 choice
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine Stage 2 state based on transition matrix probabilities
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using same beta as per fitting code)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 choice
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = 1 if np.random.random() < reward_probs[s2][a2] else 0
        
        # --- Learning Updates ---
        # Update Stage 1 Q-value based on Stage 2 Q-value (SARSA-style TD error)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 Q-value based on Reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # --- Bookkeeping ---
        prev_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
