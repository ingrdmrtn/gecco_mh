import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free model with Choice Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness = parameters
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the two choices in the first stage
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the two states (planets) x two choices (aliens)
    q_stage2_mf = np.zeros((2, 2)) 
    
    # Transition matrix (fixed for this task structure)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Previous action tracker for stickiness
    prev_action_1 = -1
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Define current reward probabilities for the second stage
        # State 0 has drift1, drift2; State 1 has drift3, drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # 2. Stage 1 Decision
        logits_1 = beta * q_stage1_mf.copy()
        
        # Add stickiness bonus if there was a previous action
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Decision
        # Calculate probabilities based on Q-values for the specific state arrived at
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Outcome
        # Determine reward based on the drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 6. Learning (Update Q-values)
        # Update Stage 2 Q-value (TD prediction error based on reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-value (TD prediction error based on Stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state, action] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update history
        prev_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
