import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Stage-Specific Learning Rates Model.
    
    This model assumes the participant learns the value of spaceships (Stage 1)
    and aliens (Stage 2) with different learning rates.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate1, learning_rate2, beta, w]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate1, learning_rate2, beta, w = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the 2 options in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the 2 states in stage 2, each having 2 options
    q_stage2_mf = np.zeros((2, 2))

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for easy indexing: reward_probs[state][action]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)

        # --- Transition ---
        # Determine Stage 2 state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 (purely Model-Free based on Stage 2 Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        # 1. Update Stage 1 MF values based on Stage 2 Q-value (TD(0)-like prediction error)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate1 * delta_stage1

        # 2. Update Stage 2 MF values based on reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate2 * delta_stage2

        # 3. Update Stage 1 MF values again based on the final reward (Eligibility trace/TD(1)-like)
        # Note: The fitting code implies lambda=1 for this specific update step as it adds the full delta_stage2
        q_stage1_mf[a1] += learning_rate1 * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
