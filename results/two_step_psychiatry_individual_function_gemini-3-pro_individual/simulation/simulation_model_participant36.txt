import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Learner with Choice Perseveration.
    
    This model relies only on direct reinforcement history. It adds a 
    'perseveration' parameter to capture the tendency to repeat the last choice 
    made at stage 1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Transition matrix (fixed for this task structure)
    # Action 0 -> 70% State 0 (index 0), 30% State 1 (index 1)
    # Action 1 -> 30% State 0 (index 0), 70% State 1 (index 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two actions (cols) in the two states (rows)
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # Initialize with impossible action

    for t in range(n_trials):
        # Organize drift probabilities for easy indexing: reward_probs[state][action]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate net Q-values including perseveration bonus
        q_net = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # Softmax policy for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Determine transition to Stage 2
        # transition_matrix[a1] gives [prob_state0, prob_state1]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax policy for Stage 2 (based on current state s2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Model-Free Updates) ---
        # Update Stage 1 Q-value
        # Note: The fitting code uses TD(1) style logic where Stage 1 is updated by the final reward directly,
        # ignoring the transition value (common in "pure model-free" implementations for this task).
        delta_stage1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store history
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
