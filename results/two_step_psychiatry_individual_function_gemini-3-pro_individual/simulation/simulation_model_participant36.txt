import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Hybrid Learner with Choice Perseveration model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, p]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
            - State 0 (Left): Action 0 -> drift1, Action 1 -> drift2
            - State 1 (Right): Action 0 -> drift3, Action 1 -> drift4

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters based on cognitive_model1 signature
    learning_rate, beta, w, p = parameters
    
    # Initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: State 0, State 1. Cols: Action 0, Action 1.
    
    last_action_1 = -1 # Indicator for no previous action
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Perseveration Bonus
        if last_action_1 != -1:
            q_net[last_action_1] += p
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine next state based on transition matrix
        state_probs = transition_matrix[a1]
        s2 = np.random.choice([0, 1], p=state_probs)
        
        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        
        # Determine reward probability based on current state and action
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < prob_reward else 0
        
        # --- Learning Updates ---
        
        # Stage 1 MF Update (SARSA-style based on Q2)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update (Prediction Error based on Reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # --- Record Data ---
        last_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
