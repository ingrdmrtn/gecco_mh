import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the TD(lambda) Learner model.
    
    This model uses eligibility traces to update Stage 1 values based on Stage 2 outcomes,
    bridging the delay between choice and reward.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lam]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, lam = parameters
    
    # Initialize Q-values
    # q_stage1 corresponds to Q(s1, a)
    q_stage1 = np.zeros(2) + 0.5
    # q_stage2 corresponds to Q(s2, a) where s2 is the second stage state (0 or 1)
    q_stage2 = np.zeros((2, 2)) + 0.5
    
    # Transition matrix (fixed for the standard 2-step task)
    # 70% common transition, 30% rare transition
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drifts to the specific state-action pair
        # State 0 (Top): Action 0 -> drift1, Action 1 -> drift2
        # State 1 (Bottom): Action 0 -> drift3, Action 1 -> drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Learning / Updates
        
        # TD Error 1: Difference between Stage 2 Value and Stage 1 Value
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        
        # TD Error 2: Difference between Reward and Stage 2 Value
        delta_2 = r - q_stage2[s2, a2]
        
        # Update Stage 1 Q-values
        # Includes immediate update from stage transition (delta_1)
        # plus eligibility trace update from final reward (lam * delta_2)
        q_stage1[a1] += learning_rate * delta_1 + learning_rate * lam * delta_2
        
        # Update Stage 2 Q-values
        q_stage2[s2, a2] += learning_rate * delta_2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
