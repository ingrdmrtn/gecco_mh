import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Learner (TD(1)).
    This model learns values for stage 1 and stage 2 purely based on received rewards,
    ignoring the transition probabilities. It updates stage 1 values directly from the final outcome.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta = parameters
    
    # Initialize Q-values
    # q_stage1 corresponds to Values for Spaceship 0 and 1
    q_stage1 = np.zeros(2)      
    # q_stage2 corresponds to Values for (Planet 0, Planet 1) x (Alien 0, Alien 1)
    q_stage2 = np.zeros((2, 2)) 
    
    # Transition matrix (fixed structure of the task, though this model ignores it for learning)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine the second stage state based on transition probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drift inputs to the specific state/action pair
        # State 0 (Planet 0) has options drift1, drift2
        # State 1 (Planet 1) has options drift3, drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        prob_reward = current_reward_probs[s2][a2]
        r = int(rng.random() < prob_reward)
        
        # 5. Value Updates
        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-value directly from the reward (TD(1) logic)
        # Note: In the provided fitting code, q_stage1 is updated by (r - q_stage1[a1])
        delta_stage1 = r - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
