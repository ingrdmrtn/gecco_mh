import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model with Choice Perseveration.
    
    This simulation mirrors the logic of 'cognitive_model1', combining Model-Based 
    and Model-Free values with a weighting parameter 'w', and adding a sticky 
    choice bonus (perseveration) to the previously chosen Stage 1 action.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, perseveration = parameters
    
    # Initialize return arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to Q_MF(s1, a)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q_MF(s2, a), shape (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2)) 
    
    # Transition matrix (fixed as per the fitting code)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 (drift1, drift2) maps to indices 0, 1
        # State 1 (drift3, drift4) maps to indices 0, 1
        current_reward_probs = [
            [drift1[t], drift2[t]], 
            [drift3[t], drift4[t]]
        ]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits = beta * q_net
        
        # Add perseveration bonus if a previous action exists
        if last_action_1 != -1:
            logits[last_action_1] += perseveration
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 Choice
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1
        
        # --- Transition ---
        # Determine Stage 2 State based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2 (pure Model-Free)
        # Note: The fitting code uses 'state' as an index. Here s2 is 0 or 1.
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 Choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning / Updates ---
        
        # Update Stage 1 MF Q-values (SARSA-like update using Q_stage2)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
