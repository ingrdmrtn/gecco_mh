import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates Model 1: Hybrid Learner (MB/MF) with Choice Perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, perseveration = parameters

    # Setup storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf corresponds to Q_MF(s1, a1)
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf corresponds to Q(s2, a2). Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2)) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1

    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Calculate Model-Based values
        # Max Q value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # MB value for stage 1 is expected max value of stage 2
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply perseveration bonus if applicable
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition
        # Determine state 2 based on transition matrix probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Softmax selection for Stage 2 (purely based on Q_stage2_mf)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Reward Generation
        # Map drifts to the specific state/action pair
        # State 0 (s2=0) has options drift1, drift2. State 1 (s2=1) has drift3, drift4.
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])

        # 5. Learning / Updates
        
        # Update Stage 1 MF value (TD(0) style update based on Stage 2 value)
        # Note: The fitting code uses q_stage2_mf[state_idx, action_2] as the target
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 MF value based on reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
