import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid MB/MF RL with Stage 1 Stickiness.
    
    This model includes:
    - Weighted combination of Model-Based and Model-Free values at Stage 1.
    - A 'stickiness' parameter that biases the agent to repeat the last Stage 1 action.
    - Standard Q-learning updates (TD(0)) without eligibility traces (lambda=0 implicitly).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness = parameters
    
    # Transition matrix (fixed structure: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for the two actions in each of the two 2nd-stage states
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Net Q-values for Stage 1
        # Model-Based Calculation: Transition Matrix * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness (perseveration) bonus to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # 2. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # 3. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice (Softmax)
        # Note: The fitting code uses the same beta for both stages
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # 5. Reward Generation
        # Map drift arrays to the specific state/action pair
        # State 0 (s2=0): Actions lead to drift1 (a2=0) or drift2 (a2=1)
        # State 1 (s2=1): Actions lead to drift3 (a2=0) or drift4 (a2=1)
        current_reward_prob = 0.0
        if s2 == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)
        
        # 6. Learning Updates
        # Update Stage 1 MF Q-value based on Stage 2 Q-value (TD(0))
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value based on Reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for stickiness
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
