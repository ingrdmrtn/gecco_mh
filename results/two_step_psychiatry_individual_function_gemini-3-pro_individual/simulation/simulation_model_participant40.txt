import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Learner with Choice Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two choices in stage 1
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf corresponds to the Q-values for the two states in stage 2, each having 2 choices
    q_stage2_mf = np.zeros((2, 2)) 

    # Transition matrix for the 2-step task (Standard: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 has options with drift1 and drift2
        # State 1 has options with drift3 and drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_effective = q_stage1_mf.copy()
        
        # Apply perseveration bias if there was a previous trial
        if prev_action_1 != -1:
            q_stage1_effective[prev_action_1] += perseveration
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_stage1_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition to Stage 2 ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Select action based on current state Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Model-Free Update) ---
        # The fitting code uses SARSA(0) logic:
        # 1. Update Stage 1 Q-value based on Stage 2 Q-value of the chosen action
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 Q-value based on received reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update tracking variables
        prev_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
