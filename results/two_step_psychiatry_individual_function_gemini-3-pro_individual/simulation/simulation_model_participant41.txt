To convert the fitting code into simulation code, we need to invert the logic: instead of calculating the likelihood of observed actions, we generate actions based on the model's internal probabilities.

Here is the simulation code for the `cognitive_model2` (Model-Free RL with Stickiness):

```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Reinforcement Learning 
    model with Choice Perseveration (Stickiness).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, eligibility_trace, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, eligibility_trace, stickiness = parameters
    
    # Setup transition matrix (fixed for this task structure)
    # Row 0: transitions from Choice 0 -> [State 0, State 1]
    # Row 1: transitions from Choice 1 -> [State 0, State 1]
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions at stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two actions in each of the two states at stage 2
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # Initialize with no previous action
    rng = np.random.default_rng() # Initialize random number generator

    for t in range(n_trials):
        # Current trial reward probabilities for State 0 (drift1, drift2) and State 1 (drift3, drift4)
        # Organized as [State][Action]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_net_stage1 = q_stage1_mf.copy()
        
        # Apply stickiness (choice perseveration)
        if last_action_1 != -1:
            q_net_stage1[last_action_1] += stickiness
            
        # Softmax choice rule for Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition to Stage 2 ---
        # Determine state based on transition matrix probabilities
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # --- Stage 2 Decision ---
        # State index is s2 (0 or 1)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning Updates ---
        # 1. Prediction error at Stage 1 (TD(0))
        # Value of state 2 is approximated by the Q-value of the chosen action
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 2. Prediction error at Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # 3. Eligibility trace update for Stage 1 (TD(1))
        q_stage1_mf[a1] += learning_rate * eligibility_trace * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
```
