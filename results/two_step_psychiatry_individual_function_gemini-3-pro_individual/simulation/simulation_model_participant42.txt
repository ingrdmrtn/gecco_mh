import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates the Hybrid Learner with Choice Perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, perseveration = parameters

    # Fixed transition matrix as defined in the fitting code
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)       
    
    # q_stage2_mf corresponds to Q-values for stage 2: shape (2 states, 2 actions)
    # Note: fitting code uses `state[trial]` as index. Usually states are 0 and 1.
    q_stage2_mf = np.zeros((2, 2))  

    # Track previous action for perseveration
    last_action_1 = -1

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities for the 4 terminal states
        # State 0 has options -> drift1, drift2
        # State 1 has options -> drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        # Max Q-value available in each stage 2 state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values (weighted average of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply perseveration bonus if not the first trial
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine stage 2 state based on transition matrix and stage 1 choice
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 (purely Model-Free based on state)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        
        # Stage 1 MF update (TD(0))
        # Note: The fitting code updates stage 1 based on the *value* of the chosen stage 2 option
        chosen_stage2_value = q_stage2_mf[s2, a2]
        delta_stage1 = chosen_stage2_value - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 MF update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
