import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free RL with Outcome-Specific Perseverance.
    Matches the logic of cognitive_model2.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, pers_rew, pers_unrew]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, pers_rew, pers_unrew = parameters
    
    # Standard 2-step task transition matrix
    # Action 0 -> 70% State 0 (index 0), 30% State 1 (index 1)
    # Action 1 -> 30% State 0 (index 0), 70% State 1 (index 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: [Action 0, Action 1]
    q_stage1 = np.zeros(2)
    # q_stage2: [State 0/1, Action 0/1]
    q_stage2 = np.zeros((2, 2))
    
    # History tracking variables
    last_action_1 = -1
    last_reward = 0

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize current trial reward probabilities
        # State 0 corresponds to drifts 1 and 2
        # State 1 corresponds to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Choice ---
        q_hybrid = q_stage1.copy()

        # Apply perseverance bonus based on previous trial outcome
        if last_action_1 != -1:
            if last_reward == 1:
                q_hybrid[last_action_1] += pers_rew
            else:
                q_hybrid[last_action_1] += pers_unrew

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_hybrid)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition to Stage 2 ---
        # Determine state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Choice ---
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Generation ---
        r = int(rng.random() < current_reward_probs[s2_idx][a2])

        # --- Learning Updates ---
        # Note: Logic follows the fitting code provided.
        # Stage 1 update uses the Q-value of the chosen Stage 2 action (SARSA-style for this step)
        delta_stage1 = q_stage2[s2_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

        # Stage 2 update uses the actual reward
        delta_stage2 = r - q_stage2[s2_idx, a2]
        q_stage2[s2_idx, a2] += learning_rate * delta_stage2
        
        # --- Update History ---
        last_action_1 = a1
        last_reward = r

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
