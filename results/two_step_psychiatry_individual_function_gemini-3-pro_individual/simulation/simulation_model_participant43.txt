import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Learner (TD Learning).
    
    This model learns values for Stage 1 actions based on the values of the 
    Stage 2 states they lead to (Temporal Difference learning), without 
    using the transition matrix. It represents habitual behavior.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, eligibility_trace]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, eligibility_trace = parameters
    
    # Set up random number generator
    rng = np.random.default_rng()

    # Initialize Q-values
    # q_stage1_mf corresponds to Q(s1, a)
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf corresponds to Q(s2, a) where s2 is index 0 or 1
    q_stage2_mf = np.zeros((2, 2))
    
    # Define transition structure (common 70/30 split for 2-step tasks)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for trial in range(n_trials):
        # Organize drift probabilities for easy indexing: reward_probs[state][action]
        reward_probs = [[drift1[trial], drift2[trial]], 
                        [drift3[trial], drift4[trial]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (TD Updates) ---
        
        # 1. Update Stage 1 Q-value based on Stage 2 Q-value (TD Prediction Error)
        # Prediction error: Value of next state (Q(s2, a2)) - Value of current state (Q(s1, a1))
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 Q-value based on Reward (Reward Prediction Error)
        # Prediction error: Reward - Value of current state (Q(s2, a2))
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 3. Eligibility Trace Update for Stage 1
        # Propagate the Stage 2 RPE back to Stage 1 Q-value
        q_stage1_mf[a1] += learning_rate * eligibility_trace * delta_stage2

        # Store data
        stage1_choice[trial] = a1
        state2[trial] = s2
        stage2_choice[trial] = a2
        reward[trial] = r

    return stage1_choice, state2, stage2_choice, reward
