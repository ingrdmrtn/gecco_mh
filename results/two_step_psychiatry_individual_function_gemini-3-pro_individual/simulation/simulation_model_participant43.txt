import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Reinforcement Learning agent (TD-Learning).
    
    This agent learns Stage 1 values by updating them based on the value of the 
    state reached in Stage 2. It ignores transition probabilities (rare vs common).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, eligibility_trace]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, eligibility_trace = parameters
    
    # Transition matrix structure (fixed for the task)
    # Action 0 -> 70% State 0 (index 0), 30% State 1 (index 1)
    # Action 1 -> 30% State 0 (index 0), 70% State 1 (index 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf corresponds to the values of the two spaceships (Stage 1 choices)
    q_stage1_mf = np.zeros(2)       
    # q_stage2_mf corresponds to the values of the aliens (Stage 2 choices)
    # Shape (2, 2): 2 states (planets), 2 actions (aliens) per state
    q_stage2_mf = np.zeros((2, 2))  

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for easier access: [state][action]
        # State 0 (Planets A/B) uses drift1, drift2
        # State 1 (Planets C/D) uses drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning / Updates ---
        
        # 1. Update Stage 1 value based on the value of the state reached (TD(0) logic for Stage 1)
        # Prediction Error 1: Value of chosen Stage 2 option - Value of chosen Stage 1 option
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 value based on received reward
        # Prediction Error 2: Reward - Value of chosen Stage 2 option
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 3. Eligibility Trace Update
        # Allow the Stage 2 prediction error to propagate back to update Stage 1 value further
        q_stage1_mf[a1] += learning_rate * eligibility_trace * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
