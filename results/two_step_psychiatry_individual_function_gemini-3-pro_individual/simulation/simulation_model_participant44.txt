import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Learner with Passive Decay (Forgetting).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, decay_rate, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, decay_rate, beta = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1 corresponds to the Q-values for the first stage actions (0 and 1)
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the Q-values for the second stage:
    # Row 0: State 0 (from drift1/drift2), Actions 0 and 1
    # Row 1: State 1 (from drift3/drift4), Actions 0 and 1
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed 70/30 structure standard for this task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on action 1 and transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s_idx][a2])
        
        # --- Learning / Updates ---
        
        # Update chosen action in Stage 1
        q_stage1[a1] += learning_rate * (r - q_stage1[a1])
        
        # Decay unchosen action in Stage 1
        unchosen_a1 = 1 - a1
        q_stage1[unchosen_a1] *= (1 - decay_rate)

        # Update chosen action in Stage 2 (specific to the state visited)
        q_stage2[s_idx, a2] += learning_rate * (r - q_stage2[s_idx, a2])
        
        # Decay unchosen action in Stage 2 (specific to the state visited)
        unchosen_a2 = 1 - a2
        q_stage2[s_idx, unchosen_a2] *= (1 - decay_rate)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
