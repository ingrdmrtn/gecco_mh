import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    specified in cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseverance]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters corresponding to cognitive_model1
    learning_rate, beta, w, perseverance = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Model constants and containers
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    # Output storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # Organize reward probabilities for the current trial
        # State 0: drift1, drift2; State 1: drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Hybrid value
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Calculate logits
        logits = beta * q_hybrid
        
        # Apply perseverance bonus to the previously chosen action
        if prev_action_1 != -1:
            logits[prev_action_1] += perseverance

        # Softmax for Stage 1
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine next state based on transition matrix probabilities for action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Calculate logits for Stage 2 (using same beta as per fitting code)
        logits_2 = beta * q_stage2_mf[s2]
        
        # Softmax for Stage 2
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        
        # Update Stage 1 MF Q-value (TD learning using Q of next state/action)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1

        # Update Stage 2 MF Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] = q_stage2_mf[s2, a2] + learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous action for next trial
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
