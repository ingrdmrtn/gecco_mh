def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Model with Direct Reinforcement (TD(1)).
    
    This model ignores the transition structure (Model-Free). It updates Stage 1 Q-values
    directly based on the final reward received (TD-lambda=1), rather than bootstrapping
    off the Stage 2 value. This creates a strong tendency to repeat successful Stage 1
    actions regardless of the transition type (common/rare).
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - decay: [0, 1] Decay rate for unchosen options (passive forgetting) to allow flexibility.
    """
    learning_rate, beta, decay = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)     # Values for spaceships
    q_stage2_mf = np.zeros((2, 2)) # Values for aliens

    for trial in range(n_trials):

        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]


        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2



        delta_stage1_direct = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1_direct

        unchosen_1 = 1 - action_1[trial]
        q_stage1_mf[unchosen_1] *= (1 - decay)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
