Here are three cognitive models based on the provided template, ranging from pure model-free learning to a hybrid model-based/model-free approach, and finally a model incorporating perseveration.

### Model 1: Pure Model-Free TD Learning (SARSA-like)
This model assumes the participant does not use knowledge of the transition structure (which spaceship goes where). Instead, they learn the value of the first-stage spaceships solely based on the eventual reward, chaining values backward using Temporal Difference (TD) learning.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Learner (TD-0).
    
    This model ignores the transition structure (Model-Based) and learns purely 
    by chaining prediction errors. It updates Stage 1 values based on the value 
    of the chosen Stage 2 state, and Stage 2 values based on the reward.

    Parameters:
    learning_rate: [0,1] Rate at which Q-values are updated (alpha).
    beta: [0,10] Inverse temperature for softmax choice (exploration/exploitation).
    """
    learning_rate, beta = model_parameters
    n_trials = len(action_1)
  
    # Initialize Q-values
    # q_stage1_mf: Values for Spaceship A (0) and U (1)
    q_stage1_mf = np.zeros(2)  # Initially 0.5 or 0 is fine, 0 is standard
    # q_stage2_mf: Values for Planet X (0) -> Aliens 0/1, Planet Y (1) -> Aliens 0/1
    q_stage2_mf = np.zeros((2, 2))

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy (Softmax on MF values) ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Observe state transition
        state_idx = state[trial] # 0 for Planet X, 1 for Planet Y

        # --- Stage 2 Policy (Softmax on Stage 2 MF values) ---
        # Select the row corresponding to the current planet (state_idx)
        current_planet_qs = q_stage2_mf[state_idx, :]
        exp_q2 = np.exp(beta * current_planet_qs)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updating ---
        # 1. Update Stage 1 Q-value based on the value of the state we arrived at (TD-0)
        # Note: We use the value of the action actually taken in stage 2 as the target
        chosen_stage2_val = q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = chosen_stage2_val - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # 2. Update Stage 2 Q-value based on the reward received
        delta_stage2 = reward[trial] - chosen_stage2_val
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model-Based / Model-Free Learner
This is the classic "Daw et al. (2011)" style model. It calculates two values for the first stage: a Model-Free value (from direct experience) and a Model-Based value (calculated by combining the transition matrix with the values of the second-stage aliens). A mixing parameter `w` determines the balance between these two strategies.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Learner.
    
    Combines a Model-Free (TD) estimate with a Model-Based (planning) estimate
    for the first-stage choice.
    
    Parameters:
    learning_rate: [0,1] Alpha for updating Q-values.
    beta: [0,10] Inverse temperature.
    w: [0,1] Mixing weight. 0 = Pure Model-Free, 1 = Pure Model-Based.
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as described in prompt (Spaceship A->X (0.7), U->Y (0.7))
    # Row 0: Spaceship A probs [to X, to Y], Row 1: Spaceship U probs [to X, to Y]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # Model-Based Value: Expected value of best option at next stage * transition prob
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Max value available on Planet X, Planet Y
        q_stage1_mb = transition_matrix @ max_q_stage2 
        
        # Hybrid Value: weighted sum
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 values
        current_planet_qs = q_stage2_mf[state_idx, :]
        exp_q2 = np.exp(beta * current_planet_qs)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updating ---
        # Update Stage 1 MF value (TD-0)
        chosen_stage2_val = q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = chosen_stage2_val - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 MF value (Reward Prediction Error)
        delta_stage2 = reward[trial] - chosen_stage2_val
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with Choice Perseveration
This model extends the Hybrid approach by adding a "perseveration" parameter. Humans often have a "stickiness" bias where they tend to repeat their previous Stage 1 choice regardless of reward, or switch more often than expected. This model captures that low-level motor or cognitive repetition.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Learner with Choice Perseveration.
    
    Adds a 'stickiness' parameter to the first stage choice, biasing the agent
    to repeat the previously selected spaceship.
    
    Parameters:
    learning_rate: [0,1] Alpha.
    beta: [0,10] Inverse temperature.
    w: [0,1] Model-based weight.
    perseveration: [0,5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Track previous choice (initially -1 or dummy)
    last_action_1 = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add Perseveration Bonus
        # We create a temporary Q vector for decision making only
        q_decision = q_net.copy()
        if last_action_1 != -1:
            q_decision[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store action for next trial's perseveration
        last_action_1 = action_1[trial]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        current_planet_qs = q_stage2_mf[state_idx, :]
        exp_q2 = np.exp(beta * current_planet_qs)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updating ---
        chosen_stage2_val = q_stage2_mf[state_idx, action_2[trial]]
        
        # Stage 1 Update
        delta_stage1 = chosen_stage2_val - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - chosen_stage2_val
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
