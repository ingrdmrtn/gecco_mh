Here are three cognitive models designed to explain the participant's behavior in the two-step decision task.

### Model 1: Hybrid Learner (Model-Based + Model-Free)
This is the classic "Hybrid" model often used for this task. It assumes the participant uses a mixture of Model-Based (MB) planning (using the transition structure) and Model-Free (MF) learning (reinforcement from direct experience) to make their Stage 1 choice.

*   **Key Concept:** The Stage 1 Q-value is a weighted sum of $Q_{MB}$ and $Q_{MF}$.
*   **Parameters:**
    *   `learning_rate`: Controls how fast Q-values update.
    *   `beta_1`: Inverse temperature for Stage 1 choice (exploration vs. exploitation).
    *   `beta_2`: Inverse temperature for Stage 2 choice.
    *   `w`: The mixing weight (0 = Pure Model-Free, 1 = Pure Model-Based).

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Learner.
    
    This model assumes the agent combines model-free values (learned from direct 
    experience) and model-based values (computed using the transition matrix) 
    to make the first-stage decision.
    
    Bounds:
    learning_rate: [0, 1]
    beta_1: [0, 10]
    beta_2: [0, 10]
    w: [0, 1] (Weight for Model-Based control)
    """
    learning_rate, beta_1, beta_2, w = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition structure: A->X (0->0) is 0.7, etc.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for Stage 1 actions (0 or 1)
    q_stage2_mf = np.zeros((2, 2)) # Values for Stage 2 actions (aliens) per state (planets)

    for trial in range(n_trials):
        # --- STAGE 1 POLICY ---
        # 1. Calculate Model-Based values
        # Max value available at stage 2 for each state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Expected value of stage 1 actions based on transition probs
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Softmax choice probability
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- STAGE 2 POLICY ---
        state_idx = state[trial] # 0 or 1 (Planet X or Y)
        
        # Standard softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- LEARNING ---
        # Note: In standard TD(0) or SARSA for this task, we often update Stage 1 MF 
        # using the value of the state actually reached (Q_stage2[state, chosen_action]).
        
        # Update Stage 1 MF value
        # Prediction error: Value of state reached - Value of chosen spaceship
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 MF value
        # Prediction error: Reward received - Value of chosen alien
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free with Eligibility Traces (TD($\lambda$))
This model assumes the participant does *not* use the transition structure (transition matrix) explicitly. Instead, they rely purely on reinforcement learning but use an eligibility trace ($\lambda$) to allow the final reward to "trickle back" to the first-stage choice directly.

*   **Key Concept:** The reward at the end updates the Stage 2 choice, but also updates the Stage 1 choice scaled by $\lambda$.
*   **Parameters:**
    *   `learning_rate`: Learning rate.
    *   `beta`: Inverse temperature (shared for both stages).
    *   `lambd`: Eligibility trace parameter (0 = TD(0), 1 = Monte Carlo).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Learner with Eligibility Traces (TD-Lambda).
    
    This model assumes no knowledge of the transition structure. It updates
    Stage 1 values based on the final reward, scaled by the eligibility trace parameter lambda.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    lambd: [0, 1] (Eligibility trace decay)
    """
    learning_rate, beta, lambd = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- STAGE 1 POLICY ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- LEARNING ---
        
        # 1. Stage 1 prediction error (TD error)
        # Based on the value of the state reached (Q of the chosen stage 2 action)
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        
        # Update Stage 1 based on immediate transition
        q_stage1[action_1[trial]] += learning_rate * delta_1
        
        # 2. Stage 2 prediction error
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        # Update Stage 2
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        # 3. Eligibility Trace Update
        # The Stage 2 error (delta_2) also updates Stage 1, scaled by lambda
        q_stage1[action_1[trial]] += learning_rate * lambd * delta_2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Based with Perseveration
This model is primarily Model-Based (calculating values via the transition matrix) but includes a "Perseveration" parameter. This accounts for a common heuristic where participants tend to repeat their previous Stage 1 choice regardless of reward or structure (stickiness), or switch (alternation).

*   **Key Concept:** Choices are driven by MB planning, but modified by a tendency to repeat the last action.
*   **Parameters:**
    *   `learning_rate`: Learning rate for Stage 2 values.
    *   `beta`: Inverse temperature.
    *   `perseveration`: Bonus added to the logits of the previously chosen action. Positive values = sticky, Negative = alternating.
    *   `w`: Weighting for MB vs MF (to allow partial MB logic, though we focus on MB here, we keep the hybrid structure to make it robust).

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Choice Perseveration.
    
    Adds a 'stickiness' parameter. If positive, the agent tends to repeat 
    the previous Stage 1 action. If negative, they tend to switch.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    perseveration: [0, 5] (Bias added to previous choice logits)
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous action for perseveration (initialize randomly or with -1)
    last_action_1 = -1

    for trial in range(n_trials):

        # --- STAGE 1 POLICY ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits = beta * q_net
        
        # Add Perseveration Bonus
        if last_action_1 != -1:
            logits[last_action_1] += perseveration
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store action for next trial
        last_action_1 = action_1[trial]
        
        state_idx = state[trial]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- LEARNING ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
