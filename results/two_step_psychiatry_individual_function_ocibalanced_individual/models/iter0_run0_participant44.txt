Here are three cognitive models analyzing the participant's behavior in the two-step task. Each model hypothesizes a different balance of model-based planning, model-free learning, and perseveration.

### Model 1: Pure Model-Based Reinforcement Learning
This model assumes the participant builds a mental map of the task structure (transition probabilities) and plans their first-stage choices by calculating the expected value of the second stage. It ignores direct reinforcement of the first-stage choice (Model-Free Stage 1) and relies entirely on the computed values derived from the second stage.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Reinforcement Learning Model.
    
    This model assumes the participant acts purely based on a model of the task structure.
    They learn the values of the second-stage aliens (Model-Free Q-learning at Stage 2)
    and compute the value of Stage 1 choices by multiplying the transition matrix 
    by the max value of the destination states (Bellman equation).
    
    Parameters:
    - learning_rate: [0, 1] Rate at which second-stage Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice (exploration-exploitation trade-off).
    """
    learning_rate, beta = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition matrix as described in task (70% common, 30% rare)
    # Row 0: Spaceship A (0) -> [Planet X (0), Planet Y (1)]
    # Row 1: Spaceship U (1) -> [Planet X (0), Planet Y (1)]
    # Note: The prompt implies A->X and U->Y are common. 
    # If A=0, U=1, X=0, Y=1, then 0->0 is 0.7, 1->1 is 0.7.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values for Stage 2: 2 planets x 2 aliens
    q_stage2_mf = np.zeros((2, 2)) 
    # Stage 1 values are calculated on the fly (Model-Based), so no stored Q-table needed.

    for trial in range(n_trials):

        # --- STAGE 1 POLICY (Model-Based) ---
        # Calculate expected value of each spaceship based on transition matrix and best stage 2 option
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Max value available at each planet
        q_stage1_mb = transition_matrix @ max_q_stage2 # Bellman equation
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # Which planet did we land on?

        # --- STAGE 2 POLICY (Model-Free) ---
        # Standard value-based choice at the second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- LEARNING ---
        # Only Stage 2 Q-values are updated based on reward.
        # Stage 1 values change implicitly because q_stage2_mf changes.
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model-Based / Model-Free RL
This is the classic "Daw two-step" model. It posits that participants use a weighted combination of Model-Free learning (repeating what was just rewarded, regardless of transition type) and Model-Based learning (planning based on task structure). A mixing parameter `w` controls the balance between these two systems.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning Model.
    
    This model combines a habit-based system (Model-Free) and a planning system (Model-Based).
    The final Q-value for Stage 1 is a weighted sum of both systems.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for both stage 1 and stage 2 Q-values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting parameter. 1 = Pure Model-Based, 0 = Pure Model-Free.
    """
    learning_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)      # Model-Free values for spaceships
    q_stage2_mf = np.zeros((2, 2)) # Model-Free values for aliens

    for trial in range(n_trials):

        # --- STAGE 1 POLICY (Hybrid) ---
        # 1. Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Integrate with Model-Free values using weight w
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- LEARNING ---
        # Update Stage 1 Model-Free value (TD(1)-like update using the final reward)
        # Note: In simple hybrid models, we often update Q1 based on Q2 (SARSA) or Reward directly.
        # Here we use the direct reward to update the MF value of the chosen spaceship.
        # This captures the "habit" of repeating a rewarded choice.
        delta_stage1 = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 Model-Free value
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid RL with Choice Perseveration
This model extends the Hybrid model by adding a "perseveration" parameter. Participants often have a "stickiness" bias where they tend to repeat their previous Stage 1 choice regardless of reward or model-based expectations. This captures motor perserveration or cognitive inertia.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid RL Model with Choice Perseveration.
    
    Adds a 'stickiness' parameter to the hybrid model. This accounts for the tendency 
    to repeat the previous Stage 1 action regardless of outcomes.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - perseveration: [0, 5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous action (initialize with -1 or handle first trial logic)
    last_action_1 = -1

    for trial in range(n_trials):

        # --- STAGE 1 POLICY ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add perseveration bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store action for next trial's perseveration
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- LEARNING ---
        # Update Stage 1 MF
        delta_stage1 = reward[trial] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
