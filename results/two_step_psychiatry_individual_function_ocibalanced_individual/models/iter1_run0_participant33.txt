Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task. These models introduce variations in learning rates, eligibility traces, and distinct temperature parameters for different stages.

### Model 1: Separate Learning Rates for Stages
This model hypothesizes that the participant learns at different speeds for the first-stage choice (choosing a spaceship) versus the second-stage choice (choosing an alien). The first stage involves learning about transitions and aggregated value, while the second stage is purely reward-based.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Learning Rates for Stage 1 and Stage 2.
    
    This model allows the agent to update values at different rates for 
    the spaceship choice (stage 1) vs the alien choice (stage 2).
    
    Bounds:
    lr_stage1: [0, 1]
    lr_stage2: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    """
    lr_stage1, lr_stage2, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1 MF values using Stage 2 values (SARSA-style TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_stage1 * delta_stage1
        
        # Update Stage 2 values using reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_stage2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: SARSA($\lambda$) Model-Free Only
This model removes the model-based component entirely (w=0 implicitly) but introduces an eligibility trace parameter ($\lambda$). This allows the reward received at the end of the trial to directly influence the value of the first-stage action, rather than just relying on the second-stage value as a proxy. This is a pure temporal difference learning model with eligibility traces.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free SARSA(lambda) Model.
    
    Instead of a hybrid MB/MF mix, this model assumes pure model-free learning
    but uses an eligibility trace (lambda) to allow the stage 2 reward to 
    directly impact stage 1 values, bridging the temporal gap.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    lambd: [0, 1] (Eligibility trace decay)
    """
    learning_rate, beta, lambd = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix not used in pure MF, but kept for structural consistency if needed
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice (Pure MF)
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Prediction Error 1: Difference between Stage 2 Value and Stage 1 Value
        pe_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        
        # Prediction Error 2: Difference between Reward and Stage 2 Value
        pe_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        # Update Stage 2
        q_stage2[state_idx, action_2[trial]] += learning_rate * pe_2
        
        # Update Stage 1
        # It gets updated by its own PE, plus a fraction (lambda) of the second stage PE
        q_stage1[action_1[trial]] += learning_rate * (pe_1 + lambd * pe_2)
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Separate Temperatures
This model acknowledges that the stochasticity of choice might differ between the high-level planning stage (Stage 1) and the concrete bandit stage (Stage 2). It uses a hybrid Model-Based/Model-Free architecture but assigns a distinct inverse temperature (`beta`) to each stage.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Inverse Temperatures (Betas).
    
    This model assumes the level of exploration/exploitation differs 
    between the spaceship choice (Stage 1) and the alien choice (Stage 2).
    
    Bounds:
    learning_rate: [0, 1]
    beta_1: [0, 10] (Inverse temp for Stage 1)
    beta_2: [0, 10] (Inverse temp for Stage 2)
    w: [0, 1]
    """
    learning_rate, beta_1, beta_2, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use beta_1 for stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        # Use beta_2 for stage 2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
