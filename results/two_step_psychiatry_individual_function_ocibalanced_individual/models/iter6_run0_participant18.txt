Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Learner with Reward-Based Stickiness
This model extends the standard hybrid (MB/MF) model by hypothesizing that "stickiness" (perseverance) is not just a blind repetition of the last choice, but is modulated by whether the previous trial was rewarded. A win-stay, lose-shift dynamic is often observed alongside reinforcement learning.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Reward-Modulated Stickiness.

    This model combines Model-Based (MB) and Model-Free (MF) systems with a 
    stickiness parameter that varies based on the previous outcome.
    It tests the hypothesis that participants are more likely to repeat a 
    Stage 1 choice if the previous trial resulted in a reward.

    Parameters:
    learning_rate: [0, 1] - Rate at which MF Q-values are updated.
    beta: [0, 10] - Inverse temperature for softmax choice rule.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    stick_reward: [0, 5] - Bonus for repeating the previous choice if it was REWARDED.
    stick_unreward: [0, 5] - Bonus for repeating the previous choice if it was UNREWARDED.
    """
    learning_rate, beta, w, stick_reward, stick_unreward = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply reward-dependent stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                q_net_stage1[last_action_1] += stick_reward
            else:
                q_net_stage1[last_action_1] += stick_unreward

        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            p_choice_1[trial] = probs_1[action_1[trial]]
            last_action_1 = action_1[trial]
        else:
            p_choice_1[trial] = 1.0
            last_action_1 = -1

        # Skip update if data is missing
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            last_reward = 0 # Default reset
            continue
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Action value updating
        # TD(0) for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # TD(0) for Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Based with Forgetting
This model investigates if the participant relies solely on a Model-Based strategy (planning based on transition probabilities) but suffers from memory decay (forgetting) of the second-stage values. Instead of a hybrid `w` parameter, it assumes `w=1` but introduces a decay rate `phi` that pulls Q-values back to 0.5 (uncertainty) or 0 on every trial.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner with Passive Forgetting.

    This model assumes the participant uses a pure Model-Based strategy for Stage 1
    (calculating values based on the transition matrix). However, the stored values 
    for the second stage (the aliens) decay over time toward zero, representing 
    forgetting or passive value erosion.

    Parameters:
    learning_rate: [0, 1] - Rate at which Stage 2 Q-values are updated from rewards.
    beta: [0, 10] - Inverse temperature for softmax choice rule.
    phi: [0, 1] - Decay rate. 0 = no decay, 1 = full decay (forgetting) per trial.
    stickiness: [0, 5] - Tendency to repeat the previous Stage 1 choice.
    """
    learning_rate, beta, phi, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Only Stage 2 Q-values are learned directly; Stage 1 is derived.
    q_stage2_mf = np.zeros((2, 2)) 
    
    last_action_1 = -1

    for trial in range(n_trials):

        # policy for the first choice: Pure Model-Based Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Apply stickiness to the MB values
        if last_action_1 != -1:
            q_stage1_mb[last_action_1] += stickiness

        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            p_choice_1[trial] = probs_1[action_1[trial]]
            last_action_1 = action_1[trial]
        else:
            p_choice_1[trial] = 1.0
            last_action_1 = -1

        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            # Apply decay even on missing trials
            q_stage2_mf = (1 - phi) * q_stage2_mf
            continue
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Update Stage 2 values based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Apply Forgetting / Decay to ALL Stage 2 values
        # This pulls unvisited states/actions towards 0
        q_stage2_mf = (1 - phi) * q_stage2_mf

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with Separate Learning Rates for Stages
Standard models often use a single learning rate for both Stage 1 (MF update) and Stage 2 (MF update). This model proposes that learning about the immediate transition (Stage 1) might happen at a different rate than learning about the final reward (Stage 2). It keeps the hybrid `w` parameter to arbitrate between MB and MF control.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Separate Learning Rates.

    This model allows for different plasticity in the two stages of the task.
    'lr_1' controls the update of the Model-Free value of the spaceships (Stage 1),
    while 'lr_2' controls the update of the aliens (Stage 2). This captures 
    scenarios where the participant learns about the aliens quickly but updates 
    their habitual spaceship preference slowly (or vice versa).

    Parameters:
    lr_1: [0, 1] - Learning rate for Stage 1 MF values (Spaceships).
    lr_2: [0, 1] - Learning rate for Stage 2 MF values (Aliens).
    beta: [0, 10] - Inverse temperature for softmax choice rule.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    lr_1, lr_2, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            p_choice_1[trial] = probs_1[action_1[trial]]
        else:
            p_choice_1[trial] = 1.0

        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Action value updating
        
        # Stage 1 Update: Uses lr_1
        # The prediction error is driven by the value of the state arrived at (Stage 2 Q)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_1 * delta_stage1

        # Stage 2 Update: Uses lr_2
        # The prediction error is driven by the actual reward received
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
