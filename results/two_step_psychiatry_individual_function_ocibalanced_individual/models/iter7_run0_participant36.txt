Here are three new cognitive models for the two-step task, implemented as requested.

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Passive Decay.
    This model assumes that the values of unchosen options decay over time (forgetting),
    rather than staying static until chosen again.
    
    Parameters:
    learning_rate: [0,1] - Update rate for chosen options.
    beta: [0,10] - Inverse temperature (softmax sensitivity).
    w: [0,1] - Weighting between model-based and model-free values (0=MF, 1=MB).
    decay: [0,1] - Rate at which unchosen action values decay to 0 (0=no decay, 1=instant forgetting).
    """
    learning_rate, beta, w, decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # ACTION VALUE UPDATING FOR CHOICE 1
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        # Decay the unchosen option
        q_stage1_mf[1 - action_1[trial]] *= (1 - decay)

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # ACTION VALUE UPDATING FOR CHOICE 2
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        # Decay the unchosen option in the current state
        q_stage2_mf[state_idx, 1 - action_2[trial]] *= (1 - decay)
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Dynamic Transition Learning.
    Unlike standard models that assume fixed 70/30 transitions, this agent
    updates their internal model of spaceship-planet transitions based on experience.
    
    Parameters:
    lr_val: [0,1] - Learning rate for Q-values (reward learning).
    lr_trans: [0,1] - Learning rate for the transition matrix (structure learning).
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting between model-based and model-free values.
    """
    lr_val, lr_trans, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # ACTION VALUE UPDATING FOR CHOICE 1
        q_stage1_mf[action_1[trial]] += lr_val * delta_stage1
        
        # Update transition matrix: Move probability toward the observed transition
        act = action_1[trial]
        st = state_idx
        transition_matrix[act, st] += lr_trans * (1 - transition_matrix[act, st])
        transition_matrix[act, 1-st] += lr_trans * (0 - transition_matrix[act, 1-st])

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # ACTION VALUE UPDATING FOR CHOICE 2
        q_stage2_mf[state_idx, action_2[trial]] += lr_val * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Eligibility Traces and Choice Stickiness.
    Combines hybrid valuation with eligibility traces (allowing the Stage 2 outcome 
    to directly update Stage 1 values via lambda) and a tendency to repeat the previous Stage 1 choice.
    
    Parameters:
    learning_rate: [0,1] - Update rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF weight.
    lam: [0,1] - Eligibility trace decay (lambda). Allows Stage 2 RPE to update Stage 1 Q-values.
    stickiness: [0,5] - Choice perseveration bonus added to the previously chosen option.
    """
    learning_rate, beta, w, lam, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if trial > 0:
            q_net[action_1[trial-1]] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # ACTION VALUE UPDATING FOR CHOICE 1
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # ACTION VALUE UPDATING FOR CHOICE 2
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace update: Stage 1 value is also updated by Stage 2 RPE
        q_stage1_mf[action_1[trial]] += learning_rate * lam * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
