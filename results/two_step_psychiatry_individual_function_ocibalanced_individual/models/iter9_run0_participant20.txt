Here are three new cognitive models exploring different mechanisms for learning and decision-making in the two-step task.

### Model 1: Hybrid Learner with Inverse Temperature Modulation
This model posits that the degree of exploration (inverse temperature `beta`) is not static but is modulated by the most recent reward outcome. A large reward might encourage exploitation (higher beta), while the absence of reward encourages exploration (lower beta).

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Reward-Modulated Inverse Temperature.
    
    The inverse temperature (beta) is scaled by a parameter `phi` based on the previous trial's outcome.
    If the previous trial was rewarded, beta increases (exploitation). If unrewarded, beta stays at baseline.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta_base: [0,10] Baseline inverse temperature.
    - phi: [0,5] Modulation factor. Effective beta = beta_base * (1 + phi * prev_reward).
    - w: [0,1] Weighting parameter (0=Pure MF, 1=Pure MB).
    """
    learning_rate, beta_base, phi, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_reward = 0.0 # Initialize previous reward

    for trial in range(n_trials):
        # Modulate beta based on previous reward
        current_beta = beta_base * (1.0 + phi * prev_reward)

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update values
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Learner with Forgetting (Value Decay)
This model introduces a `decay_rate` parameter. Instead of just updating the chosen action's value, the unchosen actions' values decay toward 0 (or a neutral point) on every trial. This captures the idea that memory of unvisited states fades over time.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Passive Value Decay (Forgetting).
    
    Unchosen options decay towards 0 at a rate determined by `decay_rate`.
    This simulates forgetting of action values for options not recently taken.
    
    Parameters:
    - learning_rate: [0,1] Update rate for chosen actions.
    - decay_rate: [0,1] Decay rate for unchosen actions (0=no decay, 1=instant forgetting).
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weighting parameter (0=Pure MF, 1=Pure MB).
    """
    learning_rate, decay_rate, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update chosen values
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Decay unchosen values
        # Stage 1 unchosen
        unchosen_1 = 1 - action_1[trial]
        q_stage1_mf[unchosen_1] *= (1 - decay_rate)
        
        # Stage 2 unchosen (in the current state)
        unchosen_2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_2] *= (1 - decay_rate)
        
        # Note: We do not decay the values of the unvisited state in stage 2 here
        # to keep the parameter count low and logic simple, though variations exist.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with Outcome-Specific Learning Rates
This model splits the learning rate into two separate parameters based on the sign of the prediction error (or the nature of the outcome). Specifically, it uses `alpha_reward` for rewarded trials (positive outcomes) and `alpha_punish` for unrewarded trials (zero outcomes), allowing for asymmetry in how participants learn from gains versus losses (or lack thereof).

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Asymmetric Learning Rates for Reward vs. No-Reward.
    
    Uses different learning rates depending on whether the trial resulted in a reward (1)
    or no reward (0). This captures potential optimism/pessimism biases.
    
    Parameters:
    - alpha_reward: [0,1] Learning rate when reward == 1.
    - alpha_noreward: [0,1] Learning rate when reward == 0.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weighting parameter (0=Pure MF, 1=Pure MB).
    """
    alpha_reward, alpha_noreward, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Determine which learning rate to use
        if reward[trial] == 1:
            lr = alpha_reward
        else:
            lr = alpha_noreward

        # Update values
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
