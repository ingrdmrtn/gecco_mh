import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Reinforcement Learning 
    with Choice Perseveration (Stickiness) model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Transition matrix (fixed for the 2-step task)
    # Action 0 -> 70% State 1 (idx 0), 30% State 2 (idx 1)
    # Action 1 -> 30% State 1 (idx 0), 70% State 2 (idx 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two actions in the two stage 2 states
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # To track the previous action for perseveration
    last_action_1 = -1 

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Stage 1 Decision
        q_stage1_decision = q_stage1_mf.copy()
        
        # Add perseveration bonus if a previous action exists
        if last_action_1 != -1:
            q_stage1_decision[last_action_1] += perseveration

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # 2. State Transition
        # Determine the second stage state based on transition probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Softmax selection for Stage 2 based on current state s2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Reward Generation
        # Map drifts to the specific state/action pair
        # State 0 (s2=0) uses drift1 (act 0) and drift2 (act 1)
        # State 1 (s2=1) uses drift3 (act 0) and drift4 (act 1)
        if s2 == 0:
            current_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_prob)

        # 5. Learning (Q-value updates)
        
        # TD(0) update for Stage 1:
        # Prediction error uses the Q-value of the chosen stage 2 action/state as the target
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # TD(0) update for Stage 2:
        # Prediction error uses the actual reward received
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update tracking variables
        last_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
