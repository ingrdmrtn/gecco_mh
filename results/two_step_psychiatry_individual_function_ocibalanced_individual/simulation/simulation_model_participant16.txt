Here is the simulation code corresponding to the Dual Learning Rate Model-Free Learner described in `cognitive_model1`.

```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Dual Learning Rate Model-Free Learner.
    
    This model assumes the participant updates their expectations for Stage 1 (spaceships)
    at a different rate than Stage 2 (aliens). It uses a standard TD(1) update logic
    where the Stage 1 value is updated by both the prediction error at the transition 
    and the prediction error at the reward.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_1, alpha_2, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    alpha_1, alpha_2, beta = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Transition matrix (fixed structure)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)       # Q-values for Stage 1 actions (0, 1)
    q_stage2_mf = np.zeros((2, 2))  # Q-values for Stage 2 states (0, 1) x actions (0, 1)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize current trial reward probabilities
        # state 0 corresponds to drifts 1 and 2
        # state 1 corresponds to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate softmax probabilities for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Determine transition to Stage 2 state
        # transition_matrix[a1] gives [prob_state0, prob_state1]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Calculate softmax probabilities for Stage 2 (given state s2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # Observe Reward
        r = int(rng.random() < current_reward_probs[s2][a2])

        # --- Learning Updates ---
        
        # 1. Prediction error at transition (Stage 1 -> Stage 2)
        # Using the value of the chosen stage 2 option as the target (SARSA-like logic implied by fitting code)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha_1 * delta_stage1

        # 2. Prediction error at reward (Stage 2 -> Outcome)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += alpha_2 * delta_stage2

        # 3. Propagate Stage 2 prediction error back to Stage 1 (TD(1) logic)
        q_stage1_mf[a1] += alpha_1 * delta_stage2

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
