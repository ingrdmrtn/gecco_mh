import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using Model-Free Reinforcement Learning with Choice Perseveration.
    
    This model relies on simple TD-learning but includes a 'choice kernel' parameter
    that biases the agent to repeat (or switch) their previous stage 1 action.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stickiness = parameters
    
    # Task structure: Transition probabilities
    # Action 0 -> 70% State 0 (Common), 30% State 1 (Rare)
    # Action 1 -> 30% State 0 (Rare), 70% State 1 (Common)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for Stage 1 actions (0 and 1)
    q_stage1_mf = np.zeros(2) + 0.5
    # q_stage2_mf: Q-values for Stage 2 states (0 and 1) and actions (0 and 1)
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Track the previous action for stickiness
    last_action_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 (index 0) has options with drift1 and drift2
        # State 1 (index 1) has options with drift3 and drift4
        reward_probs = [[drift1[t], drift2[t]], 
                        [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_effective = q_stage1_mf.copy()
        
        # Apply stickiness bonus if there was a previous action
        if last_action_1 != -1:
            q_stage1_effective[last_action_1] += stickiness

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_stage1_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using just the MF values for that state)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (TD Updates) ---
        
        # 1. Update Stage 1 Q-value based on Stage 2 Q-value (TD(0) prediction error)
        # Note: The fitting code uses q_stage2_mf[state, action2] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 Q-value based on reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 3. Update Stage 1 Q-value again based on Stage 2 RPE (Eligibility trace logic, effectively lambda=1)
        q_stage1_mf[a1] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
