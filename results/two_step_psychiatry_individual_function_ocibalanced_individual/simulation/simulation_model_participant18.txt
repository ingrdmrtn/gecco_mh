import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Based model with Choice Stickiness 
    and Reward-Modulated Beta.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, beta_boost, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_base, beta_boost, stickiness = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize model variables
    # Transition matrix (fixed as per task structure)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values for Stage 2 (Model-Free learning occurs here)
    # Shape: (2 states, 2 actions) corresponding to states 1 and 2 (indices 0 and 1)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # state 0 corresponds to drift1/drift2, state 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values for Stage 1
        # Max Q-value from stage 2 for each state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation: Transition matrix * max future value
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Apply Stickiness (repetition bias)
        if last_action_1 != -1:
            q_stage1_mb[last_action_1] += stickiness
            
        # Determine Beta (Inverse Temperature)
        # Beta is boosted if the previous trial was rewarded
        current_beta = beta_base + (beta_boost if last_reward == 1 else 0.0)
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(current_beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 (using only beta_base, as per fitting code logic)
        exp_q2 = np.exp(beta_base * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2_idx][a2])
        
        # --- Learning ---
        
        # Update Stage 2 Q-values (Model-Free update)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Update history for next trial
        last_action_1 = a1
        last_reward = r
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx  # 0 or 1
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
