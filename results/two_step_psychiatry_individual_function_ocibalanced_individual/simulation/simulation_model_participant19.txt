import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model-Based/Model-Free Learner 
    with Choice Perseverance (Stickiness).
    
    This function mirrors the logic of cognitive_model3.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseverance]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, perseverance = parameters
    
    # Fixed transition matrix as defined in the fitting code
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: state (0 or 1), Cols: action (0 or 1)
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    
    # Initialize random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Model-Based values for Stage 1
        # Max Q-value for each second-stage state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Apply Perseverance (Stickiness)
        if last_action_1 != -1:
            q_net[last_action_1] += perseverance
            
        # 4. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 5. Transition to Stage 2
        # State 0 (drift1/2) or State 1 (drift3/4). Note: fitting code uses state as index.
        # Transition probabilities depend on action a1.
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # 6. Stage 2 Choice (Softmax)
        # Note: Fitting code uses same beta for both stages
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 7. Reward Generation
        # Map drift arrays to the current state
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 8. Learning (TD Updates)
        # Note: This model uses a simpler update structure than the example provided in the prompt.
        # It updates stage 1 based on stage 2 value, and stage 2 based on reward.
        # It does NOT include an eligibility trace (lambda) update for stage 1 based on reward.
        
        # Prediction error for stage 1: Difference between Stage 2 Q-value and Stage 1 Q-value
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Prediction error for stage 2: Difference between Reward and Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        q_stage1_mf[a1] += learning_rate * delta_stage1
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
