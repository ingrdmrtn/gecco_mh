import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Learner with Choice Perseverance (Stickiness).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, pers]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, pers = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the two Stage 1 actions (spaceship choice)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the two Stage 2 states, each with 2 actions (aliens)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Model-Based values for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # 2. Combine MB and MF values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # 3. Apply Perseverance Bonus
        q_decision = q_net.copy()
        if last_action_1 != -1:
            q_decision[last_action_1] += pers
            
        # 4. Stage 1 Decision (Softmax)
        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 5. Transition to Stage 2
        # State 0 (common for a1=0, rare for a1=1) or State 1 (rare for a1=0, common for a1=1)
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # 6. Stage 2 Decision (Softmax based on MF only)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 7. Reward Generation
        # Map drifts to the specific state/action pair
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        prob_reward = current_reward_probs[s2][a2]
        r = int(rng.random() < prob_reward)
        
        # 8. Learning / Updates
        # Note: This model uses a simpler update rule than full eligibility traces (lambda=0 implicitly)
        # Update Stage 1 MF Q-value
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] = q_stage2_mf[s2, a2] + learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for next trial's perseverance
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
