import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Reinforcement Learning model 
    with Choice Perseverance (Stickiness).

    This model assumes the participant is purely Model-Free (no planning)
    but has a tendency to repeat (or switch) their previous Stage 1 choice
    regardless of the outcome.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for the two options in Stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two options in each of the two Stage 2 states
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed for the standard 2-step task)
    # Row 0: transitions from Action 0 -> [State 0, State 1]
    # Row 1: transitions from Action 1 -> [State 0, State 1]
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    last_choice_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_biased = q_stage1_mf.copy()
        
        # Apply stickiness bonus if there was a previous choice
        if last_choice_1 != -1:
            q_stage1_biased[last_choice_1] += stickiness

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # --- Stage 2 Decision ---
        # Select action based on simple MF Q-values for that state
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning Updates ---
        # 1. Update Stage 1 Q-value based on Stage 2 Q-value (TD(0) style)
        # Note: The fitting code uses: delta = Q2[s2, a2] - Q1[a1]
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 Q-value based on reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_choice_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
