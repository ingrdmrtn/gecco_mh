import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Reinforcement Learning (TD(0)) model.
    
    This model relies on caching values for actions at both stages. Stage 1 values
    are updated based on the value of the state reached in Stage 2 (TD error), and
    also updated via an eligibility trace from the final reward.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, eligibility]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, eligibility = parameters
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # Value of spaceships (Stage 1)
    q_stage2_mf = np.zeros((2, 2)) # Value of aliens (Stage 2)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Define transition matrix (Standard for two-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Use a random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine the second state based on the transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # s2 is 0 or 1, which maps directly to the row index of q_stage2_mf
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Map drift inputs to the specific state/action pair
        # State 0 (s2=0) has actions leading to drift1 and drift2
        # State 1 (s2=1) has actions leading to drift3 and drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Learning Updates (TD(0) logic from fitting code)
        
        # Update Stage 1 Q-value based on Stage 2 Q-value (TD Prediction Error 1)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 Q-value based on Reward (TD Prediction Error 2)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-value again based on eligibility trace of Stage 2 error
        q_stage1_mf[a1] += learning_rate * eligibility * delta_stage2
        
        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
