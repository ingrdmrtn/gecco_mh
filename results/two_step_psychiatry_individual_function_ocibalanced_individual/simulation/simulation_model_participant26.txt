import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Based Reinforcement Learning model.
    
    This model calculates stage 1 values solely via the transition matrix and 
    learned stage 2 values (no stage 1 model-free learning).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta = parameters

    # Fixed transition matrix as defined in the fitting code
    # [P(State 1 | Action 0), P(State 2 | Action 0)]
    # [P(State 1 | Action 1), P(State 2 | Action 1)]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values for Stage 2 (model-free)
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2)) + 0.5 

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize current trial reward probabilities
        # state 0 corresponds to drifts 1 & 2, state 1 corresponds to drifts 3 & 4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based value for Stage 1
        # V(s') = max_a Q(s', a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Q_MB(s, a) = T(s, a, s') * V(s')
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix probabilities
        # Note: transition_matrix[a1] gives [prob of state 0, prob of state 1]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard Q-learning choice based on stored MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning ---
        # Update Stage 2 Q-values (Standard Q-learning / Delta rule)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Note: No Stage 1 update occurs here because this is a "Pure Model-Based" model
        # where Stage 1 values are computed on the fly, not stored/updated via TD error.

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
