import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model-Based / Model-Free Learner 
    with Choice Perseveration (cognitive_model3).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    rng = np.random.default_rng()
    
    # Unpack parameters
    learning_rate, beta, w, perseveration = parameters
    
    # Fixed transition matrix (common in 2-step tasks)
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for the two spaceships (stage 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two aliens on each of the two planets (stage 2)
    # Shape (2, 2) corresponds to [planet_index, alien_index]
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # 1. Calculate Model-Based values for Stage 1
        # Max Q-value available at each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2

        # 2. Combine MB and MF values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # 3. Apply Perseveration
        # Create a copy for decision making so we don't corrupt the learning Q-values
        q_decision = q_net.copy()
        if last_action_1 != -1:
            q_decision[last_action_1] += perseveration
            
        # 4. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 5. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 6. Stage 2 Choice (Softmax)
        # Note: The fitting code uses the same beta for both stages
        current_planet_qs = q_stage2_mf[s2, :]
        exp_q2 = np.exp(beta * current_planet_qs)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 7. Outcome
        # Get reward probability for the specific planet (s2) and alien (a2)
        # Mapping drifts to states: State 0 uses drift1/drift2, State 1 uses drift3/drift4
        if s2 == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)

        # 8. Learning (TD Updates)
        
        # Stage 1 Update: TD(0)
        # The fitting code updates stage 1 MF Q-value based on the value of the *chosen* stage 2 option
        chosen_stage2_val = q_stage2_mf[s2, a2]
        delta_stage1 = chosen_stage2_val - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 Update: TD(0)
        delta_stage2 = r - chosen_stage2_val
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store history
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
