import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates the TD(lambda) Model.

    This model uses an eligibility trace (lambda) to update Stage 1 values based on 
    Stage 2 prediction errors, bridging the two stages without a distinct Model-Based component.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lam]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, lam = parameters
    
    # Transition matrix: 70% common, 30% rare transition
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1 corresponds to the two options in the first stage (0 or 1)
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two states (rows) and two options (cols) in the second stage
    q_stage2 = np.zeros((2, 2))

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # 2. Transition to Stage 2
        # State 0 (drift1/2) or State 1 (drift3/4) based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Outcome
        # Get current reward probabilities for the specific state arrived at
        current_probs = [drift1[t], drift2[t]] if s2 == 0 else [drift3[t], drift4[t]]
        r = int(rng.random() < current_probs[a2])

        # 5. Learning (TD(lambda) Logic)
        
        # Stage 1 Prediction Error (TD error at transition)
        # Value of state arrived at (Q2) minus value of state left (Q1)
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]

        # Stage 2 Prediction Error (TD error at reward)
        # Reward received minus value of state/action chosen
        delta_stage2 = r - q_stage2[s2, a2]

        # Update Stage 2 Q-values
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values
        # The update includes the immediate error (delta1) plus the propagated error (lambda * delta2)
        q_stage1[a1] += learning_rate * (delta_stage1 + lam * delta_stage2)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
