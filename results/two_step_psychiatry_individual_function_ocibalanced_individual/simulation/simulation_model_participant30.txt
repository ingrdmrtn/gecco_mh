import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free TD(lambda) Learner.
    
    This model simulates a pure model-free agent that uses eligibility traces to 
    connect Stage 1 choices directly to final rewards, bypassing the transition 
    structure of the task.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lambda_trace]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, lambda_trace = parameters
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the two choices in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the two states in stage 2, each having 2 choices
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed for this task structure, usually 70/30)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine the second state based on the transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Select action based on Q-values for the specific state arrived at (s2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get the reward probability for the specific terminal state (state s2, action a2)
        # Mapping: s2=0 -> (drift1, drift2), s2=1 -> (drift3, drift4)
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        prob_reward = current_drifts[s2][a2]
        r = int(rng.random() < prob_reward)
        
        # 5. Learning / Updates
        
        # TD Error 1: Difference between Stage 2 Q-value and Stage 1 Q-value
        # Note: The fitting code uses the Q-value of the CHOSEN action in stage 2 as the target
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 Q-value based on immediate transition (TD(0) part)
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Error 2: Difference between Reward and Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # Update Stage 2 Q-value
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update: Update Stage 1 based on Stage 2 prediction error
        # This allows the final reward to propagate back to the first choice
        q_stage1_mf[a1] += learning_rate * lambda_trace * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
