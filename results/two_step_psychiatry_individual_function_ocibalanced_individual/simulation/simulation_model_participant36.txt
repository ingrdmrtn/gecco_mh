import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Learner with Eligibility Traces and Choice Stickiness.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, lam, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.
        
    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1, corresponding to indices 0 and 1 in the transition matrix).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters based on cognitive_model3 definition
    learning_rate, beta, w, lam, stickiness = parameters
    
    # Setup storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the two options in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the two states (rows) and two options (cols) in stage 2
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Random number generator
    rng = np.random.default_rng()
    
    # Initialize previous action for stickiness (not applied on first trial)
    prev_a1 = -1

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 has options with drift1 and drift2
        # State 1 has options with drift3 and drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness if not the first trial
        if t > 0:
            q_net[prev_a1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (using only MF values for that state)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning / Updates ---
        
        # SARSA-like update for Stage 1 (TD(0) part)
        # Note: cognitive_model3 uses q_stage2_mf[state_idx, action_2] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Eligibility trace update for Stage 1 (TD(lambda) part)
        q_stage1_mf[a1] += learning_rate * lam * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous action for next trial's stickiness
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
