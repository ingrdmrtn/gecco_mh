import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Decay-Based Model-Free Learner.
    Unchosen action values decay toward 0 at each step, simulating forgetting.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, decay_rate, beta, lambda_trace]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, decay_rate, beta, lambda_trace = parameters
    
    # Initialize Q-values
    # Stage 1 has 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2 has 2 states (0 and 1), each with 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed for the 2-step task)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Create a random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for easier access: reward_probs[state][action]
        # State 0 corresponds to drifts 1 and 2
        # State 1 corresponds to drifts 3 and 4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on action 1 and transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning and Decay ---
        
        # 1. Decay unchosen actions
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] = (1 - decay_rate) * q_stage1_mf[unchosen_a1]

        unchosen_a2 = 1 - a2
        q_stage2_mf[s2, unchosen_a2] = (1 - decay_rate) * q_stage2_mf[s2, unchosen_a2]

        # 2. TD(0) update for Stage 1 (Prediction Error based on Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 3. TD(0) update for Stage 2 (Prediction Error based on Reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 4. Eligibility Trace update for Stage 1 (TD(lambda))
        q_stage1_mf[a1] += learning_rate * lambda_trace * delta_stage2

        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
