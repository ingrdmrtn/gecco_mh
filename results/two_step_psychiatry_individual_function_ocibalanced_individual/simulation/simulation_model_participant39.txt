import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Based RL with Perseveration model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Fixed transition matrix as defined in the fitting code
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values for Stage 2 (Model-Free)
    # Shape (2, 2): 2 states (State 1, State 2), 2 actions per state
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # Organize current trial reward probabilities
        # state 0 corresponds to drift1/drift2, state 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], 
                        [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1
        # V(s') = max_a Q(s', a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Q_MB(s, a) = T(s, a, s') * V(s')
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Calculate logits
        logits = beta * q_stage1_mb

        # Add perseveration bonus if it's not the first trial
        if last_action_1 != -1:
            logits[last_action_1] += perseveration
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine next state based on transition matrix
        # s2 is the index of the second stage state (0 or 1)
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 using Model-Free Q-values
        # Note: The fitting code uses the same beta for both stages
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning ---
        # Update Stage 2 Q-values (Model-Free update)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update history for next trial
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
