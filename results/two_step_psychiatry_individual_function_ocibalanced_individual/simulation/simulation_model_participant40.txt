import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Free Learner with Choice Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, perseveration = parameters
    
    # Standard 2-step task transition matrix
    # Action 0 -> 70% State 0 (index 1), 30% State 1 (index 2)
    # Action 1 -> 30% State 0 (index 1), 70% State 1 (index 2)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1

    for t in range(n_trials):
        # 1. Stage 1 Choice
        logits = beta * q_stage1_mf.copy()
        
        # Apply perseveration bonus if a previous action exists
        if last_action_1 != -1:
            logits[last_action_1] += perseveration
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Calculate probabilities based on simple softmax of Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Get current reward probabilities for the specific state
        # State 0 corresponds to drifts 1 and 2, State 1 corresponds to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Learning Updates (Model-Free SARSA-like updates)
        
        # Update Stage 1 Q-value based on Stage 2 Q-value (TD-error 1)
        # Note: In the fitting code, this update happens *after* stage 2 choice is known
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 Q-value based on Reward (TD-error 2)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
