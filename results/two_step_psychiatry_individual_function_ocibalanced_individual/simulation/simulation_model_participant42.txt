import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Learner with Value Decay.

    This model assumes no knowledge of the transition structure (pure MF).
    However, it includes a decay parameter. On every trial, the Q-values of 
    all actions decay towards 0 before the update for the chosen action is applied.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, decay]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, decay = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    # Transition matrix (fixed for the standard 2-step task)
    # Action 0 -> 70% State 0 (index 0), 30% State 1 (index 1)
    # Action 1 -> 30% State 0 (index 0), 70% State 1 (index 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows), 2 actions (cols)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current reward probabilities for the second stage
        # State 0 corresponds to drift1 and drift2
        # State 1 corresponds to drift3 and drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)

        # --- Stage 2 Choice ---
        # Select action based on Q-values for the specific state arrived at
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning and Decay ---
        
        # 1. Apply Decay
        # In the fitting code: q_stage1_mf *= decay, q_stage2_mf *= decay
        # This happens before the specific update for the chosen action in the loop logic
        # (or effectively, all values decay, then the chosen one is updated on top of the decayed value, 
        # but standard RL usually updates based on the value *at decision time*. 
        # Looking closely at the fitting code:
        #   p_choice calculated -> decay applied -> update applied.
        # This means the update `q += lr * (r - q)` uses the *decayed* q if we aren't careful, 
        # but the fitting code writes `q_stage1_mf[action] += ...` after `q_stage1_mf *= decay`.
        # This implies the prediction error is calculated against the decayed value.
        
        q_stage1_mf *= decay
        q_stage2_mf *= decay

        # 2. Update Stage 1 Q-value
        # Note: The fitting code uses a simple MF update based on the final reward, 
        # effectively treating it like a bandit or ignoring the temporal difference structure 
        # between stage 1 and stage 2 (no eligibility trace parameter or TD(0) logic shown, 
        # just direct reward update).
        q_stage1_mf[a1] += learning_rate * (r - q_stage1_mf[a1])

        # 3. Update Stage 2 Q-value
        q_stage2_mf[s2, a2] += learning_rate * (r - q_stage2_mf[s2, a2])

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
