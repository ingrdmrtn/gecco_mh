```python
import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Model with Direct Reinforcement (TD(1)).
    
    This model ignores transition structure and updates Stage 1 values directly based on the final
    reward received. It includes a decay parameter for unchosen Stage 1 options.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, decay]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, decay = parameters
    
    # Transition matrix (fixed for the task, though this model ignores it for learning)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Values for the two Stage 1 actions (spaceships)
    q_stage1_mf = np.zeros(2)     
    # q_stage2_mf: Values for the four Stage 2 actions (2 states * 2 aliens)
    # Rows are states (0 or 1), columns are actions (0 or 1)
    q_stage2_mf = np.zeros((2, 2)) 

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Setup Reward Probabilities for this trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # 2. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # 3. Transition to Stage 2
        # Determine state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice (Softmax)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 5. Outcome
        r = int(rng.random() < reward_probs[s2][a2])

        # 6. Learning Updates
        
        # Update Stage 2 Q-values (Standard Rescorla-Wagner)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (Direct Reinforcement / TD(1))
        # The update is driven by the final reward 'r', not the value of the next state.
        delta_stage1_direct = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1_direct

        # Decay for unchosen Stage 1 option
        unchosen_1 = 1 - a1
        q_stage1_mf[unchosen_1] *= (1 - decay)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
