Here are three cognitive models based on the two-step task structure and the participant's behavior.

### Cognitive Model 1: Hybrid Model-Based / Model-Free Agent
This model assumes the participant uses a combination of two strategies: a **Model-Based (MB)** strategy that plans using the transition probabilities (understanding that Spaceship A leads to Planet X), and a **Model-Free (MF)** strategy that learns purely from reward history (Spaceship A was good last time). A mixing parameter `w` determines the balance between these strategies.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based and Model-Free Reinforcement Learning agent.
    
    This model calculates values for the first stage using a weighted average
    of Model-Based (planning via transition matrix) and Model-Free (TD learning)
    estimates.
    
    Parameters:
    - learning_rate: [0,1] Rate at which Q-values are updated.
    - beta_1: [0,10] Inverse temperature for Stage 1 softmax (exploration/exploitation).
    - beta_2: [0,10] Inverse temperature for Stage 2 softmax.
    - w: [0,1] Weighting parameter. 1 = Pure Model-Based, 0 = Pure Model-Free.
    """
    learning_rate, beta_1, beta_2, w = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix: Row 0 is Spaceship 0 (A), Row 1 is Spaceship 1 (U)
    # Col 0 is Planet 0 (X), Col 1 is Planet 1 (Y). 
    # Common transitions: 0->0 and 1->1 (0.7 prob).
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Model-Free values
    q_stage1_mf = np.zeros(2)      # Values for Spaceships
    q_stage2_mf = np.zeros((2, 2)) # Values for Aliens (Rows: Planet, Cols: Alien)

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            continue

        # --- STAGE 1 POLICY ---
        # 1. Calculate Model-Based values: V(S1) = T * max(Q(S2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Softmax choice probability for Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # Current Planet

        # --- STAGE 2 POLICY ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATES ---
        # Prediction Error 1: Difference between value of state reached (max Q2) and chosen spaceship
        # Note: Using max_q_stage2[state_idx] is a proxy for TD(0) with state value
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Prediction Error 2: Reward received vs expected value of alien
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    # Sum log likelihoods, ignoring trials where probability is 0 (missing data)
    log_loss = -(np.sum(np.log(p_choice_1[p_choice_1 > 0] + eps)) + 
                 np.sum(np.log(p_choice_2[p_choice_2 > 0] + eps)))
    return log_loss
```

### Cognitive Model 2: Model-Free Agent with Choice Perseveration
The participant data shows streaks of repeating the same spaceship choice (e.g., trials 18-24). This model adds a **stickiness (perseveration)** parameter. It assumes the participant is primarily Model-Free but has a bias to repeat their previous Stage 1 choice, regardless of the reward outcome.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Choice Perseveration (Stickiness).
    
    Adds a 'stickiness' bonus to the logits of the previously chosen action
    in Stage 1, capturing the tendency to repeat choices.
    
    Parameters:
    - learning_rate: [0,1]
    - beta: [0,10] Inverse temperature (used for both stages).
    - stickiness: [0,5] Bonus added to the value of the previously chosen spaceship.
    """
    learning_rate, beta, stickiness = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- STAGE 1 POLICY ---
        # Add stickiness bonus to the previously chosen action
        logits_1 = beta * q_stage1_mf.copy()
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATES ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[p_choice_1 > 0] + eps)) + 
                 np.sum(np.log(p_choice_2[p_choice_2 > 0] + eps)))
    return log_loss
```

### Cognitive Model 3: Asymmetric Learning Rate Agent
This model hypothesizes that the participant learns differently from positive outcomes (finding gold) versus negative outcomes (no gold). It implements **separate learning rates** for positive and negative prediction errors. This captures potential risk-seeking or risk-averse behavior in the learning process.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Agent with Asymmetric Learning Rates.
    
    Uses different learning rates depending on whether the prediction error
    (delta) is positive (better than expected) or negative (worse than expected).
    
    Parameters:
    - alpha_pos: [0,1] Learning rate for positive prediction errors.
    - alpha_neg: [0,1] Learning rate for negative prediction errors.
    - beta: [0,10] Inverse temperature.
    """
    alpha_pos, alpha_neg, beta = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- STAGE 1 POLICY ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- UPDATES ---
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        # Select learning rate based on sign of delta
        lr_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += lr_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        # Select learning rate based on sign of delta
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[p_choice_1 > 0] + eps)) + 
                 np.sum(np.log(p_choice_2[p_choice_2 > 0] + eps)))
    return log_loss
```
