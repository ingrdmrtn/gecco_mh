Here are three cognitive models explaining the participant's behavior in the two-step task.

### Model 1: Hybrid Model-Based / Model-Free with Eligibility Traces and Perseveration
This model (based on Daw et al., 2011) assumes the participant uses a combination of Model-Based (planning using the transition matrix) and Model-Free (learning from reward history) strategies. It includes an eligibility trace parameter (`lambda`) to modulate how much the Stage 2 outcome updates Stage 1 values directly, and a perseveration parameter to account for the participant's tendency to repeat choices (stickiness).

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free agent with eligibility traces and perseveration.
    
    Parameters:
    learning_rate: rate of value updating [0,1]
    beta: inverse temperature for softmax (exploration/exploitation) [0,10]
    w: weight of model-based system (0=pure MF, 1=pure MB) [0,1]
    lambda_eligibility: eligibility trace decay (0=TD(0), 1=TD(1)) [0,1]
    perseveration: bonus added to repeating the previous choice [0,5]
    """
    learning_rate, beta, w, lambda_eligibility, perseveration = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: A(0)->X(0) is common (0.7), U(1)->Y(1) is common (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # Stage 1: Spaceships (0, 1)
    q_stage1_mf = np.zeros(2) # Model-Free values
    # Stage 2: Aliens (State 0: Alien 0,1; State 1: Alien 0,1)
    q_stage2_mf = np.zeros((2, 2)) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Choice ---
        # Model-Based Value: Transition * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Perseveration (Stickiness)
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseveration
            
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        
        # --- Stage 2 Choice ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning Updates ---
        # Prediction Errors
        # Stage 1 RPE: Difference between Q_stage2 chosen and Q_stage1 chosen
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Stage 2 RPE: Difference between Reward and Q_stage2 chosen
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1 MF (TD(lambda) update)
        # Q1 += lr * delta1 + lr * lambda * delta2
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1 + learning_rate * lambda_eligibility * delta_stage2
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Asymmetric Learning Rate Hybrid Model
This model hypothesizes that the participant learns differently from positive outcomes (finding gold) versus negative outcomes (no gold). This "valence-dependent" learning is common in clinical populations. It maintains the Hybrid structure but splits the learning rate into `lr_pos` (for wins) and `lr_neg` (for losses). It assumes `lambda=1` (full eligibility trace) for the Model-Free component to capture the strong Win-Stay behavior observed.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid model with asymmetric learning rates for wins and losses.
    
    Parameters:
    lr_pos: learning rate when reward is received (Win) [0,1]
    lr_neg: learning rate when no reward is received (Loss) [0,1]
    beta: inverse temperature [0,10]
    w: weight of model-based system [0,1]
    perseveration: stickiness to previous choice [0,5]
    """
    lr_pos, lr_neg, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Determine learning rate based on outcome (Win vs Loss)
        current_lr = lr_pos if reward[trial] == 1 else lr_neg
        
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1 (Assume TD(1) / lambda=1 logic for strong reinforcement)
        # We propagate the full reward prediction error back
        q_stage1_mf[action_1[trial]] += current_lr * (delta_stage1 + delta_stage2)
        
        # Update Stage 2
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Pure Model-Free TD($\lambda$) with Perseveration
This model tests the hypothesis that the participant ignores the transition probabilities entirely (no Model-Based planning). It relies solely on temporal difference learning with eligibility traces. If this model fits best, it suggests the participant relies on direct reinforcement (e.g., "I chose Spaceship A and got gold, so I'll choose A again") regardless of whether the transition was rare or common.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Q-Learning (TD-Lambda) with perseveration.
    Ignores transition structure (w=0).
    
    Parameters:
    learning_rate: rate of value updating [0,1]
    beta: inverse temperature [0,10]
    lambda_eligibility: eligibility trace decay [0,1]
    perseveration: stickiness to previous choice [0,5]
    """
    learning_rate, beta, lambda_eligibility, perseveration = model_parameters
    n_trials = len(action_1)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy (Pure Model-Free) ---
        q_net_1 = q_stage1_mf.copy()
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # TD(Lambda) update for Stage 1
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1 + learning_rate * lambda_eligibility * delta_stage2
        
        # TD update for Stage 2
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
