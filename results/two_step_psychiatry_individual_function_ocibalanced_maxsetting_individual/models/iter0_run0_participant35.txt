Here are three cognitive models analyzing the participant's behavior in the two-step task.

### Model 1: Hybrid MB/MF with Eligibility Traces
This model represents the standard hypothesis for this task: participants use a combination of Model-Based (planning using the transition matrix) and Model-Free (habitual, TD-learning) strategies. It includes an eligibility trace ($\lambda$) to allow the stage 2 reward to directly update stage 1 values.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free reinforcement learning model with eligibility traces.
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta_1: [0, 10] Inverse temperature for stage 1 choice (exploration/exploitation).
    - beta_2: [0, 10] Inverse temperature for stage 2 choice.
    - w: [0, 1] Weighting parameter (0 = pure Model-Free, 1 = pure Model-Based).
    - lam: [0, 1] Eligibility trace parameter (lambda) connecting stage 2 reward to stage 1 choice.
    """
    learning_rate, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1 (actions 0, 1)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (states 0, 1; actions 0, 1)
    
    for trial in range(n_trials):
        # Handle missing/invalid data (e.g., trial 191)
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Policy for the first choice ---
        # Model-Based Value: Expected max value of next stage weighted by transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Net Value: Weighted sum of MB and MF values
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax selection
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Policy for the second choice ---
        state_idx = state[trial]
        q_current_stage2 = q_stage2_mf[state_idx]
        
        exp_q2 = np.exp(beta_2 * q_current_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Updates ---
        # Stage 1 Prediction Error: Difference between value of state reached and stage 1 value
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Stage 2 Prediction Error: Difference between reward received and stage 2 value
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1 MF value: Uses delta1 (TD(0)) and delta2 scaled by lambda (TD(lambda))
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1 + learning_rate * lam * delta_stage2
        
        # Update Stage 2 MF value
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free with Perseverance
This model assumes the participant relies solely on habitual (Model-Free) learning but exhibits "perseverance" (a tendency to repeat the previous choice regardless of reward). This tests if the behavior is better explained by simple habit + repetition rather than planning.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free reinforcement learning model with Perseverance.
    Assumes no usage of the transition matrix (w=0).
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - lam: [0, 1] Eligibility trace parameter.
    - pers: [0, 5] Perseverance bonus added to the value of the previously chosen action.
    """
    learning_rate, beta_1, beta_2, lam, pers = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Policy for the first choice ---
        # Calculate perseverance bonus (add to Q-value of last chosen action)
        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = pers
            
        # Net value is purely Model-Free + Perseverance
        q_net_1 = q_stage1_mf + pers_bonus
        
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        
        # --- Policy for the second choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1 + learning_rate * lam * delta_stage2
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Based with Perseverance
This model assumes the participant is purely a planner (Model-Based) who understands the transition structure perfectly, but also exhibits perseverance. It ignores the direct reinforcement of stage 1 choices (Model-Free caching), relying only on the computed value from stage 2.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Based reinforcement learning model with Perseverance.
    Uses transition structure to compute stage 1 values from stage 2 values.
    Does not maintain separate Model-Free values for stage 1 (w=1).
    
    Parameters:
    - learning_rate: [0, 1] Rate for updating stage 2 values.
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - pers: [0, 5] Perseverance bonus.
    """
    learning_rate, beta_1, beta_2, pers = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage2_mf = np.zeros((2, 2)) # Stage 2 values must still be learned
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Policy for the first choice ---
        # Pure Model-Based calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add perseverance
        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = pers
            
        q_net_1 = q_stage1_mb + pers_bonus
        
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        
        # --- Policy for the second choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Updates ---
        # Only update Stage 2 values; Stage 1 values are derived dynamically
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
