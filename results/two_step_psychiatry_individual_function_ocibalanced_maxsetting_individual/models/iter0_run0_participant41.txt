Here are three cognitive models based on the two-step task paradigm and the specific behavior observed in the participant data.

### Model 1: Hybrid Model-Based/Model-Free Reinforcement Learning
This is the standard baseline model for this task. It assumes the participant's choices are a weighted mixture of a Model-Based (MB) strategy (planning based on the transition matrix) and a Model-Free (MF) strategy (habitual caching of values). Given the participant data shows some sensitivity to transitions but strong habitual streaks, this model tests the balance between the two systems.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Reinforcement Learning.
    
    This model assumes the agent computes values using both a Model-Based (planning)
    system and a Model-Free (habit) system. The final choice is a weighted combination.
    
    Parameters:
    learning_rate: [0,1] - Rate at which Q-values are updated.
    beta_1: [0,10] - Inverse temperature for Stage 1 choice (exploration/exploitation).
    beta_2: [0,10] - Inverse temperature for Stage 2 choice.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta_1, beta_2, w = model_parameters
    n_trials = len(action_1)
  
    # Transition matrix: T[0,0] is P(State 0 | Action 0), etc.
    # A (0) -> X (0) is 0.7; U (1) -> Y (1) is 0.7
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)       # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2))  # Values for stage 2 (State x Action)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # 1. Calculate Model-Based values
        # V_MB(S1) = T(S1, S2) * max_a Q(S2, a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Softmax choice probability
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Current state (planet) reached
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Updates ---
        # Prediction Error 1 (MF): Value of chosen stage 2 state vs stage 1 value
        # Note: We use the value of the chosen action in stage 2 (SARSA-like logic for stage 1)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Prediction Error 2: Reward vs stage 2 value
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free with Eligibility Traces (TD-Lambda)
The participant data shows behavior where rewards obtained after "rare" transitions (e.g., Ship A -> Planet Y -> Reward) often lead to repeating Ship A. This is characteristic of Model-Free learning, where the transition structure is ignored. This model implements TD($\lambda$), allowing the reward at Stage 2 to directly reinforce the Stage 1 choice via an eligibility trace, skipping the Model-Based planning component entirely.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free Q-Learning with Eligibility Traces (TD-Lambda).
    
    This model ignores the transition structure (0% Model-Based). 
    It uses an eligibility trace (lambda) to allow the reward at the second stage 
    to directly update the value of the first stage choice, capturing 
    temporal credit assignment without a world model.
    
    Parameters:
    learning_rate: [0,1] - Rate at which Q-values are updated.
    beta_1: [0,10] - Inverse temperature for Stage 1 choice.
    beta_2: [0,10] - Inverse temperature for Stage 2 choice.
    lambda_param: [0,1] - Eligibility trace decay (0 = TD(0), 1 = Monte Carlo).
    """
    learning_rate, beta_1, beta_2, lambda_param = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy (Pure MF) ---
        # No Model-Based component here, purely based on cached Q-values
        exp_q1 = np.exp(beta_1 * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Value Updates ---
        
        # Stage 1 Prediction Error (TD error at step 1)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        # Standard TD update for Stage 1
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Prediction Error (TD error at step 2)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 2 value
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility Trace Update:
        # The Stage 2 RPE (delta_stage2) also updates Stage 1 value, scaled by lambda
        q_stage1_mf[action_1[trial]] += learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Choice Perseveration
The participant displays significant "stickiness" (e.g., trials 3-15 choosing Ship 0 repeatedly, trials 24-28 choosing Ship 1). Even when rewards are intermittent, the participant tends to repeat the previous Stage 1 action. This model adds a "perseveration" parameter to the Hybrid model, which increases the log-odds of repeating the previous action, independent of value learning.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF Model with Choice Perseveration (Stickiness).
    
    This model extends the Hybrid model by adding a perseveration bonus.
    If the agent chose action A on trial t-1, the probability of choosing A 
    on trial t is boosted, capturing the 'streaks' seen in the data.
    
    Parameters:
    learning_rate: [0,1] - Rate at which Q-values are updated.
    beta_1: [0,10] - Inverse temperature for Stage 1 choice.
    beta_2: [0,10] - Inverse temperature for Stage 2 choice.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    perseveration: [0,5] - "Stickiness" bonus added to the logits of the repeated action.
    """
    learning_rate, beta_1, beta_2, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous action for perseveration (initialize with -1 or valid index)
    last_action = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net value mixture
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits (Value * Beta)
        logits = beta_1 * q_net
        
        # Add Perseveration Bonus
        if trial > 0 and last_action != -1:
            logits[last_action] += perseveration
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store action for next trial's perseveration
        last_action = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
