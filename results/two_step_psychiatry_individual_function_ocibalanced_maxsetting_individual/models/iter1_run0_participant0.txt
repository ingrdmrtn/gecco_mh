Here are three new cognitive models for the two-step task, designed to capture the participant's behavior, particularly the strong tendency for perseveration (stickiness) and potential model-free learning strategies.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Asymmetric Learning Rates and Perseveration.
    
    This model assumes a Model-Free strategy but distinguishes between learning 
    from positive prediction errors (gains) and negative prediction errors (losses).
    It also includes a perseveration parameter to account for the participant's 
    tendency to repeat Stage 1 choices.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors (RPE > 0).
    - alpha_neg: [0, 1] Learning rate for negative prediction errors (RPE < 0).
    - beta_1: [0, 10] Inverse temperature (softness) for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature (softness) for Stage 2 choice.
    - perseveration: [0, 10] Bonus added to the logit of the previously chosen action.
    """
    alpha_pos, alpha_neg, beta_1, beta_2, perseveration = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2)) # state x action
    
    log_likelihood = 0.0
    eps = 1e-10
    
    last_action_1 = -1
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])
        
        # --- Stage 1 Policy ---
        logits_1 = beta_1 * q_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Value Updates ---
        # Calculate Prediction Errors
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        delta_2 = r - q_stage2[s2, a2]
        
        # Combine errors for Stage 1 (TD(1)-like update for strong reward link)
        total_delta_1 = delta_1 + delta_2
        
        # Asymmetric Update for Stage 1
        if total_delta_1 > 0:
            q_stage1[a1] += alpha_pos * total_delta_1
        else:
            q_stage1[a1] += alpha_neg * total_delta_1
            
        # Asymmetric Update for Stage 2
        if delta_2 > 0:
            q_stage2[s2, a2] += alpha_pos * delta_2
        else:
            q_stage2[s2, a2] += alpha_neg * delta_2
            
        last_action_1 = a1
        
    return -log_likelihood

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Control with Perseveration.
    
    Combines a Model-Based value (derived from the known transition matrix) 
    with a Model-Free value. The contributions are weighted by w_mb.
    Includes a perseveration bonus, which was missing in previous hybrid attempts.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w_mb: [0, 1] Weight of Model-Based component (0 = Pure MF, 1 = Pure MB).
    - perseveration: [0, 10] Stickiness bonus added to previously chosen action.
    """
    learning_rate, beta_1, beta_2, w_mb, perseveration = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: Row=Action, Col=State.
    # Action 0 -> State 0 (0.7), State 1 (0.3)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])
        
        # --- Stage 1 Policy ---
        # Model-Based Value: Expected max Q-value of next stage
        max_q2 = np.max(q_stage2, axis=1) # Max value for each state
        q_stage1_mb = transition_matrix @ max_q2
        
        # Integrated Value
        q_net = w_mb * q_stage1_mb + (1 - w_mb) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Value Updates ---
        # MF Update (TD(1) style for Stage 1 to capture direct reinforcement)
        delta_1 = q_stage2[s2, a2] - q_stage1_mf[a1]
        delta_2 = r - q_stage2[s2, a2]
        
        # Update Stage 1 MF
        q_stage1_mf[a1] += learning_rate * (delta_1 + delta_2)
        
        # Update Stage 2
        q_stage2[s2, a2] += learning_rate * delta_2
        
        last_action_1 = a1

    return -log_likelihood

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Choice Kernel.
    
    Uses a 'Choice Kernel' to track the history of chosen actions, creating a 
    dynamic 'habit' effect that decays over time. This differs from simple 
    perseveration by integrating history rather than just the last choice.
    
    Parameters:
    - learning_rate: [0, 1] Q-value update rate.
    - beta_1: [0, 10] Inverse temperature for Q-values in Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - ck_lr: [0, 1] Learning rate for the choice kernel (decay/update rate).
    - ck_weight: [0, 10] Weight of the choice kernel in the decision policy.
    """
    learning_rate, beta_1, beta_2, ck_lr, ck_weight = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    choice_kernel = np.zeros(2)
    
    log_likelihood = 0.0
    eps = 1e-10
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])
        
        # --- Stage 1 Policy ---
        # Policy mixes Q-values and Choice Kernel
        logits_1 = beta_1 * q_stage1 + ck_weight * choice_kernel
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Value Updates ---
        # Q-Learning Update
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        delta_2 = r - q_stage2[s2, a2]
        
        q_stage1[a1] += learning_rate * (delta_1 + delta_2)
        q_stage2[s2, a2] += learning_rate * delta_2
        
        # Choice Kernel Update
        # CK[a] = CK[a] + alpha * (1 - CK[a])
        # CK[other] = CK[other] * (1 - alpha)
        choice_kernel[a1] += ck_lr * (1 - choice_kernel[a1])
        choice_kernel[1 - a1] *= (1 - ck_lr)
        
    return -log_likelihood
```
