Here are 3 new cognitive models based on the participant data and the two-step task structure.

### Model 1: Hybrid MB/MF with Choice Stickiness
**Rationale:** The participant data shows significant "streaks" (e.g., Trials 190-200) where they repeat the same Stage 1 choice despite outcomes. This suggests a "stickiness" or perseverance bias, where the simple act of choosing an option increases the probability of choosing it again, independent of reward history. This model adds a `stickiness` parameter to the standard Hybrid model.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free model with Choice Stickiness.
    
    Adds a 'stickiness' parameter to the standard hybrid model. This captures
    behavioral inertia (repeating the previous choice) which is often observed
    in two-step tasks alongside reinforcement learning.

    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature (exploration/exploitation trade-off).
    - w: [0, 1] Weight of Model-Based system (0=MF, 1=MB).
    - lam: [0, 1] Eligibility trace for Stage 1 MF update.
    - stickiness: [0, 5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, w, lam, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: A->X (0->0) is 0.7, A->Y (0->1) is 0.3, etc.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize values
    q_mf = np.zeros(2)         # Stage 1 Model-Free values
    q2 = np.zeros((2, 2))      # Stage 2 values (State x Alien)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1 # Track previous choice for stickiness
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # Handle missing data
        if a1 == -1 or s == -1 or a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            last_action_1 = -1
            continue

        # --- STAGE 1 CHOICE ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        
        # Net Value: Weighted sum of MB and MF
        q_net = w * q_mb + (1 - w) * q_mf
        
        # Add Stickiness to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 CHOICE ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        # Stage 1 PE (Model-Free)
        delta_1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta_1
        
        # Stage 2 PE
        delta_2 = r - q2[s, a2]
        q2[s, a2] += learning_rate * delta_2
        
        # Eligibility Trace: Update Stage 1 MF value based on Stage 2 outcome
        q_mf[a1] += learning_rate * lam * delta_2
        
        # Update history
        last_action_1 = a1

    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with Asymmetric Learning Rates
**Rationale:** Participants often update beliefs differently following rewards (positive prediction errors) versus omissions (negative prediction errors). This model splits the learning rate into `lr_pos` and `lr_neg`. This allows the model to capture behavior where the participant might be quick to learn good options but slow to discard them when they fail (or vice-versa), which is relevant given the slow drift of reward probabilities in the task.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Asymmetric Learning Rates.
    
    Distinguishes between learning from positive prediction errors (better than expected)
    and negative prediction errors (worse than expected).
    
    Parameters:
    - lr_pos: [0, 1] Learning rate when prediction error is positive.
    - lr_neg: [0, 1] Learning rate when prediction error is negative.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - lam: [0, 1] Eligibility trace.
    """
    lr_pos, lr_neg, beta, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf = np.zeros(2)
    q2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        if a1 == -1 or s == -1 or a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- STAGE 1 CHOICE ---
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        q_net = w * q_mb + (1 - w) * q_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 CHOICE ---
        exp_q2 = np.exp(beta * q2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        # Stage 1 Update
        delta_1 = q2[s, a2] - q_mf[a1]
        eff_lr_1 = lr_pos if delta_1 > 0 else lr_neg
        q_mf[a1] += eff_lr_1 * delta_1
        
        # Stage 2 Update
        delta_2 = r - q2[s, a2]
        eff_lr_2 = lr_pos if delta_2 > 0 else lr_neg
        q2[s, a2] += eff_lr_2 * delta_2
        
        # Eligibility Trace Update (uses the Stage 2 effective LR)
        q_mf[a1] += eff_lr_2 * lam * delta_2

    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with Separate Stage Temperatures
**Rationale:** The decision dynamics in Stage 1 (choosing a spaceship/strategy) and Stage 2 (choosing an alien/bandit) are different. Stage 1 involves planning and uncertainty about transitions, while Stage 2 is a direct bandit task. This model uses two separate beta parameters (`beta_1` and `beta_2`) to allow for different levels of exploration/randomness in the two stages.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Separate Inverse Temperatures for each stage.
    
    Allows the model to capture different levels of exploration/exploitation
    fidelity in the spaceship choice (Stage 1) versus the alien choice (Stage 2).
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta_1: [0, 10] Inverse temperature for Stage 1 (Spaceship choice).
    - beta_2: [0, 10] Inverse temperature for Stage 2 (Alien choice).
    - w: [0, 1] Weight of Model-Based system.
    - lam: [0, 1] Eligibility trace.
    """
    learning_rate, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf = np.zeros(2)
    q2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        if a1 == -1 or s == -1 or a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- STAGE 1 CHOICE (Uses beta_1) ---
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        q_net = w * q_mb + (1 - w) * q_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 CHOICE (Uses beta_2) ---
        exp_q2 = np.exp(beta_2 * q2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        delta_1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta_1
        
        delta_2 = r - q2[s, a2]
        q2[s, a2] += learning_rate * delta_2
        
        q_mf[a1] += learning_rate * lam * delta_2

    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
