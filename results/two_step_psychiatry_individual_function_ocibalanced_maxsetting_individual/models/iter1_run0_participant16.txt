Here are three cognitive models based on the participant data and the two-step task structure.

### Model 1: Hybrid Model-Based / Model-Free RL
This model combines Model-Based (MB) planning with Model-Free (MF) learning. The participant likely uses knowledge of the transition structure (70%/30%) to plan decisions (MB), while also learning from direct reward experience (MF). A mixing weight parameter `w` controls the balance between these two systems, and an eligibility trace `lam` allows stage 2 outcomes to reinforce stage 1 choices.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL with Eligibility Trace.
    
    Parameters:
    - learning_rate: [0, 1] Weight for prediction error updates.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Mixing weight (0 = Pure MF, 1 = Pure MB).
    - lam: [0, 1] Eligibility trace (credit assignment to Stage 1).
    """
    learning_rate, beta, w, lam = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        # Calculate Model-Based values using the transition matrix and max Stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Action value updating
        # Stage 1 TD error (SARSA-style for the transition)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Prediction Error (Reward outcome)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace: Update Stage 1 based on Stage 2 outcome
        q_stage1_mf[action_1[trial]] += learning_rate * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free RL with Choice Perseveration (Stickiness)
The participant data shows "runs" of repeating the same spaceship (e.g., trials 1-22, 23-32). This suggests a behavioral tendency to repeat the previous choice regardless of the reward value, known as "perseveration" or "stickiness." This model adds a `stickiness` parameter to the softmax function for the Stage 1 decision, biasing the agent to repeat the last action.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Choice Perseveration (Stickiness).
    
    Parameters:
    - learning_rate: [0, 1] Weight for prediction error updates.
    - beta: [0, 10] Inverse temperature.
    - lam: [0, 1] Eligibility trace.
    - stickiness: [0, 5] Bonus added to the log-odds of repeating the previous choice.
    """
    learning_rate, beta, lam, stickiness = model_parameters
    n_trials = len(action_1)
  
    # transition_matrix is calculated in template but not used in pure MF
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):

        # policy for the first choice
        # Calculate max Q for template consistency (unused in logic)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2 

        # Add stickiness bonus to the previously chosen action
        stickiness_bonus = np.zeros(2)
        if last_action_1 != -1:
            stickiness_bonus[last_action_1] = stickiness

        exp_q1 = np.exp(beta * q_stage1_mf + stickiness_bonus)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Action value updating
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility trace
        q_stage1_mf[action_1[trial]] += learning_rate * lam * delta_stage2
        
        # Update history
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Free RL with Separate Stage Learning Rates
This model hypothesizes that the participant learns about the immediate bandit task (aliens, Stage 2) at a different rate than the distal choice (spaceships, Stage 1). By separating `alpha_stage1` and `alpha_stage2`, the model can capture if the participant is slow to update spaceship preferences but quick to adapt to changing alien coin probabilities.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Separate Learning Rates for Stage 1 and Stage 2.
    
    Parameters:
    - alpha_1: [0, 1] Learning rate for Stage 1 (Spaceships).
    - alpha_2: [0, 1] Learning rate for Stage 2 (Aliens).
    - beta: [0, 10] Inverse temperature.
    - lam: [0, 1] Eligibility trace.
    """
    alpha_1, alpha_2, beta, lam = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Action value updating
        # Stage 1 Update uses alpha_1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += alpha_1 * delta_stage1
        
        # Stage 2 Update uses alpha_2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += alpha_2 * delta_stage2
        
        # Eligibility trace: Propagates Stage 2 error to Stage 1
        # Uses alpha_1 for the update scaling
        q_stage1_mf[action_1[trial]] += alpha_1 * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
