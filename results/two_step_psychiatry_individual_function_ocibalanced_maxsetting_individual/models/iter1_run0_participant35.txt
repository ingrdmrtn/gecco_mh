Here are three cognitive models implemented as Python functions, designed to explain the participant's behavior in the two-step task.

### Cognitive Model 1: Hybrid Model-Based/Model-Free with Perseverance
This model implements the canonical "Daw et al. (2011)" hybrid architecture. It assumes the participant uses a weighted combination of Model-Based (MB) planning and Model-Free (MF) reinforcement learning. The MB system calculates values using the known transition matrix and current Stage 2 values. The MF system uses TD-learning with eligibility traces. A perseverance bonus explains the observed stickiness in choice.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL with Perseverance.
    Combines MB planning (using fixed transition matrix) and MF learning (TD-lambda).
    
    Parameters:
    - learning_rate: [0, 1] Rate at which MF Q-values are updated.
    - beta_1: [0, 10] Inverse temperature for stage 1 choices.
    - beta_2: [0, 10] Inverse temperature for stage 2 choices.
    - w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    - lam: [0, 1] Eligibility trace parameter for MF updates.
    - pers: [0, 5] Perseverance bonus for repeating the previous Stage 1 choice.
    """
    learning_rate, beta_1, beta_2, w, lam, pers = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: T[action, next_state]
    # Action 0 -> 0.7 prob State 0, 0.3 prob State 1
    # Action 1 -> 0.3 prob State 0, 0.7 prob State 1
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf_1 = np.zeros(2)
    q_mf_2 = np.zeros((2, 2)) # Shared Q-values for Stage 2 (MB uses these too)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Choice ---
        # Model-Based Values: Expected max value of next stage
        # max_q2[s] = max(Q_stage2[s, :])
        max_q2 = np.max(q_mf_2, axis=1)
        q_mb_1 = transition_matrix @ max_q2
        
        # Perseverance Bonus
        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = pers
            
        # Net Value: Weighted sum of MB and MF + Perseverance
        q_net_1 = w * q_mb_1 + (1 - w) * q_mf_1 + pers_bonus
        
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        
        exp_q2 = np.exp(beta_2 * q_mf_2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning Updates ---
        # Prediction errors
        # Delta 1: Difference between Stage 2 value and Stage 1 MF value
        delta_1 = q_mf_2[state_idx, action_2[trial]] - q_mf_1[action_1[trial]]
        
        # Delta 2: Difference between Reward and Stage 2 value
        delta_2 = reward[trial] - q_mf_2[state_idx, action_2[trial]]
        
        # Update Stage 1 MF (TD-Lambda)
        # Update combines direct error (delta_1) and eligibility trace of second error (lambda * delta_2)
        q_mf_1[action_1[trial]] += learning_rate * delta_1 + learning_rate * lam * delta_2
        
        # Update Stage 2 MF
        q_mf_2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Asymmetric Learning Rate MF with Perseverance
This model tests the hypothesis that the participant learns differently from positive outcomes (rewards) versus negative outcomes (omissions). It is a Model-Free algorithm where the learning rate depends on the sign of the prediction error. This asymmetry is applied to both Stage 1 and Stage 2 updates.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Model-Free RL with Perseverance.
    Uses different learning rates for positive and negative prediction errors.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors (better than expected).
    - lr_neg: [0, 1] Learning rate for negative prediction errors (worse than expected).
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - lam: [0, 1] Eligibility trace parameter.
    - pers: [0, 5] Perseverance bonus.
    """
    lr_pos, lr_neg, beta_1, beta_2, lam, pers = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = pers
            
        q_net_1 = q_stage1 + pers_bonus
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Delta 1
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        lr_1 = lr_pos if delta_1 > 0 else lr_neg
        
        # Delta 2
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        lr_2 = lr_pos if delta_2 > 0 else lr_neg
        
        # Update Stage 1: Direct update + Eligibility trace
        # Note: The effective learning rate for the eligibility trace part depends on delta_2
        q_stage1[action_1[trial]] += lr_1 * delta_1 + lr_2 * lam * delta_2
        
        # Update Stage 2
        q_stage2[state_idx, action_2[trial]] += lr_2 * delta_2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Model-Free with Perseverance and Decay
This model incorporates a "forgetting" or "decay" mechanism. Since the alien reward probabilities change slowly over time, older Q-values become unreliable. This model decays the Q-value of the *unchosen* alien in Stage 2 towards 0 on each trial, helping the model adapt to the non-stationary environment better than standard RL.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Perseverance and Value Decay.
    Unchosen Stage 2 actions decay towards 0 to handle non-stationarity.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for updates.
    - decay: [0, 1] Decay rate for unchosen Stage 2 actions (0 = no decay, 1 = instant forget).
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - lam: [0, 1] Eligibility trace parameter.
    - pers: [0, 5] Perseverance bonus.
    """
    learning_rate, decay, beta_1, beta_2, lam, pers = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        pers_bonus = np.zeros(2)
        if last_action_1 != -1:
            pers_bonus[last_action_1] = pers
            
        q_net_1 = q_stage1 + pers_bonus
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        q_stage1[action_1[trial]] += learning_rate * delta_1 + learning_rate * lam * delta_2
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        # --- Decay Unchosen Stage 2 Action ---
        # The unchosen alien on the current planet has its value decayed
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_action_2] *= (1.0 - decay)
        
        # Note: We do not typically decay actions on the unvisited planet in this paradigm,
        # but one could. Here we decay the unchosen option of the visited state.
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
