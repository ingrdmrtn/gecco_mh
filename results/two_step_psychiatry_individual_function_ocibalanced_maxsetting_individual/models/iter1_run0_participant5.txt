Here are the three proposed cognitive models.

### Model 1: Hybrid MB/MF with Stickiness
This model combines Model-Based (planning) and Model-Free (habitual) value estimation, a standard approach in two-step tasks. To specifically address the participant's tendency for long streaks of repeated choices (e.g., trials 15-30, 173-181), it includes a **stickiness** parameter (choice autocorrelation) that biases the agent to repeat the previous Stage 1 choice.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free Control with Stickiness.

    Combines a Model-Based planner (using the known transition matrix)
    with a Model-Free learner (SARSA(0)). A 'stickiness' parameter is added
    to the Stage 1 choice policy to account for the participant's tendency 
    to persevere with the same spaceship choice.

    Parameters:
    - learning_rate: [0, 1] Update rate for Model-Free Q-values.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice (exploration/exploitation).
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - w: [0, 1] Mixing weight (0 = Pure MF, 1 = Pure MB).
    - stickiness: [0, 5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta_1, beta_2, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix: 
    # Space 0 -> Planet 0 (0.7), Planet 1 (0.3)
    # Space 1 -> Planet 1 (0.7), Planet 0 (0.3)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Model-Free Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    prev_a1 = -1

    for trial in range(n_trials):
        # Handle invalid trials (missing data)
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            prev_a1 = -1
            continue

        # --- Stage 1 Policy ---
        # Model-Based: Planning using transition matrix and max Stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Mix
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness (Choice Autocorrelation)
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF Update (SARSA-style)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    # Add epsilon to probabilities to avoid log(0)
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    
    log_loss = -(np.sum(np.log(p_choice_1)) + np.sum(np.log(p_choice_2)))
    return log_loss
```

### Model 2: Model-Free with Asymmetric Learning Rates
This model hypothesizes that the participant learns differently from positive outcomes (rewards) versus negative outcomes (omissions). By splitting the learning rate into `alpha_pos` and `alpha_neg`, the model can capture biases such as being quick to learn from gains but slow to unlearn from losses, or vice versa.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Asymmetric Learning Rates.
    
    This model uses separate learning rates for positive and negative prediction 
    errors (alpha_pos, alpha_neg). This allows the model to capture potential 
    biases in how the participant processes rewards versus omissions.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors (better than expected).
    - alpha_neg: [0, 1] Learning rate for negative prediction errors (worse than expected).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    """
    alpha_pos, alpha_neg, beta_1, beta_2 = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta_1 * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        lr_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[a1] += lr_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, a2] += lr_2 * delta_stage2

    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    log_loss = -(np.sum(np.log(p_choice_1)) + np.sum(np.log(p_choice_2)))
    return log_loss
```

### Model 3: Model-Free with Forgetting (Decay)
This model implements a "use it or lose it" mechanism. While chosen actions are updated based on reinforcement, unchosen actions slowly decay in value. This can explain why a participant might eventually switch away from a preferred option if its value isn't reinforced, or conversely, why they might return to an option if the alternative degrades.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Forgetting (Value Decay).
    
    Q-values for chosen actions are updated via standard RL.
    Q-values for unchosen actions decay towards 0 at each step. This models 
    memory decay or a heuristic where unobserved options lose value over time.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for chosen actions.
    - decay_rate: [0, 1] Decay rate for unchosen actions (0 = no decay, 1 = instant forgetting).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    """
    learning_rate, decay_rate, beta_1, beta_2 = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta_1 * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        # Decay unchosen Stage 1
        q_stage1_mf[1-a1] *= (1 - decay_rate)
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        # Decay unchosen Stage 2 (in the active state)
        q_stage2_mf[state_idx, 1-a2] *= (1 - decay_rate)

    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    log_loss = -(np.sum(np.log(p_choice_1)) + np.sum(np.log(p_choice_2)))
    return log_loss
```
