Here are three new cognitive models for the two-step decision task, designed to capture the participant's specific behavioral patterns such as choice perseveration and response to rewards.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF Model with Outcome-Dependent Stickiness.
    
    This model splits the 'stickiness' (perseveration) parameter into two:
    one for when the previous trial was rewarded (stick_win) and one for when
    it was unrewarded (stick_loss). This captures the participant's tendency
    to repeat choices differently after success versus failure (Win-Stay/Lose-Stay).
    
    Parameters:
    lr (float): Learning rate [0, 1].
    beta_1 (float): Inverse temperature Stage 1 [0, 10].
    beta_2 (float): Inverse temperature Stage 2 [0, 10].
    w (float): MB/MF mixing weight [0, 1].
    lam (float): Eligibility trace parameter [0, 1].
    stick_win (float): Stickiness bonus after a reward [0, 5].
    stick_loss (float): Stickiness bonus after no reward [0, 5].
    """
    lr, beta_1, beta_2, w, lam, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0
    eps = 1e-10
    
    last_a1 = -1
    last_r = 0 # Default to unrewarded state for first trial logic
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta_1 * q_net
        # Apply outcome-dependent stickiness
        if last_a1 != -1:
            if last_r == 1:
                logits[last_a1] += stick_win
            else:
                logits[last_a1] += stick_loss
        
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        if a1 != -1:
            log_loss -= np.log(probs_1[a1] + eps)
            last_a1 = a1
        else:
            last_a1 = -1
            continue
            
        if action_2[trial] == -1:
            last_r = 0
            continue
            
        # Stage 2 Policy
        state_idx = state[trial]
        a2 = action_2[trial]
        
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # Value Updates
        r = reward[trial]
        if r == -1: r = 0 # Treat missing/penalty as 0
        last_r = r
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += lr * delta_stage1 + lr * lam * delta_stage2
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF Model with Choice Trace (Kernel) Stickiness.
    
    Instead of a simple 1-step stickiness, this model maintains a decaying
    'choice trace' or kernel. This allows the model to capture stronger
    perseveration after repeated choices (momentum) or longer-term habits.
    
    Parameters:
    lr (float): Learning rate [0, 1].
    beta_1 (float): Inverse temperature Stage 1 [0, 10].
    beta_2 (float): Inverse temperature Stage 2 [0, 10].
    w (float): MB/MF mixing weight [0, 1].
    lam (float): Eligibility trace parameter [0, 1].
    trace_decay (float): Decay rate of choice trace [0, 1]. (1=no decay, 0=instant).
    trace_weight (float): Weight of choice trace in decision [0, 5].
    """
    lr, beta_1, beta_2, w, lam, trace_decay, trace_weight = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    choice_trace = np.zeros(2)
    
    log_loss = 0
    eps = 1e-10
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add choice trace bonus
        logits = beta_1 * q_net + trace_weight * choice_trace
        
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        if a1 != -1:
            log_loss -= np.log(probs_1[a1] + eps)
            # Update Trace: Decay existing trace, then add 1 to chosen action
            choice_trace = choice_trace * trace_decay
            choice_trace[a1] += 1.0
        else:
            choice_trace = choice_trace * trace_decay
            continue
            
        if action_2[trial] == -1:
            continue
            
        # Stage 2 Policy
        state_idx = state[trial]
        a2 = action_2[trial]
        
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # Value Updates
        r = reward[trial]
        if r == -1: r = 0
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += lr * delta_stage1 + lr * lam * delta_stage2
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF Model with Dual-Stage Stickiness.
    
    This model includes perseveration parameters for both the Stage 1 choice
    (Spaceship) and the Stage 2 choice (Alien). This captures potential
    motor repetition biases in the second stage independent of the planet.
    
    Parameters:
    lr (float): Learning rate [0, 1].
    beta_1 (float): Inverse temperature Stage 1 [0, 10].
    beta_2 (float): Inverse temperature Stage 2 [0, 10].
    w (float): MB/MF mixing weight [0, 1].
```
