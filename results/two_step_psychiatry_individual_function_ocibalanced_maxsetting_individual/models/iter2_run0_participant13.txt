Here are three cognitive models based on the participant data and the two-step task structure.

### Model 1: Hybrid MB/MF with Separate Decision Noise
This model hypothesizes that the participant combines Model-Based (MB) and Model-Free (MF) strategies, but uses different levels of exploration (decision noise) for the spaceship choice (Stage 1) versus the alien choice (Stage 2). The `w` parameter controls the balance between MB and MF control.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model 1: Hybrid MB/MF with Separate Decision Noise (Beta 1 & Beta 2).
    
    This model combines Model-Based (transition-aware) and Model-Free (TD-0) 
    value estimation. It allows for distinct inverse temperatures (betas) for 
    Stage 1 and Stage 2, capturing different levels of exploration/exploitation 
    at each step.
    
    Parameters:
    learning_rate: [0, 1] - Rate at which Q-values are updated.
    beta_1: [0, 10] - Inverse temperature for Stage 1 (Spaceship choice).
    beta_2: [0, 10] - Inverse temperature for Stage 2 (Alien choice).
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    perseveration: [-3, 3] - Bias for repeating the previous Stage 1 choice.
    """
    learning_rate, beta_1, beta_2, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    # Fixed transition probabilities as per task structure
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values to 0.5 (neutral, as rewards are 0/1)
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        # Model-Based Value: Expected max value of next stage weighted by transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Shape (2,) for State 0 and State 1
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits with Stage 1 specific beta
        logits_1 = beta_1 * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        
        # Calculate logits with Stage 2 specific beta
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Stage 1 MF Update (TD-0): Driven by the value of the state actually reached
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 MF Update: Driven by reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: TD($\lambda$) with Outcome-Dependent Perseveration
This model extends the TD($\lambda$) framework by refining the perseveration mechanism. Instead of a simple "stickiness" bias, it differentiates between "Win-Stay" and "Lose-Shift" behaviors. It adds a specific bias `pers_win` if the previous trial was rewarded, and `pers_loss` if it was not.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model 2: TD(lambda) Q-Learning with Outcome-Dependent Perseveration.
    
    This model uses eligibility traces (TD-lambda) for learning, but splits the 
    perseveration bias into two components: one for when the previous trial was 
    rewarded (Win-Stay) and one for when it was unrewarded (Lose-Stay/Shift).
    
    Parameters:
    learning_rate: [0, 1] - Base learning rate.
    lambda_eligibility: [0, 1] - Eligibility trace decay (0=TD-0, 1=Monte Carlo).
    beta: [0, 10] - Inverse temperature.
    pers_win: [-3, 3] - Bias to repeat Stage 1 choice after a Reward (Win-Stay).
    pers_loss: [-3, 3] - Bias to repeat Stage 1 choice after No Reward.
    """
    learning_rate, lambda_eligibility, beta, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
  
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    prev_reward = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        logits_1 = beta * q_stage1_mf
        
        # Apply Outcome-Dependent Perseveration
        if prev_a1 != -1:
            if prev_reward == 1:
                logits_1[prev_a1] += pers_win
            else:
                logits_1[prev_a1] += pers_loss
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates (TD-Lambda) ---
        # TD-0 error for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # TD error for Stage 2 (Reward Prediction Error)
        current_delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        # Update Stage 1: Mix of TD-0 (Step 1 error) and TD-1 (Step 2 error)
        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + lambda_eligibility * current_delta_stage2)

        # Update Stage 2
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * current_delta_stage2
        
        prev_a1 = action_1[trial]
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with Stage 2 Forgetting
This model posits that while the participant builds a Model-Based map, the values of the aliens (Stage 2) are subject to passive decay or "forgetting". When an alien is not chosen, its value decays towards a neutral prior (0.5). This affects the Model-Based value calculation for the spaceships, as the MB system relies on accurate Stage 2 estimates.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model 3: Hybrid MB/MF with Stage 2 Forgetting (Decay).
    
    This model includes a decay mechanism for the unchosen alien in Stage 2.
    The Model-Based system uses these decaying values to estimate spaceship values.
    This captures the difficulty of maintaining accurate value representations 
    for unvisited states.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for chosen options.
    decay_rate: [0, 1] - Rate at which unchosen alien values decay to 0.5.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight (0=MF, 1=MB).
    perseveration: [-3, 3] - Bias for repeating the previous Stage 1 choice.
    """
    learning_rate, decay_rate, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # MB Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Mixing
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Update Chosen Alien (Stage 2)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Decay Unchosen Alien (Stage 2) towards neutral 0.5
        unchosen_a2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_a2] = (1 - decay_rate) * q_stage2_mf[state_idx, unchosen_a2] + decay_rate * 0.5
        
        # Update Stage 1 MF (TD-0)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
