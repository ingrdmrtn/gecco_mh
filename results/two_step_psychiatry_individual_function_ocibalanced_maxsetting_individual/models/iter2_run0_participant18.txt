Here are three new cognitive models based on the participant data and the two-step task structure.

### Model 1: Hybrid Model with Separate Stage-Specific Exploration (Dual Beta)
**Hypothesis:** The participant may exhibit different levels of exploration/exploitation (randomness) in the high-level planning stage (Stage 1) compared to the immediate reward stage (Stage 2). Stage 1 requires integrating Model-Based probabilities, while Stage 2 is a simple bandit task; this cognitive load difference often results in distinct "temperatures" for decision-making.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with separate inverse temperatures (betas) for Stage 1 and Stage 2.
    
    Parameters:
    lr (float): Learning rate for value updates [0,1].
    beta1 (float): Inverse temperature for Stage 1 (Spaceship choice) [0,10].
    beta2 (float): Inverse temperature for Stage 2 (Alien choice) [0,10].
    w (float): Weighting parameter (0 = Pure MF, 1 = Pure MB) [0,1].
    """
    lr, beta1, beta2, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Shape: (State/Planet, Action/Alien)

    for trial in range(n_trials):
        # Skip missing data
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q_Stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax using beta1
        exp_q1 = np.exp(beta1 * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        # Softmax using beta2 (Pure MF at this stage)
        exp_q2 = np.exp(beta2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # Prediction errors
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        # Update Stage 1 MF (TD(1) logic implicitly)
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        q_stage1_mf[action_1[trial]] += lr * delta_stage2

        # Update Stage 2 MF
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Dynamic Transition Learning
**Hypothesis:** Although the task instructions imply fixed transition probabilities (70/30), participants often track these statistics dynamically. If they experience a streak of "rare" transitions, they may update their internal model of the spaceship-planet mapping. This model estimates the transition matrix `T` trial-by-trial rather than using a static constant.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model where the participant learns the transition matrix over time.
    
    Parameters:
    lr_val (float): Learning rate for Q-values [0,1].
    lr_trans (float): Learning rate for transition probabilities [0,1].
    beta (float): Inverse temperature [0,10].
    w (float): MB/MF Weighting [0,1].
    """
    lr_val, lr_trans, beta, w = model_parameters
    n_trials = len(action_1)
  
    # Initialize belief about transitions (start with instructed 0.7/0.3 or 0.5 neutral)
    # Here we initialize at instructions: 0.7 for common
    estimated_trans = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        # Use CURRENT estimated transition matrix for MB calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = estimated_trans @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[chosen_a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        q_stage1_mf[chosen_a1] += lr_val * delta_stage1
        q_stage1_mf[chosen_a1] += lr_val * delta_stage2
        q_stage2_mf[state_idx, action_2[trial]] += lr_val * delta_stage2

        # --- Transition Matrix Update ---
        # State prediction error: Observed state is 1, unobserved is 0
        # Update the row corresponding to the chosen spaceship
        # If we went A -> X (0->0), boost T[0,0], lower T[0,1]
        
        # Vectorized update for the chosen row
        # Target vector is one-hot encoding of the observed state
        target = np.zeros(2)
        target[state_idx] = 1.0
        
        estimated_trans[chosen_a1] += lr_trans * (target - estimated_trans[chosen_a1])

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 3: Surprise-Modulated Learning Rate Model
**Hypothesis:** Learning may be modulated by the "surprise" of the transition. If a participant experiences a "Rare" transition, they might learn differently (either discounting the trial as noise or boosting learning due to salience) compared to a "Common" transition. This model uses two distinct learning rates depending on whether the transition was expected.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid model where learning rate depends on whether the transition was Common or Rare.
    
    Parameters:
    lr_common (float): Learning rate after a Common transition [0,1].
    lr_rare (float): Learning rate after a Rare transition [0,1].
    beta (float): Inverse temperature [0,10].
    w (float): MB/MF Weighting [0,1].
    """
    lr_common, lr_rare, beta, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Determine Learning Rate ---
        # Common: (0->0) or (1->1). Rare: (0->1) or (1->0)
        is_common = (a1 == state_idx)
        current_lr = lr_common if is_common else lr_rare

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        # Use the context-specific learning rate for this trial's updates
        q_stage1_mf[a1] += current_lr * delta_stage1
        q_stage1_mf[a1] += current_lr * delta_stage2
        
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_stage2

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```
