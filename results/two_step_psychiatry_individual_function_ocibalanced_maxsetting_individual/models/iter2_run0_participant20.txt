Here are three new cognitive models for the two-step decision task, designed to capture different potential mechanisms underlying the participant's behavior, such as eligibility traces, asymmetric learning, and memory decay.

### Cognitive Model 1: Hybrid MB/MF with Eligibility Traces and Stickiness
This model extends the standard hybrid learner by including an **eligibility trace** ($\lambda$). This allows the reward prediction error from the second stage (aliens) to directly update the first stage (spaceship) model-free values, bridging the gap between the final outcome and the initial choice. It also includes **stickiness** to account for the observed perseverance in spaceship choices.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based/Model-Free agent with Eligibility Traces and Stickiness.
    
    incorporates 'lambda_param' to modulate how much the Stage 2 RPE affects 
    Stage 1 Q-values (eligibility trace), alongside choice stickiness.

    Parameters:
    - learning_rate: [0,1] Learning rate for Q-value updates.
    - beta_1: [0,10] Inverse temperature for Stage 1 softmax.
    - beta_2: [0,10] Inverse temperature for Stage 2 softmax.
    - w: [0,1] Weighting parameter (0=MF, 1=MB).
    - lambda_param: [0,1] Eligibility trace parameter.
    - stickiness: [0,10] Choice perseverance bonus.
    """
    learning_rate, beta_1, beta_2, w, lambda_param, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness to logits
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx] - np.max(beta_2 * q_stage2_mf[state_idx]))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 RPE
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 RPE
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility Trace Update: Propagate Stage 2 RPE to Stage 1 MF value
        q_stage1_mf[action_1[trial]] += learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    # Filter out trials where probabilities were not calculated (missing data)
    valid_indices = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_indices] + eps)) + np.sum(np.log(p_choice_2[valid_indices] + eps)))
    return log_loss
```

### Cognitive Model 2: Asymmetric Model-Free with Stickiness
This model tests the hypothesis that the participant relies on a **Model-Free** strategy (ignoring the transition structure, effectively $w=0$) but learns differently from positive versus negative prediction errors (**Asymmetric Learning Rates**). This is often relevant in reward processing tasks where sensitivity to gains and losses differs.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Free agent with Asymmetric Learning Rates and Stickiness.
    
    Differentiates learning from positive (alpha_pos) and negative (alpha_neg)
    prediction errors. Assumes no Model-Based planning (w=0).

    Parameters:
    - alpha_pos: [0,1] Learning rate for positive prediction errors.
    - alpha_neg: [0,1] Learning rate for negative prediction errors.
    - beta_1: [0,10] Inverse temperature for Stage 1 softmax.
    - beta_2: [0,10] Inverse temperature for Stage 2 softmax.
    - stickiness: [0,10] Choice perseverance bonus for Stage 1.
    """
    alpha_pos, alpha_neg, beta_1, beta_2, stickiness = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy (Pure MF) ---
        logits_1 = beta_1 * q_stage1_mf
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx] - np.max(beta_2 * q_stage2_mf[state_idx]))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        eff_alpha_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += eff_alpha_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        eff_alpha_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += eff_alpha_2 * delta_stage2
        

    eps = 1e-10
    valid_indices = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_indices] + eps)) + np.sum(np.log(p_choice_2[valid_indices] + eps)))
    return log_loss
```

### Cognitive Model 3: Hybrid MB/MF with Stickiness and Decay
Since the task description notes that alien reward probabilities "changed slowly over trials," this model introduces a **Decay** parameter. Unchosen options in the second stage decay in value, allowing the agent to "forget" old values and adapt to the drifting probabilities more effectively than standard learning updates alone.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based/Model-Free agent with Stickiness and Value Decay.
    
    Includes a decay parameter that reduces the value of unchosen actions 
    in Stage 2, helping the agent adapt to drifting reward probabilities.

    Parameters:
    - learning_rate: [0,1] Rate at which Q-values are updated.
    - beta_1: [0,10] Inverse temperature for Stage 1 softmax.
    - beta_2: [0,10] Inverse temperature for Stage 2 softmax.
    - w: [0,1] Weighting parameter (0=MF, 1=MB).
    - stickiness: [0,10] Choice perseverance bonus.
    - decay: [0,1] Decay rate for unchosen Stage 2 actions.
    """
    learning_rate, beta_1, beta_2, w, stickiness, decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx] - np.max(beta_2 * q_stage2_mf[state_idx]))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Apply Decay to the unchosen alien on the current planet
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action_2] *= (1.0 - decay)

    eps = 1e-10
    valid_indices = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_indices] + eps)) + np.sum(np.log(p_choice_2[valid_indices] + eps)))
    return log_loss
```
