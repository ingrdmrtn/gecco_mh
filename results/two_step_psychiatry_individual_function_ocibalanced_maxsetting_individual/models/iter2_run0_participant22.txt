Here are three new cognitive models based on the participant data and the two-step task structure.

### Cognitive Model 1: Outcome-Dependent Stickiness
This model hypothesizes that the participant's tendency to repeat a choice (stickiness) is not constant but depends on the previous outcome. The participant may be more likely to "stick" to a spaceship after a win (Win-Stay) compared to a loss. This separates the mechanical perseveration from reward-sensitive repetition.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Outcome-Dependent Stickiness.
    
    Differentiates 'stickiness' (repetition bias) based on the previous trial's outcome.
    Standard stickiness assumes a constant bias; this model assumes the bias 
    changes if the previous trial was rewarded (Win-Stay) vs unrewarded.

    Parameters:
    lr: [0,1] - Learning rate.
    beta_1: [0,10] - Inverse temperature for Stage 1 (Spaceship).
    beta_2: [0,10] - Inverse temperature for Stage 2 (Alien).
    w: [0,1] - Weighting between Model-Based and Model-Free values (1=MB).
    stick_win: [0,5] - Stickiness bonus applied to Stage 1 choice if previous trial was Rewarded.
    stick_loss: [0,5] - Stickiness bonus applied to Stage 1 choice if previous trial was Unrewarded.
    """
    lr, beta_1, beta_2, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    prev_reward = -1

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        
        # Apply outcome-dependent stickiness
        if prev_action_1 != -1:
            if prev_reward == 1:
                logits_1[prev_action_1] += stick_win
            elif prev_reward == 0:
                logits_1[prev_action_1] += stick_loss
            # If prev_reward was -1 (missing), no stickiness applied or neutral
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # MF Stage 1 Update (TD Error 1)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        # MF Stage 2 Update (TD Error 2)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        prev_action_1 = action_1[trial]
        prev_reward = reward[trial]

    eps = 1e-10
    # Filter out trials where probability wasn't calculated (missing data)
    valid_trials = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Cognitive Model 2: Decaying Choice Trace
The participant data shows long streaks of choosing the same spaceship (e.g., trials 25-30, 102-126). A simple 1-back stickiness parameter often fails to capture the momentum of these streaks. This model uses a "Choice Trace" that accumulates when an action is chosen and decays slowly, creating a stronger, more persistent habit effect.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Decaying Choice Trace.
    
    Replaces simple 1-back stickiness with an accumulating 'Choice Trace'.
    Repeatedly choosing an option builds up a 'habit' (trace) that decays 
    gradually, explaining long streaks of behavior better than simple repetition.

    Parameters:
    lr: [0,1] - Learning rate.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - MB/MF weighting.
    stick_weight: [0,5] - Weight of the choice trace in the decision logic.
    trace_decay: [0,1] - Rate at which the choice trace persists (1=no decay, 0=instant decay).
    """
    lr, beta_1, beta_2, w, stick_weight, trace_decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace vector for the two spaceships
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        
        # Add the accumulated choice trace to the logits
        logits_1 += stick_weight * choice_trace
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Update Choice Trace
        # Decay existing trace
        choice_trace *= trace_decay
        # Strengthen trace for the chosen action
        choice_trace[action_1[trial]] += 1.0

    eps = 1e-10
    valid_trials = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Cognitive Model 3: Eligibility Trace (Lambda) with Passive Decay
This model combines sophisticated credit assignment with a mechanism for handling non-stationarity. The eligibility trace (`lam`) allows the reward at Stage 2 to directly influence the Stage 1 choice (Model-Free), while the decay parameter ensures that old Q-values fade over time, which is crucial given the drifting reward probabilities in the task.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Eligibility Traces (Lambda) and Decay.
    
    Combines 'Lambda' (allowing Stage 2 outcomes to directly update Stage 1 values)
    with 'Decay' (forgetting unchosen option values). This addresses both the 
    temporal credit assignment problem and the drifting reward probabilities.

    Parameters:
    lr: [0,1] - Learning rate.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - MB/MF weighting.
    lam: [0,1] - Eligibility trace parameter (0 = TD(0), 1 = Monte Carlo).
    decay: [0,1] - Decay rate for unchosen Q-values (0=no decay, 1=instant forgetting).
    """
    lr, beta_1, beta_2, w, lam, decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 TD error
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Stage 2 TD error (Reward Prediction Error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1 MF: Includes direct update from Stage 2 RPE weighted by lambda
        q_stage1_mf[action_1[trial]] += lr * delta_stage1 + lr * lam * delta_stage2
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Apply Decay to unchosen options to handle non-stationarity
        q_stage1_mf[1 - action_1[trial]] *= (1 - decay)
        q_stage2_mf[state_idx, 1 - action_2[trial]] *= (1 - decay)

    eps = 1e-10
    valid_trials = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
