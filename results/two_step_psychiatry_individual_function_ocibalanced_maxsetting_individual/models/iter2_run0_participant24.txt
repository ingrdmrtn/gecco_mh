Here are three new cognitive models for the two-step decision task, designed to capture different potential learning mechanisms such as asymmetric learning from gains/losses, dynamic learning of transition structures, and memory decay.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model 1: Asymmetric Model-Free TD(lambda).
    This model extends the standard Model-Free Q-learning by employing separate 
    learning rates for positive and negative prediction errors. This accounts 
    for potential asymmetries in how the participant learns from better-than-expected 
    versus worse-than-expected outcomes (e.g., learning more from losses than gains).
    
    Parameters:
    alpha_pos: [0, 1] - Learning rate for positive prediction errors (delta > 0).
    alpha_neg: [0, 1] - Learning rate for negative prediction errors (delta < 0).
    beta: [0, 10] - Inverse temperature (exploration/exploitation balance).
    lam: [0, 1] - Eligibility trace parameter (weight of Stage 2 reward on Stage 1 choice).
    pers: [0, 5] - Perseverance bonus for repeating the previous Stage 1 choice.
    """
    alpha_pos, alpha_neg, beta, lam, pers = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    log_loss = 0.0
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            continue
            
        a1, s, a2, r = int(action_1[trial]), int(state[trial]), int(action_2[trial]), float(reward[trial])
        
        # Stage 1 Policy
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
        
        exp_q1 = np.exp(beta * q_net_1)
        prob_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(prob_1[a1] + 1e-10)
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2[s])
        prob_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(prob_2[a2] + 1e-10)
        
        # Value Updates
        # Calculate Prediction Errors
        delta1 = q_stage2[s, a2] - q_stage1[a1]
        delta2 = r - q_stage2[s, a2]
        
        # Update Stage 1 Value (TD(0) component)
        alpha1 = alpha_pos if delta1 > 0 else alpha_neg
        q_stage1[a1] += alpha1 * delta1
        
        # Update Stage 2 Value
        alpha2 = alpha_pos if delta2 > 0 else alpha_neg
        q_stage2[s, a2] += alpha2 * delta2
        
        # Update Stage 1 Value (Eligibility Trace component)
        # The effective error backpropagated is delta2
        alpha_elig = alpha_pos if delta2 > 0 else alpha_neg
        q_stage1[a1] += alpha_elig * lam * delta2
        
        last_action_1 = a1
        
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model 2: Dynamic Hybrid Model.
    This model combines Model-Based (MB) and Model-Free (MF) strategies. 
    Crucially, it assumes the participant does not assume fixed transition 
    probabilities (0.7/0.3) but actively learns them over time (Dynamic MB).
    
    Parameters:
    lr_val: [0, 1] - Learning rate for Q-value updates.
    lr_trans: [0, 1] - Learning rate for transition probability updates.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    lam: [0, 1] - Eligibility trace for the MF component.
    pers: [0, 5] - Perseverance bonus.
    """
    lr_val, lr_trans, beta, w, lam, pers = model_parameters
    n_trials = len(action_1)
    
    # Initialize transitions: A(0)->X(0) 0.7, U(1)->Y(1) 0.7
    # Rows: Action 0, Action 1. Cols: State 0, State 1.
    trans_probs = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf1 = np.zeros(2)
    q_mf2 = np.zeros((2, 2))
    log_loss = 0.0
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            continue
            
        a1, s, a2, r = int(action_1[trial]), int(state[trial]), int(action_2[trial]), float(reward[trial])
        
        # Model-Based Valuation
        # MB value of stage 1 is expected max value of stage 2 given learned transitions
        max_q2 = np.max(q_mf2, axis=1)
        q_mb1 = trans_probs @ max_q2
        
        # Net Stage 1 Value (Hybrid)
        q_net_1 = w * q_mb1 + (1 - w) * q_mf1
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_net_1)
        prob_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(prob_1[a1] + 1e-10)
        
        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_mf2[s])
        prob_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(prob_2[a2] + 1e-10)
        
        # Model-Free Updates
        delta1 = q_mf2[s, a2] - q_mf1[a1]
        q_mf1[a1] += lr_val * delta1
        
        delta2 = r - q_mf2[s, a2]
        q_mf2[s, a2] += lr_val * delta2
        
        q_mf1[a1] += lr_val * lam * delta2
        
        # Transition Probability Update (Dynamic MB)
        # Update the row corresponding to the chosen spaceship (a1)
        # Increase prob of observed state (s), decrease prob of unobserved (1-s)
        trans_probs[a1, s] += lr_trans * (1.0 - trans_probs[a1, s])
        trans_probs[a1, 1-s] += lr_trans * (0.0 - trans_probs[a1, 1-s])
        
        last_action_1 = a1
        
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model 3: Model-Free TD(lambda) with Memory Decay.
    This model assumes that learned values are not permanent but decay over time.
    This simulates forgetting or a "recency" bias where older experiences matter less.
    All Q-values decay towards 0 at the end of each trial.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for Q-value updates.
    beta: [0, 10] - Inverse temperature.
    lam: [0, 1] - Eligibility trace.
    pers: [0, 5] - Perseverance bonus.
    decay: [0, 1] - Decay rate applied to all Q-values at the end of each trial.
    """
    learning_rate, beta, lam, pers, decay = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    log_loss = 0.0
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            continue
            
        a1, s, a2, r = int(action_1[trial]), int(state[trial]), int(action_2[trial]), float(reward[trial])
        
        # Stage 1 Choice
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        exp_q1 = np.exp(beta * q_net_1)
        prob_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(prob_1[a1] + 1e-10)
        
        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s])
        prob_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(prob_2[a2] + 1e-10)
        
        # Updates
        delta1 = q_stage2[s, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta1
        
        delta2 = r - q_stage2[s, a2]
        q_stage2[s, a2] += learning_rate * delta2
        
        q_stage1[a1] += learning_rate * lam * delta2
        
        # Memory Decay
        # All Q-values decay towards 0
        q_stage1 *= (1.0 - decay)
        q_stage2 *= (1.0 - decay)
        
        last_action_1 = a1
        
    return log_loss
```
