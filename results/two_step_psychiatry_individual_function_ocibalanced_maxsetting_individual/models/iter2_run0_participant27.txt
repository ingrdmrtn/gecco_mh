Here are three new cognitive models for the two-step task, implemented as Python functions.

### Model 1: Hybrid Model with Outcome-Dependent Stickiness
This model refines the concept of choice stickiness by differentiating between "Win-Stay" and "Lose-Stay" behaviors. Instead of a single perseveration parameter, it uses two parameters: one applied when the previous trial was rewarded (`stick_rew`) and one when it was unrewarded (`stick_unrew`). This allows the model to capture strategies where the participant's tendency to repeat a choice depends on the outcome.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Outcome-Dependent Stickiness.
    
    This model splits the stickiness parameter into two: one for when the previous 
    trial was rewarded (Win-Stay) and one for when it was not (Lose-Stay/Shift).
    This allows the model to capture different perseveration tendencies based on outcome.

    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-updating.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Weight for Model-Based (1) vs Model-Free (0).
    - stick_rew: [0, 5] Stickiness bonus if previous trial was rewarded.
    - stick_unrew: [0, 5] Stickiness bonus if previous trial was unrewarded.
    """
    learning_rate, beta_1, beta_2, w, stick_rew, stick_unrew = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix for MB
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1
    last_reward = 0
    
    for trial in range(n_trials):
        # Skip missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1 # Reset stickiness continuity on missing data
            continue
            
        # --- Stage 1 Decision ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Outcome-Dependent Stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                q_net[last_action_1] += stick_rew
            else:
                q_net[last_action_1] += stick_unrew
                
        # Softmax Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Stage 1 Update (SARSA-like for MF)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update (Q-learning)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]
        
    eps = 1e-10
    mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Global Q-Value Decay
This model introduces a forgetting mechanism. In the standard model, Q-values persist indefinitely until updated. Here, all Stage 2 Q-values decay towards 0 (uncertainty) on every trial. This allows the model to account for the participant potentially perceiving the environment as volatile, where older information becomes less reliable over time.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Global Q-Value Decay.
    
    Implements a forgetting mechanism where all Q-values in Stage 2 decay towards 0 
    on every trial. This allows the model to adapt to changing reward probabilities 
    by diminishing the weight of old experiences.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_1: [0, 10] Stage 1 inverse temperature.
    - beta_2: [0, 10] Stage 2 inverse temperature.
    - w: [0, 1] MB/MF weight.
    - stickiness: [0, 5] Stage 1 choice stickiness.
    - decay: [0, 1] Decay rate for Stage 2 values (0 = no decay, 1 = full reset).
    """
    learning_rate, beta_1, beta_2, w, stickiness, decay = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue

        # Decay step: All Stage 2 values decay
        q_stage2_mf *= (1 - decay)

        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Dual-Stage Stickiness
The standard model only includes stickiness for the first stage (spaceship choice). However, participants often exhibit perseveration in the second stage (alien choice) as well. This model adds a second stickiness parameter (`stickiness_2`) that biases the participant towards repeating the same alien choice within a given planet.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Dual-Stage Stickiness.
    
    Includes perseveration (stickiness) in both Stage 1 (Spaceship choice) and 
    Stage 2 (Alien choice). Stage 2 stickiness is state-dependent (tracks last 
    choice per planet).
    
    Parameters:
    - learning_rate: [0, 1]
    - beta_1: [0, 10]
    - beta_2: [0, 10]
    - w: [0, 1]
    - stickiness_1: [0, 5] Stickiness for Stage 1.
    - stickiness_2: [0, 5] Stickiness for Stage 2.
    """
    learning_rate, beta_1, beta_2, w, stickiness_1, stickiness_2 = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_action_2_per_state = np.array([-1, -1]) # Track last alien chosen for Planet 0 and Planet 1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            # Reset Stage 2 stickiness on interruption to be safe
            last_action_2_per_state = np.array([-1, -1])
            continue

        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_1
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 ---
        state_idx = state[trial]
        q_stage2_adjusted = q_stage2_mf[state_idx].copy()
        
        # Apply Stage 2 Stickiness
        if last_action_2_per_state[state_idx] != -1:
            q_stage2_adjusted[last_action_2_per_state[state_idx]] += stickiness_2
            
        exp_q2 = np.exp(beta_2 * q_stage2_adjusted)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        last_action_2_per_state[state_idx] = action_2[trial]
        
    eps = 1e-10
    mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
