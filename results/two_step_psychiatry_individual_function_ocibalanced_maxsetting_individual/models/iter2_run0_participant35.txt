Here are three cognitive models based on the participant's behavior and the feedback provided.

The participant data suggests a mix of **Model-Based** behavior (switching after rare transitions followed by rewards) and **non-stationarity** (drifting reward probabilities on planets). The feedback indicates that a **decay mechanism** on Stage 2 values and **perseverance** on Stage 1 choices are highly effective.

The three models proposed are:
1.  **Hybrid MB/MF with Value Decay**: A full hybrid model incorporating Model-Based planning and Model-Free learning, with the critical addition of decay to handle drifting Stage 2 rewards.
2.  **Asymmetric Model-Free with Decay**: A Model-Free agent that learns differently from positive and negative prediction errors (risk/optimism bias), combined with decay.
3.  **Pure Model-Based with Decay**: A parsimonious model assuming the participant relies entirely on the transition structure (no MF Stage 1 learning) but updates Stage 2 estimates with decay.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF with Value Decay.
    Combines Model-Based planning (using transition matrix) with Model-Free learning.
    Includes decay on Stage 2 values to handle drifting probabilities.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - lam: [0, 1] Eligibility trace parameter for MF stage 1 update.
    - decay: [0, 1] Decay rate for unchosen Stage 2 actions.
    - pers: [0, 5] Perseverance bonus for stage 1.
    """
    learning_rate, beta_1, beta_2, w, lam, decay, pers = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Choice ---
        # Model-Based Value: Expected value based on transition matrix and current Stage 2 max values
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted mixture of MB and MF values
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Perseverance bonus
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # MF Stage 1 Update (TD(lambda))
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        q_stage1_mf[action_1[trial]] += learning_rate * delta_1 + learning_rate * lam * delta_2
        
        # Stage 2 Update
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        # Decay unchosen stage 2 values to handle non-stationarity
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_action_2] *= (1.0 - decay)
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free with Asymmetric Learning Rates and Decay.
    Differentiates learning from positive vs negative prediction errors (e.g., optimism/pessimism).
    Includes decay for unchosen actions.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors.
    - lr_neg: [0, 1] Learning rate for negative prediction errors.
    - decay: [0, 1] Decay rate for unchosen Stage 2 actions.
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - lam: [0, 1] Eligibility trace parameter.
    - pers: [0, 5] Perseverance bonus.
    """
    lr_pos, lr_neg, decay, beta_1, beta_2, lam, pers = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Choice ---
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        # Asymmetric update for Stage 1
        # Apply lr_pos/neg based on the sign of the error terms
        term1 = (lr_pos if delta_1 > 0 else lr_neg) * delta_1
        term2 = (lr_pos if delta_2 > 0 else lr_neg) * lam * delta_2
        q_stage1[action_1[trial]] += term1 + term2
        
        # Asymmetric update for Stage 2
        q_stage2[state_idx, action_2[trial]] += (lr_pos if delta_2 > 0 else lr_neg) * delta_2
        
        # Decay unchosen stage 2
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_action_2] *= (1.0 - decay)
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based with Value Decay.
    Assumes the participant relies solely on the transition structure for Stage 1.
    Stage 1 values are computed on-the-fly from Stage 2 values.
    Stage 2 values are learned via RL with decay.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Stage 2 values.
    - decay: [0, 1] Decay rate for unchosen Stage 2 actions.
    - beta_1: [0, 10] Inverse temperature for stage 1.
    - beta_2: [0, 10] Inverse temperature for stage 2.
    - pers: [0, 5] Perseverance bonus.
    """
    learning_rate, decay, beta_1, beta_2, pers = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Choice (Pure MB) ---
        # Calculate Q1 directly from Q2 via transition matrix
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Stage 2 Update ---
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        # Decay unchosen stage 2
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_action_2] *= (1.0 - decay)
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
