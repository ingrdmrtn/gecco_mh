def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Learning Rate Model with Perseveration.
    
    This model distinguishes between learning from positive prediction errors 
    (rewards better than expected) and negative prediction errors (outcomes 
    worse than expected). It also includes a perseveration parameter to account 
    for the participant's tendency to repeat spaceship choices.
    
    Parameters:
    lr_pos: Learning rate for positive prediction errors [0,1]
    lr_neg: Learning rate for negative prediction errors [0,1]
    beta: Inverse temperature (randomness) [0,10]
    w: Weight mixing MB and MF (0=Pure MF, 1=Pure MB) [0,1]
    p: Perseveration (stickiness) to previous Stage 1 choice [0,1]
    lam: Eligibility trace parameter [0,1]
    """
    lr_pos, lr_neg, beta, w, p, lam = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration bonus to the previously chosen spaceship
        if prev_action_1 != -1:
            q_net_stage1[prev_action_1] += p
            
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        s1_choice = action_1[trial]
        state_idx = state[trial]
        prev_action_1 = s1_choice
        
        # --- Stage 2 Decision ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        s2_choice = action_2[trial]
        r = reward[trial]
        
        # --- Learning Updates ---
        
        # Stage 1 Prediction Error
        delta_stage1 = q_stage2_mf[state_idx, s2_choice] - q_stage1_mf[s1_choice]
        # Use lr_pos if PE is positive, else lr_neg
        lr_1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[s1_choice] += lr_1 * delta_stage1
        
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[state_idx, s2_choice]
        # Use lr_pos if PE is positive, else lr_neg
        lr_2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[state_idx, s2_choice] += lr_2 * delta_stage2
        
        # Eligibility Trace Update for Stage 1 (using Stage 2 PE)
        q_stage1_mf[s1_choice] += lr_2 * lam * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Dual Inverse Temperature Model with Perseveration.
    
    This model uses separate inverse temperature (beta) parameters for Stage 1 
    (Spaceship) and Stage 2 (Alien) decisions. This allows the model to capture 
    different levels of exploration/exploitation randomness in the two stages. 
    It also includes a perseveration parameter for Stage 1.
    
    Parameters:
    lr: Learning rate [0,1]
    beta_1: Inverse temperature for Stage 1 [0,10]
    beta_2: Inverse temperature for Stage 2 [0,10]
    w: Weight mixing MB and MF [0,1]
    p: Perseveration (stickiness) to previous Stage 1 choice [0,1]
    lam: Eligibility trace parameter [0,1]
    """
    lr, beta_1, beta_2, w, p, lam = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net_stage1[prev_action_1] += p
            
        # Use beta_1 for Stage 1
        exp_q1 = np.exp(beta_1 * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        s1_choice = action_1[trial]
        state_idx = state[trial]
        prev_action_1 = s1_choice
        
        # --- Stage 2 Decision ---
        qs_stage2 = q_stage2_mf[state_idx]
        # Use beta_2 for Stage 2
        exp_q2 = np.exp(beta_2 * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        s2_choice = action_2[trial]
        r = reward[trial]
        
        # --- Learning Updates ---
        delta_stage1 = q_stage2_mf[state_idx, s2_choice] - q_stage1_mf[s1_choice]
        q_stage1_mf[s1_choice] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, s2_choice]
        q_stage2_mf[state_idx, s2_choice] += lr * delta_stage2
        
        q_stage1_mf[s1_choice] += lr * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dual Perseveration Model.
    
    This model includes perseveration (stickiness) parameters for BOTH Stage 1 
    (Spaceship) and Stage 2 (Alien) choices. This captures the tendency to 
    repeat motor responses or selections at both levels of the task hierarchy.
    
    Parameters:
    lr: Learning rate [0,1]
    beta: Inverse temperature [0,10]
    w: Weight mixing MB and MF [0,1]
    p_1: Perseveration for Stage 1 choice [0,1]
    p_2: Perseveration for Stage 2 choice [0,1]
    lam: Eligibility trace parameter [0,1]
    """
    lr, beta, w, p_1, p_2, lam = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    prev_action_2 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Stage 1 Perseveration
        if prev_action_1 != -1:
            q_net_stage1[prev_action_1] += p_1
            
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        s1_choice = action_1[trial]
        state_idx = state[trial]
        prev_action_1 = s1_choice
        
        # --- Stage 2 Decision ---
        # Copy to avoid modifying the learned Q-values with transient stickiness
        qs_stage2 = q_stage2_mf[state_idx].copy()
        
        # Stage 2 Perseveration (Response Stickiness)
        if prev_action_2 != -1:
             qs_stage2[prev_action_2] += p_2
             
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        s2_choice = action_2[trial]
        prev_action_2 = s2_choice
        r = reward[trial]
        
        # --- Learning Updates ---
        delta_stage1 = q_stage2_mf[state_idx, s2_choice] - q_stage1_mf[s1_choice]
        q_stage1_mf[s1_choice] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, s2_choice]
        q_stage2_mf[state_idx, s2_choice] += lr * delta_stage2
        
        q_stage1_mf[s1_choice] += lr * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
