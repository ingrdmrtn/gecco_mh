Here are three cognitive models expressed as Python functions. These models focus on Model-Free reinforcement learning mechanisms, which appear to best describe the participant's behavior (strong Win-Stay patterns regardless of transition type), while introducing distinct mechanisms like separate learning rates, eligibility traces with decay, and lapse rates to capture specific dynamics of the data.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Separate Learning Rates for Stage 1 and Stage 2.
    
    This model assumes that the participant learns the value of the spaceships (Stage 1)
    and the aliens (Stage 2) at different speeds. For instance, the value of a spaceship
    might be more stable (lower alpha) or updated more aggressively (higher alpha) than 
    the specific alien rewards. This separation allows for different plasticity in 
    higher-level vs lower-level choices.
    
    Parameters:
    - alpha_1: [0, 1] Learning rate for Stage 1 (Spaceship choice).
    - alpha_2: [0, 1] Learning rate for Stage 2 (Alien choice).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - stickiness: [0, 5] Perseverance bonus added to the previously chosen spaceship.
    """
    alpha_1, alpha_2, beta_1, beta_2, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Initialize Q-values
    # Stage 1: Spaceship 0, Spaceship 1
    q_stage1 = np.zeros(2)
    # Stage 2: Planet 0 (Alien 0, 1), Planet 1 (Alien 0, 1)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        # Softmax Stage 1
        exp_q1 = np.exp(beta_1 * (q_net_1 - np.max(q_net_1)))
        prob_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[t] != -1:
            p_choice_1[t] = prob_1[action_1[t]]
            last_action_1 = action_1[t]
        else:
            p_choice_1[t] = 0.5
            last_action_1 = -1
            continue 

        # --- Stage 2 Decision ---
        current_state = state[t]
        if current_state == -1:
            p_choice_2[t] = 0.5
            continue
            
        q_net_2 = q_stage2[current_state]
        exp_q2 = np.exp(beta_2 * (q_net_2 - np.max(q_net_2)))
        prob_2 = exp_q2 / np.sum(exp_q2)
        
        if action_2[t] != -1:
            p_choice_2[t] = prob_2[action_2[t]]
        else:
            p_choice_2[t] = 0.5
            continue

        # --- Learning ---
        # Prediction Error 1: Difference between value of state reached and chosen spaceship
        pe_1 = q_stage2[current_state, action_2[t]] - q_stage1[action_1[t]]
        
        # Prediction Error 2: Difference between reward received and chosen alien
        pe_2 = reward[t] - q_stage2[current_state, action_2[t]]
        
        # Update Stage 1 using alpha_1
        # We propagate the full reward signal back (TD(1)-like logic)
        q_stage1[action_1[t]] += alpha_1 * (pe_1 + pe_2)
        
        # Update Stage 2 using alpha_2
        q_stage2[current_state, action_2[t]] += alpha_2 * pe_2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Eligibility Traces and Value Decay.
    
    This model combines 'forgetting' (decay of unchosen action values) with an eligibility 
    trace parameter (lambda). The decay helps the agent track non-stationary rewards by 
    reducing confidence in old estimates. The eligibility trace allows the agent to modulate 
    how strongly the final reward reinforces the first-stage choice, bridging the gap between 
    TD(0) and Monte Carlo updating.
    
    Parameters:
    - learning_rate: [0, 1] Shared learning rate for updates.
    - decay_rate: [0, 1] Rate at which unchosen Q-values decay toward 0.
    - lambda_eligibility: [0, 1] Strength of the eligibility trace (0=TD(0), 1=TD(1)).
    - beta: [0, 10] Shared inverse temperature (noise) for both stages.
    - stickiness: [0, 5] Perseverance bonus for Stage 1.
    """
    learning_rate, decay_rate, lambda_eligibility, beta, stickiness = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for t in range(n_trials):
        # Stage 1
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
        
        exp_q1 = np.exp(beta * (q_net_1 - np.max(q_net_1)))
        prob_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[t] != -1:
            p_choice_1[t] = prob_1[action_1[t]]
            last_action_1 = action_1[t]
        else:
            p_choice_1[t] = 0.5
            last_action_1 = -1
            continue

        # Stage 2
        current_state = state[t]
        if current_state == -1:
            p_choice_2[t] = 0.5
            continue
            
        q_net_2 = q_stage2[current_state]
        exp_q2 = np.exp(beta * (q_net_2 - np.max(q_net_2)))
        prob_2 = exp_q2 / np.sum(exp_q2)
        
        if action_2[t] != -1:
            p_choice_2[t] = prob_2[action_2[t]]
        else:
            p_choice_2[t] = 0.5
            continue
            
        # Updates
        pe_1 = q_stage2[current_state, action_2[t]] - q_stage1[action_1[t]]
        pe_2 = reward[t] - q_stage2[current_state, action_2[t]]
        
        # Stage 1 Update: TD(0) update + Eligibility Trace part
        q_stage1[action_1[t]] += learning_rate * pe_1 + learning_rate * lambda_eligibility * pe_2
        
        # Stage 2 Update
        q_stage2[current_state, action_2[t]] += learning_rate * pe_2
        
        # Decay Unchosen Values
        # Stage 1
        q_stage1[1 - action_1[t]] *= (1 - decay_rate)
        # Stage 2 (current planet)
        q_stage2[current_state, 1 - action_2[t]] *= (1 - decay_rate)
        # Stage 2 (other planet)
        q_stage2[1 - current_state, :] *= (1 - decay_rate)
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Learner with Lapse Rate (Epsilon-Greedy Mixture).
    
    This model assumes the participant generally follows a softmax policy based on 
    Model-Free Q-values, but has a fixed probability `epsilon` of making a random 
    choice (a 'lapse'). This mixture model is robust to occasional outliers or 
    errors (like unrewarded switches or input errors) that would otherwise severely 
    penalize the likelihood of a high-precision (high beta) model.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - stickiness: [0, 5] Perseverance bonus for Stage 1.
    - epsilon: [0, 1] Probability of a random choice (lapse rate).
    """
    learning_rate, beta_1, beta_2, stickiness, epsilon = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for t in range(n_trials):
        # Stage 1
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * (q_net_1 - np.max(q_net_1)))
        prob_1 = exp_q1 / np.sum(exp_q1)
        
        # Mixture: (1-eps)*Softmax + eps*Random
        prob_1 = (1 - epsilon) * prob_1 + epsilon * 0.5
        
        if action_1[t] != -1:
            p_choice_1[t] = prob_1[action_1[t]]
            last_action_1 = action_1[t]
        else:
            p_choice_1[t] = 0.5
            last_action_1 = -1
            continue
            
        # Stage 2
        current_state = state[t]
        if current_state == -1:
            p_choice_2[t] = 0.5
            continue
            
        q_net_2 = q_stage2[current_state]
        exp_q2 = np.exp(beta_2 * (q_net_2 - np.max(q_net_2)))
        prob_2 = exp_q2 / np.sum(exp_q2)
        
        # Mixture
        prob_2 = (1 - epsilon) * prob_2 + epsilon * 0.5
        
        if action_2[t] != -1:
            p_choice_2[t] = prob_2[action_2[t]]
        else:
            p_choice_2[t] = 0.5
            continue
            
        # Updates (Standard MF TD(1))
        pe_1 = q_stage2[current_state, action_2[t]] - q_stage1[action_1[t]]
        pe_2 = reward[t] - q_stage2[current_state, action_2[t]]
        
        q_stage1[action_1[t]] += learning_rate * (pe_1 + pe_2)
        q_stage2[current_state, action_2[t]] += learning_rate * pe_2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
