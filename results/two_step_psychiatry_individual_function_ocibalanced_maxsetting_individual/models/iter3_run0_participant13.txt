Here are three new cognitive models for the two-step task, designed based on the participant's behavior and previous model performance.

### Model 1: Hybrid Model with Outcome-Dependent Perseveration
This model integrates the "best so far" perseveration mechanism (Win-Stay, Lose-Shift/Stay) into the hybrid Model-Based / Model-Free framework. It tests whether the participant combines knowledge of the transition structure (Model-Based) with simple reinforcement learning (Model-Free), while also exhibiting distinct biases depending on whether the previous trial was rewarded.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model 1: Hybrid Model (MB/MF) with Outcome-Dependent Perseveration.
    
    Combines Model-Based (transition matrix) and Model-Free learning.
    Distinguishes perseveration bias based on whether the previous trial was rewarded.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for Q-values.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting between Model-Based (1) and Model-Free (0).
    pers_win: [-3, 3] - Perseveration bonus after a reward (Win-Stay).
    pers_loss: [-3, 3] - Perseveration bonus after no reward (Lose-Stay/Shift).
    """
    learning_rate, beta, w, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: 
    # Spaceship A (0) -> Planet X (0) w/ prob 0.7
    # Spaceship U (1) -> Planet Y (1) w/ prob 0.7
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    prev_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        # V(S') = max_a Q(S', a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        # Add Outcome-Dependent Perseveration
        if prev_a1 != -1:
            if prev_reward == 1:
                logits_1[prev_a1] += pers_win
            else:
                logits_1[prev_a1] += pers_loss
        
        # Softmax
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning (Model-Free Updates) ---
        # Update Stage 1 MF based on Stage 2 value (TD error)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 MF based on Reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_a1 = action_1[trial]
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Asymmetric Learning with Choice Kernel
This model hypothesizes that the participant's perseveration is not just a one-trial "stickiness" but a longer-term habit (Choice Kernel). Additionally, it allows for asymmetric learning from positive versus negative prediction errors, capturing potential risk-seeking or risk-averse behavior in value updates.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model 2: Asymmetric Learning Q-Learning with Choice Kernel.
    
    Model-Free algorithm with separate learning rates for positive and negative 
    prediction errors, combined with a decaying choice kernel (habit) instead of 
    simple 1-back perseveration.
    
    Parameters:
    alpha_pos: [0, 1] - Learning rate for positive prediction errors.
    alpha_neg: [0, 1] - Learning rate for negative prediction errors.
    beta: [0, 10] - Inverse temperature.
    decay_k: [0, 1] - Decay rate for the choice kernel (0 = no memory, 1 = full memory/no decay).
    weight_k: [0, 10] - Weight of the choice kernel in the decision.
    """
    alpha_pos, alpha_neg, beta, decay_k, weight_k = model_parameters
    n_trials = len(action_1)
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    # Choice kernel for stage 1
    choice_kernel = np.zeros(2)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Value + Habit (Kernel)
        logits_1 = beta * q_stage1_mf + weight_k * choice_kernel
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Choice Kernel
        # Decay existing traces
        choice_kernel = choice_kernel * (1 - decay_k)
        # Reinforce chosen action
        choice_kernel[action_1[trial]] += 1.0
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        lr_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += lr_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid TD($\lambda$) Model
This model combines the sophisticated Model-Based planning with the eligibility trace mechanism (TD-$\lambda$) from the "best so far" model. This allows the model to assign credit for rewards directly to the first-stage choice (via $\lambda$) while also maintaining the transition map structure (via $w$).

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model 3: Hybrid TD(lambda) Model.
    
    Combines Model-Based planning with a TD(lambda) Model-Free component.
    This allows the model to use eligibility traces for credit assignment 
    while also maintaining a map of the task structure.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    lambda_eligibility: [0, 1] - Eligibility trace decay parameter.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Weighting between Model-Based (1) and Model-Free (0).
    perseveration: [-3, 3] - Simple repetition bias (1-back).
    """
    learning_rate, lambda_eligibility, beta, w, perseveration = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # MB Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning with Eligibility Traces ---
        # TD error at stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # TD error at stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1 MF: direct update from stage 2 transition + eligibility trace of stage 2 reward
        # Q(s1, a1) += alpha * (delta1 + lambda * delta2)
        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + lambda_eligibility * delta_stage2)
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
