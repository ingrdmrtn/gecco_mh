Here are three new cognitive models for the two-step decision task, designed to capture different potential mechanisms underlying the participant's behavior.

### Model 1: Hybrid with Stage-Specific Learning Rates and Perseverance
This model extends the standard hybrid model by allowing for different learning rates in Stage 1 (spaceship choice) and Stage 2 (alien choice). This accounts for the possibility that the participant learns the direct bandit values (aliens) at a different speed than the second-order values (spaceships), while also maintaining a perseverance bonus to capture the observed stickiness in spaceship choice.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner with separate learning rates for Stage 1 and Stage 2, plus perseverance.
    
    Parameters:
    - lr_s1: [0, 1] Learning rate for Stage 1 (Spaceship values).
    - lr_s2: [0, 1] Learning rate for Stage 2 (Alien values).
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    - perseverance: [0, 5] Bonus added to the previously chosen spaceship.
    """
    lr_s1, lr_s2, beta, w, perseverance = model_parameters
    n_trials = len(action_1)

    # Initial transition matrix (fixed for this model)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf = np.zeros(2)          # Stage 1 Model-Free values
    q_stage2 = np.zeros((2, 2)) # Stage 2 values (Aliens)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value: Expected value of next stage
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2

        # Net Value
        q_net = w * q_mb + (1 - w) * q_mf

        # Perseverance Bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseverance

        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        s_idx = state[trial] # Actual planet reached

        # --- Stage 2 Decision ---
        # Softmax Choice 2
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # Stage 1 MF Update (TD Error 1)
        # Using chosen stage 2 value for SARSA-like update
        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += lr_s1 * delta1

        # Stage 2 Update (TD Error 2)
        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += lr_s2 * delta2

        # Direct reinforcement of Stage 1 choice by Stage 2 reward (Eligibility Trace logic)
        q_mf[a1] += lr_s1 * delta2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Dual Beta Hybrid Model
This model hypothesizes that the participant's level of exploration (or decision noise) differs between the high-level planning stage (choosing a spaceship) and the low-level consumption stage (choosing an alien). It uses two separate `beta` parameters.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner with separate inverse temperatures (beta) for Stage 1 and Stage 2.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for value learning.
    - beta_s1: [0, 10] Inverse temperature for Stage 1 (Spaceship) choice.
    - beta_s2: [0, 10] Inverse temperature for Stage 2 (Alien) choice.
    - w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta_s1, beta_s2, w = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2

        q_net = w * q_mb + (1 - w) * q_mf

        # Softmax with beta_s1
        exp_q1 = np.exp(beta_s1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        s_idx = state[trial]

        # --- Stage 2 Decision ---
        # Softmax with beta_s2
        exp_q2 = np.exp(beta_s2 * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta1

        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2
        
        # MF update from reward
        q_mf[a1] += learning_rate * delta2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Dynamic Transition Learning Hybrid
This model assumes the participant does not trust the transition probabilities to be fixed (0.7/0.3) but learns them online. If a series of "rare" transitions occurs, the participant updates their internal model of the spaceship-planet mapping, which affects the Model-Based value calculation.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner that updates the transition matrix based on experience.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for value learning.
    - lr_trans: [0, 1] Learning rate for transition matrix update.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting parameter.
    """
    learning_rate, lr_trans, beta, w = model_parameters
    n_trials = len(action_1)

    # Initialize with the true probabilities, but allow them to drift
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Use current estimate of transition matrix for MB calculation
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2

        q_net = w * q_mb + (1 - w) * q_mf

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        s_idx = state[trial]
        a1 = action_1[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Values ---
        a2 = action_2[trial]
        r = reward[trial]

        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta1

        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2
        
        q_mf[a1] += learning_rate * delta2

        # --- Learning Transitions ---
        # Update P(state | action_1) towards 1 for the observed state
        # and towards 0 for the unobserved state.
        
        # Update observed transition probability
        transition_matrix[a1, s_idx] += lr_trans * (1.0 - transition_matrix[a1, s_idx])
        
        # Update unobserved transition probability (complement)
        transition_matrix[a1, 1 - s_idx] = 1.0 - transition_matrix[a1, s_idx]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
