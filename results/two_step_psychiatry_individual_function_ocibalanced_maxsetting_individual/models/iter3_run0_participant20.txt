Here are three cognitive models based on the participant data and the two-step task structure.

### Model 1: Hybrid MB/MF with Counterfactual Stage 2 Learning
**Rationale:** The participant data shows successful learning of changing reward probabilities. In binary choice tasks, participants often employ counterfactual updating: if the chosen alien gives 0 coins (reward=0), they infer the other alien would likely have given 1 (reward=1), and vice versa. This model adds a `cf_weight` parameter to update the unchosen option in Stage 2, allowing for faster adaptation to the drifting probabilities.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Stickiness and Counterfactual Updating in Stage 2.
    
    Updates the unchosen option in the second stage based on the inverse of the 
    received reward, weighted by a counterfactual parameter.

    Parameters:
    - learning_rate: [0,1] Standard learning rate for chosen options.
    - beta_1: [0,10] Inverse temperature for Stage 1.
    - beta_2: [0,10] Inverse temperature for Stage 2.
    - w: [0,1] MB/MF weighting (1=MB, 0=MF).
    - stickiness: [0,10] Choice repetition bonus.
    - cf_weight: [0,1] Weight of counterfactual update (0 = none, 1 = full update).
    """
    learning_rate, beta_1, beta_2, w, stickiness, cf_weight = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Skip missing data
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # Stage 1 MF Update (TD(1))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # Stage 2 Update (Chosen)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Stage 2 Update (Unchosen / Counterfactual)
        unchosen_action_2 = 1 - action_2[trial]
        # Assume unchosen would have yielded (1 - reward)
        fictive_reward = 1 - reward[trial]
        delta_cf = fictive_reward - q_stage2_mf[state_idx, unchosen_action_2]
        q_stage2_mf[state_idx, unchosen_action_2] += learning_rate * cf_weight * delta_cf

    eps = 1e-10
    # Filter valid trials for likelihood calculation
    valid_idx = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_idx] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_idx] + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with Separate Stage Learning Rates
**Rationale:** The participant may learn the value of the spaceships (Stage 1) at a different rate than the value of the aliens (Stage 2). For instance, the transition structure is stable, but alien rewards drift. This model decouples the learning rates `alpha_1` and `alpha_2`. It retains `stickiness` because the participant data shows significant choice repetition (runs of choosing the same spaceship).

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with separate learning rates for Stage 1 and Stage 2,
    plus choice stickiness.

    Parameters:
    - alpha_1: [0,1] Learning rate for Stage 1 (Spaceships).
    - alpha_2: [0,1] Learning rate for Stage 2 (Aliens).
    - beta_1: [0,10] Inverse temperature for Stage 1.
    - beta_2: [0,10] Inverse temperature for Stage 2.
    - w: [0,1] MB/MF weighting.
    - stickiness: [0,10] Choice repetition bonus.
    """
    alpha_1, alpha_2, beta_1, beta_2, w, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness

        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 uses alpha_1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += alpha_1 * delta_stage1
        
        # Stage 2 uses alpha_2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += alpha_2 * delta_stage2
        

    eps = 1e-10
    valid_idx = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_idx] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_idx] + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with Dynamic Habit Strength
**Rationale:** The participant exhibits very long streaks of identical choices (e.g., trials 87-94, 145-151). Simple stickiness adds a constant bonus to the last choice. A "Habit" model tracks a separate habit strength variable that accumulates over repeated choices and decays when not chosen. This allows the model to capture the *momentum* of the participant's behavior more dynamically than a simple 1-back stickiness parameter.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with a dynamic Habit strength variable.
    
    Habit strength accumulates for the chosen action and decays for the unchosen 
    action, creating a 'momentum' effect for repeated choices.

    Parameters:
    - learning_rate: [0,1] Q-value learning rate.
    - beta_1: [0,10] Inverse temperature for Stage 1.
    - beta_2: [0,10] Inverse temperature for Stage 2.
    - w: [0,1] MB/MF weighting.
    - habit_lr: [0,1] Learning rate for habit strength accumulation/decay.
    - habit_str: [0,10] Weight of the habit strength in the decision logic.
    """
    learning_rate, beta_1, beta_2, w, habit_lr, habit_str = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Habit strength for spaceship 0 and 1. Initialized at 0.
    habit_values = np.zeros(2) 

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB, MF, and Habit
        # Note: Habit is added to logits essentially as an independent influence
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net + habit_str * habit_values
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Habit Strength
        # Chosen action habit moves toward 1, unchosen moves toward 0
        chosen = action_1[trial]
        unchosen = 1 - chosen
        habit_values[chosen] += habit_lr * (1 - habit_values[chosen])
        habit_values[unchosen] += habit_lr * (0 - habit_values[unchosen])

        state_idx = state[trial]

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    valid_idx = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_idx] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_idx] + eps)))
    return log_loss
```
