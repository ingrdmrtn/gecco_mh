Here are the three proposed cognitive models.

### Model 1: Hybrid MB/MF with Subjective Reward Valuation
This model hypothesizes that the participant does not value the "no reward" outcome (0 coins) as neutral (value 0) but assigns it a subjective value, `reward_bias`. If `reward_bias` is positive, it explains the tendency to persist (stay) even after receiving 0 coins, as the outcome is still perceived as partially rewarding.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF with Subjective Reward Valuation (Reward Bias).
    
    This model assumes the participant perceives the 'no reward' outcome (0 coins)
    not necessarily as 0 value, but as a subjective value `reward_bias`. 
    This can explain persistence in choosing non-rewarding options if the 
    bias is positive (treating 0 as partial success) or negative (loss aversion).
    
    Parameters:
    learning_rate: [0, 1] Rate of Q-value updating.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2.
    w: [0, 1] Weighting between MB (1) and MF (0).
    stickiness: [0, 5] Choice stickiness for Stage 1.
    reward_bias: [0, 1] Subjective value assigned to receiving 0 coins.
    """
    learning_rate, beta_1, beta_2, w, stickiness, reward_bias = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 states, 2 actions
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        # Softmax for Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        qs_current_state = q_stage2_mf[state_idx]
        
        # Softmax for Stage 2
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning / Updating ---
        # Calculate Effective Reward based on Subjective Bias
        # If reward is 1, use 1. If 0, use reward_bias.
        r_actual = reward[trial]
        r_effective = r_actual * 1.0 + (1.0 - r_actual) * reward_bias

        # Stage 1 MF Update (TD(0) style using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # Stage 2 MF Update (using Effective Reward)
        delta_stage2 = r_effective - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with Stochastic Planning
Standard model-based agents assume they will act optimally in the second stage (using `max` Q-value). This model assumes the agent simulates the future using a specific "simulation" inverse temperature (`beta_sim`). This decouples the agent's planning confidence from their execution noise (`beta_2`), allowing for "soft" planning where the agent considers the expected value over a stochastic policy rather than just the maximum value.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF with Stochastic Planning (Simulation Beta).
    
    Standard MB agents assume optimal future choices (max Q) during planning.
    This model assumes the agent simulates the second stage using a specific
    inverse temperature `beta_sim`. This allows the agent to be pessimistic 
    (low beta_sim, assuming randomness) or optimistic (high beta_sim) about 
    their own future performance, decoupling planning from execution (`beta_2`).
    
    Parameters:
    learning_rate: [0, 1] Rate of Q-value updating.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2 execution.
    w: [0, 1] Weighting between MB (1) and MF (0).
    stickiness: [0, 5] Choice stickiness for Stage 1.
    beta_sim: [0, 10] Inverse temperature used for value estimation during planning.
    """
    learning_rate, beta_1, beta_2, w, stickiness, beta_sim = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Expected_Value(Q_stage2 | beta_sim)
        # Calculate Expected Value for each state under beta_sim policy
        values_stage2 = np.zeros(2)
        for s in range(2):
            qs = q_stage2_mf[s]
            # Probabilities under simulation policy
            exp_qs_sim = np.exp(beta_sim * qs)
            probs_sim = exp_qs_sim / np.sum(exp_qs_sim)
            # Expected value
            values_stage2[s] = np.sum(probs_sim * qs)
            
        q_stage1_mb = transition_matrix @ values_stage2

        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        qs_current_state = q_stage2_mf[state_idx]
        
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with Subjective Transition Beliefs
This model relaxes the assumption that the participant perfectly understands the transition probabilities (0.7/0.3). Instead, it fits a parameter `p_common` representing the participant's belief about the probability of common transitions. This can capture behaviors where the participant perceives the structure as more deterministic (`p_common` close to 1) or more random (`p_common` close to 0.5) than it actually is.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF with Subjective Transition Beliefs.
    
    The participant may not believe the transition probabilities are exactly 0.7/0.3.
    This model fits a subjective `p_common` parameter representing the agent's 
    belief about the probability of the common transition.
    
    Parameters:
    learning_rate: [0, 1] Rate of Q-value updating.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2.
    w: [0, 1] Weighting between MB (1) and MF (0).
    stickiness: [0, 5] Choice stickiness for Stage 1.
    p_common: [0, 1] Perceived probability of common transitions (0.7 is truth).
    """
    learning_rate, beta_1, beta_2, w, stickiness, p_common = model_parameters
    n_trials = len(action_1)

    # Construct subjective transition matrix
    # T[0,0] = p_common (A->X), T[0,1] = 1-p (A->Y)
    # T[1,0] = 1-p (U->X), T[1,1] = p (U->Y)
    transition_matrix = np.array([[p_common, 1.0 - p_common], 
                                  [1.0 - p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Use subjective transition matrix for MB calculation
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        qs_current_state = q_stage2_mf[state_idx]
        
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
