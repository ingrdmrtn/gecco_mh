Here are the three proposed cognitive models.

### Model 1: Hybrid Learner with Dynamic Transition Learning
This model assumes the participant does not start with perfect knowledge of the spaceship-planet transition probabilities (70/30) but learns them over time. The Model-Based component uses this evolving transition matrix to calculate values.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Dynamic Transition Learning.
    
    The agent learns the transition matrix (Spaceship -> Planet) over time
    using a transition learning rate, rather than assuming fixed probabilities.
    
    Parameters:
    lr_v: [0, 1] Learning rate for Q-values (value updating).
    lr_t: [0, 1] Learning rate for transition probabilities.
    beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    lam: [0, 1] Eligibility trace (lambda).
    """
    lr_v, lr_t, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)

    # Initialize transition matrix to flat priors (0.5 probability)
    # Row: Action 1 (Spaceship), Col: State (Planet)
    transition_matrix = np.ones((2, 2)) * 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 states (planets), 2 actions (aliens)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Use current estimate of transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Softmax
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Transition ---
        chosen_spaceship = action_1[trial]
        arrived_planet = state[trial]
        
        # Update Transition Matrix (State Prediction Error)
        # Increase prob of the observed transition, decrease prob of the other
        transition_matrix[chosen_spaceship, arrived_planet] += lr_t * (1 - transition_matrix[chosen_spaceship, arrived_planet])
        transition_matrix[chosen_spaceship, 1 - arrived_planet] = 1 - transition_matrix[chosen_spaceship, arrived_planet]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[arrived_planet])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Value Updates ---
        # Stage 1 Prediction Error (using MF value of state arrived at)
        delta_stage1 = q_stage2_mf[arrived_planet, action_2[trial]] - q_stage1_mf[chosen_spaceship]
        
        # Stage 2 Prediction Error (Reward)
        delta_stage2 = reward[trial] - q_stage2_mf[arrived_planet, action_2[trial]]

        # Update Stage 1 MF (with eligibility trace)
        q_stage1_mf[chosen_spaceship] += lr_v * (delta_stage1 + lam * delta_stage2)

        # Update Stage 2 MF
        q_stage2_mf[arrived_planet, action_2[trial]] += lr_v * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Dual Stickiness
This model incorporates "stickiness" (perseveration) parameters for *both* the spaceship choice (Stage 1) and the alien choice (Stage 2). This captures the tendency to repeat motor actions or choices regardless of reward history.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Dual Stickiness.
    
    Adds perseveration (stickiness) biases to both Stage 1 and Stage 2 choices.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] MB weight.
    lam: [0, 1] Eligibility trace.
    stick_1: [0, 10] Stickiness parameter for Stage 1 (Spaceship).
    stick_2: [0, 10] Stickiness parameter for Stage 2 (Alien).
    """
    lr, beta_1, beta_2, w, lam, stick_1, stick_2 = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_action_2 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to logits
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stick_1
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Record choice
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        if last_action_2 != -1:
            logits_2[last_action_2] += stick_2
            
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        last_action_2 = action_2[trial]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        q_stage1_mf[action_1[trial]] += lr * (delta_stage1 + lam * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Decay and Lambda
This model introduces a memory decay mechanism where all Q-values slowly decay towards zero on every trial ("forgetting"), combined with eligibility traces (`lam`) to maintain the two-step credit assignment. This helps the agent adapt to the drifting reward probabilities of the aliens.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Value Decay and Eligibility Traces.
    
    Q-values decay over time (forgetting), allowing the agent to discard 
    outdated information about drifting rewards. Includes eligibility traces.
    
    Parameters:
    lr: [0, 1] Learning rate for chosen options.
    decay: [0, 1] Decay rate applied to all Q-values per trial.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] MB weight.
    lam: [0, 1] Eligibility trace.
    """
    lr, decay, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Decay ---
        # Apply decay to ALL values before updating the chosen one
        # This simulates forgetting and keeps values from growing indefinitely
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)

        # --- Updates ---
        # Calculate errors based on decayed values
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        # Standard update adds to the chosen action (counteracting decay for the chosen one)
        q_stage1_mf[action_1[trial]] += lr * (delta_stage1 + lam * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
