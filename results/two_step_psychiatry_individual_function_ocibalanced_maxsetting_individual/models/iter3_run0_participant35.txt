Here are three new cognitive models for the two-step decision task, designed to capture different psychological mechanisms such as stage-specific learning speeds, outcome-dependent perseverance, and model-based planning with memory decay.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Separate Learning Rates for Stage 1 and Stage 2.
    Hypothesis: The participant learns the values of spaceships (distal choice) and 
    aliens (proximal choice) at different rates.
    
    Parameters:
    - lr_1: [0, 1] Learning rate for Stage 1 (spaceship choice) updates.
    - lr_2: [0, 1] Learning rate for Stage 2 (alien choice) updates.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - lam: [0, 1] Eligibility trace parameter connecting Stage 2 reward to Stage 1 choice.
    - pers: [0, 5] Perseverance bonus for Stage 1 repetition.
    """
    lr_1, lr_2, beta_1, beta_2, lam, pers = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # Calculate prediction errors
        # delta_1: Error between Stage 1 value and Stage 2 value
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        # delta_2: Error between Stage 2 value and received reward
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        # Update Stage 1 Q-values using lr_1
        # Includes eligibility trace (lam) from Stage 2 error
        q_stage1[action_1[trial]] += lr_1 * delta_1 + lr_1 * lam * delta_2
        
        # Update Stage 2 Q-values using lr_2
        q_stage2[state_idx, action_2[trial]] += lr_2 * delta_2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Outcome-Dependent Perseverance.
    Hypothesis: "Stickiness" (perseverance) is not constant but depends on the previous outcome.
    Distinguishes between 'Win-Stay' (pers_pos) and 'Lose-Stay' (pers_neg).
    
    Parameters:
    - learning_rate: [0, 1] Shared learning rate.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - lam: [0, 1] Eligibility trace.
    - pers_pos: [0, 5] Perseverance bonus after a reward (Win-Stay).
    - pers_neg: [0, 5] Perseverance bonus after no reward (Lose-Stay).
    """
    learning_rate, beta_1, beta_2, lam, pers_pos, pers_neg = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0 
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        q_net_1 = q_stage1.copy()
        # Apply different perseverance bonus based on previous reward
        if last_action_1 != -1:
            if last_reward == 1:
                q_net_1[last_action_1] += pers_pos
            else:
                q_net_1[last_action_1] += pers_neg
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        q_stage1[action_1[trial]] += learning_rate * delta_1 + learning_rate * lam * delta_2
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with Value Decay.
    Hypothesis: The participant uses a Model-Based strategy (planning using the transition matrix)
    but suffers from forgetting (decay) of the specific alien values (Stage 2 Q-values).
    No eligibility trace (lam) is used, relying on the MB component for structure.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - decay: [0, 1] Decay rate for unchosen Stage 2 actions (forgetting).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight (0 = Pure MF, 1 = Pure MB).
    - pers: [0, 5] Perseverance bonus.
    """
    learning_rate, decay, beta_1, beta_2, w, pers = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2 = np.zeros((2, 2)) # Used for both MF values and MB planning
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation: T * max(Q2)
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += pers
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # MF Stage 1 update (Standard TD(0), no lambda)
        delta_1 = q_stage2[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_1
        
        # Stage 2 update
        delta_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        # Decay unchosen Stage 2 actions (Forgetting)
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_action_2] *= (1.0 - decay)
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
