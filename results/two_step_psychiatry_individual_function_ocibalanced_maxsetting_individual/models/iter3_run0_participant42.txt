Here are the 3 proposed cognitive models.

### Model 1: Hybrid Agent with Choice Kernel (Habit)
This model extends the concept of "stickiness" (repeating the last choice) to a "habit" or "choice kernel" that accumulates over time. Instead of just looking at $t-1$, the agent builds a trace of their choice history. This allows the model to capture the participant's tendency to stick with a spaceship for blocks of trials (inertia) even after a single bad outcome, or to switch only after the habit strength has decayed sufficiently.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Agent with Choice Kernel (Habit).
    
    Incorporates a 'Choice Kernel' that tracks the history of stage 1 choices.
    Unlike simple stickiness (which only considers the previous trial), the kernel 
    accumulates over time, representing a habit strength that decays slowly.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weight for Model-Based values (0=MF, 1=MB).
    - lambda_coef: [0, 1] Eligibility trace for Stage 1 MF update.
    - ck_decay: [0, 1] Decay rate for the Choice Kernel (1 = full decay/instant, 0 = no decay).
    - ck_weight: [0, 10] Weight of the Choice Kernel in the decision (Habit strength).
    """
    learning_rate, beta, w, lambda_coef, ck_decay, ck_weight = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    # Choice Kernel for Stage 1 actions (initialized to 0)
    choice_kernel = np.zeros(2)

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Choice Kernel influence to logits
        logits = beta * q_net + ck_weight * choice_kernel
        
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Choice Kernel: K(t+1) = (1-decay)*K(t) + decay*Indicator(choice)
        # This keeps the kernel bounded between 0 and 1.
        choice_kernel *= (1 - ck_decay)
        choice_kernel[action_1[trial]] += ck_decay
        
        state_idx = state[trial]

        # Stage 2 Policy
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2 - np.max(beta * q_s2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Learning
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + lambda_coef * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Agent with Subjective Loss Aversion
The participant data shows rapid switching after some losses. This model hypothesizes that the participant perceives a "0 coin" outcome not as a neutral zero reward, but as a negative "loss". This "Loss Aversion" or subjective penalty drives stronger negative prediction errors than standard RL, facilitating quicker abandonment of unrewarded options.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Agent with Subjective Loss Aversion.
    
    This model assumes the participant perceives a lack of reward (0 coins) 
    not as neutral, but as a loss. The 'loss_penalty' parameter defines the 
    magnitude of this negative utility, leading to stronger avoidance learning 
    after unrewarded trials compared to standard RL.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - lambda_coef: [0, 1] Stage 1 eligibility trace.
    - stickiness: [0, 10] Choice repetition bonus.
    - loss_penalty: [0, 5] Subjective penalty added when reward is 0.
    """
    learning_rate, beta, w, lambda_coef, stickiness, loss_penalty = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # Stage 2 Policy
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2 - np.max(beta * q_s2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Reward Transformation
        # If reward is 1, subjective val is 1.
        # If reward is 0, subjective val is -loss_penalty.
        r_eff = 1.0 if reward[trial] == 1 else -loss_penalty

        # Learning
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = r_eff - q_stage2_mf[state_idx, action_2[trial]]

        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + lambda_coef * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Agent with Selective Forgetting (Unchosen Decay)
Unlike global forgetting (which decays all Q-values), this model decays only the Q-values of *unchosen* actions. This reflects a mechanism where the agent becomes increasingly uncertain about options they haven't explored recently, causing their values to revert towards zero (or a baseline). This helps in non-stationary environments where the "road not taken" might change value over time.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Agent with Selective Forgetting (Decay of Unchosen).
    
    Instead of global forgetting, this model only decays the Q-values of 
    unchosen actions (both in Stage 1 and the unchosen alien in Stage 2).
    This simulates uncertainty growing for options that are not observed,
    reverting them towards 0.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - lambda_coef: [0, 1] Stage 1 eligibility trace.
    - stickiness: [0, 10] Choice repetition bonus.
    - unchosen_decay: [0, 1] Decay rate for unchosen options.
    """
    learning_rate, beta, w, lambda_coef, stickiness, unchosen_decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # Stage 2 Policy
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2 - np.max(beta * q_s2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Decay Unchosen Actions
        # Stage 1 unchosen
        unchosen_a1 = 1 - action_1[trial]
        q_stage1_mf[unchosen_a1] *= (1 - unchosen_decay)
        
        # Stage 2 unchosen (in current state only)
        unchosen_a2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_a2] *= (1 - unchosen_decay)

        # Update Chosen Actions
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]

        q_stage1_mf[action_1[trial]] += learning_rate * (delta_stage1 + lambda_coef * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
