Here are three new cognitive models for the two-step task, designed to capture specific behavioral nuances like perseverance (stickiness), outcome-dependent repetition, and eligibility trace modulation.

### Model 1: Hybrid with Asymmetric Learning and Perseverance
This model extends the "best model so far" (which used asymmetric learning rates for positive and negative outcomes) by adding a **perseverance (stickiness)** parameter. Perseverance captures the tendency to repeat the previous choice regardless of reward, a phenomenon commonly observed in this task that is distinct from value-based learning.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner with Asymmetric Learning Rates and Choice Perseverance.
    
    Combines the optimism/pessimism bias (alpha_pos/neg) with the tendency to repeat 
    previous choices (stickiness). This accounts for both valence-dependent updating 
    and habitual repetition.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate when Reward = 1.
    - alpha_neg: [0, 1] Learning rate when Reward = 0.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - stickiness: [-5, 5] Bonus added to the logit of the previously chosen action.
    """
    alpha_pos, alpha_neg, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        # Apply perseverance bonus to the previously chosen spaceship
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        if reward[trial] == 1:
            current_lr = alpha_pos
        else:
            current_lr = alpha_neg

        # Update Stage 1 MF (Direct reinforcement from outcome, effectively lambda=1)
        q_stage1_mf[action_1[trial]] += current_lr * (reward[trial] - q_stage1_mf[action_1[trial]])

        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid with Outcome-Dependent Perseverance
This model investigates whether the "stickiness" is modulated by the previous outcome. Instead of a single perseverance parameter, it separates stickiness into **Win-Stay** (`stick_win`) and **Loss-Stay** (`stick_loss`). This allows the model to capture behavior where a participant might reflexively repeat a winning action (motor reinforcement) but switch after a loss, independent of the actual value estimation.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner with Outcome-Dependent Perseverance (Win-Stay/Loss-Stay).
    
    Differentiates perseverance based on the previous trial's outcome. 
    Allows for different tendencies to repeat a choice after a reward vs. an omission,
    separating 'motor reinforcement' from value learning.
    
    Parameters:
    - lr: [0, 1] Learning rate (symmetric).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - stick_win: [-5, 5] Perseverance bonus after Reward = 1.
    - stick_loss: [-5, 5] Perseverance bonus after Reward = 0.
    """
    lr, beta, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        # Apply outcome-dependent perseverance
        if last_action_1 != -1:
            if last_reward == 1:
                logits[last_action_1] += stick_win
            else:
                logits[last_action_1] += stick_loss
                
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Standard symmetric learning for values
        q_stage1_mf[action_1[trial]] += lr * (reward[trial] - q_stage1_mf[action_1[trial]])
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Hybrid with Eligibility Trace (Lambda)
This model combines the asymmetric learning rates of the best model with an **eligibility trace parameter (`lam`)**. While the "best model" effectively assumed `lam=1` (updating stage 1 directly from the reward), this model allows `lam` to vary between 0 (TD(0), updating stage 1 based on stage 2 value) and 1 (TD(1), updating stage 1 based on final reward). This captures the degree to which the participant relies on the immediate transition versus the final outcome for the first-stage update.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid learner with Asymmetric Learning Rates and Eligibility Trace (Lambda).
    
    Combines asymmetric updating (optimism/pessimism) with an eligibility trace 
    parameter (lambda). This controls whether the Stage 1 MF value is updated 
    based on the Stage 2 value prediction (TD(0)) or the final reward (TD(1)).
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate when Reward = 1.
    - alpha_neg: [0, 1] Learning rate when Reward = 0.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - lam: [0, 1] Eligibility trace decay (0=TD(0), 1=TD(1)).
    """
    alpha_pos, alpha_neg, beta, w, lam = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        if reward[trial] == 1:
            current_lr = alpha_pos
        else:
            current_lr = alpha_neg

        # TD(lambda) Update Logic
        # Delta 1: Prediction error at stage 1 (based on stage 2 value - TD(0) component)
        delta_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        
        # Delta 2: Prediction error at stage 2 (based on reward)
        delta_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Update Stage 1 MF: Combines immediate error and propagated error via lambda
        q_stage1_mf[action_1[trial]] += current_lr * (delta_1 + lam * delta_2)
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
