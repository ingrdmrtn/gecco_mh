Here are the three cognitive models based on the participant data and task structure.

### Model 1: Dynamic Transition Learning
This model hypothesizes that the participant does not assume the transition probabilities (Spaceship $\to$ Planet) are fixed, but continuously learns them. This allows the "Model-Based" component to adapt if the participant perceives "rare" transitions as a change in the environment structure.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Dynamic Transition Learning.
    
    The agent updates the transition matrix (Spaceship -> Planet) trial-by-trial
    based on observed transitions, rather than using a fixed matrix.
    
    Parameters:
    learning_rate: [0,1] - Value learning rate (MF).
    lr_trans: [0,1] - Learning rate for transition probabilities.
    beta: [0,10] - Inverse temperature for softmax.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    stickiness: [0,5] - Choice stickiness for Stage 1.
    """
    learning_rate, lr_trans, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
  
    # Initialize transition matrix with priors (0.7 common, 0.3 rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Alien
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness to Stage 1 choice
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Update Transition Matrix ---
        # Update P(State | Action_1)
        # T[a, s] <- T[a, s] + alpha_trans * (1 - T[a, s])
        chosen_a1 = action_1[trial]
        observed_s = state[trial]
        
        transition_matrix[chosen_a1, observed_s] += lr_trans * (1.0 - transition_matrix[chosen_a1, observed_s])
        transition_matrix[chosen_a1, 1 - observed_s] = 1.0 - transition_matrix[chosen_a1, observed_s]
        
        # Handle missing data (timeout) in Stage 2
        if action_2[trial] == -1:
            prev_action_1 = chosen_a1
            continue

        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Value Updates ---
        # Stage 1 MF Update (SARSA-like using Q_stage2)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        # Stage 2 MF Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_action_1 = chosen_a1

    eps = 1e-10
    # Sum log likelihoods (excluding missed trials for Stage 2)
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
```

### Model 2: Value Decay (Forgetting)
This model incorporates a forgetting mechanism. Since the reward probabilities of the aliens drift slowly over time, older estimates become unreliable. This model decays the Q-values of unchosen aliens towards zero, encouraging the agent to re-explore or rely more on recent outcomes.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Passive Decay (Forgetting).
    
    Q-values of unchosen actions in Stage 2 decay over time, 
    simulating forgetting or uncertainty increase in a drifting environment.
    
    Parameters:
    learning_rate: [0,1] - Value learning rate for chosen options.
    decay_rate: [0,1] - Decay rate for unchosen options (1 = full forget, 0 = no forget).
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    stickiness: [0,5] - Stage 1 choice stickiness.
    """
    learning_rate, decay_rate, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            continue
            
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Value Updates ---
        # Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # Stage 2 MF (Update chosen, Decay unchosen)
        chosen_a2 = action_2[trial]
        unchosen_a2 = 1 - chosen_a2
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # Decay the unchosen alien's value
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
```

### Model 3: Dual-Stage Stickiness
Standard models typically only model perseveration (stickiness) for the first decision (Spaceship). This model adds a separate stickiness parameter for the second decision (Alien), hypothesizing that the participant might also exhibit motor perseveration or a bias to repeat the same alien choice within a planet.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Stickiness in both Stage 1 and Stage 2.
    
    Captures perseveration not just for the spaceship choice, but also for 
    the alien choice within each planet.
    
    Parameters:
    learning_rate: [0,1] - Value learning rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    stick_s1: [0,5] - Stickiness for Spaceship choice (Stage 1).
    stick_s2: [0,5] - Stickiness for Alien choice (Stage 2).
    """
    learning_rate, beta, w, stick_s1, stick_s2 = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    # Track previous alien choice per state (Planet 0, Planet 1)
    prev_action_2_per_state = [-1, -1] 

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick_s1
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            continue
            
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        q_stage2_net = q_stage2_mf[state_idx].copy()
        
        # Apply Stage 2 stickiness if there is a history for this planet
        last_a2 = prev_action_2_per_state[state_idx]
        if last_a2 != -1:
            q_stage2_net[last_a2] += stick_s2
            
        exp_q2 = np.exp(beta * q_stage2_net)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_action_1 = action_1[trial]
        prev_action_2_per_state[state_idx] = action_2[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
```
