Here are three new cognitive models for the two-step decision-making task, designed to explore mechanisms like asymmetric learning, eligibility traces, and choice history kernels, building upon the insights from previous models.

### Model 1: Hybrid Model with Asymmetric Stage 2 Learning
This model combines Model-Based and Model-Free learning but introduces **asymmetric learning rates** specifically for the second stage. The participant may weigh positive outcomes (finding gold) differently than negative outcomes (no gold) when updating the value of the aliens, while using a standard update for the spaceships.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Asymmetric Stage 2 Learning and Perseveration.
    
    Combines Model-Based (MB) planning with Model-Free (MF) learning.
    Stage 2 value updates distinguish between positive (reward=1) and negative (reward=0) 
    outcomes via separate learning rates (alpha_2_pos, alpha_2_neg).
    
    Parameters:
    - alpha_1: [0, 1] MF Learning rate for Stage 1 (Spaceships).
    - alpha_2_pos: [0, 1] MF Learning rate for Stage 2 (Aliens) when reward is 1.
    - alpha_2_neg: [0, 1] MF Learning rate for Stage 2 (Aliens) when reward is 0.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - w_mb: [0, 1] Weight of Model-Based control (0 = pure MF, 1 = pure MB).
    - perseveration: [0, 10] Stickiness bonus for repeating the last Stage 1 action.
    """
    alpha_1, alpha_2_pos, alpha_2_neg, beta_1, beta_2, w_mb, perseveration = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: Action 0 -> State 0 (0.7), Action 1 -> State 1 (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_1 = np.zeros(2)
    q_mf_2 = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])
        
        # --- Stage 1 Decision ---
        # Model-Based Value: Transition Matrix * Max(Q_Stage2)
        max_q_stage2 = np.max(q_mf_2, axis=1)
        q_mb_1 = transition_matrix @ max_q_stage2
        
        # Hybrid Value Integration
        q_net_1 = w_mb * q_mb_1 + (1 - w_mb) * q_mf_1
        
        logits_1 = beta_1 * q_net_1
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Decision ---
        logits_2 = beta_2 * q_mf_2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # Stage 1 MF Update (TD(0) using Stage 2 Q-value)
        delta_1 = q_mf_2[s2, a2] - q_mf_1[a1]
        q_mf_1[a1] += alpha_1 * delta_1
        
        # Stage 2 MF Update (Asymmetric)
        delta_2 = r - q_mf_2[s2, a2]
        if r > 0:
            q_mf_2[s2, a2] += alpha_2_pos * delta_2
        else:
            q_mf_2[s2, a2] += alpha_2_neg * delta_2
            
        last_action_1 = a1

    return -log_likelihood
```

### Model 2: Model-Free with Separate Alphas and Eligibility Trace
This model extends the "best model so far" by adding an **eligibility trace (`lambda`)**. While the previous best model used a simple TD(0) update (Stage 1 value tracks Stage 2 value), this model allows the final reward outcome to directly influence the Stage 1 choice via the trace, bridging the temporal gap more effectively.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Separate Alphas, Eligibility Traces, and Perseveration.
    
    Uses separate learning rates for Stage 1 and Stage 2. Adds an eligibility trace 
    (lambda) to allow the reward prediction error at Stage 2 to propagate directly 
    back to update Stage 1 values (TD(lambda)).
    
    Parameters:
    - alpha_1: [0, 1] Learning rate for Stage 1.
    - alpha_2: [0, 1] Learning rate for Stage 2.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - lambda_eligibility: [0, 1] Eligibility trace decay parameter (0=TD(0), 1=Monte Carlo).
    - perseveration: [0, 10] Stickiness bonus for Stage 1.
    """
    alpha_1, alpha_2, beta_1, beta_2, lambda_eligibility, perseveration = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])
        
        # --- Stage 1 Choice ---
        logits_1 = beta_1 * q_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Choice ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # Stage 1 Prediction Error: Q2(s2,a2) - Q1(a1)
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        
        # Stage 2 Prediction Error: r - Q2(s2,a2)
        delta_2 = r - q_stage2[s2, a2]
        
        # Update Stage 2
        q_stage2[s2, a2] += alpha_2 * delta_2
        
        # Update Stage 1
        # Q1 is updated by its own error (delta_1) plus the scaled error from Stage 2
        q_stage1[a1] += alpha_1 * (delta_1 + lambda_eligibility * delta_2)
        
        last_action_1 = a1
        
    return -log_likelihood
```

### Model 3: Model-Free with Separate Alphas and Choice Kernel
The participant data shows long "streaks" of choosing the same spaceship (e.g., trials 82-119). A simple 1-step perseveration parameter might not capture this inertia as well as a **Choice Kernel**, which accumulates a history of past choices and decays slowly, effectively modeling habit formation.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Separate Alphas and Choice Kernel.
    
    Replaces simple perseveration with a Choice Kernel (CK) for Stage 1.
    The CK accumulates a history of choices, allowing the model to capture 
    stronger habits or 'streaks' observed in the participant data.
    
    Parameters:
    - alpha_1: [0, 1] Learning rate for Stage 1.
    - alpha_2: [0, 1] Learning rate for Stage 2.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - ck_decay: [0, 1] Decay rate of the choice kernel (0 = reset, 1 = no decay).
    - ck_weight: [0, 10] Weight of the choice kernel in the decision logic.
    """
    alpha_1, alpha_2, beta_1, beta_2, ck_decay, ck_weight = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    choice_kernel = np.zeros(2) # Tracks history of Stage 1 choices
    
    log_likelihood = 0.0
    eps = 1e-10
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])
        
        # --- Stage 1 Choice ---
        # Decision combines Value (Q) and Habit (Choice Kernel)
        logits_1 = beta_1 * q_stage1 + ck_weight * choice_kernel
        
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Choice ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # Update Choice Kernel: Decay old value, add new choice
        # CK(t+1) = decay * CK(t) + (1-decay) * Indicator(choice)
        choice_kernel *= ck_decay
        choice_kernel[a1] += (1.0 - ck_decay)
        
        # Update Q-values (Model-Free with separate alphas)
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += alpha_1 * delta_1
        
        delta_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha_2 * delta_2
        
    return -log_likelihood
```
