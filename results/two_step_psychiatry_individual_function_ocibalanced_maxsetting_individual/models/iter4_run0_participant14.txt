Here are the three proposed cognitive models. They introduce specific mechanisms (Stage-Specific Learning Rates, Memory Decay, and Stage-Specific Noise) combined with Choice Stickiness, building upon the insights from the participant's data and previous model performance.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Stage-Specific Learning Rates and Stickiness.
    
    This model hypothesizes that the participant learns at different rates for the 
    high-level spaceship choice (Stage 1) compared to the specific alien choice (Stage 2),
    while also exhibiting choice perseveration (stickiness).

    Parameters:
    - lr_1: [0, 1] Learning rate for Stage 1 (Spaceships).
    - lr_2: [0, 1] Learning rate for Stage 2 (Aliens).
    - beta: [0, 10] Inverse temperature (exploration/exploitation trade-off).
    - w: [0, 1] Weight of Model-Based system (0=MF, 1=MB).
    - lam: [0, 1] Eligibility trace (how much Stage 2 error updates Stage 1 value).
    - stickiness: [0, 5] Choice perseveration bonus for Stage 1.
    """
    lr_1, lr_2, beta, w, lam, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix as per task structure
    # State 0 (Planet X) is common for Action 0 (Spaceship A)
    # State 1 (Planet Y) is common for Action 1 (Spaceship U)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf = np.zeros(2)         # Stage 1 Model-Free values
    q2 = np.zeros((2, 2))      # Stage 2 values (Aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        # Handle missing data
        if a1 == -1:
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value: Expected max value of next stage
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        
        # Net Value: Weighted sum of MB and MF
        q_net = w * q_mb + (1 - w) * q_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 1 Prediction Error (SARSA-style)
        delta_1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += lr_1 * delta_1
        
        # Stage 2 Prediction Error
        delta_2 = r - q2[s, a2]
        q2[s, a2] += lr_2 * delta_2
        
        # Eligibility Trace Update for Stage 1
        # Updates Stage 1 MF value based on Stage 2 outcome
        q_mf[a1] += lr_1 * lam * delta_2
        
        last_action_1 = a1

    mask = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss


def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Value Decay and Stickiness.
    
    This model incorporates a forgetting mechanism (decay) for unchosen aliens,
    accounting for the non-stationary nature of reward probabilities, 
    alongside choice stickiness.

    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - lam: [0, 1] Eligibility trace.
    - stickiness: [0, 5] Choice perseveration bonus.
    - decay: [0, 1] Decay rate for unchosen Stage 2 values (forgetting).
    """
    lr, beta, w, lam, stickiness, decay = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf = np.zeros(2)
    q2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        if a1 == -1:
            continue

        # --- Stage 1 Policy ---
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        q_net = w * q_mb + (1 - w) * q_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += lr * delta_1
        
        delta_2 = r - q2[s, a2]
        q2[s, a2] += lr * delta_2
        
        q_mf[a1] += lr * lam * delta_2
        
        # --- Decay ---
        # Decay the value of the unchosen alien on the current planet
        # This pushes the value towards 0, simulating forgetting
        unchosen_a2 = 1 - a2
        q2[s, unchosen_a2] *= (1 - decay)
        
        last_action_1 = a1

    mask = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
