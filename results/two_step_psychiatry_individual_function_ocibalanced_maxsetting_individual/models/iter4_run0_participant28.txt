Here are three new cognitive models for the two-step decision task, designed to capture specific mechanisms like asymmetric learning, choice perseverance, and counterfactual reasoning, all built upon the successful hybrid architecture with value decay.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Asymmetric Learning Rates, Decay, and Eligibility Traces.
    
    This model distinguishes between learning from positive rewards (wins) and 
    negative outcomes (losses/omissions). It posits that the participant may 
    update their value estimates differently when they receive a coin vs. when 
    they do not. It retains value decay to handle drifting probabilities.
    
    Parameters:
    lr_pos: [0, 1] Learning rate for rewarded trials (R=1).
    lr_neg: [0, 1] Learning rate for unrewarded trials (R=0).
    decay: [0, 1] Decay rate applied to all Q-values per trial.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] Model-based weight (0=MF, 1=MB).
    lam: [0, 1] Eligibility trace connecting outcome to Stage 1 choice.
    """
    lr_pos, lr_neg, decay, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Decay ---
        # Forgetting: values decay towards 0
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)
        
        # --- Learning ---
        # Asymmetric learning: choose rate based on outcome
        current_lr = lr_pos if reward[trial] == 1 else lr_neg
        
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        q_stage1_mf[action_1[trial]] += current_lr * (delta_stage1 + lam * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Choice Stickiness, Decay, and Eligibility Traces.
    
    Adds a perseverance bonus (stickiness) to the previously chosen Stage 1 action.
    This captures the tendency to repeat choices (motor or cognitive inertia) 
    independent of the value of those choices.
    
    Parameters:
    lr: [0, 1] Learning rate.
    decay: [0, 1] Decay rate applied to all Q-values per trial.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] Model-based weight.
    lam: [0, 1] Eligibility trace.
    stick: [0, 10] Choice stickiness bonus added to the logits of the repeated action.
    """
    lr, decay, beta_1, beta_2, w, lam, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits and add stickiness to previous choice
        logits_1 = beta_1 * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stick
            
        # Softmax with stability
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) 
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Decay ---
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)
        
        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        q_stage1_mf[action_1[trial]] += lr * (delta_stage1 + lam * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Counterfactual Updating, Decay, and Eligibility Traces.
    
    This model assumes the participant engages in "fictitious play" or counterfactual
    reasoning in Stage 2. If they receive a reward from one alien, they update the 
    unchosen alien as if it would have yielded no reward (and vice versa), assuming 
    anticorrelated reward structures.
    
    Parameters:
    lr: [0, 1] Learning rate for the chosen option.
    lr_cf: [0, 1] Counterfactual learning rate for the unchosen option.
    decay: [0, 1] Decay rate.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] Model-based weight.
    lam: [0, 1] Eligibility trace.
    """
    lr, lr_cf, decay, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Decay ---
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)
        
        # --- Learning ---
        # Update chosen actions
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        q_stage1_mf[action_1[trial]] += lr * (delta_stage1 + lam * delta_stage2)
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Update unchosen action in Stage 2 (Counterfactual)
        # Assume anticorrelation: if reward=1, cf_reward=0; if reward=0, cf_reward=1
        unchosen_a2 = 1 - action_2[trial]
        cf_reward = 1 - reward[trial]
        delta_cf = cf_reward - q_stage2_mf[state_idx, unchosen_a2]
        q_stage2_mf[state_idx, unchosen_a2] += lr_cf * delta_cf
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
