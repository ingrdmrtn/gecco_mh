Here are the three proposed cognitive models.

### Model 1: Subjective Loss Perception Model
This model modifies the reward function. Instead of treating a "no reward" outcome (0 coins) as a neutral zero-value update, the agent perceives it as a subjective loss. This captures the psychological asymmetry where the absence of an expected reward is experienced as a punishment, potentially driving faster unlearning or switching than standard extinction.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Subjective Loss Perception Model.
    
    This model assumes that the participant perceives the absence of reward (0 coins)
    not as a neutral event, but as a negative outcome (loss). 
    The effective reward is R if R=1, else -loss_magnitude.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for value updates.
    beta: [0, 10] - Inverse temperature for softmax choice.
    w: [0, 1] - Weight for Model-Based values (0 = pure MF, 1 = pure MB).
    loss_magnitude: [0, 5] - The subjective negative value assigned to a 0 reward outcome.
    """
    learning_rate, beta, w, loss_magnitude = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action

    for trial in range(n_trials):
        # Skip missing trials
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Decision ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Transition
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        # Softmax Choice 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # Calculate Effective Reward
        # If reward is 1, use 1. If reward is 0, use -loss_magnitude.
        # If reward is -1 (rare penalty), keep as -1.
        r_eff = reward[trial]
        if r_eff == 0:
            r_eff = -loss_magnitude
        
        # Stage 1 MF Update (TD(0) style using Stage 2 Q-value)
        chosen_a1 = action_1[trial]
        chosen_a2 = action_2[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        # Stage 2 MF Update
        delta_stage2 = r_eff - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Global Forgetting Model
This model implements a "leaky integrator" mechanism where all Q-values (both Model-Free and implicitly Model-Based via Stage 2 values) decay towards zero on every trial. This reflects limited memory capacity or a belief in a highly volatile environment where past information rapidly becomes irrelevant, distinct from specific unchosen action decay.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Global Forgetting Model.
    
    In this model, all action values decay passively towards zero on every trial,
    representing a global forgetting process or a belief in environmental volatility.
    This affects both chosen and unchosen actions.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for prediction error updates.
    beta: [0, 10] - Inverse temperature for softmax choice.
    w: [0, 1] - Weight for Model-Based values.
    forgetting_rate: [0, 1] - The rate at which all Q-values decay per trial (0=no decay).
    """
    learning_rate, beta, w, forgetting_rate = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Global Forgetting (Decay) ---
        # Apply decay to all Q-values before decision and learning
        q_stage1_mf *= (1 - forgetting_rate)
        q_stage2_mf *= (1 - forgetting_rate)

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        chosen_a1 = action_1[trial]
        chosen_a2 = action_2[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Outcome-Dependent Exploration Model
This model hypothesizes that the participant's level of exploration (randomness) is modulated by the most recent outcome. A "Win" (reward) induces a higher `beta` (exploitation/stay), while a "Loss" (no reward) induces a lower `beta` (exploration/randomness). This captures a dynamic "Win-Exploit, Lose-Explore" strategy.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Outcome-Dependent Exploration Model.
    
    This model allows the inverse temperature (beta) to vary based on the 
    outcome of the previous trial. A win may lead to more deterministic 
    behavior (higher beta), while a loss may increase randomness (lower beta).
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta_win: [0, 10] - Inverse temperature after a rewarded trial.
    beta_loss: [0, 10] - Inverse temperature after an unrewarded trial.
    w: [0, 1] - Weight for Model-Based values.
    """
    learning_rate, beta_win, beta_loss, w = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize previous reward (assume neutral/loss start or 0)
    prev_reward = 0

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # Determine current beta based on previous reward
        if prev_reward == 1:
            current_beta = beta_win
        else:
            current_beta = beta_loss

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        # Using the same outcome-dependent beta for consistency in exploration state
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        chosen_a1 = action_1[trial]
        chosen_a2 = action_2[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # Update previous reward for next trial
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
