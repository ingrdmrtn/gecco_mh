Here are three new cognitive models for the two-step task, designed to capture different psychological mechanisms that might explain the participant's behavior.

### Model 1: Hybrid Agent with Split Betas and Reward-Dependent Stickiness
This model posits that the participant's "randomness" or exploration-exploitation trade-off differs between the two stages of the task. Stage 1 (choosing a spaceship) is a planning step involving model-based reasoning, while Stage 2 (choosing an alien) is a simpler bandit task. The model uses separate inverse temperature parameters (`beta_1` and `beta_2`) for each stage, combined with the reward-dependent stickiness that proved effective in previous iterations.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Agent with Split Betas and Reward-Dependent Stickiness.
    
    Separates the exploration/exploitation trade-off (inverse temperature) for 
    Stage 1 (planning/transition) and Stage 2 (bandit/terminal).
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-values.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - w: [0, 1] Mixing weight (1 = pure MB, 0 = pure MF).
    - stick_win: [-5, 5] Stickiness bonus after a rewarded trial.
    - stick_loss: [-5, 5] Stickiness bonus after an unrewarded trial.
    """
    learning_rate, beta_1, beta_2, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()

        # Apply reward-dependent stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                logits[last_action_1] += stick_win
            else:
                logits[last_action_1] += stick_loss
            
        # Use beta_1 for Stage 1
        exp_q1 = np.exp(beta_1 * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        # Use beta_2 for Stage 2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Update Stage 1 MF values (SARSA-like update)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        # Update Stage 2 MF values (Reward prediction error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Agent with Side Bias and Reward-Dependent Stickiness
This model tests the hypothesis that the participant has a fixed, static preference for one spaceship over the other (e.g., a bias towards the right/left button or spaceship 1), independent of the values learned during the task. This `side_bias` acts as a constant offset to the decision logits in Stage 1, distinct from the dynamic `stickiness` which depends on the previous trial's action and outcome.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Agent with Side Bias and Reward-Dependent Stickiness.
    
    Includes a static bias parameter for the Stage 1 choice (e.g., preference for 
    Spaceship 1), in addition to dynamic stickiness.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - stick_win: [-5, 5] Stickiness bonus after a rewarded trial.
    - stick_loss: [-5, 5] Stickiness bonus after an unrewarded trial.
    - side_bias: [-5, 5] Static bias towards Action 1 (Spaceship 1). Positive values favor 1.
    """
    learning_rate, beta, w, stick_win, stick_loss, side_bias = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) 

    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()

        # Apply static side bias (bias towards action 1)
        logits[1] += side_bias

        # Apply dynamic reward-dependent stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                logits[last_action_1] += stick_win
            else:
                logits[last_action_1] += stick_loss
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Agent with Asymmetric Learning and Reward-Dependent Stickiness
This model investigates whether the participant learns differently from positive prediction errors (outcomes better than expected) versus negative prediction errors (outcomes worse than expected). This asymmetry is a common feature in psychiatric computational modeling. It is combined with reward-dependent stickiness to ensure that perseveration effects are accounted for while isolating the learning asymmetry.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Agent with Asymmetric Learning Rates and Reward-Dependent Stickiness.
    
    Differentiates learning from positive prediction errors (better than expected)
    and negative prediction errors (worse than expected).
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors.
    - alpha_neg: [0, 1] Learning rate for negative prediction errors.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - stick_win: [-5, 5] Stickiness bonus after a rewarded trial.
    - stick_loss: [-5, 5] Stickiness bonus after an unrewarded trial.
    """
    alpha_pos, alpha_neg, beta, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()

        if last_action_1 != -1:
            if last_reward == 1:
                logits[last_action_1] += stick_win
            else:
                logits[last_action_1] += stick_loss
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning with Asymmetric Rates ---
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        if delta_stage1 > 0:
            q_stage1_mf[action_1[trial]] += alpha_pos * delta_stage1
        else:
            q_stage1_mf[action_1[trial]] += alpha_neg * delta_stage1

        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        if delta_stage2 > 0:
            q_stage2_mf[state_idx, action_2[trial]] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[state_idx, action_2[trial]] += alpha_neg * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
