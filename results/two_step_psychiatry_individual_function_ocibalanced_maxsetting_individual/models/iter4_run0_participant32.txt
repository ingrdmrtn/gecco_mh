Here are three new cognitive models for the two-step task.

### Model 1: Hybrid Model with Separate Betas and Perseveration
This model combines Model-Based (MB) and Model-Free (MF) control, similar to the classic Daw et al. (2011) hybrid model, but introduces **separate inverse temperature parameters (`beta1`, `beta2`)** for the two stages. This allows the model to capture different levels of exploration/exploitation or noise in the spaceship choice versus the alien choice. It also includes a perseveration bonus for the first stage.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Betas and Perseveration.
    
    Combines Model-Based and Model-Free values using a weighting parameter 'w'.
    Crucially, it uses separate inverse temperatures for Stage 1 and Stage 2,
    allowing for distinct noise levels in spaceship vs. alien choices.
    
    Parameters:
    - lr: [0, 1] Learning rate for Q-value updates.
    - beta1: [0, 10] Inverse temperature for Stage 1 (Spaceship).
    - beta2: [0, 10] Inverse temperature for Stage 2 (Alien).
    - w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    - perseveration: [0, 5] Stickiness bonus for repeating the previous Stage 1 choice.
    """
    lr, beta1, beta2, w, perseveration = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Expected value of next state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Weighted mix of MB and MF
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta1 * q_net_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        logits_2 = beta2 * q_stage2_mf[state_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Stage 1 MF Update (TD0): Updates towards value of state reached
        pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * pe_1
        
        # Stage 2 MF Update: Updates towards reward
        pe_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * pe_2
            
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Based with Reward-Dependent Perseveration
This model assumes the participant is purely Model-Based (relying on the transition matrix) but exhibits a "Win-Stay, Lose-Shift" heuristic implemented via **asymmetric perseveration**. Instead of a single perseveration parameter, it uses `pers_win` (applied if the previous trial was rewarded) and `pers_loss` (applied if unrewarded). This allows the model to capture dynamic stickiness based on recent outcomes.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based with Reward-Dependent Perseveration.
    
    Assumes choices are driven by the transition structure (MB).
    Replaces standard perseveration with outcome-dependent stickiness:
    participants may stick to a choice more after a win than after a loss.
    
    Parameters:
    - lr: [0, 1] Learning rate for Stage 2 values.
    - beta1: [0, 10] Inverse temperature for Stage 1.
    - beta2: [0, 10] Inverse temperature for Stage 2.
    - pers_win: [0, 5] Stickiness bonus after a Reward=1 trial.
    - pers_loss: [0, 5] Stickiness bonus after a Reward=0 trial.
    """
    lr, beta1, beta2, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        logits_1 = beta1 * q_stage1_mb
        
        # Apply reward-dependent perseveration
        if last_action_1 != -1:
            if last_reward == 1:
                logits_1[last_action_1] += pers_win
            else:
                logits_1[last_action_1] += pers_loss
        
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        logits_2 = beta2 * q_stage2_mf[state_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        pe = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * pe
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Pure Model-Based with Value Decay
This model addresses the non-stationary nature of the reward probabilities (drifting values). It is a Pure Model-Based model that includes a **decay parameter**. While chosen options are updated via prediction error, unchosen options (and options in the unvisited state) slowly decay towards a neutral value (0.5). This helps the model "forget" outdated information about aliens it hasn't visited in a while.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based with Passive Value Decay.
    
    Incorporates a decay mechanism to handle drifting reward probabilities.
    Unchosen Q-values decay towards 0.5 (chance) on each trial.
    
    Parameters:
    - lr: [0, 1] Learning rate for chosen option.
    - beta1: [0, 10] Inverse temperature for Stage 1.
    - beta2: [0, 10] Inverse temperature for Stage 2.
    - decay: [0, 1] Decay rate for unchosen options (0 = no decay, 1 = instant reset).
    - perseveration: [0, 5] Stickiness bonus for Stage 1.
    """
    lr, beta1, beta2, decay, perseveration = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize values at 0.5 (center of 0-1 reward range)
    q_stage2_mf = np.ones((2, 2)) * 0.5 
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        logits_1 = beta1 * q_stage1_mb
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        logits_2 = beta2 * q_stage2_mf[state_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Update chosen alien
        pe = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * pe
        
        # Decay unchosen alien in current state
        unchosen_action = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action] = (1 - decay) * q_stage2_mf[state_idx, unchosen_action] + decay * 0.5
        
        # Decay both aliens in the other (unvisited) state
        other_state = 1 - state_idx
        q_stage2_mf[other_state, :] = (1 - decay) * q_stage2_mf[other_state, :] + decay * 0.5
            
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
