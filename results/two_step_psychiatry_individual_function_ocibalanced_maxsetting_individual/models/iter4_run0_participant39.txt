Here are the three proposed cognitive models.

### Cognitive Model 1: Hybrid Learner with Uncertainty-Based Exploration Bonus
This model hypothesizes that the participant is driven not just by reward maximization and habit, but also by a curiosity or uncertainty reduction drive. It adds an "exploration bonus" to the value of options that have not been selected recently. This helps explain the block-switching behavior (switching spaceships after a long period) as the uncertainty about the unchosen spaceship grows over time.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model-Based / Model-Free learner with Uncertainty-Based Exploration.
    
    Adds an exploration bonus to options based on the time since they were last chosen.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta_1: Inverse temperature for stage 1 [0,10]
    - beta_2: Inverse temperature for stage 2 [0,10]
    - w: Mixing weight (0=MF, 1=MB) [0,1]
    - stick: Choice stickiness [0,5]
    - phi: Exploration bonus weight [0,5]
    """
    lr, beta_1, beta_2, w, stick, phi = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 planets, 2 aliens
    
    # Track time since last selection for exploration bonus
    # Initialize high to encourage initial exploration
    time_since_choice_1 = np.zeros(2) 
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # Update time counters
        time_since_choice_1 += 1
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Stickiness
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
            
        # Add Exploration Bonus (Uncertainty)
        # Bonus increases as time_since_choice increases
        q_net += phi * time_since_choice_1
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        log_loss -= np.log(probs_1[a1] + eps)
        
        # Reset timer for chosen action
        time_since_choice_1[a1] = 0
        prev_action_1 = a1
        
        # --- Stage 2 Policy ---
        a2 = action_2[trial]
        if a2 == -1:
            continue
            
        state_idx = state[trial]
        qs_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        r = reward[trial]
        
        # Update Stage 2 (TD-0)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Update Stage 1 (TD-1 / SARSA-style update using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
    return log_loss
```

### Cognitive Model 2: Hybrid Learner with Counterfactual Stage 2 Updating
This model posits that the participant uses counterfactual reasoning in the second stage. If they choose Alien A and receive a reward (1), they assume Alien B would have yielded no reward (0), and vice versa. This effectively couples the learning of the two aliens on a planet, potentially speeding up learning or creating specific biases if the anticorrelation assumption is wrong.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Counterfactual Updating in Stage 2.
    
    Updates the unchosen alien's value based on the inverse of the observed reward.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta_1: Inverse temperature for stage 1 [0,10]
    - beta_2: Inverse temperature for stage 2 [0,10]
    - w: Mixing weight (0=MF, 1=MB) [0,1]
    - stick: Choice stickiness [0,5]
    - cf_weight: Weight of counterfactual update (0=none, 1=full) [0,1]
    """
    lr, beta_1, beta_2, w, stick, cf_weight = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        log_loss -= np.log(probs_1[a1] + eps)
        prev_action_1 = a1
        
        # --- Stage 2 Policy ---
        a2 = action_2[trial]
        if a2 == -1:
            continue
            
        state_idx = state[trial]
        qs_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        r = reward[trial]
        
        # Update Chosen Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Update Unchosen Stage 2 (Counterfactual)
        # Assume unchosen would have yielded (1-r)
        unchosen_a2 = 1 - a2
        cf_reward = 1.0 - r
        delta_cf = cf_reward - q_stage2_mf[state_idx, unchosen_a2]
        q_stage2_mf[state_idx, unchosen_a2] += lr * cf_weight * delta_cf
        
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
    return log_loss
```

### Cognitive Model 3: Hybrid Learner with Outcome-Dependent Stickiness
Standard models assume a single "stickiness" parameter regardless of the previous outcome. This model splits stickiness into "Win-Stay" (stickiness after a reward) and "Lose-Stay" (stickiness after no reward). This allows the model to capture if the participant is particularly prone to repeating choices after success versus perseverating after failure.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Outcome-Dependent Stickiness.
    
    Differentiates stickiness based on whether the previous trial was rewarded.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta_1: Inverse temperature for stage 1 [0,10]
    - beta_2: Inverse temperature for stage 2 [0,10]
    - w: Mixing weight (0=MF, 1=MB) [0,1]
    - stick_win: Stickiness after a rewarded trial [0,5]
    - stick_loss: Stickiness after an unrewarded trial [0,5]
    """
    lr, beta_1, beta_2, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    prev_reward = -1 # Initialize as no reward context
    
    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Outcome-Dependent Stickiness
        if prev_action_1 != -1:
            if prev_reward == 1:
                q_net[prev_action_1] += stick_win
            else:
                q_net[prev_action_1] += stick_loss
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        log_loss -= np.log(probs_1[a1] + eps)
        prev_action_1 = a1
        
        # --- Stage 2 Policy ---
        a2 = action_2[trial]
        if a2 == -1:
            prev_reward = 0 # Assume no reward if trial invalid
            continue
            
        state_idx = state[trial]
        qs_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        r = reward[trial]
        prev_reward = r
        
        # Update Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
    return log_loss
```
