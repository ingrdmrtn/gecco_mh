Here are three new cognitive models based on the participant data and the feedback provided.

### Model 1: Model-Free Q-Learning with Passive Decay
This model modifies the standard Model-Free Q-learning by introducing a **decay parameter**. While the chosen action is updated via prediction errors, the *unchosen* action's Q-value decays toward 0 (or a neutral state). This captures a "forgetting" process where the participant becomes less certain about options they haven't tried in a while, potentially encouraging re-exploration or explaining shifts in preference after long streaks.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Eligibility Traces and Passive Decay.
    
    The unchosen action in Stage 1 decays towards 0, simulating forgetting.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    lambda_param: [0,1] - Eligibility trace decay.
    decay_rate: [0,1] - Rate at which unchosen Q-values decay to 0.
    """
    learning_rate, beta_1, beta_2, lambda_param, decay_rate = model_parameters
    n_trials = len(action_1)
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        # Pure Model-Free choice based on learned Q-values
        exp_q1 = np.exp(beta_1 * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of the observed action
        if np.isnan(action_1[trial]):
            p_choice_1[trial] = 0.5
            continue
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if np.isnan(action_2[trial]):
            p_choice_2[trial] = 0.5
            continue
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Prediction Error Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Decay the unchosen action
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        # Prediction Error Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update for Stage 1 (TD-Lambda)
        q_stage1_mf[a1] += learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Based Learning with Stickiness
Previous attempts used Hybrid models (mixing MB and MF). This model tests the hypothesis that the participant is **purely Model-Based** (using the known transition structure to calculate Stage 1 values) but exhibits **choice stickiness**. This removes the Model-Free Stage 1 learning entirely, assuming the participant relies solely on the transition matrix and the learned values of the aliens, biased only by their previous choice.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learning with Choice Stickiness.
    
    Assumes the agent calculates Stage 1 values entirely from the transition 
    matrix and Stage 2 values (no MF learning at Stage 1), plus a stickiness bias.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Stage 2 Q-values.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    stickiness: [0,10] - Bonus added to the previously chosen action.
    """
    learning_rate, beta_1, beta_2, stickiness = model_parameters
    n_trials = len(action_1)
  
    # Known transition matrix: A->X (0.7), U->Y (0.7)
    # Row 0: Space A (to X, to Y), Row 1: Space U (to X, to Y)
    # Note: State 0 is X, State 1 is Y.
    # A(0) -> X(0) is 0.7, A(0) -> Y(1) is 0.3
    # U(1) -> X(0) is 0.3, U(1) -> Y(1) is 0.7
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Only Stage 2 values are learned via RPE
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        # Calculate Model-Based values: V(s1) = T * max(Q(s2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Max value of each planet
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        logits_1 = beta_1 * q_stage1_mb
        
        # Apply Stickiness
        if prev_a1 != -1:
            logits_1[prev_a1] += stickiness

        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if np.isnan(action_1[trial]):
            p_choice_1[trial] = 0.5
            prev_a1 = -1
            continue
            
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        prev_a1 = a1
        
        state_idx = int(state[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if np.isnan(action_2[trial]):
            p_choice_2[trial] = 0.5
            continue
            
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Updates ---
        # Only update Stage 2 values based on reward
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # No Stage 1 MF update, No Lambda

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Free with Outcome-Dependent Stickiness
The participant data shows streaks of choices (e.g., trials 1-2, 16-18) that seem sensitive to rewards. This model refines the generic "stickiness" parameter into two separate components: **Win-Stay** (stickiness after a reward) and **Lose-Stay/Switch** (stickiness after no reward). This allows the model to capture if the participant perseverates differently depending on whether the previous trial was successful.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Outcome-Dependent Stickiness.
    
    Splits the stickiness parameter into 'sticky_win' (applied if previous 
    trial was rewarded) and 'sticky_lose' (applied if unrewarded).
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    lambda_param: [0,1] - Eligibility trace decay.
    sticky_win: [0,10] - Stickiness bonus after a Reward=1.
    sticky_lose: [0,10] - Stickiness bonus after a Reward=0.
    """
    learning_rate, beta_1, beta_2, lambda_param, sticky_win, sticky_lose = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_reward = -1

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        logits_1 = beta_1 * q_stage1_mf
        
        # Outcome-Dependent Stickiness
        if prev_a1 != -1:
            if prev_reward == 1:
                logits_1[prev_a1] += sticky_win
            else:
                logits_1[prev_a1] += sticky_lose
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if np.isnan(action_1[trial]):
            p_choice_1[trial] = 0.5
            prev_a1 = -1
            continue
            
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        prev_a1 = a1
        
        state_idx = int(state[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if np.isnan(action_2[trial]):
            p_choice_2[trial] = 0.5
            continue
            
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        prev_reward = r

        # --- Updates ---
        # TD(Lambda) Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
