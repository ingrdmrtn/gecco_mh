def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free RL with Asymmetric Learning Rates, Eligibility Traces, and Perseveration.
    
    This model assumes the participant is purely Model-Free (no planning) but learns 
    differently from positive versus negative prediction errors (valence-based learning).
    It includes an eligibility trace to propagate Stage 2 outcomes to Stage 1 values, 
    and a perseveration bonus for repeated Stage 1 choices.

    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors (better than expected).
    - alpha_neg: [0, 1] Learning rate for negative prediction errors (worse than expected).
    - beta_1: [0, 10] Inverse temperature (softmax randomness) for Stage 1.
    - beta_2: [0, 10] Inverse temperature (softmax randomness) for Stage 2.
    - lambda_eligibility: [0, 1] Eligibility trace decay parameter (controls credit assignment to Stage 1).
    - perseveration: [0, 10] Stickiness bonus added to the log-odds of repeating the previous Stage 1 choice.
    """
    alpha_pos, alpha_neg, beta_1, beta_2, lambda_eligibility, perseveration = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2)) # State x Action
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])

        # --- Stage 1 Policy ---
        logits_1 = beta_1 * q_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
        
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Learning ---
        # Prediction error at Stage 1 (State 1 -> State 2)
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        
        # Prediction error at Stage 2 (State 2 -> Reward)
        delta_2 = r - q_stage2[s2, a2]
        
        # Update Stage 2 Values (Asymmetric)
        eff_alpha_2 = alpha_pos if delta_2 > 0 else alpha_neg
        q_stage2[s2, a2] += eff_alpha_2 * delta_2
        
        # Update Stage 1 Values (Asymmetric with Eligibility Trace)
        # The eligibility trace allows delta_2 to affect Stage 1 immediately
        total_delta_1 = delta_1 + lambda_eligibility * delta_2
        eff_alpha_1 = alpha_pos if total_delta_1 > 0 else alpha_neg
        q_stage1[a1] += eff_alpha_1 * total_delta_1
        
        last_action_1 = a1

    return -log_likelihood

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MF/MB with Separate Stage Alphas and Eligibility Trace.
    
    This model combines Model-Based (MB) planning with a Model-Free (MF) system. 
    Crucially, it uses separate learning rates for Stage 1 and Stage 2 (allowing for 
    different plasticity at different task stages) and uses an eligibility trace 
    for the MF component to improve credit assignment.
    
    Parameters:
    - alpha_1: [0, 1] Learning rate for Stage 1 MF values.
    - alpha_2: [0, 1] Learning rate for Stage 2 values (used by both MF and MB).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w_mb: [0, 1] Weight of Model-Based control (0 = Pure MF, 1 = Pure MB).
    - lambda_eligibility: [0, 1] Eligibility trace parameter for the MF system.
    """
    alpha_1, alpha_2, beta_1, beta_2, w_mb, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    
    q_mf_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    log_likelihood = 0.0
    eps = 1e-10
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])

        # --- Model-Based Valuation ---
        # MB computes Stage 1 values using the transition matrix and max Stage 2 values
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        # --- Integrated Stage 1 Values ---
        q_net_stage1 = w_mb * q_mb_stage1 + (1 - w_mb) * q_mf_stage1
        
        # --- Stage 1 Policy ---
        logits_1 = beta_1 * q_net_stage1
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Learning ---
        # Stage 2 Update
        delta_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha_2 * delta_2
        
        # Stage 1 MF Update (TD(lambda))
        delta_1 = q_stage2[s2, a2] - q_mf_stage1[a1]
        q_mf_stage1[a1] += alpha_1 * (delta_1 + lambda_eligibility * delta_2)
        
    return -log_likelihood

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MF/MB with Asymmetric Learning and Perseveration.
    
    A hybrid agent that employs both Model-Based planning and Model-Free learning.
    It utilizes global asymmetric learning rates (different sensitivity to wins vs losses)
    across both stages, and includes perseveration (stickiness) at Stage 1.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors.
    - alpha_neg: [0, 1] Learning rate for negative prediction errors.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w_mb: [0, 1] Weight of Model-Based control.
    - perseveration: [0, 10] Stickiness bonus for Stage 1.
    """
    alpha_pos, alpha_neg, beta_1, beta_2, w_mb, perseveration = model_parameters
    n_trials = len(action_1)
    
    q_mf_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1
    
    for t in range(n_trials):
        a1 = int(action_1[t])
        s2 = int(state[t])
        a2 = int(action_2[t])
        r = int(reward[t])

        # --- Model-Based Valuation ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        # --- Integrated Stage 1 Values ---
        q_net_stage1 = w_mb * q_mb_stage1 + (1 - w_mb) * q_mf_stage1
        
        # --- Stage 1 Policy ---
        logits_1 = beta_1 * q_net_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration
            
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        logits_2 = beta_2 * q_stage2[s2]
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        log_likelihood += np.log(probs_2[a2] + eps)
        
        # --- Learning ---
        # Stage 2 Update (Asymmetric)
        delta_2 = r - q_stage2[s2, a2]
        eff_alpha_2 = alpha_pos if delta_2 > 0 else alpha_neg
        q_stage2[s2, a2] += eff_alpha_2 * delta_2
        
        # Stage 1 MF Update (Asymmetric, TD(0))
        # Note: Without lambda, MF relies on cached values from previous trials
        delta_1 = q_stage2[s2, a2] - q_mf_stage1[a1]
        eff_alpha_1 = alpha_pos if delta_1 > 0 else alpha_neg
        q_mf_stage1[a1] += eff_alpha_1 * delta_1
        
        last_action_1 = a1
        
    return -log_likelihood
