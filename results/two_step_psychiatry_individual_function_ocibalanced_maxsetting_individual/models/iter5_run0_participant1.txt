Here are the three proposed cognitive models.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Transition Hybrid Model.

    A hybrid Model-Based/Model-Free model where the participant actively learns 
    the transition probabilities (Spaceship -> Planet) instead of assuming a 
    fixed 70/30 structure.

    Parameters:
    learning_rate: [0, 1] - Learning rate for Q-value updates (Stage 1 MF and Stage 2).
    lr_trans: [0, 1] - Learning rate for updating the transition matrix.
    beta: [0, 10] - Inverse temperature for softmax choice.
    w: [0, 1] - Mixing weight (0 = pure MF, 1 = pure MB).
    stickiness: [0, 5] - Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, lr_trans, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Initialize transition matrix (rows=action1, cols=state)
    # Start with standard prior but allow deviation
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_stage1 = np.zeros(2)
    q_mf_stage2 = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        a1 = action_1[trial]
        s_idx = state[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # --- Stage 1 Choice ---
        # Model-Based Value: Expected max value of next stage using learned transition
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2

        # Net Value
        q_net_stage1 = w * q_mb_stage1 + (1 - w) * q_mf_stage1
        
        # Apply Stickiness
        if last_action_1 != -1:
            q_net_stage1[last_action_1] += stickiness

        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Choice ---
        # Standard MF choice at stage 2
        exp_q2 = np.exp(beta * q_mf_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if a2 != -1:
            log_loss -= np.log(probs_2[a2] + eps)
            
            # --- Updates ---
            if r != -1: # Valid reward
                # 1. Update Transition Matrix
                # Move probability towards the observed transition (a1 -> s_idx)
                # T[a1, s_idx] increases, T[a1, other] decreases
                transition_matrix[a1, s_idx] += lr_trans * (1 - transition_matrix[a1, s_idx])
                transition_matrix[a1, 1 - s_idx] = 1.0 - transition_matrix[a1, s_idx]

                # 2. Update Stage 2 Values (MF)
                delta_2 = r - q_mf_stage2[s_idx, a2]
                q_mf_stage2[s_idx, a2] += learning_rate * delta_2
                
                # 3. Update Stage 1 Values (MF - TD(1))
                q_mf_stage1[a1] += learning_rate * (r - q_mf_stage1[a1])

        last_action_1 = a1

    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Eligibility Traces (TD-lambda) and Stickiness.

    Combines Model-Based planning with a Model-Free system that uses eligibility 
    traces (lambda). This allows the Model-Free component to vary continuously 
    between TD(0) (update from state value) and TD(1) (update from reward),
    modulated by a mixing weight and choice stickiness.

    Parameters:
    learning_rate: [0, 1] - Learning rate for Q-values.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight between MB and MF.
    lam: [0, 1] - Eligibility trace decay parameter (lambda). 0=TD(0), 1=TD(1).
    stickiness: [0, 5] - Choice perseverance bonus.
    """
    learning_rate, beta, w, lam, stickiness = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_stage1 = np.zeros(2)
    q_mf_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        a1 = action_1[trial]
        s_idx = state[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_mb_stage1 + (1 - w) * q_mf_stage1
        
        if last_action_1 != -1:
            q_net_stage1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if a2 != -1:
            log_loss -= np.log(probs_2[a2] + eps)
            
            if r != -1:
                # --- Updates ---
                # Stage 2 Prediction Error (RPE)
                delta_2 = r - q_mf_stage2[s_idx, a2]
                
                # Stage 1 Prediction Error (TD error)
                # Note: For choice a1, the value of next state is Q(s_idx, a2)
                delta_1 = q_mf_stage2[s_idx, a2] - q_mf_stage1[a1]
                
                # Update Stage 2
                q_mf_stage2[s_idx, a2] += learning_rate * delta_2
                
                # Update Stage 1 with Eligibility Trace
                # Q(s1,a1) += alpha * (Delta1 + lambda * Delta2)
                q_mf_stage1[a1] += learning_rate * (delta_1 + lam * delta_2)
                
        last_action_1 = a1

    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Choice Trace Stickiness.

    Replaces the simple 1-back stickiness with a "Choice Trace" mechanism.
    A trace of past Stage 1 choices is maintained and decays over time. 
    This captures longer-term perseveration or "runs" of choices.

    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight (0=MF, 1=MB).
    trace_decay: [0, 1] - Decay rate of the choice trace (0 = resets every trial, 1 = no decay).
    stick_weight: [0, 5] - Weight of the choice trace influence on value.
    """
    learning_rate, beta, w, trace_decay, stick_weight = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_stage1 = np.zeros(2)
    q_mf_stage2 = np.zeros((2, 2))
    
    # Choice trace vector for the two spaceships
    choice_trace = np.zeros(2)
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        a1 = action_1[trial]
        s_idx = state[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_mb_stage1 + (1 - w) * q_mf_stage1
        
        # Add Choice Trace Stickiness
        q_net_stage1 += stick_weight * choice_trace
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Update Choice Trace
        # Decay existing trace
        choice_trace *= trace_decay
        # Increment trace for chosen action
        choice_trace[a1] += 1.0

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if a2 != -1:
            log_loss -= np.log(probs_2[a2] + eps)
            
            if r != -1:
                # --- Value Updates ---
                delta_2 = r - q_mf_stage2[s_idx, a2]
                q_mf_stage2[s_idx, a2] += learning_rate * delta_2
                
                # Simple TD(1) for Stage 1 MF
                q_mf_stage1[a1] += learning_rate * (r - q_mf_stage1[a1])

    return log_loss
```
