Here are three new cognitive models based on the participant's data and the two-step task structure.

### Model 1: Hybrid Learner with Forgetting (Decay)
This model extends the standard hybrid (MB/MF) model with perseverance by adding a **forgetting** mechanism for the second-stage values. The participant data suggests that reward probabilities for aliens change over time ("slowly changed over trials"). A decay parameter allows the model to "forget" the value of unchosen aliens, pushing their estimates back towards neutrality or zero, which can be adaptive in volatile environments.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF learner with Perseverance and Decay (Forgetting).
    
    This model includes a decay parameter that degrades the value of unchosen 
    second-stage options (aliens), allowing the model to adapt to changing 
    reward probabilities by reducing the influence of outdated information.

    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-value updates.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) control.
    - perseverance: [0, 5] Stickiness bonus for repeating the previous Stage 1 choice.
    - decay: [0, 1] Decay rate for unchosen Stage 2 values (0 = no decay, 1 = instant forgetting).
    """
    learning_rate, beta, w, perseverance, decay = model_parameters
    n_trials = len(action_1)

    # Fixed transition matrix for MB calculation (0.7 common, 0.3 rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)        # Stage 1 Model-Free values
    q_stage2_mf = np.zeros((2, 2))   # Stage 2 values (States x Aliens)
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Expected max value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted sum of MB and MF
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Perseverance Bonus
        if last_action_1 != -1:
            q_net_s1[last_action_1] += perseverance
            
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        # Standard Softmax on Stage 2 values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # TD(0) update for Stage 1 MF value
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(1) / Eligibility Trace update for Stage 1 (passing reward back)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage1_mf[a1] += learning_rate * delta_stage2 # lambda=1 implicit

        # Update Chosen Stage 2 Value
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Decay Unchosen Stage 2 Values
        # Apply decay to all options not chosen this trial to simulate forgetting
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == a2):
                    q_stage2_mf[s, a] *= (1.0 - decay)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Learner with Reward-Modulated Perseverance
The participant data shows strong "win-stay" behavior. This model splits the perseverance parameter into a baseline stickiness (habit) and a reward-dependent stickiness. This allows the model to distinguish between simply repeating a choice due to inertia versus repeating a choice because it was previously rewarded, offering a more granular explanation of the observed stability.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF learner with Reward-Modulated Perseverance.
    
    Distinguishes between general repetition bias (habit) and reinforcement-driven 
    repetition (Win-Stay). The stickiness bonus applied to the previous choice 
    depends on whether that choice resulted in a reward.

    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-value updates.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) control.
    - pers_base: [0, 5] Baseline stickiness (applied regardless of outcome).
    - pers_reward: [0, 5] Additional stickiness applied only if the previous trial was rewarded.
    """
    learning_rate, beta, w, pers_base, pers_reward = model_parameters
    n_trials = len(action_1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Reward-Modulated Perseverance
        if last_action_1 != -1:
            bonus = pers_base
            if last_reward == 1:
                bonus += pers_reward
            q_net_s1[last_action_1] += bonus
            
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        last_reward = r

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage1_mf[a1] += learning_rate * delta_stage2 # Eligibility trace
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with Transition Learning
Standard hybrid models assume the agent knows the transition probabilities (0.7/0.3). However, the prompt notes that "rare transitions occurred," and participants may learn this structure online. This model includes a parameter to update the transition matrix based on experienced state transitions, allowing the Model-Based component to rely on a learned internal model rather than a fixed ground truth.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF learner with Online Transition Learning.
    
    The Model-Based component does not use a fixed transition matrix. Instead, 
    it learns the transition probabilities (Spaceship -> Planet) trial-by-trial. 
    This accounts for subjective uncertainty about the task structure.

    Parameters:
    - learning_rate_q: [0, 1] Learning rate for Q-values (reward learning).
    - learning_rate_t: [0, 1] Learning rate for transition probabilities (structure learning).
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) control.
    - perseverance: [0, 5] Stickiness bonus for repeating the previous Stage 1 choice.
    """
    learning_rate_q, learning_rate_t, beta, w, perseverance = model_parameters
    n_trials = len(action_1)

    # Initialize transition matrix to uniform (max entropy) or slight prior
    # Rows: Actions (Spaceships), Cols: States (Planets)
    est_trans_matrix = np.ones((2, 2)) * 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Calculate MB values using the *estimated* transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = est_trans_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_s1[last_action_1] += perseverance
            
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # 1. Update Q-values (Reward Learning)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate_q * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage1_mf[a1] += learning_rate_q * delta_stage2
        q_stage2_mf[state_idx, a2] += learning_rate_q * delta_stage2
        
        # 2. Update Transition Matrix (Structure Learning)
        # Move probability towards the observed state (state_idx) for the chosen spaceship (a1)
        # Update P(s'|a1)
        est_trans_matrix[a1, state_idx] += learning_rate_t * (1.0 - est_trans_matrix[a1, state_idx])
        # Update P(other_s|a1) -> decreases
        est_trans_matrix[a1, 1-state_idx] += learning_rate_t * (0.0 - est_trans_matrix[a1, 1-state_idx])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
