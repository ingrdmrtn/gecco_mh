Here are three new cognitive models based on the participant data and feedback.

### Cognitive Model 1
**Title:** Hybrid MB/MF Model with Decay and Outcome-Dependent Perseveration
**Description:** This model extends the best-performing "Decay + Perseveration" model by splitting the perseveration parameter into two: `pers_win` and `pers_loss`. This hypothesizes that the participant's tendency to repeat their Stage 1 choice (stickiness) depends on whether the previous trial resulted in a reward (Win-Stay) or no reward. This captures potential "Win-Stay/Lose-Shift" heuristics interacting with the hybrid reinforcement learning process.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Value Decay and Outcome-Dependent Perseveration.
    
    Parameters:
    lr (float): Learning rate [0,1].
    beta (float): Inverse temperature [0,10].
    w (float): MB/MF Weighting [0,1].
    decay (float): Decay rate for Q-values per trial [0,1].
    pers_win (float): Perseveration bonus after a rewarded trial [0,5].
    pers_loss (float): Perseveration bonus after an unrewarded trial [0,5].
    """
    lr, beta, w, decay, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_reward = 0

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # Decay Q-values
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)

        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Stage 1 Value
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Calculate Stage 1 Logits with Outcome-Dependent Perseveration
        logits_1 = beta * q_net_stage1
        if prev_a1 != -1:
            if prev_reward == 1:
                logits_1[prev_a1] += pers_win
            else:
                logits_1[prev_a1] += pers_loss

        # Stage 1 Choice Probabilities
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice Probabilities
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # TD Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Eligibility Trace (implicitly lambda=1 for Stage 1 update from Stage 2 RPE)
        q_stage1_mf[action_1[trial]] += lr * delta_stage2
        
        prev_a1 = action_1[trial]
        prev_reward = reward[trial]

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Cognitive Model 2
**Title:** Hybrid MB/MF Model with Stage-Specific Learning Rates and Decay
**Description:** This model hypothesizes that the participant learns at different rates for the two stages of the task. Stage 1 (Spaceship choice) involves learning stable transition structures and integrating values, while Stage 2 (Alien choice) involves tracking drifting reward probabilities. A separate learning rate `lr_s2` is used for the second stage updates, while `lr_s1` is used for the first stage. This is combined with the successful `decay` and `pers` mechanisms.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Separate Learning Rates for Stage 1 & 2, Decay, and Perseveration.
    
    Parameters:
    lr_s1 (float): Learning rate for Stage 1 MF values [0,1].
    lr_s2 (float): Learning rate for Stage 2 MF values [0,1].
    beta (float): Inverse temperature [0,10].
    w (float): MB/MF Weighting [0,1].
    decay (float): Decay rate for Q-values per trial [0,1].
    pers (float): Perseveration bonus for repeating the previous stage 1 choice [0,5].
    """
    lr_s1, lr_s2, beta, w, decay, pers = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # Decay Q-values
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)

        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Stage 1 Value
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Stage 1 Choice Probabilities with Perseveration
        logits_1 = beta * q_net_stage1
        if prev_a1 != -1:
            logits_1[prev_a1] += pers

        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice Probabilities
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # TD Updates with separate learning rates
        # Update Stage 1 MF value using Stage 1 Learning Rate
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_s1 * delta_stage1

        # Update Stage 2 MF value using Stage 2 Learning Rate
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_s2 * delta_stage2
        
        # Eligibility Trace update for Stage 1 (using Stage 1 LR)
        q_stage1_mf[action_1[trial]] += lr_s1 * delta_stage2
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Cognitive Model 3
**Title:** Hybrid MB/MF Model with Decay, Eligibility Trace (Lambda), and Perseveration
**Description:** The previous best model implicitly assumed an eligibility trace of $\lambda=1$ (full transfer of Stage 2 reward prediction error to Stage 1 values). This model introduces `lam` as a free parameter to modulate how much the reward prediction error at the second stage affects the value of the spaceship chosen at the first stage. This is integrated with the `decay` and `pers` mechanisms that were found to be effective.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Value Decay, Eligibility Trace (Lambda), and Perseveration.
    
    Parameters:
    lr (float): Learning rate [0,1].
    beta (float): Inverse temperature [0,10].
    w (float): MB/MF Weighting [0,1].
    decay (float): Decay rate for Q-values per trial [0,1].
    lam (float): Eligibility trace parameter (Lambda) [0,1].
    pers (float): Perseveration bonus for repeating the previous stage 1 choice [0,5].
    """
    lr, beta, w, decay, lam, pers = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        # Decay Q-values
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)

        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Stage 1 Value
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Stage 1 Choice Probabilities with Perseveration
        logits_1 = beta * q_net_stage1
        if prev_a1 != -1:
            logits_1[prev_a1] += pers

        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice Probabilities
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # TD Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Eligibility Trace Update (Scaled by lambda)
        q_stage1_mf[action_1[trial]] += lr * lam * delta_stage2
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```
