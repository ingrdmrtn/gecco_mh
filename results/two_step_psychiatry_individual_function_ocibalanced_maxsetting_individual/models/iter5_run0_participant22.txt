Here are three new cognitive models for the two-step decision task.

### Model 1: Hybrid Model with Eligibility Traces (Lambda) and Stickiness
This model adds an eligibility trace parameter (`lam`) to the standard hybrid model. The eligibility trace allows the reward received at the second stage to directly reinforce the first-stage choice, bridging the temporal gap. This creates a more robust Model-Free learning signal (TD-lambda) alongside the Model-Based planning. It also includes choice stickiness to account for the participant's perseveration.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Eligibility Traces (Lambda) and Stickiness.
    
    Combines Model-Based planning with a Model-Free system that uses eligibility traces 
    (TD-lambda) to assign credit for the final reward directly to the first stage choice.
    
    Parameters:
    lr: [0,1] - Learning rate for MF Q-values.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - Weighting between MB (1) and MF (0) values.
    lam: [0,1] - Eligibility trace decay (lambda). Controls how much Stage 2 reward updates Stage 1 value.
    stick: [0,5] - Choice stickiness for Stage 1 (perseveration).
    """
    lr, beta_1, beta_2, w, lam, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue
            
        # Stage 1 Policy
        # MB Value: Expected value of next stage max Q
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Stage 1 PE
        delta_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_1
        
        # Stage 2 PE
        delta_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_2
        
        # Eligibility Trace Update for Stage 1 (TD-lambda)
        # The trace of state 1 decays by lambda, then is updated by the stage 2 error
        q_stage1_mf[action_1[trial]] += lr * lam * delta_2
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_trials = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Transition-Dependent Learning Rates
This model hypothesizes that the participant updates their Model-Free values differently depending on whether the transition to the second stage was "Common" (expected) or "Rare" (unexpected). It splits the Stage 1 learning rate into `lr_common` and `lr_rare`. This allows the model to capture behavior where participants might discount or over-weight the information gained from rare transitions.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Transition-Dependent Learning Rates.
    
    Uses different learning rates for the Stage 1 Model-Free update depending on whether 
    the transition to the second stage was 'Common' or 'Rare'. This allows the model 
    to gate learning based on the likelihood of the transition.
    
    Parameters:
    lr_common: [0,1] - Learning rate for Stage 1 update after a common transition.
    lr_rare: [0,1] - Learning rate for Stage 1 update after a rare transition.
    lr_s2: [0,1] - Learning rate for Stage 2 updates (reward learning).
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - MB/MF weighting.
    stick: [0,5] - Choice stickiness for Stage 1.
    """
    lr_common, lr_rare, lr_s2, beta_1, beta_2, w, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue
            
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Choice
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Determine if transition was common based on transition matrix
        trans_prob = transition_matrix[action_1[trial], state_idx]
        is_common = trans_prob > 0.5
        
        current_lr_s1 = lr_common if is_common else lr_rare
        
        # Stage 1 MF Update
        delta_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += current_lr_s1 * delta_1
        
        # Stage 2 MF Update
        delta_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_s2 * delta_2
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_trials = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Differential Forgetting
This model builds on the success of decay models but introduces *differential forgetting* for chosen and unchosen actions. It allows the model to separate the "leaky integration" of chosen values (`decay_c`) from the "forgetting" of unchosen alternatives (`decay_u`). This can capture complex memory dynamics where unvisited states lose value at a different rate than visited ones.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Differential Forgetting.
    
    Incorporates separate decay rates for chosen and unchosen actions. This allows the model
    to exhibit different memory dynamics for experienced vs. forgone outcomes.
    
    Parameters:
    lr: [0,1] - Learning rate.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - MB/MF weighting.
    stick: [0,5] - Choice stickiness for Stage 1.
    decay_c: [0,1] - Decay rate for chosen actions (leaky integration).
    decay_u: [0,1] - Decay rate for unchosen actions (forgetting).
    """
    lr, beta_1, beta_2, w, stick, decay_c, decay_u = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue
            
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Choice
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Stage 1
        delta_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_1
        
        # Apply differential decay to Stage 1
        q_stage1_mf[action_1[trial]] *= (1 - decay_c)
        q_stage1_mf[1 - action_1[trial]] *= (1 - decay_u)
        
        # Stage 2
        delta_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_2
        
        # Apply differential decay to Stage 2 (current state only)
        q_stage2_mf[state_idx, action_2[trial]] *= (1 - decay_c)
        q_stage2_mf[state_idx, 1 - action_2[trial]] *= (1 - decay_u)
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_trials = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
