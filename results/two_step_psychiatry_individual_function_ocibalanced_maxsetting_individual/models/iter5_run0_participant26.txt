Here are the 3 proposed cognitive models as Python functions.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner with Outcome-Dependent Stickiness.
    
    This model assumes the participant uses a Model-Based strategy (calculating
    expected values using the fixed transition matrix) but exhibits different
    perseveration (stickiness) biases depending on whether the previous trial
    resulted in a reward or not. This allows capturing behaviors like 
    'Win-Switch' (negative stick_win) or 'Loss-Stay' (positive stick_loss).
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Stage 2 alien values.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - stick_win: [-5, 5] Bias added to the previous Stage 1 choice if it was rewarded.
    - stick_loss: [-5, 5] Bias added to the previous Stage 1 choice if it was unrewarded.
    """
    learning_rate, beta_1, beta_2, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix: 
    # Row 0 (Action 0): [P(State 0), P(State 1)] = [0.7, 0.3]
    # Row 1 (Action 1): [P(State 0), P(State 1)] = [0.3, 0.7]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values for Stage 2 (Aliens): 2 states x 2 actions
    q_stage2 = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    
    last_action_1 = -1
    last_reward = -1
    
    for trial in range(n_trials):
        # Skip missing data
        if action_1[trial] == -1:
            continue
            
        # --- Stage 1 Decision (Model-Based) ---
        # V(State) = max Q_stage2(State, action)
        max_q_stage2 = np.max(q_stage2, axis=1)
        # Q_MB(Action) = Transition_Matrix * V(States)
        q_stage1 = transition_matrix @ max_q_stage2
        
        # Add Outcome-Dependent Stickiness to the previously chosen action
        if last_action_1 != -1:
            if last_reward == 1:
                q_stage1[last_action_1] += stick_win
            else:
                q_stage1[last_action_1] += stick_loss
                
        # Softmax Choice Probability
        exp_q1 = np.exp(beta_1 * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        chosen_a1 = int(action_1[trial])
        log_loss -= np.log(probs_1[chosen_a1] + eps)
        
        # --- Stage 2 Decision ---
        current_state = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_loss -= np.log(probs_2[chosen_a2] + eps)
        
        # --- Learning ---
        r = reward[trial]
        delta_stage2 = r - q_stage2[current_state, chosen_a2]
        q_stage2[current_state, chosen_a2] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = chosen_a1
        last_reward = r
        
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner with Subjective Transition Belief.
    
    This model assumes the participant is Model-Based but relies on a subjective
    estimate of the transition probabilities (parameter 'subjective_prob') rather 
    than the true probabilities (0.7). It also includes an alternation bias 
    to account for switching behavior.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Stage 2 alien values.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - subjective_prob: [0, 1] Subjective probability of the Common transition (A->X, U->Y).
    - alternation_bias: [0, 5] Bonus added to the unchosen Stage 1 action (switching bias).
    """
    learning_rate, beta_1, beta_2, subjective_prob, alternation_bias = model_parameters
    n_trials = len(action_1)
    
    # Subjective Transition Matrix constructed from parameter p
    # Assumes symmetry: P(Common) = p
    p = subjective_prob
    transition_matrix = np.array([[p, 1-p], [1-p, p]])
    
    q_stage2 = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1 = transition_matrix @ max_q_stage2
        
        # Alternation Bias: Bonus to the option NOT chosen last time
        if last_action_1 != -1:
            unchosen_action = 1 - last_action_1
            q_stage1[unchosen_action] += alternation_bias
            
        exp_q1 = np.exp(beta_1 * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        chosen_a1 = int(action_1[trial])
        log_loss -= np.log(probs_1[chosen_a1] + eps)
        
        # --- Stage 2 Decision ---
        current_state = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_loss -= np.log(probs_2[chosen_a2] + eps)
        
        # --- Learning ---
        r = reward[trial]
        delta_stage2 = r - q_stage2[current_state, chosen_a2]
        q_stage2[current_state, chosen_a2] += learning_rate * delta_stage2
        
        last_action_1 = chosen_a1
        
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based Learner with Asymmetric Learning Rates.
    
    This model assumes a Model-Based strategy but updates Stage 2 values with
    different learning rates for positive prediction errors (better than expected)
    and negative prediction errors (worse than expected). Includes alternation bias.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors.
    - lr_neg: [0, 1] Learning rate for negative prediction errors.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - alternation_bias: [0, 5] Bonus added to the unchosen Stage 1 action.
    """
    lr_pos, lr_neg, beta_1, beta_2, alternation_bias = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage2 = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1 = transition_matrix @ max_q_stage2
        
        # Alternation Bias
        if last_action_1 != -1:
            unchosen_action = 1 - last_action_1
            q_stage1[unchosen_action] += alternation_bias
            
        exp_q1 = np.exp(beta_1 * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        chosen_a1 = int(action_1[trial])
        log_loss -= np.log(probs_1[chosen_a1] + eps)
        
        # --- Stage 2 Decision ---
        current_state = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_loss -= np.log(probs_2[chosen_a2] + eps)
        
        # --- Learning ---
        r = reward[trial]
        delta_stage2 = r - q_stage2[current_state, chosen_a2]
        
        # Asymmetric Update
        if delta_stage2 > 0:
            q_stage2[current_state, chosen_a2] += lr_pos * delta_stage2
        else:
            q_stage2[current_state, chosen_a2] += lr_neg * delta_stage2
        
        last_action_1 = chosen_a1
        
    return log_loss
```
