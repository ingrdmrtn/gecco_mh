Here are the 3 proposed cognitive models.

### Model 1: Hybrid Model with Global Decay and Side Bias
This model extends the best-performing hybrid model (which included Q-value decay) by adding a **Side Bias** parameter. The participant data shows a strong, sustained preference for Spaceship 0 (trials 3-54, 85-200), which standard reinforcement learning might struggle to capture if the values are close. The bias parameter adds a constant bonus to the value of Spaceship 0, allowing the model to account for this baseline preference while still learning from rewards.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Global Decay and Side Bias.
    
    Incorporates a fixed bias towards Spaceship 0 to account for the participant's 
    observed preference, alongside a global decay mechanism for Stage 2 Q-values 
    to handle changing reward probabilities.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-value updates.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choices.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choices.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) values.
    - stickiness: [0, 5] Choice perseveration bonus for Stage 1.
    - decay: [0, 1] Decay rate for Stage 2 Q-values on each trial.
    - bias_0: [0, 5] Constant value bonus added to Spaceship 0.
    """
    learning_rate, beta_1, beta_2, w, stickiness, decay, bias_0 = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue

        # Global decay of Stage 2 values
        q_stage2_mf *= (1 - decay)

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Apply Side Bias to Spaceship 0
        q_net[0] += bias_0
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # Stage 2 Policy
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Subjective Transition Belief
Standard models assume the agent knows the true transition matrix (0.7/0.3). However, participants may have a distorted belief about the transition probabilities (e.g., believing they are deterministic or random). This model treats the **Subjective Transition Probability** as a free parameter used in the Model-Based calculation, allowing the model to fit the participant's specific understanding of the task structure.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Subjective Transition Belief.
    
    The agent uses a learned subjective transition probability (p_trans) for the 
    Model-Based value calculation instead of the fixed true probabilities.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_1: [0, 10] Stage 1 inverse temperature.
    - beta_2: [0, 10] Stage 2 inverse temperature.
    - w: [0, 1] MB/MF weight.
    - stickiness: [0, 5] Stage 1 choice stickiness.
    - p_trans: [0, 1] Subjective probability of common transitions (A->X, U->Y).
    """
    learning_rate, beta_1, beta_2, w, stickiness, p_trans = model_parameters
    n_trials = len(action_1)
    
    # Subjective transition matrix based on parameter
    # p_trans is the belief that A->X and U->Y
    transition_matrix = np.array([[p_trans, 1 - p_trans], [1 - p_trans, p_trans]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Use subjective matrix for MB calculation
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # Stage 2 Policy
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 3: Probability Mixture (Independent Agents)
Instead of averaging the Q-values (Value Mixture) of the Model-Based and Model-Free systems, this model assumes two independent agents that propose choice probabilities. The final policy is a weighted mixture of these **Probabilities**. This allows each system to have its own confidence level (inverse temperature), separating the sharpness of the Model-Based predictions from the Model-Free habits.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Probability Mixture Model (Independent MB/MF Agents).
    
    Mixes the choice probabilities generated by separate Model-Based and Model-Free 
    systems, each with its own inverse temperature (beta), rather than mixing Q-values.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_mb: [0, 10] Inverse temperature for the MB system.
    - beta_mf: [0, 10] Inverse temperature for the MF system.
    - beta_2: [0, 10] Stage 2 inverse temperature.
    - w_mix: [0, 1] Weight of the MB agent's probability distribution in the final mix.
    - stickiness: [0, 5] Stickiness bonus applied within the MF system.
    """
    learning_rate, beta_mb, beta_mf, beta_2, w_mix, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue

        # --- Stage 1: Independent Probability Generation ---
        
        # 1. Model-Based Agent
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        exp_mb = np.exp(beta_mb * q_stage1_mb)
        probs_mb = exp_mb / np.sum(exp_mb)
        
        # 2. Model-Free Agent (with stickiness)
        q_mf_aug = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_mf_aug[last_action_1] += stickiness
        exp_mf = np.exp(beta_mf * q_mf_aug)
        probs_mf = exp_mf / np.sum(exp_mf)
        
        # Mixture of Probabilities
        probs_1 = w_mix * probs_mb + (1 - w_mix) * probs_mf
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2: Standard Model-Free ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
