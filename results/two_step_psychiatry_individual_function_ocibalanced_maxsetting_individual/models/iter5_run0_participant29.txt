Here are three new cognitive models for the two-step task, designed to capture different hypotheses about the participant's behavior, such as decision noise differences between stages, forgetting mechanisms combined with model-based reasoning, and transition-dependent learning updates.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dual-Beta Hybrid Model.
    This model assumes the participant uses a hybrid of Model-Based (MB) and Model-Free (MF) 
    strategies, but applies different levels of decision noise (beta) to the Spaceship choice (Stage 1) 
    and the Alien choice (Stage 2).
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values in both stages.
    - beta1: [0, 10] Inverse temperature for Stage 1 (Spaceship) choice.
    - beta2: [0, 10] Inverse temperature for Stage 2 (Alien) choice.
    - w: [0, 1] Weight for Model-Based values in Stage 1 (0 = Pure MF, 1 = Pure MB).
    """
    learning_rate, beta1, beta2, w = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix (Assumed structure: A->X (0.7), U->Y (0.7))
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # [Planet, Alien]
    
    log_loss = 0.0
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Calculate Model-Based value for Stage 1 based on current Stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix MB and MF values
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Choice probability using beta1
        exp_q1 = np.exp(beta1 * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[action_1[trial]] + 1e-10)
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        q_s2 = q_stage2_mf[state_idx]
        
        # Choice probability using beta2
        exp_q2 = np.exp(beta2 * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[action_2[trial]] + 1e-10)
        
        # --- Updates ---
        # Stage 2 Update (Reward Prediction Error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Stage 1 MF Update (TD Prediction Error)
        # Using the value of the chosen state/action in stage 2 as the target
        target_stage1 = q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = target_stage1 - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Forgetting.
    A Hybrid MB/MF model that incorporates a forgetting mechanism. Unchosen options 
    in both Stage 1 (MF) and Stage 2 (Aliens) decay towards zero, representing 
    memory loss or passive decay of value for unvisited states.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for choices.
    - w: [0, 1] Weight for Model-Based values in Stage 1.
    - forget_rate: [0, 1] Decay rate for unchosen Q-values (0 = no forget, 1 = instant reset).
    """
    learning_rate, beta, w, forget_rate = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[action_1[trial]] + 1e-10)
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[action_2[trial]] + 1e-10)
        
        # --- Updates ---
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Stage 2 Forgetting (Decay unchosen alien)
        unchosen_a2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - forget_rate)
        
        # Stage 1 Update
        target_stage1 = q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = target_stage1 - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 1 Forgetting (Decay unchosen spaceship)
        unchosen_a1 = 1 - action_1[trial]
        q_stage1_mf[unchosen_a1] *= (1.0 - forget_rate)
        
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free Q-Learning with Transition-Dependent Learning Rates.
    This model is purely Model-Free but approximates structure learning by using 
    different learning rates for Stage 1 updates depending on whether the transition 
    was Common (expected) or Rare (unexpected).
    
    Parameters:
    - alpha_common: [0, 1] Learning rate for Stage 1 updates after a Common transition.
    - alpha_rare: [0, 1] Learning rate for Stage 1 updates after a Rare transition.
    - beta: [0, 10] Inverse temperature.
    - p: [0, 5] Perseveration bonus added to the previously chosen Stage 1 option.
    """
    alpha_common, alpha_rare, beta, p = model_parameters
    n_trials = len(action_1)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    prev_choice_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        q_net = q_stage1_mf.copy()
        
        # Apply perseveration bonus
        if prev_choice_1 != -1:
            q_net[prev_choice_1] += p
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[action_1[trial]] + 1e-10)
        
        prev_choice_1 = action_1[trial]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[action_2[trial]] + 1e-10)
        
        # --- Updates ---
        # Stage 2 Update (Standard Q-learning, using alpha_common as the base rate)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += alpha_common * delta_stage2
        
        # Stage 1 Update (Transition-Dependent)
        a1 = action_1[trial]
        s_next = state[trial]
        
        # Determine if transition was common: 0->0 or 1->1 are Common; 0->1 or 1->0 are Rare
        is_common = (a1 == s_next)
        
        # Select appropriate learning rate
        current_alpha = alpha_common if is_common else alpha_rare
        
        target_stage1 = q_stage2_mf[state_idx, action_2[trial]]
        delta_stage1 = target_stage1 - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_alpha * delta_stage1
        
    return log_loss
```
