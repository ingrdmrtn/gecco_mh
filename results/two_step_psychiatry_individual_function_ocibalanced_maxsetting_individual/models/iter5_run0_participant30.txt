Here are three new cognitive models for the two-step decision task, designed to capture different hypotheses about the participant's strategy (structural learning, asymmetric reward processing, and causal credit assignment).

### Model 1: Subjective Transition Hybrid Model
This model tests the hypothesis that the participant combines Model-Based (MB) and Model-Free (MF) planning but maintains a subjective belief about the transition probabilities (e.g., how reliable the spaceships are) that may differ from the true 70/30 structure. It also includes decay and perseveration.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Subjective Transition Hybrid Model.
    Combines Model-Based and Model-Free values, but parameterizes the agent's 
    belief about the transition matrix rather than assuming fixed 0.7/0.3.
    
    Parameters:
    learning_rate: rate for MF and Stage 2 updates [0,1]
    beta: inverse temperature [0,10]
    w: mixing weight (0=MF, 1=MB) [0,1]
    prob_common: subjective probability of common transition [0,1]
    decay: rate at which Q-values decay to 0.5 [0,1]
    perseveration: stickiness to previous choice [0,5]
    """
    learning_rate, beta, w, prob_common, decay, perseveration = model_parameters
    n_trials = len(action_1)

    # Subjective transition matrix based on parameter
    # Assumes symmetry: A->X and U->Y are 'common' with prob_common
    transition_matrix = np.array([[prob_common, 1.0 - prob_common], 
                                  [1.0 - prob_common, prob_common]])

    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    last_action_1 = -1

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # Decay Q-values toward 0.5
        q_stage1_mf = q_stage1_mf * (1 - decay) + 0.5 * decay
        q_stage2_mf = q_stage2_mf * (1 - decay) + 0.5 * decay

        # --- Stage 1 Choice ---
        # Model-Based Value: Expected max value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Perseveration
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Stage 1 MF Update (SARSA-style)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Note: Standard Hybrid often omits eligibility traces for simplicity when w is present,
        # relying on the MB component for structure and MF for cached value.
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model with Asymmetric Learning and Decay
This model integrates the successful mechanisms from the previous best model (asymmetric learning rates, decay, eligibility traces) into a Hybrid MB/MF architecture. It hypothesizes that the asymmetry in learning (optimism/pessimism) occurs primarily at the reward stage but propagates to both the MF and MB systems.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Asymmetric Learning, Eligibility, and Decay.
    Combines Model-Based planning with a sophisticated Model-Free system that 
    learns differently from positive/negative errors and decays unused values.
    
    Parameters:
    lr_pos: learning rate for positive prediction errors [0,1]
    lr_neg: learning rate for negative prediction errors [0,1]
    beta: inverse temperature [0,10]
    w: mixing weight (0=MF, 1=MB) [0,1]
    lambda_eligibility: strength of eligibility trace [0,1]
    decay: rate at which Q-values decay to 0.5 [0,1]
    perseveration: stickiness to previous choice [0,5]
    """
    lr_pos, lr_neg, beta, w, lambda_eligibility, decay, perseveration = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition structure for MB
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # Decay
        q_stage1_mf = q_stage1_mf * (1 - decay) + 0.5 * decay
        q_stage2_mf = q_stage2_mf * (1 - decay) + 0.5 * decay

        # --- Stage 1 Choice ---
        # MB Component
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Mix
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Stage 1 MF Update (Asymmetric)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        lr_s1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[action_1[trial]] += lr_s1 * delta_stage1

        # Stage 2 Update (Asymmetric)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        lr_s2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[state_idx, action_2[trial]] += lr_s2 * delta_stage2
        
        # Eligibility Trace Update for Stage 1 (Asymmetric based on Stage 2 error)
        # The trace update usually follows the sign of the Stage 2 error
        lr_trace = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage1_mf[action_1[trial]] += lr_trace * lambda_eligibility * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Transition-Gated Model-Free Model
This model represents a heuristic "smart" Model-Free agent. Instead of calculating full Model-Based values, it modulates the eligibility trace based on whether the transition was Common or Rare. This allows the agent to reinforce Stage 1 choices strongly after Common transitions but suppress reinforcement (or reinforce differently) after Rare transitions, approximating Model-Based behavior with lower computational cost.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Transition-Gated Model-Free Model.
    A pure MF model that modulates the eligibility trace (credit assignment) 
    depending on whether the transition was Common or Rare.
    
    Parameters:
    learning_rate: learning rate [0,1]
    beta: inverse temperature [0,10]
    lambda_common: eligibility trace strength for common transitions [0,1]
    lambda_rare: eligibility trace strength for rare transitions [0,1]
    decay: rate at which Q-values decay to 0.5 [0,1]
    perseveration: stickiness to previous choice [0,5]
    """
    learning_rate, beta, lambda_common, lambda_rare, decay, perseveration = model_parameters
    n_trials = len(action_1)

    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # Decay
        q_stage1_mf = q_stage1_mf * (1 - decay) + 0.5 * decay
        q_stage2_mf = q_stage2_mf * (1 - decay) + 0.5 * decay

        # --- Stage 1 Choice ---
        q_net_1 = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseveration
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # Determine if transition was common (Action 0->State 0 or Action 1->State 1)
        # Using 0.7/0.3 structure implied by task description
        is_common = (action_1[trial] == state_idx)
        
        # Select eligibility trace based on transition type
        current_lambda = lambda_common if is_common else lambda_rare

        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Gated Trace Update
        q_stage1_mf[action_1[trial]] += learning_rate * current_lambda * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
