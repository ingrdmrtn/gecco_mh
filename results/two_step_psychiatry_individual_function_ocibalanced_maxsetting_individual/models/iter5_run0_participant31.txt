Here are three new cognitive models for the two-step decision task, designed to capture different potential strategies like transition learning, stage-specific learning rates, and outcome-dependent exploration.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Transition Learning Model.
    
    The agent actively updates their belief about the transition probabilities 
    (Model-Based) based on observed transitions, rather than assuming a fixed 
    70/30 structure.
    
    Parameters:
    - lr_val: [0, 1] Learning rate for Q-values (Model-Free).
    - lr_trans: [0, 1] Learning rate for transition probabilities (Model-Based).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (1 = pure MB, 0 = pure MF).
    - stick: [-5, 5] Stickiness bias for Stage 1 choice.
    """
    lr_val, lr_trans, beta, w, stick = model_parameters
    n_trials = len(action_1)
    
    # Initialize transition matrix (Row: Action 0/1, Col: State 0/1)
    # Start with the common prior but allow it to drift
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # MB Value calculation using dynamic transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stick
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Value Updates ---
        # Stage 1 MF update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_val * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_val * delta_stage2
        
        # --- Transition Matrix Update ---
        # Update P(state|action) towards the observed state (1) and complement towards 0
        a1 = action_1[trial]
        s_obs = state[trial]
        
        # Increase probability of the observed transition
        transition_matrix[a1, s_obs] += lr_trans * (1 - transition_matrix[a1, s_obs])
        # Decrease probability of the unobserved transition (complement)
        transition_matrix[a1, 1-s_obs] += lr_trans * (0 - transition_matrix[a1, 1-s_obs])
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Dual Learning Rates with Split Stickiness.
    
    Uses separate learning rates for Stage 1 (navigation) and Stage 2 (bandit),
    allowing the model to capture different adaptation speeds for the two distinct
    sub-tasks. Also includes outcome-dependent stickiness.
    
    Parameters:
    - lr_s1: [0, 1] Learning rate for Stage 1 MF values.
    - lr_s2: [0, 1] Learning rate for Stage 2 MF values.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - stick_win: [-5, 5] Stickiness after reward.
    - stick_loss: [-5, 5] Stickiness after no reward.
    """
    lr_s1, lr_s2, beta, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    last_reward = -1
    
    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            if last_reward == 1:
                logits[last_action_1] += stick_win
            else:
                logits[last_action_1] += stick_loss
        
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Stage 1 uses lr_s1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_s1 * delta_stage1
        
        # Stage 2 uses lr_s2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_s2 * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Outcome-Dependent Exploration (Beta-Win/Loss).
    
    The agent's exploration-exploitation balance (beta) changes depending on 
    whether the previous trial was rewarded. This captures strategies like 
    "Win-Exploit, Lose-Random".
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta_win: [0, 10] Inverse temperature after a win.
    - beta_loss: [0, 10] Inverse temperature after a loss.
    - w: [0, 1] Mixing weight.
    - stick: [-5, 5] Stickiness bias.
    """
    lr, beta_win, beta_loss, w, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    last_reward = 0 # Default to 0 (loss state) or neutral for first trial
    
    for trial in range(n_trials):
        # Determine beta based on last reward
        current_beta = beta_win if last_reward == 1 else beta_loss
        
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stick
            
        exp_q1 = np.exp(current_beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2
        state_idx = state[trial]
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        last_action_1 = action_1[trial]
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
