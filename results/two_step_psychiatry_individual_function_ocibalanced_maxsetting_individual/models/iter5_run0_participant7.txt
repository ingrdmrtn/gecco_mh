def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dual Learning Rate Model with Stage 1 Stickiness.
    
    Distinguishes between the learning rate for Stage 1 (Spaceships) and Stage 2 (Aliens),
    acknowledging that the reward dynamics (aliens) drift while transition dynamics (spaceships) 
    are stable. Includes stickiness for spaceship choice.
    
    Parameters:
    - lr_1: Learning rate for Stage 1 Q-values [0, 1]
    - lr_2: Learning rate for Stage 2 Q-values [0, 1]
    - beta_1: Inverse temperature for Stage 1 softmax [0, 10]
    - beta_2: Inverse temperature for Stage 2 softmax [0, 10]
    - w: Mixing weight between Model-Based and Model-Free values [0, 1]
    - lam: Eligibility trace parameter [0, 1]
    - stick: Choice stickiness bonus for repeating the previous Stage 1 choice [0, 10]
    """
    lr_1, lr_2, beta_1, beta_2, w, lam, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        logits_1 = beta_1 * q_net
        
        if trial > 0:
            logits_1[action_1[trial-1]] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # Stage 2 Policy
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        # Stage 1 MF update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_1 * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_2 * delta_stage2
        
        # Eligibility trace update for Stage 1 (using lr_1 as it updates Stage 1 value)
        q_stage1_mf[action_1[trial]] += lr_1 * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Asymmetric Eligibility Traces and Stage 1 Stickiness.
    
    Allows the eligibility trace (how much Stage 2 outcome affects Stage 1 value) to 
    differ based on whether the Stage 2 outcome was better or worse than expected 
    (Positive vs Negative RPE). This captures potential valence-dependent credit assignment.
    
    Parameters:
    - learning_rate: Rate at which Q-values are updated [0, 1]
    - beta_1: Inverse temperature for Stage 1 softmax [0, 10]
    - beta_2: Inverse temperature for Stage 2 softmax [0, 10]
    - w: Mixing weight between Model-Based and Model-Free values [0, 1]
    - lam_pos: Eligibility trace strength for positive Stage 2 RPE [0, 1]
    - lam_neg: Eligibility trace strength for negative Stage 2 RPE [0, 1]
    - stick: Choice stickiness bonus for repeating the previous Stage 1 choice [0, 10]
    """
    learning_rate, beta_1, beta_2, w, lam_pos, lam_neg, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        logits_1 = beta_1 * q_net
        
        if trial > 0:
            logits_1[action_1[trial-1]] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # Stage 2 Policy
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Asymmetric Eligibility Trace
        eff_lam = lam_pos if delta_stage2 >= 0 else lam_neg
        q_stage1_mf[action_1[trial]] += learning_rate * eff_lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Dual Stickiness and Stage 2 Decay.
    
    Combines Stage 1 stickiness (spaceship perseveration), Stage 2 stickiness (alien perseveration),
    and Stage 2 passive decay. This assumes the agent perseveres in both motor choices but 
    also forgets unvisited alien values, helping with the non-stationary rewards.
    
    Parameters:
    - learning_rate: Rate at which Q-values are updated [0, 1]
    - beta_1: Inverse temperature for Stage 1 softmax [0, 10]
    - beta_2: Inverse temperature for Stage 2 softmax [0, 10]
    - w: Mixing weight between Model-Based and Model-Free values [0, 1]
    - lam: Eligibility trace parameter [0, 1]
    - decay: Rate at which unchosen Stage 2 Q-values decay towards 0.5 [0, 1]
    - stick_1: Stickiness for Stage 1 choice [0, 10]
    - stick_2: Stickiness for Stage 2 choice [0, 10]
    """
    learning_rate, beta_1, beta_2, w, lam, decay, stick_1, stick_2 = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        logits_1 = beta_1 * q_net
        
        if trial > 0:
            logits_1[action_1[trial-1]] += stick_1
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # Stage 2 Policy
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        
        # Apply Stage 2 stickiness (motor perseveration on the second step)
        if trial > 0:
            logits_2[action_2[trial-1]] += stick_2
            
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        q_stage1_mf[action_1[trial]] += learning_rate * lam * delta_stage2
        
        # Decay for Stage 2 unchosen options
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == action_2[trial]):
                    q_stage2_mf[s, a] = (1 - decay) * q_stage2_mf[s, a] + decay * 0.5

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
