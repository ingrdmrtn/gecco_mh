Here are three new cognitive models for the two-step task. These models build upon the success of the "unchosen decay" mechanism by integrating it with distinct, unexplored psychological hypotheses regarding subjective probability, context-dependent habit, and reward valence.

### Model 1: Subjective Transition Hybrid Model
This model tests the hypothesis that the participant may not perceive the transition probabilities as exactly 0.7/0.3, but instead operates with a **subjective transition probability** (`p_trans`). This subjective belief (e.g., believing transitions are more deterministic or more random than reality) biases the Model-Based value calculation.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Subjective Transition Hybrid Model with Decay.
    
    Incorporates a subjective belief about transition probabilities which may differ 
    from the true environmental probabilities. Includes unchosen decay and stickiness.
    
    Parameters:
    - learning_rate: [0,1] Rate of updating value based on reward.
    - decay_rate: [0,1] Rate at which unchosen action values decay to 0.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weight of Model-Based system.
    - stickiness: [0,5] Bonus added to the previously chosen action's value.
    - p_trans: [0,1] Subjective probability of the common transition (e.g., A->X).
    """
    learning_rate, decay_rate, beta, w, stickiness, p_trans = model_parameters
    n_trials = len(action_1)
  
    # Subjective transition matrix based on p_trans parameter
    # Row 0: Space A -> [Prob(X), Prob(Y)]
    # Row 1: Space B -> [Prob(X), Prob(Y)]
    # We assume symmetry: P(A->X) = P(B->Y) = p_trans
    transition_matrix = np.array([[p_trans, 1 - p_trans], [1 - p_trans, p_trans]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Decision ---
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # Decay Unchosen
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Transition-Dependent Stickiness Model
This model hypothesizes that the tendency to repeat a choice (stickiness) is context-dependent. Specifically, it tests if a "surprise" (Rare transition) disrupts the habitual motor repetition compared to an expected (Common) transition. It splits the stickiness parameter into `stick_common` and `stick_rare`.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Transition-Dependent Stickiness Model with Decay.
    
    Applies different stickiness bonuses depending on whether the previous trial's
    transition was Common or Rare. This captures whether surprise affects perseveration.
    
    Parameters:
    - learning_rate: [0,1] Rate of updating value based on reward.
    - decay_rate: [0,1] Rate at which unchosen action values decay to 0.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weight of Model-Based system.
    - stick_common: [0,5] Stickiness bonus after a common transition.
    - stick_rare: [0,5] Stickiness bonus after a rare transition.
    """
    learning_rate, decay_rate, beta, w, stick_common, stick_rare = model_parameters
    n_trials = len(action_1)
  
    # Fixed environmental transition matrix for MB calc
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_transition_type = -1 # 0 for Common, 1 for Rare

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net_stage1
        if last_action_1 != -1:
            if last_transition_type == 0: # Common
                logits_1[last_action_1] += stick_common
            else: # Rare
                logits_1[last_action_1] += stick_rare
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Decision ---
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # Decay
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        # --- Update History ---
        last_action_1 = a1
        # Determine transition type: 
        # A(0)->X(0) is Common, U(1)->Y(1) is Common.
        # A(0)->Y(1) is Rare, U(1)->X(0) is Rare.
        if a1 == state_idx:
            last_transition_type = 0 # Common
        else:
            last_transition_type = 1 # Rare

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning Rate Model
This model introduces valence-dependent learning rates (`lr_pos` vs `lr_neg`) combined with the effective decay mechanism. It tests if the participant learns differently from "better than expected" outcomes (positive RPE) versus "worse than expected" outcomes (negative RPE), which is a common feature in risk-sensitive reinforcement learning.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Learning Rate Model with Decay.
    
    Differentiates learning from positive prediction errors (better than expected)
    and negative prediction errors (worse than expected). Includes unchosen decay
    and stickiness.
    
    Parameters:
    - lr_pos: [0,1] Learning rate for positive prediction errors.
    - lr_neg: [0,1] Learning rate for negative prediction errors.
    - decay_rate: [0,1] Rate at which unchosen action values decay to 0.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weight of Model-Based system.
    - stickiness: [0,5] Bonus added to the previously chosen action's value.
    """
    lr_pos, lr_neg, decay_rate, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net_stage1
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]

        # --- Stage 2 Decision ---
        state_idx = state[trial]
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        lr_s1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[a1] += lr_s1 * delta_stage1
        
        # Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        lr_s2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[state_idx, a2] += lr_s2 * delta_stage2

        # Decay
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
