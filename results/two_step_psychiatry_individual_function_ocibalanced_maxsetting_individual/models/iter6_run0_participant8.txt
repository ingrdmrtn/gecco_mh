import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Dual Learning Rate Hybrid Model with Eligibility Traces.
    
    This model distinguishes between the learning rates for Stage 1 (Spaceship choice)
    and Stage 2 (Alien choice), positing that plasticity may differ between navigation
    and harvesting tasks. It incorporates an eligibility trace (lambda) to allow 
    Stage 2 reward prediction errors to update Stage 1 values directly.
    
    Parameters:
    lr_1: [0,1] - Learning rate for Stage 1 (Spaceship).
    lr_2: [0,1] - Learning rate for Stage 2 (Alien).
    beta: [0,10] - Inverse temperature for softmax.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    lam: [0,1] - Eligibility trace parameter (0 = TD(0), 1 = TD(1)).
    stickiness: [0,5] - Choice perseveration bonus.
    """
    lr_1, lr_2, beta, w, lam, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        chosen_a1 = action_1[trial]
        prev_a1 = chosen_a1
        state_idx = state[trial]
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        r = reward[trial]
        
        # Updates
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += lr_1 * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += lr_2 * delta_stage2
        
        # Eligibility Trace: Propagate Stage 2 RPE to Stage 1
        q_stage1_mf[chosen_a1] += lr_1 * lam * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Model-Based Learning with Reward Decay.
    
    This model assumes the participant does not trust the transition probabilities 
    to be fixed and learns them over time (`lr_trans`). It also incorporates 
    decay for unchosen options to handle the drifting reward probabilities of the aliens.
    
    Parameters:
    lr_rew: [0,1] - Learning rate for reward Q-values.
    lr_trans: [0,1] - Learning rate for the transition matrix.
    decay: [0,1] - Decay rate for unchosen Q-values.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    stickiness: [0,5] - Choice perseveration bonus.
    """
    lr_rew, lr_trans, decay, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Initialize transition matrix to instructions, but allow updates
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy (using dynamic transition matrix)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        chosen_a1 = action_1[trial]
        prev_a1 = chosen_a1
        state_idx = state[trial]
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        r = reward[trial]
        
        # Update Transition Matrix based on observed state
        # Increases prob of the observed transition, decreases the other
        transition_matrix[chosen_a1, state_idx] += lr_trans * (1 - transition_matrix[chosen_a1, state_idx])
        transition_matrix[chosen_a1, 1 - state_idx] = 1 - transition_matrix[chosen_a1, state_idx]
        
        # Value Updates
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += lr_rew * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += lr_rew * delta_stage2
        
        # Decay unchosen options
        q_stage1_mf[1 - chosen_a1] *= (1 - decay)
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == chosen_a2):
                    q_stage2_mf[s, a] *= (1 - decay)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Reward-Dependent Stickiness and Decay.
    
    This model implements differential perseveration (stickiness) depending on
    whether the previous trial was rewarded or unrewarded. This captures complex
    'Win-Stay, Lose-Shift' dynamics combined with RL. It also includes decay 
    to handle drifting reward probabilities.
    
    Parameters:
    lr: [0,1] - Learning rate.
    decay: [0,1] - Decay rate for unchosen options.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting parameter.
    stick_rew: [0,5] - Stickiness bonus applied if previous trial was rewarded.
    stick_unrew: [0,5] - Stickiness bonus applied if previous trial was unrewarded.
    """
    lr, decay, beta, w, stick_rew, stick_unrew = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    prev_reward = 0
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply differential stickiness based on previous outcome
        if prev_a1 != -1:
            if prev_reward == 1:
                q_net[prev_a1] += stick_rew
            else:
                q_net[prev_a1] += stick_unrew
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        chosen_a1 = action_1[trial]
        prev_a1 = chosen_a1
        state_idx = state[trial]
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        r = reward[trial]
        prev_reward = r
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += lr * delta_stage2
        
        # Decay unchosen options
        q_stage1_mf[1 - chosen_a1] *= (1 - decay)
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == chosen_a2):
                    q_stage2_mf[s, a] *= (1 - decay)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
