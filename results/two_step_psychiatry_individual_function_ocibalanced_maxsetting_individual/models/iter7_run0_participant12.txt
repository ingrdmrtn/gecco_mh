Here are three new cognitive models for the two-step task.

### Model 1: Counterfactual Update Model
This model incorporates **counterfactual learning** in the second stage. Given the task structure (two aliens per planet), participants often infer that if one alien yields no reward, the other is likely to yield a reward (and vice versa). This model updates the unchosen alien's Q-value towards the opposite of the received reward (`1 - reward`), controlled by a specific learning rate `lr_cf`.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Counterfactual Update Model.
    
    Assumes that participants update the unchosen alien's value in the opposite 
    direction of the received reward. If the chosen alien yields a reward (1), 
    the unchosen one is updated towards 0, and if the chosen yields 0, the 
    unchosen is updated towards 1. This captures belief in anti-correlated rewards.
    
    Parameters:
    lr (float): Learning rate for the chosen option [0, 1].
    beta (float): Inverse temperature [0, 10].
    w (float): MB/MF mixing weight [0, 1].
    lam (float): Eligibility trace parameter [0, 1].
    stick (float): Choice stickiness for Stage 1 [0, 5].
    lr_cf (float): Counterfactual learning rate for unchosen Stage 2 option [0, 1].
    """
    lr, beta, w, lam, stick, lr_cf = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1
    log_loss = 0
    eps = 1e-10
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        logits_1 = beta * q_net_1
        
        if last_action_1 != -1:
            logits_1[last_action_1] += stick
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            log_loss -= np.log(probs_1[action_1[trial]] + eps)
            last_action_1 = action_1[trial]
            
        state_idx = state[trial]
        if state_idx == -1:
            continue
            
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = action_2[trial]
        if a2 != -1:
            log_loss -= np.log(probs_2[a2] + eps)
            
            r = reward[trial]
            if r == -1: r = 0
            
            a1 = action_1[trial]
            
            # Prediction errors
            delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
            delta_stage2 = r - q_stage2_mf[state_idx, a2]
            
            # Update Stage 1 MF
            q_stage1_mf[a1] += lr * delta_stage1 + lr * lam * delta_stage2
            
            # Update Stage 2 MF (Chosen)
            q_stage2_mf[state_idx, a2] += lr * delta_stage2
            
            # Update Stage 2 MF (Unchosen - Counterfactual)
            # Target is (1 - r). 
            unchosen_a2 = 1 - a2
            delta_cf = (1.0 - r) - q_stage2_mf[state_idx, unchosen_a2]
            q_stage2_mf[state_idx, unchosen_a2] += lr_cf * delta_cf
            
    return log_loss
```

### Model 2: Dual Stickiness with Decay
This model extends the successful "Stickiness + Decay" approach by acknowledging that stickiness occurs in **both** stages. The participant data shows repeated choices of specific aliens within specific planets (Stage 2), not just repeated spaceship choices (Stage 1). This model includes separate stickiness parameters for Stage 1 and Stage 2, alongside the decay mechanism for unchosen options.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Dual Stickiness with Decay Model.
    
    Incorporates separate stickiness parameters for Stage 1 (spaceship) and 
    Stage 2 (alien) choices. Stickiness in Stage 2 is state-dependent (tracking 
    last choice per planet). Also includes decay for unchosen Stage 2 options 
    to handle non-stationarity.
    
    Parameters:
    lr (float): Learning rate [0, 1].
    beta (float): Inverse temperature [0, 10].
    w (float): MB/MF mixing weight [0, 1].
    lam (float): Eligibility trace parameter [0, 1].
    stick_1 (float): Stickiness for Stage 1 [0, 5].
    stick_2 (float): Stickiness for Stage 2 [0, 5].
    decay (float): Decay rate for unchosen Stage 2 action [0, 1].
    """
    lr, beta, w, lam, stick_1, stick_2, decay = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_action_2 = [-1, -1] # Track last action for each state (planet)
    
    log_loss = 0
    eps = 1e-10
    
    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net_1
        if last_action_1 != -1:
            logits_1[last_action_1] += stick_1
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            log_loss -= np.log(probs_1[action_1[trial]] + eps)
            last_action_1 = action_1[trial]
            
        state_idx = state[trial]
        if state_idx == -1: continue
            
        # Stage 2
        logits_2 = beta * q_stage2_mf[state_idx]
        # Apply stickiness for the specific state if previously visited
        if last_action_2[state_idx] != -1:
            logits_2[last_action_2[state_idx]] += stick_2
            
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = action_2[trial]
        if a2 != -1:
            log_loss -= np.log(probs_2[a2] + eps)
            last_action_2[state_idx] = a2 # Update last action for this state
            
            r = reward[trial]
            if r == -1: r = 0
            
            a1 = action_1[trial]
            
            delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
            delta_stage2 = r - q_stage2_mf[state_idx, a2]
            
            q_stage1_mf[a1] += lr * delta_stage1 + lr * lam * delta_stage2
            q_stage2_mf[state_idx, a2] += lr * delta_stage2
            
            # Decay unchosen
            q_stage2_mf[state_idx, 1 - a2] *= (1.0 - decay)
            
    return log_loss
```

### Model 3: Mixture Model with Lapse (Epsilon-Greedy)
This model adds a **lapse rate (`epsilon`)** to the best-performing configuration (Stickiness + Decay). The decision policy is modeled as a mixture of the softmax value-based choice and a uniform random choice. This accounts for attentional lapses or random exploration, making the model more robust to outliers in the participant's behavior.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Mixture Model with Lapse (Epsilon-Greedy).
    
    Combines the mechanisms of Stickiness and Decay with a lapse rate (epsilon). 
    The policy is a mixture of the softmax distribution and a uniform distribution, 
    accounting for random exploration or errors (lapses) in decision making.
    
    Parameters:
    lr (float): Learning rate [0, 1].
    beta (float): Inverse temperature [0, 10].
    w (float): MB/MF mixing weight [0, 1].
    lam (float): Eligibility trace parameter [0, 1].
    stick (float): Stage 1 stickiness [0, 5].
    decay (float): Decay rate for unchosen Stage 2 action [0, 1].
    epsilon (float): Lapse rate (probability of random choice) [0, 1].
    """
    lr, beta, w, lam, stick, decay, epsilon = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    log_loss = 0
    eps = 1e-10
    
    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net_1
        if last_action_1 != -1:
            logits_1[last_action_1] += stick
            
        exp_q1 = np.exp(logits_1)
        softmax_probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Mixture policy: (1-eps)*Softmax + eps*Uniform
        probs_1 = (1 - epsilon) * softmax_probs_1 + epsilon / 2.0
        
        if action_1[trial] != -1:
            log_loss -= np.log(probs_1[action_1[trial]] + eps)
            last_action_1 = action_1[trial]
            
        state_idx = state[trial]
        if state_idx == -1: continue
            
        # Stage 2
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2)
        softmax_probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Mixture policy
        probs_2 = (1 - epsilon) * softmax_probs_2 + epsilon / 2.0
        
        a2 = action_2[trial]
        if a2 != -1:
            log_loss -= np.log(probs_2[a2] + eps)
            
            r = reward[trial]
            if r == -1: r = 0
            
            a1 = action_1[trial]
            
            delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
            delta_stage2 = r - q_stage2_mf[state_idx, a2]
            
            q_stage1_mf[a1] += lr * delta_stage1 + lr * lam * delta_stage2
            q_stage2_mf[state_idx, a2] += lr * delta_stage2
            
            q_stage2_mf[state_idx, 1 - a2] *= (1.0 - decay)
            
    return log_loss
```
