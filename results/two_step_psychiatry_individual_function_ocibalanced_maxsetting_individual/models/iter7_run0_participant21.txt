Here are three new cognitive models for the two-step task.

### Model 1: Subjective Reward Valuation Model
This model tests the hypothesis that the participant perceives receiving 0 coins not as a neutral event, but as a loss. By parameterizing the "negative value" of a 0-coin outcome, the model allows the agent to learn stronger avoidance of unrewarded options than standard Q-learning (which typically converges to 0 for unrewarded options).

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Subjective Reward Valuation Model.
    
    This model posits that the participant perceives a 0-coin outcome not as neutral,
    but as a loss (negative value). This is parameterized by `zero_val`.
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-values.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2.
    w: [0, 1] Weight for Model-Based control (0=MF, 1=MB).
    stickiness: [0, 5] Choice stickiness for Stage 1.
    zero_val: [0, 5] The negative magnitude assigned to a 0 reward (perceived as -zero_val).
    """
    learning_rate, beta_1, beta_2, w, stickiness, zero_val = model_parameters
    n_trials = len(action_1)
    
    # Static transition matrix as per instructions
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 planets, 2 aliens
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        # Model-Based Value: Expected max value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Calculate effective reward: 1 if coin, -zero_val if no coin
        r_eff = 1.0 if reward[trial] == 1 else -zero_val
        
        # TD Error Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # TD Error Stage 2
        delta_stage2 = r_eff - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Probability Distortion Model (Prospect Theory)
This model investigates if the participant systematically distorts the transition probabilities (0.7 vs 0.3) used in Model-Based planning. It applies a Prospect Theory weighting function controlled by `gamma`. If `gamma < 1`, the agent overweights the rare transition (0.3), potentially explaining behavior where they choose a "bad" spaceship hoping for a rare jump to a "good" planet.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Probability Distortion Model (Prospect Theory).
    
    The participant distorts the transition probabilities (0.7/0.3) using a 
    Prospect Theory weighting function during Model-Based valuation.
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-values.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2.
    w: [0, 1] Weight for Model-Based control.
    stickiness: [0, 5] Choice stickiness for Stage 1.
    gamma: [0, 5] Probability weighting parameter (gamma < 1 overweights rare events).
    """
    learning_rate, beta_1, beta_2, w, stickiness, gamma = model_parameters
    n_trials = len(action_1)
    
    # Base transition matrix
    base_transitions = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Apply probability weighting: w(p) = p^gamma / (p^gamma + (1-p)^gamma)^(1/gamma)
    # We apply this to the transition matrix used for MB calculation
    def weight_prob(p, g):
        # Add epsilon to avoid divide by zero if g is very high/low
        return (p**g) / ((p**g + (1-p)**g)**(1/g))
        
    distorted_transitions = weight_prob(base_transitions, gamma)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # MB uses distorted transitions
        q_stage1_mb = distorted_transitions @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Dynamic Transition Learning with Planet Stickiness
This model builds on the previous "Best Model" (Dynamic Transition Learning) by adding a specific heuristic: "Planet Stickiness". It hypothesizes that the participant has a bias to return to the *Planet* they just visited (State Stickiness), distinct from repeating the *Spaceship* choice (Action Stickiness). This explains behavior where a participant might switch spaceships in an attempt to get back to a preferred or recently visited planet.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Transition Learning with Planet Stickiness.
    
    Combines learning of the transition matrix with a bias to return to the 
    previously visited planet (Planet Stickiness).
    
    Parameters:
    lr_reward: [0, 1] Learning rate for reward Q-values.
    lr_trans: [0, 1] Learning rate for transition probabilities.
    beta_1: [0, 10] Inverse temperature for Stage 1.
    beta_2: [0, 10] Inverse temperature for Stage 2.
    w: [0, 1] Weight for Model-Based control.
    stick_action: [0, 5] Stickiness for Spaceship choice (Stage 1).
    stick_planet: [0, 5] Stickiness for Planet (State) in MB valuation.
    """
    lr_reward, lr_trans, beta_1, beta_2, w, stick_action, stick_planet = model_parameters
    n_trials = len(action_1)
    
    # Initialize transitions (flat prior 0.5)
    learned_transitions = np.array([[0.5, 0.5], [0.5, 0.5]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_planet = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        # Calculate Max Q for Stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Add Planet Stickiness to the values used by MB system
        # If we were at a planet last trial, we bias the value of that planet
        mb_values = max_q_stage2.copy()
        if last_planet != -1:
            mb_values[last_planet] += stick_planet
            
        q_stage1_mb = learned_transitions @ mb_values
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stick_action
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        act = action_1[trial]
        # Transition learning
        learned_transitions[act, state_idx] += lr_trans * (1 - learned_transitions[act, state_idx])
        learned_transitions[act, 1 - state_idx] += lr_trans * (0 - learned_transitions[act, 1 - state_idx])
        
        # MF Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_reward * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_reward * delta_stage2
        
        last_action_1 = action_1[trial]
        last_planet = state_idx
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
