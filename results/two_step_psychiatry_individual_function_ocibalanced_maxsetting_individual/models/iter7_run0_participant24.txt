Here are 3 new cognitive models for the two-step task.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model 1: Model-Based with Coupled Transition Learning.
    
    This model assumes the participant utilizes a symmetric "coupled" transition 
    model. Instead of learning P(X|A) and P(Y|U) independently, the model learns 
    a single 'common transition probability' (p_common) that governs both 
    spaceships (i.e., A->X and U->Y are both governed by p_common). This enforces 
    the structural symmetry of the task.
    
    Parameters:
    lr_val: [0, 1] - Learning rate for Stage 2 Q-values (Aliens).
    lr_common: [0, 1] - Learning rate for the common transition probability.
    beta: [0, 10] - Inverse temperature.
    pers: [0, 5] - Perseverance bonus for Stage 1 choice.
    """
    lr_val, lr_common, beta, pers = model_parameters
    n_trials = len(action_1)
    
    # Q-values for Stage 2 (State 0: Aliens 0,1; State 1: Aliens 0,1)
    q_stage2 = np.zeros((2, 2))
    
    # Common transition probability (Prob of A->X and U->Y)
    # Initialize at 0.5 (uncertainty)
    p_common = 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_next = int(state[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])
        
        # --- Stage 1 Policy ---
        # Construct transition matrix from p_common
        # A (0) -> X (0) is p_common, A -> Y (1) is 1-p_common
        # U (1) -> Y (1) is p_common, U -> X (0) is 1-p_common
        trans_probs = np.array([[p_common, 1.0 - p_common], 
                                [1.0 - p_common, p_common]])
        
        # Model-Based Values: T * max(Q_stage2)
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = trans_probs @ max_q_s2
        
        # Add perseverance
        q_net_s1 = q_mb.copy()
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers
            
        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # 1. Update Stage 2 Q-values
        delta_2 = r - q_stage2[s_next, a2]
        q_stage2[s_next, a2] += lr_val * delta_2
        
        # 2. Update Common Transition Probability
        # Determine if transition was "common" (0->0 or 1->1) or "rare" (0->1 or 1->0)
        is_common_transition = (a1 == s_next)
        
        if is_common_transition:
            # Observed common transition, update p_common towards 1
            p_common += lr_common * (1.0 - p_common)
        else:
            # Observed rare transition, update p_common towards 0
            p_common += lr_common * (0.0 - p_common)
            
        last_action_1 = a1
        
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model 2: Model-Based with Choice Kernel (Trace Perseverance).
    
    Instead of a simple "stick to last choice" perseverance, this model maintains 
    a "Choice Kernel" that integrates past choices with an exponential decay.
    This allows for longer-term habit formation or stickiness that fades over time.
    
    Parameters:
    lr_val: [0, 1] - Learning rate for Stage 2 Q-values.
    lr_trans: [0, 1] - Learning rate for transition probabilities.
    beta: [0, 10] - Inverse temperature.
    ck_decay: [0, 1] - Decay rate of the choice kernel (1 = only last choice, 0 = no decay).
    ck_weight: [0, 5] - Weight of the choice kernel in Stage 1 decision.
    """
    lr_val, lr_trans, beta, ck_decay, ck_weight = model_parameters
    n_trials = len(action_1)
    
    q_stage2 = np.zeros((2, 2))
    trans_probs = np.ones((2, 2)) * 0.5
    choice_kernel = np.zeros(2)
    
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_next = int(state[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])
        
        # --- Stage 1 Policy ---
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = trans_probs @ max_q_s2
        
        # Combine MB value and Choice Kernel
        logits_s1 = beta * q_mb + ck_weight * choice_kernel
        
        # Softmax Stage 1 (using subtraction for numerical stability)
        exp_q1 = np.exp(logits_s1 - np.max(logits_s1)) 
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # 1. Update Stage 2 Q-values
        q_stage2[s_next, a2] += lr_val * (r - q_stage2[s_next, a2])
        
        # 2. Update Transition Probabilities (Independent)
        trans_probs[a1, s_next] += lr_trans * (1.0 - trans_probs[a1, s_next])
        trans_probs[a1, 1 - s_next] += lr_trans * (0.0 - trans_probs[a1, 1 - s_next])
        
        # 3. Update Choice Kernel
        # Decay trace for both options
        choice_kernel *= (1.0 - ck_decay)
        # Add 1 to the chosen option
        choice_kernel[a1] += 1.0
        
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model 3: Model-Based with Asymmetric Transition Learning.
    
    This model assumes that the participant updates their internal model of the 
    spaceship-planet transitions differently depending on whether the trial 
    resulted in a reward or not. Rewarded trials might reinforce the transition 
    belief more strongly (or differently) than unrewarded trials.
    
    Parameters:
    lr_val: [0, 1] - Learning rate for Stage 2 Q-values.
    lr_trans_pos: [0, 1] - Learning rate for transitions after a Reward (1).
    lr_trans_neg: [0, 1] - Learning rate for transitions after No Reward (0).
    beta: [0, 10] - Inverse temperature.
    pers: [0, 5] - Perseverance bonus.
    """
    lr_val, lr_trans_pos, lr_trans_neg, beta, pers = model_parameters
    n_trials = len(action_1)
    
    q_stage2 = np.zeros((2, 2))
    trans_probs = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_next = int(state[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])
        
        # --- Stage 1 Policy ---
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = trans_probs @ max_q_s2
        
        q_net_s1 = q_mb.copy()
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers
            
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # 1. Update Stage 2 Q-values
        q_stage2[s_next, a2] += lr_val * (r - q_stage2[s_next, a2])
        
        # 2. Update Transition Probabilities with Asymmetric Learning Rate
        # Select transition learning rate based on reward outcome
        if r > 0:
            lr_curr = lr_trans_pos
        else:
            lr_curr = lr_trans_neg
            
        trans_probs[a1, s_next] += lr_curr * (1.0 - trans_probs[a1, s_next])
        trans_probs[a1, 1 - s_next] += lr_curr * (0.0 - trans_probs[a1, 1 - s_next])
        
        last_action_1 = a1
        
    return log_loss
```
