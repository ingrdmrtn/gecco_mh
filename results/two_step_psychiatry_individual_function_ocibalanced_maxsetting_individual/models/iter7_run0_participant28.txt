def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Transition-Dependent Eligibility Traces and Stickiness.
    
    This model modulates the eligibility trace (lambda) based on whether the 
    Stage 1 -> Stage 2 transition was common or rare. This allows the model-free 
    system to be sensitive to the transition structure: the agent may reduce 
    credit assignment to the Stage 1 choice if the transition was "surprising" 
    (rare). It also includes choice stickiness to capture perseveration.

    Parameters:
    lr: [0, 1] Learning rate.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] Model-based weight.
    lam_c: [0, 1] Eligibility trace for common transitions.
    lam_r: [0, 1] Eligibility trace for rare transitions.
    stick: [0, 5] Stickiness parameter (choice perseveration).
    """
    lr, beta_1, beta_2, w, lam_c, lam_r, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1 

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to logits
        logits_1 = beta_1 * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) 
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Determine if transition was common or rare
        # A0->S0 and A1->S1 are common (match index)
        is_common = (action_1[trial] == state_idx)
        lam = lam_c if is_common else lam_r
        
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Eligibility trace update for Stage 1 using transition-dependent lambda
        q_stage1_mf[action_1[trial]] += lr * lam * delta_stage2
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Counterfactual Updating and Stickiness.
    
    This model combines counterfactual learning with choice stickiness. 
    It assumes the participant updates the unchosen alien in Stage 2 as if 
    it would have yielded the opposite reward (1-R), helping to track the 
    anticorrelated structure. Instead of decay, it uses stickiness to explain 
    persistence in Stage 1 choices.
    
    Parameters:
    lr: [0, 1] Learning rate for chosen option.
    lr_cf: [0, 1] Learning rate for unchosen option (counterfactual).
    stick: [0, 5] Stickiness parameter.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] Model-based weight.
    lam: [0, 1] Eligibility trace.
    """
    lr, lr_cf, stick, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += stick
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        q_stage1_mf[action_1[trial]] += lr * lam * delta_stage2
        
        # Counterfactual Update
        unchosen_a2 = 1 - action_2[trial]
        cf_reward = 1 - reward[trial]
        delta_cf = cf_reward - q_stage2_mf[state_idx, unchosen_a2]
        q_stage2_mf[state_idx, unchosen_a2] += lr_cf * delta_cf
        
        prev_a1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Independent Decay for Stage 1 and Stage 2.
    
    This model separates the decay (forgetting) processes for the abstract 
    Stage 1 choices (Spaceships) and the concrete Stage 2 values (Aliens).
    This allows the agent to forget cached model-free preferences at a different 
    rate than the changing alien reward probabilities.
    
    Parameters:
    lr: [0, 1] Learning rate.
    decay_1: [0, 1] Decay rate for Stage 1 Q-values.
    decay_2: [0, 1] Decay rate for Stage 2 Q-values.
    beta_1: [0, 10] Stage 1 inverse temperature.
    beta_2: [0, 10] Stage 2 inverse temperature.
    w: [0, 1] Model-based weight.
    lam: [0, 1] Eligibility trace.
    """
    lr, decay_1, decay_2, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Independent Decay
        q_stage1_mf *= (1 - decay_1)
        q_stage2_mf *= (1 - decay_2)
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        q_stage1_mf[action_1[trial]] += lr * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
