Here are three new cognitive models for the two-step decision-making task.

### Model 1: Subjective Transition Belief Model
This model tests the hypothesis that the participant's internal model of the transition probabilities differs from the objective truth (0.7/0.3). Instead of fixing the Model-Based transition matrix, it learns a subjective probability parameter `trans_prob`.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Subjective Transition Belief Model.
    
    The agent employs a hybrid Model-Based/Model-Free strategy but holds a 
    subjective belief about the transition probabilities (`trans_prob`) that 
    may differ from the objective truth (0.7).
    
    Parameters:
    - lr: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Weight of Model-Based system (0=MF, 1=MB).
    - stick: [-5, 5] Choice stickiness/perseveration.
    - trans_prob: [0, 1] Subjective probability of common transitions.
    """
    lr, beta, w, stick, trans_prob = model_parameters
    n_trials = len(action_1)
    
    # Subjective transition matrix based on parameter
    # Row 0: Action 0 -> [prob(State0), prob(State1)]
    # Row 1: Action 1 -> [prob(State0), prob(State1)]
    # Assuming symmetry in belief for A and U
    transition_matrix = np.array([[trans_prob, 1-trans_prob], 
                                  [1-trans_prob, trans_prob]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # MF values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # MF values for stage 2 (State x Action)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value: Expected max value of next stage using subjective matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to logits
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stick
            
        # Softmax Stage 1
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning / Updates ---
        # Stage 1 Update (TD)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        # Stage 2 Update (Reward Prediction Error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Counterfactual Stage 2 Learning Model
This model assumes the participant engages in counterfactual learning ("fictitious play") at the second stage. If they choose an alien and receive a reward (or no reward), they update the *unchosen* alien's value towards the opposite outcome, scaled by `lr_cf`.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Counterfactual Stage 2 Learning Model.
    
    In addition to standard updates, the agent updates the unchosen alien in Stage 2
    assuming a counterfactual outcome (opposite of received reward).
    
    Parameters:
    - lr: [0, 1] Learning rate for chosen actions.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - stick: [-5, 5] Stickiness.
    - lr_cf: [0, 1] Learning rate for the unchosen (counterfactual) option.
    """
    lr, beta, w, stick, lr_cf = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stick
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        # Chosen update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Counterfactual update for unchosen option
        # Assume if reward=1, other was 0. If reward=0, other was 1.
        unchosen_a2 = 1 - action_2[trial]
        cf_reward = 1 - reward[trial]
        delta_cf = cf_reward - q_stage2_mf[state_idx, unchosen_a2]
        q_stage2_mf[state_idx, unchosen_a2] += lr_cf * delta_cf
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Transition-Dependent Stickiness Model
This model posits that the agent's tendency to repeat their Stage 1 choice (stickiness) is modulated by the "surprise" of the previous transition. A common transition might reinforce the habit (`stick_common`), while a rare transition might disrupt it (`stick_rare`).

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Transition-Dependent Stickiness Model.
    
    The degree of perseveration (stickiness) depends on whether the previous 
    transition was 'Common' (expected) or 'Rare' (unexpected).
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - stick_common: [-5, 5] Stickiness after a common transition.
    - stick_rare: [-5, 5] Stickiness after a rare transition.
    """
    lr, beta, w, stick_common, stick_rare = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_transition_common = True # Default assumption for first trial
    
    for trial in range(n_trials):
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            # Apply stickiness based on previous transition type
            stick_val = stick_common if last_transition_common else stick_rare
            logits[last_action_1] += stick_val
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Determine if this transition was common or rare for next trial
        # Common: Action 0 -> State 0, Action 1 -> State 1
        is_common = (action_1[trial] == state[trial])
        
        last_action_1 = action_1[trial]
        last_transition_common = is_common

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
