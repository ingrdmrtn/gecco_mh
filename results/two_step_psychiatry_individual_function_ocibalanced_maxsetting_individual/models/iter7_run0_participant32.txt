Here are three new cognitive models for the two-step task.

### Model 1: Hybrid Model (MB+MF) with Passive Decay and Outcome-Dependent Perseveration
This model re-introduces the Model-Free (MF) component (`w`) to the "best model so far" configuration. It tests whether a mixture of strategies (Hybrid) is better than Pure Model-Based when combined with value decay and nuanced perseveration (Win-Stay vs. Lose-Stay/Shift). It uses a simplified TD(1)-style update for the MF component at Stage 1, driven directly by the final reward.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model (MB+MF) with Passive Decay and Outcome-Dependent Perseveration.
    
    Combines Model-Based and Model-Free controls.
    - MB uses transition matrix and Stage 2 values.
    - MF learns Stage 1 values directly from Reward (TD(1)-like).
    - Unchosen Stage 2 values decay to 0.5.
    - Perseveration bias depends on the previous outcome (Win-Stay, Lose-Stay/Shift).
    
    Parameters:
    - lr: [0, 1] Learning rate for value updates.
    - beta1: [0, 10] Inverse temperature for Stage 1.
    - beta2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight (1 = Pure MB, 0 = Pure MF).
    - decay: [0, 1] Decay rate for unchosen Stage 2 options.
    - pers_win: [0, 5] Perseveration bonus after reward.
    - pers_loss: [0, 5] Perseveration bonus after no reward.
    """
    lr, beta1, beta2, w, decay, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_mf1 = np.ones(2) * 0.5
    q_mf2 = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    last_reward = -1
    
    for trial in range(n_trials):
        # Stage 1 Choice
        # MB Component
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        # Net Q-value
        q_net1 = w * q_mb1 + (1 - w) * q_mf1
        
        logits_1 = beta1 * q_net1
        
        # Perseveration
        if last_action_1 != -1:
            if last_reward == 1:
                logits_1[last_action_1] += pers_win
            else:
                logits_1[last_action_1] += pers_loss
                
        # Softmax Stage 1
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        a1 = action_1[trial]
        s_idx = state[trial]
        
        # Stage 2 Choice
        logits_2 = beta2 * q_mf2[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]
        
        # Updates
        # Stage 2 MF Update (Standard Q-learning)
        pe2 = r - q_mf2[s_idx, a2]
        q_mf2[s_idx, a2] += lr * pe2
        
        # Stage 1 MF Update (TD(1) style - driven by reward)
        pe1 = r - q_mf1[a1]
        q_mf1[a1] += lr * pe1
        
        # Decay unchosen Stage 2
        unchosen_a2 = 1 - a2
        q_mf2[s_idx, unchosen_a2] = (1 - decay) * q_mf2[s_idx, unchosen_a2] + decay * 0.5
        
        other_state = 1 - s_idx
        q_mf2[other_state, :] = (1 - decay) * q_mf2[other_state, :] + decay * 0.5
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Based with Asymmetric Learning and Decay
This model refines the Pure Model-Based approach by allowing asymmetric learning rates for positive (`lr_pos`) and negative (`lr_neg`) prediction errors. This captures potential biases in how the participant processes wins versus losses (optimism/pessimism bias) while maintaining the decay and perseveration mechanisms.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Pure Model-Based with Asymmetric Learning, Decay, and Outcome-Dependent Perseveration.
    
    Pure MB model (w=1) that updates Stage 2 values with different learning rates
    for positive and negative prediction errors. Includes decay and specific perseveration.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive PE.
    - lr_neg: [0, 1] Learning rate for negative PE.
    - beta1: [0, 10] Inverse temperature for Stage 1.
    - beta2: [0, 10] Inverse temperature for Stage 2.
    - decay: [0, 1] Decay rate for unchosen options.
    - pers_win: [0, 5] Perseveration bonus after reward.
    - pers_loss: [0, 5] Perseveration bonus after no reward.
    """
    lr_pos, lr_neg, beta1, beta2, decay, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf2 = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    last_reward = -1
    
    for trial in range(n_trials):
        # Stage 1 (MB)
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        logits_1 = beta1 * q_mb1
        
        if last_action_1 != -1:
            if last_reward == 1:
                logits_1[last_action_1] += pers_win
            else:
                logits_1[last_action_1] += pers_loss
        
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        a1 = action_1[trial]
        s_idx = state[trial]
        
        # Stage 2
        logits_2 = beta2 * q_mf2[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]
        
        # Update
        pe = r - q_mf2[s_idx, a2]
        if pe >= 0:
            q_mf2[s_idx, a2] += lr_pos * pe
        else:
            q_mf2[s_idx, a2] += lr_neg * pe
            
        # Decay
        unchosen_a2 = 1 - a2
        q_mf2[s_idx, unchosen_a2] = (1 - decay) * q_mf2[s_idx, unchosen_a2] + decay * 0.5
        
        other_state = 1 - s_idx
        q_mf2[other_state, :] = (1 - decay) * q_mf2[other_state, :] + decay * 0.5
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with Separate Stage Learning Rates
This model hypothesizes that the participant learns the simple Stage 1 "habit" and the complex Stage 2 "values" at different rates. It uses `lr_stg1` for the Model-Free update at Stage 1 and `lr_stg2` for value updates at Stage 2 (which informs the Model-Based controller). This decouples the speed of habit formation from value tracking.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Stage Learning Rates and Decay.
    
    Allows different learning rates for the habitual Stage 1 (MF) and the 
    value-based Stage 2 (used by MB). This decouples the speed of habit formation
    from the speed of tracking changing reward probabilities.
    
    Parameters:
    - lr_stg1: [0, 1] Learning rate for Stage 1 MF values.
    - lr_stg2: [0, 1] Learning rate for Stage 2 values.
    - beta1: [0, 10] Inverse temperature for Stage 1.
    - beta2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight.
    - decay: [0, 1] Decay rate for unchosen Stage 2 options.
    - pers: [0, 5] General perseveration bonus.
    """
    lr_stg1, lr_stg2, beta1, beta2, w, decay, pers = model_parameters
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.ones(2) * 0.5
    q_mf2 = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        q_net1 = w * q_mb1 + (1 - w) * q_mf1
        
        logits_1 = beta1 * q_net1
        
        if last_action_1 != -1:
            logits_1[last_action_1] += pers
            
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        a1 = action_1[trial]
        s_idx = state[trial]
        
        # Stage 2
        logits_2 = beta2 * q_mf2[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]
        
        # Updates
        # Stage 2 (Bandit) - uses lr_stg2
        pe2 = r - q_mf2[s_idx, a2]
        q_mf2[s_idx, a2] += lr_stg2 * pe2
        
        # Stage 1 (Habit/MF) - uses lr_stg1
        pe1 = r - q_mf1[a1]
        q_mf1[a1] += lr_stg1 * pe1
        
        # Decay unchosen Stage 2
        unchosen_a2 = 1 - a2
        q_mf2[s_idx, unchosen_a2] = (1 - decay) * q_mf2[s_idx, unchosen_a2] + decay * 0.5
        
        other_state = 1 - s_idx
        q_mf2[other_state, :] = (1 - decay) * q_mf2[other_state, :] + decay * 0.5
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
