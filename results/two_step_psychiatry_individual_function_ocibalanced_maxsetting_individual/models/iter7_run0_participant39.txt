Here are three new cognitive models for the two-step decision task, designed to capture specific mechanisms of learning and decision-making that might explain the participant's behavior.

### Model 1: Hybrid MB/MF with Subjective Transition Probability
This model tests the hypothesis that the participant's "Model-Based" system relies on a distorted belief about the transition probabilities (e.g., they might believe the transitions are more deterministic or more random than they actually are), rather than the true 0.7/0.3 probabilities.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Subjective Transition Probability.
    
    The participant uses a fixed subjective probability for the 'common' transition
    in the Model-Based calculation, which may differ from the true 0.7. This allows
    the model to capture if the participant mis-estimates the transition structure.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta_1: Inverse temperature stage 1 [0,10]
    - beta_2: Inverse temperature stage 2 [0,10]
    - w: Mixing weight (0=MF, 1=MB) [0,1]
    - stick: Choice stickiness [0,5]
    - prob_common: Subjective probability of common transition [0,1]
    """
    lr, beta_1, beta_2, w, stick, prob_common = model_parameters
    n_trials = len(action_1)
    
    # Subjective transition matrix based on parameter
    # 0->0 and 1->1 are common
    transition_matrix = np.array([[prob_common, 1-prob_common], 
                                  [1-prob_common, prob_common]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 planets, 2 aliens
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Expected max value of next stage using subjective probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Stickiness
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
            
        # Softmax
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        log_loss -= np.log(probs_1[a1] + eps)
        
        prev_action_1 = a1
        
        # --- Stage 2 Choice ---
        a2 = action_2[trial]
        if a2 == -1: # Handle missing data
            continue
            
        state_idx = state[trial]
        qs_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 Update (TD)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update (TD)
        # Using SARSA-like update for MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
    return log_loss
```

### Model 2: Hybrid MB/MF with Rare Transition Penalty
This model introduces a penalty mechanism for rare transitions. If a participant experiences a rare transition, they might treat it as an aversive "error" or "bad luck" event, penalizing the action that led to it within the Model-Free system, regardless of the subsequent reward. This captures a specific form of risk/surprise aversion.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Rare Transition Penalty.
    
    A penalty is applied to the Model-Free value update of the chosen spaceship
    if a rare transition occurs. This discourages repeating actions that lead 
    to surprising outcomes, independent of the reward.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta_1: Inverse temperature stage 1 [0,10]
    - beta_2: Inverse temperature stage 2 [0,10]
    - w: Mixing weight (0=MF, 1=MB) [0,1]
    - stick: Choice stickiness [0,5]
    - rare_penalty: Penalty subtracted from update on rare transitions [0,2]
    """
    lr, beta_1, beta_2, w, stick, rare_penalty = model_parameters
    n_trials = len(action_1)
    
    # True transition matrix for MB part
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        log_loss -= np.log(probs_1[a1] + eps)
        prev_action_1 = a1
        
        # --- Stage 2 Choice ---
        a2 = action_2[trial]
        if a2 == -1:
            continue
            
        state_idx = state[trial]
        qs_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Learning ---
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Check transition type
        # Common: 0->0, 1->1. Rare: 0->1, 1->0
        is_rare = (a1 == 0 and state_idx == 1) or (a1 == 1 and state_idx == 0)
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        if is_rare:
            # Apply penalty to the delta, effectively reducing the value
            delta_stage1 -= rare_penalty
            
        q_stage1_mf[a1] += lr * delta_stage1
        
    return log_loss
```

### Model 3: Hybrid MB/MF with Separate Learning Rates for Transition Types
This model hypothesizes that the participant updates their Model-Free values differently depending on whether the transition was common or rare. For instance, they might ignore rare transitions (treating them as noise) or learn from them aggressively.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Separate Learning Rates for Common/Rare Transitions.
    
    The Model-Free update for the first stage uses different learning rates depending
    on whether the transition was common or rare.
    
    Parameters:
    - lr_c: Learning rate for common transitions (and stage 2) [0,1]
    - lr_r: Learning rate for rare transitions (stage 1 only) [0,1]
    - beta_1: Inverse temperature stage 1 [0,10]
    - beta_2: Inverse temperature stage 2 [0,10]
    - w: Mixing weight (0=MF, 1=MB) [0,1]
    - stick: Choice stickiness [0,5]
    """
    lr_c, lr_r, beta_1, beta_2, w, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = action_1[trial]
        log_loss -= np.log(probs_1[a1] + eps)
        prev_action_1 = a1
        
        # --- Stage 2 Choice ---
        a2 = action_2[trial]
        if a2 == -1:
            continue
            
        state_idx = state[trial]
        qs_2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 uses lr_c (standard learning rate)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_c * delta_stage2
        
        # Stage 1
        is_rare = (a1 == 0 and state_idx == 1) or (a1 == 1 and state_idx == 0)
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Select learning rate based on transition type
        current_lr = lr_r if is_rare else lr_c
        
        q_stage1_mf[a1] += current_lr * delta_stage1
        
    return log_loss
```
