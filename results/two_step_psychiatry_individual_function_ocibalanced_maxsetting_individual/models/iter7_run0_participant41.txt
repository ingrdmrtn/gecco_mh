import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner with Subjective Transition Belief.
    
    A Hybrid Model-Based/Model-Free agent. Unlike the standard model which assumes 
    perfect knowledge of the transition probabilities (0.7/0.3), this model estimates 
    a subjective probability `p_common` for the common transition. This tests if the 
    participant's behavior is driven by an incorrect or biased internal model of the 
    spaceship-planet transitions.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values (MF and Stage 2).
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - Weighting parameter (0=Pure MF, 1=Pure MB).
    p_common: [0,1] - Subjective probability of the common transition (e.g., S0->P0).
    """
    learning_rate, beta_1, beta_2, w, p_common = model_parameters
    n_trials = len(action_1)
    
    # Construct subjective transition matrix based on parameter
    # Row 0: S0 -> [P0, P1], Row 1: S1 -> [P0, P1]
    # S0 commonly goes to P0; S1 commonly goes to P1.
    transition_matrix = np.array([[p_common, 1-p_common], [1-p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        
        # --- Stage 1 Choice ---
        # Model-Based Value: Expected max value of next stage using subjective transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if np.isnan(action_1[trial]):
            p_choice_1[trial] = 0.5
            continue
            
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if np.isnan(action_2[trial]):
            p_choice_2[trial] = 0.5
            continue
            
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # --- Updates ---
        # Model-Free TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Risk-Sensitive Q-Learning with Outcome-Dependent Stickiness.
    
    A pure Model-Free agent that learns asymmetrically from positive and negative 
    prediction errors (Risk Sensitivity). It also includes outcome-dependent 
    perseveration (Stickiness), where the tendency to repeat a choice differs 
    depending on whether the previous trial was rewarded or not.
    
    Parameters:
    alpha_pos: [0,1] - Learning rate for positive prediction errors.
    alpha_neg: [0,1] - Learning rate for negative prediction errors.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    sticky_win: [0,10] - Stickiness bonus applied to previous choice if Reward=1.
    sticky_lose: [0,10] - Stickiness bonus applied to previous choice if Reward=0.
    """
    alpha_pos, alpha_neg, beta_1, beta_2, sticky_win, sticky_lose = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_reward = -1
    
    for trial in range(n_trials):
        
        # --- Stage 1 Choice ---
        logits_1 = beta_1 * q_stage1
        
        # Apply outcome-dependent stickiness
        if prev_a1 != -1:
            if prev_reward == 1:
                logits_1[prev_a1] += sticky_win
            else:
                logits_1[prev_a1] += sticky_lose
                
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if np.isnan(action_1[trial]):
            p_choice_1[trial] = 0.5
            prev_a1 = -1
            continue
            
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        prev_a1 = a1
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if np.isnan(action_2[trial]):
            p_choice_2[trial] = 0.5
            continue
            
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        prev_reward = r
        
        # --- Updates ---
        # Stage 1 Update (TD-0)
        delta1 = q_stage2[state_idx, a2] - q_stage1[a1]
        lr1 = alpha_pos if delta1 > 0 else alpha_neg
        q_stage1[a1] += lr1 * delta1
        
        # Stage 2 Update
        delta2 = r - q_stage2[state_idx, a2]
        lr2 = alpha_pos if delta2 > 0 else alpha_neg
        q_stage2[state_idx, a2] += lr2 * delta2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dual-Learning Rate Hybrid Model.
    
    A Hybrid agent that uses distinct learning rates for the two stages.
    This captures the potential difference in plasticity between the high-level 
    spaceship preference (Stage 1 MF) and the specific alien reward values (Stage 2).
    
    Parameters:
    lr_stage1: [0,1] - Learning rate for Stage 1 Model-Free values.
    lr_stage2: [0,1] - Learning rate for Stage 2 values.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - Weighting parameter (0=Pure MF, 1=Pure MB).
    """
    lr_stage1, lr_stage2, beta_1, beta_2, w = model_parameters
    n_trials = len(action_1)
    
    # Fixed objective transition matrix
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        
        # --- Stage 1 Choice ---
        # MB Component
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Mix
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if np.isnan(action_1[trial]):
            p_choice_1[trial] = 0.5
            continue
            
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if np.isnan(action_2[trial]):
            p_choice_2[trial] = 0.5
            continue
            
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # --- Updates ---
        # Stage 1 MF Update using lr_stage1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Stage 2 Update using lr_stage2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
