Here are the three proposed cognitive models.

### Cognitive Model 1
**Hypothesis:** The participant acts as a Model-Free learner who updates values differently for the two stages (spaceship vs. alien) and uses an eligibility trace to propagate reward information directly to the first stage choice.
**Key Mechanism:** Separate learning rates (`alpha_1`, `alpha_2`) allow for distinct plasticity in spaceship vs. alien preferences. The eligibility trace (`lambda_elig`) allows the reward at the end of the trial to reinforce the spaceship choice, bridging the gap between the two stages.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free TD(lambda) with Separate Stage Learning Rates.

    Parameters:
    - alpha_1: [0, 1] Learning rate for Stage 1 (Spaceship choice).
    - alpha_2: [0, 1] Learning rate for Stage 2 (Alien choice).
    - lambda_elig: [0, 1] Eligibility trace parameter (weight of Stage 2 RPE on Stage 1 update).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - stickiness: [0, 5] Bonus for repeating the last Stage 1 choice.
    """
    alpha_1, alpha_2, lambda_elig, beta_1, beta_2, stickiness = model_parameters
    n_trials = len(action_1)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Q-values
    q_stage1_mf = np.zeros(2)      # Values for Spaceships 0 and 1
    q_stage2_mf = np.zeros((2, 2)) # Values for Aliens (Planet 0: [Alien0, Alien1], Planet 1: ...)

    last_action_1 = -1

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Decision ---
        q_net_1 = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
        
        exp_q1 = np.exp(beta_1 * (q_net_1 - np.max(q_net_1)))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        current_state = state[trial]

        # --- Stage 2 Decision ---
        q_net_2 = q_stage2_mf[current_state]
        exp_q2 = np.exp(beta_2 * (q_net_2 - np.max(q_net_2)))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        # 1. Prediction Error at Transition (Stage 1 -> Stage 2)
        # Value of the state arrived at (max Q or Q of chosen? Standard SARSA uses Q of chosen)
        # Here we use Q of chosen action at stage 2 for SARSA-like update
        q_next = q_stage2_mf[current_state, action_2[trial]]
        delta_stage1 = q_next - q_stage1_mf[action_1[trial]]
        
        # Update Stage 1 Value
        q_stage1_mf[action_1[trial]] += alpha_1 * delta_stage1

        # 2. Prediction Error at Outcome (Stage 2 -> Reward)
        delta_stage2 = reward[trial] - q_stage2_mf[current_state, action_2[trial]]
        
        # Update Stage 2 Value
        q_stage2_mf[current_state, action_2[trial]] += alpha_2 * delta_stage2
        
        # 3. Eligibility Trace Update for Stage 1
        # The Stage 1 choice also learns from the Stage 2 RPE, scaled by lambda
        q_stage1_mf[action_1[trial]] += alpha_1 * lambda_elig * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2
**Hypothesis:** The participant employs a Hybrid strategy (mixing Model-Based and Model-Free) but exhibits significant perseveration (stickiness) at **both** stages of the task.
**Key Mechanism:** While standard models often only include stickiness for the spaceship choice, this model includes `stick_2`, a perseveration bonus for the alien choice. This accounts for streaks where the participant repeatedly chooses the same alien on a specific planet regardless of reward.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Learner (MB+MF) with Dual-Stage Stickiness.

    Parameters:
    - alpha: [0, 1] Learning rate for MF values (Stage 1 and Stage 2).
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Weight of Model-Based control (0=Pure MF, 1=Pure MB).
    - stick_1: [0, 5] Stickiness for Stage 1 (Spaceship).
    - stick_2: [0, 5] Stickiness for Stage 2 (Alien).
    """
    alpha, beta_1, beta_2, w, stick_1, stick_2 = model_parameters
    n_trials = len(action_1)
    
    # Transition probabilities: A(0)->X(0) 0.7, U(1)->Y(1) 0.7
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Used for both MF choice and MB calculation

    last_action_1 = -1
    last_action_2 = np.array([-1, -1]) # Track last alien choice per planet

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Decision (Hybrid) ---
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness 1
        if last_action_1 != -1:
            q_integrated[last_action_1] += stick_1
            
        exp_q1 = np.exp(beta_1 * (q_integrated - np.max(q_integrated)))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        current_state = state[trial]

        # --- Stage 2 Decision ---
        q_net_2 = q_stage2_mf[current_state].copy()
        
        # Apply Stickiness 2 (specific to the planet)
        if last_action_2[current_state] != -1:
            q_net_2[last_action_2[current_state]] += stick_2
            
        exp_q2 = np.exp(beta_2 * (q_net_2 - np.max(q_net_2)))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        last_action_2[current_state] = action_2[trial]

        # --- Learning ---
        # Update Stage 1 MF
        delta_stage1 = q_stage2_mf[current_state, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += alpha * delta_stage1
        
        # Update Stage 2 MF (which also updates the basis for MB)
        delta_stage2 = reward[trial] - q_stage2_mf[current_state, action_2[trial]]
        q_stage2_mf[current_state, action_2[trial]] += alpha * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3
**Hypothesis:** The participant is a Model-Free learner with asymmetric learning rates for positive and negative prediction errors, combined with a decay mechanism for unchosen options.
**Key Mechanism:** `alpha_pos` and `alpha_neg` allow the model to learn more from rewards (optimism) or omissions (pessimism). `decay_rate` handles non-stationarity by slowly forgetting the value of options that are not currently being sampled, causing values to drift toward zero.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model-Free with Asymmetric Learning and Value Decay.

    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors.
    - alpha_neg: [0, 1] Learning rate for negative prediction errors.
    - decay_rate: [0, 1] Decay rate for unchosen Q-values.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - stickiness: [0, 5] Bonus for repeating the last Stage 1 choice.
    """
    alpha_pos, alpha_neg, decay_rate, beta_1, beta_2, stickiness = model_parameters
    n_trials = len(action_1)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Decision ---
        q_net_1 = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * (q_net_1 - np.max(q_net_1)))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        current_state = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta_2 * (q_stage2_mf[current_state] - np.max(q_stage2_mf[current_state])))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning ---
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[current_state, action_2[trial]] - q_stage1_mf[action_1[trial]]
        alpha_1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[action_1[trial]] += alpha_1 * delta_stage1

        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[current_state, action_2[trial]]
        alpha_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[current_state, action_2[trial]] += alpha_2 * delta_stage2

        # --- Decay ---
        # Decay unchosen options in Stage 1
        unchosen_1 = 1 - action_1[trial]
        q_stage1_mf[unchosen_1] *= (1 - decay_rate)
        
        # Decay unchosen options in Stage 2 (on the current planet)
        unchosen_2 = 1 - action_2[trial]
        q_stage2_mf[current_state, unchosen_2] *= (1 - decay_rate)
        
        # Decay options on the unvisited planet?
        # Standard decay usually applies to all un-updated values, or just unchosen available ones.
        # Here we decay the unvisited planet's values as well to simulate general forgetting.
        other_planet = 1 - current_state
        q_stage2_mf[other_planet, 0] *= (1 - decay_rate)
        q_stage2_mf[other_planet, 1] *= (1 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
