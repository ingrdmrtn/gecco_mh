Here are three new cognitive models for the two-step decision task, designed to explore different mechanisms of learning and perseverance.

### Model 1: Hybrid MB/MF with Asymmetric Learning and Stage 2 Decay
This model hypothesizes that the participant learns differently from positive outcomes (better than expected) versus negative outcomes (worse than expected), captured by `lr_pos` and `lr_neg`. Additionally, it includes a `decay_rate` for unchosen Stage 2 options to handle the slowly drifting alien reward probabilities, allowing the agent to "forget" outdated values.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Asymmetric Learning Rates and Decay.
    
    This model splits the learning rate into positive (lr_pos) and negative (lr_neg)
    prediction error updates. It also decays the Q-values of unchosen Stage 2 
    options to account for drifting reward probabilities.
    
    Parameters:
    lr_pos: [0,1] - Learning rate for positive prediction errors.
    lr_neg: [0,1] - Learning rate for negative prediction errors.
    decay_rate: [0,1] - Decay rate for unchosen Stage 2 options.
    beta: [0,10] - Inverse temperature for softmax.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    stickiness: [0,5] - Choice perseverance bonus for Stage 1.
    """
    lr_pos, lr_neg, decay_rate, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness
        q_net_stick = q_net.copy()
        if prev_action_1 != -1:
            q_net_stick[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Handle missing stage 2 data
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            continue
            
        state_idx = state[trial]
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Stage 1 Update (MF)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        eff_lr_s1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[action_1[trial]] += eff_lr_s1 * delta_stage1
        
        # Stage 2 Update (MF)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        eff_lr_s2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[state_idx, action_2[trial]] += eff_lr_s2 * delta_stage2
        
        # Decay unchosen Stage 2 option
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action_2] *= (1 - decay_rate)
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with Separate Stage Learning Rates and Eligibility Trace
This model acknowledges that Stage 1 (choosing spaceships) involves a stable transition structure, while Stage 2 (choosing aliens) involves dynamic reward probabilities. Therefore, it assigns separate learning rates (`lr_s1`, `lr_s2`) to each stage. It also includes an eligibility trace (`lambda_elig`) to allow the Stage 2 outcome to directly reinforce the Stage 1 choice.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Separate Stage Learning Rates and Eligibility Trace.
    
    Uses distinct learning rates for the stable Stage 1 transitions and the 
    drifting Stage 2 rewards. Includes eligibility traces to propagate 
    Stage 2 reward prediction errors back to Stage 1 choices.
    
    Parameters:
    lr_s1: [0,1] - Learning rate for Stage 1 (MF values).
    lr_s2: [0,1] - Learning rate for Stage 2 (MF values).
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    lambda_elig: [0,1] - Eligibility trace decay parameter.
    stickiness: [0,5] - Choice perseverance bonus for Stage 1.
    """
    lr_s1, lr_s2, beta, w, lambda_elig, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        q_net_stick = q_net.copy()
        if prev_action_1 != -1:
            q_net_stick[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            continue
            
        state_idx = state[trial]
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_s1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_s2 * delta_stage2
        
        # Eligibility Trace: Update Stage 1 based on Stage 2 outcome
        # We use lr_s1 for the update weight on Stage 1 Q-values
        q_stage1_mf[action_1[trial]] += lr_s1 * lambda_elig * delta_stage2
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with Dynamic Transition Learning and Eligibility Trace
This model relaxes the assumption that the agent uses a fixed transition matrix. Instead, the agent learns the transition probabilities (`lr_trans`) between spaceships and planets based on experience. This dynamic belief is used for the Model-Based calculation, combined with standard MF learning and eligibility traces.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Dynamic Transition Learning and Eligibility Trace.
    
    The agent updates their internal model of the transition matrix based on 
    observed transitions, rather than assuming fixed probabilities. 
    
    Parameters:
    learning_rate: [0,1] - Learning rate for reward values (MF).
    lr_trans: [0,1] - Learning rate for transition probabilities.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    lambda_elig: [0,1] - Eligibility trace decay.
    stickiness: [0,5] - Choice perseverance bonus for Stage 1.
    """
    learning_rate, lr_trans, beta, w, lambda_elig, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Initialize transitions with prior knowledge (or uniform 0.5)
    # Using 0.7/0.3 as starting point but allowing drift
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy (MB uses dynamic transition matrix)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        q_net_stick = q_net.copy()
        if prev_action_1 != -1:
            q_net_stick[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            continue
            
        state_idx = state[trial]
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Update Transition Matrix
        # Update probability of reaching observed state from chosen action towards 1
        # Update probability of reaching unobserved state from chosen action towards 0
        observed_s = state_idx
        unobserved_s = 1 - state_idx
        act1 = action_1[trial]
        
        transition_matrix[act1, observed_s] += lr_trans * (1 - transition_matrix[act1, observed_s])
        transition_matrix[act1, unobserved_s] += lr_trans * (0 - transition_matrix[act1, unobserved_s])
        
        # Stage 1 Update (MF)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update (MF)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility Trace
        q_stage1_mf[action_1[trial]] += learning_rate * lambda_elig * delta_stage2
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
```
