### Cognitive Model 1
```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model 1: Hybrid MF/MB with Coupled Transition Learning.
    
    This model combines the Model-Based (MB) "coupled" transition learning mechanism 
    with a Model-Free (MF) habitual system. 
    - The MB system learns a symmetric transition probability (p_common) and 
      computes values based on the learned structure and Stage 2 values.
    - The MF system learns Stage 1 values directly from rewards, ignoring structure.
    - The final Stage 1 choice is a weighted mixture of MB and MF values.
    
    Parameters:
    lr_val: [0, 1] - Learning rate for Stage 2 Q-values (Aliens).
    lr_mf: [0, 1] - Learning rate for Stage 1 MF Q-values (Spaceships).
    lr_common: [0, 1] - Learning rate for the common transition probability.
    w: [0, 1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    beta: [0, 10] - Inverse temperature for softmax.
    pers: [0, 5] - Perseverance bonus for Stage 1 choice.
    """
    lr_val, lr_mf, lr_common, w, beta, pers = model_parameters
    n_trials = len(action_1)

    # Initialize Q-values
    q_stage1_mf = np.zeros(2) # MF values for spaceships
    q_stage2 = np.zeros((2, 2)) # Values for aliens (Planet x Alien)
    
    # Initialize transition probability (p_common)
    # Represents P(Planet X | Spaceship A) and P(Planet Y | Spaceship U)
    p_common = 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        # Skip missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_next = int(state[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])

        # --- Stage 1 Policy ---
        # 1. Model-Based Value Calculation
        # Transition matrix based on coupled p_common
        # T[0,0] = p_common (A->X), T[0,1] = 1-p (A->Y)
        # T[1,0] = 1-p (U->X), T[1,1] = p_common (U->Y)
        trans_probs = np.array([[p_common, 1.0 - p_common], 
                                [1.0 - p_common, p_common]])
        
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = trans_probs @ max_q_s2
        
        # 2. Hybrid Mixture
        q_net_s1 = w * q_mb + (1.0 - w) * q_stage1_mf
        
        # 3. Perseverance
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # --- Learning ---
        # 1. Update Stage 2 Q-values (Standard Q-learning)
        delta_2 = r - q_stage2[s_next, a2]
        q_stage2[s_next, a2] += lr_val * delta_2
        
        # 2. Update Stage 1 MF Q-values (Direct Reward Learning)
        # Note: In 2-step tasks, MF often updates from Reward directly (TD(1)-like)
        delta_mf = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_mf * delta_mf
        
        # 3. Update Coupled Transition Probability
        # If A->X (0->0) or U->Y (1->1), it's a "common" type transition event
        is_common_transition = (a1 == s_next)
        if is_common_transition:
            p_common += lr_common * (1.0 - p_common)
        else:
            p_common += lr_common * (0.0 - p_common)
            
        last_action_1 = a1
        
    return log_loss
```

### Cognitive Model 2
```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model 2: Model-Based with Coupled Transition Learning and Fictive Updating.
    
    This model assumes the participant uses a Model-Based strategy with coupled 
    transition learning, but adds "Fictive" (Counterfactual) updating for the 
    aliens. When the participant chooses an alien and observes a reward (or lack 
    thereof), they also update the value of the *unchosen* alien on that planet, 
    assuming it would have yielded the opposite outcome (or simply updating it 
    towards the unobserved reward).
    
    Parameters:
    lr_val: [0, 1] - Learning rate for chosen Alien.
    lr_fict: [0, 1] - Learning rate for unchosen Alien (Fictive update).
    lr_common: [0, 1] - Learning rate for the common transition probability.
    beta: [0, 10] - Inverse temperature.
    pers: [0, 5] - Perseverance bonus for Stage 1 choice.
    """
    lr_val, lr_fict, lr_common, beta, pers = model_parameters
    n_trials = len(action_1)

    q_stage2 = np.zeros((2, 2))
    p_common = 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_next = int(state[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])

        # --- Stage 1 Policy (Model-Based Coupled) ---
        trans_probs = np.array([[p_common, 1.0 - p_common], 
                                [1.0 - p_common, p_common]])
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = trans_probs @ max_q_s2

        q_net_s1 = q_mb.copy()
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers

        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # --- Learning ---
        # 1. Update Chosen Alien
        delta_2 = r - q_stage2[s_next, a2]
        q_stage2[s_next, a2] += lr_val * delta_2
        
        # 2. Update Unchosen Alien (Fictive)
        # Assume unchosen option would have given (1-r)
        # This implements a "counterfactual" belief that rewards are anti-correlated
        unchosen_a2 = 1 - a2
        fictive_reward = 1.0 - r
        delta_fict = fictive_reward - q_stage2[s_next, unchosen_a2]
        q_stage2[s_next, unchosen_a2] += lr_fict * delta_fict

        # 3. Update Coupled Transition
        is_common_transition = (a1 == s_next)
        if is_common_transition:
            p_common += lr_common * (1.0 - p_common)
        else:
            p_common += lr_common * (0.0 - p_common)
            
        last_action_1 = a1
        
    return log_loss
```

### Cognitive Model 3
```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model 3: Model-Based with Reward-Modulated Coupled Transition Learning.
    
    This model investigates if the learning of the transition structure (p_common)
    is modulated by the outcome of the trial. The hypothesis is that participants 
    may update their internal model of the spaceship transitions more strongly 
    when the trial results in a reward (reinforcement of the path) versus when 
    it does not.
    
    Parameters:
    lr_val: [0, 1] - Learning rate for Stage 2 Q-values.
    lr_common_rew: [0, 1] - Learning rate for p_common when Reward = 1.
    lr_common_unrew: [0, 1] - Learning rate for p_common when Reward = 0.
    beta: [0, 10] - Inverse temperature.
    pers: [0, 5] - Perseverance bonus for Stage 1 choice.
    """
    lr_val, lr_common_rew, lr_common_unrew, beta, pers = model_parameters
    n_trials = len(action_1)

    q_stage2 = np.zeros((2, 2))
    p_common = 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1 or reward[trial] == -1:
            last_action_1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_next = int(state[trial])
        a2 = int(action_2[trial])
        r = float(reward[trial])

        # --- Stage 1 Policy (Model-Based Coupled) ---
        trans_probs = np.array([[p_common, 1.0 - p_common], 
                                [1.0 - p_common, p_common]])
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = trans_probs @ max_q_s2

        q_net_s1 = q_mb.copy()
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers

        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # --- Learning ---
        # 1. Update Stage 2 Q-values
        delta_2 = r - q_stage2[s_next, a2]
        q_stage2[s_next, a2] += lr_val * delta_2
        
        # 2. Update Coupled Transition with Reward Modulation
        # Determine effective learning rate based on reward
        if r > 0.5: # Reward = 1
            lr_c = lr_common_rew
        else:       # Reward = 0
            lr_c = lr_common_unrew
            
        is_common_transition = (a1 == s_next)
        if is_common_transition:
            p_common += lr_c * (1.0 - p_common)
        else:
            p_common += lr_c * (0.0 - p_common)
            
        last_action_1 = a1
        
    return log_loss
```
