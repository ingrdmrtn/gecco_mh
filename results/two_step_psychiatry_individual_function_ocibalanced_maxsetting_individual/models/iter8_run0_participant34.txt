Here are three new cognitive models for the two-step decision task, designed to capture specific behavioral patterns such as outcome-dependent strategy shifting, loss aversion, and transition-specific learning.

### Model 1: Outcome-Dependent Model-Based Weighting
This model hypothesizes that the participant's reliance on the Model-Based (planning) system versus the Model-Free (habit) system changes depending on whether the previous trial was rewarded. A win might reinforce the current mode of control, while a loss might trigger a shift.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Outcome-Dependent Model-Based Weighting.
    
    The weight `w` controlling the balance between Model-Based and Model-Free 
    systems is dynamic, switching between `w_win` and `w_loss` based on the 
    outcome of the previous trial.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choices.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choices.
    - w_win: [0, 1] Weight for Model-Based control after a Reward (Win).
    - w_loss: [0, 1] Weight for Model-Based control after No Reward (Loss).
    - stickiness: [0, 5] Choice perseverance bonus.
    """
    learning_rate, beta_1, beta_2, w_win, w_loss, stickiness = model_parameters
    n_trials = len(action_1)
    
    # Transition matrix: A(0)->X(0) is 0.7, U(1)->Y(1) is 0.7
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)
    q_s2 = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0 # Assume 0 (loss) state for the very first trial logic
    
    for t in range(n_trials):
        # Select w based on previous outcome
        w = w_win if last_reward == 1 else w_loss
        
        # --- Stage 1 Policy ---
        max_q_s2 = np.max(q_s2, axis=1)
        q_mb1 = transition_matrix @ max_q_s2
        
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        q_net_stable = q_net - np.max(q_net)
        exp_q1 = np.exp(beta_1 * q_net_stable)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        a1 = action_1[t]
        s_next = state[t]
        last_action_1 = a1
        
        # --- Stage 2 Policy ---
        q_s2_curr = q_s2[s_next]
        q_s2_stable = q_s2_curr - np.max(q_s2_curr)
        exp_q2 = np.exp(beta_2 * q_s2_stable)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        a2 = action_2[t]
        r = reward[t]
        last_reward = r
        
        # --- Updates ---
        # MF Stage 1 update
        pe1 = q_s2[s_next, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * pe1
        
        # Stage 2 update
        pe2 = r - q_s2[s_next, a2]
        q_s2[s_next, a2] += learning_rate * pe2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Loss Aversion Model
This model incorporates a subjective valuation of "no reward". Instead of treating 0 coins as a neutral event (value 0), the participant may perceive it as a loss. This model transforms the reward signal, which affects the asymptotic values of options and the magnitude of prediction errors.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Loss Aversion Model.
    
    Assumes the participant treats 0 coins as a negative outcome (loss) rather than 
    neutral. The effective reward is 1 for a win, and `-loss_magnitude` for a loss.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Weight for Model-Based control.
    - stickiness: [0, 5] Choice perseverance.
    - loss_magnitude: [0, 5] Subjective penalty for receiving 0 coins.
    """
    learning_rate, beta_1, beta_2, w, stickiness, loss_magnitude = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)
    q_s2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_s2 = np.max(q_s2, axis=1)
        q_mb1 = transition_matrix @ max_q_s2
        
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        q_net_stable = q_net - np.max(q_net)
        exp_q1 = np.exp(beta_1 * q_net_stable)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        a1 = action_1[t]
        s_next = state[t]
        last_action_1 = a1
        
        # --- Stage 2 Policy ---
        q_s2_curr = q_s2[s_next]
        q_s2_stable = q_s2_curr - np.max(q_s2_curr)
        exp_q2 = np.exp(beta_2 * q_s2_stable)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        a2 = action_2[t]
        r = reward[t]
        
        # --- Updates ---
        # Transform reward: 0 becomes negative loss
        effective_reward = 1.0 if r == 1 else -loss_magnitude
        
        pe1 = q_s2[s_next, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * pe1
        
        pe2 = effective_reward - q_s2[s_next, a2]
        q_s2[s_next, a2] += learning_rate * pe2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Transition-Dependent Learning Rates
This model suggests that the participant updates their Model-Free values differently depending on whether the transition to the planet was "Common" or "Rare". This captures a heuristic where participants might discount (or over-weight) outcomes resulting from rare transitions when updating their spaceship preference.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Transition-Dependent Learning Rates Model.
    
    Applies different learning rates for the Stage 1 Model-Free update depending 
    on whether the transition (Spaceship -> Planet) was Common or Rare.
    
    Parameters:
    - lr_s2: [0, 1] Learning rate for Stage 2 (Alien reward learning).
    - lr_common: [0, 1] Stage 1 Learning rate after a Common transition.
    - lr_rare: [0, 1] Stage 1 Learning rate after a Rare transition.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Weight for Model-Based control.
    - stickiness: [0, 5] Choice perseverance.
    """
    lr_s2, lr_common, lr_rare, beta_1, beta_2, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)
    q_s2 = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_s2 = np.max(q_s2, axis=1)
        q_mb1 = transition_matrix @ max_q_s2
        
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        q_net_stable = q_net - np.max(q_net)
        exp_q1 = np.exp(beta_1 * q_net_stable)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        a1 = action_1[t]
        s_next = state[t]
        last_action_1 = a1
        
        # --- Stage 2 Policy ---
        q_s2_curr = q_s2[s_next]
        q_s2_stable = q_s2_curr - np.max(q_s2_curr)
        exp_q2 = np.exp(beta_2 * q_s2_stable)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        a2 = action_2[t]
        r = reward[t]
        
        # --- Updates ---
        # Determine if transition was Common or Rare
        # Common: A(0)->X(0) or U(1)->Y(1). Rare: Mixed.
        is_common = (a1 == s_next)
        
        current_lr_s1 = lr_common if is_common else lr_rare
        
        pe1 = q_s2[s_next, a2] - q_mf1[a1]
        q_mf1[a1] += current_lr_s1 * pe1
        
        pe2 = r - q_s2[s_next, a2]
        q_s2[s_next, a2] += lr_s2 * pe2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
