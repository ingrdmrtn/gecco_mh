def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Learning Hybrid Model with Eligibility Traces.
    
    This model distinguishes between positive and negative prediction errors (learning rates),
    allowing for optimistic or pessimistic learning biases. It combines this with a 
    hybrid MB/MF architecture including eligibility traces (lambda) to update Stage 1 
    MF values based on Stage 2 outcomes.

    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors.
    - lr_neg: [0, 1] Learning rate for negative prediction errors.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - lam: [0, 1] Eligibility trace (lambda) for Stage 1 updates.
    """
    lr_pos, lr_neg, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix (A->X common, U->Y common)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    q_stage2_mf.fill(0.5) # Initialize to neutral expectation
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue
            
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        s_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # Stage 1 TD(0) Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        curr_lr_1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[a1] += curr_lr_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        curr_lr_2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[s_idx, a2] += curr_lr_2 * delta_stage2
        
        # Eligibility Trace Update (Stage 1 from Stage 2 RPE)
        # Uses the learning rate associated with the Stage 2 error
        q_stage1_mf[a1] += curr_lr_2 * lam * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Dual Learning Rate Hybrid Model with Stickiness.
    
    This model separates the learning rates for Stage 1 (navigational/state-action values)
    and Stage 2 (bandit/reward values), allowing the agent to adapt to reward drift 
    at a different speed than value propagation. It also includes a stickiness parameter
    to account for choice perseveration in Stage 1.
    
    Parameters:
    - lr_1: [0, 1] Learning rate for Stage 1 MF values.
    - lr_2: [0, 1] Learning rate for Stage 2 values.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - stick: [0, 10] Stickiness parameter for Stage 1 choice.
    """
    lr_1, lr_2, beta_1, beta_2, w, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    q_stage2_mf.fill(0.5)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue
            
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits and add stickiness
        logits_1 = beta_1 * q_net_1
        if prev_a1 != -1:
            logits_1[prev_a1] += stick
            
        # Softmax with stability
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        s_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_1 * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_2 * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic Transition Hybrid Model with Eligibility Traces.
    
    This model assumes the agent learns the transition matrix dynamically based on 
    observed transitions, rather than using a fixed prior. It combines this with 
    eligibility traces (lambda) to allow Stage 2 reward prediction errors to 
    directly update Stage 1 MF values.
    
    Parameters:
    - lr_val: [0, 1] Learning rate for Q-values (both stages).
    - lr_trans: [0, 1] Learning rate for transition probabilities.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - lam: [0, 1] Eligibility trace (lambda).
    """
    lr_val, lr_trans, beta_1, beta_2, w, lam = model_parameters
    n_trials = len(action_1)
    
    # Initialize transitions to uniform (maximum uncertainty)
    trans_probs = np.array([[0.5, 0.5], [0.5, 0.5]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    q_stage2_mf.fill(0.5)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue
            
        # Stage 1 Policy (MB uses dynamic transition matrix)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = trans_probs @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        s_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_val * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_val * delta_stage2
        
        # Eligibility Trace Update
        q_stage1_mf[a1] += lr_val * lam * delta_stage2
        
        # Dynamic Transition Update
        # Update P(s_idx | a1)
        trans_probs[a1, s_idx] += lr_trans * (1.0 - trans_probs[a1, s_idx])
        trans_probs[a1, 1 - s_idx] = 1.0 - trans_probs[a1, s_idx]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
