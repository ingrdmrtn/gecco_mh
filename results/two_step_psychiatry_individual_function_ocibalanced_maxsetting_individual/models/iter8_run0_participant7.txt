Here are three new cognitive models for the two-step decision task.

### Model 1: Transition-Dependent Eligibility Trace (Gated Lambda)
This model hypothesizes that the participant modulates the strength of the eligibility trace (the link between Stage 2 outcome and Stage 1 choice) based on whether the transition was common or rare. A "smart" heuristic agent might suppress the eligibility trace after a rare transition to avoid reinforcing the wrong Stage 1 action (a hallmark of Model-Based behavior), while a standard Model-Free agent would reinforce it. This model captures this via `lam_common` and `lam_rare`.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Transition-Dependent Eligibility Trace Model.
    
    Distinguishes between common and rare transitions for the eligibility trace update.
    This allows the model to modulate the 'Model-Free' credit assignment based on the
    transition type. If lam_rare is low, the agent avoids reinforcing Stage 1 choices
    that led to rewards via rare transitions.
    
    Parameters:
    - learning_rate: Rate at which Q-values are updated [0, 1]
    - beta_1: Inverse temperature for Stage 1 [0, 10]
    - beta_2: Inverse temperature for Stage 2 [0, 10]
    - w: Mixing weight between Model-Based and Model-Free values [0, 1]
    - lam_common: Eligibility trace for common transitions [0, 1]
    - lam_rare: Eligibility trace for rare transitions [0, 1]
    - stick: Choice stickiness bonus [0, 10]
    - decay: Decay rate for unchosen options [0, 1]
    """
    learning_rate, beta_1, beta_2, w, lam_common, lam_rare, stick, decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)

    for trial in range(n_trials):

        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if trial > 0:
            logits_1[action_1[trial-1]] += stick

        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        # Stage 1 update (TD)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Eligibility Trace Update for Stage 1
        # Check transition type (0->0 and 1->1 are common)
        is_common = (action_1[trial] == state_idx)
        lam = lam_common if is_common else lam_rare
        
        q_stage1_mf[action_1[trial]] += learning_rate * lam * delta_stage2

        # Decay unchosen Stage 1
        for a in range(2):
            if a != action_1[trial]:
                q_stage1_mf[a] = (1 - decay) * q_stage1_mf[a] + decay * 0.5

        # Decay unchosen Stage 2 (all states/actions not visited)
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == action_2[trial]):
                    q_stage2_mf[s, a] = (1 - decay) * q_stage2_mf[s, a] + decay * 0.5

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Counterfactual Stage 2 Learning
This model assumes the participant engages in fictitious play or counterfactual reasoning. When an alien is chosen and yields a specific outcome (e.g., reward), the participant updates the value of the *unchosen* alien in the same state assuming the opposite outcome (e.g., no reward), scaled by `alpha_cf`. This captures a belief in negative correlations between options (e.g., "if this one didn't have it, the other one must have").

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Counterfactual Updating Model.
    
    Implements a counterfactual update for the unchosen alien in Stage 2.
    The agent assumes that if the chosen alien yielded reward R, the unchosen
    alien would have yielded 1-R. This is scaled by alpha_cf.
    
    Parameters:
    - learning_rate: Rate at which Q-values are updated [0, 1]
    - beta_1: Inverse temperature for Stage 1 [0, 10]
    - beta_2: Inverse temperature for Stage 2 [0, 10]
    - w: Mixing weight between Model-Based and Model-Free values [0, 1]
    - lam: Eligibility trace parameter [0, 1]
    - alpha_cf: Learning rate scaling for the counterfactual update [0, 1]
    - stick: Choice stickiness bonus [0, 10]
    - decay: Decay rate for unchosen Stage 1 values [0, 1]
    """
    learning_rate, beta_1, beta_2, w, lam, alpha_cf, stick, decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)

    for trial in range(n_trials):

        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if trial > 0:
            logits_1[action_1[trial-1]] += stick
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Chosen Stage 2 update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Counterfactual Stage 2 update (Unchosen option)
        unchosen_a2 = 1 - action_2[trial]
        # Assumed reward for unchosen is 1 - actual_reward
        assumed_reward = 1 - reward[trial]
        delta_cf = assumed_reward - q_stage2_mf[state_idx, unchosen_a2]
        q_stage2_mf[state_idx, unchosen_a2] += learning_rate * alpha_cf * delta_cf
        
        # Trace update
        q_stage1_mf[action_1[trial]] += learning_rate * lam * delta_stage2

        # Decay Stage 1 unchosen
        for a in range(2):
            if a != action_1[trial]:
                q_stage1_mf[a] = (1 - decay) * q_stage1_mf[a] + decay * 0.5
        
        # Decay Stage 2 unvisited states (counterfactual handles the current state unchosen)
        for s in range(2):
            if s != state_idx:
                for a in range(2):
                    q_stage2_mf[s, a] = (1 - decay) * q_stage2_mf[s, a] + decay * 0.5

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Dual Learning Rate Model (MB/MF Split)
This model maintains separate value estimates for the Model-Based and Model-Free systems with distinct learning rates (`lr_mb` and `lr_mf`). This allows the model to capture scenarios where the participant's explicit knowledge (used for planning) updates at a different speed than their habitual values. For example, they might understand quickly that a planet is good (high `lr_mb`) but have slow-to-change habits (low `lr_mf`).

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dual Learning Rate Model (MB/MF Split).
    
    Maintains separate value estimates for the Model-Based and Model-Free systems
    with distinct learning rates. This allows the model to capture differences
    in how quickly the 'planning' values update versus the 'habitual' values.
    
    Parameters:
    - lr_mf: Learning rate for Model-Free values [0, 1]
    - lr_mb: Learning rate for Model-Based values [0, 1]
    - beta_1: Inverse temperature for Stage 1 [0, 10]
    - beta_2: Inverse temperature for Stage 2 [0, 10]
    - w: Mixing weight between MB and MF systems [0, 1]
    - lam: Eligibility trace parameter [0, 1]
    - stick: Choice stickiness bonus [0, 10]
    - decay: Decay rate for unchosen values [0, 1]
    """
    lr_mf, lr_mb, beta_1, beta_2, w, lam, stick, decay = model_parameters
    n_trials = len(action_1)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Separate value stores
    q_stage1_mf = np.full(2, 0.5)
    q_stage2_mf = np.full((2, 2), 0.5)
    q_stage2_mb = np.full((2, 2), 0.5) # MB tracks values its own way

    for trial in range(n_trials):

        # Stage 1 Choice
        # MB uses its own stage 2 values
        max_q_stage2_mb = np.max(q_stage2_mb, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2_mb
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net_1
        if trial > 0:
            logits_1[action_1[trial-1]] += stick
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice
        # Combine estimates for stage 2 as well
        q_net_2 = w * q_stage2_mb[state_idx] + (1 - w) * q_stage2_mf[state_idx]
        
        logits_2 = beta_2 * q_net_2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Updates
        
        # 1. Update MF system
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_mf * delta_stage1
        
        delta_stage2_mf = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_mf * delta_stage2_mf
        
        q_stage1_mf[action_1[trial]] += lr_mf * lam * delta_stage2_mf # Trace uses MF error
        
        # 2. Update MB system (only updates Stage 2 values based on reward)
        delta_stage2_mb = reward[trial] - q_stage2_mb[state_idx, action_2[trial]]
        q_stage2_mb[state_idx, action_2[trial]] += lr_mb * delta_stage2_mb

        # Decay
        for a in range(2):
            if a != action_1[trial]:
                q_stage1_mf[a] = (1 - decay) * q_stage1_mf[a] + decay * 0.5
        
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == action_2[trial]):
                    q_stage2_mf[s, a] = (1 - decay) * q_stage2_mf[s, a] + decay * 0.5
                    q_stage2_mb[s, a] = (1 - decay) * q_stage2_mb[s, a] + decay * 0.5

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
