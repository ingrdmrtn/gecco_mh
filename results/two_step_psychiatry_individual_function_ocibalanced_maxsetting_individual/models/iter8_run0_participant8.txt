Here are the three new cognitive models, designed to capture specific mechanisms like stage-specific exploration, asymmetric learning, and outcome-dependent perseveration.

def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Separate Betas and Eligibility Trace.
    
    Distinguishes decision noise (beta) between Stage 1 (Spaceship) and Stage 2 (Alien),
    while using an eligibility trace (lambda) to propagate Stage 2 errors to Stage 1.
    This accounts for potentially different exploration levels in the navigation vs. 
    bandit phases of the task.
    
    Parameters:
    lr: [0,1] - Learning rate.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    lam: [0,1] - Eligibility trace parameter (0 = TD(0), 1 = TD(1)).
    stickiness: [0,5] - Choice perseveration bonus for Stage 1.
    """
    lr, beta_1, beta_2, w, lam, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        chosen_a1 = action_1[trial]
        chosen_a2 = action_2[trial]
        r = reward[trial]
        
        # MF Update Stage 1 (TD(0))
        delta_1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += lr * delta_1
        
        # MF Update Stage 2
        delta_2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += lr * delta_2
        
        # Eligibility Trace Update (TD(1) logic for Stage 1)
        q_stage1_mf[chosen_a1] += lr * lam * delta_2
        
        prev_a1 = chosen_a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Asymmetric Learning Rates and Separate Betas.
    
    Uses different learning rates for positive and negative prediction errors (valence-dependent learning),
    and different inverse temperatures for Stage 1 and Stage 2. This does not use an eligibility trace,
    relying instead on the separated learning dynamics to capture behavior.
    
    Parameters:
    lr_pos: [0,1] - Learning rate for positive prediction errors.
    lr_neg: [0,1] - Learning rate for negative prediction errors.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    beta_2: [0,10] - Inverse temperature for Stage 2.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    stickiness: [0,5] - Choice perseveration bonus for Stage 1.
    """
    lr_pos, lr_neg, beta_1, beta_2, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        chosen_a1 = action_1[trial]
        chosen_a2 = action_2[trial]
        r = reward[trial]
        
        # MF Update Stage 1
        delta_1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        eff_lr_1 = lr_pos if delta_1 > 0 else lr_neg
        q_stage1_mf[chosen_a1] += eff_lr_1 * delta_1
        
        # MF Update Stage 2
        delta_2 = r - q_stage2_mf[state_idx, chosen_a2]
        eff_lr_2 = lr_pos if delta_2 > 0 else lr_neg
        q_stage2_mf[state_idx, chosen_a2] += eff_lr_2 * delta_2
        
        prev_a1 = chosen_a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid Model with Decay, Eligibility Trace, and Outcome-Dependent Stickiness.
    
    Features reward-dependent perseveration (win-stay vs lose-stay logic via separate stickiness params),
    passive decay for unchosen options to handle drifting rewards, and eligibility traces for temporal credit assignment.
    
    Parameters:
    lr: [0,1] - Learning rate.
    decay: [0,1] - Decay rate for unchosen options.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting parameter (0 = Pure MF, 1 = Pure MB).
    lam: [0,1] - Eligibility trace parameter.
    stick_rew: [0,5] - Stickiness bonus if previous trial was rewarded.
    stick_unrew: [0,5] - Stickiness bonus if previous trial was unrewarded.
    """
    lr, decay, beta, w, lam, stick_rew, stick_unrew = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    prev_reward = 0
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            if prev_reward == 1:
                q_net[prev_a1] += stick_rew
            else:
                q_net[prev_a1] += stick_unrew
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        chosen_a1 = action_1[trial]
        chosen_a2 = action_2[trial]
        r = reward[trial]
        
        delta_1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += lr * delta_1
        
        delta_2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += lr * delta_2
        
        # Eligibility trace update
        q_stage1_mf[chosen_a1] += lr * lam * delta_2
        
        # Decay unchosen
        q_stage1_mf[1 - chosen_a1] *= (1 - decay)
        for s in range(2):
            for a in range(2):
                if not (s == state_idx and a == chosen_a2):
                    q_stage2_mf[s, a] *= (1 - decay)
        
        prev_a1 = chosen_a1
        prev_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
