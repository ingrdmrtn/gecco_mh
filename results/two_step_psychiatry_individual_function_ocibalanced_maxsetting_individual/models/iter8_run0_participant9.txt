def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Decay, Separate Betas for Stage 1/2, and Stickiness.
    
    This model distinguishes between the exploration/exploitation trade-off in the 
    strategic Stage 1 (spaceship choice) versus the bandit-like Stage 2 (alien choice) 
    by using separate inverse temperature parameters (beta_1, beta_2).
    It incorporates decay for unchosen Stage 2 options to handle drifting reward probabilities.
    
    Parameters:
    learning_rate: [0,1] - Learning rate for value updates.
    decay_rate: [0,1] - Decay rate for unchosen Stage 2 options (1 = full reset).
    beta_1: [0,10] - Inverse temperature for Stage 1 choice.
    beta_2: [0,10] - Inverse temperature for Stage 2 choice.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    stickiness: [0,5] - Choice perseverance bonus for Stage 1.
    """
    learning_rate, decay_rate, beta_1, beta_2, w, stickiness = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta_1 * q_net
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            continue
            
        state_idx = state[trial]
        logits_2 = beta_2 * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        # Stage 1 MF update (Standard TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Decay unchosen Stage 2 options
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action_2] *= (1 - decay_rate)
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Decay and Choice Kernel.
    
    Replaces simple stickiness with a Choice Kernel (CK) that integrates 
    choice history over time. This allows the model to capture more complex 
    persistence dynamics than a simple 1-back stickiness.
    
    Parameters:
    learning_rate: [0,1] - Learning rate for value updates.
    decay_rate: [0,1] - Decay rate for unchosen Stage 2 options.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    ck_lr: [0,1] - Learning rate for the choice kernel (decay of trace).
    ck_weight: [0,5] - Weight of the choice kernel in Stage 1 decision.
    """
    learning_rate, decay_rate, beta, w, ck_lr, ck_weight = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    choice_kernel = np.zeros(2)
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Combine value and choice kernel
        logits_1 = beta * q_net + ck_weight * choice_kernel
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Choice Kernel (accumulate chosen, decay both)
        choice_kernel *= (1 - ck_lr)
        choice_kernel[action_1[trial]] += ck_lr * 1
        
        # Stage 2 Policy
        if action_2[trial] == -1:
            continue
            
        state_idx = state[trial]
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action_2] *= (1 - decay_rate)

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF model with Decay and Outcome-Dependent Stickiness.
    
    Splits the stickiness parameter into 'win-stay' and 'lose-stay' components.
    This allows the model to capture behavior where the participant is more likely 
    to repeat a choice after a reward (win) compared to a punishment/omission (loss).
    
    Parameters:
    learning_rate: [0,1] - Learning rate.
    decay_rate: [0,1] - Decay rate for unchosen Stage 2 options.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    stick_win: [0,5] - Stickiness bonus applied if previous trial was rewarded.
    stick_loss: [0,5] - Stickiness bonus applied if previous trial was unrewarded.
    """
    learning_rate, decay_rate, beta, w, stick_win, stick_loss = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    prev_reward = 0
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if prev_action_1 != -1:
            # Apply different stickiness based on previous outcome
            bonus = stick_win if prev_reward > 0 else stick_loss
            logits_1[prev_action_1] += bonus
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Stage 2 Policy
        if action_2[trial] == -1:
            prev_action_1 = action_1[trial]
            prev_reward = 0
            continue
            
        state_idx = state[trial]
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        unchosen_action_2 = 1 - action_2[trial]
        q_stage2_mf[state_idx, unchosen_action_2] *= (1 - decay_rate)
        
        prev_action_1 = action_1[trial]
        prev_reward = reward[trial]

    eps = 1e-10
    valid_mask_2 = action_2 != -1
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2[valid_mask_2] + eps)))
    return log_loss
