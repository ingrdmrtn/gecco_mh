Here are three new cognitive models based on the participant data and feedback.

### Model 1: Split-Learning-Rate TD($\lambda$) with Decay
This model extends the successful TD($\lambda$) with Decay approach by assigning separate learning rates to the first and second stages. The participant data suggests a stable, slow-changing strategy for spaceship choice (Stage 1) but a more reactive strategy for alien choice (Stage 2) given the changing reward probabilities.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Split-Learning-Rate TD(lambda) with Decay.
    
    Uses separate learning rates for Stage 1 and Stage 2 to capture different 
    timescales of adaptation. Includes eligibility traces (lambda) and 
    decay of unchosen Stage 2 values to handle changing reward probabilities.
    
    Parameters:
    lr_1:         [0, 1] Learning rate for Stage 1 (Spaceships).
    lr_2:         [0, 1] Learning rate for Stage 2 (Aliens).
    beta:         [0, 10] Inverse temperature (softmax noise).
    lambda_param: [0, 1] Eligibility trace decay (credit assignment).
    decay:        [0, 1] Decay rate for unchosen Stage 2 values.
    """
    lr_1, lr_2, beta, lambda_param, decay = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1 = np.zeros(2)      
    q_stage2 = np.zeros((2, 2)) 

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5 
            p_choice_2[trial] = 0.5 
            continue

        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Prediction Errors
        delta_stage1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2[state_idx, action_2[trial]]

        # Update Stage 2 (using lr_2)
        q_stage2[state_idx, action_2[trial]] += lr_2 * delta_stage2
        
        # Decay unchosen Stage 2 values
        unchosen_a2 = 1 - action_2[trial]
        q_stage2[state_idx, unchosen_a2] *= (1 - decay)
        
        # Decay values for the unvisited state
        unchosen_state = 1 - state_idx
        q_stage2[unchosen_state, :] *= (1 - decay)

        # Update Stage 1 (using lr_1 and lambda)
        combined_delta = delta_stage1 + lambda_param * delta_stage2
        q_stage1[action_1[trial]] += lr_1 * combined_delta

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: Hybrid Model-Based/Model-Free with Decay and Perseveration
This model combines Model-Based (MB) planning with the Model-Free (MF) decay mechanism. The MB system calculates values based on the transition matrix and the *decaying* Stage 2 values. It also includes perseveration on Stage 1, which was a key component of previous successful models but not yet combined with Hybrid+Decay.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Hybrid MB/MF with Decay and Perseveration.
    
    Combines a Model-Based planner (using fixed transition probs) and a 
    Model-Free learner. Stage 2 values decay to simulate forgetting.
    Stage 1 includes a perseveration bonus.
    
    Parameters:
    lr:            [0, 1] Learning rate for Stage 2 and Stage 1 MF.
    beta:          [0, 10] Inverse temperature.
    w:             [0, 1] Weight of Model-Based system (1 = pure MB).
    perseveration: [0, 5] Bonus for repeating the previous Stage 1 choice.
    decay:         [0, 1] Decay rate for unchosen Stage 2 values.
    """
    lr, beta, w, perseveration, decay = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf_stage1 = np.zeros(2)
    q_mf_stage2 = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5 
            p_choice_2[trial] = 0.5 
            continue

        # Calculate Model-Based values for Stage 1
        # MB value is expected value of Stage 2 options given transitions
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        # Combined Net Value
        q_net_stage1 = w * q_mb_stage1 + (1 - w) * q_mf_stage1
        
        # Apply Perseveration
        if prev_action_1 != -1:
            q_net_stage1[prev_action_1] += perseveration

        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice (Pure MF)
        exp_q2 = np.exp(beta * q_mf_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_mf_stage2[state_idx, action_2[trial]]
        q_mf_stage2[state_idx, action_2[trial]] += lr * delta_stage2
        
        # Decay unchosen Stage 2 values
        unchosen_a2 = 1 - action_2[trial]
        q_mf_stage2[state_idx, unchosen_a2] *= (1 - decay)
        unchosen_state = 1 - state_idx
        q_mf_stage2[unchosen_state, :] *= (1 - decay)

        # Update Stage 1 MF (TD(1) style using direct reward)
        delta_stage1_mf = reward[trial] - q_mf_stage1[action_1[trial]]
        q_mf_stage1[action_1[trial]] += lr * delta_stage1_mf
        
        prev_action_1 = action_1[trial]

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: TD($\lambda$) with Habit Learning (Choice Kernel)
This model replaces simple one-step perseveration with a "Habit" mechanism (Choice Kernel). The participant data shows long blocks of repeating the same spaceship choice, which suggests a slowly building habit strength rather than just repeating the immediately preceding choice.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    TD(lambda) with Habit Learning.
    
    Standard TD(lambda) augmented with a Habit Choice Kernel. 
    The kernel tracks the frequency of past choices with a separate learning rate,
    creating a 'momentum' effect that explains long streaks of same-choice behavior.
    
    Parameters:
    lr:           [0, 1] Learning rate for values.
    beta:         [0, 10] Inverse temperature.
    lambda_param: [0, 1] Eligibility trace decay.
    habit_lr:     [0, 1] Learning rate for the habit kernel (how fast habits form/decay).
    habit_w:      [0, 10] Weight of the habit kernel in the decision.
    """
    lr, beta, lambda_param, habit_lr, habit_w = model_parameters
    n_trials = len(action_1)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1 = np.zeros(2)      
    q_stage2 = np.zeros((2, 2)) 
    
    # Habit kernel for Stage 1 choices initialized to 0
    habit_kernel = np.zeros(2)

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5 
            p_choice_2[trial] = 0.5 
            continue

        # Stage 1 Choice: Value + Habit
        # Note: Habit is added to the log-logits (Q-values) before softmax
        net_value_1 = q_stage1 + habit_w * habit_kernel
        
        exp_q1 = np.exp(beta * net_value_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # TD Updates
        delta_stage1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        delta_stage2 = reward[trial] - q_stage2[state_idx, action_2[trial]]

        q_stage2[state_idx, action_2[trial]] += lr * delta_stage2
        
        combined_delta = delta_stage1 + lambda_param * delta_stage2
        q_stage1[action_1[trial]] += lr * combined_delta
        
        # Update Habit Kernel
        # Chosen action moves toward 1, unchosen toward 0
        chosen_a1 = action_1[trial]
        habit_kernel[chosen_a1] += habit_lr * (1 - habit_kernel[chosen_a1])
        habit_kernel[1 - chosen_a1] += habit_lr * (0 - habit_kernel[1 - chosen_a1])

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
