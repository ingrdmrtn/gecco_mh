Here are three new cognitive models for the two-step decision task, designed based on the participant's data and the constraint to explore novel parameter combinations.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Model 1: Outcome-Dependent Weight Hybrid Model.
    
    This model posits that the balance between Model-Based (goal-directed) and 
    Model-Free (habitual) control shifts based on the outcome of the previous trial.
    For example, a win might reinforce the current strategy (MF dominance), while a 
    loss might trigger re-evaluation using the model (MB dominance), or vice versa.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for value updates.
    beta: [0, 10] - Inverse temperature (exploration/exploitation trade-off).
    w_win: [0, 1] - Weight of Model-Based values after a rewarded trial.
    w_loss: [0, 1] - Weight of Model-Based values after an unrewarded trial.
    perseveration: [-3, 3] - Stickiness to the previous Stage 1 choice.
    """
    learning_rate, beta, w_win, w_loss, perseveration = model_parameters
    n_trials = len(action_1)
    
    # Fixed transition matrix (as per task description: A->X (0.7), U->Y (0.7))
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_stage1 = np.zeros(2) + 0.5
    q_mf_stage2 = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    prev_reward = 0 
    
    for t in range(n_trials):
        # --- Stage 1 Policy ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        # Select mixing weight based on previous outcome
        w = w_win if prev_reward == 1 else w_loss
        
        # Combined Net Value
        q_net = w * q_mb_stage1 + (1 - w) * q_mf_stage1
        
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        a1 = action_1[t]
        s_idx = state[t]
        
        # --- Stage 2 Policy ---
        logits_2 = beta * q_mf_stage2[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        a2 = action_2[t]
        r = reward[t]
        
        # --- Updates ---
        # Update Stage 2 MF values (Standard Q-learning)
        delta_2 = r - q_mf_stage2[s_idx, a2]
        q_mf_stage2[s_idx, a2] += learning_rate * delta_2
        
        # Update Stage 1 MF values (TD(0))
        delta_1 = q_mf_stage2[s_idx, a2] - q_mf_stage1[a1]
        q_mf_stage1[a1] += learning_rate * delta_1
        
        prev_a1 = a1
        prev_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Model 2: TD(lambda) with Outcome-Dependent Beta.
    
    This model combines the "Outcome-Dependent Beta" mechanism (which was successful
    in previous iterations) with Eligibility Traces (TD(lambda)). 
    The eligibility trace allows the outcome at Stage 2 (Reward) to directly update 
    the value of the Stage 1 choice, bridging the gap between Model-Free and Monte-Carlo 
    learning. The exploration rate (beta) varies by outcome confidence.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for value updates.
    lambda_elig: [0, 1] - Decay rate of the eligibility trace (0=TD, 1=Monte Carlo).
    beta_win: [0, 10] - Inverse temperature after a rewarded trial.
    beta_loss: [0, 10] - Inverse temperature after an unrewarded trial.
    perseveration: [-3, 3] - Stickiness to the previous Stage 1 choice.
    """
    learning_rate, lambda_elig, beta_win, beta_loss, perseveration = model_parameters
    n_trials = len(action_1)
    
    q_stage1 = np.zeros(2) + 0.5
    q_stage2 = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    prev_reward = 0
    
    for t in range(n_trials):
        # --- Stage 1 Policy ---
        current_beta = beta_win if prev_reward == 1 else beta_loss
        
        logits_1 = current_beta * q_stage1
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        a1 = action_1[t]
        s_idx = state[t]
        
        # --- Stage 2 Policy ---
        logits_2 = current_beta * q_stage2[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        a2 = action_2[t]
        r = reward[t]
        
        # --- Updates with Eligibility Trace ---
        # 1. Prediction error at Stage 1 (Transition)
        # delta_1 = V(s') - Q(s,a)
        delta_1 = q_stage2[s_idx, a2] - q_stage1[a1]
        
        # Update Stage 1 immediately based on transition (TD(0) part)
        q_stage1[a1] += learning_rate * delta_1
        
        # 2. Prediction error at Stage 2 (Reward)
        # delta_2 = R - Q(s',a')
        delta_2 = r - q_stage2[s_idx, a2]
        
        # Update Stage 2
        q_stage2[s_idx, a2] += learning_rate * delta_2
        
        # Update Stage 1 again based on Reward, scaled by lambda
        # This effectively passes the R back to Stage 1
        q_stage1[a1] += learning_rate * delta_2 * lambda_elig
        
        prev_a1 = a1
        prev_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Model 3: Independent Betas Hybrid Model.
    
    Instead of a mixing weight 'w' that creates a weighted average of Q-values,
    this model assigns independent inverse temperatures (betas) to the Model-Based
    and Model-Free systems. This allows the agent to have different sensitivities
    to the two sources of information (e.g., precise MB planning vs noisy MF habits)
    simultaneously.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for MF value updates.
    beta_mb: [0, 10] - Inverse temperature for the Model-Based system.
    beta_mf: [0, 10] - Inverse temperature for the Model-Free system.
    perseveration: [-3, 3] - Stickiness to the previous Stage 1 choice.
    """
    learning_rate, beta_mb, beta_mf, perseveration = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_stage1 = np.zeros(2) + 0.5
    q_mf_stage2 = np.zeros((2, 2)) + 0.5
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        # Policy integrates both values with separate betas
        logits_1 = (beta_mb * q_mb_stage1) + (beta_mf * q_mf_stage1)
        
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        a1 = action_1[t]
        s_idx = state[t]
        
        # --- Stage 2 Policy ---
        # Stage 2 is purely driven by MF values (no transition structure to plan over)
        # We use the sum of betas or just one beta? Standard is to use a beta.
        # Here we use beta_mf as it's direct value retrieval.
        logits_2 = beta_mf * q_mf_stage2[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        a2 = action_2[t]
        r = reward[t]
        
        # --- Updates ---
        delta_2 = r - q_mf_stage2[s_idx, a2]
        q_mf_stage2[s_idx, a2] += learning_rate * delta_2
        
        delta_1 = q_mf_stage2[s_idx, a2] - q_mf_stage1[a1]
        q_mf_stage1[a1] += learning_rate * delta_1
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
