Here are three new cognitive models for the two-step task, designed based on the participant's data patterns and the feedback provided.

### Cognitive Model 1: Subjective Transition Model
**Rationale:** The participant may not believe or learn the "true" transition probabilities (0.7/0.3). Instead, they might operate with a fixed **subjective belief** (`p_common`) about the transition structure. For instance, they might treat the transitions as deterministic (`p_common` $\approx$ 1.0) or random (`p_common` $\approx$ 0.5), which heavily biases the Model-Based component of their decision-making.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Subjective Transition Model.
    The participant uses a fixed subjective belief (p_common) about the transition 
    probabilities for Model-Based planning, rather than the objective 0.7 or learning it.

    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w: [0, 1] Weighting parameter (0 = Pure MF, 1 = Pure MB).
    - perseverance: [0, 5] Stickiness to the previously chosen spaceship.
    - p_common: [0, 1] Subjective probability of the common transition (0.5 = random, 1.0 = deterministic).
    """
    learning_rate, beta, w, perseverance, p_common = model_parameters
    n_trials = len(action_1)

    # Subjective transition matrix based on p_common parameter
    transition_matrix = np.array([[p_common, 1 - p_common], 
                                  [1 - p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf = np.zeros(2)          # Stage 1 Model-Free values
    q_stage2 = np.zeros((2, 2)) # Stage 2 values (Aliens)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_mb + (1 - w) * q_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += perseverance

        # Safer softmax
        Q1 = beta * q_net
        Q1 -= np.max(Q1)
        exp_q1 = np.exp(Q1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        s_idx = state[trial]
        Q2 = beta * q_stage2[s_idx]
        Q2 -= np.max(Q2)
        exp_q2 = np.exp(Q2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        # TD(0) update for Stage 1 MF
        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta1

        # TD(0) update for Stage 2
        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2
        
        # TD(1) / Eligibility trace update for Stage 1 MF
        q_mf[a1] += learning_rate * delta2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Asymmetric Perseverance (Win-Stay vs. Loss-Stay)
**Rationale:** Standard models assume a single perseverance parameter. However, participants often exhibit different levels of stickiness depending on the previous outcome. This model splits perseverance into `pers_win` (Win-Stay) and `pers_loss` (Loss-Stay/Stubbornness), allowing the model to capture if the participant repeats choices strategically after rewards or habitually after losses.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    Asymmetric Perseverance Model.
    Applies distinct perseverance bonuses depending on whether the previous trial 
    resulted in a reward (Win) or no reward (Loss).

    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for MB vs MF.
    - pers_win: [0, 5] Perseverance bonus applied after a rewarded trial.
    - pers_loss: [0, 5] Perseverance bonus applied after an unrewarded trial.
    """
    learning_rate, beta, w, pers_win, pers_loss = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf = np.zeros(2)       
    q_stage2 = np.zeros((2, 2)) 
    
    last_action_1 = -1
    last_reward = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_mb + (1 - w) * q_mf
        
        # Apply conditional perseverance
        if last_action_1 != -1:
            if last_reward == 1:
                q_net[last_action_1] += pers_win
            else:
                q_net[last_action_1] += pers_loss

        Q1 = beta * q_net
        Q1 -= np.max(Q1)
        exp_q1 = np.exp(Q1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        s_idx = state[trial]
        Q2 = beta * q_stage2[s_idx]
        Q2 -= np.max(Q2)
        exp_q2 = np.exp(Q2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta1

        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2
        
        q_mf[a1] += learning_rate * delta2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Dynamic MB-MF Weighting
**Rationale:** The balance between Model-Based and Model-Free control (`w`) may not be static. Participants often shift strategies over time, for example, starting with goal-directed planning (high MB) and slipping into habits (high MF), or vice versa as they learn the structure. This model allows `w` to drift linearly from `w_start` to `w_end` across the session.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Dynamic MB-MF Weighting Model.
    The mixing weight 'w' changes linearly over the course of the experiment, 
    capturing a shift between Model-Based and Model-Free strategies (e.g., habituation).

    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w_start: [0, 1] Initial MB weight at trial 0.
    - w_end: [0, 1] Final MB weight at the last trial.
    - perseverance: [0, 5] Stickiness parameter.
    """
    learning_rate, beta, w_start, w_end, perseverance = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf = np.zeros(2)       
    q_stage2 = np.zeros((2, 2)) 
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Calculate dynamic w for the current trial
        w = w_start + (w_end - w_start) * (trial / n_trials)
        
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_mb + (1 - w) * q_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += perseverance

        Q1 = beta * q_net
        Q1 -= np.max(Q1)
        exp_q1 = np.exp(Q1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        s_idx = state[trial]
        Q2 = beta * q_stage2[s_idx]
        Q2 -= np.max(Q2)
        exp_q2 = np.exp(Q2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]

        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta1

        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2
        
        q_mf[a1] += learning_rate * delta2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
