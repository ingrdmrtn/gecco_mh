Here are three new cognitive models for the two-step task, expressed as Python functions.

### Model 1: Probability Mixture (Strategy Selection)
This model hypothesizes that instead of integrating Model-Based (MB) and Model-Free (MF) values into a single "net" Q-value (value integration), the agent selects a strategy (MB or MF) on each trial with probability $w$ (strategy selection). The final choice probability is a linear mixture of the probabilities generated by each system independently. This architecture can capture distinct modes of behavior that value-mixing cannot.

```python
def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """
    Probability Mixture Model (Strategy Selection).
    
    The agent chooses between using a Model-Based strategy or a Model-Free strategy 
    on each trial, rather than integrating their values. The final policy is a 
    weighted average of the policies (probabilities) of the two systems.
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature (shared across systems).
    - w: [0, 1] Probability weight for the Model-Based strategy (1 = Pure MB).
    - stick: [-5, 5] Stickiness bias applied to the Model-Free system.
    """
    lr, beta, w, stick = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        
        # 1. Model-Based System (Policy)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # MB Probabilities
        exp_q1_mb = np.exp(beta * q_stage1_mb)
        probs_mb = exp_q1_mb / np.sum(exp_q1_mb)
        
        # 2. Model-Free System (Policy)
        logits_mf = q_stage1_mf.copy()
        if last_action_1 != -1:
            logits_mf[last_action_1] += stick
            
        exp_q1_mf = np.exp(beta * logits_mf)
        probs_mf = exp_q1_mf / np.sum(exp_q1_mf)
        
        # 3. Probability Mixture
        probs_1 = w * probs_mb + (1 - w) * probs_mf
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        
        # Update Stage 1 MF values (SARSA-style)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        # Update Stage 2 MF values (Q-learning/Bandit)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: State Stickiness (Planet Perseveration)
This model introduces a bias to return to the previously visited **State** (Planet), distinct from the bias to repeat the previous **Action** (Spaceship). Since actions lead to states probabilistically, a "State Stickiness" bias motivates the agent to choose the action most likely to lead to the previous planet. This captures behavior where a participant tries to "get back" to a specific context.

```python
def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """
    State Stickiness Model (Planet Perseveration).
    
    Incorporates a bias to return to the previously visited state (planet).
    The agent receives a 'bonus' for actions that transition to the previous 
    state, weighted by the transition matrix.
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based values.
    - stick_act: [-5, 5] Standard action stickiness (motor repetition).
    - stick_state: [-5, 5] State stickiness (bias to return to previous planet).
    """
    lr, beta, w, stick_act, stick_state = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    last_state = -1
    
    for trial in range(n_trials):

        # Calculate hybrid Q-values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        logits = q_net.copy()
        
        # 1. Action Stickiness
        if last_action_1 != -1:
            logits[last_action_1] += stick_act
            
        # 2. State Stickiness (Bonus based on transition prob to last state)
        if last_state != -1:
            # Add bonus proportional to probability of reaching last_state
            # logits[0] += stick_state * T(0 -> last_state)
            # logits[1] += stick_state * T(1 -> last_state)
            logits[0] += stick_state * transition_matrix[0, last_state]
            logits[1] += stick_state * transition_matrix[1, last_state]

        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        
        last_action_1 = action_1[trial]
        last_state = state_idx

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Choice Kernel (Decaying Stickiness)
This model extends the concept of stickiness from a 1-trial memory to a decaying trace (Choice Kernel) of past actions. This allows the model to capture longer-term trends such as streaks (strong positive kernel) or persistent alternation (strong negative kernel), which a simple 1-back stickiness parameter cannot fully capture.

```python
def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """
    Choice Kernel Model (Decaying Stickiness).
    
    Instead of just repeating the last action, the agent maintains a 'Choice Kernel'
    that integrates past choices with a decay rate. This models longer-term 
    perseveration or alternation biases.
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based values.
    - decay_k: [0, 1] Decay rate of the choice kernel (0 = only last choice matters).
    - stick_k: [-5, 5] Weight of the choice kernel (positive=perseveration, negative=alternation).
    """
    lr, beta, w, decay_k, stick_k = model_parameters
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize Choice Kernel (trace of past actions)
    choice_kernel = np.zeros(2)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):

        # Policy for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Choice Kernel bonus to logits
        logits = q_net + stick_k * choice_kernel
        
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Choice Kernel
        # Decay existing trace
        choice_kernel *= decay_k
        # Add current choice (indicator function)
        choice_kernel[action_1[trial]] += 1.0
        
        state_idx = state[trial]

        # Policy for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
