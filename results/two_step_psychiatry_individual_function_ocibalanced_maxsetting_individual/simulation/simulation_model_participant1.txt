import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Independent MB/MF Betas Model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_mb, beta_mf, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        
    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta_mb, beta_mf, stickiness = parameters
    
    # Transition matrix: A1=0 -> [0.7, 0.3], A1=1 -> [0.3, 0.7]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_mf_stage1 = np.zeros(2)
    q_mf_stage2 = np.zeros((2, 2)) # 2 states (indices 0 and 1), 2 actions
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    
    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Calculate Model-Based values: T * max(Q_MF_S2)
        max_q_stage2 = np.max(q_mf_stage2, axis=1)
        q_mb_stage1 = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values with independent betas
        logits_stage1 = (beta_mb * q_mb_stage1) + (beta_mf * q_mf_stage1)
        
        # Add stickiness if not the first trial
        if last_action_1 != -1:
            logits_stage1[last_action_1] += stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state index (0 or 1) based on transition probabilities
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Logits based on MF values and beta_mf (shared beta for MF components)
        logits_stage2 = beta_mf * q_mf_stage2[s_idx]
        
        # Softmax for Stage 2
        exp_q2 = np.exp(logits_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drifts to the specific state/action
        # State 0 corresponds to drifts 1 and 2
        # State 1 corresponds to drifts 3 and 4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        p_reward = current_drifts[s_idx][a2]
        
        r = 1 if np.random.random() < p_reward else 0
        
        # 5. Learning Updates
        # Update Stage 2 Q-values (Standard Q-learning)
        delta_2 = r - q_mf_stage2[s_idx, a2]
        q_mf_stage2[s_idx, a2] += learning_rate * delta_2
        
        # Update Stage 1 Q-values (Direct reinforcement from reward)
        q_mf_stage1[a1] += learning_rate * (r - q_mf_stage1[a1])
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        
    return stage1_choice, state2, stage2_choice, reward
