import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using Model 3: Outcome-Dependent Beta Model-Free Controller.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_win, beta_loss, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta_win, beta_loss, perseveration = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    # Initialize Q-values (matching fitting code initialization of 0.5)
    q_stage1 = np.zeros(2) + 0.5
    q_stage2 = np.zeros((2, 2)) + 0.5
    
    # Initialize history variables
    prev_a1 = -1
    prev_reward = 0  # Matches fitting code assumption
    
    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # 1. Determine Beta based on previous reward
        current_beta = beta_win if prev_reward == 1 else beta_loss
        
        # 2. Stage 1 Choice
        logits_1 = current_beta * q_stage1
        if prev_a1 != -1:
            logits_1[prev_a1] += perseveration
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # Standard 2-step task transition probabilities
        # Choice 0 -> 70% State 0, 30% State 1
        # Choice 1 -> 30% State 0, 70% State 1
        transition_probs = [0.7, 0.3] if a1 == 0 else [0.3, 0.7]
        s_idx = rng.choice([0, 1], p=transition_probs)
        
        # 4. Stage 2 Choice
        logits_2 = current_beta * q_stage2[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Outcome
        # Map state/action to specific drift parameter
        if s_idx == 0:
            current_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < current_prob else 0
        
        # 6. Value Updates (TD Learning)
        # Update Stage 2 Q-value
        delta_2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_2
        
        # Update Stage 1 Q-value using updated Stage 2 value
        delta_1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1
        
        # 7. Update History
        prev_a1 = a1
        prev_reward = r
        
        # 8. Store Data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
