import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with 
    Asymmetric Learning Rates and Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_pos, lr_neg, beta, w, lam, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr_pos, lr_neg, beta, w, lam, stickiness = parameters
    
    # Initialize Random Number Generator
    rng = np.random.default_rng()

    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])

    # Initialize Q-values
    q_mf = np.zeros(2)         # Model-free values for Stage 1
    q2 = np.zeros((2, 2))      # Values for Stage 2 (State x Action)
    
    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # Organize reward probabilities for the current trial
        # State 0 (index 0): drift1 (action 0), drift2 (action 1)
        # State 1 (index 1): drift3 (action 0), drift4 (action 1)
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        q_net = w * q_mb + (1 - w) * q_mf

        # Apply stickiness to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax for Stage 2 using the same beta
        exp_q2 = np.exp(beta * q2[s2])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)

        # --- Reward ---
        r = int(rng.random() < current_reward_probs[s2][a2])

        # --- Learning Updates ---
        
        # 1. Stage 1 Prediction Error (TD0)
        delta_1 = q2[s2, a2] - q_mf[a1]
        
        # Asymmetric learning rate for Stage 1 update
        lr_1 = lr_pos if delta_1 > 0 else lr_neg
        q_mf[a1] += lr_1 * delta_1

        # 2. Stage 2 Prediction Error
        delta_2 = r - q2[s2, a2]
        
        # Asymmetric learning rate for Stage 2 update
        lr_2 = lr_pos if delta_2 > 0 else lr_neg
        q2[s2, a2] += lr_2 * delta_2

        # 3. Eligibility Trace Update for Stage 1
        # Uses the learning rate determined by the Stage 2 error (lr_2)
        q_mf[a1] += lr_2 * lam * delta_2

        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
