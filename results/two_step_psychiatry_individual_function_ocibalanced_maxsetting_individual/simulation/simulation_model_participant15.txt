import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Model-Based RL model with 
    Asymmetric Learning and Forgetting.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, forget, beta_mb, beta_2, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    alpha_pos, alpha_neg, forget, beta_mb, beta_2, stickiness = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values for Stage 2 (2 states, 2 actions)
    # As per fitting code, initialized to 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    last_action_1 = -1
    
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Passive Forgetting
        # Decay Stage 2 values towards 0.5 before decision making
        q_stage2_mf = (1 - forget) * q_stage2_mf + forget * 0.5

        # 2. Stage 1: Model-Based Calculation
        # Calculate max value of Stage 2 states
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation: Q_MB = T * V_stage2
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate logits
        logits_1 = beta_mb * q_stage1_mb
        
        # Add stickiness bonus if not the first trial
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 Choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        logits_2 = beta_2 * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Outcome
        # Map drifts to the specific state/action
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 6. Asymmetric Learning Update
        delta = r - q_stage2_mf[s2, a2]
        
        if delta >= 0:
            q_stage2_mf[s2, a2] += alpha_pos * delta
        else:
            q_stage2_mf[s2, a2] += alpha_neg * delta
            
        # Update history
        last_action_1 = a1
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
