import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid Model-Based/Model-Free learner
    defined in cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseverance]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters matching cognitive_model1 signature
    learning_rate, beta, w, perseverance = parameters

    # Initialization matches cognitive_model1
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Value initialization
    q_mf = np.zeros(2)          # Stage 1 Model-Free values
    q_stage2 = np.zeros((2, 2)) # Stage 2 values (Aliens)
    
    last_action_1 = -1

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_mb = transition_matrix @ max_q_stage2

        q_net = w * q_mb + (1 - w) * q_mf

        # Apply perseverance bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseverance

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine next state based on transition matrix
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Get reward probability for the specific state and action
        p_reward = current_drifts[s_idx][a2]
        r = int(np.random.random() < p_reward)

        # --- Learning Updates ---
        # Matches logic in cognitive_model1 exactly
        delta1 = q_stage2[s_idx, a2] - q_mf[a1]
        q_mf[a1] += learning_rate * delta1

        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2

        q_mf[a1] += learning_rate * delta2

        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
