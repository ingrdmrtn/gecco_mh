import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Dual Stage Stickiness Model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_1, beta_2, w, stick_1, stick_2]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_1, beta_2, w, stick_1, stick_2 = parameters
    
    # Initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Stage 2 (2 states x 2 actions)
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Stickiness trackers (initialized to -1 to indicate no previous action)
    last_action_1 = -1
    last_action_2 = -1
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        # Max Q-value for each second-stage state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Expected value based on transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits_1 = beta_1 * q_net
        
        # Apply Stickiness for Stage 1
        if last_action_1 != -1:
            logits_1[last_action_1] += stick_1
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine State 2 based on transition matrix and action 1
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # --- Stage 2 Decision ---
        
        # Get Q-values for the specific state arrived at
        logits_2 = beta_2 * q_stage2_mf[s2]
        
        # Apply Stickiness for Stage 2 (Global motor perseveration)
        if last_action_2 != -1:
            logits_2[last_action_2] += stick_2
            
        # Softmax for Stage 2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Determine reward based on drift probability for State 2 / Action 2
        r_prob = current_drifts[s2][a2]
        r = 1 if rng.random() < r_prob else 0
        
        # --- Value Updates ---
        
        # Stage 1 Update (TD/SARSA logic as per source model)
        # Difference between Q of chosen stage 2 option and Q of chosen stage 1 option
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update (Rescorla-Wagner)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for stickiness
        last_action_1 = a1
        last_action_2 = a2

    return stage1_choice, state2, stage2_choice, reward
