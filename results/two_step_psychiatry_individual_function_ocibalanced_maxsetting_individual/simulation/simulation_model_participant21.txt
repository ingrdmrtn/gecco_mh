import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Dynamic Transition Learning Model.
    
    The agent learns the transition matrix from experience rather than using 
    fixed probabilities, combined with a hybrid MB/MF strategy.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_reward, lr_trans, beta_1, beta_2, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr_reward, lr_trans, beta_1, beta_2, w, stickiness = parameters
    
    # Initialize agent state
    # Agent starts with uniform beliefs about transitions
    learned_transitions = np.array([[0.5, 0.5], [0.5, 0.5]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 planets, 2 aliens per planet
    
    last_action_1 = -1
    
    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # True environment transition matrix (Standard 2-step task physics)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    true_transition_matrix = np.array([[0.7, 0.3],
                                       [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value: Expected value of next stage based on learned transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = learned_transitions @ max_q_stage2
        
        # Net Value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Action Selection Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition to Stage 2 ---
        
        # Determine state based on true environmental probabilities
        s2_idx = rng.choice([0, 1], p=true_transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        qs_current_state = q_stage2_mf[s2_idx]
        exp_q2 = np.exp(beta_2 * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        
        # Determine reward probability based on current drift values
        if s2_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < p_reward else 0
        
        # --- Learning Updates ---
        
        # 1. Update Transition Matrix (Dynamic Learning)
        # Increase prob of observed transition, decrease prob of unobserved
        learned_transitions[a1, s2_idx] += lr_trans * (1 - learned_transitions[a1, s2_idx])
        learned_transitions[a1, 1 - s2_idx] += lr_trans * (0 - learned_transitions[a1, 1 - s2_idx])
        
        # 2. Stage 1 MF Update (TD Learning)
        # Value of state 2 (Q-value of chosen action) - Value of state 1
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_reward * delta_stage1
        
        # 3. Stage 2 MF Update (Reward Prediction Error)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += lr_reward * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
