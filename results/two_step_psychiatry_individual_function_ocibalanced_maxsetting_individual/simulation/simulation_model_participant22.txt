import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Hybrid Model with Decay to Equilibrium Bias.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta_1, beta_2, w, stick, decay, q_bias]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters based on cognitive_model3 definition
    lr, beta_1, beta_2, w, stick, decay, q_bias = parameters
    
    # Task constants
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_action_1 = -1

    for t in range(n_trials):
        # Organize reward probabilities for the current trial
        # State 0 uses drift1, drift2; State 1 uses drift3, drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits and apply stickiness
        logits_1 = beta_1 * q_net
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stick
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix and action 1
        # transition_matrix[a1] gives [p(s2=0|a1), p(s2=1|a1)]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Calculate logits for the specific state
        logits_2 = beta_2 * q_stage2_mf[s2]
        
        # Softmax for Stage 2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward based on drifts
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning and Updates ---
        
        # Stage 1 MF Update (SARSA-like: using Q-value of chosen stage 2 option)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 1 Decay (Unchosen option decays to q_bias)
        unchosen_1 = 1 - a1
        q_stage1_mf[unchosen_1] = q_stage1_mf[unchosen_1] * (1 - decay) + q_bias * decay
        
        # Stage 2 MF Update (Reward Prediction Error)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2
        
        # Stage 2 Decay (Unchosen option in current state decays to q_bias)
        unchosen_2 = 1 - a2
        q_stage2_mf[s2, unchosen_2] = q_stage2_mf[s2, unchosen_2] * (1 - decay) + q_bias * decay
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
