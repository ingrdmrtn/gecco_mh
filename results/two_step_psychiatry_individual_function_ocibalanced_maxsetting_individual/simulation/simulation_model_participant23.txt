import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Stage-Specific Stickiness Model-Free Q-Learning model.
    
    This model incorporates stickiness (perseveration) for both Stage 1 and 
    Stage 2 choices independently.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stick_s1, stick_s2]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stick_s1, stick_s2 = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1: values for actions at the spaceship stage
    q_stage1 = np.zeros(2) + 0.5
    # q_stage2: values for actions at the planet stage (2 planets x 2 aliens)
    q_stage2 = np.zeros((2, 2)) + 0.5
    
    # Initialize stickiness trackers
    prev_action_1 = -1
    # Track last choice for each state (Planet 0, Planet 1) separately
    prev_action_2 = np.array([-1, -1]) 
    
    # Transition Matrix (Standard 2-step task structure)
    # Row 0 (Action 0): 0.7 prob to State 0, 0.3 prob to State 1
    # Row 1 (Action 1): 0.3 prob to State 0, 0.7 prob to State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        logits_1 = beta * q_stage1
        if prev_action_1 != -1:
            logits_1[prev_action_1] += stick_s1
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state (planet) based on transition probabilities
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        logits_2 = beta * q_stage2[s_idx]
        
        # Apply stickiness specific to the current planet (s_idx)
        if prev_action_2[s_idx] != -1:
            logits_2[prev_action_2[s_idx]] += stick_s2
            
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        # Select drift probability based on state (planet) and action (alien)
        if s_idx == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(np.random.random() < prob_reward)
        
        # --- Value Updates ---
        # Stage 1 update: SARSA-style, using the value of the state-action pair chosen in Stage 2
        target_stage1 = q_stage2[s_idx, a2]
        q_stage1[a1] += learning_rate * (target_stage1 - q_stage1[a1])
        
        # Stage 2 update: Standard Q-learning using observed reward
        q_stage2[s_idx, a2] += learning_rate * (r - q_stage2[s_idx, a2])
        
        # --- Update History ---
        prev_action_1 = a1
        prev_action_2[s_idx] = a2
        
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
