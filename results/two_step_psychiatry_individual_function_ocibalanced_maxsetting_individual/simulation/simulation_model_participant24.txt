import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using Model 1: Model-Based with Coupled Transition Learning.
    
    The agent learns a common transition probability (p_common) and uses model-based 
    planning to make Stage 1 decisions. It assumes a symmetric transition structure.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_val, lr_common, beta, pers]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr_val, lr_common, beta, pers = parameters

    # Initialization
    # Stage 2 Q-values: 2 states x 2 actions
    q_stage2 = np.zeros((2, 2))
    
    # Common transition probability (agent's belief), initialized to 0.5 (uncertain)
    p_common = 0.5
    
    # Perseverance tracking
    last_action_1 = -1
    
    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # True environment transition matrix (Standard 2-step task configuration)
    # 70% chance of common transition (0->0, 1->1), 30% rare
    true_transition_matrix = np.array([[0.7, 0.3],
                                       [0.3, 0.7]])

    for t in range(n_trials):
        # 1. --- Stage 1 Decision ---
        
        # Construct Agent's internal transition model based on p_common
        # Symmetric assumption: A->X and B->Y are governed by p_common
        agent_trans_probs = np.array([[p_common, 1.0 - p_common], 
                                      [1.0 - p_common, p_common]])

        # Model-Based Value Calculation: T * max(Q_s2)
        max_q_s2 = np.max(q_stage2, axis=1)
        q_mb = agent_trans_probs @ max_q_s2

        # Add Perseverance Bonus
        q_net_s1 = q_mb.copy()
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = np.random.choice([0, 1], p=probs_1)

        # 2. --- Transition to Stage 2 ---
        
        # Determine next state based on TRUE environment probabilities
        s_next = np.random.choice([0, 1], p=true_transition_matrix[a1])

        # 3. --- Stage 2 Decision ---
        
        # Softmax for Stage 2 (using same beta as per fitting code)
        exp_q2 = np.exp(beta * q_stage2[s_next])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = np.random.choice([0, 1], p=probs_2)

        # 4. --- Outcome / Reward ---
        
        # Get reward probability for current state/action
        # State 0: drift1 (action 0), drift2 (action 1)
        # State 1: drift3 (action 0), drift4 (action 1)
        if s_next == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < prob_reward else 0

        # 5. --- Learning Updates ---

        # Update Stage 2 Q-values (Value Learning)
        delta_2 = r - q_stage2[s_next, a2]
        q_stage2[s_next, a2] += lr_val * delta_2

        # Update Transition Model (Coupled Transition Learning)
        # Check if the observed transition was "common" (0->0 or 1->1)
        is_common_transition = (a1 == s_next)
        
        if is_common_transition:
            # Move p_common towards 1
            p_common += lr_common * (1.0 - p_common)
        else:
            # Move p_common towards 0
            p_common += lr_common * (0.0 - p_common)

        # 6. --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s_next
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
