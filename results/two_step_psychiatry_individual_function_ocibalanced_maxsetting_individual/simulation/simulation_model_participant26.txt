import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Based Learner with Outcome-Dependent Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_1, beta_2, stick_win, stick_loss]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_1, beta_2, stick_win, stick_loss = parameters
    
    # Initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage2 = np.zeros((2, 2)) # Q-values for Stage 2 (2 states, 2 actions)
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # History tracking for stickiness
    last_action_1 = -1
    last_reward = -1
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # 1. Model-Based Value Calculation
        # Calculate expected value of best action in stage 2 for each state
        max_q_stage2 = np.max(q_stage2, axis=1)
        # Map back to stage 1 actions using transition matrix
        q_stage1 = transition_matrix @ max_q_stage2
        
        # 2. Outcome-Dependent Stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                q_stage1[last_action_1] += stick_win
            else:
                q_stage1[last_action_1] += stick_loss
                
        # 3. Softmax Choice Stage 1
        exp_q1 = np.exp(beta_1 * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine next state based on action and transition matrix
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # 1. Softmax Choice Stage 2
        exp_q2 = np.exp(beta_2 * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        
        # Determine reward probability based on current drift values
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        prob_reward = current_drifts[s2][a2]
        
        r = int(np.random.random() < prob_reward)
        
        # --- Learning ---
        
        # Update Stage 2 Q-values (Model-Based agent updates the underlying model values)
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
