import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Independent Beta Hybrid Model with Forgetting.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_mb, beta_mf, forget_rate]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta_mb, beta_mf, forget_rate = parameters
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 Decision ---
        # Calculate Model-Based component (expected max value of stage 2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF components with separate betas
        logits_1 = (beta_mb * q_stage1_mb) + (beta_mf * q_stage1_mf)
        
        # Softmax for Stage 1
        probs_1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 /= np.sum(probs_1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        state_idx = s2
        
        # --- Stage 2 Decision ---
        # Driven by MF beta
        logits_2 = beta_mf * q_stage2_mf[state_idx]
        
        # Softmax for Stage 2
        probs_2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 /= np.sum(probs_2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = 1 if rng.random() < reward_probs[state_idx][a2] else 0
        
        # --- Learning Updates ---
        
        # Update Stage 2 Q-values
        delta2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta2
        # Forgetting for unchosen Stage 2 action
        q_stage2_mf[state_idx, 1-a2] *= (1.0 - forget_rate)
        
        # Update Stage 1 Q-values
        # Note: The fitting model uses the *updated* Stage 2 Q-value as the target
        target_s1 = q_stage2_mf[state_idx, a2]
        delta1 = target_s1 - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1
        # Forgetting for unchosen Stage 1 action
        q_stage1_mf[1-a1] *= (1.0 - forget_rate)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
