import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Outcome-Dependent Exploration (Beta-Win/Loss) model.
    
    The agent's exploration-exploitation balance (beta) changes depending on 
    whether the previous trial was rewarded.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta_win, beta_loss, w, stick]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr, beta_win, beta_loss, w, stick = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: State (0 or 1), Cols: Action (0 or 1)
    
    # Data storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    last_reward = 0 # Default to 0 (loss state) or neutral for first trial
    
    for t in range(n_trials):
        # 1. Determine Beta based on previous outcome
        current_beta = beta_win if last_reward == 1 else beta_loss
        
        # 2. Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness to logits
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stick
            
        exp_q1 = np.exp(current_beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        exp_q2 = np.exp(current_beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Outcome
        # Map state/action to specific drift parameter
        if s2_idx == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s2_idx == 1
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < prob_reward else 0
        
        # 6. Learning Updates
        # Stage 1 Update (TD-like using max Stage 2 Q-value logic from source code, 
        # though source uses q_stage2_mf[state, action])
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += lr * delta_stage2
        
        # 7. Store Data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
