import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Adaptive Model-Based agent with 
    Transition Learning, Value Decay, and Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_trans, lr_val, beta1, beta2, decay, pers_win, pers_loss]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters corresponding to cognitive_model3
    lr_trans, lr_val, beta1, beta2, decay, pers_win, pers_loss = parameters

    # Initialize agent state
    # Agent's internal transition model (starts with standard assumption)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Stage 2 Model-Free Q-values (initialized to 0.5)
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    # History tracking for perseveration
    last_action_1 = -1
    last_reward = -1

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # --- Stage 1: Choice ---
        
        # Calculate Model-Based values: T * max(Q_MF)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        logits_1 = beta1 * q_stage1_mb
        
        # Apply Perseveration
        if last_action_1 != -1:
            if last_reward == 1:
                logits_1[last_action_1] += pers_win
            else:
                logits_1[last_action_1] += pers_loss
        
        # Softmax Stage 1
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition to Stage 2 ---
        
        # Simulate environment transition (Standard 2-step probabilities)
        # Action 0 -> 70% State 0, 30% State 1
        # Action 1 -> 30% State 0, 70% State 1
        if a1 == 0:
            transition_probs = [0.7, 0.3]
        else:
            transition_probs = [0.3, 0.7]
        
        # Determine actual state outcome
        s_idx = np.random.choice([0, 1], p=transition_probs)

        # --- Stage 2: Choice ---
        
        logits_2 = beta2 * q_stage2_mf[s_idx]
        
        # Softmax Stage 2
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Outcome ---
        
        # Determine reward probability based on state and action
        if s_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(np.random.random() < p_reward)

        # --- Learning and Updates ---

        # 1. Update Stage 2 Q-value (Standard Q-learning)
        pe = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_val * pe

        # 2. Value Decay for unchosen options
        unchosen_a2 = 1 - a2
        # Decay unchosen action in current state
        q_stage2_mf[s_idx, unchosen_a2] = (1 - decay) * q_stage2_mf[s_idx, unchosen_a2] + decay * 0.5
        # Decay both actions in the other state
        other_state = 1 - s_idx
        q_stage2_mf[other_state, :] = (1 - decay) * q_stage2_mf[other_state, :] + decay * 0.5

        # 3. Update Transition Matrix (Adaptive MB)
        # Agent updates belief based on observed transition a1 -> s_idx
        transition_matrix[a1, s_idx] += lr_trans * (1 - transition_matrix[a1, s_idx])
        transition_matrix[a1, 1-s_idx] = 1.0 - transition_matrix[a1, s_idx]

        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
