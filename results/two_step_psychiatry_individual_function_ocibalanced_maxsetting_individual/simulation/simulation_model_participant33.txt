import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Dual Stickiness Hybrid Model with Decay.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, decay_rate, beta, w, stick_1, stick_2]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        
    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, decay_rate, beta, w, stick_1, stick_2 = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()
    
    # Task structure
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Value initialization
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # History for stickiness
    last_action_1 = -1
    last_action_2 = -1
    
    # Data storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value: Expected max value of stage 2 weighted by transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted sum of MB and MF
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits_1 = beta * q_net_stage1
        
        # Apply Stickiness to previously chosen action (if exists)
        if last_action_1 != -1:
            logits_1[last_action_1] += stick_1
            
        # Softmax Probability
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state (0 or 1) based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Get MF values for current state
        logits_2 = beta * q_stage2_mf[s_idx]
        
        # Apply Stickiness to previously chosen action (motor perseveration)
        if last_action_2 != -1:
            logits_2[last_action_2] += stick_2
            
        # Softmax Probability
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Get reward probability based on state (s_idx) and action (a2)
        # State 0 uses drift1 (action 0) and drift2 (action 1)
        # State 1 uses drift3 (action 0) and drift4 (action 1)
        if s_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < p_reward else 0
        
        # --- Learning Updates ---
        
        # Stage 1 Update (SARSA-style: using value of chosen stage 2 action)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        # Decay unchosen stage 1 action
        q_stage1_mf[1-a1] *= (1.0 - decay_rate)
        
        # Stage 2 Update (Prediction Error based on reward)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        # Decay unchosen stage 2 action in the current state
        q_stage2_mf[s_idx, 1-a2] *= (1.0 - decay_rate)
        
        # --- Update History ---
        last_action_1 = a1
        last_action_2 = a2
        
        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
