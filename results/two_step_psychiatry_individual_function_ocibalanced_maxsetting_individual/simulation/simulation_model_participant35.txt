import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Pure Model-Free Reinforcement Learning (TD-Learning) model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, eligibility_trace]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, eligibility_trace = parameters
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the first stage actions (0 and 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the second stage states (0 and 1) and actions (0 and 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (standard for the 2-step task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. First Stage Choice
        # Calculate probabilities using Softmax on Stage 1 Q-values
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Second Stage
        # Determine state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Second Stage Choice
        # Calculate probabilities using Softmax on Stage 2 Q-values for the specific state
        qs_current_planet = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * qs_current_planet)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drift inputs to the specific state/action pair
        # State 0 (s2=0) has actions mapping to drift1 and drift2
        # State 1 (s2=1) has actions mapping to drift3 and drift4
        current_reward_prob = 0.0
        if s2 == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)
        
        # 5. Learning Updates (TD-Learning)
        
        # Stage 1 Update: TD error based on Stage 2 Q-value
        # delta_stage1 = Q_stage2(s', a') - Q_stage1(s, a)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update: TD error based on Reward
        # delta_stage2 = Reward - Q_stage2(s', a')
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update: Propagate Stage 2 error back to Stage 1
        q_stage1_mf[a1] += learning_rate * eligibility_trace * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
