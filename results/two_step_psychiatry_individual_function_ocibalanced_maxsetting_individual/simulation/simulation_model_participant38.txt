import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Dynamic Transition Learning Hybrid Model with Decay.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_val, lr_trans, beta_1, beta_2, w, decay]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr_val, lr_trans, beta_1, beta_2, w, decay = parameters

    # Initialize agent's internal model
    # Agent starts with uniform transition beliefs (0.5/0.5), not the true 0.7/0.3
    trans_probs_agent = np.array([[0.5, 0.5], [0.5, 0.5]])

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    q_stage2_mf.fill(0.5)

    # Initialize environment physics (Standard 2-step task transitions)
    # This determines the actual state transition, distinct from what the agent believes
    env_transition_matrix = np.array([[0.7, 0.3],
                                      [0.3, 0.7]])

    # Data storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Stage 1 Choice
        # Calculate Model-Based value based on learned transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = trans_probs_agent @ max_q_stage2
        
        # Net Q-value (Hybrid)
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Stage 1
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Sample Action 1
        a1 = np.random.choice([0, 1], p=probs_1)

        # 2. Transition to Stage 2
        # Use the *environment's* transition matrix to determine the actual state
        s2 = np.random.choice([0, 1], p=env_transition_matrix[a1])

        # 3. Stage 2 Choice
        # Softmax Stage 2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Sample Action 2
        a2 = np.random.choice([0, 1], p=probs_2)

        # 4. Outcome
        # Get reward probability for current state/action from drifts
        # drift1/2 correspond to State 0 (Action 0/1), drift3/4 to State 1 (Action 0/1)
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r_prob = current_reward_probs[s2][a2]
        r = 1 if np.random.random() < r_prob else 0

        # 5. Learning Updates
        
        # Stage 2 Value Update (TD)
        delta_2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr_val * delta_2

        # Decay for unchosen Stage 2 option (decays towards 0.5)
        unchosen_a2 = 1 - a2
        q_stage2_mf[s2, unchosen_a2] = (1 - decay) * q_stage2_mf[s2, unchosen_a2] + decay * 0.5

        # Stage 1 Value Update (TD using updated Q2 value)
        delta_1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_val * delta_1

        # Transition Probability Update (Agent learns the structure)
        # Increases probability of the observed transition (a1 -> s2)
        trans_probs_agent[a1, s2] += lr_trans * (1 - trans_probs_agent[a1, s2])
        # Ensure probabilities sum to 1
        trans_probs_agent[a1, 1 - s2] = 1.0 - trans_probs_agent[a1, s2]

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
