import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model-Based / Model-Free learner 
    with Uncertainty-Based Exploration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta_1, beta_2, w, stick, phi]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr, beta_1, beta_2, w, stick, phi = parameters
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 states, 2 options
    
    # Exploration and stickiness tracking
    time_since_choice_1 = np.zeros(2)
    prev_action_1 = -1
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # Organize drift probabilities for current trial
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 ---
        
        # Increment uncertainty counter
        time_since_choice_1 += 1
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Weighted MB + MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Stickiness
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
            
        # Add Exploration Bonus
        q_net += phi * time_since_choice_1
        
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = np.argmax(np.random.multinomial(1, probs_1))
        
        # Reset uncertainty for chosen option
        time_since_choice_1[a1] = 0
        prev_action_1 = a1
        
        # --- Transition ---
        
        # Determine State 2 based on transition matrix
        state_probs = transition_matrix[a1]
        state_idx = np.argmax(np.random.multinomial(1, state_probs))
        
        # --- Stage 2 ---
        
        # Get Q-values for current state
        qs_2 = q_stage2_mf[state_idx]
        
        # Softmax Policy Stage 2
        exp_q2 = np.exp(beta_2 * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = np.argmax(np.random.multinomial(1, probs_2))
        
        # --- Reward ---
        
        # Stochastic reward based on drift probabilities
        r = 1 if np.random.rand() < reward_probs[state_idx][a2] else 0
        
        # --- Learning Updates ---
        
        # Update Stage 2 Q-values (MF)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Update Stage 1 Q-values (MF)
        # Uses the updated Stage 2 value as the target (following fitting code logic)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Record data
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
