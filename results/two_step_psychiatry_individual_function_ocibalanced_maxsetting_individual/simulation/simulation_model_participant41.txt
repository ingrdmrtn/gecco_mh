import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Risk-Sensitive Q-Learning model 
    with Outcome-Dependent Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, beta_1, beta_2, sticky_win, sticky_lose]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    alpha_pos, alpha_neg, beta_1, beta_2, sticky_win, sticky_lose = parameters
    
    # Initialization
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2)) # state x action
    
    prev_a1 = -1
    prev_reward = -1
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Transition matrix (standard 2-step task configuration)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        # Calculate logits based on Q-values
        logits_1 = beta_1 * q_stage1.copy()
        
        # Apply Outcome-Dependent Stickiness
        if prev_a1 != -1:
            if prev_reward == 1:
                logits_1[prev_a1] += sticky_win
            else:
                logits_1[prev_a1] += sticky_lose
        
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        logits_2 = beta_2 * q_stage2[s2]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        # Determine reward probability based on current state, action, and drift
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s2 == 1
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # --- Learning (Risk Sensitive) ---
        
        # Stage 1 Update (TD(0))
        # Note: Uses Q-value of chosen stage 2 action (SARSA-like logic in fitting code)
        delta1 = q_stage2[s2, a2] - q_stage1[a1]
        lr1 = alpha_pos if delta1 > 0 else alpha_neg
        q_stage1[a1] += lr1 * delta1
        
        # Stage 2 Update
        delta2 = r - q_stage2[s2, a2]
        lr2 = alpha_pos if delta2 > 0 else alpha_neg
        q_stage2[s2, a2] += lr2 * delta2
        
        # Update history
        prev_a1 = a1
        prev_reward = r
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
