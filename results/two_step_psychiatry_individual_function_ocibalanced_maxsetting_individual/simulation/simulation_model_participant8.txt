import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid Model with Decay and Eligibility Traces.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, decay, beta, w, lam, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr, decay, beta, w, lam, stickiness = parameters
    
    # Initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_a1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        # Map state/action to specific drift array
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < p_reward else 0
        
        # --- Updates ---
        
        # Stage 1 TD Error (TD(0))
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 TD Error
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2
        
        # Eligibility Trace Update (TD(lambda) effect on Stage 1)
        q_stage1_mf[a1] += lr * lam * delta_stage2
        
        # Decay Unchosen Options
        # Decay unchosen stage 1 action
        q_stage1_mf[1 - a1] *= (1 - decay)
        
        # Decay all stage 2 Q-values except the specific state-action pair visited
        for s in range(2):
            for a in range(2):
                if not (s == s2 and a == a2):
                    q_stage2_mf[s, a] *= (1 - decay)
        
        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_a1 = a1
        
    return stage1_choice, state2, stage2_choice, reward
