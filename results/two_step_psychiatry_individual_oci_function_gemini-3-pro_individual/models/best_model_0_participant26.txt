def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Driven Perseveration (Stickiness).
    
    This model assumes OCI modulates choice stickiness (perseveration).
    It uses a purely Model-Based planner for Stage 1 values, but adds a 
    stickiness bonus to the previously chosen option, the magnitude of which 
    depends on the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature.
    - stick_base: [0, 5] Baseline tendency to repeat the last choice.
    - oci_stick_mod: [0, 5] Additional stickiness modulated by OCI score.
    """
    learning_rate, beta, stick_base, oci_stick_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage2_mf = np.zeros((2, 2))

    last_choice_1 = -1


    total_stickiness = stick_base + (current_oci * oci_stick_mod)

    for trial in range(n_trials):


        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        logits = beta * q_stage1_mb
        if last_choice_1 != -1:
            logits[last_choice_1] += total_stickiness
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]

        last_choice_1 = int(action_1[trial])
        
        state_idx = int(state[trial])


        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]


        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
