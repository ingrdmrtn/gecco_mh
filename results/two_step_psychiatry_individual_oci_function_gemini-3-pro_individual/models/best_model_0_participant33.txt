def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free with OCI-modulated Choice Stickiness.
    
    This model computes a net Q-value as a weighted sum of Model-Based (MB) and 
    Model-Free (MF) values. It adds a "stickiness" bonus to the action chosen 
    in the previous trial. The magnitude of this stickiness is modulated by the OCI score.

    Parameters:
    - learning_rate: Rate of value updating [0, 1]
    - beta: Inverse temperature for softmax [0, 10]
    - w: Mixing weight for MB vs MF systems [0, 1] (0=Pure MF, 1=Pure MB)
    - stickiness_base: Base tendency to repeat the previous Stage 1 choice [-5, 5]
    - stickiness_oci_mod: Effect of OCI score on stickiness [-5, 5]
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    eff_stickiness = stickiness_base + (stickiness_oci_mod * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1

    for trial in range(n_trials):


        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf

        q_net = q_hybrid.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += eff_stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]

        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        r = reward[trial]

        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
