Here are three cognitive models that incorporate the OCI score to explain participant behavior in the two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that obsessive-compulsive traits influence the balance between goal-directed (model-based) and habitual (model-free) control. Higher OCI scores might be associated with a stronger reliance on habitual (model-free) systems or a deficit in model-based planning. Here, the `w` parameter (mixing weight) is modulated by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid learner where the balance (w) between Model-Based (MB) and 
    Model-Free (MF) control is modulated by the OCI score.
    
    Hypothesis: Higher OCI scores shift the balance towards MF (habitual) control 
    or reduce the efficacy of MB planning.
    
    Parameters:
    learning_rate (alpha): [0, 1] Speed of value updating.
    beta: [0, 10] Inverse temperature for softmax choice.
    w_base: [0, 1] Baseline mixing weight (0=pure MF, 1=pure MB).
    oci_sensitivity: [0, 1] How strongly OCI reduces the model-based weight.
    """
    learning_rate, beta, w_base, oci_sensitivity = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w based on OCI
    # We hypothesize OCI reduces model-based control (w). 
    # We clip to ensure it stays in [0, 1].
    w = w_base * (1.0 - (oci_score * oci_sensitivity))
    w = np.clip(w, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Calculation: V(state) = max(Q_stage2)
        # We assume the transition matrix is fixed/known as per standard 2-step models
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of the chosen action
        # Ensure indices are integers
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        qs_stage2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updating ---
        r = reward[trial]
        
        # Stage 2 Update (standard Q-learning)
        # Prediction error stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD(0))
        # Prediction error stage 1
        # Using the value of the state actually reached
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Perseveration (Stickiness)
This model focuses on "stickiness" or choice perseveration. A core feature of OC symptoms is repetitive behavior or difficulty switching sets. This model posits that the OCI score directly scales a perseveration parameter, making the participant more likely to repeat their previous Stage 1 choice regardless of reward history.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Reinforcement Learning with OCI-modulated Perseveration.
    
    Hypothesis: Higher OCI scores lead to higher choice stickiness (repetitive 
    selection of the same spaceship), independent of reward outcomes.
    
    Parameters:
    learning_rate: [0, 1] Speed of value updating.
    beta: [0, 10] Inverse temperature.
    persev_base: [0, 5] Baseline tendency to repeat the last choice.
    oci_persev_scale: [0, 5] Additional stickiness scaled by OCI score.
    """
    learning_rate, beta, persev_base, oci_persev_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective perseveration bonus
    # Total stickiness = base + (OCI * scale)
    stickiness = persev_base + (oci_score * oci_persev_scale)
    
    # We simplify by using a pure Model-Free architecture for the base learning
    # to isolate the effect of perseveration on the first stage.
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Indicator for previous choice

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Calculate logits: Q-value + Stickiness Bonus
        logits_1 = beta * q_stage1.copy()
        
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Softmax
        # Subtract max for numerical stability
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        logits_2 = beta * q_stage2[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updating ---
        r = reward[trial]
        
        # Stage 2 Update (Q-learning)
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (SARSA-style / TD)
        # Using the value of the state visited
        v_next = q_stage2[s_idx, a2]
        delta_stage1 = v_next - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Loss Aversion
This model investigates if OCI scores correlate with an altered sensitivity to negative outcomes (receiving 0 coins). It splits the learning rate into positive (`alpha_pos`) and negative (`alpha_neg`) components, where the negative learning rate is modulated by the OCI score. This reflects the hypothesis that individuals with higher OC traits might be hyper-sensitive to errors or missed rewards.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Dual Learning Rate Model where OCI modulates learning from non-reward (loss).
    
    Hypothesis: OCI is associated with hyper-sensitivity to errors/losses (0 coins).
    The learning rate for negative prediction errors is scaled by OCI.
    
    Parameters:
    alpha_pos: [0, 1] Learning rate for positive prediction errors (RPE > 0).
    alpha_neg_base: [0, 1] Base learning rate for negative prediction errors (RPE < 0).
    beta: [0, 10] Inverse temperature.
    oci_loss_amp: [0, 2] Multiplier for OCI to increase negative learning rate.
    """
    alpha_pos, alpha_neg_base, beta, oci_loss_amp = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate
    # alpha_neg increases with OCI, capped at 1.0
    alpha_neg = alpha_neg_base * (1.0 + (oci_score * oci_loss_amp))
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    # Fixed transition matrix for simple Model-Based estimation (if we were using it)
    # But here we use a simple Model-Free structure to isolate learning rate effects.
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updating ---
        r = reward[trial]
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2[s_idx, a2]
        
        # Select learning rate based on sign of prediction error
        lr_2 = alpha_pos if delta_stage2 >= 0 else alpha_neg
        q_stage2[s_idx, a2] += lr_2 * delta_stage2
        
        # Stage 1 Update
        # Using simple TD(0) update based on stage 2 value
        v_next = q_stage2[s_idx, a2]
        delta_stage1 = v_next - q_stage1[a1]
        
        lr_1 = alpha_pos if delta_stage1 >= 0 else alpha_neg
        q_stage1[a1] += lr_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
