Here are three cognitive models designed to explain the participant's behavior in the two-step task, incorporating their OCI score into the decision-making or learning process.

### Model 1: Hybrid Learner with OCI-Modulated Mixing Weight
This model assumes that the participant uses a hybrid of Model-Based (MB) and Model-Free (MF) reinforcement learning strategies. The key hypothesis here is that the balance between these strategies (`w`) is influenced by the participant's OCI score. Even though this participant has a low score, the functional form allows the model to scale this influence across a population.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the mixing weight 'w' is 
    modulated by the OCI score.
    
    Model Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w_base: [0, 1] Base weight for Model-Based control (0 = pure MF, 1 = pure MB).
    - w_oci_slope: [0, 1] Sensitivity of the mixing weight to the OCI score.
    """
    learning_rate, beta, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate mixing weight w, bounded between 0 and 1.
    # We hypothesize OCI might push towards habit (MF) or planning (MB).
    # Here, we model it as an additive effect on the base weight.
    w = w_base + (w_oci_slope * current_oci)
    w = np.clip(w, 0.0, 1.0)

    # Transition matrix (fixed for this task structure)
    # 0 -> 0 (common 70%), 0 -> 1 (rare 30%)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (Aliens)

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE ---
        # Model-Based Value Calculation: V(state) = max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Observe state transition
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- STAGE 2 CHOICE ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Observe Reward
        r = reward[trial]
        a2 = int(action_2[trial])

        # --- UPDATES ---
        # Stage 2 Update (TD Learning)
        # Prediction Error 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD Learning)
        # Prediction Error 1 (using the value of the state actually reached)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Perseveration (Stickiness)
This model posits that OCI scores relate to "stickiness" or perseverationâ€”the tendency to repeat the previous choice regardless of reward history. A low OCI score (like this participant's) might imply very low stickiness, or standard flexibility, whereas high scores might imply rigid repetition.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with choice perseveration (stickiness) modulated by OCI.
    
    Model Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - stickiness_base: [0, 5] Base bonus added to the previously chosen action.
    - stickiness_oci_mod: [0, 5] Additional stickiness scaled by OCI score.
    """
    learning_rate, beta, stickiness_base, stickiness_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective stickiness parameter
    # Higher OCI -> Higher tendency to repeat choices (if parameter is positive)
    perseveration_weight = stickiness_base + (stickiness_oci_mod * current_oci)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness (initialize as -1 or random)
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE ---
        # Add stickiness bonus to Q-values before softmax
        q_stage1_adjusted = q_stage1_mf.copy()
        if prev_action_1 != -1:
            q_stage1_adjusted[prev_action_1] += perseveration_weight
            
        exp_q1 = np.exp(beta * q_stage1_adjusted)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        prev_action_1 = a1 # Update previous action

        # --- STAGE 2 CHOICE ---
        # No stickiness modeled for stage 2 to keep parameters low/focused
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        r = reward[trial]
        a2 = int(action_2[trial])

        # --- UPDATES ---
        # Standard SARSA/Q-learning updates
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Learning Rate Asymmetry
This model investigates if OCI affects how participants learn from positive versus negative outcomes. Specifically, it tests if the OCI score modulates the learning rate for *negative* prediction errors (disappointments), potentially reflecting a sensitivity to failure or checking behavior often associated with OC traits.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with asymmetric learning rates for positive and negative
    prediction errors, where the negative learning rate is modulated by OCI.
    
    Model Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors (better than expected).
    - alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    - alpha_neg_oci_scale: [0, 1] Scaling factor for OCI's impact on negative learning rate.
    - beta: [0, 10] Inverse temperature.
    """
    alpha_pos, alpha_neg_base, alpha_neg_oci_scale, beta = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective negative learning rate
    # We allow OCI to increase sensitivity to negative outcomes
    alpha_neg = alpha_neg_base + (alpha_neg_oci_scale * current_oci)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- STAGE 2 CHOICE ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        r = reward[trial]
        a2 = int(action_2[trial])

        # --- UPDATES ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Apply asymmetric learning rates
        if delta_stage2 >= 0:
            q_stage2_mf[state_idx, a2] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[state_idx, a2] += alpha_neg * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        if delta_stage1 >= 0:
            q_stage1_mf[a1] += alpha_pos * delta_stage1
        else:
            q_stage1_mf[a1] += alpha_neg * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
