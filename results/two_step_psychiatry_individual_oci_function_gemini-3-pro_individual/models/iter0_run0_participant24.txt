Here are three cognitive models designed to capture decision-making in the two-step task, incorporating the OCI-R score to modulate specific mechanisms like stickiness, model-based control, and learning rates.

### Model 1: OCI-Modulated Model-Based/Model-Free Hybrid
This model hypothesizes that individuals with higher OCI scores rely more heavily on habit-based (Model-Free) learning rather than goal-directed (Model-Based) planning. The `w` parameter (mixing weight) is modulated by the OCI score. A higher OCI score reduces `w`, pushing the agent towards Model-Free control.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free reinforcement learning model where the balance 
    between systems (w) is modulated by the OCI score.
    
    Hypothesis: Higher OCI scores lead to reduced Model-Based control (lower w),
    reflecting a reliance on habitual behaviors.

    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w_base: [0, 1] Baseline weight for Model-Based control (0=MF, 1=MB).
    - oci_sens: [0, 5] Sensitivity of the weighting parameter to the OCI score.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1]
    oci_sens: [0, 5]
    """
    learning_rate, beta, w_base, oci_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] # Extract scalar value

    # Modulate w based on OCI. We clip to ensure it stays in [0, 1].
    # Higher OCI reduces w (less model-based).
    w = w_base - (oci_sens * current_oci)
    w = np.clip(w, 0.0, 1.0)

    # Fixed transition matrix for Model-Based calculation
    # A -> X (0.7), Y (0.3); U -> X (0.3), Y (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (State x Action)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Calculate Model-Based values: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value: w * MB + (1-w) * MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax policy
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        # Standard Model-Free Q-learning for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # TD Error Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # TD Error Stage 1 (SARSA-style update using Stage 2 value)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Choice Stickiness (Perseveration)
This model posits that OCI symptoms manifest as "stickiness" or perseverationâ€”a tendency to repeat the previous action regardless of the reward outcome. The OCI score directly scales the `stickiness` parameter, making high-OCI individuals more likely to repeat their Stage 1 choice.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-learning with Choice Stickiness modulated by OCI.
    
    Hypothesis: OCI is associated with behavioral rigidity or perseveration.
    The OCI score increases the 'stickiness' parameter, making the agent more 
    likely to repeat the previous Stage 1 action regardless of reward.

    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - stickiness_base: [-2, 2] Baseline tendency to repeat choices.
    - oci_perseveration: [0, 5] How strongly OCI increases stickiness.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    stickiness_base: [0, 2]
    oci_perseveration: [0, 5]
    """
    learning_rate, beta, stickiness_base, oci_perseveration = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective stickiness
    # Higher OCI -> Higher stickiness (stronger urge to repeat previous choice)
    stickiness_param = stickiness_base + (oci_perseveration * current_oci)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize invalid action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Add stickiness bonus to the Q-value of the previously chosen action
        q_net = q_stage1.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_param
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Record this action for the next trial's stickiness
        last_action_1 = int(action_1[trial])
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Update Stage 2
        delta_stage2 = r - q_stage2[state_idx, a2]
        q_stage2[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 (TD-0 using Stage 2 value)
        delta_stage1 = q_stage2[state_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Punishment Sensitivity
This model suggests that OCI relates to anxiety and an increased sensitivity to negative outcomes (lack of reward). Here, the learning rate is split into positive (`lr_pos`) and negative (`lr_neg`) components. The OCI score amplifies `lr_neg`, meaning the agent updates its values more drastically after a loss (0 coins) than a win.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dual Learning Rate Model (Positive/Negative) where OCI modulates negative learning rate.
    
    Hypothesis: OCI correlates with error-sensitivity or fear of failure.
    Higher OCI scores amplify the learning rate for negative prediction errors 
    (when reward is 0), causing faster avoidance learning.

    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors (Reward > Expectation).
    - lr_neg_base: [0, 1] Baseline learning rate for negative prediction errors.
    - beta: [0, 10] Inverse temperature.
    - oci_anxiety: [0, 5] Multiplier for OCI effect on negative learning rate.
    
    Bounds:
    lr_pos: [0, 1]
    lr_neg_base: [0, 1]
    beta: [0, 10]
    oci_anxiety: [0, 5]
    """
    lr_pos, lr_neg_base, beta, oci_anxiety = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective negative learning rate
    # We constrain it to be <= 1.0
    lr_neg = lr_neg_base + (oci_anxiety * current_oci)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Based for Stage 1 value estimation (simplification to focus on LR)
    # But we track MF values to apply the learning rates
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Calculate MB values based on current Stage 2 estimates
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning Updates ---
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Calculate Prediction Error
        pe = r - q_stage2_mf[state_idx, a2]
        
        # Apply asymmetric learning rates
        if pe >= 0:
            q_stage2_mf[state_idx, a2] += lr_pos * pe
        else:
            q_stage2_mf[state_idx, a2] += lr_neg * pe
            
        # Note: In this specific formulation, Stage 1 values are derived purely 
        # from the transition matrix and Stage 2 values (Pure MB), so we do not 
        # update a separate q_stage1_mf array.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
