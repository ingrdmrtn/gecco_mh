Here are three cognitive models designed to explain the participant's behavior, incorporating their Obsessive-Compulsive Inventory (OCI) score.

### Cognitive Model 1: Hybrid Learner with OCI-Modulated Mixing
This model assumes that the balance between Model-Based (planning) and Model-Free (habitual) learning is influenced by the participant's OCI score. Specifically, it posits that even low-OCI individuals (like this participant) have a baseline mixing weight `w` that shifts based on their compulsive tendencies. Since this participant has a low score (0.1875), the model tests if lower OCI leads to a stronger reliance on Model-Based planning (higher `w`) or Model-Free habits.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model where the balance between Model-Based (MB) and Model-Free (MF) 
    systems is modulated by the OCI score.
    
    Parameters:
    learning_rate: [0,1] - Rate of updating Q-values.
    beta: [0,10] - Inverse temperature for softmax choice.
    w_base: [0,1] - Baseline weight for Model-Based control (0=Pure MF, 1=Pure MB).
    oci_sensitivity: [0, 5] - How strongly OCI affects the mixing weight w.
    """
    learning_rate, beta, w_base, oci_sensitivity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] # 0.1875 for this participant
    
    # Transition matrix: A->X (0.7), A->Y (0.3), U->X (0.3), U->Y (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for Step 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Step 2 (State x Action)
    
    # Calculate OCI-modulated mixing weight
    # We use a logistic-style modulation or simple linear clip. 
    # Here: w = w_base - (oci * sensitivity). 
    # Hypothesis: Higher OCI might reduce MB control (planning), reducing w.
    w = w_base - (current_oci * oci_sensitivity)
    w = np.clip(w, 0.0, 1.0) # Ensure w stays valid

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        curr_state = int(state[trial])
        curr_a1 = int(action_1[trial])
        
        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[curr_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        curr_a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[curr_a2]
        
        # --- Learning / Updating ---
        # Prediction error for Stage 2 (Reward driven)
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[curr_state, curr_a2]
        q_stage2_mf[curr_state, curr_a2] += learning_rate * delta_stage2
        
        # Prediction error for Stage 1 (TD-learning driven by Stage 2 value)
        # Note: In hybrid models, MF update uses the value of the state actually reached
        delta_stage1 = q_stage2_mf[curr_state, curr_a2] - q_stage1_mf[curr_a1]
        q_stage1_mf[curr_a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Driven Perseveration
This model hypothesizes that OCI scores relate to "stickiness" or perseveration (the tendency to repeat the previous action regardless of reward). Even though this participant has a low score, the model allows the data to determine if their specific level of OCI contributes to a non-zero perseveration bonus.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with OCI-modulated choice perseveration (stickiness).
    
    Parameters:
    learning_rate: [0,1] - Rate of updating Q-values.
    beta: [0,10] - Inverse temperature for softmax choice.
    perseveration_base: [0, 5] - Baseline tendency to repeat the last choice.
    oci_pers_factor: [0, 5] - How much OCI amplifies perseveration.
    """
    learning_rate, beta, perseveration_base, oci_pers_factor = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective perseveration bonus
    # Hypothesis: Higher OCI leads to higher repetition compulsion.
    pers_bonus = perseveration_base + (current_oci * oci_pers_factor)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize as invalid

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Add stickiness bonus to Q-values before softmax
        q_stage1_effective = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_effective[last_action_1] += pers_bonus
            
        exp_q1 = np.exp(beta * q_stage1_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        curr_a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[curr_a1]
        
        curr_state = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[curr_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        curr_a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[curr_a2]
        
        # --- Learning ---
        # SARSA / TD(0) style updates
        delta_stage1 = q_stage2[curr_state, curr_a2] - q_stage1[curr_a1]
        q_stage1[curr_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2[curr_state, curr_a2]
        q_stage2[curr_state, curr_a2] += learning_rate * delta_stage2
        
        last_action_1 = curr_a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Learning Rates
This model suggests that OCI affects the *flexibility* of belief updating (learning rate). High OCI is sometimes associated with rigid beliefs (low learning rate) or hyper-sensitivity to error (high learning rate). This model tests if the participant's learning rate is a function of their OCI score, allowing separate learning rates for the first stage (spaceship choice) and second stage (alien choice).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Based learner where learning rates are modulated by OCI.
    Tests if OCI affects plasticity (speed of updating values).
    
    Parameters:
    lr_base: [0,1] - Base learning rate.
    beta: [0,10] - Inverse temperature.
    oci_lr_mod: [0, 2] - Multiplier for OCI's effect on learning rate.
                         (Can dampen or accelerate learning).
    """
    lr_base, beta, oci_lr_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective learning rate
    # We model this as lr = lr_base / (1 + oci * mod) 
    # This assumes OCI might induce rigidity (lower LR), but parameters can fit otherwise.
    # If oci_lr_mod is 0, it reverts to standard LR.
    effective_lr = lr_base / (1.0 + (current_oci * oci_lr_mod))
    effective_lr = np.clip(effective_lr, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # We only track Stage 2 values, as Stage 1 is derived purely Model-Based here
    q_stage2 = np.zeros((2, 2)) 

    for trial in range(n_trials):
        # --- Stage 1 Policy (Pure Model-Based) ---
        # Value of action 1 is expected value of states it leads to
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        curr_a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[curr_a1]
        
        curr_state = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[curr_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        curr_a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[curr_a2]
        
        # --- Learning ---
        # Update Stage 2 values based on reward
        delta_stage2 = reward[trial] - q_stage2[curr_state, curr_a2]
        q_stage2[curr_state, curr_a2] += effective_lr * delta_stage2
        
        # No Stage 1 update needed for pure MB, as it's calculated on the fly 
        # from transition matrix and current Stage 2 values.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
