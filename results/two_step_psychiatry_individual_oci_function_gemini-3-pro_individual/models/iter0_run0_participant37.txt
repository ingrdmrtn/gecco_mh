Here are three cognitive models proposing different mechanisms for how Obsessive-Compulsive Inventory (OCI) scores might influence reinforcement learning in this two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that individuals with higher OCI scores rely more on habitual (model-free) control at the expense of goal-directed (model-based) planning. This aligns with theories suggesting compulsivity is linked to overactive habit formation. The mixing parameter `w` is modulated by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-modulated Model-Based/Model-Free Balance.
    
    This model assumes that the balance between Model-Based (planning) and 
    Model-Free (habitual) systems is influenced by the participant's OCI score.
    Higher OCI scores reduce the weight (w) of the Model-Based system, leading to 
    more habitual behavior.
    
    Parameters:
    learning_rate: [0, 1] Rate at which Q-values are updated.
    beta: [0, 10] Inverse temperature for softmax choice.
    w_base: [0, 1] Baseline mixing weight (1 = pure MB, 0 = pure MF).
    oci_sensitivity: [0, 2] How strongly OCI reduces the model-based weight.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_base: [0,1]
    oci_sensitivity: [0,2]
    """
    learning_rate, beta, w_base, oci_sensitivity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    # Calculate effective mixing weight w based on OCI
    # We constrain w to be between 0 and 1.
    # Higher OCI subtracts from the base weight, pushing towards Model-Free (0).
    w = w_base - (oci_sensitivity * current_oci)
    if w < 0: w = 0
    if w > 1: w = 1

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 planets, 2 aliens)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values for the current state
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updating ---
        # SARSA(0) / TD-learning
        
        # Prediction error for stage 1 (using the value of the state actually reached)
        # Note: In standard 2-step models, stage 1 MF is updated using the value of the *chosen* stage 2 action
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Prediction error for stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] = q_stage2_mf[s_idx, a2] + learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Perseveration
This model posits that OCI symptoms manifest as "stickiness" or perseverationâ€”a tendency to repeat the previous choice regardless of reward. Here, the OCI score directly scales a perseveration parameter added to the Q-values, making high-OCI participants more likely to repeat their Stage 1 choice.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Driven Perseveration.
    
    This model assumes OCI is linked to behavioral rigidity or 'stickiness'.
    The OCI score scales a perseveration parameter (stickiness) which biases 
    the participant to repeat their previous Stage 1 choice, regardless of the outcome.
    
    Parameters:
    learning_rate: [0, 1] Update rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based control (fixed).
    pers_scaling: [0, 5] How strongly OCI increases choice repetition/stickiness.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1]
    pers_scaling: [0,5]
    """
    learning_rate, beta, w, pers_scaling = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Effective perseveration bonus based on OCI
    eff_stickiness = pers_scaling * current_oci
    
    last_action_1 = -1 # Indicator for previous action

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the Q-value of the action taken on the previous trial
        q_net_sticky = q_net.copy()
        if last_action_1 != -1:
            q_net_sticky[last_action_1] += eff_stickiness

        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updating ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] = q_stage2_mf[s_idx, a2] + learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Enhanced Learning Rate (Hyper-learning)
This model suggests that OCI is associated with hyper-sensitivity to prediction errors. Individuals with higher OCI scores might over-update their value estimates in response to outcomes. The learning rate `alpha` is boosted by the OCI score, leading to more volatile value estimations.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Enhanced Learning Rate.
    
    This model tests the hypothesis that higher OCI scores lead to higher learning rates
    (hyper-learning), reflecting an over-sensitivity to recent prediction errors or 
    an inability to integrate rewards stably over time.
    
    Parameters:
    alpha_base: [0, 1] Baseline learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based control.
    oci_boost: [0, 1] Additional learning rate magnitude scaled by OCI.
    
    Bounds:
    alpha_base: [0,1]
    beta: [0,10]
    w: [0,1]
    oci_boost: [0,1]
    """
    alpha_base, beta, w, oci_boost = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate effective learning rate
    # We cap the learning rate at 1.0 to maintain mathematical validity.
    effective_alpha = alpha_base + (oci_boost * current_oci)
    if effective_alpha > 1.0:
        effective_alpha = 1.0

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updating ---
        # Use the OCI-enhanced learning rate for updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + effective_alpha * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] = q_stage2_mf[s_idx, a2] + effective_alpha * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
