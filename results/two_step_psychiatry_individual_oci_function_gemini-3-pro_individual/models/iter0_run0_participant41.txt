Here are three cognitive models proposing different mechanisms for how Obsessive-Compulsive Inventory (OCI) scores might influence decision-making in this two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that OCI scores influence the balance between Model-Based (planning-oriented) and Model-Free (habit-oriented) control. Literature often suggests that compulsivity relates to an over-reliance on habitual (Model-Free) control or a deficit in goal-directed (Model-Based) control. Here, the `w` parameter (mixing weight) is modulated by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Based/Model-Free Balance.
    
    This model assumes that the balance between Model-Based (MB) and Model-Free (MF) 
    systems is influenced by the participant's OCI score. Higher OCI scores might 
    shift the balance towards habitual (MF) control.
    
    The mixing weight 'w' is calculated as: w_base + (w_oci_slope * oci).
    w = 1 implies pure Model-Based, w = 0 implies pure Model-Free.
    
    Parameters:
    learning_rate: [0, 1] Rate at which Q-values are updated.
    beta: [0, 10] Inverse temperature for softmax choice (exploration/exploitation).
    w_base: [0, 1] Baseline mixing weight for MB vs MF.
    w_oci_slope: [-1, 1] How strongly OCI affects the MB/MF balance.
    """
    learning_rate, beta, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective w constrained between 0 and 1
    w = w_base + (w_oci_slope * oci_val)
    w = np.clip(w, 0.0, 1.0)

    # Transition matrix (fixed for this task structure)
    # A -> X (0.7), A -> Y (0.3); U -> X (0.3), U -> Y (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # MF values for Stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # MF values for Stage 2 (Aliens per Planet)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Bellman equation using transition matrix and max stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax policy for stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of the chosen action (avoiding index errors)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial]) # 0 for Planet X, 1 for Planet Y

        # --- Stage 2 Choice ---
        # Standard Model-Free Q-learning at Stage 2
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning Updates ---
        r = reward[trial]
        
        # Stage 2 Update (standard Q-learning)
        # Q2(s, a2) = Q2(s, a2) + alpha * (r - Q2(s, a2))
        pe_2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * pe_2
        
        # Stage 1 Update (TD-learning)
        # Q1(a1) = Q1(a1) + alpha * (Q2(s, a2) - Q1(a1))
        # Note: We use the updated Q2 value or the value before update? 
        # SARSA usually uses value of next state. Here we use the value of the state we landed in.
        pe_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Perseveration (Stickiness)
This model posits that OCI symptoms manifest as behavioral rigidity or "stickiness" (perseveration). Participants with higher OCI scores might be more likely to repeat their previous Stage 1 choice regardless of the reward outcome, reflecting a compulsion to repeat actions.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Driven Perseveration (Stickiness).
    
    This model assumes that OCI scores increase choice perseveration (stickiness).
    A 'stickiness' bonus is added to the Q-value of the previously chosen action.
    The magnitude of this bonus is scaled by the OCI score.
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-values.
    beta: [0, 10] Inverse temperature.
    stickiness_base: [0, 5] Baseline tendency to repeat the previous choice.
    stickiness_oci: [0, 5] Additional stickiness multiplier based on OCI.
    """
    learning_rate, beta, stickiness_base, stickiness_oci = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective stickiness
    # Higher OCI leads to higher tendency to repeat
    eff_stickiness = stickiness_base + (stickiness_oci * oci_val)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Based Q-values for Stage 1 logic (simplified for this model focus)
    # We track MF values for stage 2 to feed into MB calculation
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add Stickiness Bonus
        q_net = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += eff_stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1 # Update for next trial
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning Updates ---
        r = reward[trial]
        
        # Only updating Stage 2 MF values here to keep model focus on stickiness + MB planning
        pe_2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * pe_2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Learning Rate Asymmetry
This model suggests that OCI scores affect how participants learn from positive versus negative prediction errors. High OCI might be associated with a "perfectionist" or anxiety-driven learning style where negative outcomes (missing a coin) drive learning more strongly than positive ones, or vice versa.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Learning Rate Asymmetry.
    
    This model separates learning rates for positive (alpha_pos) and negative (alpha_neg)
    prediction errors. The OCI score modulates the ratio between these two.
    Specifically, we test if OCI amplifies learning from negative outcomes (avoidance).
    
    alpha_pos = base_alpha
    alpha_neg = base_alpha * (1 + oci_penalty_mult * oci)
    
    Parameters:
    base_alpha: [0, 1] Baseline learning rate.
    beta: [0, 10] Inverse temperature.
    oci_penalty_mult: [0, 5] Multiplier for negative prediction errors based on OCI.
    """
    base_alpha, beta, oci_penalty_mult = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Define learning rates
    alpha_pos = base_alpha
    # If OCI is high, alpha_neg becomes larger than alpha_pos
    alpha_neg = base_alpha * (1.0 + oci_penalty_mult * oci_val)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0) # Ensure it stays valid

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Using a simple hybrid: Average of MB and MF for the base policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = 0.5 * q_stage1_mb + 0.5 * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning Updates ---
        r = reward[trial]
        
        # Stage 2 Update
        pe_2 = r - q_stage2_mf[state_idx, a2]
        lr_2 = alpha_pos if pe_2 >= 0 else alpha_neg
        q_stage2_mf[state_idx, a2] += lr_2 * pe_2
        
        # Stage 1 Update
        pe_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        lr_1 = alpha_pos if pe_1 >= 0 else alpha_neg
        q_stage1_mf[a1] += lr_1 * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
