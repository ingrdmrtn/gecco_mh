import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated asymmetric learning rates.
    
    Hypothesis: Participants with higher OCI scores may be more sensitive to negative outcomes 
    (failure to retrieve gold), leading to stronger value updates after losses (0 coins).
    
    Parameters:
    learning_rate: [0,1] - Base learning rate for rewards.
    beta: [0,10] - Inverse temperature for softmax.
    w: [0,1] - Weight for Model-Based control (1=Pure MB, 0=Pure MF).
    oci_neg_lr_boost: [0,10] - How much OCI amplifies the learning rate for non-rewards.
    """
    learning_rate, beta, w, oci_neg_lr_boost = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate negative learning rate based on OCI
    # If reward is 0, learning rate is boosted by OCI factor
    lr_neg = learning_rate * (1.0 + current_oci * oci_neg_lr_boost)
    # Clip to ensure stability
    if lr_neg > 1.0: lr_neg = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-Free values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-Free values for stage 2 (Aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial]) # Planet
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Determine effective learning rate
        eff_lr = learning_rate if r > 0 else lr_neg
        
        # Stage 2 Update (TD)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += eff_lr * delta_stage2
        
        # Stage 1 Update (TD)
        # Using Q-learning update (max of next stage) for Stage 1 MF
        # Note: Standard two-step task MF update usually uses the value of the state reached
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += eff_lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Inverse Temperature (Beta).
    
    Hypothesis: OCI scores correlate with decision noise or anxiety. Higher OCI leads to 
    lower choice consistency (lower Beta), flattening the softmax curve.
    
    Parameters:
    learning_rate: [0,1] - Learning rate for Q-values.
    beta_max: [0,10] - Maximum inverse temperature (at OCI=0).
    w: [0,1] - Weight for Model-Based control.
    oci_noise_scale: [0,10] - Scaling factor for OCI-induced noise (reduces beta).
    """
    learning_rate, beta_max, w, oci_noise_scale = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective beta
    # Higher OCI -> Larger denominator -> Lower Beta -> More noise
    beta_effective = beta_max / (1.0 + current_oci * oci_noise_scale)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_effective * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        exp_q2 = np.exp(beta_effective * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Eligibility Trace (Lambda).
    
    Hypothesis: OCI affects the credit assignment depth in the Model-Free system.
    High OCI might lead to stronger 'habitual' caching where the final outcome 
    strongly reinforces the first stage choice directly (High Lambda), 
    bypassing the step-by-step structure.
    
    Parameters:
    learning_rate: [0,1] - Learning rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weight for Model-Based control.
    lambda_base: [0,1] - Baseline eligibility trace.
    oci_lambda_sens: [0, 5] - Sensitivity of lambda to OCI.
    """
    learning_rate, beta, w, lambda_base, oci_lambda_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective lambda
    lambda_eff = lambda_base + (current_oci * oci_lambda_sens)
    if lambda_eff > 1.0: lambda_eff = 1.0
    if lambda_eff < 0.0: lambda_eff = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates (TD-Lambda style) ---
        r = reward[trial]
        
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Stage 1 Prediction Error (using state value of stage 2)
        # In a standard TD(lambda) for this task, the Stage 1 value is updated 
        # by its own error PLUS lambda * stage 2 error.
        # Estimate of stage 2 value for stage 1 update:
        v_stage2 = q_stage2_mf[s_idx, a2] 
        delta_stage1 = v_stage2 - q_stage1_mf[a1]
        
        # Update Stage 1 MF
        # Q(s1,a1) += lr * (delta1 + lambda * delta2)
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_eff * delta_stage2)
        
        # Update Stage 2 MF
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
