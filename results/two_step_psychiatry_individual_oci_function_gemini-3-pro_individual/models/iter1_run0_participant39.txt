Here are three new cognitive models for the two-step decision task, incorporating the OCI score into different mechanisms of learning and decision-making.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid Model-Based/Model-Free with OCI-modulated Inverse Temperature.
    
    Hypothesis: OCI correlates with the rigidity or determinism of choice behavior.
    In this Hybrid model, we allow the OCI score to scale the inverse temperature (beta).
    A higher OCI score leads to a higher beta, resulting in more deterministic (exploitation-heavy) 
    choices and less exploration.
    
    Parameters:
    lr: [0, 1] Learning rate for the Model-Free component.
    w: [0, 1] Mixing weight (0 = Pure MF, 1 = Pure MB).
    beta_base: [0, 10] Baseline inverse temperature.
    beta_oci_slope: [0, 10] Sensitivity of beta to the OCI score.
    """
    lr, w, beta_base, beta_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta based on OCI
    # We ensure beta stays non-negative.
    beta = beta_base + (beta_oci_slope * oci_score)
    beta = np.maximum(0, beta)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: V_MB(S1) = Transition * max(Q_MF(S2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Weighted sum of MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        qs_stage2 = q_stage2_mf[state_idx]
        
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning ---
        current_a1 = int(action_1[trial])
        current_a2 = int(action_2[trial])
        
        # Update Stage 1 MF (SARSA-style update using Q(S2))
        delta_stage1 = q_stage2_mf[state_idx, current_a2] - q_stage1_mf[current_a1]
        q_stage1_mf[current_a1] += lr * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, current_a2]
        q_stage2_mf[state_idx, current_a2] += lr * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: TD(lambda) Learning with OCI-modulated Eligibility Trace.
    
    Hypothesis: OCI affects credit assignment depth. A higher OCI might correlate 
    with 'magical thinking' or attributing outcomes to distal causes.
    This is modeled by the eligibility trace parameter (lambda). 
    Higher OCI increases lambda, allowing the Stage 2 reward to more strongly 
    update the Stage 1 choice directly.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    lambda_base: [0, 1] Base eligibility trace decay.
    lambda_oci_slope: [0, 1] Modulation of lambda by OCI.
    """
    lr, beta, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective lambda and clip to valid range [0, 1]
    effective_lambda = lambda_base + (lambda_oci_slope * oci_score)
    effective_lambda = np.clip(effective_lambda, 0.0, 1.0)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        
        # --- Updates ---
        # TD Error 1: Difference between Value of S2 and Value of S1
        delta1 = q_stage2[state_idx, a2] - q_stage1[a1]
        
        # TD Error 2: Difference between Reward and Value of S2
        delta2 = reward[trial] - q_stage2[state_idx, a2]
        
        # Update S2 Value
        q_stage2[state_idx, a2] += lr * delta2
        
        # Update S1 Value:
        # Standard TD update (lr * delta1) + Eligibility Trace update (lr * lambda * delta2)
        # This passes the Stage 2 outcome back to Stage 1 proportional to lambda.
        q_stage1[a1] += lr * (delta1 + effective_lambda * delta2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Model-Free RL with OCI-modulated Negative Learning Rate.
    
    Hypothesis: OCI is associated with altered sensitivity to negative outcomes 
    (punishment or omission of reward).
    We split the learning rate into positive (lr_pos) and negative (lr_neg).
    The negative learning rate is specifically scaled by the OCI score, 
    allowing the model to learn differently from disappointments vs rewards.
    
    Parameters:
    beta: [0, 10] Inverse temperature.
    lr_pos: [0, 1] Learning rate for positive prediction errors.
    lr_neg_base: [0, 1] Base learning rate for negative prediction errors.
    lr_neg_oci_slope: [0, 1] Sensitivity of negative learning rate to OCI.
    """
    beta, lr_pos, lr_neg_base, lr_neg_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate
    lr_neg = lr_neg_base + (lr_neg_oci_slope * oci_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        
        # --- Updates ---
        
        # Stage 1 Update (using Q(S2) as target)
        prediction_error_1 = q_stage2[state_idx, a2] - q_stage1[a1]
        if prediction_error_1 >= 0:
            q_stage1[a1] += lr_pos * prediction_error_1
        else:
            q_stage1[a1] += lr_neg * prediction_error_1
            
        # Stage 2 Update (using Reward as target)
        prediction_error_2 = reward[trial] - q_stage2[state_idx, a2]
        if prediction_error_2 >= 0:
            q_stage2[state_idx, a2] += lr_pos * prediction_error_2
        else:
            q_stage2[state_idx, a2] += lr_neg * prediction_error_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
