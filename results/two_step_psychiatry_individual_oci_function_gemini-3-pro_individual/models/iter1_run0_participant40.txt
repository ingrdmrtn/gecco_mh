Here are three new cognitive models that explore different mechanisms by which Obsessive-Compulsive Inventory (OCI) scores might modulate decision-making in the two-step task.

### Model 1: Hybrid MB/MF with OCI-Modulated Learning Rate
This model posits that OCI scores relate to the *plasticity* of the learner. It tests whether individuals with higher or lower OCI symptoms update their value estimates more rapidly (high learning rate) or rigidly (low learning rate).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free learning with OCI-modulated Learning Rate.
    
    Hypothesis: OCI score correlates with the learning rate (plasticity). 
    Individuals with different OCI scores may update their value estimates at different speeds.
    
    Parameters:
    alpha_base: [0, 1] Base learning rate.
    alpha_oci_slope: [-1, 1] Change in learning rate per unit of OCI.
    beta: [0, 10] Inverse temperature (softness of choice).
    w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    """
    alpha_base, alpha_oci_slope, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective learning rate modulated by OCI
    learning_rate = alpha_base + (alpha_oci_slope * oci_score)
    learning_rate = np.clip(learning_rate, 0.0, 1.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # 1. Model-Based Value: Expected value of stage 2 state max Q
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value Calculation
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Action Selection (Softmax)
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        # Standard MF choice at stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning / Updates ---
        chosen_s1 = int(action_1[trial])
        chosen_s2 = int(action_2[trial])
        
        # Update Stage 1 MF Q-value (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, chosen_s2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_s2]
        q_stage2_mf[state_idx, chosen_s2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with OCI-Modulated Inverse Temperature
This model explores whether OCI affects the *exploration-exploitation* trade-off. It tests if OCI symptoms lead to more deterministic choice behavior (high beta) or more stochastic/exploratory behavior (low beta).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free with OCI-modulated Inverse Temperature (Beta).
    
    Hypothesis: OCI score affects the exploration-exploitation balance.
    Higher or lower OCI might lead to more deterministic (high beta) or random (low beta) choices.
    
    Parameters:
    learning_rate: [0, 1] Update rate.
    beta_base: [0, 10] Base inverse temperature.
    beta_oci_slope: [-10, 10] Modulation of beta by OCI.
    w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    """
    learning_rate, beta_base, beta_oci_slope, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta modulated by OCI
    beta = beta_base + (beta_oci_slope * oci_score)
    beta = np.clip(beta, 0.0, 10.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        chosen_s1 = int(action_1[trial])
        chosen_s2 = int(action_2[trial])
        
        delta_stage1 = q_stage2_mf[state_idx, chosen_s2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += learning_rate * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_s2]
        q_stage2_mf[state_idx, chosen_s2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with OCI-Modulated Stickiness
This model combines the Hybrid framework with a choice perseveration (stickiness) parameter. Unlike previous attempts that used a purely Model-Free baseline, this model tests if OCI drives a primitive repetition bias *on top of* a sophisticated Hybrid valuation system.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free with OCI-modulated Choice Stickiness.
    
    Hypothesis: OCI drives a tendency to repeat the previous choice (perseveration),
    which acts as a bias on top of the hybrid valuation system.
    
    Parameters:
    learning_rate: [0, 1] Update rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    stickiness_base: [0, 5] Base stickiness bonus.
    stickiness_oci_slope: [-5, 5] Modulation of stickiness by OCI.
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness
    eff_stickiness = stickiness_base + (stickiness_oci_slope * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness bonus to the previously chosen action
        if prev_action_1 != -1:
            q_net_stage1[prev_action_1] += eff_stickiness
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        chosen_s1 = int(action_1[trial])
        chosen_s2 = int(action_2[trial])
        
        delta_stage1 = q_stage2_mf[state_idx, chosen_s2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += learning_rate * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_s2]
        q_stage2_mf[state_idx, chosen_s2] += learning_rate * delta_stage2
        
        prev_action_1 = chosen_s1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
