Here are three new cognitive models for the two-step task.

### Model 1: Eligibility Trace Modulation
This model hypothesizes that OCI traits influence the **Eligibility Trace ($\lambda$)**. The eligibility trace controls how much the outcome at the second stage (reward) reinforces the choice made at the first stage. High OCI (compulsivity) might be associated with a stronger link between distal outcomes and initial choices (higher $\lambda$), or conversely, a disconnect (lower $\lambda$).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the Eligibility Trace (lambda).
    
    Hypothesis: The strength of the link between the final reward and the 
    Stage 1 choice (credit assignment) is modulated by OCI.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] (Weighting of Model-Based vs Model-Free)
    lambda_base: [0, 1] (Base eligibility trace)
    lambda_oci_mod: [-1, 1] (Modulation of lambda by OCI)
    """
    learning_rate, beta, w, lambda_base, lambda_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    # Calculate effective eligibility trace lambda
    eff_lambda = lambda_base + (lambda_oci_mod * oci_score)
    eff_lambda = np.clip(eff_lambda, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Prediction Error Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Prediction Error Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Update Stage 1 MF using eligibility trace
        # The Stage 1 value is updated by its own PE plus the weighted Stage 2 PE
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + eff_lambda * delta_stage2)
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Stage-Specific Exploration Modulation
This model hypothesizes that OCI affects the **Inverse Temperature ($\beta$) specifically at Stage 2**, independent of Stage 1. While Stage 1 involves planning and uncertainty about transitions, Stage 2 is a direct reward choice. High OCI might drive more rigid/deterministic behavior (higher $\beta$) specifically when facing the immediate reward source.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI modulates the Inverse Temperature (beta) specifically 
    at the second stage, allowing for differential exploration/exploitation 
    between planning (Stage 1) and harvesting (Stage 2).
    
    Bounds:
    learning_rate: [0, 1]
    w: [0, 1]
    beta_1: [0, 10] (Fixed temperature for Stage 1)
    beta_2_base: [0, 10] (Base temperature for Stage 2)
    beta_2_oci_mod: [-5, 5] (Modulation of Stage 2 beta by OCI)
    """
    learning_rate, w, beta_1, beta_2_base, beta_2_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    # Calculate effective Beta for Stage 2
    eff_beta_2 = beta_2_base + (beta_2_oci_mod * oci_score)
    eff_beta_2 = np.clip(eff_beta_2, 0.0, 10.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use beta_1 for first choice
        exp_q1 = np.exp(beta_1 * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        # Use OCI-modulated eff_beta_2 for second choice
        exp_q2 = np.exp(eff_beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Value Decay (Forgetting) Modulation
This model introduces a passive **decay parameter** for unchosen options, which is modulated by OCI. This tests the hypothesis that OCI traits relate to the persistence of value representations (e.g., "sticky thoughts" or inability to update/forget old values).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with Value Decay (Forgetting) modulated by OCI.
    Unchosen options decay towards zero at a rate determined by OCI.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    decay_base: [0, 1] (Base decay rate per trial)
    decay_oci_mod: [-1, 1] (Modulation of decay by OCI)
    """
    learning_rate, beta, w, decay_base, decay_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective decay rate
    eff_decay = decay_base + (decay_oci_mod * oci_score)
    eff_decay = np.clip(eff_decay, 0.0, 1.0)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        
        # 1. Decay mechanism for unchosen options
        # Stage 1 unchosen option decays
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - eff_decay)
        
        # Stage 2 unchosen option (in the current state) decays
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - eff_decay)
        
        # 2. Standard Reinforcement for chosen options
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
