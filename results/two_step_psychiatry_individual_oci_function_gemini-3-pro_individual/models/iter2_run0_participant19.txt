Here are three new cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-Learning with OCI-modulated Eligibility Traces.
    
    Hypothesis: OCI scores modulate the eligibility trace parameter (lambda). 
    This parameter controls whether the agent learns Stage 1 values primarily via 
    Temporal Difference (TD) from the next state (low lambda) or via direct reinforcement 
    from the final reward (high lambda, Monte Carlo-like).
    
    Parameters:
    lr: [0, 1] Learning rate for value updates.
    beta: [0, 10] Inverse temperature for softmax choice.
    lambda_base: [0, 1] Baseline eligibility trace decay.
    oci_lambda_mod: [-2, 2] OCI modulation of lambda.
    """
    lr, beta, lambda_base, oci_lambda_mod = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective lambda
    eff_lambda = lambda_base + (oci_val * oci_lambda_mod)
    eff_lambda = np.clip(eff_lambda, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Learning
        r = reward[trial]
        
        # Prediction error for stage 2 (driven by reward)
        delta2 = r - q_stage2[state_idx, a2]
        
        # Prediction error for stage 1 (driven by stage 2 value)
        delta1 = q_stage2[state_idx, a2] - q_stage1[a1]
        
        # Update Stage 2
        q_stage2[state_idx, a2] += lr * delta2
        
        # Update Stage 1: Standard TD update + Eligibility Trace portion (lambda * delta2)
        q_stage1[a1] += lr * (delta1 + eff_lambda * delta2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-Learning with Asymmetric Learning Rates modulated by OCI.
    
    Hypothesis: OCI scores correlate with a bias in learning from negative prediction errors 
    (disappointment/punishment) versus positive ones. This model uses a separate learning
    rate for negative updates that is modulated by OCI.
    
    Parameters:
    beta: [0, 10] Inverse temperature.
    lr_pos: [0, 1] Learning rate for positive prediction errors.
    lr_neg_base: [0, 1] Baseline learning rate for negative prediction errors.
    oci_neg_mod: [-1, 1] Modulation of negative learning rate by OCI.
    """
    beta, lr_pos, lr_neg_base, oci_neg_mod = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective negative learning rate
    lr_neg = lr_neg_base + (oci_val * oci_neg_mod)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # Stage 1
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # Stage 2
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # Update Stage 2
        delta2 = r - q_stage2[state_idx, a2]
        lr2 = lr_pos if delta2 >= 0 else lr_neg
        q_stage2[state_idx, a2] += lr2 * delta2
        
        # Update Stage 1
        # Calculate delta based on updated stage 2 value (SARSA-like consistency)
        delta1 = q_stage2[state_idx, a2] - q_stage1[a1]
        lr1 = lr_pos if delta1 >= 0 else lr_neg
        q_stage1[a1] += lr1 * delta1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Transition Beliefs.
    
    Hypothesis: OCI affects the Model-Based component's belief about the transition matrix.
    High OCI might reflect rigid beliefs (transitions perceived as deterministic) or 
    uncertain beliefs, distorting the Model-Based value calculation.
    
    Parameters:
    lr: [0, 1] Learning rate for MF values.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (0=MF, 1=MB).
    oci_trans_mod: [-1, 1] Modulation of the common transition probability (base 0.7).
    """
    lr, beta, w, oci_trans_mod = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Base transition probability is 0.7. OCI modulates this belief.
    p_common = 0.7 + (oci_val * oci_trans_mod)
    p_common = np.clip(p_common, 0.0, 1.0)
    p_rare = 1.0 - p_common
    
    # Perceived transition matrix
    # T[0,0] is A->X (Common), T[0,1] is A->Y (Rare)
    # T[1,0] is B->X (Rare), T[1,1] is B->Y (Common)
    trans_matrix = np.array([[p_common, p_rare], [p_rare, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)
    q_mf2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # Model-Based Values
        # Assume MB uses MF stage 2 values as proxy for reward expectation
        max_q_stage2 = np.max(q_mf2, axis=1) 
        q_mb1 = trans_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # Stage 2 Choice (Pure MF usually)
        exp_q2 = np.exp(beta * q_mf2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # Updates (MF only, MB is static structure)
        delta2 = r - q_mf2[state_idx, a2]
        q_mf2[state_idx, a2] += lr * delta2
        
        # Update MF Stage 1
        delta1 = q_mf2[state_idx, a2] - q_mf1[a1]
        q_mf1[a1] += lr * delta1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
