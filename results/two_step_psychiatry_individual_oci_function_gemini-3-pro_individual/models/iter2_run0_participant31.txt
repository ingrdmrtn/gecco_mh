Here are three new cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid Model with OCI-Modulated Inverse Temperature.

    Hypothesis: OCI scores influence the exploration-exploitation trade-off (decision noise).
    While the agent uses a standard hybrid (MB+MF) learning strategy, their ability to 
    deterministically select the best option is modulated by their OCI score.
    High OCI might lead to more rigid/deterministic behavior (higher beta) or 
    anxiety-driven noise (lower beta).

    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - w: [0, 1] Weight for Model-Based control (1=Pure MB, 0=Pure MF).
    - beta_base: [0, 10] Baseline inverse temperature.
    - beta_oci_slope: [-5, 5] Modulation of beta by OCI.
    """
    learning_rate, w, beta_base, beta_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective beta (ensure non-negative)
    beta = beta_base + (beta_oci_slope * oci_score)
    beta = max(0.0, beta)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial]) 

        # --- Stage 2 Policy ---
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        chosen_s1 = int(action_1[trial])
        chosen_s2 = int(action_2[trial])
        
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[state_idx, chosen_s2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_s2]
        q_stage2_mf[state_idx, chosen_s2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Dynamic Transition Learning with OCI Modulation.

    Hypothesis: The agent actively updates their internal model of spaceship-planet transitions,
    rather than assuming fixed probabilities. OCI modulates the learning rate of this 
    transition model (`lr_transition`). High OCI might correspond to hyper-vigilance 
    (fast updating of the world model) or rigidity (slow updating).

    Parameters:
    - lr_reward: [0, 1] Learning rate for reward values (Stage 2).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - lr_trans_base: [0, 1] Baseline learning rate for transitions.
    - lr_trans_oci: [-1, 1] Modulation of transition learning rate by OCI.
    """
    lr_reward, beta, w, lr_trans_base, lr_trans_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Effective transition learning rate (clamped [0,1])
    lr_trans = lr_trans_base + (lr_trans_oci * oci_score)
    lr_trans = np.clip(lr_trans, 0.0, 1.0)

    # Initialize transition matrix with prior knowledge
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]]) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Transition Update ---
        chosen_s1 = int(action_1[trial]) # 0 or 1 (Spaceship)
        actual_state = int(state[trial]) # 0 or 1 (Planet)
        
        # Create one-hot vector for the observed outcome state
        outcome_vector = np.zeros(2)
        outcome_vector[actual_state] = 1.0
        
        # Delta rule update for the transition probabilities of the chosen spaceship
        transition_matrix[chosen_s1] += lr_trans * (outcome_vector - transition_matrix[chosen_s1])

        # --- Stage 2 Policy ---
        q_s2 = q_stage2_mf[actual_state]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Value Updates ---
        chosen_s2 = int(action_2[trial])
        
        # MF Stage 1 Update
        delta_stage1 = q_stage2_mf[actual_state, chosen_s2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += lr_reward * delta_stage1
        
        # MF Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[actual_state, chosen_s2]
        q_stage2_mf[actual_state, chosen_s2] += lr_reward * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Model-Free with OCI-Enhanced Punishment Sensitivity.

    Hypothesis: OCI scores relate to the subjective impact of negative outcomes (0 coins).
    Rather than just neutral, a lack of reward is perceived as a punishment, 
    scaled by the OCI score. This creates "loss aversion" or "regret" where 
    0-reward outcomes drive Q-values below zero.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - stickiness: [0, 5] Tendency to repeat stage 1 choice.
    - punishment_sensitivity: [0, 5] How negative a 0-reward feels per unit OCI.
    """
    learning_rate, beta, stickiness, punishment_sensitivity = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective value of a "0" reward
    # If reward is 1, value is 1. If reward is 0, value is negative based on OCI
    neg_val = -1.0 * punishment_sensitivity * oci_score

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Pure MF with stickiness
        q_net = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = int(action_1[trial])
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        chosen_s1 = int(action_1[trial])
        chosen_s2 = int(action_2[trial])
        
        # Determine effective reward
        r_actual = reward[trial]
        if r_actual == 0:
            r_eff = neg_val
        else:
            r_eff = r_actual

        # MF Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, chosen_s2] - q_stage1_mf[chosen_s1]
        q_stage1_mf[chosen_s1] += learning_rate * delta_stage1
        
        # MF Stage 2 Update using Effective Reward
        delta_stage2 = r_eff - q_stage2_mf[state_idx, chosen_s2]
        q_stage2_mf[state_idx, chosen_s2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
