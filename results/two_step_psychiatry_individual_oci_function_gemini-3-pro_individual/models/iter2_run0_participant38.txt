Here are three new cognitive models exploring different mechanisms for how OCI scores might influence decision-making in the two-step task, specifically focusing on the balance between model-based/model-free control and learning rate modulation.

### Model 1: Hybrid Model with OCI-modulated Model-Based Weight
This model tests the hypothesis that higher compulsivity (OCI) is associated with a shift away from goal-directed (model-based) control towards habitual (model-free) control. The weighting parameter `w` determines the mix of these systems, and here `w` is a linear function of the OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid MB/MF with OCI-modulated Weighting.
    
    Hypothesis: OCI score modulates the trade-off between Model-Based (MB) and 
    Model-Free (MF) control. Higher OCI might lead to reduced MB control (lower w).
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature for softmax choice [0, 10]
    - w_base: Baseline weight for Model-Based control [0, 1]
    - w_oci_mod: Modulation of weight by OCI [-1, 1] (clipped to ensure w in [0,1])
    """
    learning_rate, beta, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate the mixing weight w based on OCI
    # We clip w to be between 0 and 1
    w = w_base + (w_oci_mod * current_oci)
    w = np.clip(w, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value Calculation
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net_s1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Policy
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 update (Standard Q-learning)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 update (TD(1) / SARSA-like update for MF)
        # Note: In standard 2-step models, stage 1 MF is often updated via the stage 2 value
        # or the reward directly. Here we use the standard TD chain.
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # We update using both the prediction error at stage 1 and the prediction error at stage 2
        # This is equivalent to updating Q1 towards the final reward
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * delta_stage2 

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Pure Model-Free with OCI-modulated Learning Rates
This model assumes the participant operates purely in a model-free manner (ignoring transition probabilities) but that their OCI score influences how quickly they update their beliefs. Specifically, it tests if OCI affects the learning rate `alpha`.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Pure Model-Free with OCI-modulated Learning Rate.
    
    Hypothesis: OCI affects the plasticity of value updates. 
    High OCI might be associated with rigid beliefs (low alpha) or hyper-reactivity (high alpha).
    
    Parameters:
    - alpha_base: Base learning rate [0, 1]
    - alpha_oci_mod: Modulation of learning rate by OCI [-1, 1]
    - beta: Inverse temperature [0, 10]
    - eligibility: Eligibility trace parameter for stage 1 update [0, 1]
    """
    alpha_base, alpha_oci_mod, beta, eligibility = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Modulate learning rate
    alpha = alpha_base + (alpha_oci_mod * current_oci)
    alpha = np.clip(alpha, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        qs_stage2 = q_stage2[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # Prediction errors
        # Stage 2 PE
        pe_2 = r - q_stage2[state_idx, a2]
        # Stage 1 PE (driven by Stage 2 Q-value)
        pe_1 = q_stage2[state_idx, a2] - q_stage1[a1]
        
        # Update Stage 2
        q_stage2[state_idx, a2] += alpha * pe_2
        
        # Update Stage 1
        # Using eligibility trace logic: Q1 updates from its own PE plus the subsequent PE scaled by lambda
        q_stage1[a1] += alpha * pe_1 + (alpha * eligibility * pe_2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with OCI-modulated Inverse Temperature (Beta)
This model posits that the balance between systems (`w`) is constant, but the *randomness* or *exploration* in choice is driven by OCI. Higher anxiety/compulsivity might lead to more deterministic choices (higher beta) or more erratic behavior (lower beta).

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Hybrid MB/MF with Fixed Weight but OCI-modulated Beta.
    
    Hypothesis: OCI scores impact the exploration-exploitation trade-off.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - w: Fixed weight for Model-Based control [0, 1]
    - beta_base: Base inverse temperature [0, 10]
    - beta_oci_mod: Modulation of beta by OCI [-5, 5]
    """
    learning_rate, w, beta_base, beta_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Modulate beta
    beta = beta_base + (beta_oci_mod * current_oci)
    # Ensure beta stays within reasonable positive bounds
    beta = np.clip(beta, 0.0, 10.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted mixture
        q_net_s1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        state_idx = int(state[trial])
        
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
