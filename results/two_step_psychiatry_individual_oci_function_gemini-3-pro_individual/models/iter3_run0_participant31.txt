Here are three new cognitive models that incorporate OCI scores into the decision-making process, following the provided template and constraints.

### Model 1: Model-Based Transition Belief Distortion
This model posits that OCI scores relate to a distortion in the internal model of the environment. Specifically, high OCI scores (associated with rigidity or black-and-white thinking) may lead participants to perceive the "common" transition as more deterministic than it actually is (e.g., believing it happens 90% of the time instead of 70%), which biases their Model-Based planning values.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Model-Based Transition Belief Distortion.
    
    Hypothesis: OCI scores correlate with a distorted belief about the transition probabilities 
    used in Model-Based planning. High OCI leads to an overestimation of the 
    reliability of the common transition (rigidity), affecting q_stage1_mb.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for Model-Free values.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting parameter (0=Pure MF, 1=Pure MB).
    belief_distortion: [0, 1] - How much OCI increases the perceived common transition probability.
                                p_common_perceived = 0.7 + (distortion * OCI).
    """
    learning_rate, beta, w, belief_distortion = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    # Calculate perceived transition probability based on OCI
    # Base is 0.7. Distortion adds to it, capped at 0.99 to prevent math errors.
    p_common = min(0.99, 0.7 + (belief_distortion * oci_score))
    
    # The participant's internal model (distorted)
    transition_matrix = np.array([[p_common, 1-p_common], [1-p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice (Hybrid MB/MF)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # policy for the second choice
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Updates
        # Stage 1 MF Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Stage 2 Perseveration
This model hypothesizes that obsessive-compulsive symptoms manifest as "stickiness" or perseveration specifically at the level of the final action (choosing the alien). While the planning stage (choosing a spaceship) might be goal-directed, the second stage choice is subject to a compulsive urge to repeat the last action taken in that state, with the strength of this urge scaled by the OCI score.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Stage 2 Perseveration.
    
    Hypothesis: OCI scores predict stickiness (perseveration) specifically in the second stage 
    (choosing the alien). This reflects compulsive repetition of the final action, 
    independent of the planning stage.
    
    Parameters:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] - MB/MF weight.
    stick_base: [0, 2] - Base stickiness bonus.
    stick_oci_slope: [0, 5] - OCI modulation of stickiness. 
                              Total stickiness = base + slope * OCI.
    """
    learning_rate, beta, w, stick_base, stick_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate stickiness bonus
    stickiness = stick_base + (stick_oci_slope * oci_score)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track last choice for each state (Planet X and Planet Y)
    # Initialize as -1 (no previous choice)
    last_choice_s2 = [-1, -1] 

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # policy for the second choice
        q_s2_vals = q_stage2_mf[state_idx].copy()
        
        # Apply stickiness if there was a previous choice in this specific state
        prev_choice = last_choice_s2[state_idx]
        if prev_choice != -1:
            q_s2_vals[prev_choice] += stickiness
            
        exp_q2 = np.exp(beta * q_s2_vals)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Update tracker
        last_choice_s2[state_idx] = int(action_2[trial])
  
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Stage-Specific Learning Rates
This model splits the learning process into two distinct rates: one for the high-level spaceship choice (Stage 1) and one for the concrete alien choice (Stage 2). It tests the hypothesis that OCI specifically modulates the learning rate of the first stage (planning/transition prediction errors), while the second stage (direct reward learning) remains independent.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Stage-Specific Learning Rates with OCI Modulation.
    
    Hypothesis: High OCI participants may update high-level values (Stage 1, Spacehips) 
    differently from concrete values (Stage 2, Aliens). 
    This model allows separate learning rates, where Stage 1 LR is modulated by OCI.
    
    Parameters:
    lr_stage2: [0, 1] - Learning rate for Stage 2 (Aliens).
    beta: [0, 10]
    w: [0, 1] - MB/MF weight.
    lr_stage1_base: [0, 1] - Base learning rate for Stage 1.
    lr_stage1_oci_mod: [-1, 1] - Modulation of Stage 1 LR by OCI. 
                                 lr1 = base + mod * OCI (clipped 0-1).
    """
    lr_stage2, beta, w, lr_stage1_base, lr_stage1_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Stage 1 Learning Rate
    lr_stage1 = lr_stage1_base + (lr_stage1_oci_mod * oci_score)
    lr_stage1 = np.clip(lr_stage1, 0.0, 1.0) # Ensure valid bounds
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # policy for the second choice
        q_s2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Updates using separate learning rates
        
        # Stage 1 update uses lr_stage1 (OCI modulated)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr_stage1 * delta_stage1
        
        # Stage 2 update uses lr_stage2 (fixed)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr_stage2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
