Here are three new cognitive models for the two-step task, incorporating the OCI score into different mechanisms:

### Cognitive Model 1: Hybrid Model with OCI-Modulated Stickiness
This model combines the standard Hybrid (Model-Based + Model-Free) architecture with a choice "stickiness" parameter. This tests the hypothesis that OCI symptoms correlate with a tendency to repeat previous choices (perseveration), while controlling for model-based planning capabilities.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free with OCI-modulated Choice Stickiness.
    
    This model computes a net Q-value as a weighted sum of Model-Based (MB) and 
    Model-Free (MF) values. It adds a "stickiness" bonus to the action chosen 
    in the previous trial. The magnitude of this stickiness is modulated by the OCI score.

    Parameters:
    - learning_rate: Rate of value updating [0, 1]
    - beta: Inverse temperature for softmax [0, 10]
    - w: Mixing weight for MB vs MF systems [0, 1] (0=Pure MF, 1=Pure MB)
    - stickiness_base: Base tendency to repeat the previous Stage 1 choice [-5, 5]
    - stickiness_oci_mod: Effect of OCI score on stickiness [-5, 5]
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective stickiness based on OCI
    eff_stickiness = stickiness_base + (stickiness_oci_mod * oci_score)
    
    # Fixed transition matrix (70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Expected value of next stage states given transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Weighted mix of MB and MF
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Stickiness to the previously chosen action
        q_net = q_hybrid.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += eff_stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # Update Stage 2 MF values (Direct Reward)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF values (TD(0) from Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Model-Free with OCI-Modulated Passive Decay
This model tests the hypothesis that OCI relates to memory retention or rumination. It implements a decay mechanism where unchosen options slowly lose their value (forgetting). The rate of this decay is modulated by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with OCI-modulated Passive Decay of Unchosen Options.
    
    Standard Q-learning for chosen options. Unchosen options decay towards 0 
    at a rate determined by the OCI score. This captures potential differences 
    in value retention or forgetting.

    Parameters:
    - learning_rate: Rate of value updating for chosen actions [0, 1]
    - beta: Inverse temperature [0, 10]
    - decay_base: Base decay rate for unchosen actions [0, 1]
    - decay_oci_mod: How OCI modifies the decay rate [-1, 1]
    """
    learning_rate, beta, decay_base, decay_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate decay rate bounded between 0 and 1
    decay_rate = decay_base + (decay_oci_mod * oci_score)
    decay_rate = np.clip(decay_rate, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # --- Updates ---
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Decay unchosen option in Stage 2
        unchosen_a2 = 1 - a2
        q_stage2_mf[s_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Decay unchosen option in Stage 1
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Hybrid with OCI-Modulated Stage 2 Beta
This model investigates if OCI specifically impacts decision noise (exploration/exploitation) at the second stage (direct reward collection), distinct from the planning stage. It allows for different inverse temperatures for Stage 1 and Stage 2, with OCI modulating only the second stage.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with Differential OCI Modulation on Stage 2 Inverse Temperature.
    
    Allows for different exploration/exploitation trade-offs in the planning stage (Stage 1)
    versus the harvesting stage (Stage 2). OCI modulates the precision (beta) 
    of choices specifically at Stage 2.

    Parameters:
    - learning_rate: [0, 1]
    - beta_1: Stage 1 inverse temperature (fixed) [0, 10]
    - beta_2_base: Stage 2 base inverse temperature [0, 10]
    - beta_2_oci_mod: OCI modulation for Stage 2 beta [-5, 5]
    - w: Mixing weight [0, 1]
    """
    learning_rate, beta_1, beta_2_base, beta_2_oci_mod, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Stage 2 beta, ensuring it stays non-negative
    beta_2 = beta_2_base + (beta_2_oci_mod * oci_score)
    beta_2 = max(0.0, beta_2)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy (Hybrid) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_hybrid)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy (MF) ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
