Here are the three proposed cognitive models.

### Model 1: Hybrid Learner with OCI-Modulated Eligibility Traces
This model hypothesizes that OCI symptoms affect "credit assignment"â€”specifically, how much the outcome at the second stage (the coin) reinforces the choice made at the first stage (the spaceship). A high eligibility trace ($\lambda$) implies a strong link between the final reward and the initial choice. This model tests if OCI scores correlate with a tendency to over-associate (or under-associate) distal rewards with proximal choices.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid MB/MF with OCI-modulated Eligibility Trace.
    
    This model assumes a Hybrid Model-Based/Model-Free architecture.
    The eligibility trace parameter (lambda), which controls how much the Stage 2 
    reward updates the Stage 1 value, is modulated by the OCI score.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature for softmax choice [0, 10]
    - w: Mixing weight (0=MF, 1=MB) [0, 1]
    - lambda_base: Base eligibility trace [0, 1]
    - lambda_oci_mod: Modulation of eligibility trace by OCI [-1, 1]
    """
    learning_rate, beta, w, lambda_base, lambda_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective lambda and clip to valid range [0, 1]
    lambda_eff = lambda_base + (lambda_oci_mod * current_oci)
    if lambda_eff < 0: lambda_eff = 0.0
    if lambda_eff > 1: lambda_eff = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # Stage 1 RPE (SARSA-style for MF)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 MF value using OCI-modulated eligibility trace
        # If lambda is 1, Stage 1 is fully updated by Stage 2 outcome (TD(1))
        # If lambda is 0, Stage 1 is only updated by Stage 1 prediction error (TD(0))
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_eff * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Learner with OCI-Modulated Perseveration
While the "best model so far" used perseveration in a purely Model-Free context, this model integrates perseveration into the full Hybrid (Model-Based + Model-Free) framework. It posits that OCI symptoms drive a "stickiness" or repetition compulsion that overrides both goal-directed (MB) and habit-based (MF) value calculations.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Hybrid MB/MF with OCI-modulated Perseveration.
    
    This model combines Model-Based and Model-Free values but adds a 
    perseveration bonus to the previously chosen action in Stage 1.
    The magnitude of this 'stickiness' is determined by the OCI score.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature for softmax choice [0, 10]
    - w: Mixing weight (0=MF, 1=MB) [0, 1]
    - pers_base: Base perseveration bonus [0, 5]
    - pers_oci_slope: Modulation of perseveration by OCI [-5, 5]
    """
    learning_rate, beta, w, pers_base, pers_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate perseveration bonus
    pers_bonus = pers_base + (pers_oci_slope * current_oci)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix MB and MF
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration bonus to the net Q-value of the previous action
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers_bonus
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Standard TD(1) update for the MF component
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with OCI-Modulated Stage 2 Learning Rate
This model proposes that OCI symptoms specifically impact the learning rate for the terminal reward stage (Stage 2), distinct from the general learning rate or the Stage 1 transition learning. It separates the learning process into two streams: one for the first decision (controlled by a fixed rate) and one for the alien/coin outcome (controlled by an OCI-modulated rate). This tests if OCI is associated with hyper-sensitivity or hypo-sensitivity to immediate reward feedback.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Hybrid MB/MF with Differential Stage 2 Learning Rate.
    
    This model separates the learning rate for Stage 1 (spaceships) and 
    Stage 2 (aliens). The Stage 2 learning rate is modulated by OCI,
    testing if OCI affects sensitivity to terminal reward outcomes specifically.
    
    Parameters:
    - lr_stage1: Learning rate for Stage 1 updates [0, 1]
    - lr_stage2_base: Base learning rate for Stage 2 updates [0, 1]
    - lr_stage2_oci_mod: Modulation of Stage 2 learning rate by OCI [-1, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight (0=MF, 1=MB) [0, 1]
    """
    lr_stage1, lr_stage2_base, lr_stage2_oci_mod, beta, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate Stage 2 specific learning rate
    lr_stage2 = lr_stage2_base + (lr_stage2_oci_mod * current_oci)
    if lr_stage2 < 0: lr_stage2 = 0.0
    if lr_stage2 > 1: lr_stage2 = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        # Update Stage 2 using the OCI-modulated rate
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2

        # Update Stage 1 using the fixed Stage 1 rate
        # Note: We use the *updated* q_stage2 value for the MB calculation next turn,
        # but for the MF update here, we use the values as they were.
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # We apply TD(1) logic but with potentially different learning rates for the two steps
        q_stage1_mf[a1] += lr_stage1 * delta_stage1 + lr_stage1 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
