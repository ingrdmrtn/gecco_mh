Here are three new cognitive models for the two-step task, exploring different mechanisms for how OCI might influence decision-making, specifically focusing on mixing weights, learning rate modulation, and eligibility traces.

### Model 1: OCI-Modulated Model-Based/Model-Free Mixing
This model tests the hypothesis that higher OCI scores lead to a stronger reliance on Model-Based (habit-free, goal-directed) control versus Model-Free (habitual) control. It parameterizes the mixing weight `w` as a logistic function of OCI.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner where the balance between Model-Based (MB) and Model-Free (MF) 
    control is modulated by OCI.
    
    Hypothesis: OCI scores influence the weighting (w) between goal-directed (MB) 
    and habitual (MF) systems. Higher OCI might reflect more rigid (MB) or more 
    habitual (MF) control depending on the fitted slope.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w_intercept: [0, 10] Intercept for the logistic function determining w.
    - w_slope: [-5, 5] Slope for OCI effect on w.
    """
    learning_rate, beta, w_intercept, w_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate mixing weight w using a sigmoid function to keep it in [0, 1]
    # w = 1 implies pure Model-Based; w = 0 implies pure Model-Free
    w_logit = w_intercept + w_slope * oci_score
    w = 1.0 / (1.0 + np.exp(-w_logit))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Q_MB(s1, a1) = T(s1, a1, s2) * max_a2 Q(s2, a2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = action_1[trial]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]

        # --- Value Updates ---
        # Stage 2 update (TD)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 update (TD)
        # Using the reward directly for simple TD(1)-like update often used in these tasks
        delta_stage1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Eligibility Trace (Lambda)
This model introduces an eligibility trace parameter `lambda` that controls how much the Stage 1 Model-Free values are updated based on the Stage 2 outcome. OCI modulates this `lambda` parameter.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner where the eligibility trace (lambda) is modulated by OCI.
    
    Hypothesis: OCI affects how credit is assigned to the first stage action based 
    on the second stage outcome. A higher lambda connects the final reward more 
    strongly to the initial choice (TD(1)), while a lower lambda relies on the 
    state prediction error (TD(0)).
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight for MB vs MF.
    - lambda_base: [0, 1] Base eligibility trace parameter.
    - lambda_oci_slope: [-1, 1] Effect of OCI on lambda.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate lambda, clamped between 0 and 1
    eligibility_lambda = lambda_base + lambda_oci_slope * oci_score
    eligibility_lambda = np.clip(eligibility_lambda, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]

        # --- Value Updates ---
        # Prediction error at stage 2 (reward - Q2)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Prediction error at stage 1 (Q2 - Q1)
        # Note: In standard TD(lambda), the update to stage 1 depends on both 
        # the immediate transition error (delta_stage1_immediate) and the 
        # subsequent error (delta_stage2) scaled by lambda.
        delta_stage1_immediate = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 2
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1
        # Q1 = Q1 + alpha * ( (Q2 - Q1) + lambda * (r - Q2) )
        # This simplifies to Q1 += alpha * (delta_stage1_immediate + lambda * delta_stage2)
        q_stage1_mf[a1] += learning_rate * (delta_stage1_immediate + eligibility_lambda * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Transition Learning Rate
This model hypothesizes that OCI affects how quickly participants update their internal model of the transition matrix (Model-Based learning), rather than the reward values.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner with dynamic transition matrix learning modulated by OCI.
    
    Hypothesis: Instead of a fixed 70/30 transition matrix, the participant learns 
    transition probabilities online. OCI modulates the learning rate for these 
    transitions (lr_trans). High OCI might correlate with hyper-vigilance to 
    structural changes (high lr_trans) or rigidity (low lr_trans).
    
    Parameters:
    - learning_rate_rew: [0, 1] Learning rate for reward values (Q-values).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight for MB vs MF.
    - lr_trans_base: [0, 1] Base learning rate for transition probabilities.
    - lr_trans_oci: [-1, 1] Modulation of transition learning rate by OCI.
    """
    learning_rate_rew, beta, w, lr_trans_base, lr_trans_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate transition learning rate
    lr_trans = lr_trans_base + lr_trans_oci * oci_score
    lr_trans = np.clip(lr_trans, 0.0, 1.0)

    # Initialize transition counts or probs. 
    # We initialize to the true probs [0.7, 0.3] but allow them to drift
    trans_probs = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # MB uses the dynamically learned transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = trans_probs @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        a1 = action_1[trial]
        state_idx = int(state[trial]) # Observed state (0 or 1)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = action_2[trial]
        r = reward[trial]

        # --- Value Updates ---
        # Standard TD updates for reward
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate_rew * delta_stage2
        
        delta_stage1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate_rew * delta_stage1
        
        # --- Transition Probability Update ---
        # State counting / delta rule for transitions
        # Update the row corresponding to action a1
        # Target is 1 for the observed state, 0 for the unobserved
        target = np.zeros(2)
        target[state_idx] = 1.0
        
        # trans_probs[a1] = trans_probs[a1] + lr_trans * (target - trans_probs[a1])
        # We ensure it sums to 1 automatically by the nature of the update if initialized normalized
        trans_probs[a1] += lr_trans * (target - trans_probs[a1])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
