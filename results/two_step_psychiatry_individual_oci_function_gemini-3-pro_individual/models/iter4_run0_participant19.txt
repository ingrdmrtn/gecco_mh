Here are three new cognitive models, expressed as Python functions, that incorporate the OCI-R score into the two-step decision-making task.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Reward-Dependent Stickiness (Win-Stay).
    
    Hypothesis: OCI score modulates the tendency to repeat a choice specifically 
    when it was previously rewarded (Win-Stay), rather than general perseveration 
    (repeating regardless of outcome). High OCI may drive compulsive repetition 
    of rewarded actions.
    
    Parameters:
    lr: [0, 1] Learning rate for value updates.
    beta: [0, 10] Inverse temperature (decision consistency).
    w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    ws_base: [0, 5] Baseline reward-dependent stickiness bonus.
    oci_mod: [0, 5] Modulation of stickiness by OCI score.
    """
    lr, beta, w, ws_base, oci_mod = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective stickiness for rewarded trials
    # This bonus is applied ONLY if the previous trial was a 'win' (reward > 0)
    k_win_stay = ws_base + (oci_mod * oci_val)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    prev_a1 = -1
    prev_r = 0

    for trial in range(n_trials):
        # Stage 1 Policy
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mixed Value
        q_mixed = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Reward-Dependent Stickiness
        if prev_a1 != -1 and prev_r > 0:
            q_mixed[prev_a1] += k_win_stay
            
        exp_q1 = np.exp(beta * q_mixed)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # Updates
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Stage 1 MF Update (using Stage 2 value estimate)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        prev_a1 = a1
        prev_r = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with Asymmetric Learning Rates modulated by OCI.
    
    Hypothesis: OCI individuals may show differential sensitivity to negative prediction errors 
    (losses/omissions) compared to positive ones. This model allows the learning rate 
    for negative prediction errors to be modulated by OCI, creating asymmetry.
    
    Parameters:
    lr_pos: [0, 1] Learning rate for positive prediction errors.
    lr_neg_base: [0, 1] Baseline learning rate for negative prediction errors.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (0=MF, 1=MB).
    oci_neg_mod: [-5, 5] Modulation of negative learning rate by OCI.
    """
    lr_pos, lr_neg_base, beta, w, oci_neg_mod = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective negative learning rate, bounded [0, 1]
    lr_neg = lr_neg_base + (oci_neg_mod * oci_val)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_mixed = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_mixed)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Updates
        # Stage 2 Update with Asymmetry
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        alpha2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[state_idx, a2] += alpha2 * delta_stage2
        
        # Stage 1 Update with Asymmetry
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        alpha1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[a1] += alpha1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with Stage-Specific Decision Noise modulated by OCI.
    
    Hypothesis: OCI affects decision consistency differently at the second stage 
    (proximal to reward/aliens) compared to the first stage. This model separates
    inverse temperature (beta) for the two stages, with OCI modulating Stage 2.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta1: [0, 10] Inverse temperature for Stage 1 (Spaceship choice).
    beta2_base: [0, 10] Baseline inverse temperature for Stage 2 (Alien choice).
    w: [0, 1] Mixing weight.
    oci_beta2_mod: [-5, 5] Modulation of Stage 2 beta by OCI.
    """
    lr, beta1, beta2_base, w, oci_beta2_mod = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective Stage 2 beta, bounded [0, 10]
    beta2 = beta2_base + (oci_beta2_mod * oci_val)
    beta2 = np.clip(beta2, 0.0, 10.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy (Uses beta1)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_mixed = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta1 * q_mixed)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # Stage 2 Policy (Uses beta2 modulated by OCI)
        exp_q2 = np.exp(beta2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Updates
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
