Here are three new cognitive models that explore different mechanisms by which OCI scores might influence decision-making in the two-step task, specifically focusing on how obsessive-compulsive traits might alter learning rates, eligibility traces, or the balance between model-based and model-free control.

### Model 1: OCI-Modulated Eligibility Trace (Lambda)
This model hypothesizes that OCI scores affect the "eligibility trace" parameter ($\lambda$), which controls how much the second-stage reward directly updates the first-stage value. High OCI might be associated with a more rigid or direct assignment of credit (higher $\lambda$), linking outcomes strongly to the initial choice, effectively bypassing the intermediate state structure.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-Learning with OCI-modulated Eligibility Trace (Lambda).
    
    Hypothesis: OCI affects how strongly the second-stage outcome updates the first-stage choice 
    directly (eligibility trace). High OCI might lead to stronger direct reinforcement (higher lambda).
    
    Parameters:
    learning_rate: [0, 1] Update rate (alpha).
    beta: [0, 10] Inverse temperature for softmax.
    lambda_base: [0, 1] Base eligibility trace parameter.
    lambda_oci_slope: [-1, 1] Modulation of lambda by OCI score.
    """
    learning_rate, beta, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate lambda based on OCI
    eligibility_lambda = lambda_base + (lambda_oci_slope * oci_score)
    eligibility_lambda = np.clip(eligibility_lambda, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Q(s1, a1)
    q_stage2_mf = np.zeros((2, 2)) # Q(s2, a2)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        # 1. Prediction error at stage 2 (outcome)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # 2. Prediction error at stage 1 (transition)
        # Standard SARSA-style update: value of state 2 (Q(s2,a2)) - value of state 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 value:
        # It gets updated by its own error, PLUS the eligibility trace of the stage 2 error
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * eligibility_lambda * delta_stage2
        
        # Update Stage 2 value
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Model-Based Weight ($w$) with Fixed Parameters
This model is the classic "Hybrid" model where behavior is a mix of Model-Based (planning) and Model-Free (habit) systems. Here, we test the hypothesis that OCI specifically modulates the mixing weight $w$. High OCI is often associated with habit formation (Model-Free), so we might expect higher OCI to decrease $w$ (less Model-Based control).

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model (MB/MF) with OCI-modulated Weight (w).
    
    Hypothesis: OCI shifts the balance between goal-directed (Model-Based) and 
    habitual (Model-Free) control.
    
    Parameters:
    learning_rate: [0, 1] Common learning rate for MF values.
    beta: [0, 10] Inverse temperature.
    w_base: [0, 1] Base weighting for Model-Based system (0=MF, 1=MB).
    w_oci_slope: [-1, 1] Effect of OCI on w.
    """
    learning_rate, beta, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate mixing weight w
    w = w_base + (w_oci_slope * oci_score)
    w = np.clip(w, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Used by both MF and MB (as terminal values)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value Calculation
        # V_MB(s1) = T(s1, s2) * max(Q(s2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        # Usually purely MF at stage 2 as it's the final step
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning (MF updates only) ---
        # MB system is static in transition knowledge here, only values update
        
        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 RPE (SARSA)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Note: Often Hybrid models also include the eligibility trace (lambda)
        # but to keep parameters <=5, we omit lambda here (effectively lambda=0 for MF part)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Punishment Sensitivity
This model investigates if OCI is related to an asymmetry in learning from positive versus negative outcomes. Obsessive-compulsive symptoms often involve anxiety and harm avoidance. This model allows the learning rate for negative rewards (punishment/loss) to differ based on the OCI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-Learning with OCI-modulated Punishment Learning Rate.
    
    Hypothesis: OCI is related to enhanced sensitivity to negative outcomes (punishment).
    This model splits learning rate into alpha_pos and alpha_neg, where alpha_neg depends on OCI.
    
    Parameters:
    alpha_pos: [0, 1] Learning rate for positive prediction errors (or rewards).
    alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    alpha_neg_oci_slope: [-1, 1] Modulation of negative learning rate by OCI.
    beta: [0, 10] Inverse temperature.
    stickiness: [0, 5] Choice perseveration.
    """
    alpha_pos, alpha_neg_base, alpha_neg_oci_slope, beta, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate punishment learning rate
    alpha_neg = alpha_neg_base + (alpha_neg_oci_slope * oci_score)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        q_net = q_stage1_mf.copy()
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        # We calculate RPE at stage 2 to determine if it's "good" or "bad" news
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Select learning rate based on sign of prediction error
        if delta_stage2 >= 0:
            lr = alpha_pos
        else:
            lr = alpha_neg
            
        # Update Stage 2
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Update Stage 1 (Direct reinforcement from Stage 2 outcome, effectively lambda=1)
        # We use the same lr logic for the full chain for simplicity in this variant
        q_stage1_mf[a1] += lr * delta_stage2
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
