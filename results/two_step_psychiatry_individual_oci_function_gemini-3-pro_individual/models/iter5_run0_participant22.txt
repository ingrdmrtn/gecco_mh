def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated choice stickiness.
    
    This model posits that participants use a mix of Model-Based (planning) and 
    Model-Free (habit) strategies, but their tendency to perseverate (repeat choices)
    is modulated by their OCI score.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature (exploration/exploitation).
    w: [0,1] - Weight of Model-Based control (1=Pure MB, 0=Pure MF).
    stick_oci_scale: [0,10] - Scaling factor for stickiness based on OCI.
                              Effective stickiness = stick_oci_scale * oci.
    """
    learning_rate, beta, w, stick_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness bonus
    stickiness = stick_oci_scale * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Stage 1 Policy
        # 1. Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value calculation
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # 3. Add Stickiness to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = int(action_1[trial])
        state_idx = int(state[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        # Stage 1 Update (Model-Free TD update)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated asymmetric learning rates.
    
    This model tests the hypothesis that OCI correlates with altered sensitivity 
    to negative prediction errors (outcomes worse than expected).
    
    Parameters:
    learning_rate: [0,1] - Base learning rate (used for positive prediction errors).
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF mixing weight.
    asym_oci_scale: [0,10] - Modulates learning rate for negative prediction errors.
                             lr_neg = lr_base * (1 + asym_oci_scale * oci).
                             (Bounded at 1.0).
    """
    learning_rate, beta, w, asym_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Define asymmetric learning rates
    lr_pos = learning_rate
    # Calculate negative LR: High OCI -> Higher learning rate on failure (hypersensitivity to error)
    lr_neg_raw = learning_rate * (1.0 + asym_oci_scale * oci_score)
    lr_neg = 1.0 if lr_neg_raw > 1.0 else lr_neg_raw
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        # Stage 1
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        eff_lr_s1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[int(action_1[trial])] += eff_lr_s1 * delta_stage1
        
        # Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        eff_lr_s2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[state_idx, int(action_2[trial])] += eff_lr_s2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated transition uncertainty.
    
    This model assumes that OCI correlates with "doubt" about the environmental structure.
    High OCI flattens the Model-Based transition matrix towards uniformity (50/50),
    representing uncertainty about where spaceships lead.
    
    Parameters:
    learning_rate: [0,1] - Value update rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF mixing weight.
    uncertainty_oci_scale: [0,5] - Determines how much OCI flattens the transition matrix.
                                   eff_uncertainty = uncertainty_oci_scale * oci.
                                   T_subj = (1-u)*T_true + u*T_flat.
    """
    learning_rate, beta, w, uncertainty_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate subjective transition matrix
    # Bound uncertainty to 1.0 (max uncertainty)
    uncertainty = uncertainty_oci_scale * oci_score
    if uncertainty > 1.0: uncertainty = 1.0
    
    t_true = np.array([[0.7, 0.3], [0.3, 0.7]])
    t_flat = np.array([[0.5, 0.5], [0.5, 0.5]])
    # Mix true knowledge with uncertainty
    t_subj = (1.0 - uncertainty) * t_true + uncertainty * t_flat
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Use subjective matrix for MB calculation
        q_stage1_mb = t_subj @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
