Here are three new cognitive models expressed as Python functions. These models explore different mechanistic roles for the OCI score, focusing on how obsessive-compulsive traits might influence the balance between Model-Based (MB) and Model-Free (MF) control, reward sensitivity, and perseveration.

### Model 1: OCI-Modulated Model-Based/Model-Free Weighting
This model tests the hypothesis that higher OCI scores relate to a deficit in goal-directed (model-based) control, leading to a stronger reliance on habitual (model-free) systems. It uses a mixing parameter `w` that is linearly modulated by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the weighting parameter 'w'
    is modulated by the OCI score.
    
    Hypothesis: Higher OCI scores shift the balance towards Model-Free (habitual) control 
    (lower w) and away from Model-Based control.

    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w_base: [0, 1] Baseline weighting for model-based control (0=pure MF, 1=pure MB).
    - w_oci_mod: [-1, 1] Modulation of 'w' by OCI score.
    """
    learning_rate, beta, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate w based on OCI, ensuring it stays within [0, 1]
    w = w_base + (current_oci * w_oci_mod)
    w = np.clip(w, 0.0, 1.0)
    
    # Transition matrix: A -> X (0.7), U -> Y (0.7)
    # State 0=X, State 1=Y. Action 0=A, Action 1=U.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 states (planets), 2 actions (aliens)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: T * max(Q2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 RPE (TD(0) update for Model-Free)
        # Note: The template implies a SARSA-like update using Q2 of chosen action
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Reward Sensitivity (Outcome Valuation)
This model posits that OCI traits affect how strongly participants value rewards versus non-rewards. Instead of modulating learning rate, OCI scales the effective magnitude of the reward signal `r` before it enters the prediction error calculation.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner where the effective reward magnitude is modulated by OCI.
    
    Hypothesis: Higher OCI scores might lead to hypersensitivity (or hyposensitivity)
    to outcomes, scaling the subjective utility of the reward.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - reward_sens_base: [0, 2] Baseline reward sensitivity (multiplier on r).
    - reward_sens_oci_slope: [-2, 2] Slope of OCI effect on reward sensitivity.
    """
    learning_rate, beta, reward_sens_base, reward_sens_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective reward sensitivity
    # We clip to ensure it doesn't become negative (inverting reward valence)
    reward_sensitivity = reward_sens_base + (current_oci * reward_sens_oci_slope)
    reward_sensitivity = np.clip(reward_sensitivity, 0.0, 5.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r_raw = reward[trial]
        
        # Modulate reward
        r_effective = r_raw * reward_sensitivity

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage2 = r_effective - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Choice Perseveration (Stickiness)
This model investigates whether OCI is associated with "stickiness" or choice perseverationâ€”a tendency to repeat the previous choice regardless of reward. Here, the stickiness parameter is a function of the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner with choice perseveration (stickiness),
    where the stickiness parameter is modulated by OCI.
    
    Hypothesis: Individuals with higher OCI scores may exhibit repetitive behaviors,
    modeled here as an increased tendency to repeat the previous Stage 1 choice.

    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weighting (fixed).
    - stick_base: [-5, 5] Baseline stickiness (tendency to repeat choice).
    - stick_oci_mod: [-5, 5] Modulation of stickiness by OCI.
    """
    learning_rate, beta, w, stick_base, stick_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate stickiness parameter
    stickiness = stick_base + (current_oci * stick_oci_mod)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    prev_choice = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the previously chosen action
        q_net_stick = q_net.copy()
        if prev_choice != -1:
            q_net_stick[prev_choice] += stickiness
        
        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update previous choice
        prev_choice = a1

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
