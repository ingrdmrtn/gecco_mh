Here are three new cognitive models for the two-step task, incorporating OCI scores into different computational mechanisms.

### Cognitive Model 1: Hybrid Learner with OCI-modulated Model-Based Weight and Fixed Stickiness
This model posits that OCI symptoms affect the balance between Goal-Directed (Model-Based) and Habitual (Model-Free) control, represented by the mixing weight `w`. Crucially, this model includes a fixed `stickiness` parameter to account for basic choice perseveration, ensuring that any effect found on `w` is not simply capturing motor repetition.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where OCI modulates the mixing weight 'w',
    controlled for fixed choice stickiness.
    
    Parameters:
    learning_rate: [0, 1] Value update rate.
    beta: [0, 10] Inverse temperature (softmax noise).
    stickiness: [0, 5] Constant choice perseveration bonus.
    w_base: [0, 1] Base mixing weight (0=MF, 1=MB).
    w_oci_mod: [-1, 1] Modulation of w by OCI score.
    """
    learning_rate, beta, stickiness, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate mixing weight w based on OCI
    w = w_base + (w_oci_mod * oci_score)
    w = np.clip(w, 0.0, 1.0)

    # Fixed transition matrix for the Model-Based component
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-Free Q-values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Q-values for Stage 2 (State x Action)
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value: V(State) = max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to previously chosen action
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness

        # Softmax Policy
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # SARSA / TD(0) updates for Model-Free values
        # Stage 1 MF update (using Stage 2 Q-value as proxy for reward)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF update (using actual reward)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Dynamic Transition Learning with OCI Modulation
This model assumes the participant does *not* trust the fixed transition probabilities (70/30) but learns them dynamically. OCI modulates the `transition_learning_rate`, testing the hypothesis that high-OCI individuals might be hyper-sensitive to "rare" transitions (updating their internal model too quickly) or rigid (updating too slowly), affecting their Model-Based valuations.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Based learner that dynamically learns the transition matrix.
    OCI modulates the learning rate for these transitions.
    
    Parameters:
    lr_reward: [0, 1] Learning rate for reward values (Q-values).
    lr_transition_base: [0, 1] Base learning rate for state transitions.
    lr_transition_oci_mod: [-1, 1] OCI modulation of transition learning rate.
    beta: [0, 10] Inverse temperature.
    """
    lr_reward, lr_transition_base, lr_transition_oci_mod, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate transition learning rate
    lr_trans = lr_transition_base + (lr_transition_oci_mod * oci_score)
    lr_trans = np.clip(lr_trans, 0.0, 1.0)
    
    # Initialize dynamic transition matrix (start with uniform or slight bias)
    # Rows: Action A, Action U. Cols: Planet X, Planet Y.
    # Start assuming 0.5 (max uncertainty) or slight prior. Using 0.5 for pure learning.
    est_transition_matrix = np.array([[0.5, 0.5], [0.5, 0.5]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # We only need Stage 2 values for the MB calculation
    q_stage2 = np.zeros((2, 2)) 

    for trial in range(n_trials):
        # --- Stage 1 Decision (Pure Model-Based using Dynamic Transitions) ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = est_transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # 1. Update Value Estimates (Stage 2)
        delta_stage2 = r - q_stage2[state_idx, a2]
        q_stage2[state_idx, a2] += lr_reward * delta_stage2
        
        # 2. Update Transition Matrix Estimates
        # If we took action a1 and landed in state_idx, increase prob of that link
        # and decrease prob of the other link.
        # Current prob of going to observed state given action a1:
        curr_prob = est_transition_matrix[a1, state_idx]
        # Update toward 1.0
        est_transition_matrix[a1, state_idx] += lr_trans * (1.0 - curr_prob)
        # The other state must sum to 1
        other_state = 1 - state_idx
        est_transition_matrix[a1, other_state] = 1.0 - est_transition_matrix[a1, state_idx]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Model-Free Learner with OCI-modulated Memory Decay
This model tests if OCI is related to the stability of value representations. It introduces a `decay` parameter. Unlike standard Q-learning where unchosen options remain static, here unchosen options decay toward zero. OCI modulates this decay rate. High OCI could imply "active forgetting" or instability of value representations over time.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with memory decay on unchosen options.
    OCI modulates the decay rate.
    
    Parameters:
    learning_rate: [0, 1] Update rate for chosen options.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Fixed mixing weight (allows hybrid context, though focus is on MF decay).
    decay_base: [0, 1] Base decay rate for unchosen options.
    decay_oci_mod: [-1, 1] Modulation of decay by OCI.
    """
    learning_rate, beta, w, decay_base, decay_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate decay rate
    decay = decay_base + (decay_oci_mod * oci_score)
    decay = np.clip(decay, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value calculation
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Standard Update for CHOSEN options
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Decay for UNCHOSEN options (Forgetting)
        # Stage 1 unchosen
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay)
        
        # Stage 2 unchosen (in the current state)
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay)
        
        # Note: Options in the unvisited state are usually not decayed in standard models,
        # but one could argue they decay too. Here we only decay the unchosen option 
        # in the visited state to represent 'alternative value' decay.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
