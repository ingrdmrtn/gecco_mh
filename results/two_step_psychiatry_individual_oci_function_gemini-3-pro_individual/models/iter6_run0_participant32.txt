Here are three new cognitive models for the two-step task, incorporating OCI scores into the decision-making process.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Subjective Transition Belief.
    
    Hypothesis: High OCI individuals perceive the environment's transition structure 
    as more deterministic (rigid) than it actually is. They overestimate the 
    probability of the 'common' transition in their Model-Based planning, believing
    rules are stricter than reality.
    
    The true transition matrix is [0.7, 0.3; 0.3, 0.7].
    The subjective matrix becomes [p, 1-p; 1-p, p] where p = 0.7 + (oci * rigidity).
    
    Parameters:
    learning_rate: [0, 1] - Model-Free learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Weighting between MB (1) and MF (0).
    belief_rigidity: [0, 1] - How much OCI increases the perceived common transition probability.
                              Scaled such that max OCI pushes p towards 1.0.
    """
    learning_rate, beta, w, belief_rigidity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Subjective Transition Matrix Calculation
    # Base common prob is 0.7. We allow OCI to push this belief towards 1.0 (determinism).
    p_common = 0.7 + (belief_rigidity * current_oci * 0.29) 
    p_common = np.clip(p_common, 0.0, 0.99)
    
    # Subjective matrix used for MB calculations
    trans_matrix = np.array([[p_common, 1-p_common], [1-p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value calculation using subjective matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = trans_matrix @ max_q_stage2
        
        # Net Value (Hybrid)
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # MF Stage 1 Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # MF Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Reward-Dependent Stickiness.
    
    Hypothesis: High OCI individuals exhibit 'Win-Stay' compulsivity.
    They are more likely to repeat the first-stage choice if it was previously rewarded,
    compared to low OCI individuals. This is modeled as a stickiness bonus 
    conditional on the previous reward, distinct from general perseveration.
    
    Parameters:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    stick_base: [0, 5] - Baseline reward-dependent stickiness.
    oci_stick_mod: [0, 5] - Additional reward-dependent stickiness scaling with OCI.
    """
    learning_rate, beta, w, stick_base, oci_stick_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Effective stickiness applied only after a reward
    eff_rew_stick = stick_base + (current_oci * oci_stick_mod)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_a1 = -1
    last_reward = 0
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Reward-Dependent Stickiness
        if last_a1 != -1 and last_reward == 1:
            q_net_s1[last_a1] += eff_rew_stick
            
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_a1 = action_1[trial]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        last_reward = r
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Stage 1 Learning Rate.
    
    Hypothesis: OCI modulates the learning rate specifically for the first stage 
    (transition choice), reflecting altered sensitivity to structural prediction errors 
    or anxiety about the path, while Stage 2 (reward) learning remains at baseline.
    
    Parameters:
    lr_base: [0, 1] - Baseline learning rate (used for Stage 2).
    beta: [0, 10]
    w: [0, 1]
    oci_s1_boost: [0, 2] - OCI-dependent boost to Stage 1 learning rate.
    """
    lr_base, beta, w, oci_s1_boost = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Stage 1 specific learning rate
    lr_stage1 = lr_base + (current_oci * oci_s1_boost)
    lr_stage1 = np.clip(lr_stage1, 0.0, 1.0)
    
    lr_stage2 = lr_base
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # Update Stage 1 with specific LR
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Update Stage 2 with base LR
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
