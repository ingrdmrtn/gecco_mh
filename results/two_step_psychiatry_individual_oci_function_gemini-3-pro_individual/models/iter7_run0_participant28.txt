### Model 1: Post-Loss Model-Based Arbitration
This model hypothesizes that OCI symptoms specifically affect how participants react to **negative outcomes** (losses). While healthy participants might maintain a balance of Model-Based (MB) and Model-Free (MF) strategies, this model proposes that high OCI scores trigger a "reactive" shift towards Model-Based control (over-thinking/checking) immediately following a loss, attempting to cognitively "fix" the error.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Post-Loss Model-Based Arbitration.
    
    Hypothesis: OCI modulates the balance between Model-Based (MB) and Model-Free (MF) 
    control specifically after a loss (0 reward). High OCI leads to increased 
    MB weighting (w) following a failure, reflecting reactive over-analysis.
    
    Parameters:
    learning_rate: [0,1] - Value updating rate.
    beta: [0,10] - Inverse temperature (choice consistency).
    w_base: [0,1] - Baseline mixing weight (0=MF, 1=MB) after a win.
    oci_loss_w_boost: [0,1] - Additional MB weight added after a loss, scaled by OCI.
    """
    learning_rate, beta, w_base, oci_loss_w_boost = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) # MF values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # MF values for stage 2 (Aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Track previous reward to determine current 'w'. Initialize with a 'win' (1) assumption or neutral.
    prev_reward = 1.0 

    for trial in range(n_trials):
        
        # --- Stage 1: Spaceship Choice ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Determine mixing weight w based on previous outcome and OCI
        if prev_reward == 0:
            # After a loss, OCI increases MB reliance
            w_current = w_base + (current_oci * oci_loss_w_boost)
        else:
            # After a win, use baseline
            w_current = w_base
            
        # Clip w to [0, 1]
        if w_current > 1.0: w_current = 1.0
        
        # Mix MB and MF
        q_net_stage1 = w_current * q_stage1_mb + (1 - w_current) * q_stage1_mf
        
        # Softmax policy
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2: Alien Choice ---
        
        s_idx = int(state[trial])
        
        # Simple MF policy for stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        
        r = reward[trial]
        prev_reward = r # Store for next trial's arbitration
        
        # Stage 2 update (TD Error)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 update (TD Error using Stage 2 value)
        # Note: Standard hybrid models update MF Stage 1 using the value of the state reached
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Stage 1 Transfer Deficit
This model hypothesizes that OCI symptoms interfere with the **back-propagation of value** from the second stage (Aliens) to the first stage (Spaceships). While the participant may learn the immediate rewards of Aliens normally (Stage 2 learning), high OCI causes a "transfer deficit," effectively reducing the learning rate specifically for the first-stage (complex) choice.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage 1 Transfer Deficit Model.
    
    Hypothesis: OCI acts as a bottleneck for higher-order learning. While immediate 
    alien values (Stage 2) are learned at a base rate, the transfer of this value 
    back to the spaceship choice (Stage 1) is dampened by OCI.
    
    Parameters:
    lr_stage2: [0,1] - Base learning rate for Stage 2 (Alien outcome).
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    transfer_ratio_base: [0,1] - Base ratio of Stage 1 LR to Stage 2 LR.
    oci_transfer_damp: [0,1] - Amount OCI reduces the Stage 1 learning rate ratio.
    """
    lr_stage2, beta, w, transfer_ratio_base, oci_transfer_damp = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Stage 1 Learning Rate
    # It is a fraction of Stage 2 LR, reduced by OCI
    ratio = transfer_ratio_base - (current_oci * oci_transfer_damp)
    if ratio < 0: ratio = 0.0
    
    lr_stage1 = lr_stage2 * ratio
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Update Stage 2 with base rate
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_stage2 * delta_stage2
        
        # Update Stage 1 with OCI-modulated rate
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Ambiguity Aversion (UCB Penalty)
This model integrates an **Upper Confidence Bound (UCB)** style mechanism, but inverted to represent ambiguity aversion. It hypothesizes that high OCI is associated with intolerance of uncertainty. Therefore, options that have been chosen fewer times (high uncertainty) receive a value penalty, discouraging exploration and promoting "safe", familiar choices.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Ambiguity Aversion.
    
    Hypothesis: OCI correlates with intolerance of uncertainty. The model tracks 
    how often Stage 1 actions (spaceships) are chosen. A penalty is applied to 
    actions with low counts (high uncertainty). OCI scales this penalty.
    
    Parameters:
    learning_rate: [0,1] - Value learning rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    aversion_base: [0,1] - Baseline penalty for uncertainty.
    oci_aversion_sens: [0,10] - Sensitivity of uncertainty penalty to OCI.
    """
    learning_rate, beta, w, aversion_base, oci_aversion_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track choice counts for Stage 1 to calculate uncertainty
    action_counts = np.zeros(2) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Uncertainty Penalty (Ambiguity Aversion)
        # 1 / sqrt(count + 1) is large when count is low.
        uncertainty = 1.0 / np.sqrt(action_counts + 1)
        penalty_strength = aversion_base + (current_oci * oci_aversion_sens)
        
        # Apply penalty to the Net Q-values
        # Q_net = w*MB + (1-w)*MF - Penalty
        q_net_stage1 = (w * q_stage1_mb + (1 - w) * q_stage1_mf) - (penalty_strength * uncertainty)
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update count for chosen action
        action_counts[a1] += 1
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
