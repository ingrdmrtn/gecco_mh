Here are three new cognitive models, expressed as Python functions, that incorporate OCI scores to explain individual variability in the two-step task.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Planning Pessimism.
    
    This model assumes OCI modulates the 'optimism' of the Model-Based planner.
    Standard MB planning assumes the agent will take the best action at Stage 2 
    (maximizing Q). This model proposes that high-OCI individuals calculate 
    Stage 1 values based on a pessimistic mixture of the best and worst possible 
    Stage 2 outcomes, reflecting anxiety or risk aversion in planning.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for MF values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting between MB (1) and MF (0) systems.
    - oci_pessimism: [0, 5] Scales OCI to determine the weight of the 'worst' outcome in planning.
    """
    learning_rate, beta, w, oci_pessimism = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    # Calculate pessimism weight (rho) bounded [0, 1]
    # If OCI is high, rho is high -> more weight on min(Q)
    rho = min(1.0, current_oci * oci_pessimism)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        # MB Valuation: Mix of Max (Optimistic) and Min (Pessimistic)
        max_q2 = np.max(q_stage2_mf, axis=1)
        min_q2 = np.min(q_stage2_mf, axis=1)
        
        # Effective value of state 2 for planning
        val_stage2 = (1 - rho) * max_q2 + rho * min_q2
        
        q_stage1_mb = transition_matrix @ val_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        # Stage 2 MF update (standard)
        pe_2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * pe_2
        
        # Stage 1 MF update (TD(0)) using Stage 2 value
        pe_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Reward-Reactive Beta (Exploration Instability).
    
    This model assumes that OCI modulates how the agent's exploration (inverse temperature beta) 
    reacts to the previous outcome. High OCI participants may become significantly more 
    exploratory/random (lower beta) after a loss (0 reward) or more rigid (higher beta) 
    after a win, reflecting intolerance of uncertainty or emotional reactivity.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_base: [0, 10] Baseline inverse temperature.
    - w: [0, 1] MB/MF weight.
    - oci_reactivity: [0, 5] Scaling factor for OCI-driven beta modulation based on prior reward.
    """
    learning_rate, beta_base, w, oci_reactivity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_reward = 0.5 # Neutral initialization

    for trial in range(n_trials):
        # Calculate dynamic beta
        # Shift is -0.5 (loss) to +0.5 (win) -> scaled by OCI and param
        # If oci_reactivity > 0: Win -> Beta Up (Rigid), Loss -> Beta Down (Noisy)
        beta_shift = current_oci * oci_reactivity * (prev_reward - 0.5) * 2
        current_beta = max(0.0, beta_base + beta_shift)

        # Stage 1 Policy
        max_q2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        pe_2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * pe_2
        
        pe_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1
        
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Transition-Gated Model-Free Learning.
    
    This model assumes OCI modulates how the Model-Free system learns from 'Rare' transitions.
    High OCI individuals may treat rare transitions as anomalies or noise, effectively 
    dampening the Model-Free update (TD error propagation) for Stage 1 when a rare 
    transition occurs. This creates a specific 'transition sensitivity' within the MF system.
    
    Parameters:
    - learning_rate: [0, 1] Baseline learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - oci_rare_damp: [0, 5] Strength of learning suppression on rare transitions for high OCI.
    """
    learning_rate, beta, w, oci_rare_damp = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        # Stage 2 update is standard
        pe_2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * pe_2
        
        # Stage 1 Update: Gated by transition type
        # Common: a1==0 -> s0, a1==1 -> s1. Rare: a1!=s
        is_rare = (a1 != state_idx)
        
        if is_rare:
            # Dampen LR based on OCI
            # If oci_rare_damp is high, lr_eff approaches 0 for rare trials
            damp_factor = 1.0 / (1.0 + current_oci * oci_rare_damp)
            lr_eff = learning_rate * damp_factor
        else:
            lr_eff = learning_rate
            
        pe_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_eff * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
