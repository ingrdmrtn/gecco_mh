Here are three new cognitive models implemented as Python functions, incorporating OCI scores to explain individual variability in the two-step task.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Deterministic Belief Distortion.
    
    This model hypothesizes that individuals with higher OCI scores tend towards 
    "black-and-white" thinking or an overestimation of control. In the Model-Based 
    planning step, they distort the transition matrix, perceiving the common transition 
    (0.7 probability) as more deterministic (closer to 1.0) than it actually is.
    
    This results in a stronger Model-Based preference for the "correct" spaceship 
    when the values at the second stage are distinct, as the agent underestimates 
    the risk of the rare transition.

    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) systems.
    - belief_sharpness: [0, 1] How much OCI sharpens the transition belief. 
      (0 = accurate belief, 1 = maximum distortion towards determinism).
    """
    learning_rate, beta, w, belief_sharpness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Standard transition probability
    base_prob = 0.7
    
    # Distort probability based on OCI
    # If belief_sharpness is high and OCI is high, prob approaches 1.0
    # We clip to 0.99 to prevent numerical instability with log(0) if we were doing log-prob
    distorted_prob = base_prob + (belief_sharpness * oci_score * (1.0 - base_prob))
    distorted_prob = min(distorted_prob, 0.99)
    
    # Construct the subjective transition matrix
    # Row 0: Action 0 (A) -> usually State 0 (X)
    # Row 1: Action 1 (U) -> usually State 1 (Y)
    transition_matrix = np.array([
        [distorted_prob, 1.0 - distorted_prob], 
        [1.0 - distorted_prob, distorted_prob]
    ])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value Calculation using distorted matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Action Selection
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        
        # --- Stage 2 Choice ---
        q_s2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Learning ---
        # Stage 1 Update (TD-0)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Rare Transition Dampening (Dynamic Reliability).
    
    This model hypothesizes that high OCI scores correlate with an intolerance for 
    unexpected structural changes (rare transitions). When a rare transition occurs, 
    the agent loses confidence in their internal model (Model-Based system).
    
    The weight `w` (controlling MB vs MF influence) is dynamically reduced on the 
    trial immediately following a rare transition. The magnitude of this reduction 
    is scaled by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w_base: [0, 1] Baseline Model-Based weight.
    - rare_dampening: [0, 1] Sensitivity to rare transitions. 
      Higher values cause `w` to drop closer to 0 (pure MF) after a rare event, 
      specifically for high OCI participants.
    """
    learning_rate, beta, w_base, rare_dampening = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Fixed transition matrix for MB calculation
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    prev_transition_rare = False

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        
        # Calculate dynamic w based on previous trial's surprise
        w_current = w_base
        if prev_transition_rare:
            # If OCI is high, the dampening effect is stronger
            dampening_factor = rare_dampening * oci_score
            # Ensure w doesn't go below 0
            w_current = max(0.0, w_base * (1.0 - dampening_factor))
        
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w_current * q_stage1_mb + (1 - w_current) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        
        # --- Stage 2 Choice ---
        q_s2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Learning ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # --- Update History ---
        # Check if current transition was rare
        # Common: 0->0, 1->1. Rare: 0->1, 1->0.
        is_rare = (a1 != s_idx)
        prev_transition_rare = is_rare

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Aversive Miss Learning.
    
    This model assumes that for individuals with high OCI, the absence of a reward 
    (0 coins) is not merely neutral but actively aversive (perceived as a failure 
    or punishment).
    
    Instead of updating values with a reward of 0, the model updates with a 
    negative "effective reward" scaled by the OCI score. This causes the agent 
    to unlearn unrewarded actions more aggressively, "running away" from options 
    that yield nothing.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting parameter.
    - miss_aversion: [0, 1] The magnitude of the penalty applied to 0-reward outcomes,
      scaled by OCI. (Effective Reward = -1 * miss_aversion * OCI when reward is 0).
    """
    learning_rate, beta, w, miss_aversion = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        
        # --- Stage 2 Choice ---
        q_s2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Calculate Effective Reward ---
        # If reward is 0, treat it as negative based on OCI
        if r == 0:
            r_effective = -1.0 * (miss_aversion * oci_score)
        else:
            r_effective = r

        # --- Learning ---
        # Note: Standard TD updates propagate the Stage 2 value back to Stage 1.
        # If Stage 2 values become negative due to r_effective, Stage 1 MF values
        # will also decrease, reflecting the aversion.
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r_effective - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
