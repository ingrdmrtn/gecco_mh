Here are three new cognitive models that introduce OCI-dependent mechanisms not fully explored in the previous iterations, specifically focusing on how obsessive-compulsive traits might influence eligibility traces, the weighting of model-based vs. model-free control, and specific sensitivities to punishment/reward in the context of habit formation.

### Model 1: OCI-Modulated Eligibility Trace (Lambda)
This model hypothesizes that high OCI scores (associated with compulsivity and habit formation) might lead to stronger "eligibility traces" ($\lambda$). A higher $\lambda$ means the reward at the second stage reinforces the first-stage choice more strongly, effectively blurring the distinction between the two stages and making the behavior more "habit-like" or model-free in its temporal structure.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-Learning with OCI-modulated Eligibility Trace (Lambda).
    
    Hypothesis: OCI affects how strongly the second-stage outcome reinforces the first-stage choice directly.
    Higher OCI might lead to higher lambda (more habit-like "caching" of long-term value).

    Parameters:
    learning_rate: [0, 1] Learning rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    lambda_base: [0, 1] Base eligibility trace decay parameter.
    lambda_oci_slope: [-1, 1] Modulation of lambda by OCI score.
    """
    learning_rate, beta, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective lambda based on OCI
    # We use a sigmoid-like transform or simple clipping to keep it in [0, 1]
    lambda_val = lambda_base + (lambda_oci_slope * oci_score)
    lambda_val = np.clip(lambda_val, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)      # Q(s1, a1)
    q_stage2 = np.zeros((2, 2)) # Q(s2, a2)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        # Prediction error at stage 2
        delta_stage2 = r - q_stage2[state_idx, a2]
        
        # Prediction error at stage 1 (SARSA-style temporal difference)
        delta_stage1 = q_stage2[state_idx, a2] - q_stage1[a1]

        # Update Stage 2 Q-values
        q_stage2[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values
        # Standard TD update: q1 += lr * delta1
        # Eligibility trace update: q1 += lr * lambda * delta2
        q_stage1[a1] += learning_rate * (delta_stage1 + lambda_val * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Mixing Weight (w) with Inverse Modulation
Previous attempts explored `w` (model-based weight) modulation. This model refines that by testing if OCI specifically *impairs* the ability to use the model-based system (`w` decreases as OCI increases) while also including a specific stickiness parameter that is constant. This isolates the trade-off between goal-directed planning and habit without confounding it with perseveration.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model (MB/MF) where OCI modulates the mixing weight 'w'.
    
    Hypothesis: High OCI is associated with reduced Model-Based control (lower w).
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    stickiness: [0, 5] Choice perseveration.
    w_base: [0, 1] Base weight for Model-Based system.
    w_oci_mod: [-1, 1] How OCI changes w. Negative value implies OCI reduces MB control.
    """
    learning_rate, beta, stickiness, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate w (mixing weight)
    w = w_base + (w_oci_mod * oci_score)
    w = np.clip(w, 0.0, 1.0) # Ensure w stays between 0 (Pure MF) and 1 (Pure MB)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value (weighted sum of MB and MF)
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness to the net value
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        # Standard TD(1) updates for MF values
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace for stage 1 MF (lambda=1 assumed here for MF component)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Dependent Punishment Sensitivity (Asymmetric Learning Rates)
This model tests the hypothesis that OCI is related to an oversensitivity to negative outcomes (fear/avoidance) specifically in the model-free system. It separates the learning rate into `alpha_pos` (for rewards) and `alpha_neg` (for punishments/omissions), where `alpha_neg` is modulated by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-Learning with Asymmetric Learning Rates modulated by OCI.
    
    Hypothesis: OCI increases sensitivity to negative prediction errors (punishment/loss),
    leading to faster learning from zero rewards compared to positive rewards.
    
    Parameters:
    alpha_pos: [0, 1] Learning rate for positive prediction errors.
    alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    alpha_neg_oci_slope: [-1, 1] How OCI modulates negative learning rate.
    beta: [0, 10] Inverse temperature.
    stickiness: [0, 5] Choice perseveration.
    """
    alpha_pos, alpha_neg_base, alpha_neg_oci_slope, beta, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate negative learning rate based on OCI
    alpha_neg = alpha_neg_base + (alpha_neg_oci_slope * oci_score)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        q_net = q_stage1.copy()
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        # Stage 2 PE
        delta2 = r - q_stage2[state_idx, a2]
        
        # Apply asymmetric learning rate for Stage 2
        lr2 = alpha_pos if delta2 >= 0 else alpha_neg
        q_stage2[state_idx, a2] += lr2 * delta2
        
        # Stage 1 PE (using updated Stage 2 value for TD)
        delta1 = q_stage2[state_idx, a2] - q_stage1[a1]
        
        # Apply asymmetric learning rate for Stage 1
        lr1 = alpha_pos if delta1 >= 0 else alpha_neg
        q_stage1[a1] += lr1 * delta1
        
        # Note: No explicit eligibility trace connecting r to s1 here to isolate the learning rate effect
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
