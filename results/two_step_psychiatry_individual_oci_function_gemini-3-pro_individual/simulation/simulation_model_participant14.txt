import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Perseveration Model.
    
    Hypothesis: High OCI relates to behavioral rigidity. The OCI score acts as a 
    gain on a 'stickiness' parameter, increasing the probability of repeating the 
    previous Stage 1 choice regardless of the outcome.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base = parameters
    
    # Handle OCI input (scalar or list/array)
    if oci is None:
        oci_score = 0.0
    elif np.ndim(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states, 2 actions per state
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed as per fitting code)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Calculate effective stickiness based on OCI
    effective_stickiness = stickiness_base * (1.0 + oci_score)

    # Storage for simulation results
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1  # Initialize with no previous action

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_net = q_stage1_mf.copy()

        # Apply perseveration bonus if a previous action exists
        if prev_action_1 != -1:
            q_net[prev_action_1] += effective_stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Select Action 1
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine State 2 based on Action 1 and transition matrix
        probs_state = transition_matrix[a1]
        s2 = np.random.choice([0, 1], p=probs_state)

        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using same beta as fitting code)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Select Action 2
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        r_prob = reward_probs[s2][a2]
        r = int(np.random.random() < r_prob)

        # --- Learning Updates ---
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (Direct Reinforcement from final reward)
        delta_stage1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
