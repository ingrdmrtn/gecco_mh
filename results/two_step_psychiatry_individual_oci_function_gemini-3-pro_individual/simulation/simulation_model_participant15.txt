import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using the Model with Choice Stickiness (Perseveration) modulated by OCI.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, stickiness_oci_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, stickiness_oci_mod = parameters
    
    # Handle OCI input (ensure it's a scalar for calculation)
    oci_score = oci[0] if isinstance(oci, (list, np.ndarray)) else oci

    # Initialize random number generator
    rng = np.random.default_rng()

    # Initialize arrays to store simulation data
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1 = np.zeros(2)
    # Stage 2: 2 states x 2 actions
    q_stage2 = np.zeros((2, 2))

    # Transition matrix (Standard 2-step task structure)
    # Row 0: transitions from action 0 (70% to state 0, 30% to state 1)
    # Row 1: transitions from action 1 (30% to state 0, 70% to state 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Calculate effective stickiness based on OCI
    eff_stickiness = stickiness_base + (stickiness_oci_mod * oci_score)

    last_action_1 = -1 

    for t in range(n_trials):
        # --- Stage 1 ---
        q_stage1_modified = q_stage1.copy()
        
        # Apply stickiness to the previous action
        if last_action_1 != -1:
            q_stage1_modified[int(last_action_1)] += eff_stickiness

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modified)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine State 2 based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 ---
        # Softmax selection for Stage 2 (using same beta as fitting code implies)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Get reward probabilities for current trial and state
        # State 0 uses drift1, drift2; State 1 uses drift3, drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        r = int(rng.random() < current_reward_probs[s2][a2])

        # --- Learning Updates (SARSA-style logic from fitting code) ---
        
        # Update Stage 1 Q-value based on Stage 2 Q-value of chosen action
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 Q-value based on received reward
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # --- Store Data & Cleanup ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
