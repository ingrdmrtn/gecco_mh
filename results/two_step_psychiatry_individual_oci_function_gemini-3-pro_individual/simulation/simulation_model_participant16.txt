import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Reinforcement Learning with OCI-modulated Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, persev_base, oci_persev_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list/array): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, persev_base, oci_persev_scale = parameters
    
    # Handle OCI input (support both scalar and array-like as seen in fitting code)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate constant stickiness factor
    stickiness = persev_base + (oci_score * oci_persev_scale)

    # Initialize Q-values
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    # Standard 2-step task transition matrix: [P(s2=0|a1=0), P(s2=1|a1=0)]
    #                                         [P(s2=0|a1=1), P(s2=1|a1=1)]
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 # Indicator for previous choice

    for t in range(n_trials):
        # 1. Stage 1 Decision
        logits_1 = beta * q_stage1.copy()
        
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness

        # Softmax Stage 1
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # 2. Transition to Stage 2
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        logits_2 = beta * q_stage2[s2]
        
        # Softmax Stage 2
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Reward Generation
        # Map state/action to the correct drift parameter for this trial
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < p_reward else 0

        # 5. Learning Updates (matching cognitive_model2 logic)
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD-0 style using value of chosen next state)
        v_next = q_stage2[s2, a2]
        delta_stage1 = v_next - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

        # Update tracking variables
        last_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
