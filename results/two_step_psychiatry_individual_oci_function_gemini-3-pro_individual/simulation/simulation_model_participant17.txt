import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using Model-Free reinforcement learning with choice stickiness (perseveration),
    where the stickiness parameter is scaled by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the second stage.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_factor = parameters
    
    # Handle OCI input (if passed as a list/array like in the fitting code, take first element)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate the effective stickiness (phi)
    phi = stickiness_factor * oci_score

    # Initialize Q-values
    # q_stage1 corresponds to the Q-values for the first stage actions (0 and 1)
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the Q-values for the second stage states (rows) and actions (cols)
    q_stage2 = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Transition matrix for the task (common structure for 2-step tasks)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    last_action_1 = -1 # Indicator for previous choice (none at start)
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        q_net_1 = q_stage1.copy()
        
        # Apply stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_net_1[last_action_1] += phi
            
        # Softmax choice rule for Stage 1
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1

        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # 3. Stage 2 Choice
        # Softmax choice rule for Stage 2 (using q_stage2 for the specific state s_idx)
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Reward Outcome
        # Get current reward probabilities for the specific state
        # State 0 uses drift1 (action 0) and drift2 (action 1)
        # State 1 uses drift3 (action 0) and drift4 (action 1)
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        prob_reward = current_reward_probs[s_idx][a2]
        
        r = int(rng.random() < prob_reward)

        # 5. Learning (Model-Free Updates)
        
        # Second stage update (TD error)
        # delta_2 = r - Q2(s', a')
        delta_2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_2

        # First stage update (TD error)
        # delta_1 = Q2(s', a') - Q1(a)
        # Note: The fitting code uses q_stage2[s_idx, a2] as the target for the first stage Q-value
        delta_1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
