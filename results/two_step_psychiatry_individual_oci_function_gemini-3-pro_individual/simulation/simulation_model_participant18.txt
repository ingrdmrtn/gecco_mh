import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Model-Free RL model where OCI modulates choice stickiness.
    
    This model implements a standard TD-learning approach with eligibility traces,
    but adds a perseveration bonus (stickiness) to the Stage 1 Q-values. The magnitude
    of this stickiness is modulated by the participant's OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_stickiness_scale, eligibility_trace]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0 or handles gracefully.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_stickiness_scale, eligibility_trace = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, take first element, otherwise float)
    current_oci = oci[0] if (isinstance(oci, (list, np.ndarray)) and len(oci) > 0) else (oci if oci is not None else 0.0)

    # Calculate the effective stickiness bonus based on OCI
    effective_stickiness = stickiness_base * (1 + oci_stickiness_scale * current_oci)

    # Transition matrix for the two-step task (Standard: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: 2 actions (Left, Right)
    q_stage1 = np.zeros(2)
    # q_stage2: 2 states (State 1, State 2) x 2 actions
    q_stage2 = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # Indicator for previous choice

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 (drift1, drift2), State 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_modified = q_stage1.copy()
        
        # Apply stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_stage1_modified[last_action_1] += effective_stickiness

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modified)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1 # Update for next trial

        # --- Transition ---
        # Determine Stage 2 state based on transition probabilities
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning (TD Updates) ---
        
        # Prediction error 1: Difference between Stage 2 value and Stage 1 value
        delta_1 = q_stage2[s2_idx, a2] - q_stage1[a1]

        # Prediction error 2: Difference between received reward and Stage 2 value
        delta_2 = r - q_stage2[s2_idx, a2]

        # Update Stage 2 Q-values
        q_stage2[s2_idx, a2] += learning_rate * delta_2

        # Update Stage 1 Q-values with eligibility trace
        q_stage1[a1] += learning_rate * (delta_1 + eligibility_trace * delta_2)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
