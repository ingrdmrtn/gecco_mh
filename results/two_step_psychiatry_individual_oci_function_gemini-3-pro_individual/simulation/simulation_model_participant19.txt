import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the "Model-Free learner with Forgetting Rate modulated by OCI" model.
    
    Logic:
    - Pure Model-Free TD learning.
    - OCI modulates the decay rate of UNCHOSEN options.
    - Standard softmax action selection.
    - No eligibility traces or perseveration (based on the provided fitting code).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, decay_base, oci_decay_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, decay_base, oci_decay_mod = parameters
    
    # Handle OCI input (default to 0 if not provided, though the model relies on it)
    oci_val = oci if oci is not None else 0.0

    # Calculate the fixed decay rate based on OCI
    decay_rate = decay_base + (oci_decay_mod * oci_val)
    decay_rate = np.clip(decay_rate, 0.0, 1.0)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1: Values for Spaceships A(0), U(1)
    q_stage1 = np.zeros(2) 
    # q_stage2: Values for Aliens (Planet X: 0,1; Planet Y: 0,1)
    # Note: In the fitting code, state is 0 or 1.
    q_stage2 = np.zeros((2, 2)) 

    # Transition matrix (fixed for the 2-step task)
    # 70% common transition, 30% rare transition
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Map drift inputs to the specific state/action pair
        # State 0 (Planet X) has options corresponding to drift1, drift2
        # State 1 (Planet Y) has options corresponding to drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < reward_probs[s2][a2])

        # 5. Learning / Updates
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += lr * delta_stage2

        # Decay unchosen option in Stage 2
        unchosen_a2 = 1 - a2
        q_stage2[s2, unchosen_a2] *= (1.0 - decay_rate)

        # Update Stage 1 Q-values (TD-0 style update using max Q of next stage)
        max_q2_next = np.max(q_stage2[s2])
        delta_stage1 = max_q2_next - q_stage1[a1]
        q_stage1[a1] += lr * delta_stage1

        # Decay unchosen option in Stage 1
        unchosen_a1 = 1 - a1
        q_stage1[unchosen_a1] *= (1.0 - decay_rate)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
