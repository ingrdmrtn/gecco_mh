import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free learner with choice perseveration 
    (stickiness) modulated by OCI.

    Model Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - stickiness_base: [0, 5] Base bonus added to the previously chosen action.
    - stickiness_oci_mod: [0, 5] Additional stickiness scaled by OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, stickiness_oci_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stickiness_base, stickiness_oci_mod = parameters
    
    # Handle OCI input
    current_oci = oci if oci is not None else 0.0

    # Calculate the combined perseveration weight
    perseveration_weight = stickiness_base + (stickiness_oci_mod * current_oci)

    # Transition matrix (fixed for the 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two actions in the two possible stage 2 states
    q_stage2_mf = np.zeros((2, 2))

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Apply perseveration bonus to Q-values if there was a previous action
        q_stage1_adjusted = q_stage1_mf.copy()
        if prev_action_1 != -1:
            q_stage1_adjusted[prev_action_1] += perseveration_weight
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_stage1_adjusted)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine stage 2 state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        # Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (TD Error using Stage 2 Q-value as proxy for value)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1 update
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous action for next trial's perseveration
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
