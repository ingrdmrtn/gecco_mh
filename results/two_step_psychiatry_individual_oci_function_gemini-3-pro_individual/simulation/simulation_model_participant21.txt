import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free RL model with OCI-modulated choice perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_sticky_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant. 
                             If None, defaults to 0 (no OCI modulation).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_sticky_factor = parameters
    
    # Handle OCI input
    current_oci = oci if oci is not None else 0.0

    # Calculate the effective perseveration weight based on OCI
    perseveration_weight = stickiness_base + (current_oci * oci_sticky_factor)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for the two choices in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two choices in each of the two states in stage 2
    # Shape (2, 2) -> (state, action)
    q_stage2_mf = np.zeros((2, 2))

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 # No previous action on trial 0

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drifts 1 and 2, State 1 to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_vals_s1 = q_stage1_mf.copy()
        
        # Apply OCI-modulated stickiness
        if last_action_1 != -1:
            q_vals_s1[last_action_1] += perseveration_weight

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_vals_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Select action based on Q-values for the specific state arrived at
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        r = int(rng.random() < current_reward_probs[s2][a2])

        # --- Learning (Model-Free) ---
        # Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD Error)
        # Note: The fitting code uses the updated Q-value of stage 2 as the target for stage 1
        # delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for stickiness
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
