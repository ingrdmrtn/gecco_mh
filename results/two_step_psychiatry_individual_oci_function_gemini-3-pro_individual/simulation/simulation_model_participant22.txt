import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with OCI-modulated choice stickiness.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_oci_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.
                             If None, defaults to 0 (no stickiness modulation).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, w, stick_oci_scale = parameters
    
    # Handle OCI input safely
    oci_score = oci if oci is not None else 0.0
    
    # Calculate effective stickiness based on OCI
    stickiness = stick_oci_scale * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF Q-values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # Current reward probabilities for the two states
        # state 0 corresponds to drift1/drift2, state 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (purely MF based on stage 2 Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning / Updates ---
        # Stage 1 MF update (TD(0))
        # Note: The fitting code uses TD(0) for stage 1 based on stage 2 Q-value, 
        # not eligibility traces (lambda).
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 MF update
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
