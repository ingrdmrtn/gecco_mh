import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid model with Reward-Dependent Stickiness (Win-Stay) modulated by OCI.
    
    Parameters:
    - n_trials (int): Number of trials to simulate.
    - parameters (list): [learning_rate, beta, w, stick_win_base, stick_win_oci_slope]
    - drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the second stage.
    - oci (float or list/array): Obsessive-Compulsive Inventoryâ€“Revised score. If list/array, first element is used.

    Returns:
    - stage1_choice (np.ndarray): First-stage choices (0 or 1).
    - state2 (np.ndarray): Second-stage states (0 or 1).
    - stage2_choice (np.ndarray): Second-stage choices (0 or 1).
    - reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_win_base, stick_win_oci_slope = parameters
    
    # Handle OCI input (can be scalar or array-like based on fitting code signature)
    if np.ndim(oci) > 0:
        current_oci = oci[0]
    else:
        current_oci = oci if oci is not None else 0.0

    # Calculate the composite stickiness parameter
    stick_win = stick_win_base + (current_oci * stick_win_oci_slope)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Model-free Q-values for stage 1 (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free Q-values for stage 2 (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Track history for win-stay stickiness
    last_action_1 = -1
    last_reward = 0

    for t in range(n_trials):
        # 1. Stage 1 Decision
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        
        # Apply Reward-Dependent Stickiness (Win-Stay)
        # Logic: If previous trial was rewarded, boost the logits of the previously chosen action
        if t > 0 and last_reward == 1 and last_action_1 != -1:
            logits[last_action_1] += stick_win
            
        # Softmax selection
        exp_q1 = np.exp(logits - np.max(logits)) # Numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # State 0 (common for a1=0, rare for a1=1), State 1 (rare for a1=0, common for a1=1)
        # Note: The fitting code implies state is 0 or 1.
        # transition_matrix[0] = [0.7, 0.3] -> probs of state 0, state 1 given action 0
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # 3. Stage 2 Decision
        # Simple Model-Free decision based on Q-values for the specific state
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map state/action to specific drift parameter
        if s2 == 0:
            current_drift = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_drift = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_drift)
        
        # 5. Learning (Update Q-values)
        # Update Stage 2 Q-values (TD(0))
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (TD(1) / SARSA-like update using stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
