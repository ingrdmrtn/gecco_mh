import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model-Free Q-learning with Choice Stickiness 
    modulated by OCI.

    Hypothesis: OCI is associated with behavioral rigidity or perseveration.
    The OCI score increases the 'stickiness' parameter, making the agent more 
    likely to repeat the previous Stage 1 action regardless of reward.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, treated as 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_perseveration = parameters
    
    # Handle OCI input safely
    current_oci = oci if oci is not None else 0.0

    # Calculate the stickiness parameter based on OCI
    stickiness_param = stickiness_base + (oci_perseveration * current_oci)

    # Initialize Q-values
    # q_stage1: Q-values for the two options in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2: Q-values for the two options in each of the two second-stage states
    q_stage2 = np.zeros((2, 2))

    # Transition matrix (fixed for this task structure)
    # Row 0: State 1 (Common 0.7 -> State 2, Rare 0.3 -> State 3)
    # Row 1: State 1 (Common 0.7 -> State 3, Rare 0.3 -> State 2)
    # Note: State indices in simulation usually map 0->State A, 1->State B.
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # Initialize invalid action
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Decision
        q_net = q_stage1.copy()
        
        # Apply stickiness to the previous action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_param
            
        # Softmax choice rule for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1

        # 2. Transition to Stage 2
        # Determine next state based on transition matrix probabilities
        # transition_matrix[a1] gives [prob_state_0, prob_state_1]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Softmax choice rule for Stage 2 (using just Q-values, no stickiness applied here in the fitting code)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Reward Generation
        # Map drifts to the specific state/action pair
        # s2=0 corresponds to drift1/drift2, s2=1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])

        # 5. Learning (Model-Free Q-learning)
        
        # Stage 2 update (TD error)
        # delta_stage2 = r - Q2(s2, a2)
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Stage 1 update (TD error passing back from Stage 2 Q-value)
        # delta_stage1 = Q2(s2, a2) - Q1(a1)
        # Note: The fitting code uses q_stage2[state_idx, a2] as the target, not the reward directly (SARSA-like logic)
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
