import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free learner with Choice Stickiness 
    (Perseveration) modulated by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_stickiness_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_stickiness_mod = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, take first element, else use float)
    participant_oci = oci if oci is not None else 0.0
    if isinstance(participant_oci, (list, np.ndarray)):
        participant_oci = participant_oci[0]

    # Calculate total stickiness
    stickiness = stickiness_base + (oci_stickiness_mod * participant_oci)

    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q-values for the two states in stage 2, each having 2 actions
    q_stage2_mf = np.zeros((2, 2))

    # Transition matrix (fixed for this task structure)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # No previous action on trial 0
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial [state][action]
        # state 0 (drift1, drift2), state 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        logits_1 = beta * q_stage1_mf.copy()

        # Apply stickiness if there was a previous action
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_prime = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_prime])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s_prime][a2])

        # --- Learning (Model-Free SARSA-like updates) ---
        # Prediction error for stage 2
        delta_stage2 = r - q_stage2_mf[s_prime, a2]
        
        # Prediction error for stage 1 (TD(0) style, using Q-value of chosen s2 action)
        delta_stage1 = q_stage2_mf[s_prime, a2] - q_stage1_mf[a1]
        
        # Update Q-values
        q_stage2_mf[s_prime, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_prime
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
