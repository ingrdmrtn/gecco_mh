import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-Driven Perseveration (Stickiness).
    
    This model assumes OCI modulates choice stickiness (perseveration).
    It uses a purely Model-Based planner for Stage 1 values, but adds a 
    stickiness bonus to the previously chosen option.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stick_base, oci_stick_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or np.ndarray): Obsessive-Compulsive Inventory score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, stick_base, oci_stick_mod = parameters
    
    # Handle OCI input format (scalar or array)
    current_oci = oci if np.isscalar(oci) else oci[0]
    
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    # Q-values for Stage 2 (Model-Free): 2 states, 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate total stickiness (constant per subject)
    total_stickiness = stick_base + (current_oci * oci_stick_mod)
    
    last_choice_1 = -1

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Calculate Stage 1 Values (Pure Model-Based)
        # Value of Stage 1 actions is the transition probability * max value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # 2. Stage 1 Selection (Softmax with Stickiness)
        logits_1 = beta * q_stage1_mb
        
        if last_choice_1 != -1:
            logits_1[last_choice_1] += total_stickiness
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = np.random.choice([0, 1], p=probs_1)

        # 3. Transition to Stage 2
        # If a1=0 -> probs [0.7, 0.3], if a1=1 -> probs [0.3, 0.7]
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Selection (Softmax on MF values)
        logits_2 = beta * q_stage2_mf[s2, :]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = np.random.choice([0, 1], p=probs_2)

        # 5. Outcome
        # Map state/action to specific drift array
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s2 == 1
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(np.random.random() < prob_reward)

        # 6. Learning (Update Stage 2 MF values only)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 7. Update history
        last_choice_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
