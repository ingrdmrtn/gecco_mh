import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None
):
    """
    Simulates choices and rewards using Model 2: OCI-Dependent Perseveration (Sticky-Choice).
    
    Hypothesis: OCI modulates choice perseveration (stickiness). Higher OCI might lead
    to a higher tendency to repeat the previous Stage 1 choice regardless of outcome.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness_oci_slope = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, extract value, otherwise use float)
    if oci is None:
        oci_score = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate the fixed stickiness value based on OCI
    stickiness_val = stickiness_oci_slope * oci_score

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (shape: 2)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for stage 2 states/actions (shape: 2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1 # Initialize with invalid action
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Calculate Model-Based values
        # Max Q-value for each second-stage state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation for MB values
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add Stickiness/Perseveration bonus if a previous action exists
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_val

        # Softmax action selection
        exp_q1 = np.exp(beta * q_net)
        if np.sum(exp_q1) == 0: 
            probs_1 = np.array([0.5, 0.5])
        else: 
            probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Standard Softmax on Stage 2 MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        if np.sum(exp_q2) == 0: 
            probs_2 = np.array([0.5, 0.5])
        else: 
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get reward probability for the specific terminal state (s_idx, a2)
        # Mapping: State 0 (Actions 0,1) -> drift1, drift2; State 1 (Actions 0,1) -> drift3, drift4
        if s_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # 5. Learning / Updates
        
        # Prediction Error Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Prediction Error Stage 1 (TD-error based on Q-value of stage 2 choice)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Eligibility trace update (Stage 1 Q-value updated by Stage 2 RPE)
        # Note: In the fitting code provided, lambda is implicitly 1.0 because 
        # the line is `q_stage1_mf[a1] += learning_rate * delta_stage2`
        q_stage1_mf[a1] += learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for stickiness
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
