import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid Model with OCI-Modulated Value Decay.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, decay_base, oci_decay_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, decay_base, oci_decay_sens = parameters
    
    # Handle OCI input (default to 0 if None provided, though model expects a value)
    current_oci = oci if oci is not None else 0.0

    # Calculate decay rate based on OCI
    decay_rate = decay_base + (current_oci * oci_decay_sens)
    # Clamp decay rate between 0 and 1
    if decay_rate < 0.0: decay_rate = 0.0
    if decay_rate > 1.0: decay_rate = 1.0

    # Initialize task structure
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    rng = np.random.default_rng()

    # Initialize Q-values
    # Stage 1 MF values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Get current reward probabilities for the second stage states
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Weighted sum of MB and MF)
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice for Stage 1
        logits_1 = beta * q_net_stage1
        # Subtract max for numerical stability
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (purely MF based on state)
        logits_2 = beta * q_stage2_mf[s2_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning ---
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1 MF Q-values (Direct reinforcement, no TD(lambda) in source model)
        q_stage1_mf[a1] += learning_rate * (r - q_stage1_mf[a1])

        # --- Decay (Forgetting) ---
        # Apply decay to all Q-values as specified in the source model
        q_stage1_mf *= (1.0 - decay_rate)
        q_stage2_mf *= (1.0 - decay_rate)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
