import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Based Learner with OCI-modulated Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, persev_base, oci_persev_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, persev_base, oci_persev_scale = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, extract value, or default to 0 if None)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    elif oci is not None:
        oci_score = oci
    else:
        oci_score = 0.0

    # Calculate the static perseveration bonus based on OCI
    perseveration_bonus = persev_base + (oci_persev_scale * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # Note: The fitting code only tracks MF values for Stage 2 (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2)) 
    
    last_action_1 = -1 # Initialize with no previous action
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 (index 0) corresponds to drift1/drift2
        # State 1 (index 1) corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1
        # Q_MB(s1, a) = sum(P(s2|s1,a) * max(Q_MF(s2, :)))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Calculate logits
        logits_1 = beta * q_stage1_mb
        
        # Apply perseveration bonus if there was a previous action
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration_bonus
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix
        # State 0 is usually 'common' for action 0, State 1 'common' for action 1
        # The fitting code uses state indices directly from data, here we simulate them.
        # transition_matrix[0] is probs for action 0 -> [0.7, 0.3]
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 using MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < current_reward_probs[s2_idx][a2])
        
        # --- Learning ---
        # Update Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
