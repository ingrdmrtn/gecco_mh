import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: Hybrid Model with OCI-Modulated Choice Kernel.
    
    This model combines Model-Based and Model-Free learning with a "Choice Kernel"
    representing habit strength. The influence of this habit is modulated by the OCI score.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, ck_weight_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    # Unpack parameters
    learning_rate, beta, w, ck_weight_oci = parameters
    
    # Handle OCI input (default to 0 if None, though model requires it)
    oci_score = oci if oci is not None else 0.0

    # Initialize random number generator
    rng = np.random.default_rng()

    # Transition matrix (fixed)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values and Choice Kernel
    q_mf = np.zeros((2, 2))      # Stage 2 values: [State, Action]
    q_mb = np.zeros(2)           # Stage 1 MB values (calculated dynamically)
    q_stage1_mf = np.zeros(2)    # Stage 1 MF values
    choice_kernel = np.zeros(2)  # Habit strength tracker

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_mf, axis=1)
        q_mb = transition_matrix @ max_q_stage2

        # Calculate Net Q-values (MB + MF)
        q_net = w * q_mb + (1 - w) * q_stage1_mf

        # Apply OCI-modulated Habit Bonus
        habit_bonus = (ck_weight_oci * oci_score) * choice_kernel
        q_net += habit_bonus

        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Update Choice Kernel (Habit)
        # The chosen action increases towards 1, unchosen decays towards 0
        choice_kernel[a1] += learning_rate * (1 - choice_kernel[a1])
        choice_kernel[1 - a1] += learning_rate * (0 - choice_kernel[1 - a1])

        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State is 0 or 1 (corresponding to indices in q_mf)
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        q_s2 = q_mf[s2_idx]
        exp_q2 = np.exp(beta * q_s2)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning Updates ---
        
        # TD(0) update for Stage 1 MF value
        # Note: In the provided fitting code, TD(0) uses Q_MF(s2, a2) as the target
        delta_stage1 = q_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # TD(0) update for Stage 2 MF value
        delta_stage2 = r - q_mf[s2_idx, a2]
        q_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
