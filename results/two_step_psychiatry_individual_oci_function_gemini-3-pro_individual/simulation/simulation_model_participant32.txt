import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Model-Free learner with Asymmetric Learning Rates 
    modulated by OCI.
    
    Hypothesis: OCI modulates sensitivity to negative prediction errors (alpha_neg).
    Individuals with high OCI might be hyper-sensitive to errors/missing rewards.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [base_alpha, beta, oci_neg_boost, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    base_alpha, beta, oci_neg_boost, stickiness = parameters
    
    # Handle OCI input (if provided as a scalar or single-element list/array)
    current_oci = 0.0
    if oci is not None:
        if isinstance(oci, (list, np.ndarray)):
            current_oci = oci[0]
        else:
            current_oci = oci

    # Initialize Q-values
    # q_stage1: values for stage 1 actions (0, 1)
    q_stage1 = np.zeros(2)
    # q_stage2: values for stage 2 states (0, 1) and actions (0, 1)
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed for the task structure usually)
    # 70% common, 30% rare transitions
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # 1. Stage 1 Choice
        q_stage1_mod = q_stage1.copy()
        
        # Apply stickiness (perseveration) to the previously chosen action
        if last_action_1 != -1:
            q_stage1_mod[last_action_1] += stickiness
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_mod)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1

        # 2. Transition to Stage 2
        # Determine state based on transition probabilities
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        state_idx = s2  # 0 or 1
        
        # 3. Stage 2 Choice
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Outcome
        # Determine reward based on drift probabilities for the current state/action
        # Mapping: state 0 -> (drift1, drift2), state 1 -> (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        current_reward_prob = reward_probs[state_idx][a2]
        r = int(rng.random() < current_reward_prob)

        # 5. Learning
        # Calculate Prediction Errors
        # PE1: Difference between value of Stage 2 state-action pair and Stage 1 action
        pe_1 = q_stage2[state_idx, a2] - q_stage1[a1]
        # PE2: Difference between actual reward and Stage 2 value
        pe_2 = r - q_stage2[state_idx, a2]

        # Update Stage 1 Q-values
        alpha_eff_1 = base_alpha
        if pe_1 < 0:
             # OCI modulates negative learning rate
             alpha_neg = base_alpha + (current_oci * oci_neg_boost)
             alpha_eff_1 = np.clip(alpha_neg, 0, 1)
        
        q_stage1[a1] += alpha_eff_1 * pe_1

        # Update Stage 2 Q-values
        alpha_eff_2 = base_alpha
        if pe_2 < 0:
             # OCI modulates negative learning rate
             alpha_neg = base_alpha + (current_oci * oci_neg_boost)
             alpha_eff_2 = np.clip(alpha_neg, 0, 1)

        q_stage2[state_idx, a2] += alpha_eff_2 * pe_2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
