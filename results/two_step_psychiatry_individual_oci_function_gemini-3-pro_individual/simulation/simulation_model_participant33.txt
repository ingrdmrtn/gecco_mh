import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid Model-Based/Model-Free model
    with OCI-modulated Choice Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_base, stickiness_oci_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.
        oci (list or float): Obsessive-Compulsive Inventoryâ€“Revised score. If a list/array, the first element is used.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness_base, stickiness_oci_mod = parameters
    
    # Handle OCI input (fitting code assumes it might be an array/list)
    if hasattr(oci, '__len__'):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate effective stickiness based on OCI
    eff_stickiness = stickiness_base + (stickiness_oci_mod * oci_score)

    # Initialize Q-values
    # q_stage1_mf: Q-values for Stage 1 actions (MF system)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for Stage 2 states x actions (MF system)
    # Row 0: State 2A (index 0), Row 1: State 2B (index 1)
    q_stage2_mf = np.zeros((2, 2)) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities map:
        # State 0 (2A) has actions leading to drift1, drift2
        # State 1 (2B) has actions leading to drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], 
                        [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Model-Based calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Hybrid calculation
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness to net Q-values
        q_net = q_hybrid.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += eff_stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State 0 (Common for a1=0), State 1 (Common for a1=1)
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (Pure MF based on fitting code logic)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s_idx][a2])
        
        # --- Learning (Update Q-values) ---
        
        # SARSA / TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # TD update for Stage 1 MF
        # Note: Fitting code uses q_stage2_mf[s_idx, a2] as the target, not reward directly
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
