import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF Model with 
    OCI-Modulated Value Decay (Memory Distrust).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, decay_base, decay_oci_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, decay_base, decay_oci_mod = parameters

    # Handle OCI input
    # In the fitting code, oci was passed as an array where we took index 0. 
    # Here it might be a float or None. We default to 0 if None.
    oci_score = oci if oci is not None else 0.0

    # Calculate the effective decay rate
    decay_rate = decay_base + (decay_oci_mod * oci_score)
    decay_rate = np.clip(decay_rate, 0.0, 1.0)

    # Transition probabilities
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for stage 2 (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # -----------------------
        # Stage 1 Decision
        # -----------------------
        
        # Calculate Model-Based values for Stage 1
        # MB value is the transition probability * max value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values: Weighted sum of MB and MF
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # -----------------------
        # Transition to Stage 2
        # -----------------------
        
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # -----------------------
        # Stage 2 Decision
        # -----------------------
        
        # Softmax choice for Stage 2 (purely MF based on state)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # -----------------------
        # Outcome
        # -----------------------
        
        # Determine reward based on drift probabilities for current trial
        # Map state/action to specific drift array
        if s2 == 0:
            current_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_prob)

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        # -----------------------
        # Learning & Decay
        # -----------------------
        
        unchosen_a2 = 1 - a2

        # Update Stage 2 Q-value (TD error)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Decay unchosen Stage 2 option
        q_stage2_mf[s2, unchosen_a2] *= (1.0 - decay_rate)

        # Update Stage 1 MF Q-value (TD error using updated Stage 2 value)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1 update
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Decay unchosen Stage 1 option
        q_stage1_mf[1 - a1] *= (1.0 - decay_rate)

    return stage1_choice, state2, stage2_choice, reward
