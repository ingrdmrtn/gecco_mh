```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: OCI-Modulated Memory Decay.

    Hypothesis:
    OCI scores modulate the decay rate of Q-values for unchosen options.
    High OCI leads to faster forgetting (higher decay) of values not recently experienced.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, decay_base, oci_decay_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    rng = np.random.default_rng()
    
    # Unpack parameters
    learning_rate, beta, w, decay_base, oci_decay_sens = parameters
    
    # Handle OCI input (scalar or list/array)
    oci_score = 0.0
    if oci is not None:
        if np.isscalar(oci):
            oci_score = oci
        elif len(oci) > 0:
            oci_score = oci[0]

    # Calculate decay rate modulated by OCI
    decay_rate = np.clip(decay_base + (oci_decay_sens * oci_score), 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the two choices in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the two states (rows) and two choices (cols) in stage 2
    q_stage2_mf = np.zeros((2, 2))

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Model-Based value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value (Weighted MB + MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice stage 2 (purely MF based on state)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning Updates ---
        # 1. Stage 1 MF Update (TD(0) prediction error from Stage 2 value)
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 2. Stage 2 MF Update (Prediction error from Reward)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # 3. Stage 1 MF Update (Eligibility trace / TD(1) logic passing reward back)
        # Note: The original code does `q_stage1_mf[action_1[trial]] += learning_rate * delta_stage2`
        # This implies lambda=1 logic hardcoded for the second update step.
        q_stage1_mf[a1] += learning_rate * delta_stage2

        # --- Decay Logic ---
        # Unchosen Stage 1 option decays
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        # Unchosen Stage 2 option (in the current state) decays
        unchosen_a2 = 1 - a2
        q_stage2_mf[s2_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
