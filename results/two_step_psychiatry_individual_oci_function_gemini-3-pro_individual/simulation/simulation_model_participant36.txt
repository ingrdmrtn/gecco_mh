import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-driven Choice Perseveration (Stickiness).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, w, stickiness_factor = parameters
    
    # Handle OCI input (if provided as a list/array or scalar)
    if oci is None:
        oci_score = 0.0
    elif hasattr(oci, '__len__') and len(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State 0 (drift1/2) and State 1 (drift3/4)

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_choice_1 = -1

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 leads to options with drift1 and drift2
        # State 1 leads to options with drift3 and drift4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Add Stickiness Bonus
        stickiness_bonus = np.zeros(2)
        if prev_choice_1 != -1:
            stickiness_bonus[int(prev_choice_1)] = stickiness_factor * oci_score
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * (q_net + stickiness_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 (purely Model-Free Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < current_drifts[s2][a2])

        # --- Learning Updates ---
        
        # TD Error Stage 1 (SARSA logic: Q(s2, a2) - Q(s1, a1))
        # Note: The fitting code uses q_stage2_mf[state_idx, action_2] as the target for stage 1 update
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Error Stage 2 (Reward - Q(s2, a2))
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update tracking variables
        prev_choice_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
