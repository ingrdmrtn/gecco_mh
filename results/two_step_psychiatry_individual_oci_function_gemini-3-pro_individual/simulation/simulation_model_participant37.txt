import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using Model 3: OCI-Driven Win-Stay Bias.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, oci_wsls_amp]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, oci_wsls_amp = parameters
    
    # Handle OCI input format (scalar or list/array)
    if hasattr(oci, '__getitem__') and not isinstance(oci, str):
         current_oci = oci[0]
    else:
         current_oci = oci

    # Initialize arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows) x 2 actions (cols)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Bias logic variables
    last_action_1 = -1
    last_reward = 0.0
    wsls_bias = oci_wsls_amp * current_oci
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Model-Based Value Calculation (Plan)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-Modulated Win-Stay Bias
        q_choice = q_net.copy()
        if last_action_1 != -1 and last_reward == 1.0:
            q_choice[last_action_1] += wsls_bias
            
        # Softmax selection
        exp_q1 = np.exp(beta * q_choice)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Transition probabilities depend on action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Determine reward probability based on current state (s2) and action (a2)
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # 5. Learning
        
        # Update Stage 1 (SARSA update using Q-value of chosen stage 2 action)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Update Stage 2 (TD update using actual reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] = q_stage2_mf[s2, a2] + learning_rate * delta_stage2
        
        # 6. Update History
        last_action_1 = a1
        last_reward = float(r)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
