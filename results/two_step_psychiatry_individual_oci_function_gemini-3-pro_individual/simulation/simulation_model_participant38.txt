import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Hybrid MB/MF with OCI-modulated Perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, pers_base, pers_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, pers_base, pers_oci_slope = parameters
    
    # Handle OCI input (default to 0 if None, though usually expected)
    current_oci = oci if oci is not None else 0.0
    
    # Calculate the static perseveration bonus based on OCI
    pers_bonus = pers_base + (pers_oci_slope * current_oci)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (MF)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (MF)
    # Shape (2, 2) corresponds to [state_idx, action_idx]
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values: Weighted sum of MB and MF
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply perseveration bonus to the previously chosen action
        if last_action_1 != -1:
            q_net_s1[last_action_1] += pers_bonus
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1

        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        # State index is 0 or 1 (unlike the example which used 1-based indexing for q_mf array slicing)
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---

        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        
        r = int(rng.random() < reward_probs[state_idx][a2])

        # --- Learning / Update ---
        
        # Note: The fitting code uses a specific update logic where stage 1 MF 
        # is updated by both the prediction error at stage 1 (based on stage 2 value)
        # and the prediction error at stage 2 (based on reward).
        
        # 1. Update Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # 2. Update Stage 1 Q-value
        # The fitting code calculates delta_stage1 *before* the stage 2 update implicitly,
        # but uses the *old* stage 2 value for the TD(0) part, and the *new* delta_stage2
        # for the TD(1)/eligibility trace part.
        # Looking closely at the fitting code:
        # delta_stage2 = r - q_stage2_mf[state_idx, a2] (computed before update)
        # q_stage2_mf[...] += ...
        # delta_stage1 = q_stage2_mf[state_idx, a2] (this is the UPDATED value) - q_stage1_mf[a1]
        # q_stage1_mf[a1] += lr * delta_stage1 + lr * delta_stage2
        
        # However, standard TD(lambda) usually uses the pre-update value for the first PE.
        # Let's strictly follow the logic order in the provided fitting code:
        
        # Re-calculating delta_stage2 as per fitting code line:
        # delta_stage2 = r - q_stage2_mf[state_idx, a2]
        # But we need the value *before* the update for the calculation.
        q_s2_a2_old = q_stage2_mf[state_idx, a2] # Store old value
        delta_stage2 = r - q_s2_a2_old
        
        # Apply update to Stage 2
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Calculate delta_stage1 using the UPDATED stage 2 value (as per fitting code logic)
        # "delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]"
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 1
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
