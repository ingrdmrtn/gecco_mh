import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 3: OCI-Driven Choice Perseveration.
    
    Hypothesis: OCI relates to a 'stickiness' parameter. Higher OCI leads to 
    a higher tendency to repeat the immediately preceding Stage 1 choice, 
    independent of reward history.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, stickiness_base, stickiness_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    lr, beta, w, stickiness_base, stickiness_oci_slope = parameters
    
    # Handle OCI input (scalar or list/array)
    oci_score = oci if np.isscalar(oci) else oci[0]
    
    # Calculate the total stickiness value based on OCI
    stickiness_val = stickiness_base + stickiness_oci_slope * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: state (0 or 1), Cols: action (0 or 1)

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Model-Based value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness (perseveration)
        q_net_sticky = q_net.copy()
        if prev_a1 != -1:
            q_net_sticky[prev_a1] += stickiness_val
        
        # Softmax choice 1
        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Transition to Stage 2
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice 2 (purely MF based on stage 2 Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # Reward outcome
        r = int(rng.random() < reward_probs[s_idx][a2])

        # --- Learning Updates ---
        # TD(0) update for Stage 1 (using Q-value of chosen stage 2 action as target)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2

        # Update tracking variables
        prev_a1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
