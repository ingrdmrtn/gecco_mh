import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Model-Free Q-Learning model 
    with OCI-modulated Inverse Temperature (Beta) and stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, beta_oci_mod, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_base, beta_oci_mod, stickiness = parameters
    
    # Handle OCI input to match fitting logic
    if oci is None:
        oci_score = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = float(oci)

    # Calculate modulated Beta
    beta = beta_base + (beta_oci_mod * oci_score)
    beta = np.clip(beta, 0.0, 20.0)

    # Transition matrix (Standard 2-step task structure: 0.7 common, 0.3 rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1: 2 choices
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows), 2 choices (cols)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # 1. Stage 1 Choice
        q_stage1_net = q_stage1_mf.copy()
        
        # Apply stickiness (perseveration)
        if prev_action_1 != -1:
            q_stage1_net[prev_action_1] += stickiness

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_stage1_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities for chosen action a1
        s2_probs = transition_matrix[a1]
        state_idx = np.random.choice([0, 1], p=s2_probs)
        
        # 3. Stage 2 Choice
        # Softmax for Stage 2 (using same beta as per fitting code)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Get current reward probabilities from drifts
        # drift1/2 correspond to state 0, drift3/4 correspond to state 1
        if state_idx == 0:
            current_probs = [drift1[t], drift2[t]]
        else:
            current_probs = [drift3[t], drift4[t]]
            
        prob_reward = current_probs[a2]
        r = int(np.random.random() < prob_reward)

        # 5. Learning Updates
        # Update logic matches cognitive_model3 exactly
        
        # TD error 1: Difference between Stage 2 Q-value and Stage 1 Q-value
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # TD error 2: Difference between Reward and Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # Eligibility trace update: Pass Stage 2 prediction error back to Stage 1 choice
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        # Update previous action for stickiness in next trial
        prev_action_1 = a1

        # Store data
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
