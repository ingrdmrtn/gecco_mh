import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-Driven Perseveration (Stickiness).
    
    This model simulates a pure Model-Based agent at stage 1 (calculating expected values
    via the transition matrix and max stage 2 values), but adds a stickiness bonus 
    to the previously chosen action. The magnitude of this stickiness is modulated by the OCI score.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, treated as 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1, representing the two terminal states).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, stickiness_oci = parameters
    
    # Handle OCI input
    oci_val = oci if oci is not None else 0.0
    
    # Calculate effective stickiness based on OCI
    eff_stickiness = stickiness_base + (stickiness_oci * oci_val)

    # Initialize arrays for storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Transition matrix (fixed as per the fitting code)
    # [P(State 0 | Action 0), P(State 1 | Action 0)]
    # [P(State 0 | Action 1), P(State 1 | Action 1)]
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])

    # Initialize Q-values for Stage 2 (Model-Free)
    # 2 states (columns), 2 actions (rows)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize "last action" tracker for stickiness
    last_action_1 = -1 
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Calculate Model-Based values for Stage 1
        # V(s') = max_a Q_MF(s', a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Q_MB(s, a) = T * V(s')
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Apply Stickiness
        q_net = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += eff_stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        qs_current_state = q_stage2_mf[s2_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drift inputs to the specific state/action pair
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r_prob = current_reward_probs[s2_idx][a2]
        r = int(rng.random() < r_prob)
        
        # 5. Learning (Update Stage 2 Q-values)
        pe_2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * pe_2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update last action for next trial's stickiness
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
