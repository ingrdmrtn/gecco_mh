import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid learner with Choice Stickiness 
    (Perseveration) modulated by OCI.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_base, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci = parameters
    
    # Handle OCI input safely
    oci_score = oci if oci is not None else 0.0
    
    # Calculate effective stickiness based on OCI
    effective_stickiness = stickiness_base + (stickiness_oci * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 (model-free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 (model-free), shape (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drift1/drift2, State 1 to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits = beta * q_net
        
        # Apply Stickiness/Perseveration
        if last_action_1 != -1:
            logits[last_action_1] += effective_stickiness
            
        # Softmax selection
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax for stage 2 (pure model-free)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2_idx][a2])
        
        # --- Learning (Updates) ---
        
        # Update Stage 2 Q-values
        rpe_2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * rpe_2

        # Update Stage 1 Q-values (TD(1) style update based on final reward)
        # Note: The fitting code uses 'reward - q_stage1_mf', implying a direct update 
        # from the final outcome rather than a TD(0) update from stage 2 value.
        rpe_1_final = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * rpe_1_final

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
