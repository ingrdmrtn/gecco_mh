import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid RL model where the balance between 
    Model-Based (MB) and Model-Free (MF) control is modulated by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_base, oci_sensitivity]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w_base, oci_sensitivity = parameters
    
    # Handle OCI input (default to 0 if None, though usually provided)
    current_oci = oci if oci is not None else 0.0

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: MF values for stage 1 (Spaceships) - shape (2,)
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf: MF values for stage 2 (Aliens) - shape (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2)) 

    # Transition matrix (fixed as per fitting code)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Calculate mixing weight w based on OCI
    raw_w = w_base - (current_oci * oci_sensitivity)
    w = np.clip(raw_w, 0.0, 1.0)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 using MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])

        # --- Learning Updates ---
        # Update Stage 2 Q-values (SARSA / TD(0) logic from fitting code)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD logic using Stage 2 Q-value as target)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1 update
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
