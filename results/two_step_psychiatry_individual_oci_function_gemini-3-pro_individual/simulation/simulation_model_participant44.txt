import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Model-Free RL model with OCI-modulated 
    choice perseveration (stickiness).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_stick_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_stick_mod = parameters
    
    # Handle OCI input (if it's a list/array, take the first element, otherwise use directly)
    if oci is not None:
        if isinstance(oci, (list, np.ndarray)):
            oci_val = oci[0]
        else:
            oci_val = oci
    else:
        oci_val = 0.0 # Default if not provided, though logic implies it should be

    # Calculate effective stickiness based on OCI
    effective_stickiness = stickiness_base + (oci_val * oci_stick_mod)
    
    # Transition matrix (standard for the 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: values for the two actions at stage 1
    q_stage1 = np.zeros(2)
    # q_stage2: values for the two states (rows) and two actions (cols) at stage 2
    q_stage2 = np.zeros((2, 2))
    
    # Storage for simulation results
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 # No previous action on trial 0

    for t in range(n_trials):
        # Current reward probabilities for the two states
        # state 0 corresponds to drifts 1 and 2
        # state 1 corresponds to drifts 3 and 4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_modified = q_stage1.copy()
        
        # Apply stickiness bonus if there was a previous action
        if last_action_1 != -1:
            q_stage1_modified[last_action_1] += effective_stickiness
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modified)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Determine Stage 2 State based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 (no stickiness applied here in the fitting code)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_drifts[s2][a2])
        
        # --- Learning (TD updates) ---
        
        # Stage 2 Update: RPE = Reward - Q_stage2(s2, a2)
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update: RPE = Q_stage2(s2, a2) - Q_stage1(a1)
        # Note: The fitting code uses the *updated* Q-value of stage 2 as the target for stage 1
        # or simply the value of the chosen state/action pair.
        # Following the fitting code logic: q_stage2[s2_state, s2_choice] is used.
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
