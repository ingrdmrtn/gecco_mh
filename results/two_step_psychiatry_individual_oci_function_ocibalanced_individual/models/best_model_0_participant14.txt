def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with OCI-modulated Perseveration (Stickiness).
    
    This model adds a 'stickiness' bonus to the previously chosen action.
    The magnitude of this stickiness is modulated by the OCI score.
    
    Parameters:
    - learning_rate: Learning rate [0, 1].
    - beta: Inverse temperature [0, 10].
    - stickiness_base: Base tendency to repeat the last choice [0, 5].
    - oci_sticky_slope: How strongly OCI increases stickiness [0, 5].
    """
    learning_rate, beta, stickiness_base, oci_sticky_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]


    stickiness = stickiness_base + (oci_score * oci_sticky_slope)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]


        q_stage1_modulated = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_modulated[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_stage1_modulated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]


        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2


        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
