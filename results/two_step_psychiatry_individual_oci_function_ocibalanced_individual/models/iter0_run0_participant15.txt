Here are three cognitive models that incorporate the participant's OCI score into the decision-making process in the two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that the balance between goal-directed (Model-Based) and habitual (Model-Free) control is influenced by obsessive-compulsive traits. Specifically, the mixing weight `w` is modulated by the OCI score. Higher OCI scores might lead to a stronger reliance on habitual (Model-Free) systems (lower `w`), reflecting the rigid, repetitive nature of compulsions.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free Reinforcement Learning with OCI-modulated mixing weight.
    
    The mixing weight 'w' determines the balance between Model-Based (MB) and Model-Free (MF) 
    values for the first-stage choice. The OCI score linearly modulates this weight, 
    testing if symptom severity shifts the controller balance.
    
    Parameters:
    learning_rate: [0, 1] Rate at which Q-values are updated.
    beta: [0, 10] Inverse temperature for softmax choice (exploration/exploitation).
    w_base: [0, 1] Baseline mixing weight (0 = pure MF, 1 = pure MB).
    w_oci_slope: [-1, 1] How strongly OCI affects the mixing weight.
    """
    learning_rate, beta, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate the effective mixing weight based on OCI
    # We clip it to ensure it stays valid [0, 1]
    w = w_base + (w_oci_slope * oci_score)
    if w < 0: w = 0
    if w > 1: w = 1

    # Fixed transition matrix for the task (commonly used in these models)
    # A -> X (0.7), U -> Y (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 planets, 2 aliens)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based calculation: V(state) = max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value: w * MB + (1-w) * MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax selection
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of the action actually taken
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Transition to state (planet)
        state_idx = int(state[trial]) # 0 or 1

        # --- Stage 2 Policy ---
        # Standard Model-Free choice at the second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # SARSA / TD(1) style updates
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Prediction Error (using the value of the state actually reached)
        # Note: In standard hybrid models, MF stage 1 is updated by the stage 2 value
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Eligibility trace update (updating stage 1 with stage 2 outcome)
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Stickiness (Perseveration)
This model posits that OCI symptoms manifest as behavioral rigidity or "stickiness" (perseveration). Instead of altering the learning mechanism itself, the OCI score modulates a choice autocorrelation parameter (`stickiness`). A participant with higher OCI might be more likely to repeat their previous Stage 1 choice regardless of the reward outcome.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Reinforcement Learning with OCI-modulated Choice Stickiness (Perseveration).
    
    This model adds a 'stickiness' bonus to the previously chosen action at Stage 1.
    The magnitude of this bonus is scaled by the OCI score.
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-values.
    beta: [0, 10] Inverse temperature.
    stickiness_base: [0, 5] Base tendency to repeat the last choice.
    stickiness_oci: [0, 5] Additional stickiness scaling with OCI score.
    """
    learning_rate, beta, stickiness_base, stickiness_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness
    eff_stickiness = stickiness_base + (stickiness_oci * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Based for simplicity of base logic, modified by stickiness
    # (Though we track MF values for the second stage to inform MB)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice (initialize to -1 or None)
    prev_action_1 = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add stickiness bonus to the Q-values before softmax
        q_net = q_stage1_mb.copy()
        if prev_action_1 != -1:
            q_net[prev_action_1] += eff_stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update history
        prev_action_1 = a1
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Learning Rate Asymmetry
This model suggests that OCI affects how individuals learn from positive versus negative outcomes. Specifically, it tests if higher OCI scores lead to a differential learning rate for rewards (`alpha_pos`) versus omissions (`alpha_neg`). This reflects the hypothesis that compulsive behavior might be driven by an over-sensitivity to negative outcomes (avoidance) or an inability to update values after non-reward.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with OCI-modulated Asymmetric Learning Rates.
    
    The learning rate is split into positive (alpha_pos) and negative (alpha_neg)
    updating. The OCI score modulates the negative learning rate specifically,
    testing if symptom severity impacts learning from non-reward (omissions).
    
    Parameters:
    alpha_base: [0, 1] Base learning rate.
    alpha_oci_mod: [-1, 1] Modulation of learning rate by OCI.
    beta: [0, 10] Inverse temperature.
    asymmetry: [0, 1] Ratio of negative to positive learning.
    """
    alpha_base, alpha_oci_mod, beta, asymmetry = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Define effective learning rates
    # Alpha positive is base
    alpha_pos = alpha_base
    
    # Alpha negative is modulated by asymmetry and OCI
    # If asymmetry > 0.5, negative learning is boosted.
    # We allow OCI to further scale this negative learning rate.
    alpha_neg_raw = alpha_base * asymmetry + (alpha_oci_mod * oci_score)
    
    # Bounds checking
    if alpha_neg_raw < 0: alpha_neg_raw = 0
    if alpha_neg_raw > 1: alpha_neg_raw = 1
    alpha_neg = alpha_neg_raw

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Free TD(1)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        r = reward[trial]
        
        # Determine which learning rate to use based on prediction error sign
        # We calculate PE first to decide alpha
        pe2 = r - q_stage2_mf[state_idx, a2]
        lr_eff = alpha_pos if pe2 >= 0 else alpha_neg
        
        # Update Stage 2
        q_stage2_mf[state_idx, a2] += lr_eff * pe2
        
        # Update Stage 1
        # TD(1) update: Stage 1 value is updated by Stage 2 PE
        q_stage1_mf[a1] += lr_eff * pe2
        
        # Also standard TD(0) update for Stage 1 transition
        pe1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Usually we use the same LR for consistency within a trial
        q_stage1_mf[a1] += lr_eff * pe1
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
