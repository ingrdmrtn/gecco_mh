Here are three cognitive models that incorporate the OCI-R score into the decision-making process of the two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that high OCI scores (indicative of compulsivity) lead to a deficit in goal-directed (Model-Based) control, favoring habitual (Model-Free) control. The `w` parameter, which controls the balance between these systems, is modulated by the participant's OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid Model-Based/Model-Free with OCI-modulated weight (w).
    
    Hypothesis: Higher OCI scores reduce the contribution of the Model-Based system (goal-directed),
    making the participant rely more on Model-Free (habitual) values.
    
    Parameters:
    learning_rate (alpha): [0, 1] Rate at which Q-values are updated.
    beta: [0, 10] Inverse temperature for softmax choice rule.
    w_base: [0, 1] Baseline mixing weight (0 = pure MF, 1 = pure MB).
    w_oci_slope: [-1, 1] How strongly OCI affects the mixing weight w.
    """
    learning_rate, beta, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w based on OCI. Bounded between 0 and 1.
    # If slope is negative, higher OCI leads to lower w (more habitual).
    w = w_base + (w_oci_slope * oci_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (Aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE ---
        # Model-Based Value Calculation: V(S2) * Transition_Prob
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- STAGE 2 CHOICE ---
        state_idx = int(state[trial]) # 0 for Planet X, 1 for Planet Y
        
        # Standard Model-Free Q-learning for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- UPDATING ---
        # Prediction errors
        # Stage 1 MF update (SARSA-like logic using Stage 2 value)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        # Stage 2 MF update (Standard Q-learning)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Perseveration
This model hypothesizes that high OCI scores are associated with "stickiness" or perseverationâ€”a tendency to repeat the previous action regardless of reward history. The OCI score scales the perseveration parameter `p`.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Pure Model-Based with OCI-modulated Perseveration.
    
    Hypothesis: High OCI scores relate to behavioral rigidity. This model adds a 
    'perseveration bonus' to the previously chosen action, scaled by OCI.
    
    Parameters:
    learning_rate: [0, 1] Update rate for stage 2 values.
    beta: [0, 10] Inverse temperature.
    pers_base: [-2, 2] Baseline tendency to repeat choice (positive) or switch (negative).
    pers_oci_sens: [0, 5] Sensitivity of perseveration to OCI score.
    """
    learning_rate, beta, pers_base, pers_oci_sens = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective perseveration parameter
    # Higher OCI = Higher tendency to repeat the previous Stage 1 action
    perseveration_weight = pers_base + (pers_oci_sens * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage2_mf = np.zeros((2, 2)) # Values for aliens
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1 # Initialize variable to track previous choice

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE ---
        # Pure Model-Based calculation for underlying value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add perseveration bonus to the Q-values before softmax
        q_stage1_adjusted = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_stage1_adjusted[int(last_action_1)] += perseveration_weight
            
        exp_q1 = np.exp(beta * q_stage1_adjusted)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Record this action for next trial's perseveration
        last_action_1 = action_1[trial]
        
        # --- STAGE 2 CHOICE ---
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- UPDATING ---
        # Only updating Stage 2 values (Pure MB assumes fixed transition model)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Learning Rate Asymmetry
This model hypothesizes that individuals with high OCI scores might learn differently from positive versus negative prediction errors (e.g., being hyper-sensitive to errors or lack of reward). The OCI score shifts the learning rate specifically for negative prediction errors (disappointment).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Model-Free with OCI-modulated Negative Learning Rate.
    
    Hypothesis: OCI affects how quickly participants update values after a lack of reward.
    This implements a dual learning rate model (alpha_pos, alpha_neg), where alpha_neg
    is a function of OCI.
    
    Parameters:
    alpha_pos: [0, 1] Learning rate for positive prediction errors (RPE > 0).
    alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    alpha_neg_oci: [-1, 1] How OCI modifies the negative learning rate.
    beta: [0, 10] Inverse temperature.
    """
    alpha_pos, alpha_neg_base, alpha_neg_oci, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate
    alpha_neg = alpha_neg_base + (alpha_neg_oci * oci_score)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    # Pure Model-Free setup
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- STAGE 1 CHOICE ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- STAGE 2 CHOICE ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- UPDATING ---
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        
        # Apply asymmetric learning rates
        lr_2 = alpha_pos if delta_stage2 >= 0 else alpha_neg
        q_stage2_mf[state_idx, int(action_2[trial])] += lr_2 * delta_stage2
        
        # Stage 1 Update (TD(0) style, using Stage 2 Q-value as proxy for reward)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        
        # Apply asymmetric learning rates
        lr_1 = alpha_pos if delta_stage1 >= 0 else alpha_neg
        q_stage1_mf[int(action_1[trial])] += lr_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
