Here are three cognitive models designed to capture the behavior of a participant with high obsessive-compulsive symptoms (high OCI score) in the two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that high OCI scores correlate with a rigidity that favors habitual (model-free) control over goal-directed (model-based) planning. The `w` parameter (mixing weight) is modulated by the OCI score, where higher OCI pushes the agent towards model-free reinforcement learning.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid Model-Based/Model-Free RL with OCI-modulated mixing weight.
    
    Hypothesis: Higher OCI scores reduce the reliance on model-based (planning) strategies,
    pushing the participant towards model-free (habitual) control. The mixing weight 'w'
    is derived from a base parameter and shifted by the OCI score.

    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice (exploration/exploitation).
    - w_base: [0, 1] Baseline weight for model-based control (0=pure MF, 1=pure MB).
    - oci_sensitivity: [0, 1] Strength of OCI's negative impact on model-based weighting.
    """
    learning_rate, beta, w_base, oci_sensitivity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    # Transition matrix: A->X (0.7), U->Y (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)

    # Calculate effective mixing weight w. 
    # High OCI reduces w (more MF), low OCI increases w (more MB).
    # We clip to ensure it stays in [0, 1].
    w = w_base - (current_oci * oci_sensitivity)
    w = np.clip(w, 0.0, 1.0)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-based value: T * max(Q2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated value: weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of the chosen action (0 or 1)
        # Action data is 0.0 or 1.0, cast to int for indexing
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Prediction error for Stage 1 (TD(0))
        # Note: In standard hybrid models, MF updates via SARSA-like logic
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Prediction error for Stage 2
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace for Stage 1 (update stage 1 MF value based on stage 2 outcome)
        # This connects the final reward back to the first choice in the MF system
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Perseveration
This model posits that obsessive-compulsive traits manifest as "stickiness" or perseverationâ€”a tendency to repeat the previous action regardless of reward history. The OCI score directly scales the `stickiness` parameter added to the Q-values.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Reinforcement Learning with OCI-Scaled Perseveration (Stickiness).
    
    Hypothesis: High OCI scores relate to compulsive repetition of actions. 
    Here, the OCI score acts as a multiplier on a 'stickiness' bonus added to the 
    Q-value of the previously chosen action.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Softmax inverse temperature.
    - stickiness_factor: [0, 5] Base magnitude of choice repetition bonus.
    """
    learning_rate, beta, stickiness_factor = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    # Transition matrix
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    last_action_1 = -1 

    # Calculate total stickiness bonus based on OCI
    # Higher OCI = Higher tendency to repeat
    effective_stickiness = stickiness_factor * current_oci

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        # Pure Model-Free logic for simplicity, augmented with stickiness
        q_vals_1 = q_stage1_mf.copy()
        
        # Add stickiness bonus if a previous action exists
        if last_action_1 != -1:
            q_vals_1[last_action_1] += effective_stickiness

        exp_q1 = np.exp(beta * q_vals_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update last action
        last_action_1 = a1
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # TD(1) / SARSA style update logic for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Propagate Stage 2 RPE back to Stage 1
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Influenced Learning Rates (Separate Positive/Negative)
This model suggests that high OCI scores alter how individuals process positive versus negative feedback (anxiety/avoidance vs. reward seeking). The model splits the learning rate into `alpha_pos` and `alpha_neg`. The OCI score is used to bias the ratio between these two, hypothesizing that high OCI individuals might be hyper-sensitive to negative prediction errors (punishment/lack of reward).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Dual Learning Rate Model with OCI-biased Sensitivity.
    
    Hypothesis: High OCI participants may learn differently from positive (reward) 
    vs negative (no reward) outcomes. This model scales the learning rate for 
    negative prediction errors (alpha_neg) upwards based on the OCI score.
    
    Parameters:
    - alpha_base: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - neg_bias_strength: [0, 1] How much OCI amplifies learning from negative outcomes.
    """
    alpha_base, beta, neg_bias_strength = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
  
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate specific learning rates
    # Alpha pos is just the base rate
    alpha_pos = alpha_base
    
    # Alpha neg is boosted by OCI. If OCI is high, they update more aggressively on failure.
    # We constrain the resulting alpha to be <= 1.
    alpha_neg = alpha_base + (current_oci * neg_bias_strength)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)

    for trial in range(n_trials):

        # --- Stage 1 Choice (Model-Free) ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Note: We apply the split learning rate to the Second Stage update primarily,
        # as that is where the concrete Reward is received.
        
        # Stage 1 Update (Standard TD)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Use base alpha for transition prediction errors (no explicit valence)
        q_stage1_mf[a1] += alpha_base * delta_stage1
        
        # Stage 2 Update (Reward Prediction Error)
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Choose learning rate based on sign of prediction error
        if delta_stage2 >= 0:
            lr = alpha_pos
        else:
            lr = alpha_neg
            
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Propagate to Stage 1
        q_stage1_mf[a1] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
