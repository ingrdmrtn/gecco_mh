Here are three new cognitive models designed to capture decision-making in the two-step task, specifically incorporating the OCI score to modulate reinforcement learning and model-based planning processes.

### Model 1: Hybrid Model with OCI-Modulated Model-Based Weight (`w`)
This model assumes that individuals with different OCI scores rely differently on Model-Based (planning) vs. Model-Free (habitual) systems. The parameter `w` controls the balance between these two systems. Here, the OCI score directly scales the `w` parameter, testing the hypothesis that compulsivity (even at low levels like 0.3) might correlate with a shift towards habitual (model-free) control or rigid planning.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with OCI-modulated mixing weight.
    
    The balance between Model-Based (MB) and Model-Free (MF) control is determined by 
    parameter `w`. This model posits that the effective `w` is influenced by the 
    participant's OCI score.
    
    Parameters:
    - learning_rate: Learning rate for value updates [0, 1].
    - beta: Inverse temperature for softmax choice [0, 10].
    - w_base: Base weighting for Model-Based control [0, 1].
    - oci_mod: Strength of OCI modulation on the mixing weight [0, 1].
    """
    learning_rate, beta, w_base, oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective w. We constrain it to [0, 1].
    # Hypothesis: OCI might reduce flexibility/MB planning, or conversely increase rigid planning.
    # Here we model it as a modulation: w_eff = w_base * (1 +/- oci * mod). 
    # Let's assume OCI reduces MB control (shifting to habits).
    w = w_base * (1.0 - (oci_score * oci_mod))
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Q-values for stage 2 (same for MB and MF)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        # Model-Based Value Calculation: V_MB(s1) = T(s1, s2) * max(Q(s2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        # Standard Model-Free choice at stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 RPE (TD(0))
        # Note: In a full hybrid model, MF updates occur based on the transition
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseveration Model with OCI-Dependent Stickiness
This model investigates whether the OCI score relates to "stickiness" or perseverationâ€”the tendency to repeat the previous choice regardless of reward history. This is a common feature in compulsivity, where behavior can become repetitive.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with OCI-modulated Perseveration (Stickiness).
    
    This model adds a 'stickiness' bonus to the previously chosen action.
    The magnitude of this stickiness is modulated by the OCI score.
    
    Parameters:
    - learning_rate: Learning rate [0, 1].
    - beta: Inverse temperature [0, 10].
    - stickiness_base: Base tendency to repeat the last choice [0, 5].
    - oci_sticky_slope: How strongly OCI increases stickiness [0, 5].
    """
    learning_rate, beta, stickiness_base, oci_sticky_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective stickiness
    # Hypothesis: Higher OCI leads to higher perseveration.
    stickiness = stickiness_base + (oci_score * oci_sticky_slope)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    # Track previous choice (initialize with -1 or handle first trial separately)
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        # Add stickiness bonus to Q-values before softmax
        q_stage1_modulated = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_modulated[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_stage1_modulated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Standard SARSA/Q-learning updates
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        # TD(1) / Eligibility trace update for stage 1 (simplified)
        # Using the value of the state chosen in stage 2 to update stage 1
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Inverse Temperature (Exploration/Exploitation)
This model posits that OCI affects the randomness of choice (exploration vs. exploitation). A common finding in compulsivity research is either rigid behavior (high beta) or noisy/inconsistent behavior (low beta). This model allows the `beta` parameter to scale linearly with the OCI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Reinforcement Learning with OCI-modulated Inverse Temperature (Beta).
    
    The inverse temperature (beta) controls the trade-off between exploitation 
    (choosing the best option) and exploration (choosing randomly).
    This model allows the OCI score to shift the participant's position on this curve.
    
    Parameters:
    - learning_rate: Learning rate [0, 1].
    - beta_intercept: Baseline inverse temperature [0, 10].
    - beta_slope: Effect of OCI on inverse temperature [-5, 5].
    - lambd: Eligibility trace parameter [0, 1].
    """
    learning_rate, beta_intercept, beta_slope, lambd = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective beta
    # We clip to ensure it stays within reasonable positive bounds [0, 20]
    beta_eff = beta_intercept + (oci_score * beta_slope)
    beta_eff = np.clip(beta_eff, 0.0, 20.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta_eff * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta_eff * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Update Stage 2
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 with eligibility trace (lambda)
        # If lambda=0, it's pure TD(0). If lambda=1, it's Monte Carlo-like.
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        
        # Combining immediate stage 1 error with stage 2 error passed back
        q_stage1[a1] += learning_rate * delta_stage1 + (lambd * learning_rate * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
