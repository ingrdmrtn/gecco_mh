Here are three new cognitive models exploring different mechanisms by which Obsessive-Compulsive Inventory (OCI) scores might influence decision-making in the two-step task.

### Model 1: Hybrid Model with OCI-modulated Model-Based Weighting
This model posits that individuals with higher OCI scores might rely more on habitual (model-free) control rather than goal-directed (model-based) planning. The parameter `w` (mixing weight) is usually static, but here it is a function of the OCI score. A high OCI score reduces the influence of the model-based system.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid RL where OCI score modulates the balance between Model-Based (MB) 
    and Model-Free (MF) control.
    
    Hypothesis: Higher OCI scores are associated with reduced model-based control (compulsivity/habit).
    
    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature (softness of choice).
    - w_base: [0, 1] Baseline mixing weight (0=Pure MF, 1=Pure MB).
    - oci_mod: [0, 1] Strength of OCI reduction on w.
      (Effective w = w_base * (1 - oci_mod * oci_score))
    """
    learning_rate, beta, w_base, oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective mixing weight w
    # We clip to ensure w stays within [0, 1]
    w = w_base * (1.0 - (oci_mod * oci_score))
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value Calculation
        # Max Q value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # MB value is transition probability * best future value
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        chosen_a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[chosen_a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[chosen_a2]

        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 Update (TD)
        # Note: In standard hybrid models, Stage 2 Q-values are purely MF
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # Stage 1 MF Update (TD(1) / Eligibility Trace style)
        # Using the actual reward received at the end to update stage 1 MF
        delta_stage1 = r - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-modulated Learning Rate Asymmetry (Pos/Neg)
This model investigates if OCI scores relate to how participants learn from positive versus negative outcomes. Specifically, it tests if higher OCI leads to "over-learning" from failure (negative outcomes) or success, by splitting the learning rate into a base rate and an OCI-dependent modifier applied specifically to negative prediction errors.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Model-Free RL where OCI modulates learning from negative prediction errors.
    
    Hypothesis: High OCI participants may be more sensitive to "failure" (0 reward),
    resulting in a higher learning rate for negative prediction errors.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors.
    - beta: [0, 10] Inverse temperature.
    - lr_neg_base: [0, 1] Baseline learning rate for negative prediction errors.
    - oci_neg_boost: [0, 1] Additional learning rate for neg errors per unit of OCI.
    """
    lr_pos, beta, lr_neg_base, oci_neg_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective negative learning rate
    lr_neg = lr_neg_base + (oci_neg_boost * oci_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        chosen_a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[chosen_a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        chosen_a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[chosen_a2]

        r = reward[trial]

        # --- Updating Stage 2 ---
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        
        effective_alpha_2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[state_idx, chosen_a2] += effective_alpha_2 * delta_stage2
        
        # --- Updating Stage 1 ---
        # Using SARSA-like update: Q1(a1) moves toward Q2(s, a2)
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        
        effective_alpha_1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[chosen_a1] += effective_alpha_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-modulated Exploration (Inverse Temperature)
This model tests the hypothesis that OCI scores affect the exploration-exploitation trade-off. Specifically, it posits that high OCI (associated with anxiety or rigidity) might lead to lower exploration (higher `beta`), making choices more deterministic based on current values.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Model-Free RL where OCI modulates the inverse temperature (beta).
    
    Hypothesis: Higher OCI leads to more rigid/deterministic behavior (higher beta),
    reducing random exploration.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta_base: [0, 10] Baseline inverse temperature.
    - oci_stiffness: [0, 10] Increase in beta per unit of OCI.
    - eligibility: [0, 1] Eligibility trace (lambda) connecting Stage 2 reward to Stage 1.
    """
    learning_rate, beta_base, oci_stiffness, eligibility = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective beta
    # High OCI -> Higher Beta -> Less exploration (more exploitation)
    beta_eff = beta_base + (oci_stiffness * oci_score)
    # Cap beta to prevent overflow
    beta_eff = np.clip(beta_eff, 0.0, 20.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta_eff * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        chosen_a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[chosen_a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        chosen_a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[chosen_a2]

        r = reward[trial]

        # --- Updating ---
        # Stage 2 update
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # Stage 1 update using eligibility trace logic
        # Instead of just Q2-Q1, we allow the reward R to directly influence Q1 via lambda
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        
        # The update combines immediate TD error at stage 1 (which is 0 usually) 
        # and the propagated error from stage 2 scaled by eligibility
        q_stage1_mf[chosen_a1] += learning_rate * (delta_stage1 + eligibility * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
