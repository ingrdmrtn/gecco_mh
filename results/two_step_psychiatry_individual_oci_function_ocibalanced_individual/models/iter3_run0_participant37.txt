Here are three new cognitive models that explore different mechanisms for how OCI scores might influence decision-making in the two-step task, specifically focusing on mixing model-based/model-free strategies and learning rate modulations.

### Model 1: OCI-Modulated Model-Based Weighting (Linear Mixing)
This model hypothesizes that the balance between model-based (planning) and model-free (habitual) control is directly influenced by OCI symptoms. Specifically, it tests whether higher OCI scores lead to a shift in the weighting parameter `w`, potentially reflecting a reliance on rigid habits (low `w`) or over-deliberation (high `w`).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Based Weighting.
    
    This model assumes a hybrid reinforcement learning agent where the contribution 
    of the model-based system (w) is a linear function of the OCI score.
    This tests if OCI symptoms shift the balance between goal-directed (MB) 
    and habitual (MF) control.

    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_base: [0,1] - Baseline weighting for model-based values.
    w_oci_mod: [-1,1] - Modulation of w by OCI score.
    """
    learning_rate, beta, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate mixing weight w based on OCI, clamped between 0 and 1
    w = w_base + (w_oci_mod * oci_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free Q-values for stage 2

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-based valuation: V(s') = max_a Q(s', a)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid valuation: Q_net = w * Q_MB + (1-w) * Q_MF
        q_stage1_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_stage1_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0 # Handle missing data

        # Skip update if data is missing
        if a1 == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue

        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        # Stage 2 update (TD learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 update (TD learning for MF component)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    log_loss = -(np.sum(np.log(p_choice_1)) + np.sum(np.log(p_choice_2)))
    return log_loss
```

### Model 2: OCI-Dependent Learning Rate for Losses
This model investigates whether OCI is associated with a specific sensitivity to negative outcomes (absence of reward). It implements separate learning rates for rewards and non-rewards (losses), where the learning rate for losses is scaled by the OCI score. This reflects the hypothesis that high OCI individuals might over-learn from failure or lack of control.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Dependent Learning Rate for Losses.
    
    This model separates learning from positive outcomes (reward=1) and 
    neutral outcomes (reward=0). The learning rate for neutral outcomes 
    is modulated by the OCI score, testing if OCI relates to hyper-sensitivity 
    to 'failure' or missing rewards.
    
    Bounds:
    lr_pos: [0,1] - Learning rate for rewarded trials.
    lr_neg_base: [0,1] - Baseline learning rate for unrewarded trials.
    lr_neg_oci: [-1,1] - Modulation of negative learning rate by OCI.
    beta: [0,10]
    w_mb: [0,1] - Fixed model-based weight.
    """
    lr_pos, lr_neg_base, lr_neg_oci, beta, w_mb = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective learning rate for negative outcomes
    lr_neg = lr_neg_base + (lr_neg_oci * oci_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_stage1_net = w_mb * q_stage1_mb + (1 - w_mb) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_stage1_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0

        if a1 == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue

        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        # Determine which learning rate to use
        current_lr = lr_pos if r > 0 else lr_neg
        
        # Stage 2 update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += current_lr * delta_stage2
        
        # Stage 1 update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1

    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    log_loss = -(np.sum(np.log(p_choice_1)) + np.sum(np.log(p_choice_2)))
    return log_loss
```

### Model 3: OCI-Linked Inverse Temperature (Exploration/Exploitation)
This model posits that OCI affects the randomness of choices (exploration vs. exploitation). It models the inverse temperature parameter `beta` as a function of OCI. High OCI might lead to very rigid, deterministic choices (high beta) or, conversely, high uncertainty and noise (low beta).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Linked Inverse Temperature.
    
    This model tests if OCI scores impact the exploration-exploitation trade-off.
    The inverse temperature (beta), which controls choice stochasticity, is 
    modeled as a linear function of OCI. 
    
    Bounds:
    learning_rate: [0,1]
    beta_base: [0,10] - Baseline inverse temperature.
    beta_oci: [-5,5] - Modulation of beta by OCI.
    w_mb: [0,1] - Fixed model-based weight.
    """
    learning_rate, beta_base, beta_oci, w_mb = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate beta based on OCI, ensuring it stays non-negative
    beta = beta_base + (beta_oci * oci_score)
    beta = np.maximum(beta, 0.0) # Inverse temp cannot be negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_stage1_net = w_mb * q_stage1_mb + (1 - w_mb) * q_stage1_mf
        
        # Use the OCI-modulated beta here
        exp_q1 = np.exp(beta * q_stage1_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0

        if a1 == -1 or state[trial] == -1 or action_2[trial] == -1:
            continue

        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 2 Choice ---
        # Use the OCI-modulated beta here as well
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    log_loss = -(np.sum(np.log(p_choice_1)) + np.sum(np.log(p_choice_2)))
    return log_loss
```
