Here are 3 new cognitive models that introduce different mechanisms for how OCI scores might influence decision-making in this two-step task, specifically focusing on the balance between model-based and model-free control, learning rates, and exploration.

### Model 1: OCI-Modulated Model-Based/Model-Free Hybrid
This model hypothesizes that high OCI scores correlate with a rigid, habitual (model-free) strategy rather than a flexible, goal-directed (model-based) strategy. It implements a hybrid reinforcement learning agent where the weighting parameter $w$ (which arbitrates between model-based and model-free values) is directly modulated by the OCI score. A higher OCI reduces $w$, pushing the agent towards model-free control.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where OCI modulates the balance (w).
    Hypothesis: High OCI leads to reduced Model-Based control (lower w).
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1] - Baseline mixing weight (0=MF, 1=MB)
    oci_penalty: [0, 1] - Reduction in MB weight due to OCI
    """
    learning_rate, beta, w_base, oci_penalty = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective mixing weight w
    # We clip to ensure it stays within [0, 1]
    # High OCI reduces w, making behavior more Model-Free
    w = w_base - (oci_penalty * oci_score)
    w = np.clip(w, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free Q-values for stage 2 (also used for MB calculation)

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation: V_MB(s1) = T(s1, s2) * max(Q(s2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updating ---
        r = reward[trial]
        
        # Update Stage 2 Q-values (Standard Q-learning)
        # Q(s2, a2) <- Q(s2, a2) + alpha * (r - Q(s2, a2))
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF Q-values (TD(0))
        # Q(s1, a1) <- Q(s1, a1) + alpha * (Q(s2, a2) - Q(s1, a1))
        # Note: In standard hybrid models, stage 1 MF is often updated via TD(1) (lambda), 
        # but here we stick to the simpler TD(0) structure implied by the template logic.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Learning Rate Asymmetry
This model investigates whether OCI scores affect how participants learn from positive versus negative prediction errors. It posits that high OCI might be associated with a "safety" or "perfectionistic" bias where learning from negative outcomes (failures to get gold) is amplified compared to positive outcomes. This creates an asymmetry in the learning rate $\alpha$.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with asymmetric learning rates modulated by OCI.
    Hypothesis: High OCI amplifies learning from negative prediction errors (loss aversion/perfectionism).
    
    Bounds:
    alpha_pos: [0, 1] - Learning rate for positive prediction errors
    alpha_neg_base: [0, 1] - Base learning rate for negative prediction errors
    beta: [0, 10]
    oci_neg_amp: [0, 5] - Amplification of negative learning rate by OCI
    """
    alpha_pos, alpha_neg_base, beta, oci_neg_amp = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective negative learning rate
    # High OCI increases sensitivity to "bad" news (negative PE)
    alpha_neg_effective = alpha_neg_base + (oci_neg_amp * oci_score)
    alpha_neg_effective = np.clip(alpha_neg_effective, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        r = reward[trial]

        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        if delta_stage2 >= 0:
            eff_lr_s2 = alpha_pos
        else:
            eff_lr_s2 = alpha_neg_effective
        q_stage2_mf[s_idx, a2] += eff_lr_s2 * delta_stage2

        # Stage 1 Update
        # Using simple SARSA-style TD error: Q(s2, a2) - Q(s1, a1)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        if delta_stage1 >= 0:
            eff_lr_s1 = alpha_pos
        else:
            eff_lr_s1 = alpha_neg_effective
        q_stage1_mf[a1] += eff_lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Softmax Temperature (Exploration/Exploitation)
This model proposes that OCI scores influence the randomness of choice behavior (exploration vs. exploitation). Specifically, it tests the idea that high OCI leads to lower decision noise (higher $\beta$), representing a rigid adherence to the perceived best option and reduced exploration, even when uncertainty exists.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Based learner where OCI modulates the inverse temperature (beta).
    Hypothesis: High OCI leads to higher beta (more rigid/deterministic choices).
    
    Bounds:
    learning_rate: [0, 1]
    beta_base: [0, 10]
    beta_oci_slope: [0, 10] - How much OCI increases beta (rigidity)
    lambda_eligibility: [0, 1] - Eligibility trace decay (0=TD(0), 1=TD(1))
    """
    learning_rate, beta_base, beta_oci_slope, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Effective Beta: beta = base + slope * OCI
    # High OCI -> Higher Beta -> More deterministic (less exploration)
    beta_effective = beta_base + (beta_oci_slope * oci_score)
    # Cap beta to prevent overflow in exp
    beta_effective = np.clip(beta_effective, 0.0, 20.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2) # We will treat this as a purely model-free value for simplicity in this specific variant
                           # or we could make it MB. Let's stick to the template's implication of MF updating
                           # but apply the beta modulation to it.
                           # Actually, to make it distinct, let's use a pure MF agent but focus on the beta modulation.
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta_effective * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta_effective * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updating ---
        r = reward[trial]
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update with Eligibility Trace logic (TD(lambda))
        # Delta is R + Q(s2,a2) - Q(s1,a1) effectively if we view the whole sequence
        # But following the template structure:
        # Standard TD(1) update for stage 1 is often: Q(s1,a1) += alpha * (Reward - Q(s1,a1))
        # Standard TD(0) update is: Q(s1,a1) += alpha * (Q(s2,a2) - Q(s1,a1))
        # We blend them using lambda.
        
        # TD(0) target: Value of state 2
        target_0 = q_stage2[s_idx, a2]
        # TD(1) target: Actual Reward
        target_1 = r
        
        # Blended target
        target_combined = (1 - lambda_eligibility) * target_0 + lambda_eligibility * target_1
        
        delta_stage1 = target_combined - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
