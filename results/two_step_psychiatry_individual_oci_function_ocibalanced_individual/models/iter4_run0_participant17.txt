def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid Learner with OCI-Modulated Memory Decay.
    
    Hypothesis: Obsessive-compulsive symptoms are associated with an inability to 
    maintain representations of alternative (unchosen) values. High OCI leads to 
    faster decay of unchosen action values towards zero, representing a "tunnel vision" 
    or obsession with the current path and neglecting counterfactuals.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature (softmax randomness).
    - w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    - decay_base: [0, 1] Baseline decay rate for unchosen options.
    - decay_oci_slope: [0, 1] Impact of OCI on increasing decay rate.
    """
    learning_rate, beta, w, decay_base, decay_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective decay rate bounded between 0 and 1
    # decay_rate determines how much the unchosen option value reduces per trial
    decay_rate = decay_base + (decay_oci_slope * oci_score)
    if decay_rate > 1.0: decay_rate = 1.0
    if decay_rate < 0.0: decay_rate = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-Free values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-Free values for Stage 2 (2 planets x 2 aliens)

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        # Model-Based Value: Expected value of next stage based on transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Weighted sum of MB and MF
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Policy for Choice 2 ---
        # Standard Softmax on Stage 2 values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updating ---
        
        # 1. Update Stage 1 MF (TD-learning / SARSA)
        # The value of the state arrived at (max_q_stage2 or q of chosen)
        # Using simple Q-learning update for Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Decay unchosen Stage 1 option
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        # 2. Update Stage 2 MF
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Decay unchosen Stage 2 option (on the current planet)
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        # Note: We do not decay the values of the unvisited planet as they were not available.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Hybrid Learner with OCI-Modulated "Loss-Chasing" Stickiness.
    
    Hypothesis: Unlike general stickiness (repeating any choice), OCI is linked 
    to compulsivity specifically after negative feedback. High OCI participants 
    may feel a compulsion to "fix" a mistake or "chase" the loss, leading to 
    increased repetition of the Stage 1 choice specifically after receiving 0 rewards.

    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - stick_loss_base: [0, 5] Baseline tendency to repeat Stage 1 choice after a loss (0 reward).
    - stick_loss_oci: [0, 5] Additional loss-stickiness per unit of OCI.
    """
    learning_rate, beta, w, stick_loss_base, stick_loss_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Stickiness applied only if previous trial reward was 0
    stickiness_after_loss = stick_loss_base + (stick_loss_oci * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Loss-Chasing Stickiness
        if last_action_1 != -1 and last_reward == 0.0:
            q_net_stage1[last_action_1] += stickiness_after_loss
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        state_idx = int(state[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updating ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Hybrid Learner with OCI-Modulated Stage 2 Learning Rate.
    
    Hypothesis: OCI individuals may focus disproportionately on immediate, concrete 
    outcomes (Stage 2 aliens and coins) rather than the abstract predictive structure 
    (Stage 1 spaceships). This model proposes that OCI specifically boosts the 
    learning rate for the second stage (Action-outcome) relative to the first stage.

    Parameters:
    - lr_stage1: [0, 1] Learning rate for Stage 1 (Spaceships).
    - lr_stage2_base: [0, 1] Baseline learning rate for Stage 2 (Aliens).
    - lr_stage2_oci: [0, 1] Increase in Stage 2 learning rate per unit OCI.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    """
    lr_stage1, lr_stage2_base, lr_stage2_oci, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Stage 2 specific learning rate
    lr_stage2 = lr_stage2_base + (lr_stage2_oci * oci_score)
    if lr_stage2 > 1.0: lr_stage2 = 1.0
    if lr_stage2 < 0.0: lr_stage2 = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        state_idx = int(state[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updating ---
        # Update Stage 1 using lr_stage1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Update Stage 2 using OCI-modulated lr_stage2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
