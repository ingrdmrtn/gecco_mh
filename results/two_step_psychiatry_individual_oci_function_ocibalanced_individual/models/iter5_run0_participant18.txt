Here are three new cognitive models that introduce different mechanisms for how OCI scores might influence decision-making in the two-step task, specifically focusing on learning rates and model-based weighting in ways not yet tested.

### Model 1: OCI-Modulated Model-Based Weighting (Sigmoid)
This model hypothesizes that the balance between model-based (planning) and model-free (habitual) control is modulated by OCI symptoms, but non-linearly. Instead of a linear slope, we use a sigmoid function to bound the weight $w$ strictly between 0 and 1, where the inflection point is determined by the OCI score. This tests if high OCI leads to extreme rigidity (high MB) or habit reliance (high MF).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner where the Model-Based weight (w) is a sigmoid function of OCI.
    
    Hypothesis: The trade-off between goal-directed (MB) and habitual (MF) control
    changes with OCI severity. This implementation bounds w between 0 and 1 using
    a logistic function, allowing OCI to push the participant towards pure MB or pure MF.
    
    Parameters:
    - learning_rate: Learning rate for value updates.
    - beta: Inverse temperature for softmax.
    - w_bias: The intercept of the sigmoid function (baseline tendency).
    - w_oci_strength: The slope of the sigmoid with respect to OCI. 
                      Positive values mean higher OCI -> more Model-Based.
                      Negative values mean higher OCI -> more Model-Free.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_bias: [-5, 5]
    w_oci_strength: [-10, 10]
    """
    learning_rate, beta, w_bias, w_oci_strength = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate w using a sigmoid function to keep it valid [0,1]
    # w = 1 / (1 + exp(-(bias + strength * oci)))
    logit_w = w_bias + (w_oci_strength * oci_score)
    w = 1.0 / (1.0 + np.exp(-logit_w))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted combination of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # SARSA-style update for stage 1 MF
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Standard Q-learning for stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Specific Punishment Sensitivity
This model tests the hypothesis that OCI is related to an over-sensitivity to negative outcomes (punishment). Instead of modulating the mixing weight ($w$) or randomness ($\beta$), OCI modulates the learning rate specifically for unrewarded trials (losses).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner with separate learning rates for rewards and losses,
    where the loss learning rate is modulated by OCI.
    
    Hypothesis: Individuals with high OCI scores may be hyper-sensitive to 
    negative feedback (lack of reward). 
    lr_neg = lr_base + (lr_oci_scale * OCI).
    
    Parameters:
    - lr_pos: Learning rate for rewarded trials (outcome = 1).
    - lr_neg_base: Baseline learning rate for unrewarded trials (outcome = 0).
    - lr_neg_oci_scale: How much OCI increases sensitivity to losses.
    - beta: Inverse temperature.
    - w: Weighting between model-based and model-free.
    
    Bounds:
    lr_pos: [0,1]
    lr_neg_base: [0,1]
    lr_neg_oci_scale: [0,2] (Allowed to boost LR, clipped at 1 inside)
    beta: [0,10]
    w: [0,1]
    """
    lr_pos, lr_neg_base, lr_neg_oci_scale, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective negative learning rate based on OCI
    lr_neg = lr_neg_base + (lr_neg_oci_scale * oci_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0) # Ensure valid bounds

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Determine current learning rate based on outcome
        if r > 0:
            current_lr = lr_pos
        else:
            current_lr = lr_neg

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += current_lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Eligibility Trace (Lambda)
This model introduces an eligibility trace parameter ($\lambda$) that controls how much credit from the second-stage reward is assigned to the first-stage choice. We hypothesize that OCI affects this credit assignment processâ€”perhaps high OCI leads to "over-thinking" causality (high $\lambda$) or struggling to link distant outcomes (low $\lambda$).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Free learner with Eligibility Traces (TD(lambda)), where lambda is OCI-modulated.
    
    Hypothesis: The ability to link Stage 2 outcomes back to Stage 1 choices (credit assignment)
    varies with OCI. This replaces the explicit Model-Based component with a more sophisticated
    Model-Free mechanism (eligibility traces).
    
    Parameters:
    - learning_rate: Learning rate.
    - beta: Inverse temperature.
    - lambda_base: Baseline eligibility trace decay parameter.
    - lambda_oci_scale: How much OCI alters the eligibility trace.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    lambda_base: [0,1]
    lambda_oci_scale: [-1, 1]
    """
    learning_rate, beta, lambda_base, lambda_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective lambda
    eff_lambda = lambda_base + (lambda_oci_scale * oci_score)
    eff_lambda = np.clip(eff_lambda, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Policy for Choice 1 ---
        # Note: Pure MF here, no MB component
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates with Eligibility Trace ---
        # 1. Prediction error at stage 1 (driven by value of state arrived at)
        delta_1 = q_stage2[s_idx, a2] - q_stage1[a1]
        
        # 2. Prediction error at stage 2 (driven by reward)
        delta_2 = r - q_stage2[s_idx, a2]
        
        # Update Stage 1 Value:
        # It gets updated by its own immediate error (delta_1)
        # PLUS the discounted error from stage 2 (eff_lambda * delta_2)
        q_stage1[a1] += learning_rate * (delta_1 + eff_lambda * delta_2)
        
        # Update Stage 2 Value:
        q_stage2[s_idx, a2] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
