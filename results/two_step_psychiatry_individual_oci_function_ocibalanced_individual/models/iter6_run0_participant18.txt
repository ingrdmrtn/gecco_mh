Here are three new cognitive models exploring different mechanisms by which Obsessive-Compulsive Inventory (OCI) scores might influence decision-making in the two-step task.

### Model 1: OCI-Modulated Model-Based Weighting (Sigmoid)
This model hypothesizes that OCI scores influence the balance between model-based (planning) and model-free (habitual) control. High OCI might be associated with rigid, habit-like behavior (low $w$) or over-deliberation (high $w$). Here, we model $w$ as a sigmoid function of OCI to keep it strictly bounded between 0 and 1, testing if OCI shifts the arbitration point.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner where the weighting parameter 'w' is a sigmoid function of OCI.
    
    Hypothesis: OCI scores shift the balance between Model-Based (planning) and 
    Model-Free (habit) systems. The sigmoid ensures w stays in [0,1].
    
    Parameters:
    - learning_rate: Learning rate for value updates.
    - beta: Inverse temperature for softmax.
    - w_slope: Slope of the sigmoid relationship between OCI and w.
    - w_midpoint: The OCI score at which w is 0.5 (inflection point).
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_slope: [-10, 10] (Can be negative or positive)
    w_midpoint: [0, 1] (Since OCI is normalized/bounded roughly 0-1)
    """
    learning_rate, beta, w_slope, w_midpoint = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate w using a sigmoid function
    # w = 1 / (1 + exp(-slope * (oci - midpoint)))
    # If slope is positive, higher OCI -> higher w (more Model-Based)
    # If slope is negative, higher OCI -> lower w (more Model-Free)
    w = 1.0 / (1.0 + np.exp(-w_slope * (oci_score - w_midpoint)))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: V_MB(s1) = T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # SARSA(0) update for Stage 1 Model-Free
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Q-learning update for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Learning Rate Asymmetry
This model hypothesizes that OCI relates to how individuals process positive versus negative prediction errors. Specifically, high OCI might be associated with "safety behavior" or anxiety-driven avoidance, potentially leading to stronger learning from punishment (negative RPEs) than reward (positive RPEs), or vice versa.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dual learning rate model where OCI modulates the ratio between 
    positive and negative learning rates.
    
    Hypothesis: OCI affects sensitivity to positive vs negative outcomes.
    We define a base learning rate, and an asymmetry parameter modulated by OCI.
    
    Parameters:
    - lr_base: The baseline learning rate.
    - asymmetry_base: Baseline bias towards positive (>0.5) or negative (<0.5) updates.
    - asymmetry_oci: How OCI shifts this bias.
    - beta: Inverse temperature.
    - w: Model-based weight.
    
    Bounds:
    lr_base: [0,1]
    asymmetry_base: [0,1]
    asymmetry_oci: [-1, 1] 
    beta: [0,10]
    w: [0,1]
    """
    lr_base, asymmetry_base, asymmetry_oci, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective asymmetry parameter K
    # K determines the split: alpha_pos = lr * K, alpha_neg = lr * (1-K)
    # We clip K to stay in [0.01, 0.99] to avoid zero learning rates
    K = asymmetry_base + (asymmetry_oci * oci_score)
    K = np.clip(K, 0.01, 0.99)
    
    # We scale lr_base so that the effective alpha never exceeds 1
    # If K is high, alpha_pos is high. If K is low, alpha_neg is high.
    # We ensure lr_base is interpreted as the sum or max magnitude.
    # Here, let's treat lr_base as the max possible learning rate.
    alpha_pos = lr_base * K
    alpha_neg = lr_base * (1.0 - K)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        if delta_stage1 >= 0:
            q_stage1_mf[a1] += alpha_pos * delta_stage1
        else:
            q_stage1_mf[a1] += alpha_neg * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        if delta_stage2 >= 0:
            q_stage2_mf[s_idx, a2] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[s_idx, a2] += alpha_neg * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Eligibility Trace (Lambda)
This model tests if OCI affects the credit assignment process between the two stages. The parameter $\lambda$ (eligibility trace) controls how much the second-stage reward directly updates the first-stage choice. A higher OCI might imply "hyper-binding" or stronger causal attribution across time steps (high $\lambda$), or conversely, a disconnect (low $\lambda$).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with Eligibility Traces (TD(lambda)) where lambda is modulated by OCI.
    
    Hypothesis: OCI affects how credit is assigned to past actions (Stage 1) 
    based on current outcomes (Stage 2 reward). 
    Lambda = 0 is pure TD(0) (only immediate next state matters).
    Lambda = 1 is Monte Carlo (outcome fully updates start).
    
    Parameters:
    - learning_rate: Learning rate.
    - beta: Inverse temperature.
    - w: Model-based weight.
    - lambda_base: Baseline eligibility trace decay.
    - lambda_oci_slope: How OCI affects lambda.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1]
    lambda_base: [0,1]
    lambda_oci_slope: [-1, 1]
    """
    learning_rate, beta, w, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective lambda
    eff_lambda = lambda_base + (lambda_oci_slope * oci_score)
    eff_lambda = np.clip(eff_lambda, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates with Eligibility Trace ---
        
        # 1. Prediction error at Stage 1 (TD error)
        # The 'reward' for stage 1 is 0, plus the value of the next state (Q_stage2)
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1
        
        # 2. Prediction error at Stage 2
        delta2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta2
        
        # 3. Eligibility Trace update:
        # The Stage 2 RPE (delta2) also updates Stage 1 Q-value, scaled by lambda.
        # This connects the final reward back to the first choice.
        q_stage1_mf[a1] += learning_rate * eff_lambda * delta2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
