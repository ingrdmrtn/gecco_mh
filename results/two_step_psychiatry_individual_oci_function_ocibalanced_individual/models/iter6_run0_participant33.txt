Here are three new cognitive models for the two-step task, incorporating OCI scores in distinct ways inspired by computational psychiatry literature (e.g., balance between model-based/model-free control, perseveration, and uncertainty).

### Model 1: OCI-modulated Model-Based Weight ($w$)
This model hypothesizes that OCI symptoms relate to the balance between goal-directed (Model-Based) and habitual (Model-Free) control. Specifically, it tests if higher OCI scores lead to a reliance on rigid, habitual (Model-Free) strategies (lower $w$) or over-deliberation (higher $w$). The parameter `w` is modeled as a linear function of OCI.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-modulated Model-Based Weight.
    
    This model assumes the balance between Model-Based (MB) and Model-Free (MF) control
    is linearly related to the OCI score.
    w = w_base + (w_oci_slope * oci)
    w is clamped between [0, 1].
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature for softmax.
    - lambda_eligibility: [0, 1] Eligibility trace for Stage 1 updates.
    - w_base: [0, 1] Baseline mixing weight (0=MF, 1=MB) for an OCI of 0.
    - w_oci_slope: [-1, 1] Slope of OCI effect on w.
    """
    learning_rate, beta, lambda_eligibility, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate mixing weight w based on OCI
    w = w_base + (w_oci_slope * oci_score)
    # Clamp w to [0, 1]
    if w < 0.0: w = 0.0
    if w > 1.0: w = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-based value calculation
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        # Standard Model-Free Q-learning for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        # Stage 2 PE
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 PE
        # Note: SARSA-style update often used here, but using max for consistency with MB
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1 + (learning_rate * lambda_eligibility * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-modulated Choice Stickiness (Perseveration)
This model tests the hypothesis that OCI is associated with "stickiness" or perseverationâ€”a tendency to repeat the previous choice regardless of reward history. Here, the stickiness parameter in the softmax function is directly modulated by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-modulated Choice Stickiness.
    
    This model adds a 'stickiness' bonus to the previously chosen action at Stage 1.
    The magnitude of this stickiness is determined by the OCI score.
    stickiness = stick_base + (stick_oci_slope * oci)
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weighting.
    - stick_base: [-5, 5] Baseline stickiness.
    - stick_oci_slope: [-5, 5] How much OCI adds to stickiness.
    """
    learning_rate, beta, w, stick_base, stick_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate stickiness parameter
    stickiness = stick_base + (stick_oci_slope * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Indicator for previous action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness to the Q-value of the previously chosen action
        q_net_sticky = q_net.copy()
        if last_action_1 != -1:
            q_net_sticky[last_action_1] += stickiness
        
        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1 # Update for next trial
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # MF Stage 1 update (no eligibility trace in this simplified model to focus on stickiness)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-modulated Transition Learning Rate (State Inference)
This model posits that OCI affects how participants update their internal model of the world (the transition matrix). A high OCI might reflect hyper-vigilance to environmental changes, causing the participant to update their belief about transition probabilities ($P(State|Action)$) more rapidly than controls. This is distinct from reward learning rates.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-modulated Transition Learning Rate (Dynamic Transition Matrix).
    
    Instead of a fixed transition matrix, the agent learns transition probabilities.
    The rate at which they update these probabilities is modulated by OCI.
    lr_transition = lr_trans_base + (lr_trans_oci_slope * oci)
    
    Parameters:
    - learning_rate_reward: [0, 1] Learning rate for Q-values (rewards).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weighting.
    - lr_trans_base: [0, 1] Base learning rate for transition probabilities.
    - lr_trans_oci_slope: [-1, 1] OCI modulation on transition learning.
    """
    learning_rate_reward, beta, w, lr_trans_base, lr_trans_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate transition learning rate
    lr_trans = lr_trans_base + (lr_trans_oci_slope * oci_score)
    if lr_trans < 0.0: lr_trans = 0.0
    if lr_trans > 1.0: lr_trans = 1.0

    # Initialize dynamic transition matrix (State x Action)
    # Start with standard priors: Action 0 -> State 0 (0.7), Action 1 -> State 1 (0.7)
    # Rows: Action (0, 1), Cols: Planet/State (0, 1)
    # Note: The template usually defines matrix as [State x Action] or similar. 
    # Here we define P(State | Action).
    # row 0: P(S=0|A=0), P(S=1|A=0)
    # row 1: P(S=0|A=1), P(S=1|A=1)
    trans_probs = np.array([[0.7, 0.3], [0.3, 0.7]]) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # MB Calculation using DYNAMIC transition matrix
        # trans_probs[0] is probs for Action 0, trans_probs[1] for Action 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Value of best action at State 0, State 1
        
        q_stage1_mb = np.zeros(2)
        q_stage1_mb[0] = np.dot(trans_probs[0], max_q_stage2) # Expected value of A0
        q_stage1_mb[1] = np.dot(trans_probs[1], max_q_stage2) # Expected value of A1
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Learning (Reward) ---
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate_reward * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate_reward * delta_stage1
        
        # --- Learning (Transitions) ---
        # Update probability of reaching 'state_idx' given 'a1'
        # State vector is one-hot: [1, 0] if state 0, [0, 1] if state 1
        observed_transition = np.zeros(2)
        observed_transition[state_idx] = 1.0
        
        # Update row corresponding to action taken
        # New_Prob = Old_Prob + lr * (Observed - Old_Prob)
        trans_probs[a1] += lr_trans * (observed_transition - trans_probs[a1])
        
        # Normalize to ensure sum to 1 (numerical stability)
        trans_probs[a1] /= np.sum(trans_probs[a1])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
