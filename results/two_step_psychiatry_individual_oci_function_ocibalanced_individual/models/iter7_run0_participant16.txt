Here are three new cognitive models that explore different mechanisms by which OCI scores might influence reinforcement learning in the two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance with Pruning
This model hypothesizes that high OCI scores might lead to "over-pruning" or a restricted search space. It modifies the weighting parameter `w` (balance between Model-Based and Model-Free systems) based on OCI, but specifically links it to a pruning threshold where low-value model-based branches are ignored if OCI is high (simulating a rigid focus).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated MB/MF Balance with Pruning.
    
    This model assumes that OCI affects the balance between model-based (MB) and 
    model-free (MF) control. Specifically, it tests if higher OCI leads to a 
    more rigid (MF-dominant) or more calculated (MB-dominant) strategy.
    
    Parameters:
    lr: [0, 1] Learning rate for value updates.
    beta: [0, 10] Inverse temperature for softmax.
    w_base: [0, 1] Baseline weighting for Model-Based system (0=Pure MF, 1=Pure MB).
    oci_w_shift: [-1, 1] Shift in 'w' proportional to OCI score.
    """
    lr, beta, w_base, oci_w_shift = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free Q-values for stage 2 (State x Action)

    # Calculate individual w parameter based on OCI
    # Bounded between 0 and 1
    w = w_base + (oci_score * oci_w_shift)
    if w < 0: w = 0
    if w > 1: w = 1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based: Bellman equation using transition matrix and max stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning / Updates ---
        # Stage 2 Update (TD error)
        # RPE at second stage
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        
        # Stage 1 Update (TD(0))
        # RPE at first stage driven by the value of the state actually reached
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Learning Rate Asymmetry (Pos/Neg)
This model posits that OCI symptoms relate to how individuals process positive versus negative prediction errors. It allows for separate learning rates for better-than-expected outcomes (positive RPE) and worse-than-expected outcomes (negative RPE), with OCI modulating the sensitivity to negative outcomes (punishment/lack of reward).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Dependent Learning Rate Asymmetry.
    
    This model separates learning rates for positive and negative prediction errors.
    It hypothesizes that high OCI is associated with altered sensitivity to 
    negative feedback (loss/omission of reward).
    
    Parameters:
    lr_pos: [0, 1] Learning rate for positive prediction errors.
    lr_neg_base: [0, 1] Base learning rate for negative prediction errors.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF mixing weight.
    oci_neg_mod: [-1, 1] Modulation of negative learning rate by OCI.
    """
    lr_pos, lr_neg_base, beta, w, oci_neg_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective negative learning rate
    lr_neg = lr_neg_base + (oci_score * oci_neg_mod)
    if lr_neg < 0: lr_neg = 0
    if lr_neg > 1: lr_neg = 1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning with Asymmetry ---
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        curr_lr2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[state_idx, int(action_2[trial])] += curr_lr2 * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        curr_lr1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[int(action_1[trial])] += curr_lr1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Eligibility Trace (Lambda)
This model introduces an eligibility trace parameter ($\lambda$) for the temporal difference learning. $\lambda$ controls how much the Stage 2 reward updates the Stage 1 choice directly (bypassing the state value). The hypothesis is that OCI might correlate with how credit is assigned to past actionsâ€”either "over-assigning" credit to distant past actions (high $\lambda$) or failing to link sequences (low $\lambda$).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Eligibility Trace (Lambda).
    
    This model uses a TD(lambda) approach where the Stage 1 value is updated 
    partially by the Stage 2 prediction error. This links the final reward 
    directly back to the first choice. OCI modulates this 'eligibility' or 
    credit assignment strength.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF mixing weight.
    lambda_base: [0, 1] Base eligibility trace parameter.
    oci_lambda_mod: [-1, 1] Modulation of lambda by OCI.
    """
    lr, beta, w, lambda_base, oci_lambda_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate lambda based on OCI
    lam = lambda_base + (oci_score * oci_lambda_mod)
    if lam < 0: lam = 0
    if lam > 1: lam = 1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_current_state = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning with Eligibility Trace ---
        # 1. Calculate RPEs
        # RPE 1: Value of State 2 - Value of Action 1
        delta1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        
        # RPE 2: Reward - Value of Action 2
        delta2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        
        # 2. Update Stage 2
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta2
        
        # 3. Update Stage 1
        # Standard TD(0) part: lr * delta1
        # Eligibility trace part: lr * lambda * delta2 (propagating stage 2 error back)
        q_stage1_mf[int(action_1[trial])] += lr * (delta1 + lam * delta2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
