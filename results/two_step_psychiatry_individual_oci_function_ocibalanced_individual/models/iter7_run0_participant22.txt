Here are three new cognitive models for the two-step task that incorporate OCI-R scores to explain individual variability.

### Model 1: Hybrid MB/MF with Sigmoid OCI-Modulated Weighting
This model tests the hypothesis that OCI shifts the balance between Model-Based (goal-directed) and Model-Free (habitual) control in a non-linear way. By using a sigmoid function, it robustly maps the OCI score to the mixing weight $w$ within the [0, 1] bound, preventing the out-of-bounds issues of linear models.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with Sigmoid OCI-modulated Weighting.

    Hypothesis: OCI affects the balance between Model-Based (goal-directed) and 
    Model-Free (habitual) control. A sigmoid function maps OCI to the mixing 
    weight 'w', ensuring valid bounds [0,1] and allowing for a non-linear 
    transition between strategies.

    Parameters:
    - learning_rate: [0, 1] Update rate for MF values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w_logit_base: [-5, 5] Base log-odds of using MB over MF (at OCI=0).
    - w_logit_oci_slope: [-5, 5] Effect of OCI on the log-odds of MB.
    """
    learning_rate, beta, w_logit_base, w_logit_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Sigmoid calculation for w (mixing weight)
    logit = w_logit_base + (w_logit_oci_slope * current_oci)
    # Clip logit to prevent overflow
    logit = np.clip(logit, -20, 20) 
    w = 1.0 / (1.0 + np.exp(-logit))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # MB value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net value: Weighted mix of MB and MF
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        # Standard MF choice at stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updates ---
        # Stage 1 TD error (SARSA-style for stage 1)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 TD error
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace: Propagate stage 2 RPE back to stage 1 choice (Lambda=1)
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid Model with OCI-Modulated Loss Aversion
This model hypothesizes that high OCI individuals are hypersensitive to "failure" (missing a coin). Instead of treating a 0 outcome as neutral, the model treats it as a penalty whose magnitude scales with OCI. This drives stronger avoidance learning for unrewarded actions.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Loss Aversion (Zero-Outcome Penalty).

    Hypothesis: High OCI individuals are hypersensitive to missing rewards. 
    They perceive a 0-coin outcome not as neutral, but as a penalty (loss), 
    driving stronger avoidance learning.
    
    Effective Reward = Reward - (1 - Reward) * (Penalty_Slope * OCI)

    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based control.
    - zero_penalty_oci_slope: [0, 5] Magnitude of penalty for 0 outcomes per unit OCI.
    """
    learning_rate, beta, w, zero_penalty_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updates ---
        # Calculate Effective Reward
        # If r=1, eff=1. If r=0, eff = - (slope * oci)
        penalty = zero_penalty_oci_slope * current_oci
        r_effective = r - (1 - r) * penalty

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update stage 2 with effective reward
        delta_stage2 = r_effective - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace (TD(1) logic)
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Model with OCI-Modulated Lapse Rate
This model posits that OCI relates to anxiety-driven distraction or "lapses" in attention. The final choice policy is a mixture of the standard Softmax policy and a uniform random guess. The weight of the random component ($\epsilon$) is determined by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Lapse Rate (Epsilon-Greedy Mixture).

    Hypothesis: High OCI is associated with anxiety or distraction, leading to 
    random "lapses" in decision making. The policy is a mixture of Softmax 
    and uniform random choice, with the random weight (epsilon) scaling with OCI.

    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based control.
    - epsilon_base: [0, 1] Baseline lapse rate.
    - epsilon_oci_slope: [-1, 1] Change in lapse rate per unit OCI.
    """
    learning_rate, beta, w, epsilon_base, epsilon_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate epsilon (randomness weight)
    epsilon = epsilon_base + (epsilon_oci_slope * current_oci)
    epsilon = np.clip(epsilon, 0.0, 1.0 - 1e-5) # Ensure bounds

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        softmax_probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Mixture Policy: (1-eps)*Softmax + eps*Random
        mixture_probs_1 = (1 - epsilon) * softmax_probs_1 + epsilon * 0.5
        p_choice_1[trial] = mixture_probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        softmax_probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Mixture Policy for Stage 2
        mixture_probs_2 = (1 - epsilon) * softmax_probs_2 + epsilon * 0.5
        p_choice_2[trial] = mixture_probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace (TD(1))
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
