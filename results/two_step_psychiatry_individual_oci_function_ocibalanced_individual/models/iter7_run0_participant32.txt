Here are three new cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid MB/MF with OCI-modulated Learning Rate.
    
    Hypothesis: OCI scores correlate with the volatility or speed of value updating (learning rate).
    High OCI might reflect hyper-plasticity (over-adjusting to noise) or rigidity.
    This model adds a mixing weight 'w' and perseveration 'pers' as fixed controls,
    while modulating the global 'learning_rate' by OCI.

    Parameters:
    lr_base: [0, 1] Base learning rate.
    lr_oci_sens: [-1, 1] Effect of OCI on learning rate (additive).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (0=MF, 1=MB).
    pers: [-2, 2] General perseveration bonus.
    """
    lr_base, lr_oci_sens, beta, w, pers = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective learning rate and clip to valid bounds [0, 1]
    effective_lr = lr_base + (lr_oci_sens * oci_score)
    effective_lr = np.clip(effective_lr, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State (Planet), Action (Alien)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Perseveration bonus
        if last_action_1 != -1:
            q_net[int(last_action_1)] += pers

        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = action_1[trial]
        state_idx = int(state[trial])
        
        # --- Stage 2 Decision ---
        # Standard MF choice at stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        # Stage 1 MF Update (TD(0))
        curr_q2 = q_stage2_mf[state_idx, int(action_2[trial])]
        pe_1 = curr_q2 - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += effective_lr * pe_1
        
        # Stage 2 MF Update
        pe_2 = reward[trial] - curr_q2
        q_stage2_mf[state_idx, int(action_2[trial])] += effective_lr * pe_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Hybrid MB/MF with Compulsive Unrewarded Perseveration.
    
    Hypothesis: High OCI is associated with 'compulsive' repetition of actions 
    specifically when they fail (lack of reward), reflecting an inability to shift set.
    Perseveration is split: Fixed 'Win-Stay' (pers_rew) vs OCI-modulated 'Lose-Stay' (pers_loss_oci).

    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    pers_rew: [-2, 2] Perseveration bonus after a rewarded trial.
    pers_loss_oci_slope: [-5, 5] Perseveration bonus after an unrewarded trial, scaled by OCI.
    """
    learning_rate, beta, w, pers_rew, pers_loss_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    last_reward = 0 # Assume neutral start

    for trial in range(n_trials):
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Conditional Perseveration
        if last_action_1 != -1:
            if last_reward == 1:
                q_net[int(last_action_1)] += pers_rew
            else:
                # Unrewarded previous trial: OCI modulates tendency to stay
                q_net[int(last_action_1)] += (pers_loss_oci_slope * oci_score)

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = action_1[trial]
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        curr_q2 = q_stage2_mf[state_idx, int(action_2[trial])]
        pe_1 = curr_q2 - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * pe_1
        
        pe_2 = reward[trial] - curr_q2
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * pe_2
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Hybrid MB/MF with Subjective Loss Sensitivity.
    
    Hypothesis: High OCI participants treat the absence of reward (0 coins) not as neutral, 
    but as a negative outcome (loss/punishment), potentially driving avoidance learning.
    The effective reward for 0 outcomes is -1 * oci * sensitivity.

    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    pers: [-2, 2] Perseveration bonus.
    loss_sens_oci: [0, 5] Magnitude of subjective loss for 0 rewards, scaled by OCI.
    """
    learning_rate, beta, w, pers, loss_sens_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    last_action_1 = -1

    for trial in range(n_trials):
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[int(last_action_1)] += pers

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = action_1[trial]
        state_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Calculate Effective Reward
        raw_reward = reward[trial]
        if raw_reward == 0:
            # Transform 0 reward into a negative value based on OCI
            effective_reward = -1.0 * loss_sens_oci * oci_score
        else:
            effective_reward = raw_reward
        
        # Updates
        curr_q2 = q_stage2_mf[state_idx, int(action_2[trial])]
        pe_1 = curr_q2 - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * pe_1
        
        pe_2 = effective_reward - curr_q2
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * pe_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
