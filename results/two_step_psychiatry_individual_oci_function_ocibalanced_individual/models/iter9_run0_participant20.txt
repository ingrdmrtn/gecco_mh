Here are three new cognitive models exploring different mechanisms by which OCI scores might influence decision-making in the two-step task, focusing on learning rate asymmetry, eligibility traces, and transition structure learning.

### Model 1: Asymmetric Learning Rates Modulated by OCI
This model hypothesizes that individuals with higher OCI scores might exhibit a negativity bias in learning. Specifically, they may update their value estimates more strongly following a lack of reward (loss/omission) compared to a reward, reflecting a heightened sensitivity to failure or error common in obsessive-compulsive traits.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Asymmetric Learning Rates Modulated by OCI.
    
    Hypothesis: High OCI participants have a heightened sensitivity to negative outcomes.
    This is modeled by splitting the learning rate into alpha_pos (for rewards) and 
    alpha_neg (for omission of rewards). The alpha_neg is modulated by the OCI score.
    
    Parameters:
    alpha_pos: [0, 1] - Learning rate for positive prediction errors (rewards).
    alpha_neg_base: [0, 1] - Baseline learning rate for negative prediction errors.
    alpha_neg_oci_slope: [0, 1] - How much OCI increases the negative learning rate.
    beta: [0, 10] - Inverse temperature (softmax sensitivity).
    w: [0, 1] - Weighting parameter (0 = pure Model-Free, 1 = pure Model-Based).
    """
    alpha_pos, alpha_neg_base, alpha_neg_oci_slope, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate the OCI-modulated negative learning rate, capped at 1.0
    alpha_neg = min(1.0, alpha_neg_base + alpha_neg_oci_slope * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value integration
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Learning ---
        # Stage 1 Update (SARSA-style TD(0) for MF)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Use alpha_pos if delta is positive, alpha_neg if negative
        lr_s1 = alpha_pos if delta_stage1 > 0 else alpha_neg
        q_stage1_mf[a1] += lr_s1 * delta_stage1

        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        # Use alpha_pos if delta is positive, alpha_neg if negative
        lr_s2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, a2] += lr_s2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Eligibility Trace Decay Modulated by OCI
This model investigates the "stickiness" of credit assignment. In reinforcement learning, an eligibility trace ($\lambda$) determines how much credit previous states receive for current rewards. A higher $\lambda$ links the first-stage choice more strongly to the final outcome. We hypothesize that high OCI (associated with rumination or over-thinking causal chains) leads to a higher eligibility trace parameter, effectively making the "Model-Free" system behave slightly more like a "Model-Based" system by connecting distal actions to rewards more robustly.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Eligibility Trace Decay Modulated by OCI.
    
    Hypothesis: OCI scores modulate the eligibility trace parameter (lambda) in a 
    TD(lambda) reinforcement learning framework. Higher OCI leads to higher lambda,
    implying stronger credit assignment to the first-stage choice based on the second-stage outcome.
    
    Parameters:
    learning_rate: [0, 1] - Standard learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weighting.
    lambda_base: [0, 1] - Baseline eligibility trace decay.
    lambda_oci_slope: [0, 1] - How much OCI increases the eligibility trace.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate OCI-modulated lambda, capped at 1.0
    lambda_param = min(1.0, lambda_base + lambda_oci_slope * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Learning with Eligibility Trace ---
        # 1. Prediction error at stage 2 (Reward - Q_stage2)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # 2. Prediction error at stage 1 (Q_stage2 - Q_stage1)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 2 Q-value
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-value: 
        # Standard TD(0) part: learning_rate * delta_stage1
        # PLUS Eligibility trace part: learning_rate * lambda * delta_stage2
        # This allows the stage 1 value to be updated directly by the stage 2 reward error
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Subjective Transition Probability Distortion
Standard Model-Based RL assumes participants use the true transition matrix (0.7/0.3). However, anxiety and OC traits can lead to an overestimation of rare/risky events or a distrust of "common" safety. This model proposes that high OCI participants have a distorted internal model of the transition probabilities, perceiving the "common" transition as less likely than it actually is (flattening the matrix towards 0.5/0.5), which degrades the effectiveness of their Model-Based planning.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Subjective Transition Probability Distortion.
    
    Hypothesis: High OCI participants distrust the stability of the environment.
    Instead of using the objective 0.7/0.3 transition matrix for Model-Based planning,
    they use a distorted matrix where the 'common' probability is reduced based on their OCI score.
    This reflects 'uncertainty intolerance' or doubt.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for MF values.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weighting.
    distortion_oci_slope: [0, 1] - How much OCI reduces the perceived common transition probability.
    """
    learning_rate, beta, w, distortion_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Objective common probability
    true_common_prob = 0.7
    
    # Subjective common probability: 
    # Starts at 0.7, decreases as OCI increases. 
    # Lower bound clamped at 0.5 (randomness), preventing it from flipping (becoming rare).
    distortion = distortion_oci_slope * oci_score
    subjective_p = max(0.5, true_common_prob - distortion)
    
    # Construct subjective transition matrix for MB planning
    # If subjective_p is 0.6, matrix is [[0.6, 0.4], [0.4, 0.6]]
    subjective_trans_matrix = np.array([
        [subjective_p, 1 - subjective_p], 
        [1 - subjective_p, subjective_p]
    ])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # CRITICAL: Use subjective matrix for MB calculation
        q_stage1_mb = subjective_trans_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Learning ---
        # Standard MF updates (learning is based on experience, not the distorted matrix)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
