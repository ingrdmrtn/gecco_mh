import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model-Free RL with OCI-modulated Perseveration (Stickiness).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_sticky_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_sticky_slope = parameters
    
    # Handle OCI input (if provided as a list/array, take the first element, otherwise use directly)
    if hasattr(oci, '__len__'):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0.0

    # Calculate the effective stickiness based on OCI
    stickiness = stickiness_base + (oci_score * oci_sticky_slope)

    # Initialize Q-values
    # q_stage1: Q-values for the two actions in the first stage (State 0)
    q_stage1 = np.zeros(2)
    # q_stage2: Q-values for the two actions in the two possible second-stage states
    # Rows are states (0 or 1), columns are actions (0 or 1)
    q_stage2 = np.zeros((2, 2))

    # Transition matrix (fixed for the standard 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_modulated = q_stage1.copy()
        
        # Apply stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_stage1_modulated[last_action_1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modulated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (using Q-values for the specific state arrived at)
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        r = int(rng.random() < current_reward_probs[s_idx][a2])

        # --- Learning (Model-Free Updates) ---
        
        # 1. Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2

        # 2. Update Stage 1 Q-values (TD Error based on Stage 2 value)
        # Note: The fitting code uses q_stage2[s_idx, a2] as the target for stage 1 update.
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
