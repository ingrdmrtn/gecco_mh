import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-modulated Choice Stickiness model.
    
    This model is purely Model-Based (MB) in the first stage regarding value estimation,
    but adds a stickiness bonus to the previously chosen action. The magnitude of this
    bonus is modulated by the participant's OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stickiness_base, stickiness_oci = parameters
    
    # Handle OCI input safely
    oci_score = oci if oci is not None else 0.0

    # Calculate effective stickiness based on OCI
    eff_stickiness = stickiness_base + (stickiness_oci * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values for Stage 2 (Model-Free)
    # Shape: (2 states, 2 actions) corresponding to the two second-stage states
    q_stage2_mf = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values for Stage 1
        # Q_MB(a1) = sum(P(s2|a1) * max(Q_MF(s2, :)))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Apply Stickiness
        q_net = q_stage1_mb.copy()
        if prev_action_1 != -1:
            q_net[prev_action_1] += eff_stickiness
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 using Model-Free Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2_idx][a2])
        
        # --- Learning ---
        
        # Update Stage 2 Q-values (Model-Free update)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
