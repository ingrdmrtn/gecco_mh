import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-Modulated Rewarded Stickiness (Compulsive Win-Stay).

    This model adds a "stickiness" bonus to the choice logits at Stage 1,
    but ONLY if the previous trial was rewarded. The magnitude of this
    conditional stickiness is modulated by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, stick_rew_base, oci_stick_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, w, stick_rew_base, oci_stick_mod = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Handle OCI input (if it's an array/list, take first element, otherwise use as is)
    if isinstance(oci, (list, np.ndarray)):
        oci_val = oci[0]
    else:
        oci_val = oci if oci is not None else 0.0

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Model variables
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: State 1 (states 2/3 mapped to indices 0/1), Cols: Actions

    stickiness_bonus = stick_rew_base + (oci_val * oci_stick_mod)
    
    prev_choice_1 = -1
    prev_reward = 0

    for t in range(n_trials):
        # Current trial reward probabilities
        # drift1/2 correspond to State 1 (index 0 in q_stage2_mf), drift3/4 to State 2 (index 1)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits = beta * q_net
        
        # Apply OCI-modulated Rewarded Stickiness
        if prev_choice_1 != -1 and prev_reward == 1.0:
            logits[prev_choice_1] += stickiness_bonus

        # Softmax for Stage 1
        exp_q1 = np.exp(logits)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=p_stage1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State 0 (common for a1=0) or State 1 (common for a1=1)
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        qs_current_state = q_stage2_mf[s2_idx]
        exp_q2 = np.exp(beta * qs_current_state)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=p_stage2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning ---
        
        # Stage 2 update (TD)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += lr * delta_stage2
        
        # Stage 1 update (TD)
        # Note: The fitting code uses q_stage2_mf[state_idx, action_2] as the target for stage 1
        # which effectively is SARSA(0) or similar logic where the value of the chosen state-action pair is used.
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

        # Update history for stickiness
        prev_choice_1 = a1
        prev_reward = r

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx # Storing 0 or 1
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
