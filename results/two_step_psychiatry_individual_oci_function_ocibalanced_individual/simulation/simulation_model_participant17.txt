import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Model-Free Learner with OCI-modulated Choice Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_stickiness_boost]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_stickiness_boost = parameters
    
    # Handle OCI input (default to 0 if None)
    oci_score = oci if oci is not None else 0.0
    
    # Calculate effective stickiness based on OCI
    stickiness_eff = stickiness_base + (oci_stickiness_boost * oci_score)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states, 2 actions
    q_stage2_mf = np.zeros((2, 2))

    # Transition matrix (Common: 0.7, Rare: 0.3)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    last_action_1 = -1

    for t in range(n_trials):
        # --- Stage 1 Choice ---
        q_stage1_with_stick = q_stage1_mf.copy()
        
        # Apply stickiness to the previously chosen action
        if last_action_1 != -1:
            q_stage1_with_stick[last_action_1] += stickiness_eff
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_with_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition to Stage 2 ---
        # Determine state 2 based on transition matrix probabilities
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Choice ---
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        # Map drifts to specific state-action pairs
        # State 0 corresponds to drift1 (action 0) and drift2 (action 1)
        # State 1 corresponds to drift3 (action 0) and drift4 (action 1)
        if s2 == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
        
        r = 1 if np.random.random() < current_reward_prob else 0

        # --- Learning Updates ---
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values
        # Note: This model uses the updated Q-value of the chosen stage 2 action (SARSA style)
        # as the target for the stage 1 update, rather than the max (Q-learning).
        v_stage2_arrival = q_stage2_mf[s2, a2]
        delta_stage1 = v_stage2_arrival - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
