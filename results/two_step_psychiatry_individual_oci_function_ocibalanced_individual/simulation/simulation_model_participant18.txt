import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the hybrid learner where beta is modulated by OCI.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, beta_oci_scale, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None/list, handled appropriately.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize Random Number Generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta_base, beta_oci_scale, w, stickiness = parameters
    
    # Handle OCI input (can be passed as a single float, a list, or None)
    if oci is None:
        oci_score = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate effective beta based on OCI
    eff_beta = beta_base + (beta_oci_scale * oci_score)
    eff_beta = np.maximum(eff_beta, 0.0) 

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 (model-free) -> shape (2,)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 (model-free) -> shape (2, 2) representing [state_idx, action]
    # Note: In fitting code, states are 0 and 1.
    q_stage2_mf = np.zeros((2, 2))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities for state 0 (drift1, drift2) and state 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based value
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness (perseveration)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(eff_beta * q_net)
        # Handle potential overflow/underflow for stability
        if np.sum(exp_q1) == 0:
             probs_1 = np.array([0.5, 0.5])
        else:
             probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (purely model-free based on Q-values of that state)
        exp_q2 = np.exp(eff_beta * q_stage2_mf[s_idx, :])
        if np.sum(exp_q2) == 0:
            probs_2 = np.array([0.5, 0.5])
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Generate reward based on drift probabilities
        r = int(rng.random() < reward_probs[s_idx][a2])

        # --- Learning (Update Q-values) ---
        
        # TD(0) update for Stage 1 (SARSA logic: Q(s1,a1) updated by Q(s2,a2))
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
