import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Reinforcement Learning with OCI-modulated Choice Stickiness model.

    This model includes a stickiness parameter that biases the first-stage choice towards the 
    previously chosen action, with the magnitude of this bias modulated by the participant's OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stick_param]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stick_param = parameters
    
    # Handle OCI input
    current_oci = oci if oci is not None else 0.0
    
    # Define transition matrix (standard 70/30 structure for 2-step task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1 corresponds to the two actions in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two states (rows) and two actions (cols) in the second stage
    q_stage2 = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Calculate effective stickiness based on OCI
    stick_eff = stick_param * (1.0 + current_oci)
    
    prev_action_1 = -1 

    for t in range(n_trials):
        # Current trial reward probabilities for state 0 (drift1, drift2) and state 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_net = q_stage1.copy()
        
        # Add stickiness bonus if there was a previous action
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick_eff
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # No stickiness applied to Stage 2 in the fitting code provided
        q_s2 = q_stage2[s2]
        exp_q2 = np.exp(beta * q_s2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        
        # Update Stage 2 Q-values (SARSA / Q-learning logic)
        delta2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta2

        # Update Stage 1 Q-values
        # The fitting code uses v_state_reached = max(q_stage2[state_idx])
        # This implies a TD(0) update using the max value of the next state (Q-learning style update for stage 1)
        v_state_reached = np.max(q_stage2[s2])
        delta1 = v_state_reached - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
