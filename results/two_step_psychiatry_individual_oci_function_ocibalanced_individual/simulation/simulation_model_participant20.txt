import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Hybrid RL with OCI-modulated choice perseveration.
    
    Hypothesis: Higher OCI scores lead to increased 'stickiness' or perseveration 
    in the first-stage choice.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, w, stickiness_oci = parameters
    
    # Handle OCI input (can be passed as a single float or a list/array)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0

    stickiness_bonus = stickiness_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: states (0 or 1), Cols: actions
    
    last_action_1 = -1 # Indicator for the previous action

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for trial in range(n_trials):
        # Current trial reward probabilities for the two states
        reward_probs = [[drift1[trial], drift2[trial]], 
                        [drift3[trial], drift4[trial]]]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness bonus if there was a previous action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_bonus

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State 0 corresponds to matrix row 0, State 1 to row 1
        s2_probs = transition_matrix[a1]
        # The logic in the fitting code implies state is 0 or 1.
        # The transition matrix maps action -> probability of landing in state 0 vs 1
        # Specifically: transition_matrix[a1] gives [P(s2=0|a1), P(s2=1|a1)]
        s2 = rng.choice([0, 1], p=s2_probs)
        state_idx = s2 

        # --- Stage 2 Decision ---
        qs_stage2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[state_idx][a2])

        # --- Learning ---
        # 1. Update Stage 1 MF Q-value using TD error from transition
        # delta_stage1 = V(s') - Q(s, a)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 MF Q-value using Reward Prediction Error
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Store history
        stage1_choice[trial] = a1
        state2[trial] = s2
        stage2_choice[trial] = a2
        reward[trial] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
