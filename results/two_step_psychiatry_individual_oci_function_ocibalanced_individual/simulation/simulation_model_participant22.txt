import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model-Free RL with OCI-modulated Choice Autocorrelation.
    
    Hypothesis: High OCI leads to repetitive behavior (perseveration). This model
    adds a "stickiness" bonus directly to the choice logits of the previously
    selected action, with the strength of this bonus determined by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_oci_slope, stickiness_intercept]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_oci_slope, stickiness_intercept = parameters
    
    # Handle OCI input (default to 0 if None, though the prompt implies it's required)
    current_oci = oci if oci is not None else 0.0
    
    # Calculate the stickiness phi value based on OCI
    phi = stickiness_intercept + (current_oci * stickiness_oci_slope)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1 is for the two actions in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 is for the two states (rows) and two actions (cols) in the second stage
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed for the 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    prev_action_1 = -1 # Indicator for previous action
    rng = np.random.default_rng() # Random number generator

    for t in range(n_trials):
        # 1. Stage 1 Choice
        logits_1 = beta * q_stage1
        
        # Add stickiness bonus if there was a previous action
        if prev_action_1 != -1:
            logits_1[prev_action_1] += phi
            
        # Softmax for Stage 1
        logits_1 = logits_1 - np.max(logits_1) # Numerical stability
        exp_logits_1 = np.exp(logits_1)
        probs_1 = exp_logits_1 / np.sum(exp_logits_1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Calculate logits for the specific state arrived at
        logits_2 = beta * q_stage2[s2]
        
        # Softmax for Stage 2
        logits_2 = logits_2 - np.max(logits_2)
        exp_logits_2 = np.exp(logits_2)
        probs_2 = exp_logits_2 / np.sum(exp_logits_2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Get current trial's reward probabilities for the specific state/action
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Value Updates (Model-Free TD Learning)
        # Update Stage 2 Q-values
        # Prediction Error 2: Reward - Q2(s2, a2)
        delta2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta2
        
        # Update Stage 1 Q-values
        # Note: The fitting code uses SARSA-like update: Q2(s2, a2) - Q1(a1)
        # It does not use max(Q2) or eligibility traces (lambda) based on the provided fitting code.
        delta1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous action for stickiness in next trial
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
