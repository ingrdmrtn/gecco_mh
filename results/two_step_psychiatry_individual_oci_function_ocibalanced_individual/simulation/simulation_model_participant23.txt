import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid RL model with Choice Stickiness 
    (Perseveration) modulated by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness_factor = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, take first element, else use float)
    if oci is None:
        current_oci = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        current_oci = oci[0]
    else:
        current_oci = oci

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values and transition matrix
    # q_stage1_mf: values for Stage 1 actions (0 and 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: values for Stage 2 states (rows) and actions (cols)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Calculate stickiness magnitude based on OCI
    phi = stickiness_factor * current_oci 
    
    last_action_1 = -1 # Initialize with no previous action

    for t in range(n_trials):
        # Current trial reward probabilities for the two states
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Hybrid)
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # Apply Stickiness (Perseveration)
        q_net_sticky = q_net.copy()
        if last_action_1 != -1:
            q_net_sticky[last_action_1] += phi

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1

        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State is 0 or 1 (mapped to indices of q_stage2_mf)
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2 (pure Model-Free)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        r = int(rng.random() < reward_probs[state_idx][a2])

        # --- Learning ---
        
        # Stage 1 TD update
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 TD update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
