import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: Hybrid Model-Based/Model-Free RL 
    with OCI-modulated mixing weight.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_base, w_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (list or np.ndarray): Obsessive-Compulsive Inventory score (expected as a list/array, uses first element).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w_base, w_oci_slope = parameters
    
    # Handle OCI input (fitting code treats it as an array/list where index 0 is the score)
    if oci is not None:
        if isinstance(oci, (list, np.ndarray)):
             oci_score = oci[0]
        else:
             oci_score = oci
    else:
        oci_score = 0.0

    # Calculate the mixing weight based on OCI
    w = w_base + (w_oci_slope * oci_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Stage 2 (2 states, 2 actions)
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities for the 4 terminal states
        # State 0 (Planet A): actions 0, 1 correspond to drift1, drift2
        # State 1 (Planet B): actions 0, 1 correspond to drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        # Max Q-value for each second-stage state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # Softmax choice
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix probabilities
        # transition_matrix[a1] gives [p(s2=0|a1), p(s2=1|a1)]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        qs_current_state = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Stochastic reward outcome based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        
        # Update Stage 2 Q-values (SARSA / TD prediction error)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD(0) style update based on Stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target, 
        # effectively treating the value of the chosen next state/action as the reward proxy.
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
