import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Reinforcement Learning model 
    with OCI-modulated Choice Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stick_base, stick_oci_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, stick_base, stick_oci_mod = parameters
    
    # Handle OCI input (default to 0 if None, though fit code assumes it exists)
    current_oci = oci if oci is not None else 0.0
    
    # Calculate effective stickiness based on OCI
    stick_effective = stick_base + (stick_oci_mod * current_oci)

    # Transition matrix (fixed 70/30 structure standard in 2-step tasks)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (indices 0 and 1), 2 actions each
    q_stage2_mf = np.zeros((2, 2))

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_a1 = -1  # No previous action for the first trial

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        qs = q_stage1_mf.copy()
        
        # Apply stickiness bonus to the previously chosen action
        if last_a1 != -1:
            qs[last_a1] += stick_effective
            
        # Softmax calculation for Stage 1
        exp_q1 = np.exp(beta * qs)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=p_choice_1)
        last_a1 = a1 # Update for next trial stickiness

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        # Note: fitting code uses 'state' as indices 0 or 1 directly for q_stage2_mf
        state_idx = s2 

        # --- Stage 2 Decision ---
        # Softmax calculation for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice
        a2 = rng.choice([0, 1], p=p_choice_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning (Model-Free TD) ---
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (TD(1) logic: r propagates back to stage 1 via stage 2 value)
        # The fitting code uses: delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Note: In the fitting code, q_stage2_mf[state_idx, a2] is used *after* the update.
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
