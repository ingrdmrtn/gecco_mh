import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: OCI-Modulated Model-Based/Model-Free Hybrid.
    
    This model assumes the balance between Model-Based (MB) and Model-Free (MF) 
    strategies is modulated by the OCI score. A base mixing weight (w_base) is 
    adjusted by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_base, oci_sensitivity]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.
        oci (float, list, or np.ndarray): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w_base, oci_sensitivity = parameters
    
    # Handle OCI input format (scalar or array)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate the OCI-modulated mixing weight
    # Note: Logic matches fitting code: w = w_base - (sensitivity * (oci - 0.5))
    raw_w = w_base - (oci_sensitivity * (oci_score - 0.5))
    w = np.clip(raw_w, 0.0, 1.0)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (Model-Free)
    q_stage1_mf = np.zeros(2)      
    # q_stage2_mf: Q-values for stage 2 states and actions (Model-Free)
    # Shape (2, 2) corresponds to [state_index, action_index]
    q_stage2_mf = np.zeros((2, 2)) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 (index 0) corresponds to drifts 1 & 2
        # State 1 (index 1) corresponds to drifts 3 & 4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        # Note: In fitting code, state is usually 0 or 1. 
        # Here s2=0 is state 1, s2=1 is state 2.
        
        # --- Stage 2 Decision ---
        
        qs_current_state = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        
        r = int(rng.random() < current_drifts[s2][a2])

        # --- Learning (Update Q-values) ---
        
        # Update Stage 1 MF value (SARSA-like update using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 MF value (Prediction Error using reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
