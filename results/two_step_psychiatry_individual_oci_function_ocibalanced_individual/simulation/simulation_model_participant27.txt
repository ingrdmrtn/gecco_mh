import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using the Pure Model-Based Agent with OCI-modulated Stickiness.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stick_base, stick_oci_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, stick_base, stick_oci_sens = parameters
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Handle OCI input (scalar or single-element list/array)
    oci_score = oci[0] if isinstance(oci, (list, np.ndarray)) else oci

    # Calculate constant stickiness magnitude based on OCI
    stickiness_mag = stick_base + (stick_oci_sens * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values for Stage 2 (Model-Free)
    # Rows: State 0 (Drift 1/2), State 1 (Drift 3/4)
    # Cols: Action 0, Action 1
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation:
        # Project the max value of the next stage states using the transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Stickiness (Perseveration)
        stick_vector = np.zeros(2)
        if prev_action_1 != -1:
            stick_vector[prev_action_1] = stickiness_mag
            
        # Logits and Softmax
        logits_1 = beta * q_stage1_mb + stick_vector
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Model-Free decision based on current state Q-values
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Get reward probability for the specific state-action pair
        # s2=0 corresponds to drift1/drift2, s2=1 corresponds to drift3/drift4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        prob_reward = current_drifts[s2][a2]
        
        r = int(rng.random() < prob_reward)
        
        # --- Learning ---
        
        # Update Stage 2 Model-Free Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Note: Stage 1 values are not updated via Q-learning in this "Pure Model-Based" formulation;
        # they are re-calculated every step based on the evolving Stage 2 values.

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
