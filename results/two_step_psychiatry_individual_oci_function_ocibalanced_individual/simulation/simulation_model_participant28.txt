import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the hybrid RL model where the balance between 
    Model-Based (MB) and Model-Free (MF) control is modulated by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w_base, w_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (list or np.ndarray): Obsessive-Compulsive Inventoryâ€“Revised score (expects a list/array with at least one value).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, w_base, w_oci_slope = parameters
    
    # Handle OCI input (fitting code accesses oci[0])
    if oci is None:
        oci_score = 0.0 # Default if not provided, though model expects it
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate the mixing weight based on OCI
    w = w_base + w_oci_slope * (oci_score - 0.5)
    w = np.clip(w, 0.0, 1.0)

    # Setup transition matrix
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: MF values for stage 1 actions (2 actions)
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf: MF values for stage 2 (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2)) 

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (pure MF based on fitting code logic)
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        # Note: The fitting code uses simple TD(0) updates for both stages independently based on the final reward r.
        # It does not explicitly model the Stage 1 -> Stage 2 TD error (delta1) or eligibility traces (lambda).
        
        # Update Stage 2 Q-values
        pe_2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * pe_2

        # Update Stage 1 MF Q-values (direct reinforcement from outcome)
        pe_1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * pe_1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
