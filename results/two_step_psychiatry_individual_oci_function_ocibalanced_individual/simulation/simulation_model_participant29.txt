import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: Hybrid MB/MF RL with OCI-modulated mixing weight.
    
    The balance between Model-Based (planning) and Model-Free (habit) systems is determined by 'w',
    which is a function of the OCI score. It uses a single learning rate for both stages and
    TD(1) logic (updating stage 1 based on the sum of prediction errors).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness, w_base, w_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness, w_base, w_oci_slope = parameters
    
    # Handle OCI input
    current_oci = oci if oci is not None else 0.0
    
    # Calculate mixing weight w based on OCI
    w = w_base + (w_oci_slope * current_oci)
    w = np.clip(w, 0.0, 1.0)

    # Transition matrix (fixed)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 choices (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Add stickiness (perseveration)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 (purely MF based on state Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < current_reward_probs[s2_idx][a2])

        # --- Learning / Update ---
        
        # Prediction errors
        # Note: The fitting code updates Stage 1 using (delta_stage1 + delta_stage2)
        # This implies TD(1) or an eligibility trace of 1.0 for the first stage.
        
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]

        # Update Stage 2 Q-values
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
