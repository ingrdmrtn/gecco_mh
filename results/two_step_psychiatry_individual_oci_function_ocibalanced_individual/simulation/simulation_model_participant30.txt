import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: OCI-modulated Model-Based vs. Model-Free weighting.
    
    Hypothesis: High OCI participants rely more on Model-Free (habitual) learning
    and less on Model-Based (goal-directed) planning. The parameter `w` (mixing weight)
    is scaled down by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w_base = parameters
    
    # Handle OCI input safely
    current_oci = oci if oci is not None else 0.0

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize values
    # q_stage1_mf: Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2) 
    
    # q_stage2_mf: Q-values for the two states in stage 2, each having 2 actions
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2)) 

    # Fixed transition matrix as defined in the fitting code
    # Rows: Stage 1 actions (0, 1), Cols: Probability of Stage 2 states (0, 1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Calculate effective weight based on OCI modulation
    w_effective = w_base * (1.0 - current_oci)

    for t in range(n_trials):
        # Current trial reward probabilities (organized by state index 0 and 1)
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values for Stage 1
        # Max value achievable in each stage 2 state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-value is a weighted mix of MB and MF values
        q_net = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf

        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        
        # Determine second stage state based on transition probabilities
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 based on MF values for that state
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        
        # Determine reward outcome based on drift probabilities
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning Updates ---

        # 1. Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # 2. Update Stage 1 MF Q-values (TD Error based on Stage 2 value)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target,
        # effectively SARSA(0) logic for the first stage update.
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
