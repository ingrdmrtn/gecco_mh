import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Hybrid MB/MF with Compulsive Unrewarded Perseveration.
    
    Hypothesis: High OCI is associated with 'compulsive' repetition of actions 
    specifically when they fail (lack of reward).
    Perseveration is split: Fixed 'Win-Stay' (pers_rew) vs OCI-modulated 'Lose-Stay' (pers_loss_oci).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, pers_rew, pers_loss_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, pers_rew, pers_loss_oci_slope = parameters
    
    # Handle OCI input (default to 0 if not provided, though logic implies it's necessary)
    oci_score = oci if oci is not None else 0.0

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Model-free Q-values for stage 1 (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free Q-values for stage 2 (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize tracking variables for perseveration
    last_action_1 = -1
    last_reward = 0  # Assume neutral start
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Net Q-values for Stage 1
        
        # Model-Based calculation: V(s') = max(Q_MF(s', a'))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid mix
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Perseveration Bonus
        if last_action_1 != -1:
            if last_reward == 1:
                # Fixed Win-Stay bonus
                q_net[int(last_action_1)] += pers_rew
            else:
                # OCI-modulated Lose-Stay bonus
                q_net[int(last_action_1)] += (pers_loss_oci_slope * oci_score)

        # 2. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # State transitions are probabilistic based on the transition matrix
        # State 0 (Common for a1=0), State 1 (Common for a1=1)
        # Note: The fitting code uses `state` as input, here we generate it.
        # In the fitting code, state indices are usually 0 and 1.
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # 4. Stage 2 Choice (Softmax based on Q_MF only)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Outcome
        # Get current reward probabilities for the specific state
        # Mapping: State 0 uses drift1/drift2, State 1 uses drift3/drift4
        if s2 == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)
        
        # 6. Learning (SARSA / TD)
        
        # Stage 1 Update: TD(1) style without lambda in this specific model code
        # The fitting code updates stage 1 based on the value of the chosen stage 2 action (SARSA-like)
        curr_q2 = q_stage2_mf[s2, a2]
        pe_1 = curr_q2 - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1
        
        # Stage 2 Update
        pe_2 = r - curr_q2
        q_stage2_mf[s2, a2] += learning_rate * pe_2
        
        # 7. Update history for next trial
        last_action_1 = a1
        last_reward = r
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
