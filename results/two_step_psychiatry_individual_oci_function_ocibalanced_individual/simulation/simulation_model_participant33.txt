import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 3: OCI-modulated Learning Rate Asymmetry.
    
    This model assumes that OCI scores affect how strongly the agent learns from 
    negative prediction errors (worse than expected outcomes). High OCI acts as a 
    multiplier on the learning rate when the prediction error is negative.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, beta, lambda_eligibility, oci_neg_bias]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    alpha_pos, beta, lambda_eligibility, oci_neg_bias = parameters
    
    # Handle OCI input (if provided as a single float or list, ensure scalar)
    if oci is None:
        oci_score = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Q-values initialization
    # q_stage1_mf: Q-values for stage 1 actions (0 and 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states (0 and 1) and actions (0 and 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (standard 2-step task structure)
    # Row 0: State 1 (Common -> State 2, Rare -> State 3)
    # Row 1: State 1 (Common -> State 3, Rare -> State 2)
    # Note: The fitting code uses 'state' which usually maps 0->State B, 1->State C.
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine specific state index (0 or 1) based on transition probabilities
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drift arrays to the specific state/action
        # State index 0 corresponds to drifts 1 and 2
        # State index 1 corresponds to drifts 3 and 4
        if s2_idx == 0:
            current_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_prob)
        
        # 5. Learning / Updates
        
        # Calculate Prediction Errors
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        
        # Determine effective learning rate for Stage 2 based on PE sign and OCI
        if delta_stage2 < 0:
            alpha_s2 = alpha_pos * (1.0 + (oci_neg_bias * oci_score))
            if alpha_s2 > 1.0: alpha_s2 = 1.0
        else:
            alpha_s2 = alpha_pos

        # Determine effective learning rate for Stage 1 based on PE sign and OCI
        if delta_stage1 < 0:
            alpha_s1 = alpha_pos * (1.0 + (oci_neg_bias * oci_score))
            if alpha_s1 > 1.0: alpha_s1 = 1.0
        else:
            alpha_s1 = alpha_pos
            
        # Update Q-values
        q_stage2_mf[s2_idx, a2] += alpha_s2 * delta_stage2
        
        # Update Stage 1 Q-values with eligibility trace from Stage 2 error
        q_stage1_mf[a1] += alpha_s1 * delta_stage1 + (alpha_s1 * lambda_eligibility * delta_stage2)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
