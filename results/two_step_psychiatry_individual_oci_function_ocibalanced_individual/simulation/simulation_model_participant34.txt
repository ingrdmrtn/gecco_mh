import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model-Free Q-learning with an Uncertainty/Novelty Bonus
    modulated by OCI.

    Model Logic:
    - Q-values track expected rewards.
    - An uncertainty trace tracks time since an option was last chosen.
    - Uncertainty decays over time (decay_rate).
    - Choice probability is influenced by Q-values + (uncertainty_weight * uncertainty).
    - uncertainty_weight is derived from OCI: -1.0 * (oci_uncert_param * current_oci).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, decay_rate, oci_uncert_param]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, decay_rate, oci_uncert_param = parameters
    
    # Handle OCI input safely
    current_oci = oci if oci is not None else 0.0
    
    # Calculate the uncertainty modulation weight
    # As per fitting code: uncertainty_weight = -1.0 * (oci_uncert_param * current_oci)
    uncertainty_weight = -1.0 * (oci_uncert_param * current_oci)

    # Initialize Q-values
    # q_stage1: Q-values for the two actions in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2: Q-values for the two actions in each of the two second-stage states
    q_stage2 = np.zeros((2, 2))

    # Initialize uncertainty trace for stage 1 options
    uncertainty_s1 = np.zeros(2)

    # Transition matrix (fixed 70/30 structure standard for 2-step tasks)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. Stage 1 Choice
        # Calculate net Q-values including the uncertainty bonus/penalty
        q_net_stage1 = q_stage1 + (uncertainty_weight * uncertainty_s1)
        
        # Softmax choice rule
        exp_q1 = np.exp(beta * q_net_stage1)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # 2. Update Uncertainty Trace
        uncertainty_s1 += 1.0       # Increment for all options
        uncertainty_s1[a1] = 0.0    # Reset for the chosen option
        uncertainty_s1 = uncertainty_s1 * decay_rate # Apply decay
        
        # 3. Transition to Stage 2
        # Determine second stage state based on transition probabilities
        s2_state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        # Standard softmax on Q-values (no uncertainty mechanism in stage 2 per fitting code)
        exp_q2 = np.exp(beta * q_stage2[s2_state_idx])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # 5. Reward Generation
        # Get current reward probabilities from drift parameters
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        r = int(rng.random() < current_reward_probs[s2_state_idx][a2])
        
        # 6. Learning Updates (TD Learning)
        
        # Stage 2 update
        pe_2 = r - q_stage2[s2_state_idx, a2]
        q_stage2[s2_state_idx, a2] += learning_rate * pe_2
        
        # Stage 1 update (TD(0) style: update based on value of state arrived at)
        # Note: The fitting code uses q_stage2[state_idx, action_2] as the target for stage 1
        pe_1 = q_stage2[s2_state_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * pe_1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
