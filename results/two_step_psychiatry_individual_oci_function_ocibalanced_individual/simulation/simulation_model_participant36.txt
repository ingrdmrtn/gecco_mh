import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Reinforcement Learning with OCI-Scaled Perseveration.
    
    Hypothesis: High OCI scores relate to compulsive repetition of actions. 
    Here, the OCI score acts as a multiplier on a 'stickiness' bonus added to the 
    Q-value of the previously chosen action.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_factor = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, take the value, otherwise assume scalar)
    current_oci = oci[0] if (isinstance(oci, (list, np.ndarray)) and len(oci) > 0) else oci
    if current_oci is None:
        current_oci = 0.0

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for the two options in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for the two options in each of the two stage 2 states
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed as per standard 2-step task)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    effective_stickiness = stickiness_factor * current_oci
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        q_vals_1 = q_stage1_mf.copy()
        
        # Add stickiness bonus if there was a previous action
        if last_action_1 != -1:
            q_vals_1[last_action_1] += effective_stickiness
            
        # Softmax choice rule for Stage 1
        exp_q1 = np.exp(beta * q_vals_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Use simple Q-learning for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Get current trial's reward probabilities based on state and action
        # Mapping: s2=0 -> (drift1, drift2), s2=1 -> (drift3, drift4)
        if s2 == 0:
            current_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_prob)
        
        # 5. Learning / Updates
        
        # Update Stage 1 Q-value (TD(1) logic or SARSA-like update as seen in fitting code)
        # The fitting code uses: q_stage1_mf[a1] += learning_rate * (q_stage2_mf[state_idx, a2] - q_stage1_mf[a1])
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Direct update to Stage 1 based on Stage 2 RPE (Model-Free path)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update tracking variable
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
