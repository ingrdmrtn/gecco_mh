import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 1: OCI-Modulated Loss-Chasing Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_mb, stick_base, stick_loss_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).

        reward (np.ndarray): Rewards (0 or 1).

    learning_rate, beta, w_mb, stick_base, stick_loss_oci = parameters

    oci_score = oci if isinstance(oci, (float, int)) else oci[0]

    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action (State 0 and State 1)

    # Storage for history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1
    last_reward = -1

    for t in range(n_trials):
        # Current reward probabilities for the two states (0 and 1)
        # State 0 corresponds to drifts 1 and 2, State 1 to drifts 3 and 4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values
        q_net = w_mb * q_stage1_mb + (1 - w_mb) * q_stage1_mf
        
        # Calculate logits
        logits = beta * q_net

        # Apply OCI-Modulated Loss-Chasing Stickiness
        if last_action_1 != -1:

            # Add extra stickiness if previous trial was a loss
            if last_reward == 0.0:
                stickiness += stick_loss_oci * oci_score
            logits[int(last_action_1)] += stickiness

        # Softmax for Stage 1
        exp_q1 = np.exp(logits - np.max(logits)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 choice
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine Stage 2 state based on transition matrix
        # State 0 or 1 (mapped to indices 0 and 1 for array access)
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        
        # TD Update for Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # TD Update for Stage 1 (using updated Stage 2 value as target, per fitting code)

        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history for next trial
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
