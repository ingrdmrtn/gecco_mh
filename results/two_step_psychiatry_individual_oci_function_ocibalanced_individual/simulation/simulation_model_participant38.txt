import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid Model-Based/Model-Free 2-step model.
    The mixing weight 'w' is modulated by OCI: w_effective = w_base * (1 - oci).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant. 
                             Defaults to 0 if None.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, w_base = parameters
    
    # Handle OCI input safely
    participant_oci = oci if oci is not None else 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 (actions 0, 1)
    q_stage1_mf = np.zeros(2)       
    # q_stage2_mf: Model-free values for stage 2 (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))  

    w_effective = w_base * (1.0 - participant_oci)

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for easy indexing: [state][action]
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        # transition_matrix[a1] gives [prob_state0, prob_state1]
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 using simple MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < current_reward_probs[s_idx][a2])

        # --- Learning / Updates ---
        
        # Update Stage 2 Q-values (SARSA-like logic based on fitting code)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD(0) logic: driven by value of state reached)
        # Note: The fitting code uses q_stage2_mf[s_idx, a2] (post-update or pre-update depending on implementation flow, 
        # but typically TD(0) uses the value of the next state/action pair). 
        # Looking at fitting code: 
        # delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        # This occurs *after* stage 2 update in the fitting loop provided.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Record data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
