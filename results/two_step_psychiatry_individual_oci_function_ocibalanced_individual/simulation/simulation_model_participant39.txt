import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free learner with OCI-modulated perseveration.
    
    The model calculates an effective stickiness parameter based on the participant's
    OCI score: stickiness_effective = stickiness_base + (oci_amp * oci_score).
    This bonus is added to the Q-value of the previously chosen action in Stage 1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_amp]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the second stage options.
        oci (float, optional): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_amp = parameters
    
    # Handle OCI input safely
    oci_score = oci if oci is not None else 0.0
    
    # Calculate effective stickiness based on OCI
    stickiness_effective = stickiness_base + (oci_amp * oci_score)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states, 2 actions per state
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed 70/30 structure standard in this task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    last_action_1 = -1 # No previous action for first trial

    for t in range(n_trials):
        # Current trial reward probabilities
        # state 0 corresponds to drift1/drift2, state 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_with_stickiness = q_stage1_mf.copy()
        
        # Apply stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_stage1_with_stickiness[last_action_1] += stickiness_effective
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_with_stickiness)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 (no stickiness applied here in this model)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s_idx][a2])

        # --- Learning (TD Updates) ---
        
        # 1. Update Stage 2 Q-values based on reward
        # Prediction Error: Reward - Expected Value
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # 2. Update Stage 1 Q-values based on Stage 2 value (TD-0 / SARSA-style logic)
        # Note: The fitting code uses q_stage2_mf[s_idx, a2] as the target for Stage 1 update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
