import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-modulated Choice Stickiness.
    
    This model is a pure Model-Free learner but includes a 'stickiness' parameter 
    that biases the agent to repeat the last chosen action at Stage 1.
    The magnitude of this stickiness is scaled by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, stickiness_oci = parameters
    
    # Handle OCI input (scalar or list/array)
    if np.ndim(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate the effective stickiness phi
    phi = stickiness_base + (stickiness_oci * oci_score)

    # Transition matrix (standard 2-step task structure)
    # Action 0 -> 70% State 0 (index 1), 30% State 1 (index 2)
    # Action 1 -> 30% State 0 (index 1), 70% State 1 (index 2)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1 corresponds to the two actions at the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to two states, each having two actions
    q_stage2 = np.zeros((2, 2))

    # Initialize arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 # No previous action for the first trial
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities for State 0 (drifts 1&2) and State 1 (drifts 3&4)
        # Note: drift indices usually map to: state 0 (actions 0,1) -> drift1, drift2; state 1 (actions 0,1) -> drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_net = q_stage1.copy()
        
        # Apply stickiness if there was a previous action
        if last_action_1 != -1:
            q_net[last_action_1] += phi
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using Q-values for the specific state s2)
        exp_q2 = np.exp(beta * q_stage2[s2])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (TD updates) ---
        # 1. Update Stage 2 Q-values
        # Prediction error: Reward received - Expected value
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # 2. Update Stage 1 Q-values
        # Prediction error: Value of state arrived at (Q_stage2) - Expected value of action taken (Q_stage1)
        # Note: This uses SARSA-like update logic present in the fitting code: q_stage2[current_state, current_a2]
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1

        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
