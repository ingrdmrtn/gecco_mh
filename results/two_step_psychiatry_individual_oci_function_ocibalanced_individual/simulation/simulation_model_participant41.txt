Here is the simulation code corresponding to the OCI-Mediated Eligibility Trace model.

```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-Mediated Eligibility Trace.
    
    This model uses temporal difference learning where the eligibility trace (lambda)
    depends on OCI. A higher OCI score increases lambda, meaning the Stage 1 Q-value 
    is more strongly updated by the Stage 2 Reward Prediction Error.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lambda_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or np.ndarray): Obsessive-Compulsive Inventoryâ€“Revised score.
                                          Can be a single float or a list/array where the first element is used.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, lambda_scale = parameters
    
    # Handle OCI input (robustness for single float or list/array)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0.0

    # Calculate eligibility trace based on OCI
    eligibility_lambda = np.clip(lambda_scale * oci_score, 0.0, 1.0)
    
    # Initialize arrays for storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1: Q-values for the two actions in stage 1
    q_stage1 = np.full(2, 0.5)
    # q_stage2: Q-values for the two actions in each of the two stage 2 states
    # Shape is (2, 2) -> (state_idx, action_idx)
    q_stage2 = np.full((2, 2), 0.5)
    
    # Transition matrix (fixed for the standard 2-step task)
    # Row 0: State 1 transitions (70% common to State 2, 30% rare to State 3)
    # Row 1: State 1 transitions (30% rare to State 2, 70% common to State 3)
    # Note: In the output `state2` variable, 0 maps to State 2, 1 maps to State 3.
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 2 (index 0) has probs [drift1, drift2]
        # State 3 (index 1) has probs [drift3, drift4]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # If a1=0: 0.7 prob of state 0, 0.3 prob of state 1
        # If a1=1: 0.3 prob of state 0, 0.7 prob of state 1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # s2 is the index for q_stage2 (0 or 1)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        
        # 1. Prediction error for Stage 1 (TD(0) style)
        # The value of the chosen stage 2 state-action pair is the target
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        
        # 2. Prediction error for Stage 2
        delta_stage2 = r - q_stage2[s2, a2]
        
        # Update Stage 2 Q-values
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values
        # Standard TD update
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Eligibility trace update: Stage 1 choice gets credit for Stage 2 RPE
        # scaled by lambda (which is scaled by OCI)
        q_stage1[a1] += learning_rate * eligibility_lambda * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
