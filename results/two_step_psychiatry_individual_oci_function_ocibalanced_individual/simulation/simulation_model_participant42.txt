import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model-Free RL with OCI-modulated Inverse Temperature.
    
    High OCI scores increase 'beta', making the softmax function steeper.
    This leads to more deterministic/rigid choices (less exploration).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, rigidity_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize Random Number Generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta_base, rigidity_factor = parameters
    
    # Handle OCI input (default to 0 if None, though fit code assumes it exists)
    oci_score = oci if oci is not None else 0.0

    # Calculate effective beta based on OCI
    effective_beta = beta_base * (1.0 + rigidity_factor * oci_score)
    effective_beta = np.clip(effective_beta, 0.0, 20.0)

    # Initialize Q-values
    # q_stage1: Q-values for the two actions in stage 1
    q_stage1 = np.zeros(2)
    # q_stage2: Q-values for the two actions in the two states of stage 2 (shape: 2 states x 2 actions)
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed for the 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # 1. First Stage Choice
        exp_q1 = np.exp(effective_beta * q_stage1)
        # Add small epsilon to avoid division by zero if overflow happens (though clipping beta helps)
        sum_exp_q1 = np.sum(exp_q1)
        if sum_exp_q1 == 0 or np.isinf(sum_exp_q1) or np.isnan(sum_exp_q1):
             probs_1 = np.array([0.5, 0.5]) # Fallback to random
        else:
             probs_1 = exp_q1 / sum_exp_q1
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Second Stage
        # Determine state based on transition matrix probabilities for chosen action a1
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs)
        
        # 3. Second Stage Choice
        exp_q2 = np.exp(effective_beta * q_stage2[s2])
        sum_exp_q2 = np.sum(exp_q2)
        if sum_exp_q2 == 0 or np.isinf(sum_exp_q2) or np.isnan(sum_exp_q2):
             probs_2 = np.array([0.5, 0.5])
        else:
             probs_2 = exp_q2 / sum_exp_q2
             
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Get current trial reward probabilities
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # 5. Learning (Model-Free Updates)
        
        # TD Error Stage 2
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # TD Error Stage 1 (SARSA-like update based on Q-value of state 2)
        # Note: The fitting code uses: delta_stage1 = q_stage2[state_idx, a2] - q_stage1[a1]
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Eligibility Trace / Direct update from reward to stage 1
        # The fitting code does: q_stage1[a1] += learning_rate * delta_stage2
        q_stage1[a1] += learning_rate * delta_stage2

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
