To convert the fitting code into simulation code, we need to replicate the decision-making and learning logic but replace the likelihood calculation with random sampling of choices and rewards.

Here is the simulation code for Model 2 (OCI-modulated Choice Perseveration):

```python
import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-modulated Choice Perseveration (Stickiness).
    
    This model implements a pure Model-Free learner but adds a perseveration bonus to the Q-values
    of the action taken on the previous trial. The magnitude of this bonus is determined
    by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_stickiness_mult]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, stickiness_base, oci_stickiness_mult = parameters
    
    # Handle OCI input (can be a list/array or a single float)
    if hasattr(oci, '__len__'):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0.0

    # Calculate the total stickiness parameter
    stickiness = stickiness_base + (oci_stickiness_mult * oci_score)

    # Transition matrix (standard 2-step task structure)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: Q-values for the two actions in stage 1
    # q_stage2: Q-values for the two actions in the two possible states of stage 2
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with invalid action

    # Data containers
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current reward probabilities for the second stage
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_eff_1 = q_stage1.copy()
        
        # Apply stickiness bonus if there was a previous action
        if last_action_1 != -1:
            q_eff_1[last_action_1] += stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_eff_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Sample Action 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition to Stage 2 ---
        # Determine state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using Q-values for the specific state arrived at)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Sample Action 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Sample reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Model-Free TD) ---
        # Note: The fitting code uses a specific TD(0) implementation.
        # It updates Stage 1 based on the value of the CHOSEN option in Stage 2 (SARSA-like logic here).
        
        # Update Stage 1 Q-value
        # Prediction Error 1: Value of Stage 2 choice minus Value of Stage 1 choice
        pe_1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * pe_1

        # Update Stage 2 Q-value
        # Prediction Error 2: Reward minus Value of Stage 2 choice
        pe_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * pe_2

        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
