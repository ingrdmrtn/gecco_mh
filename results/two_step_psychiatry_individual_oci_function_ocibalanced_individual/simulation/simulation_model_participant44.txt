import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using Model 3: OCI-Dependent Eligibility Trace (Lambda).
    
    Hypothesis: OCI affects the efficiency of credit assignment between stages.
    High OCI might lead to stronger 'chaining' (higher lambda).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, lambda_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or np.ndarray): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, lambda_base = parameters
    
    # Handle OCI input (scalar or array)
    if oci is None:
        oci_score = 0.0
    elif np.ndim(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate OCI-dependent lambda
    eligibility_lambda = np.clip(lambda_base + (0.5 * oci_score), 0.0, 1.0)

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # 2 states (indices 0 and 1), 2 actions

    # Transition matrix (Standard 2-step task structure)
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Map state/action to specific drift parameter
        if s2 == 0:
            p_rew = drift1[t] if a2 == 0 else drift2[t]
        else: # s2 == 1
            p_rew = drift3[t] if a2 == 0 else drift4[t]
        
        r = 1 if rng.random() < p_rew else 0

        # --- Updates (Following logic of cognitive_model3) ---
        
        # 1. Calculate RPE for Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # 2. Update Stage 2 Q-values
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 3. Calculate RPE for Stage 1 (using updated Q2 value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]

        # 4. Combined update for Stage 1 using eligibility trace
        combined_update = delta_stage1 + (eligibility_lambda * delta_stage2)
        
        # 5. Update Stage 1 Q-values
        q_stage1_mf[a1] += learning_rate * combined_update

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
