def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Compulsive Perseveration Model (OCI-Modulated Trace Decay).

    Hypothesis: High OCI leads to "stickier" habits due to slower decay of choice traces.
    A choice trace accumulates when an action is chosen and decays over time.
    The decay rate is modulated by OCI: higher OCI results in slower decay (higher retention of the trace),
    leading to compulsive repetition of actions regardless of reward history.

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - lambd: [0, 1] Eligibility trace.
    - stick_weight: [0, 5] Weight of the choice trace in the decision (habit strength).
    - decay_base: [0, 1] Base decay rate for low OCI.
    """
    learning_rate, beta, w, lambd, stick_weight, decay_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    choice_trace = np.zeros(2)


    decay_eff = decay_base + (1.0 - decay_base) * oci_score

    for trial in range(n_trials):

        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + stick_weight * choice_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        choice_trace *= decay_eff
        choice_trace[a1] += 1.0

        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
