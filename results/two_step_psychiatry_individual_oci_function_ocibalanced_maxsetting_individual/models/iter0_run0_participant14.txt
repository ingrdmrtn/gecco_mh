Here are three cognitive models formulated as Python functions. Each model incorporates the Obsessive-Compulsive Inventory (OCI) score to modulate a specific mechanism of the reinforcement learning process, based on hypotheses linking compulsivity to habit formation, perseveration, and model-based control.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Determined Weight Model.
    
    This model assumes that the balance between Goal-Directed (Model-Based) and 
    Habitual (Model-Free) control is directly determined by the participant's 
    OCI score. 
    
    Hypothesis: Lower OCI scores (lower compulsivity) are associated with 
    stronger Model-Based control.
    
    Equation: w = 1.0 - OCI (clipped to [0,1]).
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta_1: [0, 10] Inverse temperature (randomness) for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature (randomness) for Stage 2 choice.
    """
    learning_rate, beta_1, beta_2 = model_parameters
    oci_val = oci[0]
    
    # Mixing weight w is determined by OCI.
    # Low OCI (e.g., 0.3) results in high w (0.7), favoring Model-Based control.
    w = 1.0 - oci_val
    w = np.clip(w, 0.0, 1.0)
    
    n_trials = len(action_1)
    
    # Transition probabilities: 
    # Row 0 (Action A) -> 0.7 prob to State 0 (X), 0.3 to State 1 (Y)
    # Row 1 (Action U) -> 0.3 prob to State 0 (X), 0.7 to State 1 (Y)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        # Handle missing data
        if a1 == -1:
            continue
            
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: Expected value of best stage 2 options weighted by transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted sum of MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        # Softmax Choice 2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 1 MF Update (SARSA-like prediction error)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update (TD(1)): Stage 1 value updated by Stage 2 outcome
        # Using full eligibility (lambda=1) as baseline for this model
        q_stage1_mf[a1] += learning_rate * delta_stage2

    # Calculate Negative Log Likelihood
    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stickiness Model.
    
    This model extends the standard hybrid learner with a "stickiness" (perseveration)
    bonus that is modulated by the OCI score.
    
    Hypothesis: High OCI scores relate to compulsive repetition. The tendency to 
    repeat the previous Stage 1 choice is scaled by the OCI score.
    
    Stickiness Bonus = stick_sensitivity * OCI * I(current_action == prev_action)
    
    Parameters:
    - learning_rate: [0, 1]
    - beta_1: [0, 10]
    - beta_2: [0, 10]
    - w: [0, 1] Mixing weight (0 = Pure MF, 1 = Pure MB).
    - stick_sensitivity: [0, 10] Scaling factor for OCI-driven stickiness.
    """
    learning_rate, beta_1, beta_2, w, stick_sensitivity = model_parameters
    oci_val = oci[0]
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            prev_a1 = -1
            continue
            
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Stickiness Bonus
        stick_bonus = np.zeros(2)
        if prev_a1 != -1:
            # Add bonus to the previously chosen action, scaled by OCI
            stick_bonus[prev_a1] = stick_sensitivity * oci_val
        
        # Softmax with stickiness
        exp_q1 = np.exp(beta_1 * (q_net + stick_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Standard eligibility trace
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        prev_a1 = a1

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Determined Eligibility Trace Model.
    
    This model assumes that the strength of the eligibility trace (lambda) -- 
    how strongly the Stage 2 outcome reinforces the Stage 1 choice -- is 
    determined by the OCI score.
    
    Hypothesis: Higher OCI scores (compulsivity) lead to stronger habit formation 
    via a stronger eligibility trace, linking distal rewards to proximal choices 
    regardless of model structure.
    
    lambda = OCI (clipped to [0,1]).
    
    Parameters:
    - learning_rate: [0, 1]
    - beta_1: [0, 10]
    - beta_2: [0, 10]
    - w: [0, 1] Mixing weight.
    """
    learning_rate, beta_1, beta_2, w = model_parameters
    oci_val = oci[0]
    
    # Eligibility trace lambda is set by OCI
    lam = np.clip(oci_val, 0.0, 1.0)
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            continue
            
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update scaled by OCI-determined lambda
        # If OCI is low, Stage 1 MF values are weakly updated by Stage 2 outcomes
        q_stage1_mf[a1] += learning_rate * lam * delta_stage2

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
