Here are three cognitive models formulated as Python functions. They explore how Obsessive-Compulsive symptoms (OCI) might modulate the balance between goal-directed and habitual control, the tendency to perseverate (stickiness), and the rate of learning.

### Model 1: OCI-Modulated Model-Based Weight
This model hypothesizes that the balance between Model-Based (goal-directed) and Model-Free (habitual) control is directly influenced by the participant's OCI score. High OCI is often associated with deficits in model-based control. Here, the mixing weight $w$ is a logistic function of the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model where the balance between Model-Based and Model-Free control (w)
    is modulated by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choice.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choice.
    - w_base: [-5, 5] Base logit for the mixing weight w.
    - w_oci_slope: [-5, 5] How strongly OCI affects w. Negative slope implies high OCI -> low MB.
    """
    learning_rate, beta_1, beta_2, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate mixing weight w based on OCI (Sigmoid transformation to keep 0 < w < 1)
    # w represents the weight of Model-Based control.
    w_logit = w_base + w_oci_slope * current_oci
    w = 1.0 / (1.0 + np.exp(-w_logit))
    
    # Transition matrix: Row 0 -> Planet 0 (70%), Planet 1 (30%); Row 1 -> Planet 0 (30%), Planet 1 (70%)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # Model-Free values for Stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-Free values for Stage 2 (Aliens per Planet)
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Handle missing/invalid data (e.g., trial 153 with -1)
        if a1 < 0 or s_idx < 0 or a2 < 0:
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value: Expected max value of next stage weighted by transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        # Standard Model-Free Softmax
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Learning / Updates ---
        # Prediction errors
        pe_1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        pe_2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 1 MF (TD(0))
        q_stage1_mf[a1] += learning_rate * pe_1
        
        # Update Stage 2 MF
        q_stage2_mf[s_idx, a2] += learning_rate * pe_2

    return -log_likelihood
```

### Model 2: OCI-Driven Perseveration
This model hypothesizes that OCI symptoms manifest as "stickiness" or perseverationâ€”a compulsion to repeat the previous action regardless of the outcome. It adds a stickiness bonus to the Q-value of the previously chosen spaceship. The magnitude of this bonus is scaled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model with choice stickiness (perseveration) modulated by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta_1: [0, 10] Inverse temperature Stage 1.
    - beta_2: [0, 10] Inverse temperature Stage 2.
    - w: [0, 1] Fixed mixing weight between MB and MF.
    - stickiness_base: [-2, 2] Base tendency to repeat choice.
    - stickiness_oci: [-2, 2] Additional stickiness scaled by OCI score.
    """
    learning_rate, beta_1, beta_2, w, stickiness_base, stickiness_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate total stickiness parameter
    # High OCI might lead to higher stickiness (positive stickiness_oci)
    total_stickiness = stickiness_base + stickiness_oci * current_oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 < 0 or s_idx < 0 or a2 < 0:
            last_action_1 = -1 # Reset memory on invalid trial
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness bonus to the Q-value of the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += total_stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Learning ---
        # TD(1) / Eligibility Trace approximation for Stage 1
        # Note: Standard Daw et al 2011 uses TD(1) for MF.
        # Here we use simple TD(1) logic: Stage 1 updates from Stage 2 RPE
        
        dt_2 = r - q_stage2_mf[s_idx, a2] # Stage 2 RPE
        q_stage2_mf[s_idx, a2] += learning_rate * dt_2
        
        # Stage 1 update: update towards the actual outcome (TD(1)) rather than just next state value
        # Value of path taken = q_stage2_mf (updated)
        q_stage1_mf[a1] += learning_rate * (q_stage2_mf[s_idx, a2] - q_stage1_mf[a1])
        
        last_action_1 = a1

    return -log_likelihood
```

### Model 3: OCI-Modulated Choice Determinism (Beta)
This model posits that OCI affects the rigidity of decision-making. Specifically, the OCI score scales the inverse temperature ($\beta$) parameters. A higher OCI might lead to higher $\beta$ (more deterministic, less exploration), reflecting the rigid behavioral patterns often seen in compulsive disorders.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model where the inverse temperature (randomness of choice)
    is modulated by OCI. High OCI -> Higher Beta -> More rigid/deterministic choices.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta_base: [0, 10] Baseline inverse temperature.
    - beta_oci_factor: [-5, 5] Multiplier for OCI effect on beta.
    - w: [0, 1] Weight for Model-Based control.
    - lam: [0, 1] Eligibility trace (decay parameter for TD-lambda).
    """
    learning_rate, beta_base, beta_oci_factor, w, lam = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective Beta
    # We ensure beta stays positive.
    # If beta_oci_factor is positive, higher OCI means more deterministic choices.
    beta_effective = max(0.0, beta_base * (1.0 + beta_oci_factor * current_oci))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 < 0 or s_idx < 0 or a2 < 0:
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_effective * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        # Assuming the same beta modulation applies to stage 2
        exp_q2 = np.exp(beta_effective * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Learning (TD-Lambda) ---
        # Stage 2 RPE
        delta_2 = r - q_stage2_mf[s_idx, a2]
        
        # Stage 1 RPE (SARSA-style for Stage 1 to State transition)
        delta_1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_2
        
        # Update Stage 1
        # Combined update: Direct TD(0) + Eligibility Trace from Stage 2
        q_stage1_mf[a1] += learning_rate * delta_1 + learning_rate * lam * delta_2

    return -log_likelihood
```
