Here are three cognitive models that explain the participant's behavior in the two-step task, incorporating the OCI score to modulate specific cognitive mechanisms.

### Model 1: OCI-Modulated Hybrid Learning (Variable Mixing Weight)
This model hypothesizes that the balance between Model-Based (goal-directed) and Model-Free (habitual) control is influenced by the participant's OCI score. Literature suggests that compulsive traits correlate with a reliance on habitual control. Here, the mixing weight $w$ is a logistic function of the OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the mixing weight 'w' 
    is modulated by the OCI score.
    
    Hypothesis: Higher OCI scores may shift the balance towards Model-Free (habitual) 
    control or alter the reliance on Model-Based (planning) strategies.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature for softmax (exploration/exploitation).
    w_base: [-5, 5] - Base logit for the mixing weight.
    w_oci_slope: [-5, 5] - Sensitivity of mixing weight to OCI score.
    lambda_param: [0, 1] - Eligibility trace (credit assignment).
    """
    learning_rate, beta, w_base, w_oci_slope, lambda_param = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate mixing weight w based on OCI (Sigmoid to keep between 0 and 1)
    # w represents the weight of Model-Based values.
    # If w_oci_slope is negative, higher OCI leads to lower w (more habitual).
    w_logit = w_base + w_oci_slope * oci_val
    w = 1.0 / (1.0 + np.exp(-w_logit))
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-Free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-Free values for stage 2 (State x Action)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation: V(State) = max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        # Standard Model-Free softmax at the second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Prediction Error 1: Q_stage2 - Q_stage1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 MF value (SARSA-like)
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Prediction Error 2: Reward - Q_stage2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Update Stage 2 MF value
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace: Propagate Stage 2 error back to Stage 1 choice
        q_stage1_mf[a1] += learning_rate * lambda_param * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Perseveration (Choice Stickiness)
This model assumes the core learning mechanism is a standard hybrid model, but adds a "stickiness" parameter to the first-stage choice. This stickiness is directly scaled by the OCI score. This reflects the hypothesis that OCD-like traits relate to behavioral rigidity or a tendency to repeat previous actions regardless of reward history.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated choice perseveration (stickiness).
    
    Hypothesis: OCI score predicts the degree of 'stickiness' (tendency to repeat 
    the previous Stage 1 choice), independent of value learning.
    
    Parameters:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] - Mixing weight (constant in this model).
    stickiness_base: [-2, 2] - Baseline tendency to repeat.
    stickiness_oci: [0, 5] - Additional stickiness scaling with OCI.
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate total stickiness
    # High OCI -> Higher tendency to repeat the previous action
    stickiness = stickiness_base + (stickiness_oci * oci_val)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the logits
        logits = beta * q_net
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        # Update last action
        last_action_1 = a1

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 MF
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # No lambda in this model (TD(0) for simplicity to isolate stickiness effect)
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Dampened Learning Rate
This model hypothesizes that OCI traits correlate with cognitive rigidity, manifested as a reduced ability to update values based on new information. The effective learning rate is modeled as a decaying function of the OCI score (higher OCI = slower learning).

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model where the learning rate is dampened by the OCI score.
    
    Hypothesis: Higher OCI indicates cognitive rigidity, resulting in a 
    lower effective learning rate (slower updating of value beliefs).
    
    Parameters:
    lr_base: [0, 1] - Maximum theoretical learning rate (at OCI=0).
    lr_decay: [0, 5] - How strongly OCI reduces the learning rate.
    beta: [0, 10]
    w: [0, 1]
    lambda_param: [0, 1]
    """
    lr_base, lr_decay, beta, w, lambda_param = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Effective learning rate decreases as OCI increases
    # Using an exponential decay curve
    learning_rate = lr_base * np.exp(-lr_decay * oci_val)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Update Stage 1 MF
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update
        q_stage1_mf[a1] += learning_rate * lambda_param * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
