Here are three cognitive models designed to explain the participant's behavior, incorporating their medium OCI score (0.4) into the decision-making process.

### Model 1: OCI-Weighted Model-Based/Model-Free Hybrid
This model hypothesizes that the balance between Model-Based (planning) and Model-Free (habitual) control is modulated by the participant's OCI score. A score of 0.4 suggests a "medium" level of compulsivity. In this model, the `w` parameter (mixing weight) is not a free parameter but is derived directly from or influenced by the OCI score, allowing the model to test if symptom severity directly sets the planning horizon.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL where the mixing weight 'w' is 
    modulated by the OCI score.
    
    The model assumes that higher OCI scores might lead to more habitual (Model-Free) 
    behavior or rigid planning. Here, we model 'w' as a logistic function of OCI 
    scaled by a free parameter, testing if OCI predicts the reliance on MB vs MF.

    Parameters:
    learning_rate: [0, 1] - Rate at which Q-values are updated.
    beta: [0, 10] - Inverse temperature for softmax action selection.
    w_scale: [0, 10] - Scaling factor for how strongly OCI affects the mixing weight.
    """
    learning_rate, beta, w_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate mixing weight w based on OCI. 
    # A simple sigmoid transformation to keep w between 0 and 1.
    # If w_scale is high, OCI has a strong effect.
    # We center OCI around 0.5 for the sigmoid.
    w = 1 / (1 + np.exp(-w_scale * (oci_score - 0.5)))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (Aliens per planet)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based value calculation: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value: weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        s1 = int(state[trial]) # The planet arrived at
        
        # Softmax for Stage 2 (purely model-free based on alien values)
        exp_q2 = np.exp(beta * q_stage2_mf[s1])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning / Updating ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # SARSA-style update for Stage 1 MF
        # TD error using the value of the state actually reached (Stage 2 value)
        delta_stage1 = q_stage2_mf[s1, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Update for Stage 2
        delta_stage2 = r - q_stage2_mf[s1, a2]
        q_stage2_mf[s1, a2] = q_stage2_mf[s1, a2] + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Stickiness (Perseveration)
This model posits that the core deficit or variation related to the OCI score isn't in the learning rate, but in *perseveration* (stickiness). Individuals with higher compulsivity scores might be more prone to repeating the same action regardless of reward history (compulsive repetition). This model adds a "stickiness" bonus to the previously chosen action, where the magnitude of this bonus is scaled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with OCI-modulated choice perseveration (stickiness).
    
    This model assumes the participant relies on Model-Free learning but has a 
    tendency to repeat previous choices (stickiness). The strength of this 
    stickiness is determined by the OCI score.
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    stickiness_factor: [0, 5] - Base multiplier for how much OCI influences repetition.
    w: [0, 1] - Standard MB/MF mixing weight (treated as free param here).
    """
    learning_rate, beta, stickiness_factor, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Track previous choice for stickiness
    last_action_1 = -1 

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus if not the first trial
        # Bonus = OCI * stickiness_factor applied to the index of the last action
        aug_q_net = q_net.copy()
        if last_action_1 != -1:
            aug_q_net[last_action_1] += (oci_score * stickiness_factor)

        exp_q1 = np.exp(beta * aug_q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Update last action
        last_action_1 = int(action_1[trial])
        
        # --- Stage 2 Decision ---
        s1 = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s1])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # TD(0) update for Stage 1 MF
        delta_stage1 = q_stage2_mf[s1, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Update for Stage 2
        delta_stage2 = r - q_stage2_mf[s1, a2]
        q_stage2_mf[s1, a2] = q_stage2_mf[s1, a2] + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Eligibility Traces (TD-Lambda)
This model introduces an eligibility trace parameter ($\lambda$) that connects the second-stage outcome back to the first-stage choice. Here, we hypothesize that OCI affects the efficiency of credit assignment. A common finding in compulsivity is a decoupling of action and outcome or over-reliance on immediate outcomes. In this model, the eligibility trace decay ($\lambda$) is a function of the OCI score, testing if higher symptoms lead to longer or shorter credit assignment windows.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with Eligibility Traces (TD-Lambda) where Lambda is OCI-dependent.
    
    The eligibility trace parameter lambda determines how much the Stage 2 reward 
    directly updates the Stage 1 choice (bypassing the Stage 2 state value estimate).
    We model lambda as being constrained by the OCI score.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    lambda_base: [0, 1] - Base eligibility trace parameter.
    oci_mod: [0, 1] - Strength of OCI modulation on lambda.
    """
    learning_rate, beta, lambda_base, oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective lambda.
    # We hypothesize OCI might dampen the eligibility trace (making credit assignment harder)
    # or strengthen it. Here we define it as: lambda_eff = lambda_base * (1 - oci_mod * oci)
    # This keeps lambda within [0, 1] assuming params are in bounds.
    lambda_eff = lambda_base * (1.0 - (oci_mod * oci_score))
    lambda_eff = np.clip(lambda_eff, 0.0, 1.0)

    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision (Pure Model-Free for this model to isolate Lambda effects) ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        s1 = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2[s1])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning with Eligibility Traces ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Prediction error at stage 2 (outcome)
        delta_2 = r - q_stage2[s1, a2]
        
        # Update Stage 2 value
        q_stage2[s1, a2] = q_stage2[s1, a2] + learning_rate * delta_2
        
        # Update Stage 1 value using Eligibility Trace
        # In standard Q-learning, Stage 1 updates based on Stage 2 Value (TD-0).
        # With Lambda=1, Stage 1 updates based on the Reward itself (Monte Carlo).
        # Update = alpha * (TD_error_stage1 + lambda * TD_error_stage2)
        
        # 1. Immediate error for Stage 1 (moving to Stage 2 state)
        delta_1 = q_stage2[s1, a2] - q_stage1[a1] # Note: using q_stage2 PRE-update is standard TD
        
        # Combined update for Stage 1
        # Q1 = Q1 + alpha * (delta_1 + lambda * delta_2)
        q_stage1[a1] = q_stage1[a1] + learning_rate * (delta_1 + lambda_eff * delta_2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
