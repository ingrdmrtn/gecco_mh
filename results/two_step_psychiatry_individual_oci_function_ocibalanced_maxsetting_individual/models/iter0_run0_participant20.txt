Here are three cognitive models designed to capture the behavior of the participant, incorporating their OCI-R score (0.366, representing medium obsessive-compulsive symptoms) into the decision-making logic.

### Cognitive Model 1: Hybrid Learner with OCI-Modulated Mixing
This model assumes that the balance between Model-Based (planning) and Model-Free (habitual) control is influenced by the participant's OCI score. A higher OCI score might push the participant towards more habitual (Model-Free) control or rigid planning. Here, we model the mixing weight `w` as a function of the OCI score, where the OCI score scales the reliance on Model-Based planning.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the mixing weight (w) is 
    modulated by the OCI score.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature for softmax choice.
    w_base: [0,1] - Baseline weight for Model-Based control.
    w_oci_scale: [0,1] - How much OCI score reduces/increases Model-Based control.
    
    The final mixing weight is calculated as: w = w_base * (1 - w_oci_scale * oci)
    This tests the hypothesis that higher OCI leads to reduced model-based flexibility.
    """
    learning_rate, beta, w_base, w_oci_scale = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate effective mixing weight based on OCI
    # If w_oci_scale is high, higher OCI reduces the model-based contribution.
    w = w_base * (1.0 - w_oci_scale * participant_oci)
    # Clamp w to [0, 1]
    if w < 0: w = 0
    if w > 1: w = 1

    # Fixed transition matrix for the task (70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (Aliens)

    for trial in range(n_trials):
        # Skip missing data (-1)
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision (Spaceship) ---
        # Model-Based Value: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision (Alien) ---
        # Standard Model-Free Q-learning for the second stage
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 Update (TD Error)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD Error using Stage 2 value as proxy for reward)
        # Note: Standard TD(0) uses the value of the state reached
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Driven Perseveration (Stickiness)
This model posits that OCI symptoms manifest as "stickiness" or perseverationâ€”a tendency to repeat the previous choice regardless of reward. The OCI score directly scales a perseveration parameter added to the Q-values, making the model more likely to repeat the previous action if OCI is high.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with OCI-modulated choice perseveration (stickiness).
    
    Parameters:
    learning_rate: [0,1]
    beta: [0,10]
    perseveration_base: [0,5] - Base tendency to repeat previous choice.
    oci_pers_sensitivity: [0,5] - How strongly OCI amplifies perseveration.
    
    The bonus added to the Q-value of the previously chosen action is:
    bonus = perseveration_base + (oci_pers_sensitivity * oci)
    """
    learning_rate, beta, perseveration_base, oci_pers_sensitivity = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    perseveration_bonus = perseveration_base + (oci_pers_sensitivity * participant_oci)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize as invalid

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            last_action_1 = -1
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        # Add stickiness bonus to the Q-values locally for decision making
        q_stage1_decision = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_decision[last_action_1] += perseveration_bonus
            
        exp_q1 = np.exp(beta * q_stage1_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision ---
        # No stickiness modeled for stage 2 here to isolate Stage 1 planning stickiness
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # SARSA-like update for Stage 1 using Stage 2 value
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Stage 2 update
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Learning Rate Asymmetry
This model suggests that OCI affects how participants learn from positive versus negative prediction errors. Specifically, it tests if higher OCI leads to a stronger emphasis on negative outcomes (loss aversion or hyper-sensitivity to failure). The learning rate is split into positive and negative components, with the negative learning rate scaled by OCI.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner where learning rates for positive and negative prediction errors
    are asymmetric, and the negative learning rate is modulated by OCI.
    
    Parameters:
    alpha_pos: [0,1] - Learning rate for positive prediction errors (PE > 0).
    alpha_neg_base: [0,1] - Base learning rate for negative prediction errors (PE < 0).
    oci_sens: [0,1] - How much OCI increases sensitivity to negative PEs.
    beta: [0,10] - Inverse temperature.
    
    Effective alpha_neg = alpha_neg_base + (oci_sens * oci)
    """
    alpha_pos, alpha_neg_base, oci_sens, beta = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate effective negative learning rate
    alpha_neg = alpha_neg_base + (oci_sens * participant_oci)
    # Clamp to [0, 1]
    if alpha_neg > 1: alpha_neg = 1
    if alpha_neg < 0: alpha_neg = 0

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2[s_idx, a2]
        if delta_stage2 >= 0:
            q_stage2[s_idx, a2] += alpha_pos * delta_stage2
        else:
            q_stage2[s_idx, a2] += alpha_neg * delta_stage2
            
        # Stage 1 Update
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        if delta_stage1 >= 0:
            q_stage1[a1] += alpha_pos * delta_stage1
        else:
            q_stage1[a1] += alpha_neg * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
