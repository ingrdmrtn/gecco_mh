Here are three cognitive models based on the two-step task and the participant's data, incorporating the OCI score.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that the balance between Model-Based (planning based on transition structure) and Model-Free (habitual) control is influenced by the participant's OCI score. The `w` parameter (weighting between MB and MF) is modulated by OCI, suggesting that obsessive-compulsive traits might bias a participant towards more rigid habit formation (MF) or excessive planning (MB).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free Reinforcement Learning with OCI modulation on the mixing weight.
    
    The balance between Model-Based (goal-directed) and Model-Free (habitual) control
    is determined by a weighting parameter 'w'. This 'w' is dynamically adjusted 
    based on the participant's OCI score.
    
    Parameters:
    learning_rate: [0, 1] - Rate at which Q-values are updated.
    beta: [0, 10] - Inverse temperature for softmax choice rule.
    w_base: [0, 1] - Base weighting for Model-Based control (0=Pure MF, 1=Pure MB).
    oci_sensitivity: [0, 1] - How strongly OCI affects the weighting 'w'.
    """
    learning_rate, beta, w_base, oci_sensitivity = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w based on OCI. 
    # Hypothesis: Higher OCI might reduce flexibility (lower MB), so we subtract scaled OCI.
    # We clip to ensure it stays in [0, 1].
    w = np.clip(w_base - (oci_sensitivity * oci_score), 0.0, 1.0)
    
    # Fixed transition matrix as per task description (A->X, U->Y commonly)
    # A=0, U=1; X=0, Y=1.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for spaceships
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for aliens (state, action)
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation: V(state) = max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of chosen action (handle missing data if any -1s exist)
        a1 = int(action_1[trial])
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0 # Ignore missing trials
            
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        if s_idx != -1:
            # Softmax Policy Stage 2
            exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
            probs_2 = exp_q2 / np.sum(exp_q2)
            
            a2 = int(action_2[trial])
            if a2 != -1:
                p_choice_2[trial] = probs_2[a2]
                
                # --- Learning ---
                r = reward[trial]
                
                # TD Error Stage 2
                delta_stage2 = r - q_stage2_mf[s_idx, a2]
                q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
                
                # TD Error Stage 1 (SARSA-style update for MF)
                if a1 != -1:
                    delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
                    q_stage1_mf[a1] += learning_rate * delta_stage1
            else:
                p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Perseveration
This model posits that OCI traits relate to "stickiness" or perseverationâ€”the tendency to repeat the previous choice regardless of reward. A participant with medium OCI might exhibit specific repetitive behaviors. This model adds a perseveration bonus to the Q-values of the previously chosen action, where the magnitude of this bonus is scaled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with OCI-modulated Perseveration (Stickiness).
    
    The model assumes the participant is primarily model-free but has a tendency 
    to repeat choices (perseveration). The strength of this perseveration is 
    a function of the OCI score.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    perseveration_base: [0, 5] - Base tendency to repeat the last choice.
    oci_scaling: [0, 5] - How much OCI amplifies the perseveration effect.
    """
    learning_rate, beta, perseveration_base, oci_scaling = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective perseveration parameter
    # Higher OCI -> Higher tendency to stick to previous choice
    pers_weight = perseveration_base + (oci_scaling * oci_score)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize previous action
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        
        # --- Stage 1 Decision ---
        # Add perseveration bonus to Q-values before softmax
        q_stage1_eff = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_eff[last_action_1] += pers_weight
            
        exp_q1 = np.exp(beta * q_stage1_eff)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
            last_action_1 = a1 # Update last action
        else:
            p_choice_1[trial] = 1.0
            last_action_1 = -1 # Reset if missing
            
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        if s_idx != -1:
            exp_q2 = np.exp(beta * q_stage2[s_idx])
            probs_2 = exp_q2 / np.sum(exp_q2)
            
            a2 = int(action_2[trial])
            if a2 != -1:
                p_choice_2[trial] = probs_2[a2]
                
                # --- Learning ---
                r = reward[trial]
                
                # Update Stage 2
                delta2 = r - q_stage2[s_idx, a2]
                q_stage2[s_idx, a2] += learning_rate * delta2
                
                # Update Stage 1 (Direct reinforcement from reward, pure MF)
                if a1 != -1:
                    delta1 = r - q_stage1[a1] 
                    q_stage1[a1] += learning_rate * delta1
            else:
                p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Learning Rate Asymmetry
This model suggests that OCI scores influence how participants learn from positive versus negative prediction errors. Individuals with obsessive-compulsive traits may be hyper-sensitive to negative outcomes (avoidance learning) or positive outcomes. Here, the OCI score shifts the balance between a `learning_rate_pos` and `learning_rate_neg`.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Reinforcement Learning with Asymmetric Learning Rates modulated by OCI.
    
    The model allows for different learning rates for positive and negative 
    prediction errors. The OCI score acts as a bias parameter, shifting sensitivity 
    towards negative prediction errors (punishment/loss of reward sensitivity).
    
    Parameters:
    lr_base: [0, 1] - Base learning rate.
    beta: [0, 10] - Inverse temperature.
    asymmetry_param: [0, 1] - Determines the split between pos/neg learning rates.
    oci_bias: [0, 1] - How much OCI biases learning towards negative errors.
    """
    lr_base, beta, asymmetry_param, oci_bias = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate specific learning rates
    # We use asymmetry_param to define a baseline ratio, then OCI shifts it.
    # If OCI is high, we increase lr_neg and decrease lr_pos (anxiety/avoidance hypothesis).
    
    # Base split
    alpha_pos = lr_base * asymmetry_param
    alpha_neg = lr_base * (1 - asymmetry_param)
    
    # Apply OCI modulation: High OCI amplifies negative learning
    modulation = oci_bias * oci_score
    alpha_pos = np.clip(alpha_pos - (modulation * 0.5), 0.0, 1.0)
    alpha_neg = np.clip(alpha_neg + (modulation * 0.5), 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        
        # --- Stage 1 Decision (Simple MF) ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0
            
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        if s_idx != -1:
            exp_q2 = np.exp(beta * q_stage2[s_idx])
            probs_2 = exp_q2 / np.sum(exp_q2)
            
            a2 = int(action_2[trial])
            if a2 != -1:
                p_choice_2[trial] = probs_2[a2]
                
                # --- Learning ---
                r = reward[trial]
                
                # Stage 2 Update
                delta2 = r - q_stage2[s_idx, a2]
                lr_2 = alpha_pos if delta2 > 0 else alpha_neg
                q_stage2[s_idx, a2] += lr_2 * delta2
                
                # Stage 1 Update (TD-0)
                if a1 != -1:
                    # Using the value of the state reached as the target
                    v_state2 = np.max(q_stage2[s_idx])
                    delta1 = v_state2 - q_stage1[a1]
                    lr_1 = alpha_pos if delta1 > 0 else alpha_neg
                    q_stage1[a1] += lr_1 * delta1
            else:
                p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
