Here are three cognitive models designed to explain the participant's behavior in the two-step task, incorporating the OCI-R score to modulate learning and decision-making parameters.

### Model 1: Hybrid Model with OCI-modulated Model-Based Weight (`w`)
This model assumes that individuals with different levels of obsessive-compulsive symptoms rely differently on Model-Based (planning) vs. Model-Free (habitual) systems. The parameter `w` controls the balance between these systems. Here, `w` is not a free parameter but is derived directly from the OCI score (normalized), allowing the OCI score to dictate the degree of goal-directed planning.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model where the balance between Model-Based (MB) and Model-Free (MF) 
    control is modulated by the OCI score.
    
    The mixing weight 'w' is determined by the OCI score. Higher OCI implies 
    potentially more rigid (habitual/MF) or more deliberative (MB) control depending 
    on the hypothesis. Here we model OCI as enhancing Model-Based control (hyper-planning).
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - eligibility_trace: [0, 1] Decay factor for eligibility traces (lambda).
    """
    learning_rate, beta, eligibility_trace = model_parameters
    n_trials = len(action_1)
    
    # Normalize OCI to be a weight between 0 and 1. 
    # OCI input is 0.55. We clamp it to ensure stability.
    # Hypothesis: Higher OCI -> Higher Model-Based control (w).
    w = np.clip(oci[0], 0.0, 1.0)
    
    # Transition matrix (fixed for this task structure)
    # A -> X (0.7), A -> Y (0.3); U -> X (0.3), U -> Y (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1 (A, U)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (State X/Y, Alien 1/2)
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: # Skip missing data
            continue
            
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation: V_MB(s1) = T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])
        if s_idx == -1: continue

        # --- Stage 2 Decision ---
        
        # Standard Model-Free choice at Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # Prediction errors
        # Stage 1 PE (SARSA-style): Q(s2, a2) - Q(s1, a1)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Stage 2 PE: R - Q(s2, a2)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 1 MF value
        # Uses eligibility trace to allow stage 2 outcome to influence stage 1 value
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * eligibility_trace * delta_stage2
        
        # Update Stage 2 MF value
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    # Filter out trials where probabilities are 0 (due to missing data logic)
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    
    return log_loss
```

### Model 2: OCI-Modulated Learning Rate Asymmetry
This model focuses on how rewards vs. punishments (omission of reward) are processed. Obsessive-compulsive traits are often associated with harm avoidance or altered sensitivity to negative feedback. This model splits the learning rate into positive (`alpha_pos`) and negative (`alpha_neg`) components, where the negative learning rate is scaled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with asymmetric learning rates for positive and negative prediction errors.
    The OCI score scales the sensitivity to negative prediction errors (learning from non-reward).
    
    Parameters:
    - alpha_base: [0, 1] Base learning rate for positive outcomes.
    - beta: [0, 10] Inverse temperature.
    - perserverance: [0, 5] Choice stickiness bonus (repeating previous choice).
    """
    alpha_base, beta, perseverance = model_parameters
    n_trials = len(action_1)
    
    # OCI Scaling: High OCI amplifies learning from negative outcomes (0 reward).
    # We map OCI (0-1) to a multiplier.
    oci_val = oci[0]
    alpha_neg = alpha_base * (1.0 + oci_val) 
    # Ensure alpha_neg stays within bounds [0, 1] logic usually, 
    # but here we just cap it at 1.0 for stability.
    alpha_neg = min(alpha_neg, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 Choice ---
        # Add perseverance bonus to the previously chosen action
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += perseverance

        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])
        if s_idx == -1: continue

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # TD(0) updates
        # Stage 2 PE
        delta2 = r - q_stage2[s_idx, a2]
        
        # Select learning rate based on sign of PE
        lr_2 = alpha_base if delta2 >= 0 else alpha_neg
        q_stage2[s_idx, a2] += lr_2 * delta2
        
        # Stage 1 PE (using updated stage 2 value as proxy for V(s'))
        # Note: Simple TD(0) often uses V(s') = max(Q(s', a')), here using Q(s', chosen)
        delta1 = q_stage2[s_idx, a2] - q_stage1[a1]
        lr_1 = alpha_base if delta1 >= 0 else alpha_neg
        q_stage1[a1] += lr_1 * delta1
        
        last_action_1 = a1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Temperature (Exploration/Exploitation)
High OCI is often associated with uncertainty intolerance. This model hypothesizes that OCI modulates the inverse temperature parameter (`beta`). A higher OCI score leads to a higher `beta`, resulting in more deterministic (exploitative) behavior and less random exploration, as the participant tries to rigidly adhere to the perceived "correct" rule.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Based model where the exploration-exploitation trade-off (beta)
    is modulated by the OCI score.
    
    Hypothesis: Higher OCI leads to more deterministic choice behavior (higher beta),
    reflecting a rigidity or intolerance of uncertainty.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta_base: [0, 10] Base inverse temperature.
    - w: [0, 1] Weight for Model-Based control (fixed parameter here, unlike Model 1).
    """
    learning_rate, beta_base, w = model_parameters
    n_trials = len(action_1)
    
    # Modulate beta with OCI. 
    # OCI (0.55) acts as a multiplier to increase rigidity.
    # Effective Beta = beta_base * (1 + OCI)
    effective_beta = beta_base * (1.0 + oci[0])
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 ---
        # Model-Based calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q-value
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Choice using OCI-modulated beta
        exp_q1 = np.exp(effective_beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])
        if s_idx == -1: continue

        # --- Stage 2 ---
        exp_q2 = np.exp(effective_beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # Update Stage 2 MF
        delta2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta2
        
        # Update Stage 1 MF (TD(1) logic simplified: update S1 based on S2 outcome)
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
