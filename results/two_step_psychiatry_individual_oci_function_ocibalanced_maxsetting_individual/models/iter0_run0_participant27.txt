Here are the three cognitive models as Python functions.

### Model 1: Hybrid MB/MF with OCI-Modulated Stickiness
This model assumes that the participant's Obsessive-Compulsive Inventory (OCI) score correlates with "stickiness" or choice perseverance. The `stickiness` parameter is scaled by `(1 + oci)`, implying that higher OCI scores lead to a stronger tendency to repeat previous choices regardless of reward history.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free model with OCI-modulated stickiness.
    
    Hypothesis: Higher OCI scores increase choice perseverance (stickiness).
    Stickiness is calculated as: stickiness_base * (1 + oci).
    
    Parameters:
    learning_rate: [0,1] - Rate of updating Q-values.
    beta: [0,10] - Inverse temperature for softmax choice policy.
    w: [0,1] - Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    stickiness_base: [0,5] - Base tendency to repeat the previous Stage 1 choice.
    lam: [0,1] - Eligibility trace parameter (updates Stage 1 based on Stage 2 RPE).
    """
    learning_rate, beta, w, stickiness_base, lam = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Scale stickiness by OCI score
    stickiness = stickiness_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # [State, Alien]

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Handle missing/invalid data (encoded as -1)
        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value: Weighted sum of MB and MF + Stickiness
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Pure Model-Free for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 Prediction Error (TD(0)) using Q(s2, a2) as target
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace: Update Stage 1 MF value based on Stage 2 outcome
        q_stage1_mf[a1] += learning_rate * lam * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with OCI-Modulated Mixing Weight
This model posits that OCI symptoms interfere with Goal-Directed (Model-Based) control. The mixing weight `w` is scaled down by the OCI score: `w = w_raw * (1 - oci)`. A higher OCI score results in a lower effective `w`, shifting behavior toward Model-Free (Habitual) control.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free model with OCI-modulated mixing weight.
    
    Hypothesis: Higher OCI scores reduce Model-Based control.
    Effective weight w is calculated as: w_raw * (1 - oci).
    
    Parameters:
    learning_rate: [0,1] - Rate of updating Q-values.
    beta: [0,10] - Inverse temperature for softmax choice.
    w_raw: [0,1] - Base mixing weight parameter.
    stickiness: [0,5] - Tendency to repeat the previous Stage 1 choice.
    lam: [0,1] - Eligibility trace parameter.
    """
    learning_rate, beta, w_raw, stickiness, lam = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Scale mixing weight by OCI (Higher OCI -> Less Model-Based)
    w = w_raw * (1.0 - oci_score)
    # Ensure w stays non-negative (OCI is usually < 1, but clipping is safe)
    if w < 0: w = 0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * lam * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning Rate Hybrid Model with OCI Stickiness
This model introduces asymmetric learning rates for positive and negative prediction errors (`lr_pos` and `lr_neg`). This captures potential biases in how the participant processes rewards versus omissions (e.g., ignoring 0-coin outcomes). It also includes the OCI-modulated stickiness from Model 1.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with asymmetric learning rates and OCI-modulated stickiness.
    
    Hypothesis: Participant learns differently from positive vs negative errors,
    and OCI scales stickiness.
    
    Parameters:
    lr_pos: [0,1] - Learning rate for positive prediction errors.
    lr_neg: [0,1] - Learning rate for negative prediction errors.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    stickiness_base: [0,5] - Base stickiness (scaled by OCI).
    """
    lr_pos, lr_neg, beta, w, stickiness_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    stickiness = stickiness_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        alpha_1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[a1] += alpha_1 * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        alpha_2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[s_idx, a2] += alpha_2 * delta_stage2
        
        # Eligibility Trace Update (Assume lambda=1 for full propagation)
        # Using alpha_2 corresponds to the learning rate of the event that caused the error
        q_stage1_mf[a1] += alpha_2 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
