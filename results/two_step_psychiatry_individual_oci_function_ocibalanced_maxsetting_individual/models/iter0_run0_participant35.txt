Here are three cognitive models formulated as Python functions. They all build upon the standard hybrid reinforcement learning framework (Model-Based + Model-Free) but incorporate the Obsessive-Compulsive Inventory (OCI) score in distinct, theoretically motivated ways.

### Model 1: OCI-Modulated Model-Based Weight
This model hypothesizes that high OCI scores correlate with a deficit in Model-Based (goal-directed) control. The parameter `w` (the weight given to the model-based system) is not fixed but is a function of the participant's OCI score. Higher OCI reduces the reliance on the transition matrix structure, shifting control toward the Model-Free (habitual) system.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Model-Based Weighting.
    
    Hypothesis: High OCI symptoms lead to reduced Model-Based (goal-directed) control 
    and increased reliance on Model-Free (habitual) control.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature (exploration/exploitation trade-off).
    - w_intercept: [0, 1] Baseline Model-Based weight for a participant with 0 OCI.
    - w_slope: [0, 1] The rate at which OCI reduces the Model-Based weight.
      Effective w = w_intercept - (w_slope * oci). Clipped to [0, 1].
    """
    learning_rate, beta, w_intercept, w_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective mixing weight based on OCI
    # High OCI reduces w, leading to more Model-Free behavior
    w = w_intercept - (w_slope * oci_score)
    if w < 0: w = 0.0
    if w > 1: w = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for Step 1 (Spaceships)
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Step 2 (Aliens per Planet)

    for trial in range(n_trials):
        # Handle missing data or timeouts (marked as -1 in data)
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0 # Neutral placeholder
            p_choice_2[trial] = 1.0
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- STEP 1 POLICY ---
        # Model-Based valuation: Transition Matrix * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q-value: Weighted sum of MB and MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STEP 2 POLICY ---
        # Standard Model-Free Softmax on the arrived planet
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        # SARSA-style update for Stage 1 (using the value of the state actually reached)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Prediction Error update for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Perseveration (Stickiness)
This model hypothesizes that OCI is linked to "stickiness" or perseverationâ€”a compulsion to repeat the previous action regardless of the outcome. The model adds a "stickiness" bonus to the Q-value of the previously chosen spaceship. The magnitude of this bonus is scaled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Perseveration (Stickiness).
    
    Hypothesis: High OCI scores are associated with compulsive repetition.
    This model adds a 'stickiness' bonus to the previously chosen action.
    The magnitude of this stickiness is determined by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1] Fixed balance between Model-Based and Model-Free.
    - stick_base: [0, 5] Baseline tendency to repeat choices.
    - stick_oci: [0, 5] Additional stickiness added per unit of OCI.
    """
    learning_rate, beta, w, stick_base, stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate total stickiness bonus
    # High OCI = higher bonus for repeating the previous action
    stickiness_bonus = stick_base + (stick_oci * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):
        if action_1[trial] == -1:
            last_action_1 = -1
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- STEP 1 POLICY ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus to the Q-value of the action taken in the previous trial
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_bonus

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update tracker for next trial
        last_action_1 = a1

        # --- STEP 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Inverse Temperature (Rigidity)
This model hypothesizes that high OCI scores relate to "rigidity" or an intolerance of uncertainty, manifesting as a higher inverse temperature ($\beta$). A higher $\beta$ makes the softmax function sharper, meaning the participant is more likely to deterministically choose the option with the slightly higher value, exploring less.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Inverse Temperature (Beta).
    
    Hypothesis: High OCI scores are associated with behavioral rigidity and 
    reduced exploration. This is modeled by increasing the inverse temperature (beta)
    as a function of the OCI score.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta_base: [0, 10] Baseline inverse temperature.
    - beta_oci: [0, 10] Additional beta added per unit of OCI.
    - w: [0, 1] Mixing weight (MB vs MF).
    """
    learning_rate, beta_base, beta_oci, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta
    # High OCI -> Higher Beta -> Sharper/More rigid choices (less exploration)
    beta_eff = beta_base + (beta_oci * oci_score)
    # Cap beta to prevent overflow in exp(), though bounds usually handle this
    if beta_eff > 20: beta_eff = 20 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- STEP 1 POLICY ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Use OCI-modulated beta
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STEP 2 POLICY ---
        # Use OCI-modulated beta here as well
        exp_q2 = np.exp(beta_eff * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
