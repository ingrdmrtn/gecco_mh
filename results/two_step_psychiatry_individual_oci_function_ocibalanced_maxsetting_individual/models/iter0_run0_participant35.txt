Here are three cognitive models designed to explain the participant's behavior, incorporating their high OCI-R score (0.75) which suggests a tendency towards habit formation or rigid model-free learning over flexible model-based planning.

### Model 1: OCI-Modulated Model-Based/Model-Free Hybrid
This model hypothesizes that the balance between Model-Based (planning) and Model-Free (habit) systems is directly influenced by the OCI score. Higher OCI scores shift the weight `w` towards Model-Free control, making the participant less sensitive to transition structures and more reliant on simple reward history.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free Reinforcement Learning where the mixing weight
    is modulated by the OCI score.
    
    Hypothesis: High OCI participants rely more on Model-Free (habitual) control.
    The mixing weight 'w' (0=MF, 1=MB) is scaled down by the OCI score.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1] (Base mixing weight before OCI modulation)
    """
    learning_rate, beta, w_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective mixing weight. 
    # If OCI is high, w shifts towards 0 (Model-Free).
    # We model this as w_base scaled by (1 - oci_score).
    w = w_base * (1.0 - oci_score)

    # Fixed transition matrix (A->X=0.7, U->Y=0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_mf1 = np.zeros(2)      # Model-free values for stage 1
    q_mb1 = np.zeros(2)      # Model-based values for stage 1
    q_mf2 = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)

    for trial in range(n_trials):
        # Skip invalid trials (indicated by -1 in data)
        if action_1[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Calculate Model-Based values: T * max(Q2)
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        # Integrated Q-value
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 RPE
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2
        
        # Stage 1 RPE (SARSA-style update for MF)
        delta_1 = q_mf2[s2, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * delta_1
        
        # Eligibility trace: Stage 1 also learns from Stage 2 outcome
        q_mf1[a1] += learning_rate * delta_2 # lambda=1 assumption for simplicity

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Stickiness (Perseveration)
This model posits that high OCI scores manifest as behavioral rigidity or "stickiness." Regardless of reward outcomes, the participant feels a compulsion to repeat the previous action. The OCI score scales a perseverance parameter added to the Q-values.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Free Reinforcement Learning with OCI-modulated Perseveration.
    
    Hypothesis: High OCI leads to 'stickiness' (compulsion to repeat choices),
    regardless of reward. The OCI score amplifies the perseverance bonus.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    pers_weight: [0, 5] (Base perseverance weight)
    """
    learning_rate, beta, pers_weight = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # The effective stickiness is the base parameter scaled by the OCI score.
    # Higher OCI -> Higher tendency to repeat the last action.
    effective_k = pers_weight * (1.0 + oci_score)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Add perseverance bonus to the Q-value of the previously chosen action
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += effective_k
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 values
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Standard TD Learning (Model-Free)
        
        # Stage 2 update
        delta_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_2
        
        # Stage 1 update
        # Using the value of the state arrived at (s2) to update action (a1)
        # We use max Q of next state as proxy for value of state s2
        v_s2 = np.max(q_stage2[s2])
        delta_1 = v_s2 - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Learning Rate Asymmetry
This model suggests that high OCI participants have biased learning from positive vs. negative outcomes. Specifically, they might be hyper-sensitive to negative outcomes (missing gold) or rigid in updating beliefs. Here, the OCI score adjusts the learning rate specifically for negative prediction errors (disappointments), reflecting an anxiety-driven learning process.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with Asymmetric Learning Rates modulated by OCI.
    
    Hypothesis: High OCI participants may over-weight negative prediction errors
    (failures to get gold) due to perfectionism/anxiety.
    The OCI score increases the learning rate for negative prediction errors (alpha_neg).
    
    Bounds:
    alpha_pos: [0, 1] (Learning rate for positive RPE)
    alpha_neg_base: [0, 1] (Base learning rate for negative RPE)
    beta: [0, 10]
    """
    alpha_pos, alpha_neg_base, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate.
    # We assume OCI increases sensitivity to negative outcomes.
    # We clamp it at 1.0 to ensure stability.
    alpha_neg = min(1.0, alpha_neg_base * (1.0 + oci_score))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        
        # Stage 2 Update
        delta_2 = r - q_stage2[s2, a2]
        lr2 = alpha_pos if delta_2 >= 0 else alpha_neg
        q_stage2[s2, a2] += lr2 * delta_2
        
        # Stage 1 Update
        # Using the value of the state arrived at (s2)
        v_s2 = np.max(q_stage2[s2])
        delta_1 = v_s2 - q_stage1[a1]
        
        # Note: We apply the asymmetry to the first stage update as well
        lr1 = alpha_pos if delta_1 >= 0 else alpha_neg
        q_stage1[a1] += lr1 * delta_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
