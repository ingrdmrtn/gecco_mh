Here are three cognitive models formulated as Python functions. These models incorporate the Obsessive-Compulsive Inventory (OCI) score to explain the participant's decision-making, specifically focusing on the high OCI score (0.866) and the observed behavior (strong perseveration/habitual choice of Spaceship 0).

### Model 1: OCI-Modulated Habitual Control
**Theory**: High OCI scores are associated with a shift from goal-directed (Model-Based) to habitual (Model-Free) control. This model eliminates the free parameter for the weighting between these systems, instead defining the weight of the Model-Free system directly by the OCI score. For this participant ($OCI \approx 0.87$), the model enforces a strong reliance on habitual value updating (Model-Free) over the transition structure (Model-Based).

```python
import numpy as np

def cognitive_model1_oci_habit(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Habitual Control.
    Hypothesis: The balance between Model-Based (MB) and Model-Free (MF) control is determined 
    by the OCI score. High OCI leads to dominance of the MF system.
    
    The mixing weight 'w' is set to the OCI score:
    Q_net = (OCI * Q_MF) + ((1 - OCI) * Q_MB)
    
    Parameters:
    - learning_rate: [0, 1] Rate of value updating.
    - beta: [0, 10] Inverse temperature (softness of choice).
    """
    learning_rate, beta = model_parameters
    oci_score = oci[0]
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    for t in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Expected value of best stage 2 option given transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted mix fixed by OCI
        # High OCI -> High MF weight
        w_mf = oci_score
        w_mb = 1.0 - oci_score
        q_net_stage1 = (w_mf * q_stage1_mf) + (w_mb * q_stage1_mb)
        
        # Softmax Policy Stage 1
        logits_1 = beta * q_net_stage1
        logits_1 = logits_1 - np.max(logits_1) # Stability
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        # --- Stage 2 Choice ---
        current_state = state[t]
        # Softmax Policy Stage 2 (Pure MF based on direct reward history)
        logits_2 = beta * q_stage2_mf[current_state]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        # --- Updates ---
        # Stage 1 MF Update (SARSA-style using Q-value of state 2)
        delta_stage1 = q_stage2_mf[current_state, action_2[t]] - q_stage1_mf[action_1[t]]
        q_stage1_mf[action_1[t]] += learning_rate * delta_stage1
        
        # Stage 2 MF Update (Prediction Error using Reward)
        delta_stage2 = reward[t] - q_stage2_mf[current_state, action_2[t]]
        q_stage2_mf[current_state, action_2[t]] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Scaled Perseveration
**Theory**: Compulsivity is often characterized by repetitive behavior ("stickiness"). This model introduces a perseveration bonus that is scaled by the OCI score. A high OCI score amplifies the tendency to repeat the previous action, regardless of reward history. This helps explain the participant's long streaks of choosing Spaceship 0 even when rewards are sparse.

```python
import numpy as np

def cognitive_model2_oci_sticky(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Scaled Perseveration.
    Hypothesis: High OCI drives "stickiness" (choice repetition).
    A perseveration bonus is added to the previously chosen action's value.
    The magnitude of this bonus is proportional to the OCI score.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w_mb: [0, 1] Weight of Model-Based system (0=MF, 1=MB).
    - stickiness_coef: [0, 5] Base stickiness. Effective bonus = stickiness_coef * OCI.
    """
    learning_rate, beta, w_mb, stickiness_coef = model_parameters
    oci_score = oci[0]
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    prev_action_1 = -1
    
    for t in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Standard Hybrid Valuation
        q_net_stage1 = (w_mb * q_stage1_mb) + ((1 - w_mb) * q_stage1_mf)
        
        # Calculate logits
        logits_1 = beta * q_net_stage1
        
        # Add OCI-dependent Perseveration Bonus
        if t > 0:
            # If OCI is high, this bonus is large
            bonus = stickiness_coef * oci_score
            logits_1[prev_action_1] += bonus
            
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        # Track previous choice
        prev_action_1 = action_1[t]
        
        # --- Stage 2 Choice ---
        current_state = state[t]
        logits_2 = beta * q_stage2_mf[current_state]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[current_state, action_2[t]] - q_stage1_mf[action_1[t]]
        q_stage1_mf[action_1[t]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[t] - q_stage2_mf[current_state, action_2[t]]
        q_stage2_mf[current_state, action_2[t]] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Constrained Learning Rigidity
**Theory**: High OCI is modeled as cognitive rigidity, represented by a reduced ability to update value expectations (low learning rate). This model constrains the effective learning rate using the OCI score. For a high OCI participant, the learning rate is forced to be low, explaining why established preferences (like choosing Spaceship 0) persist even when they yield no reward for many trials.

```python
import numpy as np

def cognitive_model3_oci_rigidity(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Constrained Learning Rigidity.
    Hypothesis: High OCI scores constrain the learning rate, representing cognitive rigidity.
    The effective learning rate is scaled down by the OCI score:
    alpha_effective = alpha_param * (1 - OCI).
    
    With OCI ~0.87, the learning rate is capped at ~13% of the base parameter,
    preventing rapid unlearning of established habits.
    
    Parameters:
    - alpha_param: [0, 1] Base learning rate parameter.
    - beta: [0, 10] Inverse temperature.
    - w_mb: [0, 1] Weight of Model-Based system.
    """
    alpha_param, beta, w_mb = model_parameters
    oci_score = oci[0]
    
    # Constrain learning rate: High OCI -> Low Alpha
    learning_rate = alpha_param * (1.0 - oci_score)
    # Ensure non-zero learning for stability
    learning_rate = max(learning_rate, 0.001)
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    for t in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = (w_mb * q_stage1_mb) + ((1 - w_mb) * q_stage1_mf)
        
        logits_1 = beta * q_net_stage1
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[t] = probs_1[action_1[t]]
        
        # --- Stage 2 Choice ---
        current_state = state[t]
        logits_2 = beta * q_stage2_mf[current_state]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[t] = probs_2[action_2[t]]
        
        # --- Updates ---
        # Use the OCI-constrained learning rate
        delta_stage1 = q_stage2_mf[current_state, action_2[t]] - q_stage1_mf[action_1[t]]
        q_stage1_mf[action_1[t]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[t] - q_stage2_mf[current_state, action_2[t]]
        q_stage2_mf[current_state, action_2[t]] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
