Here are three cognitive models formulated as Python functions. Each model incorporates the OCI score to explain specific deviations in decision-making processes associated with obsessive-compulsive traits, such as reduced model-based control, perseveration, or behavioral rigidity.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL with OCI-modulated mixing weight.
    
    Hypothesis: High OCI scores are associated with deficits in Model-Based (goal-directed) control.
    This model posits that the weight 'w' assigned to the Model-Based system decreases
    as the OCI score increases, leading to a dominance of Model-Free (habitual) control.
    
    Parameters:
    - learning_rate: Learning rate for value updates [0, 1]
    - beta: Inverse temperature for softmax choice policy [0, 10]
    - w_scale: Scaling factor for the Model-Based weight. w = w_scale * (1 - oci) [0, 1]
    - lambd: Eligibility trace parameter connecting Stage 2 outcome to Stage 1 value [0, 1]
    """
    learning_rate, beta, w_scale, lambd = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate mixing weight based on OCI
    # A high OCI score reduces w, shifting control to the Model-Free system.
    w = w_scale * (1.0 - oci_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2) # MF values for Spaceships [A, U]
    q_stage2_mf = np.zeros((2, 2)) # MF values for Planets [X, Y] -> Aliens [0, 1]

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Calculate Model-Based values using the transition matrix and max Stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values using the OCI-modulated weight
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 MF values
        qs2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Value Updates ---
        # Stage 1 TD error (SARSA-style)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 TD error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace: Propagate Stage 2 reward prediction error back to Stage 1
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with OCI-modulated Perseveration (Stickiness).
    
    Hypothesis: High OCI scores are associated with "compulsivity" or stickiness,
    defined as a tendency to repeat the previous action regardless of reward history.
    This model adds a 'stickiness' bonus to the previously chosen spaceship,
    with the magnitude of the bonus proportional to the OCI score.
    
    Parameters:
    - learning_rate: Learning rate [0, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight between MB and MF systems [0, 1]
    - stick_scale: Scaling factor for perseveration. Bonus = stick_scale * oci [0, 5]
    - lambd: Eligibility trace parameter [0, 1]
    """
    learning_rate, beta, w, stick_scale, lambd = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate stickiness bonus based on OCI
    stickiness_bonus = stick_scale * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits and add stickiness bonus to the previously chosen action
        logits = beta * q_net
        if prev_a1 != -1:
            logits[prev_a1] += stickiness_bonus
            
        # Softmax with numerical stability
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        prev_a1 = a1

        # --- Stage 2 Policy ---
        qs2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with OCI-modulated Inverse Temperature (Rigidity).
    
    Hypothesis: High OCI scores are linked to behavioral rigidity or anxiety-driven 
    reduction in exploration. This model posits that the inverse temperature (beta),
    which controls the determinism of choices, increases with the OCI score.
    
    Parameters:
    - learning_rate: Learning rate [0, 1]
    - beta_base: Baseline inverse temperature [0, 10]
    - beta_slope: Increase in beta per unit of OCI. beta = base + slope * oci [0, 10]

    - lambd: Eligibility trace parameter [0, 1]
    """
    learning_rate, beta_base, beta_slope, w, lambd = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate dynamic beta based on OCI
    # Higher OCI -> Higher beta -> More deterministic/rigid choices
    beta = beta_base + beta_slope * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
```
