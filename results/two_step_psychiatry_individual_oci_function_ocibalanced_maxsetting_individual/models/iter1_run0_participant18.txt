Here are three new cognitive models that explore different mechanisms by which Obsessive-Compulsive symptoms (OCI) might influence reinforcement learning in this two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that OCI symptoms influence the trade-off between Model-Based (planning) and Model-Free (habitual) control. High OCI is often associated with rigid, habitual behavior. This model tests if higher OCI scores lead to a lower weighting parameter ($w$) for the Model-Based system, effectively pushing the participant towards Model-Free strategies.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Based/Model-Free Balance.
    
    This model implements a hybrid Model-Based (MB) and Model-Free (MF) learner.
    The weighting parameter 'w', which controls the mix between MB and MF values,
    is modulated by the participant's OCI score.
    
    Hypothesis: Higher OCI scores might reduce the reliance on model-based planning
    (lower w), favoring habitual (MF) control.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w_intercept: [0, 1] Base weighting for Model-Based control (at OCI=0).
    - w_oci_factor: [0, 1] Factor reducing w based on OCI score.
      (w = w_intercept - w_oci_factor * oci)
    - lambda_eligibility: [0, 1] Eligibility trace for Stage 1 MF update.
    """
    learning_rate, beta, w_intercept, w_oci_factor, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w, bounded between 0 and 1
    # We subtract because the hypothesis is High OCI -> Lower MB usage
    w = w_intercept - (w_oci_factor * oci_score)
    w = np.clip(w, 0.0, 1.0)

    # Fixed transition matrix (A->X=0.7, U->Y=0.7)
    # Row 0: Spaceship A (0), Row 1: Spaceship U (1)
    # Col 0: Planet X (0), Col 1: Planet Y (1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-Free values for Stage 1
    q_stage2 = np.zeros((2, 2))    # Values for Stage 2 (State, Action)
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        # Handle missing data
        if a1 == -1:
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value calculation: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF update using Stage 2 max value (TD-0 like)
        delta_1 = q_stage2[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
        
        # Stage 2 update
        delta_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_2
        
        # Eligibility trace: Update Stage 1 again based on Stage 2 RPE
        q_stage1_mf[a1] += learning_rate * lambda_eligibility * delta_2

    eps = 1e-10
    valid_trials = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Learning Rate Asymmetry (Pos/Neg)
This model posits that individuals with varying OCI levels might learn differently from positive versus negative outcomes. Specifically, it tests if OCI modulates the *ratio* or balance between learning from reward (positive prediction errors) versus lack of reward (negative prediction errors).

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Dependent Learning Rate Asymmetry.
    
    This is a pure Model-Free learner where the learning rate depends on whether
    the prediction error is positive (better than expected) or negative.
    The asymmetry is modulated by the OCI score.
    
    Hypothesis: OCI might correlate with over-sensitivity to errors (negative PE)
    or reassurance (positive PE).
    
    Parameters:
    - lr_base: [0, 1] Base learning rate.
    - lr_asym_oci: [0, 5] How strongly OCI affects the learning rate split.
      If prediction error > 0: lr = lr_base
      If prediction error < 0: lr = lr_base * (1 + lr_asym_oci * oci)
    - beta: [0, 10] Inverse temperature.
    - lambda_eligibility: [0, 1] Eligibility trace.
    """
    lr_base, lr_asym_oci, beta, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        if a1 == -1:
            continue

        # Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        # Determine effective learning rate based on PE sign
        # We calculate PE first to know the sign, but we apply it step by step
        
        # Stage 1 Update (TD error using Stage 2 Q-value)
        pe_1 = q_stage2[s2, a2] - q_stage1[a1]
        
        # We define the asymmetry on the FINAL outcome (Stage 2 PE) usually,
        # but here we apply the logic to the learning rate generally.
        # If PE is negative, we scale the learning rate by (1 + asym * oci).
        
        lr_1 = lr_base
        if pe_1 < 0:
            lr_1 = lr_base * (1.0 + lr_asym_oci * oci_score)
            lr_1 = np.clip(lr_1, 0, 1) # Ensure bounds
            
        q_stage1[a1] += lr_1 * pe_1
        
        # Stage 2 Update
        pe_2 = r - q_stage2[s2, a2]
        
        lr_2 = lr_base
        if pe_2 < 0:
            lr_2 = lr_base * (1.0 + lr_asym_oci * oci_score)
            lr_2 = np.clip(lr_2, 0, 1)

        q_stage2[s2, a2] += lr_2 * pe_2
        
        # Eligibility Trace Update (Stage 1 updated by Stage 2 PE)
        # We use the same lr_2 for consistency with the event causing the update
        q_stage1[a1] += lr_2 * lambda_eligibility * pe_2

    eps = 1e-10
    valid_trials = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: OCI-Based "Win-Stay, Lose-Shift" Hybrid
This model combines standard reinforcement learning with a heuristic "Win-Stay, Lose-Shift" (WSLS) strategy that is specifically modulated by OCI. It posits that high OCI scores might drive a stronger adherence to explicit rules like "if it worked, do it again," overriding or supplementing value-based learning.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Based "Win-Stay, Lose-Shift" Hybrid.
    
    This model assumes the policy is a mixture of a standard Q-learning policy
    and a heuristic Win-Stay, Lose-Shift (WSLS) bias. The strength of the 
    WSLS bias is modulated by the OCI score.
    
    Hypothesis: High OCI might lead to more rigid adherence to simple rules 
    (WSLS) regarding the previous outcome, regardless of the transition type.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-values.
    - beta: [0, 10] Inverse temperature for Q-values.
    - wsls_weight_base: [0, 5] Base weight added to the Q-value of the 'Stay' or 'Shift' option.
    - wsls_oci_mod: [0, 5] Modulation of WSLS weight by OCI.
      Total WSLS bonus = wsls_weight_base + (wsls_oci_mod * oci)
    """
    learning_rate, beta, wsls_weight_base, wsls_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    wsls_bonus = wsls_weight_base + (wsls_oci_mod * oci_score)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_reward = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        if a1 == -1:
            prev_a1 = -1
            continue
            
        # --- Stage 1 Policy ---
        # Start with Q-values
        logits_1 = beta * q_stage1
        
        # Apply WSLS Bias
        if prev_a1 != -1:
            if prev_reward == 1:
                # Win-Stay: Add bonus to the previous action
                logits_1[prev_a1] += wsls_bonus
            else:
                # Lose-Shift: Add bonus to the OTHER action (1 - prev_a1)
                logits_1[1 - prev_a1] += wsls_bonus
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        logits_2 = beta * q_stage2[s2]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates (Standard Q-Learning TD-0) ---
        # Stage 1
        # Note: We use max_a Q(s', a) logic for Stage 1 update here (Q-learning)
        # rather than SARSA to keep it distinct from previous models.
        max_q2 = np.max(q_stage2[s2])
        delta_1 = max_q2 - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1
        
        # Stage 2
        delta_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_2
        
        prev_a1 = a1
        prev_reward = r

    eps = 1e-10
    valid_trials = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
