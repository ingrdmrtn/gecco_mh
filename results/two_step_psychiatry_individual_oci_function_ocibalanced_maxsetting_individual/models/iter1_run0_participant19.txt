Here are 3 new cognitive models based on the provided participant data and OCI score.

### Cognitive Model 1: OCI-Modulated Learning Rate Asymmetry
This model hypothesizes that individuals with higher OCI scores might exhibit asymmetric learning from positive versus negative prediction errors. Specifically, high OCI might be associated with a "hyper-learning" from negative outcomes (avoidance) or a rigidity where negative feedback is weighted differently than positive feedback.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with asymmetric learning rates modulated by OCI.
    
    Hypothesis: OCI score modulates the sensitivity to negative prediction errors (punishment/lack of reward).
    Higher OCI might lead to stronger learning from negative outcomes.
    
    Parameters:
    lr_pos: Learning rate for positive prediction errors [0, 1]
    lr_neg_base: Base learning rate for negative prediction errors [0, 1]
    lr_neg_oci: Scaling factor for OCI effect on negative learning rate [0, 1]
    beta: Inverse temperature [0, 10]
    w: Mixing weight (0=MF, 1=MB) [0, 1]
    """
    lr_pos, lr_neg_base, lr_neg_oci, beta, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective negative learning rate, bounded at 1.0
    lr_neg = min(1.0, lr_neg_base + (lr_neg_oci * current_oci))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2) # Initial values 0.0
    q_stage2_mf = np.zeros((2, 2)) # Initial values 0.0

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        effective_lr_s2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[s_idx, a2] += effective_lr_s2 * delta_stage2
        
        # Stage 1 Update (TD-learning)
        # Note: Using the updated stage 2 value is common in some implementations (SARSA), 
        # but standard Q-learning uses max. Here we follow the template's implication of temporal difference.
        # The template implies: delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        # We use the value of the state actually visited.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        effective_lr_s1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[a1] += effective_lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Modulated Eligibility Trace (Lambda)
This model introduces an eligibility trace parameter ($\lambda$) that determines how much the Stage 2 reward directly updates the Stage 1 choice. The hypothesis is that OCI symptoms relate to how credit is assigned across the two stages. Higher OCI might correlate with a stronger direct link between the final outcome and the initial choice (higher lambda), reflecting a more rigid or direct causal attribution.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with an eligibility trace (lambda) modulated by OCI.
    
    Hypothesis: OCI affects the eligibility trace parameter lambda.
    High OCI might lead to stronger direct reinforcement of the first stage action 
    by the second stage reward (ignoring the transition structure).
    
    Parameters:
    lr: Learning rate [0, 1]
    beta: Inverse temperature [0, 10]
    w: Mixing weight (0=MF, 1=MB) [0, 1]
    lambda_base: Base eligibility trace parameter [0, 1]
    lambda_oci: OCI scaling factor for lambda [0, 1]
    """
    lr, beta, w, lambda_base, lambda_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective lambda, bounded [0, 1]
    # We use a tanh-like bounding or simple clipping. 
    # Since inputs are usually [0,1], simple clipping is safer.
    lambda_val = min(1.0, lambda_base + (lambda_oci * current_oci))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Stage 1 Prediction Error (TD(lambda))
        # Standard TD(0) part:
        delta_stage1_td0 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # The eligibility trace allows the stage 2 PE to also update stage 1
        # Update = lr * (delta_stage1_td0 + lambda * delta_stage2)
        q_stage1_mf[a1] += lr * (delta_stage1_td0 + lambda_val * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Random Exploration (Epsilon-Greedy + Softmax)
This model assumes that OCI affects the balance between directed exploitation (softmax) and random exploration (epsilon). High OCI might lead to reduced random exploration (lower epsilon) due to a desire for control or rigid adherence to the perceived best option, or conversely, anxiety might induce noise. Here we test if OCI modulates a "slip" or random noise parameter.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated random exploration (epsilon) added to Softmax.
    
    Hypothesis: OCI affects the 'epsilon' parameter, representing random lapses or 
    undirected exploration. We model this as a mixture of Softmax and Uniform random.
    
    Parameters:
    lr: Learning rate [0, 1]
    beta: Inverse temperature [0, 10]
    w: Mixing weight (0=MF, 1=MB) [0, 1]
    epsilon_base: Base random exploration probability [0, 1]
    epsilon_oci: OCI scaling factor for epsilon [0, 1]
    """
    lr, beta, w, epsilon_base, epsilon_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective epsilon, bounded [0, 1]
    # We allow epsilon_oci to be negative (reducing noise with higher OCI) 
    # effectively by how the optimizer uses the bounds, but here we assume additive.
    # To be safe against bounds, we clip.
    epsilon = np.clip(epsilon_base + (epsilon_oci * current_oci), 0.0, 0.99)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax probabilities
        exp_q1 = np.exp(beta * q_net)
        softmax_probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Mixture with uniform random (epsilon-greedy-like modification to softmax)
        # P(a) = (1-eps) * Softmax(a) + eps * (1/n_actions)
        probs_1 = (1 - epsilon) * softmax_probs_1 + epsilon * 0.5
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        softmax_probs_2 = exp_q2 / np.sum(exp_q2)
        probs_2 = (1 - epsilon) * softmax_probs_2 + epsilon * 0.5
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
