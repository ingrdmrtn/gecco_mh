Here are three new cognitive models that explore different mechanisms by which Obsessive-Compulsive Inventory (OCI) scores might influence decision-making in the two-step task. Given the participant's "sticky" behavior (long blocks of repeating the same choice), all models include a baseline perseveration parameter, while testing different hypotheses for the specific effect of OCI.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Learning Rate modulated by OCI, with fixed Perseveration.
    
    Hypothesis: OCI score affects the learning rate (plasticity/rigidity) of value updates,
    while accounting for the participant's general tendency to stick with choices (perseveration).
    
    Parameters:
    lr_base: [0, 1] Base learning rate.
    lr_oci_slope: [-1, 1] Effect of OCI on learning rate. (lr = lr_base + slope * oci)
    beta: [0, 10] Inverse temperature (exploration/exploitation).
    w: [0, 1] Weight for Model-Based values (0=MF, 1=MB).
    pers: [-1, 1] Perseveration bonus for repeating the previous choice.
    """
    lr_base, lr_oci_slope, beta, w, pers = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate modulated learning rate, clipped to [0, 1]
    lr = np.clip(lr_base + lr_oci_slope * current_oci, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Perseveration bonus
        if prev_action_1 != -1:
            q_net[prev_action_1] += pers
            
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        # Softmax Policy Stage 2 (pure MF)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Value Updates ---
        # Stage 1 Update (TD-learning using Stage 2 value)
        v_stage2 = q_stage2_mf[s_idx, a2]
        delta_stage1 = v_stage2 - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 Update (Reward learning)
        delta_stage2 = reward[trial] - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
