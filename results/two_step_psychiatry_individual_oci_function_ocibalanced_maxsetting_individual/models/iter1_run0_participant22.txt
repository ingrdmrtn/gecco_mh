Here are the three cognitive models based on the participant's data and OCI score.

### Model 1: OCI-Modulated Model-Based Weighting
**Hypothesis:** Obsessive-compulsive symptoms are often associated with a deficit in goal-directed (Model-Based) control and a reliance on habitual (Model-Free) control. This model posits that the mixing weight parameter $w$ (which balances MB and MF systems) is negatively modulated by the OCI score. Higher OCI scores reduce $w$, leading to more habitual behavior.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where the balance between Model-Based and Model-Free control (w)
    is modulated by the OCI score.
    
    Hypothesis: Higher OCI scores reduce the weight of the Model-Based system,
    biasing the participant towards Model-Free (habitual) strategies.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta_1: [0,10] Inverse temperature for Stage 1 choices.
    - beta_2: [0,10] Inverse temperature for Stage 2 choices.
    - w_base: [0,1] Baseline Model-Based weighting for a theoretical 0-OCI participant.
    - oci_penalty: [0,1] Amount w is reduced per unit of OCI score.
    """
    learning_rate, beta_1, beta_2, w_base, oci_penalty = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w, bounded between 0 and 1
    w_effective = w_base - (oci_penalty * oci_score)
    w_effective = np.clip(w_effective, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # MF values for Stage 1 (Actions: 0, 1)
    q_stage2_mf = np.zeros((2, 2)) # MF values for Stage 2 (States: 0, 1; Actions: 0, 1)
    
    for trial in range(n_trials):
        # Handle missing data
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]) or action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue
            
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        s_idx = int(state[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf
        
        # Softmax Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # SARSA / TD(0) update for Stage 1 MF
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(0) update for Stage 2 MF
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    # Filter valid trials for log loss calculation
    valid_mask = (action_1 != -1) & (action_2 != -1) & (~np.isnan(action_1))
    
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 2: OCI-Driven Rigid Exploration (Temperature Modulation)
**Hypothesis:** High OCI individuals often display rigid behavior and intolerance of uncertainty. In computational terms, this can be modeled as a higher inverse temperature ($\beta$) parameter, leading to "greedier" exploitation of current values and reduced random exploration. This model scales the Stage 1 $\beta$ based on the OCI score. It also includes a fixed choice stickiness parameter to account for the "chunking" seen in the data.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where choice rigidity (inverse temperature beta) is modulated by OCI.
    
    Hypothesis: Higher OCI scores lead to higher beta values (lower exploration),
    resulting in more deterministic/rigid choices based on current value estimates.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta_base: [0,10] Baseline inverse temperature.
    - beta_oci_scale: [0,5] Increase in beta per unit of OCI.
    - w: [0,1] MB/MF mixing weight.
    - stickiness: [0,5] Value added to the previously chosen action (perseveration).
    """
    learning_rate, beta_base, beta_oci_scale, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta for Stage 1
    beta_1_effective = beta_base + (beta_oci_scale * oci_score)
    # Stage 2 beta is kept equal to base for parsimony, or could be separate. 
    # Here we use the effective beta for both to represent general rigidity.
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        if np.isnan(action_1[trial]) or action_1[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue
            
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        s_idx = int(state[trial])
        r = reward[trial]
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness to the previously chosen action
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta_1_effective * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_1_effective * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Update Stage 1 MF
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    valid_mask = (action_1 != -1) & (action_2 != -1) & (~np.isnan(action_1))
    
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Eligibility Trace
**Hypothesis:** This model investigates the impact of OCI on credit assignment. The eligibility trace ($\lambda$) determines how much the outcome at Stage 2 reinforces the choice made at Stage 1. A high OCI might be associated with "magical thinking" or an over-connection of sequential events, resulting in a higher $\lambda$. This makes the Stage 1 MF values more sensitive to the final reward, bypassing the state-transition structure.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with an Eligibility Trace (lambda) modulated by OCI.
    
    Hypothesis: OCI modulates the strength of the eligibility trace, determining
    how strongly the final reward updates the Stage 1 choice directly (Model-Free).
    High OCI may lead to stronger direct reinforcement (higher lambda).
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta: [0,10] Inverse temperature (shared for both stages).
    - w: [0,1] MB/MF mixing weight.
    - lambda_base: [0,1] Baseline eligibility trace parameter.
    - lambda_oci_scale: [-1,1] Modulation of lambda by OCI.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective lambda, bounded [0,1]
    lambda_eff = lambda_base + (lambda_oci_scale * oci_score)
    lambda_eff = np.clip(lambda_eff, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        if np.isnan(action_1[trial]) or action_1[trial] == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue
            
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        s_idx = int(state[trial])
        r = reward[trial]
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # 1. Update Stage 1 based on Stage 2 value (Temporal Difference)
        delta_1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
        
        # 2. Update Stage 2 based on Reward
        delta_2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_2
        
        # 3. Eligibility Trace Update: Update Stage 1 based on Stage 2 Reward
        # This allows the final reward to 'leak' back to the first choice
        # scaled by lambda.
        q_stage1_mf[a1] += learning_rate * lambda_eff * delta_2

    eps = 1e-10
    valid_mask = (action_1 != -1) & (action_2 != -1) & (~np.isnan(action_1))
    
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```
