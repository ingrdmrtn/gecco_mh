Here are three new cognitive models that incorporate the OCI score to explain the participant's behavior in the two-step task.

### Model 1: OCI-Modulated Model-Based Weight
This model hypothesizes that OCI severity affects the balance between Model-Based (planning) and Model-Free (habitual) control. High OCI might be associated with rigid, habit-like behavior (lower `w`), or conversely, an over-reliance on explicit rules (higher `w`). This model allows the mixing weight `w` to be a logistic function of the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where the mixing weight w is modulated by OCI.
    
    Hypothesis: The balance between goal-directed (MB) and habitual (MF) control
    is a function of obsessive-compulsive traits.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_slope: [-5, 5] (Slope of OCI effect on w)
    w_intercept: [-5, 5] (Baseline logit for w)
    """
    learning_rate, beta, w_slope, w_intercept = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate w using a sigmoid function to keep it between 0 and 1
    # w represents the weight of Model-Based control
    w_logit = w_intercept + w_slope * current_oci
    w = 1 / (1 + np.exp(-w_logit))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
             p_choice_1[trial] = 0.5
             p_choice_2[trial] = 0.5
             continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Updates ---
        # Update Stage 2 Q-values (Model-Free)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (Model-Free)
        # Using the value of the state reached (q_stage2_mf) as the target
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Learning Rate Asymmetry
This model suggests that OCI affects how participants learn from positive versus negative prediction errors. Specifically, individuals with higher OCI might be hyper-sensitive to errors (punishment/lack of reward) or overly rigid in holding onto beliefs despite evidence. Here, the learning rate is split into a base rate and an OCI-dependent multiplier.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Free model where learning rate is scaled by OCI.
    
    Hypothesis: OCI scores influence the speed of updating value estimates. 
    Higher OCI might lead to faster (hyper-responsive) or slower (rigid) learning.
    
    Bounds:
    lr_base: [0, 1]
    beta: [0, 10]
    oci_lr_mod: [-1, 1] (Modifies learning rate based on OCI)
    eligibility_trace: [0, 1] (Parameter lambda for stage 1 update)
    """
    lr_base, beta, oci_lr_mod, eligibility_trace = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Effective learning rate constrained to [0, 1]
    # If oci_lr_mod is positive, higher OCI increases learning rate.
    # If negative, higher OCI decreases learning rate (rigidity).
    raw_lr = lr_base + (oci_lr_mod * current_oci)
    learning_rate = np.clip(raw_lr, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
             p_choice_1[trial] = 0.5
             p_choice_2[trial] = 0.5
             continue

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Updates ---
        # SARSA(lambda) style update for Stage 1
        # Delta 1: Prediction error at stage 1 transition
        delta1 = q_stage2[state_idx, act2] - q_stage1[act1]
        q_stage1[act1] += learning_rate * delta1
        
        # Delta 2: Prediction error at reward
        delta2 = reward[trial] - q_stage2[state_idx, act2]
        q_stage2[state_idx, act2] += learning_rate * delta2
        
        # Eligibility trace: Stage 1 value is also updated by stage 2 error
        q_stage1[act1] += learning_rate * eligibility_trace * delta2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Driven "Win-Stay, Lose-Shift" Heuristic Bias
This model proposes that high OCI participants may rely more heavily on a simple heuristic strategy (Win-Stay, Lose-Shift) on top of standard reinforcement learning. The strength of this heuristic bias is scaled by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with an OCI-scaled Win-Stay/Lose-Shift (WSLS) bias.
    
    Hypothesis: Participants with higher OCI rely more on explicit, simple rules 
    like 'repeat what worked' (WSLS) rather than probabilistic value integration.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] (MB/MF mixing weight)
    wsls_scale: [0, 5] (Strength of WSLS bias scaled by OCI)
    """
    learning_rate, beta, w, wsls_scale = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # WSLS bias strength depends on OCI
    wsls_strength = wsls_scale * current_oci

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
             p_choice_1[trial] = 0.5
             p_choice_2[trial] = 0.5
             continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # WSLS Heuristic Logic
        wsls_bonus = np.zeros(2)
        if last_action_1 != -1:
            if last_reward == 1:
                # Win-Stay: Bonus to previous action
                wsls_bonus[int(last_action_1)] = wsls_strength
            else:
                # Lose-Shift: Bonus to the OTHER action (or penalty to previous)
                # Here implemented as bonus to the alternative
                wsls_bonus[1 - int(last_action_1)] = wsls_strength

        exp_q1 = np.exp(beta * (q_net + wsls_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Updates ---
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        last_action_1 = act1
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
