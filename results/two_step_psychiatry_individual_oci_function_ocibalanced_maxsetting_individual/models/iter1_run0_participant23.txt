Here are three new cognitive models that incorporate the OCI score to explain participant behavior in the two-step task.

### Model 1: OCI-Modulated Model-Based Weighting
This model tests the hypothesis that the balance between Model-Based (planning) and Model-Free (habit) control is linearly modulated by the participant's OCI score. A common finding in psychiatry is that compulsivity relates to deficits in model-based control.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with OCI-modulated Model-Based Weight (w).

    Hypothesis: The degree of Model-Based control (w) is linearly modulated by the OCI score.
    Higher OCI scores may lead to reduced Model-Based control (Gillan et al., 2016).

    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - stickiness: [0, 5] Choice perseveration bonus.
    - w_base: [0, 1] Baseline Model-Based weight.
    - w_oci_slope: [-1, 1] Modulation of w by OCI score.
    """
    learning_rate, beta, stickiness, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate w based on OCI: w = w_base + slope * oci
    w = w_base + w_oci_slope * oci_score
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Keep track of previous choice for stickiness
    prev_a1 = -1

    for trial in range(n_trials):
        if int(action_1[trial]) == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += stickiness
            
        # Softmax for Choice 1
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # policy for the second choice
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Learning Rate
This model proposes that the OCI score affects the *learning rate* (volatility). Individuals with higher OCI scores might be more sensitive to prediction errors (higher learning rate) or more rigid (lower learning rate).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with OCI-modulated Learning Rate.

    Hypothesis: OCI score impacts the learning rate, reflecting altered sensitivity to 
    prediction errors (volatility) in both stages.

    Parameters:
    - lr_base: [0, 1] Baseline learning rate.
    - lr_oci_slope: [-1, 1] Modulation of learning rate by OCI.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - stickiness: [0, 5] Choice perseveration.
    """
    lr_base, lr_oci_slope, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Learning Rate based on OCI
    lr = lr_base + lr_oci_slope * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):

    return log_loss
```

### Model 3: OCI-Modulated Eligibility Trace
This model introduces an eligibility trace parameter (`lambda`) that allows the reward at the second stage to directly reinforce the first-stage choice. The strength of this "habitual" direct reinforcement is modulated by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with OCI-modulated Eligibility Trace (Lambda).

    Hypothesis: OCI score modulates the eligibility trace (lambda), determining the extent 
    to which outcomes reinforce the first-stage choice directly (habitual strength).

    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - lambda_base: [0, 1] Baseline eligibility trace.
    - lambda_oci_slope: [-1, 1] Modulation of lambda by OCI.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Lambda based on OCI
    lam = lambda_base + lambda_oci_slope * oci_score
    lam = np.clip(lam, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if int(action_1[trial]) == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # policy for the second choice
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Eligibility trace update: 
        # The reward from stage 2 also updates stage 1 via lambda (TD(lambda) approx)
        r_prediction_error = r - q_stage2_mf[s_idx, a2]
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lam * r_prediction_error
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```
