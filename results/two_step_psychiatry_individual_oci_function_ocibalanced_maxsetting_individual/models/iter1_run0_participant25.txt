Here are three cognitive models formulated as Python functions. They explore different mechanisms by which Obsessive-Compulsive symptoms (OCI) might influence the balance between goal-directed behavior, habit formation, and perseveration.

### Model 1: OCI-Modulated Inverse Temperature (Rigidity)
This model hypothesizes that OCI scores correlate with "behavioral rigidity" or a reduction in exploration. Instead of a fixed exploration parameter ($\beta$), the inverse temperature increases with the OCI score. A higher $\beta$ makes choices more deterministic (exploiting the current highest value), which explains the long blocks of repeated choices in the participant data.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Inverse Temperature (Rigidity).
    
    Hypothesis: Higher OCI scores lead to higher inverse temperature (beta),
    resulting in more deterministic/rigid choices (less exploration).
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate for value updates.
    - beta_base: [0, 10] Baseline inverse temperature.
    - beta_oci_slope: [0, 10] Increase in beta per unit of OCI.
      Effective Beta = beta_base + (beta_oci_slope * oci).
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    """
    learning_rate, beta_base, beta_oci_slope, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective beta based on OCI (bounded to prevent overflow)
    beta_eff = beta_base + (beta_oci_slope * current_oci)
    if beta_eff > 20: beta_eff = 20 # Safety bound
    if beta_eff < 0: beta_eff = 0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2) # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Stage 2
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax with OCI-modulated Beta
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF update (TD prediction error)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Asymmetric Learning with OCI-Dampened Punishment Sensitivity
The participant data shows persistence in choosing a spaceship even after receiving zero rewards (punishments). This model hypothesizes that OCI correlates with a resistance to negative feedback. High OCI dampens the learning rate specifically for negative prediction errors, causing the participant to "ignore" failures and persist in their current strategy.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Asymmetric Learning Model with OCI-Dampened Punishment Sensitivity.
    
    Hypothesis: OCI reduces the learning rate for negative prediction errors (punishments),
    leading to persistence despite lack of reward.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors (Reward).
    - lr_neg_base: [0, 1] Baseline learning rate for negative prediction errors (No Reward).
    - lr_neg_oci_damp: [0, 1] Factor by which OCI reduces negative learning rate.
      Effective lr_neg = lr_neg_base * (1 - (lr_neg_oci_damp * oci)).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    """
    lr_pos, lr_neg_base, lr_neg_oci_damp, beta, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective negative learning rate
    # Higher OCI -> Lower learning from failure
    lr_neg = lr_neg_base * (1.0 - (lr_neg_oci_damp * current_oci))
    if lr_neg < 0: lr_neg = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates with Asymmetric Learning Rates ---
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        eff_lr_1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[a1] += eff_lr_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        eff_lr_2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[state_idx, a2] += eff_lr_2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Choice Kernel (Dynamic Perseveration)
Unlike the simple "last-choice" stickiness, a "Choice Kernel" tracks the frequency of recent choices, decaying over time. This model assumes that OCI modulates the *weight* of this historical momentum. A high OCI score implies a stronger influence of past choice frequency on the current decision, explaining the "blocky" nature of the participant's choices.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Choice Kernel (Dynamic Perseveration).
    
    Hypothesis: OCI increases the influence of a 'Choice Kernel' (frequency-dependent stickiness),
    making the participant more likely to repeat choices they have made frequently in the recent past.
    
    Parameters:
    - learning_rate: [0, 1] Value learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - k_decay: [0, 1] Decay rate of the choice kernel (1 = no decay, 0 = instant forgetting).
    - k_weight_base: [0, 5] Baseline weight of the kernel in decision making.
    - k_weight_oci: [0, 5] Additional kernel weight scaled by OCI.
    """
    learning_rate, beta, w, k_decay, k_weight_base, k_weight_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective kernel weight
    k_weight_eff = k_weight_base + (k_weight_oci * current_oci)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice Kernel for Stage 1 (tracks history of choices)
    choice_kernel = np.zeros(2) 
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB, MF, and Choice Kernel
        # The Kernel provides a 'bonus' to options chosen frequently recently
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net + (k_weight_eff * choice_kernel)
        
        exp_q1 = np.exp(logits - np.max(logits)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # --- Kernel Update ---
        # Decay existing traces
        choice_kernel *= k_decay
        # Reinforce the chosen action
        choice_kernel[a1] += 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
