Here are three new cognitive models for the two-step task, incorporating the OCI score to modulate different mechanisms of decision-making.

### Cognitive Model 1
This model hypothesizes that OCI scores relate to the **inverse temperature (beta)**, reflecting the rigidity or randomness of choices (exploration-exploitation balance). A higher or lower OCI might lead to more deterministic (rigid) or more noisy behavior. It also includes a fixed choice perseveration parameter, as this is a strong feature of participant behavior in this task.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the Inverse Temperature (Beta).
    Includes fixed Choice Perseveration (p).
    
    Hypothesis: Obsessive-compulsive symptoms (OCI) affect the rigidity of choice 
    behavior (exploration vs exploitation).
    
    Parameters:
    learning_rate: [0,1] - Learning rate for value updates.
    w: [0,1] - Weighting between Model-Based (1) and Model-Free (0) systems.
    p: [0,1] - Choice perseveration strength (stickiness), scaled internally.
    beta_base: [0,10] - Baseline inverse temperature.
    beta_oci_mod: [0,1] - Modulation of beta by OCI score.
    """
    learning_rate, w, p_param, beta_base, beta_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Parameter transformation
    p = p_param * 5.0  # Scale p to be effective
    
    # Beta modulation: Allow OCI to increase or decrease beta relative to base
    # (beta_oci_mod - 0.5) allows range [-0.5, 0.5]
    beta = beta_base + (beta_oci_mod - 0.5) * 10.0 * oci_score
    beta = np.maximum(0.0, beta) # Ensure beta is non-negative
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Logits with Perseveration
        logits_1 = beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += p
            
        # Softmax Choice 1
        # Subtract max for numerical stability
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # SARSA-style update for Stage 1 MF value
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Prediction Error update for Stage 2 MF value
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2
This model hypothesizes that OCI scores modulate the **balance between Model-Based and Model-Free control (w)**. While similar combinations have been tried, this model explicitly includes a **fixed perseveration parameter (p)**, which is crucial for model fit and was absent in the "w-modulation" models tried previously.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the Weighting (w) between systems.
    Includes fixed Choice Perseveration (p).
    
    Hypothesis: OCI score shifts the reliance between goal-directed (MB) and 
    habitual (MF) systems.
    
    Parameters:
    learning_rate: [0,1] - Learning rate.
    beta: [0,10] - Inverse temperature.
    p: [0,1] - Choice perseveration strength.
    w_base: [0,1] - Base mixing weight.
    w_oci_mod: [0,1] - Modulation of w by OCI score.
    """
    learning_rate, beta, p_param, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Parameter transformation
    p = p_param * 5.0
    
    # W modulation: centered modulation
    w = w_base + (w_oci_mod - 0.5) * 2.0 * oci_score
    w = np.clip(w, 0.0, 1.0) # Ensure w stays in [0,1]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += p
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3
This model hypothesizes that OCI scores modulate the **learning rate (alpha)**. Compulsive individuals might update their beliefs faster or slower than others. Unlike previous attempts at learning rate modulation, this model includes the critical **perseveration parameter (p)** to account for the "stickiness" observed in the data.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the Learning Rate.
    Includes fixed Choice Perseveration (p).
    
    Hypothesis: OCI score affects how rapidly participants update values based on 
    prediction errors.
    
    Parameters:
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    p: [0,1] - Choice perseveration strength.
    lr_base: [0,1] - Base learning rate.
    lr_oci_mod: [0,1] - Modulation of learning rate by OCI score.
    """
    beta, w, p_param, lr_base, lr_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Parameter transformation
    p = p_param * 5.0
    
    # Learning Rate modulation
    lr = lr_base + (lr_oci_mod - 0.5) * 2.0 * oci_score
    lr = np.clip(lr, 0.0, 1.0) # Ensure lr stays in [0,1]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += p
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
