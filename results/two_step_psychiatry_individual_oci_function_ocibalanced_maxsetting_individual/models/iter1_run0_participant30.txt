Here are 3 new cognitive models based on the participant's data and OCI profile.

### Model 1: OCI-Modulated Habit Strength (Model-Free Dominance)
This model hypothesizes that high OCI scores are associated with an over-reliance on habit (Model-Free) learning, specifically by modulating the learning rate for the first-stage Model-Free values. Rather than just mixing values at decision time (like the `w` parameter in previous attempts), this model posits that the *formation* of Model-Free values is accelerated or decelerated by OCI traits, potentially making habits form faster or slower relative to goal-directed values.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Free Learning Rate.
    
    Hypothesis: The strength of Model-Free learning (habit formation) is 
    directly modulated by OCI severity. A base learning rate is modified 
    by the OCI score. This affects how quickly stage 1 MF values are updated.
    
    Bounds:
    lr_base: [0, 1] - Base learning rate
    lr_oci_mod: [0, 1] - Modulation of learning rate by OCI
    beta: [0, 10] - Inverse temperature
    w: [0, 1] - Fixed mixing weight between MB and MF
    """
    lr_base, lr_oci_mod, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective learning rate for Model-Free updates
    # We assume OCI increases the learning rate (stickier habits) or decreases it.
    # Here we model it as a multiplicative factor or additive modifier.
    # Let's try additive: lr = base + (mod * oci), clipped.
    alpha = np.clip(lr_base + (lr_oci_mod * oci_score), 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 update (standard Q-learning)
        delta_stage2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta_stage2
        
        # Stage 1 update (TD-learning using stage 2 max value)
        # Note: This is where the OCI-modulated alpha has a compounding effect on MF values
        delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1] # SARSA-like update for consistency with some 2-step lit
        # Alternatively using max: delta_stage1 = np.max(q_stage2_mf[s]) - q_stage1_mf[a1]
        # We stick to the template style or standard TD(0)
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Choice Perseveration (Stickiness)
High OCI scores are often linked to compulsivity and repetition. This model introduces a "stickiness" parameter that is purely a function of the OCI score. Unlike previous models where stickiness might have been a base parameter, here we enforce that the *tendency to repeat the last choice* is determined by the psychiatric score, combined with a standard reinforcement learning model.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Driven Choice Perseveration (Stickiness).
    
    Hypothesis: OCI scores predict the degree of choice perseveration (stickiness)
    regardless of reward history. This model adds a 'stickiness' bonus to the 
    previously chosen action, where the magnitude of this bonus is scaled by OCI.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    stickiness_factor: [0, 5] - Magnitude of stickiness per unit of OCI
    w: [0, 1] - MB/MF balance
    """
    learning_rate, beta, stickiness_factor, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Stickiness bonus is directly proportional to OCI
    # If OCI is high, the bonus for repeating the previous action is high.
    sticky_bonus = stickiness_factor * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with invalid action

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to the Q-value of the previously chosen action
        q_net_sticky = q_net.copy()
        if last_action_1 != -1:
            q_net_sticky[last_action_1] += sticky_bonus

        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_stage2
        
        # TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Second-Stage Exploration (Beta Modulation)
This model proposes that OCI affects the exploration/exploitation balance specifically in the second stage. Individuals with high compulsivity might be more rigid in their choices once they reach a specific context (the planet), effectively having a higher inverse temperature (beta) driven by their OCI score. This reflects "deterministic" behavior in local contexts.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Second-Stage Inverse Temperature.
    
    Hypothesis: High OCI leads to more deterministic (less exploratory) behavior 
    at the second stage (choosing the alien). We model beta_stage2 as distinct 
    from beta_stage1, where beta_stage2 is amplified by the OCI score.
    
    Bounds:
    learning_rate: [0, 1]
    beta_stage1: [0, 10] - Exploration for spaceship choice
    beta_base_stage2: [0, 5] - Base exploration for alien choice
    beta_oci_sens: [0, 5] - Sensitivity of stage 2 beta to OCI
    w: [0, 1] - MB/MF balance
    """
    learning_rate, beta_stage1, beta_base_stage2, beta_oci_sens, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate stage 2 beta: Base + (Sensitivity * OCI)
    # Higher OCI -> Higher Beta -> More exploitation/Rigidity
    beta_stage2 = beta_base_stage2 + (beta_oci_sens * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy (Uses beta_stage1) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_stage1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy (Uses OCI-modulated beta_stage2) ---
        exp_q2 = np.exp(beta_stage2 * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
