Here are 3 new cognitive models based on the participant's data and OCI score. The models explore different mechanisms by which high obsessive-compulsive traits might influence reinforcement learning, specifically focusing on the balance between model-based/model-free control and perseveration.

### Cognitive Model 1: Hybrid Model with OCI-modulated Model-Based Weight
This model posits that the balance between Model-Based (planning) and Model-Free (habitual) systems is directly influenced by the OCI score. High OCI individuals are hypothesized to rely less on the computationally expensive Model-Based system, favoring Model-Free habits.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL where the mixing weight (w) is modulated by OCI.
    High OCI reduces the contribution of the Model-Based system (lower w).
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_max: [0,1]
    w_slope: [0,5]
    """
    learning_rate, beta, w_max, w_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate mixing weight w based on OCI
    # Higher OCI -> lower w (less model-based)
    # We clip to ensure w stays in [0, 1]
    w = w_max - (w_slope * oci_score)
    w = np.clip(w, 0.0, 1.0)

    # Fixed transition matrix as per task structure (A->X 70%, U->Y 70%)
    # Rows: Space A (0), Space U (1). Cols: Planet X (0), Planet Y (1).
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 planets, 2 aliens)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy (Hybrid) ---
        # Model-Based Value Calculation
        # Max value available at stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1) 
        # Bellman equation using known transition structure
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy (Pure Model-Free) ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 update (Standard Q-learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 update (TD(0) / SARSA-like based on Stage 2 value)
        # Using the value of the state actually reached
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Choice Perseveration (Stickiness) Modulated by OCI
This model focuses on "stickiness" or choice perseveration, a common finding in compulsivity research. It assumes the participant has a tendency to repeat the previous Stage 1 action regardless of reward, and the strength of this perseveration is scaled by their OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with Choice Perseveration (Stickiness) modulated by OCI.
    High OCI leads to higher stickiness (tendency to repeat stage 1 choice).
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    stick_base: [0,5]
    stick_oci_coeff: [0,5]
    """
    learning_rate, beta, stick_base, stick_oci_coeff = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate stickiness parameter
    # Higher OCI -> Higher stickiness
    stickiness = stick_base + (stick_oci_coeff * oci_score)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Add stickiness bonus to Q-value of the previously chosen action
        q_stage1_sticky = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_sticky[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_stage1_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 update
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 update (TD-learning)
        # We use the value of the state reached to update the action taken
        # Note: Stickiness is a choice bias, not learned value, so we update q_stage1
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Eligibility Trace Decay Modulated by OCI
This model introduces an eligibility trace parameter ($\lambda$) for the first-stage update. This parameter controls how much the Stage 1 value is updated by the final reward outcome, effectively bridging the gap between Model-Free TD(0) and Monte-Carlo like updates. We hypothesize that OCI affects how credit is assigned to past actions, potentially making the trace decay faster (focus on immediate state) or slower (linking distal causes).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with Eligibility Trace (Lambda) modulated by OCI.
    The lambda parameter controls how strongly the stage 1 choice is reinforced 
    by the final reward (Stage 2 outcome).
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    lambda_base: [0,1]
    lambda_oci_scale: [-1,1] 
    """
    learning_rate, beta, lambda_base, lambda_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate lambda
    # We allow lambda_oci_scale to be negative to see if OCI reduces credit assignment
    lambda_val = lambda_base + (lambda_oci_scale * oci_score)
    lambda_val = np.clip(lambda_val, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        
        # Prediction error at Stage 2 (Reward - Q_stage2)
        delta_2 = r - q_stage2[s_idx, a2]
        
        # Prediction error at Stage 1 (Q_stage2 - Q_stage1)
        # This is the standard TD error for the first step
        delta_1 = q_stage2[s_idx, a2] - q_stage1[a1]
        
        # Update Stage 2 value
        q_stage2[s_idx, a2] += learning_rate * delta_2
        
        # Update Stage 1 value
        # Q(s1, a1) += alpha * (delta_1 + lambda * delta_2)
        # If lambda=0, this is TD(0): update only based on state transition value
        # If lambda=1, this approaches Monte Carlo: update based on final reward
        q_stage1[a1] += learning_rate * (delta_1 + lambda_val * delta_2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
