Here are three new cognitive models that incorporate the OCI score to explain the participant's decision-making behavior.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where the balance (w) is modulated by OCI.
    
    Hypothesis:
    High OCI scores are associated with reduced goal-directed (Model-Based) control,
    biasing the agent towards habitual (Model-Free) behavior.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta: [0,10] - Inverse temperature (exploration vs exploitation).
    w_base: [0,1] - Baseline Model-Based weight (before OCI modulation).
    lambda_elig: [0,1] - Eligibility trace decay.
    oci_w_penalty: [0,1] - Degree to which OCI reduces w. w = w_base * (1 - oci_w_penalty * OCI).
    """
    learning_rate, beta, w_base, lambda_elig, oci_w_penalty = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective w: higher OCI reduces w (more MF).
    w = w_base * (1.0 - oci_w_penalty * current_oci)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_elig * delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with asymmetric learning rates modulated by OCI.
    
    Hypothesis:
    High OCI scores are associated with hypersensitivity to negative outcomes (losses).
    The learning rate for negative prediction errors is amplified by OCI.
    
    Parameters:
    learning_rate: [0,1] - Base learning rate for positive errors.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF balance.
    lambda_elig: [0,1] - Eligibility trace.
    oci_loss_sens: [0,5] - Multiplier for negative learning rate based on OCI.
    """
    learning_rate, beta, w, lambda_elig, oci_loss_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Negative learning rate is boosted by OCI
    lr_neg_mult = 1.0 + oci_loss_sens * current_oci * 5.0 
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Determine effective learning rate for Stage 2 update
        # If the outcome was worse than expected (negative delta), increase learning rate
        lr_eff_2 = learning_rate * lr_neg_mult if delta_stage2 < 0 else learning_rate
        lr_eff_2 = min(lr_eff_2, 1.0) # Clip to 1
        
        # Update Stage 2
        q_stage2_mf[state_idx, a2] += lr_eff_2 * delta_stage2
        
        # Update Stage 1 (using lambda and stage 2 error)
        q_stage1_mf[a1] += learning_rate * delta_stage1 + lambda_elig * lr_eff_2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated inverse temperature (Beta).
    
    Hypothesis:
    High OCI leads to behavioral rigidity, modeled as a higher Beta (inverse temperature),
    making choices more deterministic and less exploratory.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta_base: [0,5] - Baseline inverse temperature.
    w: [0,1] - MB/MF balance.
    lambda_elig: [0,1] - Eligibility trace.
    oci_rigidity: [0,5] - Scaling factor for Beta based on OCI.
    """
    learning_rate, beta_base, w, lambda_elig, oci_rigidity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Effective beta increases with OCI
    beta_eff = beta_base * (1.0 + oci_rigidity * current_oci)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy

        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_eff * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Stage 2 Policy

        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
