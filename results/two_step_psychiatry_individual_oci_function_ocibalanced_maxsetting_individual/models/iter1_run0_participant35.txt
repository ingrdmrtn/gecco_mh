Here are three cognitive models expressed as Python functions that incorporate the OCI score to explain the participant's decision-making behavior.

### Model 1: OCI-Interpolated Learning Rate
This model hypothesizes that the obsessive-compulsive symptom severity (OCI) fundamentally alters the rate at which the participant learns from prediction errors (habit formation speed). It interpolates the learning rate between a "low OCI" baseline and a "high OCI" baseline based on the participant's score. It also includes a general stickiness parameter to account for the observed repetitive behavior.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Interpolated Learning Rate.
    
    Hypothesis: The speed of habit formation (learning rate) varies with OCI severity.
    This model linearly interpolates the effective learning rate between a 
    low-symptom baseline (lr_low) and a high-symptom baseline (lr_high).
    
    Parameters:
    - lr_low: [0, 1] Learning rate for hypothetical OCI = 0.
    - lr_high: [0, 1] Learning rate for hypothetical OCI = 1.
    - beta: [0, 10] Inverse temperature (exploration/exploitation balance).
    - w: [0, 1] Weighting of Model-Based values (0=MF, 1=MB).
    - stickiness: [0, 5] General tendency to repeat the last choice.
    """
    lr_low, lr_high, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective learning rate based on OCI
    learning_rate = lr_low + (lr_high - lr_low) * oci_score
    # Ensure bounds
    learning_rate = np.clip(learning_rate, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Policy for Choice 2 ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Update Stage 1 MF values (SARSA-like)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated "Lose-Stay" Stickiness
This model investigates the compulsive nature of the behavior by differentiating between "Win-Stay" and "Lose-Stay" strategies. It hypothesizes that high OCI scores specifically increase the tendency to repeat a choice after a lack of reward (failure), reflecting a compulsive persistence or "chasing" behavior often seen in high-OCI individuals.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Punishment Stickiness (Lose-Stay).
    
    Hypothesis: High OCI individuals exhibit compulsive repetition specifically 
    after unrewarded trials (Lose-Stay), whereas Win-Stay is consistent across individuals.
    
    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - stick_win: [0, 5] Stickiness applied if the previous trial was rewarded.
    - stick_loss_base: [0, 5] Baseline stickiness if previous trial was unrewarded.
    - stick_loss_oci: [0, 5] Additional 'Lose-Stay' stickiness per unit of OCI.
    """
    learning_rate, beta, w, stick_win, stick_loss_base, stick_loss_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):
        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply differential stickiness based on previous outcome
        if last_action_1 != -1:
            if last_reward == 1.0:
                bonus = stick_win
            else:
                # OCI modulates the tendency to stick after a loss
                bonus = stick_loss_base + (stick_loss_oci * oci_score)
            q_net[last_action_1] += bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Policy for Choice 2 ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Choice Kernel
This model addresses the long "runs" of identical choices observed in the data. Instead of simple 1-step stickiness, it uses a "Choice Kernel" (exponential moving average of past choices) to model habit strength. The OCI score modulates the *weight* or influence of this habit kernel, hypothesizing that higher OCI leads to a stronger influence of long-term choice history on current decisions.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-Modulated Choice Kernel.
    
    Hypothesis: High OCI is associated with stronger habit formation over time.
    Instead of simple repetition (stickiness), this model uses a Choice Kernel 
    (decaying trace of past choices). The influence (weight) of this kernel 
    on the decision is modulated by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - decay: [0, 1] Decay rate of the choice kernel (0=instant decay, 1=no decay).
    - ck_base: [0, 5] Baseline weight of the choice kernel.
    - ck_oci: [0, 5] Additional kernel weight per unit of OCI.
    """
    learning_rate, beta, w, decay, ck_base, ck_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice kernel tracks the frequency/recency of choices
    choice_kernel = np.zeros(2)
    
    # Calculate effective kernel weight
    ck_weight = ck_base + (ck_oci * oci_score)

    for trial in range(n_trials):
        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add Choice Kernel influence
        q_net += ck_weight * choice_kernel
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Policy for Choice 2 ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Update Choice Kernel
        # Decay both, then increment chosen
        choice_kernel = choice_kernel * decay
        choice_kernel[a1] += 1.0
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
