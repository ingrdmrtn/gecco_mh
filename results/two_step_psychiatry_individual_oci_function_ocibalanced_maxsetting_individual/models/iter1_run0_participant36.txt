Here are three new cognitive models that incorporate the OCI score to explain the participant's decision-making behavior.

import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Hybrid MB/MF Control.
    Hypothesis: OCI score modulates the balance between Model-Based (goal-directed) 
    and Model-Free (habitual) control. High OCI leads to reduced Model-Based weighting,
    relying more on habitual cached values.
    Includes a fixed stickiness parameter to account for the general perseveration observed.

    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature.
    - w_mb_base: [0, 1] Baseline weight for Model-Based control.
    - oci_w_penalty: [0, 1] Reduction in MB weight per unit of OCI.
    - stickiness: [0, 5] Bonus added to the previously chosen action.
    """
    learning_rate, beta, w_mb_base, oci_w_penalty, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate mixing weight w, bounded between 0 and 1
    # Higher OCI reduces the model-based weight (w)
    w = w_mb_base - (oci_score * oci_w_penalty)
    if w < 0: w = 0
    if w > 1: w = 1
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Weighted Sum + Stickiness
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # TD update for Stage 1 MF values
        # Using the value of the state reached (Q_stage2 of chosen action) as target
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD update for Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Learning Rate Rigidity.
    Hypothesis: High OCI scores correlate with "cognitive rigidity" manifested as 
    a reduced learning rate. The participant updates beliefs more slowly in response 
    to reward prediction errors, explaining the long blocks of repeated choices.

    Parameters:
    - lr_base: [0, 1] Baseline learning rate.
    - oci_lr_damp: [0, 5] Factor by which OCI reduces the learning rate (dampening).
    - beta: [0, 10] Inverse temperature.
    - stickiness: [0, 5] Choice perserveration bonus.
    """
    lr_base, oci_lr_damp, beta, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Modulate learning rate: Higher OCI -> Lower LR
    learning_rate = lr_base / (1.0 + oci_score * oci_lr_damp)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # We assume a Model-Based framework for value calculation
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = q_stage1_mb.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a2 = action_2[trial]
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = action_1[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Determinism (Beta).

    Parameters:
