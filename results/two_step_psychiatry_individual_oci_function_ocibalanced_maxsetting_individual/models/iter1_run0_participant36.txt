Here are the three proposed cognitive models.

### Cognitive Model 1: OCI-Modulated Exploration (Beta)
This model hypothesizes that OCI score predicts the rigidity of choice behavior. High OCI participants are assumed to have a higher inverse temperature ($\beta$), making their choices more deterministic and less exploratory.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    RL model where OCI modulates the inverse temperature (beta).
    Hypothesis: High OCI leads to more rigid/deterministic choices (higher beta),
    reducing exploration.
    
    Parameters:
    - learning_rate: [0, 1]
    - w: [0, 1] Weight for model-based values.
    - lambda_coeff: [0, 1] Eligibility trace decay.
    - beta_base: [0, 10] Base inverse temperature.
    - beta_oci_scale: [0, 5] Increase in beta per unit of OCI.
    """
    learning_rate, w, lambda_coeff, beta_base, beta_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulation of Beta: Higher OCI -> Higher Beta (more exploitation/rigidity)
    beta = beta_base + (beta_oci_scale * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        logits_1 -= np.max(logits_1)
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        logits_2 = beta * q_stage2_mf[state_idx]
        logits_2 -= np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        
        a2 = int(action_2[trial])
        # Safe indexing for probability storage
        idx_2 = a2 if a2 >= 0 else 0
        p_choice_2[trial] = probs_2[idx_2]
  
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[int(action_1[trial])]
        
        # [FILL IN ACTION VALUE UPDATING FOR CHOICE 1]
        # Calculate predicted delta2 for eligibility trace update
        delta_stage2_est = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage1_mf[int(action_1[trial])] += learning_rate * (delta_stage1 + lambda_coeff * delta_stage2_est)
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        # [FILL IN ACTION VALUE UPDATING FOR CHOICE 2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Modulated Punishment Sensitivity
This model hypothesizes that high OCI participants perceive the lack of reward (0 coins) as an active punishment (failure), rather than just a neutral outcome. This "subjective punishment" magnitude scales with their OCI score, driving stronger avoidance of non-rewarding options.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    RL model where OCI modulates sensitivity to negative outcomes (0 reward).
    Hypothesis: High OCI participants perceive 0 reward as a punishment.
    The effective reward becomes negative for 0 outcomes, proportional to OCI.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - lambda_coeff: [0, 1]
    - punishment_oci_scale: [0, 5] Magnitude of 'punishment' for 0 reward per unit OCI.
    """
    learning_rate, beta, w, lambda_coeff, punishment_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    punishment = punishment_oci_scale * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        logits_1 -= np.max(logits_1)
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        logits_2 = beta * q_stage2_mf[state_idx]
        logits_2 -= np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        
        a2 = int(action_2[trial])
        idx_2 = a2 if a2 >= 0 else 0
        p_choice_2[trial] = probs_2[idx_2]
  
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[int(action_1[trial])]
        
        # [FILL IN ACTION VALUE UPDATING FOR CHOICE 1]
        # Calculate effective reward: 1 -> 1, 0 -> -punishment
        r = reward[trial]
        eff_reward = r if r == 1 else -punishment
        delta_stage2_eff = eff_reward - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[int(action_1[trial])] += learning_rate * (delta_stage1 + lambda_coeff * delta_stage2_eff)
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        # [FILL IN ACTION VALUE UPDATING FOR CHOICE 2]
        # Update stage 2 values using the effective (subjective) prediction error
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2_eff
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Eligibility Trace (Lambda)
This model hypothesizes that OCI modulates the strength of habit formation via the eligibility trace parameter ($\lambda$). High OCI leads to a higher $\lambda$, meaning the outcome of the second stage reinforces the first stage choice more strongly (model-free credit assignment), bypassing the model-based structure.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    RL model where OCI modulates the eligibility trace parameter (lambda).
    Hypothesis: High OCI leads to stronger habit formation (higher lambda),
    reinforcing the first choice based on the second stage outcome more strongly.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - lambda_base: [0, 1] Base eligibility trace.
    - lambda_oci_scale: [0, 1] OCI impact on lambda.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulation of Lambda
    lambda_val = lambda_base + (lambda_oci_scale * oci_score)
    # Clip lambda to [0, 1]
    if lambda_val > 1.0: lambda_val = 1.0
    if lambda_val < 0.0: lambda_val = 0.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        logits_1 -= np.max(logits_1)
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        logits_2 = beta * q_stage2_mf[state_idx]
        logits_2 -= np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        
        a2 = int(action_2[trial])
        idx_2 = a2 if a2 >= 0 else 0
        p_choice_2[trial] = probs_2[idx_2]
  
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[int(action_1[trial])]
        
        # [FILL IN ACTION VALUE UPDATING FOR CHOICE 1]
        # Calculate delta2 locally to use the modulated lambda
        delta_stage2_calc = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage1_mf[int(action_1[trial])] += learning_rate * (delta_stage1 + lambda_val * delta_stage2_calc)
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        # [FILL IN ACTION VALUE UPDATING FOR CHOICE 2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
