Here are three new cognitive models that incorporate the OCI score to explain the participant's decision-making in the two-step task.

### Model 1: OCI-Modulated Choice Stickiness (Perseveration)
This model hypothesizes that high OCI scores are associated with "stickiness" or perseverationâ€”a tendency to repeat the previous action regardless of reward history or model-based planning. This aligns with the compulsivity aspect of OCI, explaining the participant's long streaks of choosing the same spaceship.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Choice Perseveration (Stickiness).
    
    Hypothesis: Participants with high OCI scores exhibit 'stickiness', repeating 
    their previous Stage 1 choice regardless of value. The magnitude of this 
    stickiness is scaled by their OCI score.
    
    Parameters:
    - learning_rate: Update rate for value estimation [0, 1]
    - beta: Inverse temperature for softmax [0, 10]
    - w: Mixing weight (0=Pure MF, 1=Pure MB) [0, 1]
    - lambd: Eligibility trace for Stage 2 outcomes affecting Stage 1 [0, 1]
    - stick_oci: Perseveration bonus scaled by OCI [0, 5]
      (Effective stickiness = stick_oci * OCI)
    """
    learning_rate, beta, w, lambd, stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness based on OCI
    stickiness_bonus = stick_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Q-values initialization
    q_stage1_mf = np.zeros(2)       # MF values for Spaceships [A, U]
    q_stage2_mf = np.zeros((2, 2))  # MF values for Planets [X, Y] -> Aliens [0, 1]
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Policy for Choice 1 ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness (Perseveration)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_bonus

        # Softmax
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Record action for next trial's stickiness
        last_action_1 = a1

        # --- Policy for Choice 2 ---
        qs2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 Prediction Error (SARSA-like for MF)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace (update Stage 1 MF value based on Stage 2 outcome)
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Learning Rigidity
This model hypothesizes that high OCI scores correlate with cognitive rigidity, represented by a suppressed learning rate. A participant with high OCI updates their value estimates more slowly, making them resistant to changing their behavior even when rewards stop (explaining the perseverance without explicit stickiness parameters).

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Learning Rigidity.
    
    Hypothesis: High OCI scores lead to cognitive rigidity, modeled as a 
    reduction in the learning rate. High OCI participants update their 
    beliefs more slowly, making them resistant to feedback.
    
    Parameters:
    - lr_base: Baseline learning rate [0, 1]
    - lr_suppression: Factor by which OCI reduces learning rate [0, 1]
      (Effective lr = lr_base * (1 - lr_suppression * OCI))
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight [0, 1]
    - lambd: Eligibility trace [0, 1]
    """
    lr_base, lr_suppression, beta, w, lambd = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective learning rate (Rigidity hypothesis)
    # As OCI increases, learning rate decreases.
    learning_rate = lr_base * (1.0 - (lr_suppression * oci_score))
    learning_rate = np.clip(learning_rate, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Policy for Choice 2 ---
        qs2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Enhanced Habit Strength (Eligibility Trace)
This model posits that OCI affects the strength of the "eligibility trace" ($\lambda$). A higher eligibility trace means the outcome at Stage 2 (reward/no reward) directly reinforces the Stage 1 choice more strongly, bypassing the two-step structure. This represents a stronger formation of "habit chains" (Model-Free control) often associated with compulsive behavior.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Eligibility Trace.
    
    Hypothesis: High OCI scores strengthen the 'habit chain' between Stage 2 outcomes
    and Stage 1 choices. This is modeled by increasing the eligibility trace (lambda)
    as OCI increases, allowing direct reinforcement of the first choice by the 
    final reward (pure Model-Free characteristic).
    
    Parameters:
    - learning_rate: Learning rate [0, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight [0, 1]
    - lambd_base: Baseline eligibility trace [0, 1]
    - lambd_oci_gain: Increase in lambda per unit of OCI [0, 1]
      (Effective lambda = lambd_base + lambd_oci_gain * OCI)
    """
    learning_rate, beta, w, lambd_base, lambd_oci_gain = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective lambda
    lambd = lambd_base + (lambd_oci_gain * oci_score)
    lambd = np.clip(lambd, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Policy for Choice 2 ---
        qs2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # OCI-modulated eligibility trace update
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
