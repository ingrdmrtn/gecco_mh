Here are three new cognitive models that incorporate the OCI score to explain participant behavior in the two-step task.

### Model 1: Stage-Specific Learning Rate Modulation
This model posits that OCI specifically affects the learning rate for the high-level planning stage (Spaceship choice), while the low-level association learning (Alien choice) remains independent. This tests the hypothesis that obsessive-compulsive traits correlate with rigidity or volatility in high-level strategy updates ($LR_1$) rather than local reward learning ($LR_2$).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with separate Stage 1 and Stage 2 learning rates.
    OCI modulates the Stage 1 learning rate (Strategy Update), testing if 
    compulsivity affects high-level policy flexibility.
    
    Parameters:
    lr1_base: [0, 1] Base learning rate for Stage 1 (Spaceship choice).
    lr1_oci:  [0, 1] OCI scaling for Stage 1 learning rate.
    lr2:      [0, 1] Learning rate for Stage 2 (Alien choice).
    beta:     [0, 10] Inverse temperature (exploration/exploitation).
    w:        [0, 1] Weight for Model-Based control.
    """
    lr1_base, lr1_oci, lr2, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate OCI-modulated Stage 1 Learning Rate
    lr1 = lr1_base + lr1_oci * oci_score
    lr1 = np.clip(lr1, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 
    
    for trial in range(n_trials):
        # Handle missing/invalid data if present
        if action_1[trial] == -1: 
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        # Stage 1 MF update using OCI-modulated lr1
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr1 * delta_stage1

        # Stage 2 MF update using fixed lr2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr2 * delta_stage2
        
    eps = 1e-10
    # Only sum log-loss for valid trials where probability was recorded
    valid_indices = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_indices] + eps)) + np.sum(np.log(p_choice_2[valid_indices] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Win-Stay Bonus
This model distinguishes "Stickiness" (general repetition) from "Win-Stay" (repetition reinforced by success). It proposes that OCI scores scale the urge to repeat a *rewarded* previous action. This tests if compulsivity is driven by an over-sensitivity to reinforcing events leading to repetition loops.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Win-Stay bonus.
    Unlike simple stickiness, this bonus is only applied if the previous 
    trial was rewarded, capturing 'reinforcement-driven repetition'.
    
    Parameters:
    lr:       [0, 1] Learning rate.
    beta:     [0, 10] Inverse temperature.
    w:        [0, 1] Model-Based weight.
    ws_base:  [0, 5] Base Win-Stay bonus.
    ws_oci:   [0, 5] OCI scaling for Win-Stay bonus.
    """
    lr, beta, w, ws_base, ws_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate Win-Stay magnitude
    ws_bonus = ws_base + ws_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):
        if action_1[trial] == -1: 
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Win-Stay Bonus only if previous trial was rewarded
        if last_action_1 != -1 and last_reward == 1:
            q_net_s1[last_action_1] += ws_bonus

        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = int(action_1[trial])
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        
        last_reward = reward[trial]
        
    eps = 1e-10
    valid_indices = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_indices] + eps)) + np.sum(np.log(p_choice_2[valid_indices] + eps)))
    return log_loss
```

### Model 3: OCI-Distorted Transition Beliefs
This model suggests that OCI affects the *subjective* transition matrix used in Model-Based planning. Rather than assuming the participant knows the true 0.7/0.3 probabilities, this model allows the perceived "common" probability ($\tau$) to shift based on OCI, reflecting potential cognitive distortions (e.g., doubting the structure or assuming determinism).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model where OCI distorts the subjective Model-Based transition matrix.
    Tests if high OCI correlates with distorted probability estimation 
    (e.g., underestimating reliability of common transitions).
    
    Parameters:
    lr:        [0, 1] Learning rate.
    beta:      [0, 10] Inverse temperature.
    w:         [0, 1] Model-Based weight.
    tau_base:  [0, 1] Base subjective transition probability (Common).
    tau_oci:   [0, 1] OCI scaling for transition probability.
    """
    lr, beta, w, tau_base, tau_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate Subjective Transition Probability
    # True value is 0.7. OCI shifts this subjective belief.
    tau = tau_base + tau_oci * oci_score
    tau = np.clip(tau, 0.0, 1.0)
    
    # Subjective Transition Matrix used for MB planning
    transition_matrix = np.array([[tau, 1-tau], [1-tau, tau]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 
    
    for trial in range(n_trials):
        if action_1[trial] == -1: 
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Use distorted subjective matrix for MB planning
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        
    eps = 1e-10
    valid_indices = p_choice_1 > 0
    log_loss = -(np.sum(np.log(p_choice_1[valid_indices] + eps)) + np.sum(np.log(p_choice_2[valid_indices] + eps)))
    return log_loss
```
