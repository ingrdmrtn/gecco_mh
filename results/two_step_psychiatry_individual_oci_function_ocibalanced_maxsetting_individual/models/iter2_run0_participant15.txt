Here are three new cognitive models for the two-step task, incorporating the OCI score into different computational mechanisms based on the hypothesis that obsessive-compulsive traits may alter learning from negative outcomes, the persistence of choice history (habits), or the precision of choices at different stages.

### Model 1: Asymmetric Learning Rates (OCI-Modulated Negative Learning)
This model hypothesizes that OCI symptoms differentially affect how participants learn from unrewarded outcomes (failures/punishments) compared to rewarded ones. High OCI might lead to "over-learning" from failure (hypersensitivity to error) or "under-learning" (persistence/ignoring lack of reward).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Asymmetric Learning Rate Model.
    
    Hypothesis: OCI modulates the learning rate specifically for unrewarded trials (alpha_unrew),
    creating an asymmetry between learning from success (alpha_rew) and failure.
    
    Parameters:
    - alpha_rew: [0, 1] Learning rate for rewarded trials (R > 0).
    - alpha_unrew_mod: [-1, 1] OCI modulation on the unrewarded learning rate.
        alpha_unrew = alpha_rew + (alpha_unrew_mod * oci)
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - stickiness: [0, 5] Choice perseveration bonus.
    """
    alpha_rew, alpha_unrew_mod, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate unrewarded learning rate based on OCI
    # We clip to ensure it stays valid [0, 1]
    alpha_unrew = alpha_rew + (alpha_unrew_mod * current_oci)
    alpha_unrew = np.clip(alpha_unrew, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Skip missing data
        if a1 < 0 or s_idx < 0 or a2 < 0:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        # Model-Based value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        q_augmented = q_net.copy()
        if last_action_1 != -1:
            q_augmented[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_augmented)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Select learning rate based on outcome
        current_lr = alpha_rew if r > 0 else alpha_unrew
        
        # TD(1) style update for Stage 1 MF (updates from full outcome)
        # Note: Standard Daw implementation often uses lambda. 
        # Here we simplify to direct update from R to Q1 for robustness in MF.
        # But following the template structure with stage 2 prediction error:
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 2
        q_stage2_mf[s_idx, a2] += current_lr * delta_stage2
        
        # Update Stage 1 (TD(1) logic: update by sum of errors)
        q_stage1_mf[a1] += current_lr * (delta_stage1 + delta_stage2)
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Decaying Choice Kernel (Habit Strength)
This model replaces simple one-step stickiness with a "Choice Kernel" that decays over time. This represents a habit strength that builds up and fades. The OCI score modulates the **decay rate** of this habit.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Choice Kernel Decay Model.
    
    Hypothesis: OCI modulates the persistence of habits (choice kernel decay).
    Higher OCI might lead to slower decay (longer-lasting habits).
    
    Parameters:
    - learning_rate: [0, 1] Value learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - stick_weight: [0, 5] Weight of the choice kernel in decision making.
    - decay_base: [0, 1] Base decay rate for the choice kernel.
    - decay_oci: [-1, 1] Effect of OCI on decay rate.
    """
    learning_rate, beta, w, stick_weight, decay_base, decay_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate decay rate
    decay = decay_base + (decay_oci * current_oci)
    decay = np.clip(decay, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice Kernel (Action trace)
    choice_kernel = np.zeros(2)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 < 0 or s_idx < 0 or a2 < 0:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Choice Kernel bonus
        q_augmented = q_net + (stick_weight * choice_kernel)
            
        exp_q1 = np.exp(beta * q_augmented)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # 1. Update Choice Kernel
        choice_kernel = choice_kernel * decay # Decay all
        choice_kernel[a1] += 1.0          # Reinforce chosen
        
        # 2. Update Values
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Stage-Specific Precision (Split Beta)
This model tests if OCI affects the randomness/precision of choices differently for the high-level planning stage (Stage 1) versus the local bandit stage (Stage 2).

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage-Specific Inverse Temperature Model.
    
    Hypothesis: OCI modulates the 'beta' (precision) for the second stage choice 
    relative to the first stage. High OCI might imply more rigid/deterministic 
    behavior in the immediate environment (Stage 2) vs the planning stage.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_1: [0, 10] Inverse temperature for Stage 1 (Spaceship).
    - beta_2_ratio: [0, 5] Multiplier for Stage 2 beta based on OCI.
        beta_2 = beta_1 * (1 + beta_2_ratio * oci)
    - w: [0, 1] Mixing weight.
    - stickiness: [0, 5] Simple stickiness.
    """
    learning_rate, beta_1, beta_2_ratio, w, stickiness = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Stage 2 beta
    beta_2 = beta_1 * (1.0 + beta_2_ratio * current_oci)
    # Clip to prevent overflow or extreme determinism
    beta_2 = np.clip(beta_2, 0.0, 20.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 < 0 or s_idx < 0 or a2 < 0:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy (Uses beta_1) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        q_augmented = q_net.copy()
        if last_action_1 != -1:
            q_augmented[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_augmented)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy (Uses beta_2) ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
