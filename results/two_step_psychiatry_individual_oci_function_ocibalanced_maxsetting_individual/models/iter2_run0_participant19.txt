Here are three cognitive models that incorporate the OCI score into the decision-making process in distinct ways.

### Model 1: OCI-Modulated Model-Based Suppression
This model tests the hypothesis that high obsessive-compulsive symptoms are associated with a deficit in goal-directed (Model-Based) control. Instead of a fixed weighting parameter $w$, the effective weight is reduced proportionally to the OCI score. A high OCI score "penalizes" the contribution of the Model-Based system, forcing reliance on the Model-Free system.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Based Suppression.
    
    Hypothesis: Higher OCI scores lead to a reduction in Model-Based (goal-directed) 
    control. The weight parameter 'w' is dynamically calculated as a base maximum 
    minus a penalty derived from the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature (exploration/exploitation trade-off).
    - w_max: [0, 1] The maximum possible weight for Model-Based control (when OCI is 0).
    - oci_penalty: [0, 1] The strength with which OCI reduces the Model-Based weight.
    """
    learning_rate, beta, w_max, oci_penalty = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w based on OCI
    # If OCI is high and penalty is high, w decreases (shift to Model-Free)
    w_effective = w_max - (oci_score * oci_penalty)
    w_effective = np.clip(w_effective, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values using the OCI-modulated weight
        q_net = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Stage 1 MF Update (TD-0)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Eligibility Traces
This model hypothesizes that OCI relates to the strength of habit formation via direct reinforcement. In standard Reinforcement Learning, an eligibility trace ($\lambda$) allows the final reward to directly update the value of the first choice, bypassing the state transition structure. Here, the eligibility trace parameter $\lambda$ is not fixed but is defined by the OCI score. A high OCI implies a higher $\lambda$, meaning the participant is more likely to reinforce the Stage 1 spaceship directly based on the coin received (habitual/reflexive), ignoring which planet they actually visited.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Driven Eligibility Traces.
    
    Hypothesis: OCI dictates the strength of the eligibility trace (lambda). 
    Higher OCI leads to stronger direct reinforcement of Stage 1 choices by 
    Stage 2 rewards (ignoring the transition structure), characteristic of 
    habitual behavior.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting between Model-Based and Model-Free values.
    - lambda_scale: [0, 1.5] Scaling factor for OCI to determine lambda. 
      (lambda = OCI * lambda_scale, clipped at 1.0).
    """
    learning_rate, beta, w, lambda_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate lambda based on OCI
    eligibility_lambda = oci_score * lambda_scale
    eligibility_lambda = np.clip(eligibility_lambda, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # 1. Standard TD update for Stage 1 (State 1 -> State 2)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 2. Standard TD update for Stage 2 (State 2 -> Reward)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # 3. Eligibility Trace Update: Apply Stage 2 RPE to Stage 1 Q-value
        # If lambda is high (high OCI), the final reward strongly affects Stage 1 choice directly.
        q_stage1_mf[a1] += learning_rate * eligibility_lambda * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Sharpened Selection (Beta Modulation)
This model posits that OCI affects the decision noise (inverse temperature $\beta$). Participants with higher OCI scores may exhibit more rigid, deterministic behavior (less exploration) once values are established. Here, the base $\beta$ is amplified by the OCI score. A high OCI results in a higher $\beta$, leading to "sharper" probability distributions where the participant almost exclusively chooses the option with the slightly higher value, potentially explaining the "blocky" persistence seen in the data.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Sharpened Selection (Beta Modulation).
    
    Hypothesis: High OCI leads to reduced exploration and more deterministic 
    (compulsive) choice selection. The inverse temperature (beta) is 
    amplified by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta_base: [0, 5] The baseline inverse temperature.
    - w: [0, 1] Weighting between Model-Based and Model-Free values.
    - oci_sharpness: [0, 5] How much OCI increases beta (reduces noise).
      Effective beta = beta_base * (1 + OCI * oci_sharpness).
    """
    learning_rate, beta_base, w, oci_sharpness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta
    # Higher OCI -> Higher Beta -> Less stochasticity/More rigidity
    beta_effective = beta_base * (1.0 + oci_score * oci_sharpness)
    
    # Cap beta to prevent numerical overflow in exp()
    beta_effective = min(beta_effective, 20.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_effective * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 2 Policy ---
        # Note: We apply the sharpened beta to stage 2 as well, assuming 
        # the compulsivity affects decision making globally.
        exp_q2 = np.exp(beta_effective * q_stage2_mf[state_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
