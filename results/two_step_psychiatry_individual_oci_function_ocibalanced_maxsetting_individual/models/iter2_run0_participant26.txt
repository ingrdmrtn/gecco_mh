Here are the 3 cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Beta (Inverse Temperature).
    
    Hypothesis: 
    Obsessive-compulsive symptoms (OCI) are associated with rigidity and anxiety. 
    This model posits that higher OCI scores lead to a higher inverse temperature (beta),
    resulting in more deterministic (exploitative) choices and reduced exploration 
    once values are learned.
    
    Parameters:
    - lr: [0, 1] Learning rate for value updates.
    - beta_base: [0, 10] Base inverse temperature parameter. The effective beta is scaled by OCI.
    - w: [0, 1] Weighting between Model-Based (MB) and Model-Free (MF) control (1 = pure MB).
    """
    lr, beta_base, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulates beta: Higher OCI -> Higher Beta (more rigid/deterministic)
    beta = beta_base * (1.0 + oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        # Skip missing trials
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based value: Expected value of next state's max Q
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        # Softmax Choice 2 (Purely Model-Free at Stage 2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Updates ---
        # Stage 1 Update (TD(0))
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2

    return -log_likelihood

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated MB/MF Weighting.
    
    Hypothesis: 
    Compulsivity is linked to enhanced habit formation (Model-Free control) and 
    reduced goal-directed (Model-Based) control. This model posits that the 
    mixing weight 'w' is negatively modulated by the OCI score: higher OCI 
    results in a lower 'w' (more habitual behavior).
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w_base: [0, 1] Base mixing weight parameter. The effective w is scaled down by OCI.
    """
    lr, beta, w_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulates w: Higher OCI -> Lower w (Less Model-Based, More Model-Free)
    # We clip to ensure it stays within bounds [0, 1]
    w = w_base * (1.0 - oci_score)
    w = np.clip(w, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value using OCI-modulated w
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2

    return -log_likelihood

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Enhanced Punishment Sensitivity (Asymmetric Learning).
    
    Hypothesis: 
    Individuals with higher OCI scores may be hyper-sensitive to negative outcomes 
    (failure to obtain gold). This model uses a single learning rate parameter 
    but amplifies it by the OCI score specifically when the reward is 0 (loss), 
    leading to faster unlearning of non-rewarded actions.
    
    Parameters:
    - lr: [0, 1] Base learning rate (applied to rewarded trials).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting between MB and MF.
    """
    lr, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_likelihood += np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_likelihood += np.log(probs_2[a2] + eps)

        # --- Updates ---
        # Determine effective learning rate based on outcome and OCI
        if r == 0:
            # Enhanced learning from loss
            effective_lr = lr * (1.0 + oci_score)
            effective_lr = np.clip(effective_lr, 0.0, 1.0)
        else:
            # Base learning from win
            effective_lr = lr

        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += effective_lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += effective_lr * delta_stage2

    return -log_likelihood
```
