def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Model-Based Weighting.
    Hypothesis: High OCI scores are associated with deficits in Model-Based (goal-directed) control.
    This model scales the mixing weight 'w' by the OCI score, such that higher OCI leads to 
    reduced Model-Based influence and greater reliance on Model-Free habits.
    
    Equation: w_effective = w_max * (1 - OCI)
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-value updates.
    beta: [0, 10] Inverse temperature (softmax sensitivity).
    w_max: [0, 1] Maximum Model-Based weight (achieved when OCI is 0).
    stickiness: [0, 5] Bonus added to the Q-value of the previously chosen action.
    """
    learning_rate, beta, w_max, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Modulate w based on OCI: Higher OCI reduces w (less Model-Based)
    w = w_max * (1.0 - oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        state_idx = int(state[trial])
        a2 = int(action_2[trial])

        # Stage 1 Policy Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy Calculation
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Value Updates
        # Stage 1 Model-Free Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 Model-Free Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Inverse Temperature (Rigidity).
    Hypothesis: OCI is associated with behavioral rigidity and reduced exploration.
    This model increases the inverse temperature (beta) as OCI increases, leading to 
    more deterministic (steeper softmax) choices.
    
    Equation: beta_effective = beta_base + beta_oci * OCI
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-value updates.
    beta_base: [0, 10] Baseline inverse temperature.
    beta_oci: [0, 10] Sensitivity of beta to the OCI score.

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Forgetting (Persistence).
    Hypothesis: High OCI is associated with 'mental stickiness' or a failure to disengage.
    This model implements a forgetting mechanism for unchosen options in Stage 1, 
    where the rate of forgetting is dampened by high OCI scores (values persist longer).
    
    Equation: Q_unchosen = Q_unchosen * (1 - forgetting_rate)
              forgetting_rate = forget_base * (1 - OCI)
    
    Parameters:
    learning_rate: [0, 1] Learning rate for chosen options.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    stickiness: [0, 5] Perseveration bonus.
    forget_base: [0, 1] Base forgetting rate. Higher values mean faster decay of unchosen values.
    """
    learning_rate, beta, w, stickiness, forget_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate forgetting rate: Higher OCI -> Lower forgetting rate (more persistence)
    forgetting_rate = forget_base * (1.0 - oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        state_idx = int(state[trial])
        a2 = int(action_2[trial])

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        # Update chosen Stage 1 MF value
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Decay unchosen Stage 1 MF value (Forgetting)
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - forgetting_rate)

        # Update Stage 2 value
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
