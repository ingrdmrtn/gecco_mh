Here are 3 new cognitive models that incorporate the OCI score into the decision-making process in the two-step task.

### Model 1: Hybrid RL with OCI-modulated Model-Based Weight
This model assumes that individuals with high OCI scores might rely more heavily on a rigid, habit-like system (model-free) or conversely, might over-deliberate (model-based). Given the high OCI score (0.83) and the potential for compulsivity (habit) vs obsession (planning), this model tests the hypothesis that OCI modulates the balance between Model-Based (MB) and Model-Free (MF) control, specifically positing that high OCI might degrade the flexibility of MB control or enhance MF control.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL where OCI score modulates the mixing weight (w) between Model-Based and Model-Free values.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_intercept: [0,1]
    w_oci_slope: [-1,1]
    """
    learning_rate, beta, w_intercept, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate mixing weight w based on OCI
    # w represents the weight of Model-Based values. 
    # High OCI might reduce MB reliance (negative slope) or increase it (positive slope).
    w = w_intercept + (w_oci_slope * oci_score)
    w = np.clip(w, 0.0, 1.0) # Ensure w stays between 0 and 1

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free stage 1 values
    q_stage2_mf = np.zeros((2, 2)) # Stage 2 values (same for MB and MF usually in simple hybrid models)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        # MB value is expected value of Stage 2 states weighted by transition probs
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Max value of next states
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        # Update Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF values (TD(0))
        # Note: Standard hybrid models often use the value of the state reached (q_stage2_mf[s_idx, a2])
        # or the max of the next state. Here we use the standard TD error form.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Eligibility trace update (TD(1) logic) is often used for MF, 
        # but for simplicity and strict adherence to the template structure requested previously,
        # we stick to simple TD updates unless lambda is a parameter.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Choice Stickiness (Perseveration)
This model hypothesizes that OCI relates to "stickiness" or perseverationâ€”the tendency to repeat the previous choice regardless of reward. High OCI scores might correlate with higher behavioral rigidity (high stickiness). The stickiness parameter is a linear function of the OCI score.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with Choice Stickiness modulated by OCI.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    stick_intercept: [-5, 5]
    stick_oci_slope: [-5, 5]
    """
    learning_rate, beta, stick_intercept, stick_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate stickiness parameter
    # Positive value means tendency to repeat; Negative means tendency to switch.
    stickiness = stick_intercept + (stick_oci_slope * oci_score)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Add stickiness bonus to Q-values before softmax
        q_stage1_adjusted = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_adjusted[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_stage1_adjusted)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # No stickiness modeled for stage 2 (aliens) in this variant, just stage 1 (spaceships)
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        # Standard SARSA/Q-learning updates
        
        # Update Stage 2
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1
        # Using the value of the chosen stage 2 option to update stage 1
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Second-Stage Learning Rate (Separate Stage 2 Alpha)
This model suggests that OCI affects how quickly participants update their beliefs about the specific outcomes (the aliens/coins). Obsessive-compulsive traits might lead to "over-learning" from local prediction errors or, conversely, a rigidity that resists updating. Here, we separate the learning rate for the second stage ($\alpha_2$) and make it a function of OCI, while keeping a base learning rate for the first stage.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with separate Stage 2 learning rate modulated by OCI.
    
    Bounds:
    alpha_1: [0,1]
    alpha_2_base: [0,1]
    alpha_2_oci_mod: [-1,1]
    beta: [0,10]
    """
    alpha_1, alpha_2_base, alpha_2_oci_mod, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate Stage 2 learning rate based on OCI
    # alpha_2 determines how fast they update alien probabilities
    alpha_2 = alpha_2_base + (alpha_2_oci_mod * oci_score)
    alpha_2 = np.clip(alpha_2, 0.001, 0.999)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        
        # Update Stage 2 using OCI-modulated alpha_2
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += alpha_2 * delta_stage2
        
        # Update Stage 1 using standard alpha_1
        # Using the updated value of stage 2 to drive stage 1 learning
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += alpha_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
