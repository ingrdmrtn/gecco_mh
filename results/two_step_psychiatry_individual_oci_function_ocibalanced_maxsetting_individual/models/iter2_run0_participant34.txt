Here are the three proposed cognitive models.

### Model 1: OCI-Modulated Loss-Stickiness
This model hypothesizes that the "stickiness" (perseveration) observed in high-OCI participants is not a general tendency to repeat actions, but specifically a "compulsive checking" response triggered by failure. If the previous trial resulted in a loss (0 coins), the model adds a bonus to the previously chosen spaceship, scaled by the OCI score. This captures the behavior of repeating an unrewarded action to "fix" or "check" it.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated stickiness specifically after losses.
    Hypothesis: High OCI participants exhibit 'compulsive' repetition (stickiness) 
    primarily after unrewarded trials (checking behavior), rather than general perseveration.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta: [0,10] - Inverse temperature (exploitation vs exploration).
    w: [0,1] - Weighting between Model-Based (1) and Model-Free (0).
    lambda_elig: [0,1] - Eligibility trace decay.
    stick_loss_oci: [0,10] - Stickiness magnitude added after a loss, scaled by OCI.
    """
    learning_rate, beta, w, lambda_elig, stick_loss_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 
    
    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value Calculation
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Loss-Specific Stickiness
        # If the previous trial resulted in no reward, add a sticky bonus to the previous choice
        if last_action_1 != -1 and last_reward == 0:
            q_net_stage1[last_action_1] += stick_loss_oci * current_oci

        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        # Softmax Choice 2 (Pure Model-Free)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Prediction Errors
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Update Stage 1 MF (with eligibility trace)
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_elig * delta_stage2)
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Eligibility Trace
This model hypothesizes that high OCI scores correlate with a disruption in the credit assignment mechanism that links distal outcomes (Stage 2 rewards) to proximal choices (Stage 1). The eligibility trace parameter `lambda` is dampened by the OCI score. A lower `lambda` means the Stage 1 Model-Free values are less responsive to the final reward, potentially leading to a disconnect between action and outcome that fosters rigid habits.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where the eligibility trace (lambda) is dampened by OCI.
    Hypothesis: High OCI impairs the 'chaining' of credit assignment in the MF system,
    reducing the influence of Stage 2 outcomes on Stage 1 values.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting between MB and MF.
    lambda_base: [0,1] - Base eligibility trace parameter.
    oci_lambda_damp: [0,1] - Factor by which OCI reduces lambda: lambda = lambda_base * (1 - oci * damp).
    """
    learning_rate, beta, w, lambda_base, oci_lambda_damp = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective lambda
    lambda_eff = lambda_base * (1.0 - current_oci * oci_lambda_damp)
    if lambda_eff < 0: lambda_eff = 0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Use effective lambda modulated by OCI
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_eff * delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Stage 1 Rigidity
This model addresses the observation that the participant shows extreme rigidity in Stage 1 choices (spaceships) but relatively rapid adaptation in Stage 2 choices (aliens). It proposes separate learning rates for the two stages, where the Stage 1 learning rate is specifically dampened by the OCI score. This creates a "sticky" high-level policy while maintaining a flexible low-level policy.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with separate learning rates for Stage 1 and Stage 2.
    Stage 1 learning rate is dampened by OCI, reflecting specific rigidity in high-level choices,
    while Stage 2 learning remains intact (preserving local adaptation).
    
    Parameters:
    lr_base: [0,1] - Base learning rate (applied fully to Stage 2).
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting between MB and MF.
    lambda_elig: [0,1] - Eligibility trace.
    oci_s1_rigidity: [0,1] - Dampening factor for Stage 1 LR: lr_s1 = lr_base * (1 - oci * rigidity).
    """
    lr_base, beta, w, lambda_elig, oci_s1_rigidity = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Separate learning rates
    lr_s1 = lr_base * (1.0 - current_oci * oci_s1_rigidity)
    if lr_s1 < 0: lr_s1 = 0
    lr_s2 = lr_base
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Update Stage 1 using dampened learning rate
        q_stage1_mf[a1] += lr_s1 * (delta_stage1 + lambda_elig * delta_stage2)
        
        # Update Stage 2 using base learning rate
        q_stage2_mf[state_idx, a2] += lr_s2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
