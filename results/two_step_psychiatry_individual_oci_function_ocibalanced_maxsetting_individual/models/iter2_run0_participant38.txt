Here are three new cognitive models that explore different mechanisms for how high OCI scores might influence learning and decision-making in the two-step task.

### Model 1: Hybrid Learner with OCI-modulated Model-Based Weight
This model assumes that individuals with high OCI scores might rely more on habitual (model-free) control and less on goal-directed (model-based) planning, or vice versa. Here, the weighting parameter `w` (which balances model-based and model-free values) is directly modulated by the OCI score. A high OCI might lead to a "compulsive" reliance on habits (lower `w`).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the balance (w) is modulated by OCI.
    
    Hypothesis: High OCI participants may show an imbalance between goal-directed (MB)
    and habitual (MF) control. The mixing weight 'w' is adjusted by OCI.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature (softmax randomness).
    w_base: [0,1] - Baseline weight for Model-Based values (0=Pure MF, 1=Pure MB).
    oci_w_mod: [0,1] - Strength of OCI modulation on w.
    """
    learning_rate, beta, w_base, oci_w_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective w. We assume high OCI reduces model-based control (compulsivity hypothesis).
    # We clip to ensure it stays within [0, 1].
    w = np.clip(w_base - (oci_w_mod * oci_score), 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Used for both MF stage 2 and MB calculation

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value Calculation
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        # Stage 2 update (Standard Q-learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 MF update (TD(1) logic roughly, updating from Stage 2 value)
        # Note: In hybrid models, MF is usually updated by the best Stage 2 value or the actual value found.
        # Here we use the standard TD error based on the value of the state reached.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Asymmetric Learning Rates Modulated by OCI
This model investigates if high OCI scores relate to an asymmetry in how positive (wins) versus negative (losses/omissions) outcomes are processed. Specifically, it tests if high OCI leads to "hyper-learning" from failure (fear of mistakes/checking behavior) or success.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with separate learning rates for positive and negative outcomes.
    The negative learning rate is specifically boosted by the OCI score.
    
    Hypothesis: High OCI individuals may be hypersensitive to negative feedback (0 coins),
    updating their internal values more drastically after a loss than a win.
    
    Parameters:
    alpha_pos: [0,1] - Learning rate for positive rewards (1 coin).
    alpha_neg_base: [0,1] - Base learning rate for negative rewards (0 coins).
    beta: [0,10] - Inverse temperature.
    oci_neg_boost: [0,1] - How much OCI increases sensitivity to negative outcomes.
    """
    alpha_pos, alpha_neg_base, beta, oci_neg_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective negative learning rate
    alpha_neg = np.clip(alpha_neg_base + (oci_neg_boost * oci_score), 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        # Determine which learning rate to use based on reward
        current_alpha = alpha_pos if r > 0 else alpha_neg
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += current_alpha * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Choice Stickiness with OCI-driven Decay
This model modifies the concept of "stickiness" (perseveration). Instead of a static stickiness bonus, this model implements a "trace" of previous actions that decays over time. The *decay rate* of this trace is modulated by OCI. A high OCI might result in a slower decay of past choices (mental stickiness/rumination), meaning previous choices influence current behavior for longer.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner where choice perseveration (stickiness) decays over time.
    The decay rate is modulated by OCI.
    
    Hypothesis: High OCI relates to 'mental stickiness' or difficulty disengaging 
    from past actions. High OCI leads to slower decay (higher retention) of the choice trace.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature.
    stickiness_weight: [0,5] - Impact of the choice trace on decision values.
    decay_base: [0,1] - Base decay rate (retention) of the choice trace (0=instant forget, 1=perfect memory).
    oci_decay_boost: [0,1] - How much OCI increases the retention (slows decay) of the trace.
    """
    learning_rate, beta, stickiness_weight, decay_base, oci_decay_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective decay (retention) parameter. 
    # Higher OCI -> Higher retention (closer to 1.0) -> Slower forgetting of past choice
    decay = np.clip(decay_base + (oci_decay_boost * oci_score), 0.0, 0.99)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for stage 1 actions
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        # Add stickiness based on the decaying trace
        q_net = q_stage1_mf + (stickiness_weight * choice_trace)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Update Choice Trace
        # Decay existing trace
        choice_trace *= decay
        # Add current choice to trace (indicator function)
        choice_trace[a1] += 1.0

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
