Here are the 3 proposed cognitive models.

### Model 1: OCI-Persistence Stickiness Model
This model hypothesizes that the Obsessive-Compulsive Inventory (OCI) score determines the *duration* (decay rate) of the perseverance "trace", while the weight of this stickiness is a free parameter. High OCI leads to slower decay of choice traces (longer memory of past actions), creating a more persistent compulsion to repeat choices, whereas low OCI leads to rapid decay.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Persistence Stickiness Model.
    
    The stickiness (perseveration) trace decay rate is fixed by the OCI score.
    Decay = 0.5 + 0.5 * OCI. 
    High OCI -> High Decay (Trace persists). Low OCI -> Low Decay (Trace vanishes).
    The model fits the weight of this stickiness.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - stick_weight: [0, 10] Strength of the stickiness bias.
    """
    learning_rate, beta, w, stick_weight = model_parameters
    oci_val = oci[0]
    
    # Define decay rate based on OCI (Mapping [0,1] OCI to [0.5, 1.0] Decay)
    # This implies even low OCI has some short-term memory, but high OCI has long-term.
    stick_decay = 0.5 + 0.5 * oci_val
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Stickiness trace for Stage 1 actions
    stick_trace = np.zeros(2)
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        # Handle missing data
        if a1 == -1:
            continue
            
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus based on the trace
        q_net += stick_weight * stick_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update stickiness trace
        # Decay the trace for both actions, then increment for the chosen one
        stick_trace *= stick_decay
        stick_trace[a1] += 1.0

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Learning Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # TD(1) update for stage 1 (standard in hybrid models)
        q_stage1_mf[a1] += learning_rate * delta_stage2

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Loss Aversion Model
This model hypothesizes that OCI correlates with a heightened sensitivity to negative outcomes (missing a reward). Instead of treating a 0 reward as simply "neutral", the model treats it as a "loss" with a magnitude scaled by the OCI score. This affects the Q-values, making the agent more avoidant of options that fail to yield gold.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Loss Aversion Model.
    
    Hypothesis: OCI modulates the subjective utility of missing a reward (Reward=0).
    Effective Reward = Reward - (1 - Reward) * (loss_sensitivity * OCI).
    If Reward is 1, value is 1.
    If Reward is 0, value is -(loss_sensitivity * OCI).
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - loss_sensitivity: [0, 10] Scaling factor for the negative utility of 0 reward.
    """
    learning_rate, beta, w, loss_sensitivity = model_parameters
    oci_val = oci[0]
    
    n_trials = len(action_1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            continue
            
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        raw_r = reward[trial]
        
        # Calculate Effective Reward
        # If raw_r = 1, term is 0 -> eff_r = 1
        # If raw_r = 0, term is 1 -> eff_r = - (loss_sens * oci)
        loss_penalty = loss_sensitivity * oci_val
        eff_r = raw_r - (1.0 - raw_r) * loss_penalty

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Learning Updates with Effective Reward
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = eff_r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Transition Distortion Model
This model hypothesizes that high OCI leads to a rigid, exaggerated belief in the structure of the world. The transition matrix used for Model-Based planning is "distorted" towards determinism based on the OCI score. A high OCI inflates the probability of the common transition (e.g., from 0.7 to 0.9), making the MB system more confident (and potentially more rigid) in its state predictions.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Transition Distortion Model.
    
    Hypothesis: OCI leads to an exaggerated belief in the transition structure (rigidity).
    The Model-Based system uses a distorted transition matrix where the common probability
    is increased by the OCI score.
    P_common_distorted = 0.7 + (distortion_factor * OCI). (Clipped to max 1.0).
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - distortion_factor: [0, 1] Scaling factor for how much OCI exaggerates the transition probability.
    """
    learning_rate, beta, w, distortion_factor = model_parameters
    oci_val = oci[0]
    
    n_trials = len(action_1)
    
    # Calculate Distorted Transition Matrix
    # Base common probability is 0.7.
    p_common = 0.7 + (distortion_factor * oci_val)
    if p_common > 0.99: p_common = 0.99 # Cap to avoid numerical issues
    
    p_rare = 1.0 - p_common
    # Matrix: [[Common, Rare], [Rare, Common]]
    transition_matrix = np.array([[p_common, p_rare], [p_rare, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            continue
            
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy uses the DISTORTED transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Learning Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
