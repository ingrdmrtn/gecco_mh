Here are three new cognitive models for the two-step decision task, designed to capture the influence of Obsessive-Compulsive symptoms (OCI) on decision-making processes.

### Model 1: OCI Modulates Mixing Weight (Interpolation)
This model hypothesizes that OCI severity directly alters the balance between Model-Based (goal-directed) and Model-Free (habitual) control. High OCI scores may lead to a reliance on one system over the other (e.g., increased habit propensity). Unlike previous slope-based attempts, this uses linear interpolation between low and high OCI set-points, ensuring parameters remain within valid bounds.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the Mixing Weight (w) via interpolation.
    Hypothesis: OCI severity shifts the balance between Model-Based (Goal-Directed) 
    and Model-Free (Habitual) control.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta_1: [0, 10] Inverse temperature for Stage 1 choices.
    - beta_2: [0, 10] Inverse temperature for Stage 2 choices.
    - stickiness: [0, 5] Choice perseveration bonus (added to last choice).
    - w_low: [0, 1] Mixing weight (0=MF, 1=MB) for low OCI (0).
    - w_high: [0, 1] Mixing weight (0=MF, 1=MB) for high OCI (1).
    """
    learning_rate, beta_1, beta_2, stickiness, w_low, w_high = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Interpolate w based on OCI
    w = w_low * (1 - current_oci) + w_high * current_oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness to the net Q-value of the previously chosen action
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Policy for the second choice
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Action value updating for Choice 1 (TD-0)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Action value updating for Choice 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI Modulates Stage 2 Learning Rate
This model hypothesizes that the core deficit or variation linked to OCI lies in the basic reward learning mechanism (Stage 2), rather than the high-level planning or global learning rate. It separates the learning rates for Stage 1 (fixed) and Stage 2 (modulated by OCI).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the Stage 2 Learning Rate.
    Hypothesis: OCI affects the basic reward learning mechanism (Stage 2) specifically,
    impacting how quickly alien reward probabilities are updated.
    
    Parameters:
    - lr_s1: [0, 1] Learning rate for Stage 1 (fixed across OCI).
    - lr_s2_low: [0, 1] Learning rate for Stage 2 when OCI is low.
    - lr_s2_high: [0, 1] Learning rate for Stage 2 when OCI is high.
    - beta_1: [0, 10] Inverse temperature for Stage 1.
    - beta_2: [0, 10] Inverse temperature for Stage 2.
    - w: [0, 1] Mixing weight.
    - stickiness: [0, 5] Perseveration bonus.
    """
    lr_s1, lr_s2_low, lr_s2_high, beta_1, beta_2, w, stickiness = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Interpolate Stage 2 Learning Rate based on OCI
    lr_s2 = lr_s2_low * (1 - current_oci) + lr_s2_high * current_oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Policy for the second choice
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Action value updating for Choice 1 using fixed lr_s1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_s1 * delta_stage1
        
        # Action value updating for Choice 2 using modulated lr_s2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_s2 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI Modulates Choice Trace Decay
This model introduces a "Choice Trace" (or habit kernel) that accumulates over time and decays, rather than a simple 1-step stickiness. It hypothesizes that OCI severity affects the *persistence* of habits (how fast the choice trace decays). Low decay implies "deep" habits (compulsivity), while high decay implies fleeting habits.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with Choice Trace (Decaying Stickiness) modulated by OCI.
    Hypothesis: OCI affects the persistence of habits. Instead of simple 1-step stickiness,
    this model uses a choice trace that accumulates and decays. OCI modulates the decay rate.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_1: [0, 10] Inverse temperature Stage 1.
    - beta_2: [0, 10] Inverse temperature Stage 2.
    - w: [0, 1] Mixing weight.
    - stick_weight: [0, 5] Weight of the choice trace in decision making.
    - decay_low: [0, 1] Decay rate of choice trace for low OCI (0=no decay, 1=instant).
    - decay_high: [0, 1] Decay rate of choice trace for high OCI.
    """
    learning_rate, beta_1, beta_2, w, stick_weight, decay_low, decay_high = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Interpolate Decay Rate
    decay = decay_low * (1 - current_oci) + decay_high * current_oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for Stage 1 actions (A and U)
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add choice trace influence (accumulated habit)
        q_net_1 += stick_weight * choice_trace
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Policy for the second choice
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Action value updating for Choice 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Action value updating for Choice 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Choice Trace
        # Decay both traces
        choice_trace *= (1 - decay)
        # Increment chosen trace
        choice_trace[a1] += 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
