Here are 3 new cognitive models based on the participant data and OCI score.

### Cognitive Model 1: OCI-Modulated Model-Based Weighting
This model tests the hypothesis that higher OCI scores correlate with a rigid reliance on Model-Based (habit-like or rule-based) planning, or conversely, a deficit in Model-Based control depending on the literature interpretation. Here, we model OCI as directly shifting the balance parameter `w` between Model-Free (MF) and Model-Based (MB) control. A base weight is adjusted by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where the mixing weight 'w' is a function of OCI.
    
    Hypothesis: OCI score determines the balance between goal-directed (MB) 
    and habitual (MF) control.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1]
    w_oci_sens: [-1, 1] (Sensitivity of w to OCI score)
    """
    learning_rate, beta, w_base, w_oci_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate w based on OCI, clamping to [0, 1]
    # If w_oci_sens is positive, higher OCI -> more MB.
    # If negative, higher OCI -> more MF.
    w = w_base + (w_oci_sens * current_oci)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
             p_choice_1[trial] = 0.5
             p_choice_2[trial] = 0.5
             continue

        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Learning ---
        # Update Stage 2 Q-values
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF Q-values (TD(0))
        # Note: Standard Daw et al. 2011 uses Q(s2, a2) for the update, not max.
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Driven Punishment Sensitivity
This model investigates if high OCI is linked to an oversensitivity to negative outcomes (punishment). In the data, the participant encounters trials with -1.0 reward. This model introduces a separate learning rate for negative prediction errors, where the magnitude of this asymmetry is controlled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with OCI-modulated punishment sensitivity (asymmetric learning rates).
    
    Hypothesis: Higher OCI leads to stronger learning from negative prediction errors
    (avoidance learning).
    
    Bounds:
    lr_pos: [0, 1] (Learning rate for positive PE)
    beta: [0, 10]
    punish_sens: [0, 5] (Multiplier for negative PE learning rate based on OCI)
    """
    lr_pos, beta, punish_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate negative learning rate based on OCI
    # If punish_sens is high, OCI increases the impact of negative outcomes.
    lr_neg = lr_pos * (1.0 + punish_sens * current_oci)
    # Clamp to reasonable bounds to prevent instability, though strictly >1 is allowed in theory
    lr_neg = np.clip(lr_neg, 0.0, 1.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Free formulation for simplicity to isolate learning rate effects
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
             p_choice_1[trial] = 0.5
             p_choice_2[trial] = 0.5
             continue

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Learning ---
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2[state_idx, act2]
        lr_s2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2[state_idx, act2] += lr_s2 * delta_stage2
        
        # Stage 1 Update (SARSA-like connection to stage 2 value)
        delta_stage1 = q_stage2[state_idx, act2] - q_stage1[act1]
        lr_s1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1[act1] += lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Scaled Inverse Temperature (Exploration/Exploitation)
This model proposes that OCI affects the overall randomness of behavior (decision noise). A high OCI might lead to very deterministic behavior (high beta) due to rigidity, or very noisy behavior (low beta) due to anxiety/uncertainty. We model `beta` as a linear function of OCI.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model where the inverse temperature (beta) is modulated by OCI.
    
    Hypothesis: OCI affects the exploration-exploitation trade-off. 
    High OCI might lead to more rigid (higher beta) or more chaotic (lower beta) choices.
    
    Bounds:
    learning_rate: [0, 1]
    beta_base: [0, 10]
    beta_oci_mod: [-5, 5] (Modification of beta by OCI)
    """
    learning_rate, beta_base, beta_oci_mod = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective beta
    beta = beta_base + (beta_oci_mod * current_oci)
    # Ensure beta stays positive and within reasonable bounds
    beta = np.clip(beta, 0.0, 20.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Using a simple Model-Based formulation to see if noise explains variance better than complex hybrid dynamics
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1 or state[trial] == -1 or action_2[trial] == -1:
             p_choice_1[trial] = 0.5
             p_choice_2[trial] = 0.5
             continue

        # --- Stage 1 Choice (Pure Model-Based for this structure) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        exp_q1 = np.exp(beta * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Learning ---
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        
        # No Stage 1 MF update in this simplified MB-only + noise model

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
