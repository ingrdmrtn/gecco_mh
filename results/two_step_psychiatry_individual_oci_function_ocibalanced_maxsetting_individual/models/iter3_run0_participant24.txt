Here are three new cognitive models inspired by the participant's data and OCI score. These models explore different mechanisms for how obsessive-compulsive traits might influence reinforcement learning, specifically focusing on mixing model-based/model-free strategies, eligibility traces, and separate learning rates for positive and negative outcomes.

### Cognitive Model 1: Hybrid Model with OCI-modulated Mixing Weight
This model posits that the balance between goal-directed (Model-Based) and habitual (Model-Free) control is influenced by the OCI score. High OCI might lead to a stronger reliance on habits (Model-Free) or rigid rules, reducing the contribution of the flexible Model-Based system.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with OCI-modulated Mixing Weight.
    
    The parameter 'w' controls the balance between MB and MF control (1 = full MB, 0 = full MF).
    This model assumes the OCI score shifts this balance.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for MF value updates.
    beta: [0, 10] - Inverse temperature for softmax.
    w_base: [0, 1] - Baseline mixing weight (tendency towards MB).
    oci_w_penalty: [0, 1] - Reduction in MB weight proportional to OCI score.
    """
    learning_rate, beta, w_base, oci_w_penalty = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate mixing weight w, bounded between 0 and 1
    # Higher OCI reduces w, pushing behavior towards Model-Free
    w = w_base - (oci_w_penalty * oci_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-Free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-Free Q-values for stage 2 (also used for MB calculation)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: V_MB(s1) = T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0

        # --- Stage 2 Policy ---
        if s_idx != -1:
            exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
            probs_2 = exp_q2 / np.sum(exp_q2)
            
            if a2 != -1:
                p_choice_2[trial] = probs_2[a2]
                
                # --- Learning ---
                # Stage 2 Update (TD error)
                delta_stage2 = r - q_stage2_mf[s_idx, a2]
                q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
                
                # Stage 1 Update (TD(0) for MF)
                if a1 != -1:
                    # Note: In a pure hybrid model, stage 1 MF is often updated by stage 2 Q-value (SARSA-like)
                    # or by the reward directly (if lambda=1). Here we use simple TD(1) style for simplicity
                    # effectively updating Q1 based on the final reward.
                    delta_stage1 = r - q_stage1_mf[a1]
                    q_stage1_mf[a1] += learning_rate * delta_stage1
            else:
                p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: TD(lambda) with OCI-modulated Eligibility Trace
This model assumes the participant uses an eligibility trace parameter ($\lambda$) to connect the second-stage outcome back to the first-stage choice. The OCI score modulates this parameter, reflecting how strongly outcomes are attributed to earlier choices (a form of credit assignment). High OCI might lead to "over-attribution" or stronger temporal linking ($\lambda$ close to 1).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free TD(lambda) where OCI modulates the eligibility trace (lambda).
    
    Lambda controls how much the stage 2 reward updates the stage 1 choice.
    Lambda = 1 corresponds to direct reinforcement of stage 1 by reward.
    Lambda = 0 corresponds to stage 1 learning only from stage 2 state value.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    lambda_base: [0, 1] - Base eligibility trace parameter.
    oci_lambda_shift: [-1, 1] - Shift in lambda based on OCI (bounded 0-1).
    """
    learning_rate, beta, lambda_base, oci_lambda_shift = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective lambda
    eff_lambda = lambda_base + (oci_lambda_shift * oci_score)
    eff_lambda = np.clip(eff_lambda, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0

        # --- Stage 2 Choice ---
        if s_idx != -1:
            exp_q2 = np.exp(beta * q_stage2[s_idx])
            probs_2 = exp_q2 / np.sum(exp_q2)
            if a2 != -1:
                p_choice_2[trial] = probs_2[a2]
                
                # --- Learning ---
                # Prediction error at stage 2
                delta2 = r - q_stage2[s_idx, a2]
                
                # Update stage 2 value
                q_stage2[s_idx, a2] += learning_rate * delta2
                
                if a1 != -1:
                    # Prediction error at stage 1
                    # The value of state 2 is q_stage2[s_idx, a2] (SARSA style for stage 1 target)
                    delta1 = q_stage2[s_idx, a2] - q_stage1[a1]
                    
                    # Update stage 1: Q1 += alpha * (delta1 + lambda * delta2)
                    # This standard TD(lambda) update combines the immediate transition error
                    # and the subsequent reward error scaled by lambda.
                    q_stage1[a1] += learning_rate * (delta1 + eff_lambda * delta2)
            else:
                p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Dual Learning Rates with OCI-Specific Punishment Sensitivity
This model separates learning from positive rewards (`lr_pos`) and negative outcomes/omissions (`lr_neg`). It hypothesizes that OCI is specifically linked to an altered sensitivity to negative outcomes (or lack of reward), scaling the negative learning rate.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free RL with separate learning rates for positive and negative prediction errors.
    OCI modulates the learning rate for negative errors (punishment/omission sensitivity).
    
    Parameters:
    lr_pos: [0, 1] - Learning rate for positive prediction errors (better than expected).
    lr_neg_base: [0, 1] - Base learning rate for negative prediction errors (worse than expected).
    beta: [0, 10] - Inverse temperature.
    oci_neg_boost: [0, 2] - Multiplier for negative learning rate based on OCI.
    """
    lr_pos, lr_neg_base, beta, oci_neg_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective negative learning rate
    # Higher OCI might increase sensitivity to negative outcomes
    lr_neg = lr_neg_base * (1 + oci_neg_boost * oci_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        if a1 != -1:
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 1.0

        # --- Stage 2 Choice ---
        if s_idx != -1:
            exp_q2 = np.exp(beta * q_stage2[s_idx])
            probs_2 = exp_q2 / np.sum(exp_q2)
            if a2 != -1:
                p_choice_2[trial] = probs_2[a2]
                
                # --- Learning ---
                # Stage 2 PE
                delta2 = r - q_stage2[s_idx, a2]
                
                # Choose learning rate based on sign of PE
                alpha2 = lr_pos if delta2 > 0 else lr_neg
                q_stage2[s_idx, a2] += alpha2 * delta2
                
                if a1 != -1:
                    # Stage 1 PE (TD(1) logic: update based on final reward directly for simplicity in this variant)
                    delta1 = r - q_stage1[a1]
                    
                    alpha1 = lr_pos if delta1 > 0 else lr_neg
                    q_stage1[a1] += alpha1 * delta1
            else:
                p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
