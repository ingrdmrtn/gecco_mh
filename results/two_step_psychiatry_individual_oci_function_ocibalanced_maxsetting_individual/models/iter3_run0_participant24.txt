Here are the three proposed cognitive models.

### Model 1: Stage 2 Learning Rate Modulation
This model hypothesizes that OCI symptoms specifically influence how quickly participants update their beliefs about the immediate reward sources (the aliens at Stage 2), potentially reflecting a hyper-focus on outcomes or "checking" the validity of the terminal states.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dual Learning Rate Model where OCI modulates the Stage 2 (Alien) learning rate.
    
    Hypothesis: High OCI individuals may exhibit different learning dynamics for immediate 
    outcomes (Stage 2) compared to abstract structure (Stage 1). This model allows the 
    Stage 2 learning rate to vary with OCI, while Stage 1 learning rate is fixed.
    
    Parameters:
    - lr_stage1: [0, 1] Learning rate for Stage 1 (Spaceships).
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    - lr_stage2_base: [0, 1] Baseline learning rate for Stage 2.
    - lr_stage2_oci_slope: [-1, 1] Change in Stage 2 learning rate per unit of OCI.
    """
    lr_stage1, beta, w, lr_stage2_base, lr_stage2_oci_slope = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]

    # Calculate OCI-modulated Stage 2 learning rate
    lr_stage2 = lr_stage2_base + (lr_stage2_oci_slope * participant_oci)
    lr_stage2 = np.clip(lr_stage2, 0.0, 1.0) # Ensure bounds

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)      # Model-free values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for Stage 2 (State x Action)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value: Mixed MB and MF
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Softmax Choice 2 based on Stage 2 MF values for current state
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 Update (TD-error using Stage 2 value) - Uses fixed lr_stage1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Stage 2 Update (Prediction error using Reward) - Uses OCI-modulated lr_stage2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_stage2 * delta_stage2

    eps = 1e-10
    # Filter out 0 probabilities from missing trials before logging
    log_lik_1 = np.sum(np.log(p_choice_1[p_choice_1 > 0] + eps))
    log_lik_2 = np.sum(np.log(p_choice_2[p_choice_2 > 0] + eps))
    
    return -(log_lik_1 + log_lik_2)
```

### Model 2: Reward Sensitivity Modulation
This model hypothesizes that OCI symptoms affect the subjective valuation of rewards. High OCI might be associated with altered reward sensitivity (e.g., dampening the perceived value of a coin or amplifying it), which scales the prediction error and thus the magnitude of value updates.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Reward Sensitivity Model where OCI modulates the subjective magnitude of the reward.
    
    Hypothesis: OCI scores correlate with how rewards are perceived. Instead of a fixed 
    reward of 1.0, the participant perceives 'rho * 1.0'. This affects the prediction 
    error magnitude and asymptotic value estimates.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - rho_base: [0, 5] Baseline reward sensitivity factor.
    - rho_oci_slope: [-2, 2] Change in reward sensitivity per unit of OCI.
    """
    learning_rate, beta, w, rho_base, rho_oci_slope = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]

    # Calculate OCI-modulated Reward Sensitivity (rho)
    rho = rho_base + (rho_oci_slope * participant_oci)
    rho = np.maximum(rho, 0.0) # Sensitivity cannot be negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        raw_reward = reward[trial]
        
        # Apply subjective scaling to reward
        effective_reward = raw_reward * rho

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update using effective reward
        delta_stage2 = effective_reward - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_lik_1 = np.sum(np.log(p_choice_1[p_choice_1 > 0] + eps))
    log_lik_2 = np.sum(np.log(p_choice_2[p_choice_2 > 0] + eps))
    
    return -(log_lik_1 + log_lik_2)
```

### Model 3: Value Decay Modulation
This model hypothesizes that OCI symptoms influence the "forgetting" rate of unchosen options. High OCI might be associated with obsessive maintenance of values (low decay) or, conversely, a tunnel vision effect where unvisited states decay rapidly.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Value Decay Model where OCI modulates the decay rate of unchosen Stage 1 options.
    
    Hypothesis: OCI modulates the persistence of memory for unchosen options. 
    On each trial, the Q-value of the unchosen spaceship decays towards 0. 
    High decay implies a focus on recent experience with chosen options only.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for chosen options.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - decay_base: [0, 1] Baseline decay rate for unchosen options.
    - decay_oci_slope: [-1, 1] Change in decay rate per unit of OCI.
    """
    learning_rate, beta, w, decay_base, decay_oci_slope = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]

    # Calculate OCI-modulated Decay rate
    decay = decay_base + (decay_oci_slope * participant_oci)
    decay = np.clip(decay, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # 1. Decay unchosen Stage 1 option
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay)
        
        # 2. Standard Update for chosen Stage 1 option
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 3. Update Stage 2 option
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_lik_1 = np.sum(np.log(p_choice_1[p_choice_1 > 0] + eps))
    log_lik_2 = np.sum(np.log(p_choice_2[p_choice_2 > 0] + eps))
    
    return -(log_lik_1 + log_lik_2)
```
