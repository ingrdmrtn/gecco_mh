Here are three new cognitive models exploring different mechanisms for how Obsessive-Compulsive Inventory (OCI) scores might influence decision-making in this two-step task.

### Model 1: Hybrid Model with OCI-Modulated Model-Based Weight
This model assumes that the balance between Model-Based (planning) and Model-Free (habit) systems is influenced by OCI symptoms. Specifically, it tests the hypothesis that higher OCI scores might lead to either a rigid reliance on habits (lower `w`) or an over-deliberative planning style (higher `w`).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free learner where the mixing weight 'w' is 
    modulated by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for value estimation.
    - beta: [0, 10] Inverse temperature (softness of choice).
    - w_base: [0, 1] Baseline weight for model-based values.
    - oci_w_mod: [-1, 1] Modulation of 'w' by OCI score.
    """
    learning_rate, beta, w_base, oci_w_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate the mixing weight w based on OCI
    w = w_base + (oci_score * oci_w_mod)
    w = np.clip(w, 0.0, 1.0) # Ensure w stays within [0, 1]

    # Fixed transition matrix for Model-Based calculation
    # Row 0: transitions from Space A -> [Planet X, Planet Y]
    # Row 1: transitions from Space B -> [Planet X, Planet Y]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_mf1 = np.zeros(2)       # Model-Free Stage 1 values
    q_mf2 = np.zeros((2, 2))  # Model-Free Stage 2 values (2 planets, 2 aliens)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation: 
        # Bellman equation using known transitions and current max stage 2 values
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        # Integrated Q-value
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        # Softmax Choice Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial]) # Planet reached
        
        # Standard Model-Free Choice Stage 2
        exp_q2 = np.exp(beta * q_mf2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # TD Errors
        delta_2 = r - q_mf2[s_idx, a2]
        delta_1 = q_mf2[s_idx, a2] - q_mf1[a1] # SARSA-style update for stage 1
        
        # Updates
        q_mf2[s_idx, a2] += learning_rate * delta_2
        q_mf1[a1] += learning_rate * delta_1 
        # Note: Pure hybrid models often use TD(0) for MF stage 1 without eligibility traces 
        # to strictly separate MB and MF influence, though variants exist.

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Perseveration (Stickiness) Modulated by OCI
This model investigates if OCI scores relate to "stickiness" or perseverationâ€”the tendency to repeat the previous choice regardless of reward. This is often linked to compulsivity. The OCI score modulates how strong this repetition bias is.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with a 'stickiness' parameter (choice autocorrelation)
    that is modulated by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - stick_base: [0, 5] Base bonus added to the previously chosen action.
    - oci_stick_mod: [-5, 5] Modulation of stickiness by OCI.
    """
    learning_rate, beta, stick_base, oci_stick_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness
    stickiness = stick_base + (oci_score * oci_stick_mod)
    # No strict clipping to [0,1] needed as stickiness is an additive bias term, 
    # but let's keep it reasonable.
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with invalid action

    for trial in range(n_trials):
        
        # --- Stage 1 Decision ---
        # Add stickiness bonus to Q-values before softmax
        q_net_1 = q_stage1.copy()
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update tracker
        last_action_1 = a1
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        # No stickiness applied to stage 2 in this variant (usually applied to top-level choice)
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning (Simple TD) ---
        r = reward[trial]
        
        # Stage 2 Update
        delta_2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_2
        
        # Stage 1 Update (TD-0)
        # Using the value of the state reached (s_idx) as the target
        v_stage2_state = np.max(q_stage2[s_idx])
        delta_1 = v_stage2_state - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning Rates Modulated by OCI
This model posits that OCI symptoms might affect sensitivity to positive vs. negative prediction errors differently. For example, individuals with high compulsivity might over-learn from negative outcomes (avoidance) or fail to update properly after rewards. OCI here modulates the ratio or difference between positive (`alpha_pos`) and negative (`alpha_neg`) learning rates.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with separate learning rates for positive and negative 
    prediction errors. The negative learning rate is modulated by OCI.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors (RPE > 0).
    - alpha_neg_base: [0, 1] Base learning rate for negative prediction errors (RPE < 0).
    - oci_neg_mod: [-1, 1] Modulation of alpha_neg by OCI.
    - beta: [0, 10] Inverse temperature.
    """
    alpha_pos, alpha_neg_base, oci_neg_mod, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate
    alpha_neg = alpha_neg_base + (oci_score * oci_neg_mod)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        
        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 Prediction Error
        pe2 = r - q_stage2[s_idx, a2]
        
        # Apply asymmetric learning rate
        lr2 = alpha_pos if pe2 >= 0 else alpha_neg
        q_stage2[s_idx, a2] += lr2 * pe2
        
        # Stage 1 Prediction Error (using max Q of next state as proxy for value)
        v_next = np.max(q_stage2[s_idx])
        pe1 = v_next - q_stage1[a1]
        
        # Apply asymmetric learning rate
        lr1 = alpha_pos if pe1 >= 0 else alpha_neg
        q_stage1[a1] += lr1 * pe1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
