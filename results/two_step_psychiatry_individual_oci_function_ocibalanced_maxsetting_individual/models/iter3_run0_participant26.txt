Here are three new cognitive models that explore different mechanisms for how Obsessive-Compulsive Inventory (OCI) scores might influence decision-making in this two-step task.

### Model 1: Stickiness/Perseverance Modulated by OCI
This model hypothesizes that OCI scores relate to "stickiness" or choice perseverance. Individuals with higher OCI scores might exhibit more repetitive behaviors or a reluctance to switch choices regardless of reward history. Here, the OCI score modulates the perseverance parameter, adding a bonus to the previously chosen action.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model (Model-Based + Model-Free) with OCI-modulated choice perseverance.
    
    Hypothesis: OCI score modulates the 'stickiness' or perseverance of choice. 
    Higher OCI leads to a stronger tendency to repeat the previous action (perseverance),
    regardless of reward, reflecting compulsive repetition.
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) systems.
    - perseverance_base: [0, 5] Base perseverance bonus added to the previously chosen action.
    """
    learning_rate, beta, w, perseverance_base = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # OCI modulates the effective perseverance. 
    # Individuals with higher OCI get a larger bonus for repeating the last action.
    effective_perseverance = perseverance_base * (1.0 + current_oci)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Store previous action for perseverance bonus (initialize to -1)
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: 
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseverance bonus to the Q-value of the previous action
        if last_action_1 != -1:
            q_integrated[last_action_1] += effective_perseverance

        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Record action for next trial
        last_action_1 = a1
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        if s_idx == -1: continue

        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        # TD Error Stage 2
        delta2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta2

        # TD Error Stage 1 (Model-Free)
        # Using the value of the chosen state-action pair in stage 2
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Learning Rate Asymmetry
This model suggests that OCI affects how individuals learn from positive versus negative prediction errors. High OCI might be associated with a "better safe than sorry" approach or hyper-sensitivity to failure, leading to different learning rates for positive and negative outcomes.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-modulated asymmetric learning rates.
    
    Hypothesis: OCI affects sensitivity to negative outcomes.
    The model uses separate learning rates for positive (alpha_pos) and negative (alpha_neg) prediction errors.
    The negative learning rate is modulated by OCI, suggesting high OCI individuals might over-learn from failures (or under-learn).
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors.
    - alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting between Model-Based and Model-Free.
    """
    alpha_pos, alpha_neg_base, beta, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # OCI modulates how strongly the agent learns from negative prediction errors.
    # We assume higher OCI increases sensitivity to negative errors (anxiety/avoidance).
    # Bounding alpha_neg to be at most 1.
    alpha_neg = min(1.0, alpha_neg_base * (1.0 + current_oci))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        if s_idx == -1: continue

        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        # Stage 2 Update
        delta2 = r - q_stage2_mf[s_idx, a2]
        lr2 = alpha_pos if delta2 >= 0 else alpha_neg
        q_stage2_mf[s_idx, a2] += lr2 * delta2

        # Stage 1 Update
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        lr1 = alpha_pos if delta1 >= 0 else alpha_neg
        q_stage1_mf[a1] += lr1 * delta1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Model-Based/Model-Free Balance
This model proposes that OCI scores directly influence the balance between goal-directed (Model-Based) and habitual (Model-Free) control. High OCI is often associated with habit formation and compulsivity. This model tests if OCI shifts the weight `w` towards the Model-Free system.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI modulates the balance (w) between Model-Based and Model-Free control.
    
    Hypothesis: Higher OCI scores are associated with greater reliance on habitual (Model-Free) control
    and reduced reliance on goal-directed (Model-Based) control.
    Here, 'w_base' represents the maximum MB weight, which is attenuated by OCI.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w_base: [0, 1] The baseline weight for Model-Based control before OCI modulation.
    - lambda_eligibility: [0, 1] Eligibility trace parameter allowing Stage 2 reward to directly update Stage 1.
    """
    learning_rate, beta, w_base, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # OCI modulation: Higher OCI reduces 'w', shifting control towards Model-Free (0).
    # We model this as w decreasing linearly with OCI.
    # If OCI is 0, w = w_base. If OCI is high, w decreases.
    w = w_base * (1.0 - current_oci)
    # Ensure w stays non-negative
    w = max(0.0, w)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        if s_idx == -1: continue

        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        # Stage 2 Update
        delta2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta2

        # Stage 1 Update (TD)
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1
        
        # Stage 1 Eligibility Trace Update (Direct reinforcement from reward)
        # This reinforces the stage 1 choice based on the final reward, a hallmark of MF learning
        q_stage1_mf[a1] += learning_rate * lambda_eligibility * delta2

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
