Here are three cognitive models designed to explain the participant's behavior in the two-step task, incorporating the OCI-R score and addressing the observed behavioral streaks.

### Model 1: Hybrid Model with Separate Stage Learning Rates
This model hypothesizes that the participant learns the transition structure (Spaceship $\to$ Planet) and the reward structure (Alien $\to$ Gold) at different rates. The medium OCI score modulates the degree of choice stickiness (perseveration).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with separate learning rates for Stage 1 and Stage 2.
    
    Hypothesis: The participant updates values for the spaceship choice (Stage 1) 
    and the alien choice (Stage 2) with different learning rates.
    OCI score modulates the stickiness (perseveration) on the first-stage choice.
    
    Parameters:
    lr_s1: [0,1] - Learning rate for Stage 1 (Spaceship transitions).
    lr_s2: [0,1] - Learning rate for Stage 2 (Alien rewards).
    beta: [0,10] - Inverse temperature (exploration/exploitation).
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    stickiness_base: [0,5] - Base stickiness parameter, scaled by OCI.
    """
    lr_s1, lr_s2, beta, w, stickiness_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulates stickiness: higher OCI -> higher tendency to repeat choices
    stickiness = stickiness_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)      # Model-free values for spaceships
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for aliens (Planet x Alien)

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Handle missing data
        if a1 == -1 or s_idx == -1 or a2 == -1 or r == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 1 Prediction Error (Transition)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_s1 * delta_stage1
        
        # Stage 2 Prediction Error (Reward)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_s2 * delta_stage2
        
        # Propagate Stage 2 error to Stage 1 (Eligibility trace lambda implicitly 1)
        # We use lr_s1 for updates to stage 1 values
        q_stage1_mf[a1] += lr_s1 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Choice Kernel Perseveration
This model addresses the long streaks of identical choices observed in the data. Instead of simple one-step stickiness, it uses a "Choice Kernel" that builds up over time and decays slowly. The OCI score scales the influence of this kernel, hypothesizing that higher OCI leads to stronger habit formation/perseveration.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with Choice Kernel Perseveration.
    
    Hypothesis: The participant builds a 'choice trace' (kernel) that biases future choices.
    This explains the long streaks of repeating the same spaceship.
    OCI modulates the weight of this kernel.
    
    Parameters:
    lr: [0,1] - Learning rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight.
    k_base: [0,5] - Base weight of the choice kernel (scaled by OCI).
    phi: [0,1] - Decay rate of the choice kernel (0 = instant decay, 1 = no decay).
    """
    lr, beta, w, k_base, phi = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI scales the impact of the choice history
    k_weight = k_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    choice_kernel = np.zeros(2) # Trace for Stage 1 choices

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Choice Kernel influence
        q_net_1 += k_weight * choice_kernel
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Update Choice Kernel: Decay existing trace, reinforce current choice
        choice_kernel *= phi
        choice_kernel[a1] += (1.0 - phi)

        # Update Values
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Propagate to Stage 1
        q_stage1_mf[a1] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning with OCI-Modulated MB Weight
This model combines asymmetric learning rates (learning differently from positive vs. negative prediction errors) with the hypothesis that OCI affects the balance between Model-Based (goal-directed) and Model-Free (habitual) control.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with Asymmetric Learning Rates and OCI-modulated MB weight.
    
    Hypothesis: The participant learns differently from gains vs losses (Asymmetric LR).
    Additionally, higher OCI scores decrease the reliance on the Model-Based system 
    (lowering w), favoring Model-Free habits.
    
    Parameters:
    lr_pos: [0,1] - Learning rate for positive prediction errors.
    lr_neg: [0,1] - Learning rate for negative prediction errors.
    beta: [0,10] - Inverse temperature.
    w_base: [0,1] - Base mixing weight (Model-Basedness).
    stickiness: [0,5] - Simple choice stickiness parameter.
    """
    lr_pos, lr_neg, beta, w_base, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulates the mixing weight: Higher OCI -> Lower w (Less Model-Based)
    w = w_base * (1.0 - oci_score)
    if w < 0: w = 0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning with Asymmetry ---
        
        # Stage 1 Update (Transition PE)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        alpha_1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[a1] += alpha_1 * delta_stage1

        # Stage 2 Update (Reward PE)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        alpha_2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[s_idx, a2] += alpha_2 * delta_stage2
        
        # Propagate to Stage 1 (Lambda=1)
        # We use the Stage 2 alpha (alpha_2) for this update
        q_stage1_mf[a1] += alpha_2 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
