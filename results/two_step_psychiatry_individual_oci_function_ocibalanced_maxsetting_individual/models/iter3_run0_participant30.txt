import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Asymmetric Learning Rates.
    
    Hypothesis: High OCI participants exhibit perseveration by failing to 'unlearn' 
    actions that lead to negative outcomes. We model this by scaling the learning rate 
    for negative prediction errors (losses) by (1 - OCI).
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w: [0, 1] Mixing weight for Model-Based control (0=MF, 1=MB).
    - stickiness: [0, 5] General choice stickiness bonus (1-step).
    
    Mechanism:
    lr_neg = lr_pos * (1.0 - oci_score)
    If prediction error < 0: update uses lr_neg.
    Else: update uses lr_pos.
    """
    lr_pos, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate negative learning rate based on OCI
    # High OCI -> lr_neg approaches 0 (perseveration despite loss)
    lr_neg = lr_pos * (1.0 - oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness
        if trial > 0 and action_1[trial-1] != -1:
            prev_a1 = int(action_1[trial-1])
            q_net[prev_a1] += stickiness
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # Learning Updates
        # Stage 1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        alpha1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[a1] += alpha1 * delta_stage1
        
        # Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        alpha2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[s_idx, a2] += alpha2 * delta_stage2

    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Eligibility Trace (Lambda).
    
    Hypothesis: High OCI participants rely on direct reinforcement of the first-stage choice 
    by the final outcome (high lambda), bypassing the sequential temporal difference structure.
    
    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight for Model-Based control.
    - lambda_max: [0, 1] Maximum eligibility trace strength.
    
    Mechanism:
    lambda = lambda_max * oci_score
    Stage 1 MF values are updated via TD(lambda).
    """
    learning_rate, beta, w, lambda_max = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    lambda_val = lambda_max * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # Learning Updates
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Stage 1 Update: TD error + Eligibility Trace
        delta_stage1_td = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Q(s1,a1) += alpha * (Q(s2,a2) - Q(s1,a1)) + alpha * lambda * (r - Q(s2,a2))
        q_stage1_mf[a1] += learning_rate * delta_stage1_td + learning_rate * lambda_val * delta_stage2
        
        # Stage 2 Update
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model with OCI-modulated Habit Trace.
    
    Hypothesis: OCI drives the strength of a decaying "Choice Trace" (Habit).
    Unlike simple stickiness, this trace persists and decays, capturing longer-term perseveration.
    The influence of this trace is scaled by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - decay: [0, 1] Decay rate of the choice trace (1=no decay).
    - habit_oci_scale: [0, 5] Weight of the habit trace, scaled by OCI.
    
    Mechanism:
    habit_weight = habit_oci_scale * oci_score
    Q_net += habit_weight * trace
    trace[chosen] += 1; trace *= decay
    """
    learning_rate, beta, w, decay, habit_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    habit_weight = habit_oci_scale * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    choice_trace = np.zeros(2)
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB, MF, and OCI-weighted Habit Trace
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + habit_weight * choice_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # Learning Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Habit Trace
        choice_trace *= decay
        choice_trace[a1] += 1.0

    return log_loss
