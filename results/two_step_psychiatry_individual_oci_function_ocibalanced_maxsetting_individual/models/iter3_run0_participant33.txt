Here are three new cognitive models for the two-step task, incorporating the participant's OCI score to explain their decision-making behavior.

### Model 1: OCI-Modulated Subjective Loss Sensitivity
This model hypothesizes that high OCI scores correlate with a "fear of failure" or perfectionism, where receiving a 0 outcome is not perceived as neutral but as a subjective punishment (loss). The model introduces a parameter $\rho$ that transforms the reward $R=0$ into an effective reward $R_{eff} = -\rho$. This creates a stronger avoidance signal for unrewarded options compared to the standard model.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Subjective Loss Sensitivity Model.
    
    Hypothesis: High OCI participants may exhibit a "fear of failure", perceiving 
    a 0-coin outcome as a negative punishment rather than a neutral event. 
    This model transforms the reward function such that R=0 becomes R=-rho, 
    where rho is modulated by the OCI score. This increases the magnitude of 
    negative prediction errors and drives avoidance.
    
    Parameters:
    - lr: [0,1] Learning rate.
    - beta: [0,10] Inverse temperature (softness of choice).
    - w: [0,1] Mixing weight (0=Model-Free, 1=Model-Based).
    - lambd: [0,1] Eligibility trace parameter.
    - rho_base: [0,5] Baseline sensitivity to 0 outcome (subjective punishment).
    - rho_oci: [-5,5] Modulation of rho by OCI score.
    """
    lr, beta, w, lambd, rho_base, rho_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate subjective punishment magnitude
    rho = rho_base + rho_oci * oci_score
    rho = max(0.0, rho) # Ensure rho represents a penalty (non-negative magnitude)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if np.isnan(a1) or np.isnan(s2) or np.isnan(a2) or np.isnan(r):
            continue
            
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / (np.sum(exp_q1) + 1e-10)
        log_loss -= np.log(probs_1[a1] + 1e-10)
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / (np.sum(exp_q2) + 1e-10)
        log_loss -= np.log(probs_2[a2] + 1e-10)
        
        # --- Value Updates ---
        # Transform Reward: 0 becomes -rho
        if r == 0:
            r_eff = -rho
        else:
            r_eff = r
            
        # Stage 1 MF update (TD(1))
        delta_1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_1
        
        # Stage 2 MF update
        delta_2 = r_eff - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_2
        
        # Eligibility trace update for Stage 1
        q_stage1_mf[a1] += lr * lambd * delta_2
        
    return log_loss
```

### Model 2: OCI-Modulated Stage 1 Decision Noise
This model hypothesizes that OCD symptoms specifically affect the high-level "planning" or arbitration stage (Stage 1) differently from the concrete "action" stage (Stage 2). It allows for different inverse temperatures (betas) for the two stages, with the Stage 1 beta being modulated by OCI. This tests if high OCI is associated with specific indecisiveness (low beta) or rigidity (high beta) at the spaceship choice level.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stage 1 Decision Noise Model.
    
    Hypothesis: OCD symptoms may differentially impact the consistency of choices 
    at the abstract, high-level planning stage (Stage 1) versus the concrete 
    outcome stage (Stage 2). This model separates the inverse temperature (beta) 
    for the two stages and allows OCI to modulate the Stage 1 beta specifically.
    
    Parameters:
    - lr: [0,1] Learning rate.
    - beta_2: [0,10] Inverse temperature for Stage 2 (Action Choice).
    - w: [0,1] Mixing weight.
    - lambd: [0,1] Eligibility trace.
    - beta_1_base: [0,10] Baseline inverse temperature for Stage 1 (Spaceship Choice).
    - beta_1_oci: [-5,5] Modulation of Stage 1 beta by OCI.
    """
    lr, beta_2, w, lambd, beta_1_base, beta_1_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Stage 1 specific beta
    beta_1 = beta_1_base + beta_1_oci * oci_score
    beta_1 = max(0.0, beta_1) # Ensure non-negative
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if np.isnan(a1) or np.isnan(s2) or np.isnan(a2) or np.isnan(r):
            continue

        # --- Stage 1 Choice (Using beta_1) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / (np.sum(exp_q1) + 1e-10)
        log_loss -= np.log(probs_1[a1] + 1e-10)
        
        # --- Stage 2 Choice (Using beta_2) ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s2])
        probs_2 = exp_q2 / (np.sum(exp_q2) + 1e-10)
        log_loss -= np.log(probs_2[a2] + 1e-10)
        
        # --- Updates ---
        delta_1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_1
        
        delta_2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_2
        
        q_stage1_mf[a1] += lr * lambd * delta_2
        
    return log_loss
```

### Model 3: OCI-Modulated Counterfactual Updating
This model hypothesizes that high OCI participants engage in "fictitious play" or counterfactual reasoning. When they receive an outcome (e.g., Reward=0), they actively update the value of the *unchosen* spaceship as if it would have yielded the opposite outcome (Reward=1). This reflects a "grass is greener" bias or a belief in anti-correlated rewards, modulated by their OCI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Counterfactual Updating Model.
    
    Hypothesis: High OCI participants may update the value of the unchosen 
    Stage 1 option based on a counterfactual belief that it would have yielded 
    the opposite outcome to the chosen option (i.e., if chosen=0, unchosen=1).
    This simulates 'regret' or 'fictitious play'.
    
    Parameters:
    - lr: [0,1] Learning rate for chosen options.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Mixing weight.
    - lambd: [0,1] Eligibility trace.
    - cf_lr_base: [0,1] Baseline learning rate for unchosen Stage 1 options.
    - cf_lr_oci: [-1,1] Modulation of counterfactual learning rate by OCI.
    """
    lr, beta, w, lambd, cf_lr_base, cf_lr_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate counterfactual learning rate
    lr_cf = cf_lr_base + cf_lr_oci * oci_score
    lr_cf = np.clip(lr_cf, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_loss = 0.0
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if np.isnan(a1) or np.isnan(s2) or np.isnan(a2) or np.isnan(r):
            continue

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / (np.sum(exp_q1) + 1e-10)
        log_loss -= np.log(probs_1[a1] + 1e-10)
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / (np.sum(exp_q2) + 1e-10)
        log_loss -= np.log(probs_2[a2] + 1e-10)
        
        # --- Standard Updates ---
        delta_1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_1
        
        delta_2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_2
        
        q_stage1_mf[a1] += lr * lambd * delta_2
        
        # --- Counterfactual Update for Unchosen Stage 1 Option ---
        # Hypothesis: Agent assumes unchosen option yields (1 - R)
        unchosen_a1 = 1 - a1
        fictitious_reward = 1.0 - r
        
        # Update unchosen Stage 1 MF value towards the fictitious reward
        delta_cf = fictitious_reward - q_stage1_mf[unchosen_a1]
        q_stage1_mf[unchosen_a1] += lr_cf * delta_cf
        
    return log_loss
```
