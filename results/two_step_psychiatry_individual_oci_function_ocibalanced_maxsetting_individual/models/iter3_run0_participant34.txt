Here are three new cognitive models that incorporate the Obsessive-Compulsive Inventory (OCI) score to explain the participant's decision-making behavior.

### Model 1: OCI-Modulated Stage 1 Beta
This model hypothesizes that high OCI scores are associated with "hyper-exploitation" or rigidity specifically at the first stage of the decision (choosing a spaceship), while the second stage (choosing an alien) remains more flexible. This is implemented by scaling the inverse temperature (beta) for Stage 1 based on the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stage 1 Beta Model.
    
    Hypothesis:
    High OCI scores lead to 'hyper-exploitation' or reduced exploration specifically 
    at the first stage of the decision tree (choice of spaceship), while the second 
    stage (choice of alien) remains more flexible. This captures the rigid/compulsive 
    nature of the high-level choice.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta_base: [0,10] - Base inverse temperature for Stage 2 (and base for Stage 1).
    w: [0,1] - Weighting between Model-Based (1) and Model-Free (0).
    lambda_elig: [0,1] - Eligibility trace decay.
    oci_beta1_mult: [0,10] - Multiplier for Beta at Stage 1 based on OCI.
    
    Total parameters: 5
    """
    learning_rate, beta_base, w, lambda_elig, oci_beta1_mult = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Beta for Stage 1 is enhanced by OCI
    beta_1 = beta_base * (1.0 + oci_beta1_mult * current_oci)
    # Beta for Stage 2 is the base beta
    beta_2 = beta_base

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use beta_1 for stage 1 (potentially higher rigidity)
        exp_q1 = np.exp(beta_1 * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]

        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        # Use beta_2 for stage 2 (standard flexibility)
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_elig * delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Dual Stickiness (Base + OCI)
This model refines the concept of perseveration by separating it into a baseline tendency (common to all individuals) and a pathological component driven by OCI. This allows the model to distinguish between normal habit formation and compulsive repetition.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dual Stickiness Model (Base + OCI).
    
    Hypothesis:
    Perseveration is composed of a baseline tendency (common to all) and a 
    pathological component driven by OCI symptoms. Both contribute to repeating 
    the previous Stage 1 choice.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting between Model-Based (1) and Model-Free (0).
    lambda_elig: [0,1] - Eligibility trace decay.
    stick_base: [0,5] - Baseline stickiness parameter.
    stick_oci: [0,5] - Additional stickiness scaled by OCI score.
    
    Total parameters: 6
    """
    learning_rate, beta, w, lambda_elig, stick_base, stick_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Total stickiness is sum of base and OCI-dependent parts
    total_stickiness = stick_base + stick_oci * current_oci

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_net_stage1[last_action_1] += total_stickiness
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]

        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_elig * delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Dampened Eligibility Trace
This model hypothesizes that high OCI scores are associated with a disconnection between the final outcome and the initial choice (reduced credit assignment). This is modeled by reducing the eligibility trace (`lambda`) as OCI increases, making the Stage 1 value update rely more on the Stage 2 value prediction than the final reward.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Dampened Eligibility Trace Model.
    
    Hypothesis:
    High OCI scores are associated with a disconnection between the final outcome 
    and the initial choice (reduced credit assignment). This is modeled by 
    reducing the eligibility trace (lambda) as OCI increases.
    
    Parameters:
    learning_rate: [0,1] - Rate of value updating.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weighting between Model-Based (1) and Model-Free (0).
    lambda_base: [0,1] - Base eligibility trace.
    oci_lambda_damp: [0,5] - Factor by which OCI dampens lambda.
    
    Total parameters: 5
    """
    learning_rate, beta, w, lambda_base, oci_lambda_damp = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Lambda is dampened by OCI.
    # High OCI -> Lower Lambda -> Stage 1 update relies less on final Reward RPE
    lambda_val = lambda_base / (1.0 + oci_lambda_damp * current_oci)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]

        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Use the OCI-modulated lambda
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_val * delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
