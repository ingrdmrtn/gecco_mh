Here are the three proposed cognitive models.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Accelerated Habit Formation (Stage 1 Learning Rate Boost).
    Hypothesis: High OCI scores are associated with enhanced habit formation. 
    This model posits that OCI specifically accelerates the learning rate for 
    the Stage 1 Model-Free values (the "habit" component), while the Stage 2 
    learning rate (outcome value estimation) remains at a baseline level.
    
    Parameters:
    - learning_rate: [0, 1] Baseline learning rate (used for Stage 2).
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - habit_accel: [0, 5] Scaling factor for OCI to boost Stage 1 learning rate.
    """
    learning_rate, beta, w, habit_accel = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate Stage 1 specific learning rate
    # High OCI increases the speed at which Stage 1 actions are updated (habit strength)
    lr_stage1 = np.clip(learning_rate * (1.0 + habit_accel * oci_score), 0.0, 1.0)
    lr_stage2 = learning_rate

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        act1 = int(action_1[trial])
        state_idx = int(state[trial])
        act2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[act1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[act2]

        # --- Learning ---
        # Update Stage 2 (Goal Values) with baseline learning rate
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += lr_stage2 * delta_stage2

        # Update Stage 1 (Habit Values) with OCI-accelerated learning rate
        # Using TD(0) update
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += lr_stage1 * delta_stage1
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Distorted Transition Beliefs (Uncertainty Intolerance).
    Hypothesis: High OCI involves intolerance of uncertainty. This model suggests
    that high OCI participants perceive the transition probabilities as more 
    uncertain (closer to 50/50) than they truly are (70/30). This degrades the 
    accuracy of the Model-Based controller.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - uncertainty_bias: [0, 1] How much OCI flattens the transition matrix.
      (0 = accurate 0.7, 1 = maximum distortion towards 0.5 based on OCI).
    """
    learning_rate, beta, w, uncertainty_bias = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate subjective transition probability
    # True probability is 0.7. OCI reduces this towards 0.5.
    # Max reduction is 0.2 (0.7 -> 0.5).
    distortion = 0.2 * uncertainty_bias * oci_score
    subjective_p = 0.7 - distortion
    
    # Construct distorted transition matrix for MB planning
    # If OCI is high and bias is high, this approaches [[0.5, 0.5], [0.5, 0.5]]
    transition_matrix = np.array([[subjective_p, 1 - subjective_p], 
                                  [1 - subjective_p, subjective_p]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        act1 = int(action_1[trial])
        state_idx = int(state[trial])
        act2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # MB calculation uses the DISTORTED transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[act1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[act2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Win-Stay Heuristic Bias.
    Hypothesis: High OCI participants rely on rigid heuristic rules like "Win-Stay".
    This model adds a bias term to the decision values: if the previous action
    resulted in a reward, its value is boosted in the next trial, proportional to OCI.
    This is distinct from "stickiness" (which repeats regardless of reward).
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - ws_strength: [0, 5] Strength of the OCI-dependent Win-Stay bias.
    """
    learning_rate, beta, w, ws_strength = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Tracking previous trial outcome
    prev_act1 = -1
    prev_reward = 0.0

    for trial in range(n_trials):
        act1 = int(action_1[trial])
        state_idx = int(state[trial])
        act2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-modulated Win-Stay Bias
        # If previous trial was rewarded, boost the value of the action taken
        if trial > 0 and prev_reward == 1.0 and prev_act1 != -1:
            bias = np.zeros(2)
            bias[prev_act1] = ws_strength * oci_score
            q_net += bias
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[act1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[act2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        # Store history
        prev_act1 = act1
        prev_reward = r
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
