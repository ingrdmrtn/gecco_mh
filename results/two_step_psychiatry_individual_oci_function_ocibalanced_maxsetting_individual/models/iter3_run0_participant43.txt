import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Stage-Specific Learning Rate Modulated by OCI.
    
    Hypothesis: OCI scores influence the rate at which Stage 1 (habitual/choice) values are updated, 
    distinct from the learning rate for Stage 2 (state-value) updates. 
    High OCI may lead to accelerated habit formation (high Stage 1 learning rate) relative to 
    value estimation at Stage 2.

    Parameters:
    lr_s2: [0, 1] Learning rate for Stage 2 (Alien values).
    beta: [0, 10] Inverse temperature for both stages.
    w: [0, 1] Mixing weight (0=MF, 1=MB).
    lr_s1_base: [0, 1] Base learning rate for Stage 1.
    oci_lr1_mod: [0, 1] Additive modulation of Stage 1 learning rate by OCI.
    """
    lr_s2, beta, w, lr_s1_base, oci_lr1_mod = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate effective Stage 1 learning rate.
    # We hypothesize OCI increases the learning rate for the spaceship choice (habit strength).
    lr_s1 = lr_s1_base + (oci_lr1_mod * participant_oci)
    lr_s1 = np.clip(lr_s1, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Choice
        s_curr = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        r = reward[trial]
        
        # Stage 2 Update (Standard TD)
        pe_2 = r - q_stage2_mf[s_curr, a2]
        q_stage2_mf[s_curr, a2] += lr_s2 * pe_2
        
        # Stage 1 Update
        # 1. Update based on State prediction (TD(0) part of Stage 1)
        pe_1 = q_stage2_mf[s_curr, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_s1 * pe_1 
        
        # 2. Update based on Outcome (TD(1) / Eligibility trace part)
        # Using the specific Stage 1 learning rate here as well.
        q_stage1_mf[a1] += lr_s1 * pe_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Eligibility Trace (Lambda).
    
    Hypothesis: OCI affects the credit assignment process (eligibility trace). 
    High OCI might imply stronger linking of final rewards to initial choices (high lambda), 
    making the agent learn Model-Free values from the final outcome more robustly, 
    or conversely, a disconnection (low lambda).
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    lambda_base: [0, 1] Base eligibility trace.
    oci_lambda_mod: [0, 1] Modulation of lambda by OCI.
    """
    lr, beta, w, lambda_base, oci_lambda_mod = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate effective lambda
    lam = lambda_base + (oci_lambda_mod * participant_oci)
    lam = np.clip(lam, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2
        s_curr = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        r = reward[trial]
        
        # Stage 2 RPE
        pe_2 = r - q_stage2_mf[s_curr, a2]
        
        # Stage 1 RPE (from state transition)
        pe_1 = q_stage2_mf[s_curr, a2] - q_stage1_mf[a1]
        
        # Update Stage 2
        q_stage2_mf[s_curr, a2] += lr * pe_2
        
        # Update Stage 1
        # Part 1: Update from state transition
        q_stage1_mf[a1] += lr * pe_1
        # Part 2: Update from final reward (scaled by lambda)
        q_stage1_mf[a1] += lr * lam * pe_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Stage-Specific Inverse Temperature (Beta) with OCI Modulation on Stage 1.
    
    Hypothesis: High OCI leads to increased rigidity (higher beta) specifically at the 
    spaceship choice level (Stage 1), reflecting compulsive behavior, while Stage 2 
    (alien choice) remains more flexible or distinct. This explains perseveration without
    a specific 'stickiness' parameter.
    
    Parameters:
    lr: [0, 1] Learning rate.
    w: [0, 1] Mixing weight.
    beta_s2: [0, 10] Inverse temperature for Stage 2.
    beta_s1_base: [0, 10] Base inverse temperature for Stage 1.
    oci_beta1_mod: [0, 10] OCI scaling factor for Stage 1 beta (additive).
    """
    lr, w, beta_s2, beta_s1_base, oci_beta1_mod = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Effective Beta for Stage 1
    beta_s1 = beta_s1_base + (oci_beta1_mod * participant_oci)
    # Ensure it doesn't explode, though higher beta just means deterministic
    beta_s1 = np.clip(beta_s1, 0.0, 30.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use Stage 1 specific Beta
        exp_q1 = np.exp(beta_s1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2
        s_curr = int(state[trial])
        # Use Stage 2 specific Beta
        exp_q2 = np.exp(beta_s2 * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        r = reward[trial]
        pe_2 = r - q_stage2_mf[s_curr, a2]
        q_stage2_mf[s_curr, a2] += lr * pe_2
        
        pe_1 = q_stage2_mf[s_curr, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * pe_1
        q_stage1_mf[a1] += lr * pe_2 # Assume full lambda=1 for this model to isolate beta effect

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
