Here are 3 new cognitive models expressed as Python functions. These models introduce variations in how OCI symptoms interact with reinforcement learning mechanisms, specifically focusing on perseveration, model-based vs. model-free control, and learning rates.

### Model 1: OCI-Modulated Model-Based Suppression
This model hypothesizes that high OCI scores suppress the usage of Model-Based (planning) strategies in favor of simpler Model-Free (habitual) strategies. Instead of a fixed weight `w`, the weight is dynamically reduced by the OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Based Suppression.
    
    This model posits that high OCI symptoms reduce the reliance on Model-Based 
    (planning) strategies. The mixing weight 'w' is calculated as a base weight 
    minus a penalty proportional to the OCI score.
    
    Bounds:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w_max: [0, 1] Maximum model-based weight (for OCI=0).
    oci_suppression: [0, 1] Factor scaling how much OCI reduces MB weight.
    """
    lr, beta, w_max, oci_suppression = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective w: Higher OCI reduces w (less model-based)
    # Ensure w stays between 0 and 1
    w = w_max - (oci_suppression * oci_score)
    if w < 0: w = 0
    if w > 1: w = 1

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        r = reward[trial]
        
        # Prediction errors
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1] # SARSA-style update for MF

        # Update Q-values
        q_stage1_mf[a1] += lr * delta_stage1
        q_stage2_mf[s_idx, a2] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Driven Choice Stickiness (Decaying)
This model suggests that high OCI creates a "stickiness" or perseveration bias towards the previously chosen action, regardless of reward. Unlike standard stickiness, the strength of this bias is directly scaled by the OCI score, and we introduce a decay parameter `stick_decay` to allow this influence to persist or fade over time, capturing the "sticky" nature of compulsive thoughts.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Driven Choice Stickiness with Decay.
    
    This model assumes OCI symptoms increase perseveration (stickiness).
    A 'stickiness trace' tracks the history of choices. The influence of this
    trace on the current decision is scaled by the OCI score.
    
    Bounds:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (MB vs MF).
    stick_decay: [0, 1] How quickly the stickiness trace decays (0=instant, 1=never).
    oci_stick_scale: [0, 5] Scaling factor for OCI's effect on stickiness.
    """
    lr, beta, w, stick_decay, oci_stick_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Trace for choice stickiness (initialized to 0)
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus scaled by OCI
        # High OCI -> Higher bonus for actions with high trace values
        stickiness_bonus = oci_stick_scale * oci_score * choice_trace
        q_net_sticky = q_net + stickiness_bonus
        
        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update choice trace: decay old trace, add 1 to chosen action
        choice_trace = choice_trace * stick_decay
        choice_trace[a1] += 1.0

        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        q_stage1_mf[a1] += lr * delta_stage1
        q_stage2_mf[s_idx, a2] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Sensitive Negative Learning Rate
This model posits that individuals with high OCI symptoms are hyper-sensitive to negative outcomes (punishment/lack of reward). While they learn from rewards normally, they update their value estimates more aggressively when a reward is omitted (reward=0). This asymmetry is modulated by the OCI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Sensitive Negative Learning Rate.
    
    This model proposes that high OCI leads to hyper-sensitivity to negative outcomes.
    The learning rate for negative prediction errors (when reward is worse than expected)
    is boosted by the OCI score.
    
    Bounds:
    lr_base: [0, 1] Base learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    oci_neg_boost: [0, 5] Multiplier for OCI to increase learning rate on negative errors.
    """
    lr_base, beta, w, oci_neg_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        r = reward[trial]
        
        # Calculate Prediction Errors
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Determine Learning Rate based on error sign
        # If delta is negative (disappointment), boost LR by OCI factor
        if delta_stage2 < 0:
            current_lr_2 = lr_base + (oci_neg_boost * oci_score * lr_base)
            # Cap LR at 1.0
            if current_lr_2 > 1.0: current_lr_2 = 1.0
        else:
            current_lr_2 = lr_base

        # Apply similar logic to stage 1 (though delta1 is driven by stage 2 values)
        if delta_stage1 < 0:
            current_lr_1 = lr_base + (oci_neg_boost * oci_score * lr_base)
            if current_lr_1 > 1.0: current_lr_1 = 1.0
        else:
            current_lr_1 = lr_base

        q_stage1_mf[a1] += current_lr_1 * delta_stage1
        q_stage2_mf[s_idx, a2] += current_lr_2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
