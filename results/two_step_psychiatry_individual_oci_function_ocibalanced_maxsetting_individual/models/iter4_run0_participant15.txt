Here are the 3 proposed cognitive models.

### Model 1: OCI-Modulated Transition Belief
This model hypothesizes that high OCI scores correlate with an altered belief about the environmental structure (the transition probabilities). OCI is often associated with intolerance of uncertainty; this model tests if higher OCI leads participants to assume the "common" transition is more deterministic (closer to 1.0) or more chaotic (closer to 0.5) than reality, affecting their Model-Based value calculation.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Transition Belief.
    OCI modulates the subjective transition probability used in Model-Based updates.
    High OCI may lead to an overestimation of determinism (or randomness).
    
    Parameters:
    lr: [0, 1] Learning rate for Q-value updates.
    beta: [0, 10] Inverse temperature (softmax sensitivity).
    w: [0, 1] Weighting between Model-Based and Model-Free values.
    stick: [0, 5] Stickiness to the previously chosen spaceship.
    trans_mod_oci: [0, 1] Modulation of belief in transition probability by OCI.
                   (Subjective P = 0.7 + trans_mod_oci * OCI * direction).
                   Here modeled as increasing determinism.
    """
    lr, beta, w, stick, trans_mod_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Subjective transition probability: Base 0.7 modified by OCI
    # We allow OCI to push belief towards 1.0 (determinism)
    # The term (0.3 * trans_mod_oci * oci_score) adds to 0.7, maxing at 1.0
    p_common = 0.7 + (0.3 * trans_mod_oci * oci_score)
    if p_common > 0.99: p_common = 0.99
    
    # Subjective transition matrix used for MB calculation
    subj_trans_matrix = np.array([[p_common, 1 - p_common], 
                                  [1 - p_common, p_common]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: Uses subjective transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = subj_trans_matrix @ max_q_stage2
        
        # Net Q-value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stick
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF Update (TD(0))
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Reward Dampening
This model proposes that OCI affects the perceived magnitude of rewards (anhedonia or dampening). By scaling the effective reward down based on OCI, the learned Q-values become smaller in magnitude. Since the "stickiness" parameter adds a fixed bonus to the choice, a dampening of rewards effectively increases the relative influence of habits (stickiness) over goal-directed values, without needing a separate stickiness-OCI parameter.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Reward Dampening.
    High OCI reduces the effective reward magnitude. This shrinks Q-values
    relative to the fixed 'stickiness' bonus, making the agent relatively 
    more sticky/habitual and less sensitive to value differences.

    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    stick: [0, 5] Base stickiness bonus.
    rew_damp_oci: [0, 1] Factor by which OCI dampens the reward signal.
                   r_eff = r / (1 + rew_damp_oci * OCI * 5).
    """
    lr, beta, w, stick, rew_damp_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective reward scaler
    # If OCI is high and param is high, scaler is < 1.0
    # The factor 5 ensures the parameter has strong potential impact within [0,1]
    reward_scaler = 1.0 / (1.0 + (rew_damp_oci * oci_score * 5.0))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        
        # Scale the reward
        r_eff = reward[trial] * reward_scaler

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stick

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates using Effective Reward ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

        delta_stage2 = r_eff - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 3: OCI-Boosted Stage 2 Learning
This model posits that while Stage 1 (spaceship choice) relies on a baseline learning rate, OCI specifically boosts the learning rate for Stage 2 (aliens). This reflects a cognitive style where the participant is hyper-reactive to immediate, concrete outcomes (the aliens), leading to volatile values in the second stage that propagate back to the first stage.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Boosted Stage 2 Learning.
    OCI increases the learning rate specifically for the second stage (aliens),
    making the agent hyper-reactive to the most recent concrete outcomes,
    while Stage 1 learning remains at a baseline.

    Parameters:
    lr_base: [0, 1] Baseline learning rate (applied to Stage 1).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    stick: [0, 5] Stickiness.
    lr2_boost_oci: [0, 1] OCI-dependent boost to Stage 2 learning rate.
                   lr2 = lr_base + (lr2_boost_oci * OCI).
    """
    lr_base, beta, w, stick, lr2_boost_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Stage 1 uses base LR
    lr1 = lr_base
    
    # Stage 2 uses boosted LR
    lr2 = lr_base + (lr2_boost_oci * oci_score)
    if lr2 > 1.0: lr2 = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stick

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Update Stage 1 with base LR
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr1 * delta_stage1

        # Update Stage 2 with OCI-boosted LR
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr2 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```
