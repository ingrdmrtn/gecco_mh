def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Stage 2 Stickiness.
    
    Hypothesis: OCI symptoms drive perseveration specifically at the level of the 
    proximal reward source (the alien/Stage 2 choice). This stickiness in Stage 2 
    propagates to Stage 1 behavior via the Model-Based system (planning to reach 
    the "sticky" alien).
    
    Parameters:
    learning_rate: [0, 1] - Value update rate for both stages.
    beta: [0, 10] - Inverse temperature for softmax choice policy.
    w: [0, 1] - Weighting between Model-Based (1) and Model-Free (0) control.
    stickiness_2_scale: [0, 10] - Coefficient scaling how much OCI increases the 
                                  tendency to repeat the last chosen alien in Stage 2.
    """
    learning_rate, beta, w, stickiness_2_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Track last action taken in each state (planet) for Stage 2 stickiness
    last_action_2_per_state = [-1, -1] 

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid MB/MF value
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        # Base Q-values for the current state (planet)
        q_net_2 = q_stage2_mf[state_idx].copy()
        
        # Apply Stage 2 Stickiness modulated by OCI
        # If we have visited this planet before, boost the value of the previously chosen alien
        last_a2 = last_action_2_per_state[state_idx]
        if last_a2 != -1:
            q_net_2[last_a2] += (oci_score * stickiness_2_scale)

        exp_q2 = np.exp(beta * q_net_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Update history
        last_action_2_per_state[state_idx] = int(action_2[trial])

        # --- Value Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] = q_stage2_mf[state_idx, a2] + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Loss Aversion (Subjective Punishment).
    
    Hypothesis: Participants with higher OCI scores may exhibit a "perfectionist" 
    or anxiety-driven response to failure. They perceive the absence of a reward (0 coins) 
    not just as a neutral null outcome, but as an active negative outcome (punishment). 
    This creates a distorted reward signal.
    
    Parameters:
    learning_rate: [0, 1] - Value update rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weight.
    punishment_sensitivity: [0, 5] - Scales the magnitude of the subjective punishment 
                                     experienced when reward is 0, proportional to OCI.
    """
    learning_rate, beta, w, punishment_sensitivity = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Value Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        raw_r = reward[trial]
        
        # Transform Reward based on OCI
        # If reward is 1, perceived reward is 1.
        # If reward is 0, perceived reward is negative (punishment).
        effective_r = raw_r
        if raw_r == 0:
            effective_r = -1.0 * oci_score * punishment_sensitivity

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        delta_stage2 = effective_r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] = q_stage2_mf[state_idx, a2] + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-decoupled Stage 2 Precision.
    
    Hypothesis: OCI affects decision noise differently in the abstract planning stage (Stage 1) 
    versus the concrete harvesting stage (Stage 2). High OCI leads to hyper-precision (or rigidity) 
    specifically in the second stage, modeled by a separate, boosted inverse temperature (beta)
    for Stage 2.
    
    Parameters:
    learning_rate: [0, 1] - Value update rate.
    beta_1: [0, 10] - Inverse temperature for Stage 1 choices.
    w: [0, 1] - MB/MF weight.
    beta_2_boost: [0, 10] - Additional precision factor for Stage 2, scaled by OCI.
                            beta_2 = beta_1 * (1 + oci * beta_2_boost).
    """
    learning_rate, beta_1, w, beta_2_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Calculate Stage 2 Beta based on OCI modulation
    beta_2 = beta_1 * (1.0 + oci_score * beta_2_boost)

    for trial in range(n_trials):
        # --- Stage 1 Policy (Uses beta_1) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy (Uses boosted beta_2) ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Value Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] = q_stage2_mf[state_idx, a2] + learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
