### Model 1: OCI-Modulated Model-Based/Model-Free Hybrid
This model hypothesizes that OCI symptoms influence the balance between Model-Based (planning) and Model-Free (habitual) control. Specifically, higher OCI scores might lead to an over-reliance on habitual control (lower `w`) or rigid planning. This model tests if the mixing weight `w` is a linear function of the OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated MB/MF Hybrid.
    
    This model assumes a hybrid learner where the balance (w) between Model-Based (MB)
    and Model-Free (MF) control is determined by the participant's OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w_intercept: [0, 1] Base mixing weight (0=Pure MF, 1=Pure MB) at OCI=0.
    - w_slope: [-1, 1] How OCI score changes the mixing weight w.
    """
    learning_rate, beta, w_intercept, w_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective mixing weight w based on OCI
    # We clip w to be between 0 and 1 to ensure validity
    w = w_intercept + (w_slope * oci_score)
    w = np.clip(w, 0.0, 1.0)
    
    # Transition matrix (fixed structure of the task)
    # Row 0: Spaceship 0 (A) -> Planet 0 (X) w/ 0.7, Planet 1 (Y) w/ 0.3
    # Row 1: Spaceship 1 (U) -> Planet 0 (X) w/ 0.3, Planet 1 (Y) w/ 0.7
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)     # MF values for Stage 1
    q_stage2_mf = np.zeros((2, 2)) # MF values for Stage 2 (Planets)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        # Skip invalid trials (missing data)
        if a1 == -1:
            continue

        # --- Stage 1 Policy ---
        # Model-Based Value: V_MB(s1) = T * max(Q_MF(s2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 Update (TD Error)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD(0) for MF)
        # Note: In standard hybrid models, MF Q1 updates towards Q2 or Reward.
        # Here we use a simple TD(1)-like update where Q1 updates from the reward directly
        # or the value of the second stage choice.
        # Common implementation: Update Q1_MF based on the reward received at the end.
        delta_stage1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    valid_trials = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Learning Rate Asymmetry
This model suggests that OCI relates to how people process positive versus negative feedback. It proposes that while there is a baseline learning rate for positive outcomes, the learning rate for negative outcomes (losses/omissions) is scaled by the OCI score. This could reflect a heightened sensitivity to failure or "punishment" often seen in anxiety-related traits.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Dependent Learning Rate Asymmetry.
    
    Assumes a pure Model-Free learner, but splits learning rates for positive (reward=1)
    and negative (reward=0) outcomes. The negative learning rate is modulated by OCI.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for rewarded trials (R=1).
    - lr_neg_base: [0, 1] Base learning rate for unrewarded trials (R=0).
    - lr_neg_oci_mod: [-1, 1] Modulation of negative learning rate by OCI.
    - beta: [0, 10] Inverse temperature.
    - lambda_eligibility: [0, 1] Eligibility trace parameter (credit assignment to Stage 1).
    """
    lr_pos, lr_neg_base, lr_neg_oci_mod, beta, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective negative learning rate
    lr_neg = lr_neg_base + (lr_neg_oci_mod * oci_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        if a1 == -1:
            continue

        # Select learning rate based on outcome
        if r == 1:
            alpha = lr_pos
        else:
            alpha = lr_neg

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # SARSA / TD(lambda) style update
        
        # Stage 2 Prediction Error
        delta_2 = r - q_stage2[s2, a2]
        
        # Update Stage 2 Q-value
        q_stage2[s2, a2] += alpha * delta_2
        
        # Update Stage 1 Q-value using eligibility trace
        # The update is driven by the stage 2 prediction error
        q_stage1[a1] += alpha * lambda_eligibility * delta_2
        
        # Note: We could also include a Stage 1 prediction error (Q2 - Q1), 
        # but in many simple implementations, direct update from R via lambda is sufficient
        # to capture the main effect. Here we stick to a simple trace update.

    eps = 1e-10
    valid_trials = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: OCI-Scaled Uncertainty/Exploration (Beta Modulation)
This model posits that OCI affects the exploration-exploitation trade-off. Specifically, the inverse temperature parameter `beta` (which controls choice determinism) is modeled as a function of the OCI score. A high OCI might lead to very deterministic behavior (high beta) due to rigidity, or very random behavior (low beta) due to indecision.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Scaled Uncertainty (Beta Modulation).
    
    This model assumes a Model-Based learner where the inverse temperature (beta)
    is modulated by the OCI score. This tests if OCI affects decision noise/rigidity.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_base: [0, 10] Base inverse temperature.
    - beta_oci_slope: [-5, 5] How OCI scales beta (positive = more rigid, negative = more noisy).
    - w: [0, 1] Fixed mixing weight between MB and MF (0=MF, 1=MB).
    """
    learning_rate, beta_base, beta_oci_slope, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective beta
    beta_eff = beta_base + (beta_oci_slope * oci_score)
    # Beta must be non-negative. We use a softplus-like or clipping to ensure > 0.
    beta_eff = np.maximum(beta_eff, 0.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        if a1 == -1:
            continue

        # --- Stage 1 Policy ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-modulated Beta
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_eff * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Update Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF
        delta_stage1 = r - q_stage1_mf[a1] 
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    valid_trials = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
