Here are three cognitive models that explain the participant's behavior using their OCI score to modulate specific decision-making mechanisms.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Distorted Transition Belief.
    
    Hypothesis: High OCI participants may perceive the transition structure of the task 
    as more deterministic than it actually is (reflecting rigid or black-and-white thinking).
    This model modulates the Model-Based transition probability 'p' based on the OCI score.
    A higher distortion leads the agent to believe the common transition is nearly certain.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Model-Free Q-values.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w: [0, 1] Weighting parameter between Model-Based and Model-Free values.
    - distortion: [0, 1] Scaling factor for OCI that pushes the perceived transition probability from 0.7 towards 1.0.
    """
    learning_rate, beta, w, distortion = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate subjective transition probability
    # Base probability is 0.7. The distortion pushes this towards 1.0 based on OCI.
    p_trans = 0.7 + (0.3 * oci_score * distortion)
    # Clip to ensure probability validity (max 0.999 to avoid log(0) issues conceptually, though not used in log here)
    p_trans = np.clip(p_trans, 0.7, 0.999)
    
    # Subjective transition matrix used for Model-Based valuation
    transition_matrix = np.array([[p_trans, 1-p_trans], [1-p_trans, p_trans]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value Calculation using distorted matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # TD Update Stage 1 (Model-Free)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Update Stage 2 (Model-Free)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Reinforced Stickiness.
    
    Hypothesis: Compulsivity (OCI) is associated with a tendency to repetitively choose 
    actions that were previously rewarded ("Win-Stay" amplification), rather than 
    general motor perseveration. This model adds a stickiness bonus to the previous 
    Stage 1 choice *only if* it resulted in a reward.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - reinf_stick: [0, 5] Magnitude of stickiness bonus scaled by OCI, conditional on previous reward.
    """
    learning_rate, beta, w, reinf_stick = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # The magnitude of the bonus is determined by the parameter and the OCI score
    stickiness_mag = reinf_stick * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0.0
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness bonus only if the previous trial yielded a reward (Win-Stay)
        if last_action_1 != -1 and last_reward == 1.0:
            q_net[last_action_1] += stickiness_mag
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Choice Trace (Habit Strength).
    
    Hypothesis: High OCI participants rely on a 'choice trace' (an exponentially decaying 
    frequency of past choices) rather than just the immediate previous choice. 
    This represents a stronger, more persistent habit formation mechanism.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - trace_decay: [0, 1] Persistence of the choice trace (higher = longer memory/slower decay).
    - trace_weight: [0, 5] Influence of the choice trace on decision making, scaled by OCI.
    """
    learning_rate, beta, w, trace_decay, trace_weight = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # The weight of the habit trace is modulated by OCI
    weight_eff = trace_weight * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize choice trace for the two spaceships
    choice_trace = np.zeros(2)
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add the accumulated choice trace habit to the Q-values
        q_net += weight_eff * choice_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Choice Trace
        # Decay the trace for all actions
        choice_trace *= trace_decay
        # Increment the trace for the chosen action
        choice_trace[a1] += 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
