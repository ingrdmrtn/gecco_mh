Here are three new cognitive models, expressed as Python functions, that hypothesize different mechanisms for how OCI scores modulate decision-making in the two-step task.

### Model 1: Subjective Transition Model
This model hypothesizes that high OCI scores are associated with a distorted internal model of the environment's transition structure (e.g., perceiving the world as more deterministic or more chaotic than it is), affecting the Model-Based value calculation.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Subjective Transition Model.
    
    Hypothesis: OCI scores distort the perceived transition probabilities in the 
    Model-Based system. High OCI might lead to a distorted belief about the 
    reliability of the spaceship-planet transitions (e.g., rigid determinism 
    or uncertainty), affecting the Model-Based value calculation.
    
    Parameters:
    lr: [0, 1] Learning rate for Model-Free values.
    beta: [0, 10] Inverse temperature (softmax sensitivity).
    w: [0, 1] Weight for Model-Based values (0 = Pure MF, 1 = Pure MB).
    pers: [0, 1] Perseveration bonus weight (mapped to [0, 5]).
    trans_base: [0, 1] Base subjective probability of the common transition.
    trans_oci_slope: [0, 1] Effect of OCI on transition probability. 
                     Mapped to [-0.5, 0.5].
    """
    lr, beta, w, pers_param, trans_base, trans_oci_slope_param = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Map parameters
    pers = pers_param * 5.0
    trans_slope = trans_oci_slope_param - 0.5
    
    # Calculate subjective transition probability
    # True matrix is roughly 0.7 common. 
    # We model the subjective 'p_common' 
    p_common = trans_base + trans_slope * current_oci
    p_common = np.clip(p_common, 0.0, 1.0)
    
    # Transition matrix: [[p, 1-p], [1-p, p]]
    # Assumes Action 0 -> State 0 is common
    transition_matrix = np.array([[p_common, 1.0 - p_common], 
                                  [1.0 - p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # [State, Alien]
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration
        if prev_action_1 != -1:
            q_net[prev_action_1] += pers
            
        exp_q1 = np.exp(beta * q_net)
        # Safety for overflow
        if np.any(np.isinf(exp_q1)):
            max_val = np.max(beta * q_net)
            exp_q1 = np.exp(beta * q_net - max_val)
            
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Stage 2 Policy
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        if np.any(np.isinf(exp_q2)):
            max_val = np.max(beta * q_stage2_mf[state_idx])
            exp_q2 = np.exp(beta * q_stage2_mf[state_idx] - max_val)
            
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        # Stage 1 MF update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        
        prev_action_1 = int(action_1[trial])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Stage 1 Habit Learning Modulation Model
This model hypothesizes that OCI scores specifically affect the formation of habits (Stage 1 Model-Free values) by modulating the learning rate for the first decision, distinct from how the participant learns about the aliens (Stage 2).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage 1 Habit Learning Modulation Model.
    
    Hypothesis: OCI scores relate to the strength of habit formation. This model 
    allows OCI to specifically modulate the learning rate of the Stage 1 Model-Free 
    values (habits), while Stage 2 learning remains independent.
    
    Parameters:
    lr_s1_base: [0, 1] Base learning rate for Stage 1 (Habit).
    lr_s1_oci_slope: [0, 1] Effect of OCI on Stage 1 LR. Mapped to [-0.5, 0.5].
    lr_s2: [0, 1] Learning rate for Stage 2 (Goal values).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based values.
    pers: [0, 1] Choice perseveration weight (mapped to [0, 5]).
    """
    lr_s1_base, lr_s1_oci_slope_param, lr_s2, beta, w, pers_param = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Map parameters
    pers = pers_param * 5.0
    lr_s1_slope = lr_s1_oci_slope_param - 0.5
    lr_s1 = lr_s1_base + lr_s1_slope * current_oci
    lr_s1 = np.clip(lr_s1, 0.0, 1.0)
    
    # Fixed transition matrix for MB
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += pers
            
        exp_q1 = np.exp(beta * q_net)
        if np.any(np.isinf(exp_q1)):
            max_val = np.max(beta * q_net)
            exp_q1 = np.exp(beta * q_net - max_val)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Stage 2
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        if np.any(np.isinf(exp_q2)):
            max_val = np.max(beta * q_stage2_mf[state_idx])
            exp_q2 = np.exp(beta * q_stage2_mf[state_idx] - max_val)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        # Stage 1: Modulated LR
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr_s1 * delta_stage1
        
        # Stage 2: Independent LR
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr_s2 * delta_stage2
        
        prev_action_1 = int(action_1[trial])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Counterfactual Update Model
This model hypothesizes that OCI predicts the tendency to engage in counterfactual updating (regret or "checking"). It updates the value of the *unchosen* alien in Stage 2 based on a fictitious inverse outcome, simulating a belief that if one alien pays out, the other likely doesn't (or vice versa).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Counterfactual Update Model.
    
    Hypothesis: OCI scores predict the tendency to engage in counterfactual 
    updating (regret/checking). This model updates the unchosen alien's value 
    in Stage 2 assuming an anti-correlated reward structure (if chosen=0, unchosen=1).
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based values.
    pers: [0, 1] Choice perseveration weight (mapped to [0, 5]).
    cf_base: [0, 1] Base weight for counterfactual update.
    cf_oci_slope: [0, 1] Effect of OCI on counterfactual weight. Mapped to [-0.5, 0.5].
    """
    lr, beta, w, pers_param, cf_base, cf_oci_slope_param = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Map parameters
    pers = pers_param * 5.0
    cf_slope = cf_oci_slope_param - 0.5
    cf_weight = cf_base + cf_slope * current_oci
    cf_weight = np.clip(cf_weight, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[prev_action_1] += pers
            
        exp_q1 = np.exp(beta * q_net)
        if np.any(np.isinf(exp_q1)):
            max_val = np.max(beta * q_net)
            exp_q1 = np.exp(beta * q_net - max_val)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Stage 2
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        if np.any(np.isinf(exp_q2)):
            max_val = np.max(beta * q_stage2_mf[state_idx])
            exp_q2 = np.exp(beta * q_stage2_mf[state_idx] - max_val)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Updates
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 MF - Chosen
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Stage 2 MF - Unchosen (Counterfactual)
        unchosen_a2 = 1 - a2
        # Fictitious reward: 1 - r (assuming anti-correlation)
        fictitious_r = 1.0 - r
        delta_cf = fictitious_r - q_stage2_mf[state_idx, unchosen_a2]
        q_stage2_mf[state_idx, unchosen_a2] += lr * cf_weight * delta_cf
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
