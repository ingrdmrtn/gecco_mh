Here are three cognitive models that incorporate the participant's OCI score to explain individual variability in the two-step task.

### Model 1: Dynamic Habit Decay
This model hypothesizes that OCI scores modulate the **persistence of habits** over time. Instead of a constant "stickiness" bonus, this model maintains a dynamic "habit trace" that accumulates when a choice is made and decays on every trial. The OCI score determines the **decay rate** of this trace. High OCI scores may lead to slower decay (higher `decay_rate`), resulting in stronger, longer-lasting perseveration (compulsive repetition), consistent with the participant's tendency to stick to choices for long blocks.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dynamic Habit Decay Model.
    
    The 'stickiness' is modeled as a dynamic habit trace that decays over time.
    The OCI score modulates the decay rate of this habit trace.
    High OCI -> Slower decay -> Habits persist longer (more perseveration).
    
    Parameters:
    - learning_rate: [0,1] Rate of value updating.
    - beta: [0,10] Inverse temperature (softness of choice) for both stages.
    - w: [0,1] Weighting between Model-Based (1) and Model-Free (0).
    - habit_weight: [0,5] Scaling factor for the influence of the habit trace on choice.
    - decay_base: [0,1] Baseline decay rate of the habit trace.
    - decay_oci_sens: [-1,1] Sensitivity of decay rate to OCI score.
      (decay = decay_base + decay_oci_sens * oci).
    """
    learning_rate, beta, w, habit_weight, decay_base, decay_oci_sens = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate OCI-modulated decay rate, clipped to valid range [0, 1]
    decay_rate = np.clip(decay_base + decay_oci_sens * oci_score, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Dynamic habit trace for the two spaceships
    habit_trace = np.zeros(2)

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value: Weighted sum of MB, MF, and Habit Trace
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + habit_weight * habit_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update Habit Trace
        # 1. Strengthen chosen action
        habit_trace[a1] += 1.0
        # 2. Decay all traces
        habit_trace *= decay_rate
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        # Stage 1 MF update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF update (TD(1) equivalent)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    # Mask out missing trials in log loss calculation
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: Rare Transition "Surprise" Sensitivity
This model hypothesizes that individuals with varying OCI scores react differently to "rare" or "surprising" transitions (e.g., Spaceship A going to Planet Y). The model allows the **learning rate** to change specifically on trials where a rare transition occurred. High OCI might be associated with over-reacting to noise (rare events), leading to a higher learning rate on these trials (interpreting them as a change in structure rather than noise).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Rare Transition Learning Modulation.
    
    The learning rate is modulated by OCI specifically on trials where a 
    rare transition occurs (Common: 0->0, 1->1; Rare: 0->1, 1->0).
    
    Parameters:
    - lr_base: [0,1] Base learning rate for common transitions.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] MB/MF weight.
    - stickiness: [0,5] Choice stickiness (constant bonus).
    - rare_mult_base: [0,5] Base multiplier for learning rate on rare trials.
    - rare_mult_oci: [-5,5] OCI-dependent adjustment to the rare multiplier.
      (lr_rare = lr_base * (rare_mult_base + rare_mult_oci * oci)).
    """
    lr_base, beta, w, stickiness, rare_mult_base, rare_mult_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        # Determine if transition was rare
        # Common: 0->0, 1->1. Rare: 0->1, 1->0.
        is_rare = (a1 == 0 and state_idx == 1) or (a1 == 1 and state_idx == 0)
        
        # Calculate effective learning rate
        current_lr = lr_base
        if is_rare:
            multiplier = rare_mult_base + rare_mult_oci * oci_score
            # Ensure multiplier is non-negative
            multiplier = np.maximum(multiplier, 0.0)
            current_lr = lr_base * multiplier
        
        # Clip learning rate to [0, 1]
        current_lr = np.clip(current_lr, 0.0, 1.0)

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += current_lr * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: Stage 2 Precision Modulation
This model posits that OCI affects the **precision (inverse temperature)** of the second-stage choices relative to the first-stage. High OCI might lead to rigid, deterministic selection of the "best" alien (Stage 2) while potentially leaving Stage 1 exploration intact, or vice versa. The model uses a base beta for Stage 1 and an OCI-modulated beta for Stage 2.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage 2 Precision Modulation Model.
    
    OCI modulates the relationship between Stage 1 and Stage 2 inverse temperatures.
    This captures potential differences in 'rigidity' between the abstract 
    spaceship choice and the concrete alien choice.
    
    Parameters:
    - learning_rate: [0,1] Learning rate.
    - beta_1: [0,10] Inverse temperature for Stage 1.
    - w: [0,1] MB/MF weight.
    - stickiness: [0,5] Choice stickiness.
    - ratio_base: [0,5] Base ratio of Beta_2 / Beta_1.
    - ratio_oci: [-5,5] OCI modulation of the ratio.
      (beta_2 = beta_1 * (ratio_base + ratio_oci * oci)).
    """
    learning_rate, beta_1, w, stickiness, ratio_base, ratio_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Beta 2 based on OCI
    ratio = np.maximum(ratio_base + ratio_oci * oci_score, 0.0)
    beta_2 = beta_1 * ratio
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy (Uses beta_1) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy (Uses beta_2) ---
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
