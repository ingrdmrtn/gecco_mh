Here are 3 new cognitive models that incorporate the OCI score into the decision-making process, specifically exploring how obsessive-compulsive traits might influence learning rates, perseveration, and the balance between model-based and model-free control.

### Model 1: OCI-Modulated Learning Rate Asymmetry
This model hypothesizes that individuals with higher OCI scores might learn differently from positive versus negative prediction errors. Specifically, high OCI might be associated with a "hyper-learning" from negative outcomes (avoidance of failure) or positive outcomes (reassurance seeking). Here, we scale the learning rate for negative prediction errors based on the OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Asymmetric Learning Rates modulated by OCI.
    
    Hypothesis: OCI score modulates the sensitivity to negative prediction errors.
    Higher OCI might lead to stronger learning from punishment/omission of reward (negative PE),
    reflecting a fear of failure or enhanced error monitoring.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors.
    - alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting parameter (0 = pure MF, 1 = pure MB).
    """
    alpha_pos, alpha_neg_base, beta, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Modulate negative learning rate by OCI. 
    # If OCI is high, alpha_neg increases, making the agent more sensitive to lack of reward.
    # Capped at 1.0.
    alpha_neg = min(1.0, alpha_neg_base * (1.0 + current_oci))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])
        if s_idx == -1: continue

        # --- Stage 2 Policy ---
        # Softmax Choice 2 (purely MF at stage 2)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]

        r = reward[trial]

        # --- Updates ---
        # Stage 2 Update
        delta2 = r - q_stage2_mf[s_idx, a2]
        lr2 = alpha_pos if delta2 > 0 else alpha_neg
        q_stage2_mf[s_idx, a2] += lr2 * delta2

        # Stage 1 Update (TD-learning)
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        lr1 = alpha_pos if delta1 > 0 else alpha_neg
        q_stage1_mf[a1] += lr1 * delta1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Perseveration (Sticky Choice)
This model introduces a perseveration parameter (stickiness) that is directly modulated by the OCI score. The hypothesis is that obsessive-compulsive traits are linked to "stuck" behaviors or repetition compulsion. High OCI leads to higher perseveration (tendency to repeat the previous Stage 1 choice regardless of reward).

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Perseveration (Stickiness) modulated by OCI.
    
    Hypothesis: Higher OCI scores are associated with higher perseveration (stickiness).
    The agent is more likely to repeat the previous stage 1 action, regardless of the outcome.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-based weight.
    - stickiness_base: [0, 5] Base parameter for choice perseveration.
    """
    learning_rate, beta, w, stickiness_base = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Stickiness increases with OCI
    effective_stickiness = stickiness_base * (1.0 + current_oci)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous action for perseveration
    prev_a1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the Q-value of the previously chosen action
        q_with_stickiness = q_integrated.copy()
        if prev_a1 != -1:
            q_with_stickiness[prev_a1] += effective_stickiness
            
        exp_q1 = np.exp(beta * q_with_stickiness)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update previous action
        prev_a1 = a1
        
        s_idx = int(state[trial])
        if s_idx == -1: continue

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]

        r = reward[trial]

        # --- Updates ---
        delta2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta2

        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Model-Based/Model-Free Balance (W)
This model posits that the balance between Goal-Directed (Model-Based) and Habitual (Model-Free) control is shifted by OCI symptoms. Compulsivity is often conceptualized as an over-reliance on habits (Model-Free) or a deficit in Goal-Directed control. Here, we model `w` (the weight of Model-Based control) as a function of OCI.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: MB/MF Balance (w) modulated by OCI.
    
    Hypothesis: High OCI is associated with a shift towards habitual (Model-Free) control.
    Therefore, the weight 'w' (Model-Based contribution) decreases as OCI increases.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w_base: [0, 1] Base weight for Model-Based control.
    - lambda_eligibility: [0, 1] Eligibility trace for stage 1 update.
    """
    learning_rate, beta, w_base, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Modulate w: Higher OCI reduces model-based weight (more habitual).
    # We dampen w_base by a factor related to OCI.
    # If OCI is 0, w = w_base. If OCI is 1, w is halved (arbitrary scaling for effect).
    w = w_base / (1.0 + current_oci)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])
        if s_idx == -1: continue

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 == -1: continue
        p_choice_2[trial] = probs_2[a2]

        r = reward[trial]

        # --- Updates ---
        # Stage 2 Update
        delta2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta2

        # Stage 1 Update with eligibility trace (lambda)
        # This connects the final reward back to the first stage choice directly
        delta_lambda = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * lambda_eligibility * delta_lambda
        
        # Standard Stage 1 MF update (TD)
        delta1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * (1 - lambda_eligibility) * delta1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (state != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
