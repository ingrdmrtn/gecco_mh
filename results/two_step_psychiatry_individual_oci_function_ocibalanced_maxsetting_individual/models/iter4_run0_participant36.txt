Here are three new cognitive models that explain the participant's behavior, incorporating their OCI score to modulate specific cognitive mechanisms.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: Hybrid Model with OCI-Modulated Memory Decay.
    Hypothesis: High OCI participants may exhibit 'hoarding' of value associations,
    manifesting as reduced forgetting (decay) of unchosen options' values.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for chosen options.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    - decay_base: [0, 1] Base decay rate for unchosen MF values (0=no decay, 1=instant forget).
    - oci_decay_mod: [-1, 1] Modulation of decay by OCI score.
    """
    learning_rate, beta, w, decay_base, oci_decay_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective decay
    # High OCI might reduce decay (negative mod), implying value hoarding.
    decay = np.clip(decay_base + (oci_score * oci_decay_mod), 0.0, 1.0)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize values
    q_stage1_mf = np.zeros(2) # Model-Free Stage 1
    q_stage2_mf = np.zeros((2, 2)) # Stage 2 values (used by both MB and MF)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # Stage 2 Update (TD)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 MF Update
        # Using the value of the chosen stage 2 action as the target
        target_stage1 = q_stage2_mf[state_idx, a2] 
        delta_stage1 = target_stage1 - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Decay for unchosen Stage 1 option
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1 - decay)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: Pure Model-Free with OCI-Modulated Stage 1 Learning Rate.
    Hypothesis: OCI affects the flexibility of the high-level policy (Stage 1)
    differently from the low-level valuation (Stage 2).
    
    Parameters:
    - lr_stage2: [0, 1] Learning rate for Stage 2 (aliens).
    - beta: [0, 10] Inverse temperature.
    - stickiness: [0, 5] Choice perseveration bonus.
    - lr_stage1_base: [0, 1] Base learning rate for Stage 1 (spaceships).
    - oci_lr1_mod: [-1, 1] Modulation of Stage 1 learning rate by OCI.
    """
    lr_stage2, beta, stickiness, lr_stage1_base, oci_lr1_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Stage 1 Learning Rate
    lr_stage1 = np.clip(lr_stage1_base + (oci_score * oci_lr1_mod), 0.0, 1.0)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice (Pure MF + Stickiness) ---
        q_net = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        last_action_1 = action_1[trial]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a1 = action_1[trial]
        a2 = action_2[trial]
        r = reward[trial]
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2
        
        # Stage 1 Update
        # Using the Stage 2 value as the target (TD chaining)
        target_stage1 = q_stage2_mf[state_idx, a2]
        delta_stage1 = target_stage1 - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: Model-Based with OCI-Modulated Choice Trace Decay.
    Hypothesis: OCI relates to the persistence of behavioral sets (rigidity).
    This is modeled as a 'choice trace' that decays over time. High OCI leads to
    slower decay (longer memory of past choices), causing stronger streaks.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Stage 2 values.
    - beta: [0, 10] Inverse temperature.
    - stick_weight: [0, 5] Weight of the choice trace (generalized stickiness).
    - trace_persistence_base: [0, 1] Base persistence (1-decay) of the choice trace.
    - oci_persistence_mod: [-1, 1] Modulation of trace persistence by OCI.
    """
    learning_rate, beta, stick_weight, trace_persistence_base, oci_persistence_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate trace persistence (how much of the previous trace is kept)
    # 0.0 = only last choice matters (standard stickiness)
    # 0.9 = long history matters
    trace_persistence = np.clip(trace_persistence_base + (oci_score * oci_persistence_mod), 0.0, 0.99)
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for Stage 1 options (0 and 1)
    choice_trace = np.zeros(2)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 Choice (MB + Trace Stickiness) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add stickiness based on trace
        q_net = q_stage1_mb + (stick_weight * choice_trace)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Update Trace
        # Decay existing trace
        choice_trace *= trace_persistence
        # Add current choice
        choice_trace[action_1[trial]] += 1.0
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        a2 = action_2[trial]
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
