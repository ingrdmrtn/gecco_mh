Here are three cognitive models formulated as Python functions. Each model incorporates the participant's OCI score to explain their decision-making behavior, specifically focusing on the observed high stickiness and persistence.

### Model 1: Hybrid Model with OCI-Modulated Stickiness
This model tests the hypothesis that high obsessive-compulsive symptoms directly drive behavioral repetition (perseveration). The stickiness parameter, which increases the probability of repeating the previous choice, is modeled as a linear function of the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-modulated Stickiness.
    Hypothesis: High OCI increases the tendency to repeat the previous Stage 1 choice (perseveration).
    The stickiness parameter is a linear function of the OCI score.
    
    stickiness = stick_base + stick_oci_sens * oci
    
    Parameters:
    - learning_rate: [0,1] Learning rate for Q-value updates.
    - beta: [0,10] Inverse temperature for softmax (choice consistency).
    - lam: [0,1] Eligibility trace decay parameter.
    - w: [0,1] Mixing weight (0=MF, 1=MB).
    - stick_base: [0,5] Baseline stickiness independent of OCI.
    - stick_oci_sens: [0,5] Sensitivity of stickiness to OCI score.
    """
    learning_rate, beta, lam, w, stick_base, stick_oci_sens = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness based on OCI
    stickiness = stick_base + stick_oci_sens * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_mf_1 = np.zeros(2) # Stage 1 MF values
    q_mf_2 = np.zeros((2, 2)) # Stage 2 MF values (State x Action)
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Skip invalid trials
        if a1 < 0 or s_idx < 0 or a2 < 0:
            continue
            
        # --- Stage 1 Choice ---
        # Model-Based Value
        max_q_stage2 = np.max(q_mf_2, axis=1)
        q_mb_1 = transition_matrix @ max_q_stage2
        
        # Net Value mixing MB and MF
        q_net_1 = w * q_mb_1 + (1 - w) * q_mf_1
        
        # Add stickiness to the previously chosen action
        if prev_action_1 != -1:
            q_net_1[prev_action_1] += stickiness
            
        # Softmax policy for Stage 1
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Choice ---
        # Standard MF for stage 2
        exp_q2 = np.exp(beta * q_mf_2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # Prediction errors
        delta_1 = q_mf_2[s_idx, a2] - q_mf_1[a1]
        delta_2 = r - q_mf_2[s_idx, a2]
        
        # Update Stage 1 MF
        q_mf_1[a1] += learning_rate * delta_1
        # Eligibility trace update for Stage 1 from Stage 2 error
        q_mf_1[a1] += learning_rate * lam * delta_2
        
        # Update Stage 2 MF
        q_mf_2[s_idx, a2] += learning_rate * delta_2
        
        prev_action_1 = a1
        
    return log_loss
```

### Model 2: Hybrid Model with OCI-Modulated Negative Learning Rate
This model addresses the participant's persistence despite lack of reward. It hypothesizes that high OCI leads to a specific insensitivity to negative prediction errors (punishments/omissions). The learning rate for negative errors is dampened by the OCI score, causing the agent to maintain high values for chosen actions even when they yield 0 coins.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-modulated Negative Learning Rate Dampening.
    Hypothesis: High OCI individuals have reduced sensitivity to negative prediction errors 
    (worse-than-expected outcomes), leading to persistence in unrewarded habits.
    
    alpha_pos = learning_rate
    alpha_neg = learning_rate * (1 - dampening * oci)
    
    Parameters:
    - learning_rate: [0,1] Base learning rate (for positive prediction errors).
    - beta: [0,10] Inverse temperature.
    - lam: [0,1] Eligibility trace decay.
    - w: [0,1] Mixing weight.
    - stickiness: [0,10] Choice stickiness.
    - dampening: [0,1] Factor reducing learning rate for negative errors based on OCI.
    """
    learning_rate, beta, lam, w, stickiness, dampening = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate negative learning rate
    lr_pos = learning_rate
    lr_neg = learning_rate * (1.0 - dampening * oci_score)
    # Ensure non-negative
    if lr_neg < 0: lr_neg = 0.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_1 = np.zeros(2)
    q_mf_2 = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a1 < 0 or s_idx < 0 or a2 < 0:
            continue
            
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_mf_2, axis=1)
        q_mb_1 = transition_matrix @ max_q_stage2
        q_net_1 = w * q_mb_1 + (1 - w) * q_mf_1
        
        if prev_action_1 != -1:
            q_net_1[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 ---
        exp_q2 = np.exp(beta * q_mf_2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        delta_1 = q_mf_2[s_idx, a2] - q_mf_1[a1]
        delta_2 = r - q_mf_2[s_idx, a2]
        
        # Apply asymmetric learning rates based on sign of prediction error
        lr_1 = lr_pos if delta_1 >= 0 else lr_neg
        lr_2 = lr_pos if delta_2 >= 0 else lr_neg
        
        q_mf_1[a1] += lr_1 * delta_1
        q_mf_1[a1] += lr_2 * lam * delta_2 # Use lr_2 for the eligibility update part
        
        q_mf_2[s_idx, a2] += lr_2 * delta_2
        
        prev_action_1 = a1
        
    return log_loss
```

### Model 3: Hybrid Model with OCI-Suppressed Model-Based Control
This model tests the classic theory that compulsivity (high OCI) involves a shift from Goal-Directed (Model-Based) to Habitual (Model-Free) control. The mixing weight `w` is not a free parameter but is constrained to decrease as OCI increases.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI suppresses Model-Based control.
    Hypothesis: Higher OCI scores correlate with reduced reliance on the Model-Based system (goal-directed),
    shifting control to the Model-Free (habitual) system.
    
    w_effective = w_base * (1 - oci)
    
    Parameters:
    - learning_rate: [0,1] Learning rate.
    - beta_1: [0,10] Inverse temperature for Stage 1.
    - beta_2: [0,10] Inverse temperature for Stage 2.
    - lam: [0,1] Eligibility trace.
    - w_base: [0,1] Baseline Model-Based weight (theoretical max for OCI=0).
    - stickiness: [0,10] Choice stickiness.
    """
    learning_rate, beta_1, beta_2, lam, w_base, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w, reduced by OCI
    w_eff = w_base * (1.0 - oci_score)
    if w_eff < 0: w_eff = 0.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf_1 = np.zeros(2)
    q_mf_2 = np.zeros((2, 2))
    
    log_loss = 0.0
    eps = 1e-10
    prev_action_1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a1 < 0 or s_idx < 0 or a2 < 0:
            continue
            
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_mf_2, axis=1)
        q_mb_1 = transition_matrix @ max_q_stage2
        
        # Net value uses effective w
        q_net_1 = w_eff * q_mb_1 + (1 - w_eff) * q_mf_1
        
        if prev_action_1 != -1:
            q_net_1[prev_action_1] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 ---
        exp_q2 = np.exp(beta_2 * q_mf_2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        delta_1 = q_mf_2[s_idx, a2] - q_mf_1[a1]
        delta_2 = r - q_mf_2[s_idx, a2]
        
        q_mf_1[a1] += learning_rate * delta_1
        q_mf_1[a1] += learning_rate * lam * delta_2
        
        q_mf_2[s_idx, a2] += learning_rate * delta_2
        
        prev_action_1 = a1
        
    return log_loss
```
