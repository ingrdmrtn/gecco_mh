Here are three new cognitive models expressed as Python functions. These models focus on different mechanisms by which Obsessive-Compulsive Inventory (OCI) scores might influence reinforcement learning: through mixing model-based/model-free strategies, through loss aversion, or through uncertainty-driven exploration.

### Model 1: Hybrid Learner with OCI-modulated Weighting
This model hypothesizes that individuals with high OCI scores might rely more on habitual (Model-Free) control rather than goal-directed (Model-Based) planning, perhaps due to cognitive rigidity or stress. The parameter $w$ (mixing weight) determines the balance between these systems, and OCI scales this balance.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free learner where the mixing weight 'w' is 
    modulated by OCI score.
    
    Hypothesis: Higher OCI scores lead to reduced Model-Based control (lower w),
    favoring habitual (Model-Free) responses.
    
    Parameters:
    learning_rate: [0, 1] Learning rate for value updates.
    beta: [0, 10] Inverse temperature for softmax choice.
    w_base: [0, 1] Baseline mixing weight (0=MF, 1=MB).
    oci_mod: [0, 1] Strength of OCI modulation reducing MB control.
    """
    learning_rate, beta, w_base, oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w: higher OCI reduces w (pushes towards MF)
    # w is bounded between 0 and 1
    w = w_base * (1.0 - (oci_score * oci_mod))
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (also used for MB calculation)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- STAGE 1 CHOICE ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax policy
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 CHOICE ---
        # Standard Q-learning choice at second stage
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        # Stage 2 Update (TD error)
        # Q(s, a2) += lr * (r - Q(s, a2))
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD-0 for MF system)
        # Q(a1) += lr * (Q(s, a2) - Q(a1))
        # Note: We use the *updated* Q value of stage 2 for the TD target often, 
        # or the un-updated. Here we use the updated value as a proxy for the value of the state reached.
        # However, standard SARSA(0) uses Q(s',a'). 
        # In this task structure, r comes immediately after a2.
        # Standard implementation often drives stage 1 MF by the Q-value of the chosen stage 2 option.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Asymmetric Learning Rates Modulated by OCI
This model focuses on how feedback is processed. It hypothesizes that high OCI is associated with "perfectionism" or hyper-sensitivity to failure (negative prediction errors). Therefore, the learning rate for negative prediction errors is amplified by the OCI score.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with asymmetric learning rates for positive and negative 
    prediction errors. The negative learning rate is scaled by OCI.
    
    Hypothesis: High OCI participants are hypersensitive to negative outcomes (errors),
    leading to faster updating (unlearning) after failure.
    
    Parameters:
    lr_pos: [0, 1] Learning rate for positive prediction errors.
    lr_neg_base: [0, 1] Base learning rate for negative prediction errors.
    beta: [0, 10] Inverse temperature.
    oci_sens: [0, 5] Sensitivity factor scaling the negative learning rate with OCI.
    """
    lr_pos, lr_neg_base, beta, oci_sens = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate
    # Bounded at 1.0. High OCI increases the impact of negative outcomes.
    lr_neg = lr_neg_base * (1.0 + oci_score * oci_sens)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Free setup
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- STAGE 1 CHOICE ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 CHOICE ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        # Stage 2
        pe2 = r - q_stage2[s_idx, a2]
        if pe2 >= 0:
            q_stage2[s_idx, a2] += lr_pos * pe2
        else:
            q_stage2[s_idx, a2] += lr_neg * pe2
            
        # Stage 1
        # Using the value of the chosen second stage state-action pair as the target
        pe1 = q_stage2[s_idx, a2] - q_stage1[a1]
        if pe1 >= 0:
            q_stage1[a1] += lr_pos * pe1
        else:
            q_stage1[a1] += lr_neg * pe1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid Learner with OCI-modulated Eligibility Trace (Lambda)
This model introduces an eligibility trace parameter ($\lambda$) that connects the second-stage outcome directly to the first-stage choice. It hypothesizes that OCI affects how strongly distant outcomes (stage 2 rewards) reinforce initial choices (stage 1), effectively blending Model-Free TD(0) and Monte-Carlo-like updating.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid learner using an eligibility trace (lambda) to connect stage 2 outcomes
    to stage 1 choices. The lambda parameter is modulated by OCI.
    
    Hypothesis: High OCI participants might have a stronger 'direct' connection 
    between final outcome and initial choice (higher lambda), ignoring the 
    intermediate transition structure (Model-Free TD(1) behavior).
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    lambda_base: [0, 1] Base eligibility trace parameter.
    oci_lambda_scale: [0, 2] Scaling factor for lambda based on OCI.
    """
    learning_rate, beta, lambda_base, oci_lambda_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective lambda
    # Higher OCI -> Higher lambda (more credit assignment to first action from final reward)
    lam = lambda_base * (1.0 + oci_score * oci_lambda_scale)
    lam = np.clip(lam, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- STAGE 1 CHOICE ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 CHOICE ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- UPDATES ---
        # Stage 2 Prediction Error
        pe2 = r - q_stage2[s_idx, a2]
        
        # Stage 1 Prediction Error (based on transition to Stage 2 state value)
        # Note: Standard TD(lambda) splits this. 
        # Here we implement a simplified update:
        # Q1 is updated by the immediate transition error (pe1) AND a portion of the second stage error (pe2)
        
        # 1. Update Stage 2 value
        q_stage2[s_idx, a2] += learning_rate * pe2
        
        # 2. Update Stage 1 value
        # pe1 is the difference between the value of the state we landed in (before update) and Q1
        # We use the Q-value of the chosen option a2 as the value of state s_idx
        v_stage2_pre = q_stage2[s_idx, a2] - (learning_rate * pe2) # Retrieve pre-update value
        pe1 = v_stage2_pre - q_stage1[a1]
        
        # The update for Q1 includes the transition error plus the discounted stage 2 error
        # Q(a1) <- Q(a1) + lr * (pe1 + lambda * pe2)
        q_stage1[a1] += learning_rate * (pe1 + lam * pe2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
