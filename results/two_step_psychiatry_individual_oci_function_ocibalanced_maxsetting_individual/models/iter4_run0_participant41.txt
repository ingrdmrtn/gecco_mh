Here are three new cognitive models for the two-step task, incorporating the OCI score into different mechanisms: **Goal Stickiness**, **Lose-Stay Stickiness**, and **Stage-Specific Precision**.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model with OCI-modulated Goal Stickiness.
    
    This model posits that high OCI participants are more likely to persevere 
    towards the *outcome* (Planet) they previously visited, rather than just the action.
    This 'Goal Stickiness' is added to the Q-values based on the transition probability 
    to the previous state.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature for softmax (Stage 1 and 2).
    w: [0,1] - Weight for Model-Based values (0=MF, 1=MB).
    goal_stick_base: [0,5] - Baseline goal stickiness.
    goal_stick_oci: [0,5] - Additional goal stickiness scaling with OCI.
    """
    learning_rate, beta, w, goal_stick_base, goal_stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Transition matrix: T[action, state] -> prob of reaching state given action
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2) # MF values for stage 1 actions
    q_stage2_mf = np.zeros((2, 2)) # MF values for stage 2 states (planets) -> aliens
    
    last_state = -1
    
    goal_stickiness = goal_stick_base + goal_stick_oci * oci_score

    for trial in range(n_trials):
        # Stage 1 Policy
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value mixing
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Goal Stickiness (Bonus to actions that lead to the previous planet)
        if last_state != -1:
            # Bonus = stickiness * P(last_state | action)
            bonus = goal_stickiness * transition_matrix[:, last_state]
            q_net += bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Updates
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace update for Stage 1
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        last_state = state_idx

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL model with OCI-modulated Lose-Stay Stickiness.
    
    This model posits that high OCI participants exhibit compulsive persistence (stickiness)
    specifically after receiving no reward (Loss), failing to switch strategies.
    
    Parameters:
    learning_rate: [0,1] - Update rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF weighting.
    stick_base: [0,5] - General stickiness (repetition bias).
    stick_loss_oci: [0,5] - Additional stickiness after a loss (0 reward), scaled by OCI.
    """
    learning_rate, beta, w, stick_base, stick_loss_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action = -1
    last_reward = -1
    
    loss_stickiness_param = stick_loss_oci * oci_score

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action != -1:
            stick_val = stick_base
            # If last trial was a loss (reward == 0), add OCI-modulated stickiness
            if last_reward == 0:
                stick_val += loss_stickiness_param
            
            q_net[last_action] += stick_val
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        last_action = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dual-Beta Hybrid RL model with OCI-modulated Stage 2 Precision.
    
    This model separates the exploration/exploitation parameters for Stage 1 (Spaceship)
    and Stage 2 (Alien). High OCI is hypothesized to affect the determinism (beta) 
    of the second stage choice specifically, reflecting anxiety-driven rigidity or noise
    in the execution phase.
    
    Parameters:
    learning_rate: [0,1] - Update rate.
    beta_1: [0,10] - Inverse temperature for Stage 1.
    w: [0,1] - MB/MF weighting.
    beta_2_base: [0,10] - Baseline inverse temperature for Stage 2.
    beta_2_oci: [-5,5] - Modulation of Stage 2 beta by OCI.
    """
    learning_rate, beta_1, w, beta_2_base, beta_2_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate Stage 2 beta
    beta_2 = beta_2_base + beta_2_oci * oci_score
    # Ensure beta is non-negative
    beta_2 = max(0.0, beta_2) 

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # Stage 2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
