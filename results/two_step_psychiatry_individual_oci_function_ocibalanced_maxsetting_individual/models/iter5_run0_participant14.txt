Here are three cognitive models that incorporate the OCI score into the decision-making process in distinct, theoretically grounded ways.

### Model 1: OCI-Modulated Eligibility Trace
This model hypothesizes that OCI scores affect the "credit assignment" processâ€”specifically, how strongly the outcome at the second stage (getting a coin or not) reinforces the choice made at the first stage. A higher eligibility trace implies a stronger habitual link between the final reward and the initial spaceship choice, bypassing the model-based structure.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Eligibility Trace Model.
    
    Hypothesis: Obsessive-compulsive traits affect how 'deeply' a participant links 
    the final outcome back to the initial choice (credit assignment). 
    This model modulates the eligibility trace (lambda) by the OCI score.
    
    Update rule for Stage 1 MF:
    Q1_MF += learning_rate * (lambda_base + lambda_oci * OCI) * delta_stage2
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - lambda_base: [0, 1] Baseline eligibility trace.
    - lambda_oci_sensitivity: [0, 5] How much OCI increases/decreases the eligibility trace.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_sensitivity = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate effective lambda bounded between 0 and 1
    # We assume OCI might increase the 'habitual' smearing of reward back to stage 1
    eligibility_trace = lambda_base + (lambda_oci_sensitivity * oci_val)
    eligibility_trace = np.clip(eligibility_trace, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of chosen action
        a1 = int(action_1[trial])
        if a1 != -1: # Handle missing data if any
            p_choice_1[trial] = probs_1[a1]
        else:
            p_choice_1[trial] = 0.5 # Fallback
            continue

        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        if a2 != -1:
            p_choice_2[trial] = probs_2[a2]
        else:
            p_choice_2[trial] = 0.5
            continue

        # --- Updates ---
        r = reward[trial]
        
        # Stage 1 Prediction Error (TD0)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace Update: Propagate Stage 2 error back to Stage 1
        # Scaled by OCI-modulated eligibility trace
        q_stage1_mf[a1] += learning_rate * eligibility_trace * delta_stage2

    eps = 1e-10
    # Filter out invalid trials (marked -1 in data if present, though data looks clean)
    mask = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Loss Aversion with Stickiness
This model combines the successful "stickiness" parameter from previous iterations with a mechanism for "Loss Aversion" modulated by OCI. It hypothesizes that individuals with higher OCI scores might react more strongly to the absence of a reward (0 coins), effectively treating it as a larger negative prediction error, while also maintaining a baseline tendency to repeat choices.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Loss Aversion with Stickiness.
    
    Hypothesis: High OCI contributes to an over-sensitivity to 'failure' (0 coins).
    When the reward is 0 (negative prediction error), the learning update is amplified 
    by the OCI score. This is combined with a standard stickiness parameter.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - stickiness: [0, 5] Base perseveration bonus.
    - loss_amp_oci: [0, 10] Amplification of negative prediction errors based on OCI.
    """
    learning_rate, beta, w, stickiness, loss_amp_oci = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            prev_a1 = -1
            continue
        
        state_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus
        stick_bonus = np.zeros(2)
        if prev_a1 != -1:
            stick_bonus[prev_a1] = stickiness
            
        exp_q1 = np.exp(beta * (q_net + stick_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 1 MF update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF update with OCI-modulated Loss Aversion
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        effective_lr = learning_rate
        
        # If the outcome was a 'loss' (0 coins) relative to expectation (or just 0 reward),
        # amplify the impact of this negative signal based on OCI.
        # We check if delta is negative (disappointment) or simply if reward is 0.
        # Here we apply it when reward is 0.
        if r == 0:
            # Increase effective learning rate for losses
            # factor = 1 + (scale * OCI)
            factor = 1.0 + (loss_amp_oci * oci_val)
            effective_lr = learning_rate * factor
            # Cap LR at 1.0 to prevent instability
            if effective_lr > 1.0: effective_lr = 1.0
            
        q_stage2_mf[state_idx, a2] += effective_lr * delta_stage2
        q_stage1_mf[a1] += effective_lr * delta_stage2 # Eligibility trace = 1 assumed here
        
        prev_a1 = a1

    eps = 1e-10
    mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Win-Stay Stickiness
The previous best model used general stickiness modulated by OCI. This model refines that hypothesis: it proposes that OCI-related perseveration is not just repetition, but specifically a "Win-Stay" compulsion. The tendency to repeat a choice is split into a baseline stickiness and an OCI-dependent component that only activates after a reward (coin) was received.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Win-Stay Stickiness.
    
    Hypothesis: OCI modulates stickiness specifically in the context of positive 
    reinforcement (Win-Stay). The participant has a baseline tendency to repeat,
    but the OCI score scales an additional urge to repeat specifically when the 
    previous trial was rewarded.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - stick_base: [0, 5] Baseline tendency to repeat regardless of outcome.
    - stick_win_oci: [0, 5] Additional stickiness applied only after a win, scaled by OCI.
    """
    learning_rate, beta, w, stick_base, stick_win_oci = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_reward = 0

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            prev_a1 = -1
            continue
            
        state_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Stickiness
        stick_bonus = np.zeros(2)
        if prev_a1 != -1:
            # Base stickiness
            total_stick = stick_base
            
            # OCI-modulated Win-Stay bonus
            if prev_reward == 1.0:
                total_stick += (stick_win_oci * oci_val)
                
            stick_bonus[prev_a1] = total_stick

        exp_q1 = np.exp(beta * (q_net + stick_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        prev_a1 = a1
        prev_reward = r

    eps = 1e-10
    mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
