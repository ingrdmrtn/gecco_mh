def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Eligibility Trace (TD-Lambda).
    OCI level determines the eligibility trace parameter (lambda).
    Higher OCI leads to a higher lambda, meaning the Stage 1 choice is 
    more directly reinforced by the Stage 2 outcome (habitual/compulsive), 
    bypassing the model-based or pure TD(0) chain.

    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    stick: [0, 5] Stickiness (repetition bias).
    lambda_oci: [0, 2] Scaling factor for lambda. lambda = clip(lambda_oci * OCI, 0, 1).
    """
    lr, beta, w, stick, lambda_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate lambda based on OCI
    lam = lambda_oci * oci_score
    if lam > 1.0: lam = 1.0
    if lam < 0.0: lam = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Skip missing data
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]):
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stick

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        # TD Error Stage 1 (based on Stage 2 value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # TD Error Stage 2 (based on Reward)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Eligibility Trace Update: Stage 2 error propagates to Stage 1 value
        # This connects the final reward directly to the first choice
        q_stage1_mf[a1] += lr * lam * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Accumulating Stickiness.
    Stickiness is modeled as a decaying trace rather than just the last choice.
    OCI modulates the decay rate. Higher OCI leads to slower decay (higher retention),
    simulating compulsive repetition or "getting stuck" on a choice.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    stick_mag: [0, 5] Magnitude of the stickiness bias.
    decay_oci: [0, 1] Decay scaling factor. decay = decay_oci * OCI.
                  (If OCI=0, decay=0 -> Standard stickiness).
                  (If OCI is high, decay -> 1 -> Strong accumulation).
    """
    lr, beta, w, stick_mag, decay_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate decay rate based on OCI
    decay = decay_oci * oci_score
    if decay > 0.95: decay = 0.95 # Cap to prevent explosion
    
    stick_trace = np.zeros(2)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]):
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add accumulating stickiness trace
        q_net += stick_mag * stick_trace

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Update stickiness trace
        # Decay all traces, then increment chosen
        stick_trace *= decay
        stick_trace[a1] += 1.0

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Transition-Dependent Stickiness.
    Stickiness (perseveration) varies depending on whether the previous outcome was 
    a Common or Rare transition. OCI modulates the additional stickiness applied 
    after a Rare transition, potentially capturing a "reactionary" persistence 
    or avoidance after unexpected structure events.

    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    stick_base: [0, 5] Baseline stickiness (applied after Common transitions).
    stick_rare_mod: [0, 10] Modifier for stickiness after Rare transitions, scaled by OCI.
                    Total Rare Stickiness = stick_base + (stick_rare_mod * OCI).
    """
    lr, beta, w, stick_base, stick_rare_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_transition_rare = False

    for trial in range(n_trials):
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]):
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            # Determine effective stickiness based on previous transition type
            eff_stick = stick_base
            if last_transition_rare:
                eff_stick += stick_rare_mod * oci_score
            
            q_net[last_action_1] += eff_stick

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Track for next trial
        last_action_1 = a1
        # Check if transition was rare
        # Common transitions: Action 0 -> State 0, Action 1 -> State 1
        if a1 == s_idx:
            last_transition_rare = False
        else:
            last_transition_rare = True

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
