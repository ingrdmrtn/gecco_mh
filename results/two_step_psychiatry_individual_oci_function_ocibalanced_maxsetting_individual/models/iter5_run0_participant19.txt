Here are three new cognitive models, expressed as Python functions, that incorporate the OCI score into the decision-making process.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with Subjective Transition Probability modulated by OCI.
    Hypothesis: OCI affects the agent's internal model of the environment's structure.
    High OCI might correlate with a rigid belief in a specific transition probability 
    (e.g., overestimating the determinism of the 'common' transition), effectively 
    distorting the Model-Based value calculation.
    
    Parameters:
    lr: Learning rate for Q-values [0, 1]
    beta: Inverse temperature [0, 10]
    w: Mixing weight (0=MF, 1=MB) [0, 1]
    p_common_base: Baseline probability assigned to common transitions [0, 1]
    p_common_oci: Effect of OCI on the perceived common transition probability [0, 1]
    """
    lr, beta, w, p_common_base, p_common_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate subjective transition probability
    # We clip to ensure it stays a valid probability.
    p_common = p_common_base + p_common_oci * current_oci
    p_common = np.clip(p_common, 0.0, 1.0) 
    
    # Construct the fixed subjective transition matrix
    # A -> X (0->0) is p_common, A -> Y (0->1) is 1-p_common
    # U -> Y (1->1) is p_common, U -> X (1->0) is 1-p_common
    transition_matrix = np.array([[p_common, 1-p_common], [1-p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial]) # 0 for X, 1 for Y
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # The MB value is calculated using the subjective transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        # Stage 2 Q-learning
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Stage 1 MF SARSA/Q-learning
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where the Model-Based transition matrix is learned dynamically.
    The learning rate for the transition matrix is modulated by OCI.
    Hypothesis: OCI affects how quickly individuals update their structural knowledge 
    of the environment (transition probabilities) in response to observed transitions.
    High OCI might lead to rapid 'over-updating' of the world model or rigidity (low updating).
    
    Parameters:
    lr_q: Learning rate for Q-values [0, 1]
    beta: Inverse temperature [0, 10]
    w: Mixing weight [0, 1]
    lr_t_base: Baseline learning rate for transition probabilities [0, 1]
    lr_t_oci: Effect of OCI on transition learning rate [0, 1]
    """
    lr_q, beta, w, lr_t_base, lr_t_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    lr_t = lr_t_base + lr_t_oci * current_oci
    lr_t = np.clip(lr_t, 0.0, 1.0)

    # Initialize transition matrix with standard prior
    # Row 0: A -> [X, Y], Row 1: U -> [X, Y]
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Use the dynamically updated transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        
        # 1. Update Transition Matrix based on observed transition
        # Create a target vector (one-hot) for the state we actually arrived at
        target = np.zeros(2)
        target[s_idx] = 1.0
        # Update the row corresponding to the chosen spaceship
        transition_matrix[a1] += lr_t * (target - transition_matrix[a1])
        
        # 2. Value Updates
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_q * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_q * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with separate inverse temperatures (beta) for Stage 1 (Spaceships) and Stage 2 (Aliens).
    The Stage 2 beta is modulated by OCI.
    Hypothesis: OCI affects the exploration-exploitation balance differently at the level of 
    concrete outcomes/targets (aliens) compared to abstract choices (spaceships). 
    
    Parameters:
    lr: Learning rate [0, 1]
    w: Mixing weight [0, 1]
    beta_1: Inverse temperature for Stage 1 choice [0, 10]
    beta_2_base: Baseline inverse temperature for Stage 2 choice [0, 10]
    beta_2_oci: Effect of OCI on Stage 2 inverse temperature [0, 10]
    """
    lr, w, beta_1, beta_2_base, beta_2_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Stage 2 beta
    beta_2 = beta_2_base + beta_2_oci * current_oci
    beta_2 = np.clip(beta_2, 0.0, 10.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy (Uses fixed beta_1)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy (Uses OCI-modulated beta_2)
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Updates
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
