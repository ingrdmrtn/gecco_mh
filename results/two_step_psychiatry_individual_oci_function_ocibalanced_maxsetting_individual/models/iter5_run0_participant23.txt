Here are three new cognitive models based on the participant's data and OCI score.

### Cognitive Model 1: OCI-Modulated "Loss-Chase" Stickiness
This model hypothesizes that OCI symptoms manifest as a specific type of perseveration: **compulsive repetition after failure**. While standard "stickiness" implies repeating any choice, this model proposes that high OCI participants are specifically prone to "chasing" a loss (sticking with an option that just yielded 0 coins) to verify or correct the outcome, reflecting a "checking" compulsion.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with Loss-Specific Stickiness modulated by OCI.
    
    Hypothesis: High OCI scores correlate with a compulsion to repeat actions 
    specifically after a lack of reward (checking behavior), rather than general stickiness.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based system (0=MF, 1=MB).
    stick_base: [0, 5] General stickiness (perseveration).
    stick_loss_slope: [0, 5] Additional stickiness applied only after a loss (reward=0), scaled by OCI.
    """
    lr, beta, w, stick_base, stick_loss_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Fixed transition matrix for the MB system
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    last_reward = -1
    log_loss = 0.0
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        # Handle missing/invalid data
        if a1 == -1:
            continue
            
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            stickiness_val = stick_base
            # If the previous trial resulted in a loss (r=0), add OCI-modulated checking bonus
            if last_reward == 0:
                stickiness_val += stick_loss_slope * oci_score
            
            q_net_1[last_action_1] += stickiness_val
            
        # Softmax choice 1
        exp_q1 = np.exp(beta * q_net_1)
        if np.sum(exp_q1) == 0: exp_q1 = np.ones(2) # Prevent div by zero
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + 1e-10)

        # --- Stage 2 Policy ---
        # Softmax choice 2 (pure MF at this stage)
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        if np.sum(exp_q2) == 0: exp_q2 = np.ones(2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + 1e-10)

        # --- Learning ---
        # Stage 2 Update
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += lr * delta2
        
        # Stage 1 Update (TD)
        delta1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta1
        
        # Update history
        last_action_1 = a1
        last_reward = r

    return log_loss
```

### Cognitive Model 2: OCI-Modulated Counterfactual Rumination
This model assumes that OCI symptoms are linked to **rumination and "what-if" thinking**. High OCI participants do not just learn from the path they took; they also update the value of the path they *didn't* take (counterfactual updating). Specifically, if they receive a loss (0 coins), they assume the other option would have been a win, and vice-versa. This "grass is greener" bias destabilizes their value estimates.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with Counterfactual Updating (Rumination) modulated by OCI.
    
    Hypothesis: High OCI participants update the unchosen Stage 1 option based on 
    a counterfactual assumption (if I lost here, the other one was a win).
    
    Parameters:
    lr: [0, 1] Learning rate for chosen option.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based system.
    stickiness: [0, 5] Choice persistence.
    cf_slope: [0, 1] Learning rate scaling for the unchosen option (Counterfactual LR = lr * cf_slope * oci).
    """
    lr, beta, w, stickiness, cf_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            last_action_1 = -1
            continue
            
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        if np.sum(exp_q1) == 0: exp_q1 = np.ones(2)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + 1e-10)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        if np.sum(exp_q2) == 0: exp_q2 = np.ones(2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + 1e-10)

        # --- Learning ---
        # Stage 2 Update
        q_stage2_mf[s, a2] += lr * (r - q_stage2_mf[s, a2])
        
        # Stage 1 Update (Chosen)
        q_stage1_mf[a1] += lr * (q_stage2_mf[s, a2] - q_stage1_mf[a1])
        
        # Stage 1 Update (Unchosen / Counterfactual)
        # If OCI is high, update the unchosen arm assuming the opposite outcome
        # If r=0 (loss), assume unchosen was r=1. If r=1, assume unchosen was r=0.
        unchosen_a1 = 1 - a1
        counterfactual_r = 1.0 - r 
        # The value of the unchosen option is pushed towards the counterfactual outcome
        # We use a simplified MF update for the unchosen path
        lr_cf = lr * cf_slope * oci_score
        q_stage1_mf[unchosen_a1] += lr_cf * (counterfactual_r - q_stage1_mf[unchosen_a1])
        
        last_action_1 = a1

    return log_loss
```

### Cognitive Model 3: OCI-Modulated Subjective Punishment
This model hypothesizes that OCI relates to **perfectionism and fear of failure**. Instead of treating 0 coins as a neutral event, high OCI participants perceive it as a punishment (negative reward). This drives stronger avoidance learning than standard RL models which treat 0 as simply "no reward".

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid RL with Subjective Punishment for Failure modulated by OCI.
    
    Hypothesis: High OCI participants perceive a lack of reward (0 coins) 
    not as neutral, but as a punishment (negative reward value), driving avoidance.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based system.
    stickiness: [0, 5] Choice persistence.
    punishment_scale: [0, 5] Magnitude of negative reward for 0 outcomes, scaled by OCI.
                             (Effective Reward = -1 * punishment_scale * oci)
    """
    lr, beta, w, stickiness, punishment_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    last_action_1 = -1
    log_loss = 0.0
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1:
            last_action_1 = -1
            continue
            
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Transform Reward: 0 becomes negative based on OCI
        r_effective = r
        if r == 0:
            r_effective = -1.0 * punishment_scale * oci_score

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        if np.sum(exp_q1) == 0: exp_q1 = np.ones(2)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + 1e-10)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        if np.sum(exp_q2) == 0: exp_q2 = np.ones(2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + 1e-10)

        # --- Learning ---
        # Update both stages using the subjective effective reward
        
        # Stage 2 Update
        delta2 = r_effective - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += lr * delta2
        
        # Stage 1 Update
        delta1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta1
        
        last_action_1 = a1

    return log_loss
```
