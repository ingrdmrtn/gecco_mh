Here are three new cognitive models for the two-step task that incorporate the OCI score to modulate specific mechanisms.

### Model 1: General Perseveration Modulation
This model tests the hypothesis that OCI symptoms drive a baseline tendency to repeat actions regardless of the outcome ("mindless" repetition), while the reward-driven component of repetition remains constant across individuals. This is the inverse of the previous best model (which modulated reward-stickiness).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Two-step model where OCI modulates General Stickiness (p_gen), with fixed Reinforced Stickiness.
    
    Separates stickiness into general perseverance (p_gen) and 
    reward-dependent perseverance (p_rew). OCI modulates the general component.
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1]
    p_gen_base: [0,1] (General stickiness, scaled by 5)
    p_gen_oci_mod: [0,1]
    p_rew: [0,1] (Reinforced stickiness, scaled by 5)
    lambda_val: [0,1]
    """
    learning_rate, beta, w, p_gen_base, p_gen_oci_mod, p_rew, lambda_val = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Modulate General Stickiness: High OCI may lead to higher baseline repetition
    p_gen = p_gen_base + (p_gen_oci_mod - 0.5) * oci_score
    p_gen = np.clip(p_gen, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net

        # Stickiness Logic
        if last_action_1 != -1:
            # General stickiness (modulated by OCI)
            stickiness_bonus = p_gen * 5.0
            
            # Reward-dependent stickiness (fixed)
            if last_reward == 1.0:
                stickiness_bonus += p_rew * 5.0
            
            logits_1[last_action_1] += stickiness_bonus
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Learning
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_val * delta_stage2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Choice Trace Decay Modulation
This model implements a "Choice Trace" (exponentially decaying memory of past choices) instead of simple 1-step stickiness. OCI modulates the **decay rate** of this trace. High OCI might correlate with a slower decay (value close to 1), implying that past choices exert influence for longer periods (compulsive holding on).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Two-step model with Choice Trace Stickiness. OCI modulates the Trace Decay rate.
    
    Instead of 1-step stickiness, uses an exponentially decaying trace of past choices.
    OCI modulates how fast this trace decays (compulsive 'holding on' to past choices).
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1]
    trace_weight: [0,1] (Scaled by 5)
    decay_base: [0,1]
    decay_oci_mod: [0,1]
    lambda_val: [0,1]
    """
    learning_rate, beta, w, trace_weight, decay_base, decay_oci_mod, lambda_val = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Modulate Decay: High OCI -> Slower decay (higher value)
    decay = decay_base + (decay_oci_mod - 0.5) * oci_score
    decay = np.clip(decay, 0.0, 0.99) # Clip < 1 to ensure stability
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Trace vector initialized to zero
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        # Add Trace Stickiness
        logits_1 += trace_weight * 5.0 * choice_trace
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Update Trace: Decay existing trace and increment chosen action
        choice_trace *= decay
        choice_trace[a1] += 1.0

        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Learning
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_val * delta_stage2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Stage 2 Learning Rate Modulation
This model separates the learning rates for Stage 1 (Choice) and Stage 2 (Alien/Bandit). OCI modulates the **Stage 2 Learning Rate**. This tests if OCI affects the direct valuation of outcomes (aliens) differently from the temporal-difference updates used for the first-stage choice.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Two-step model with separate learning rates for Stage 1 and Stage 2.
    OCI modulates Stage 2 Learning Rate.
    
    Hypothesis: OCI affects the direct value learning (Stage 2) differently from 
    temporal difference updates (Stage 1).
    
    Bounds:
    lr_1: [0,1]
    lr_2_base: [0,1]
    lr_2_oci_mod: [0,1]
    beta: [0,10]
    w: [0,1]
    p: [0,1] (Stickiness, scaled by 5)
    lambda_val: [0,1]
    """
    lr_1, lr_2_base, lr_2_oci_mod, beta, w, p, lambda_val = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Modulate Stage 2 LR
    lr_2 = lr_2_base + (lr_2_oci_mod - 0.5) * oci_score
    lr_2 = np.clip(lr_2, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        if last_action_1 != -1:
            logits_1[last_action_1] += p * 5.0
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Learning
        # Stage 1 update uses lr_1 for consistency in this stage's plasticity
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += lr_1 * delta_stage1 + lr_1 * lambda_val * delta_stage2
        
        # Stage 2 update uses modulated lr_2
        q_stage2_mf[s_idx, a2] += lr_2 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
