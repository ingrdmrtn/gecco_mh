Here are 3 new cognitive models that introduce different mechanisms for how OCI scores might modulate learning and decision-making in the two-step task.

### Model 1: OCI-Modulated Model-Based Weighting with Asymmetric Learning
This model tests the hypothesis that OCI symptoms affect the balance between model-based (goal-directed) and model-free (habitual) control (`w`), but specifically couples this with separate learning rates for positive and negative prediction errors. High OCI might be associated with a stronger reliance on habits (lower `w`) and a heightened sensitivity to negative outcomes (punishment).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Model-Based Weighting with Asymmetric Learning.

    This model posits that OCI affects the balance between model-based and model-free 
    systems (w), while also allowing for asymmetric learning rates for positive and 
    negative prediction errors.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive prediction errors.
    - lr_neg_base: [0, 1] Base learning rate for negative prediction errors.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w_base: [0, 1] Base mixing weight for model-based system (0=MF, 1=MB).
    - w_oci_mod: [0, 1] Modulation of w by OCI score.
    """
    lr_pos, lr_neg_base, beta, w_base, w_oci_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # OCI modulates the weight w. 
    # Hypothesis: Higher OCI leads to more habitual behavior (lower w).
    # We constrain w to be between 0 and 1.
    w = w_base - (w_oci_mod * oci_score)
    w = min(1.0, max(0.0, w))
    
    # We use the base negative learning rate directly (could also modulate this, 
    # but keeping params <= 5 means we focus on w modulation here while allowing asymmetry).
    lr_neg = lr_neg_base 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        current_lr = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[s2, a2] += current_lr * delta_stage2

        # Stage 1 Update
        # Note: Standard TD(0) update for Stage 1 MF uses Stage 2 value
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        current_lr_s1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[a1] += current_lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Choice Perseverance (Sticky Choice)
This model investigates if OCI symptoms relate to "stickiness" or choice perseveranceâ€”the tendency to repeat the previous choice regardless of reward. High OCI (compulsivity) might manifest as a higher tendency to repeat actions (perseveration).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Choice Perseverance.

    This model adds a 'stickiness' parameter to the first-stage choice.
    The stickiness bias is modulated by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - stick_base: [0, 5] Base stickiness bonus added to the previously chosen option.
    - stick_oci: [0, 5] Additional stickiness scaling with OCI.
    """
    learning_rate, beta, w, stick_base, stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate total stickiness
    stickiness = stick_base + (stick_oci * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice (initialize to -1 or handle first trial separately)
    prev_a1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus to Q-values before softmax
        q_net_stick = q_net.copy()
        if prev_a1 != -1:
            q_net_stick[prev_a1] += stickiness
        
        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update history
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Second-Stage Update Sensitivity (Lambda)
This model introduces an eligibility trace parameter (`lambda`) that is modulated by OCI. This parameter controls how much the second-stage reward prediction error updates the first-stage model-free values. A high `lambda` means the first stage is strongly updated by the final outcome (direct reinforcement), while a low `lambda` relies more on the temporal difference at the transition. This mechanism explores if OCI affects how credit is assigned across the two stages.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Eligibility Trace (Lambda).

    This model modulates the eligibility trace parameter (lambda) with OCI.
    Lambda controls the extent to which the Stage 2 reward prediction error 
    directly updates Stage 1 values.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - lambda_base: [0, 1] Base eligibility trace parameter.
    - lambda_oci: [0, 1] Modulation of lambda by OCI.
    """
    learning_rate, beta, w, lambda_base, lambda_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate lambda based on OCI
    # We hypothesize OCI might increase direct reinforcement (higher lambda)
    # or decrease it.
    lambda_param = lambda_base + (lambda_oci * oci_score)
    lambda_param = min(1.0, max(0.0, lambda_param))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # Stage 1 RPE (standard TD error based on Stage 2 value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        
        # Update Stage 2
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1
        # The update includes the immediate TD error plus a portion of the Stage 2 RPE
        # scaled by lambda. This connects the final reward to the first choice.
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_param * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
