Here are the three proposed cognitive models.

### Model 1: Pure Model-Free with OCI-Attenuated Learning and Stickiness
This model tests the hypothesis that high OCI scores are associated with "cognitive rigidity" manifested as a reduced ability to update value estimates (low learning rate) combined with a general tendency to repeat choices (stickiness). It assumes a pure Model-Free system ($w=0$) where the learning rate is attenuated by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Free Learning with OCI-Attenuated Learning Rate.
    
    Hypothesis: High OCI participants exhibit cognitive rigidity, modeled as 
    a reduced effective learning rate (slower updating) and baseline stickiness.
    The learning rate is scaled down as OCI increases.
    
    Parameters:
    - lr_base: [0, 1] Base learning rate.
    - lr_attenuation: [0, 1] Factor by which OCI reduces the learning rate. 
      Effective LR = lr_base * (1 - lr_attenuation * OCI).
    - beta: [0, 10] Inverse temperature.
    - stickiness: [0, 5] Choice perseveration bonus (constant).
    """
    lr_base, lr_attenuation, beta, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective learning rate based on OCI
    # Bounded to be non-negative
    learning_rate = lr_base * (1.0 - lr_attenuation * oci_score)
    if learning_rate < 0:
        learning_rate = 0.0

    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    last_choice = -1
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        # Skip missing data
        if action_1[trial] == -1:
            last_choice = -1
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Stage 1 Policy: Pure MF + Stickiness
        q_net_stage1 = q_stage1_mf.copy()
        if last_choice != -1:
            q_net_stage1[last_choice] += stickiness
            
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # Q-Learning Updates (TD(0))
        # Update Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1
        # In pure MF (TD0), we update Q1 towards Q2 of the state we landed in
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        last_choice = a1

    return log_loss
```

### Model 2: Hybrid MB/MF with OCI-Weighted Cumulative Habit
This model hypothesizes that OCI-related stickiness is not just a one-step repetition bias, but a cumulative "habit" that builds up over multiple repetitions. The strength (weight) of this habit influence on the decision is proportional to the OCI score. It operates within a hybrid Model-Based/Model-Free framework.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Weighted Cumulative Habit Trace.
    
    Hypothesis: OCI drives behavior via a cumulative habit mechanism (Choice Trace).
    The weight of this habit history in the decision process is proportional to OCI.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values and Habit trace decay.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    - habit_oci_scale: [0, 5] Scaling factor for the habit weight. 
      Habit Weight = habit_oci_scale * OCI.
    """
    learning_rate, beta, w, habit_oci_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    habit_weight = habit_oci_scale * oci_score
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    # Cumulative choice trace (Habit)
    habit_trace = np.zeros(2)
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-value: Mix of MB, MF, and Cumulative Habit
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf) + (habit_weight * habit_trace)
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # Updates
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Habit Trace (Exponential Moving Average of choices)
        # Decays unchosen, strengthens chosen
        habit_trace *= (1.0 - learning_rate)
        habit_trace[a1] += learning_rate * 1.0

    return log_loss
```

### Model 3: Hybrid MB/MF with OCI-Modulated Eligibility Trace
This model extends the "eligibility trace" finding from pure MF models to a hybrid MB/MF context. It proposes that while the participant may use Model-Based planning (controlled by `w`), the direct reinforcement of Stage 1 choices by Stage 2 outcomes (the eligibility trace $\lambda$) is specifically sensitive to OCI severity.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Modulated Eligibility Trace.
    
    Hypothesis: The strength of the eligibility trace (connecting outcome to Stage 1 choice)
    is proportional to OCI severity, even within a hybrid MB/MF architecture.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight for MB/MF.
    - lambda_scale: [0, 1] Scaling factor for eligibility trace.
      Lambda = lambda_scale * OCI.
    """
    learning_rate, beta, w, lambda_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Lambda (eligibility trace) depends on OCI
    lambda_val = lambda_scale * oci_score
    # Ensure lambda stays within [0, 1]
    if lambda_val > 1.0: lambda_val = 1.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # Updates
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # MF Stage 1 update includes eligibility trace from Stage 2 prediction error
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_val * delta_stage2

    return log_loss
```
