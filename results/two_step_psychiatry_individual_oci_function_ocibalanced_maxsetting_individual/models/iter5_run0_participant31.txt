Here are the three proposed cognitive models.

### Model 1: OCI-Modulated Choice Kernel
This model hypothesizes that OCI symptoms modulate the strength of **habitual perseveration** (or alternation) over a longer timescale than simple stickiness. It uses a "Choice Kernel" that accumulates a trace of past choices with a decay rate. The impact of this kernel on the decision (the weight) is modulated by the OCI score. A positive weight implies habit formation (perseveration), while a negative weight implies a bias for alternation.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Choice Kernel Model.
    
    Tracks a 'Choice Kernel' (accumulated history of choices) with a decay parameter.
    The influence of this kernel on Stage 1 choice is modulated by OCI.
    Allows for modeling long-term perseveration (habit) or alternation biases.
    
    Parameters:
    - learning_rate: [0,1] Standard Q-learning rate.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weighting of Model-Based vs Model-Free values.
    - decay: [0,1] Decay rate of the choice kernel (0 = only last choice, 1 = no forgetting).
    - ck_weight_base: [0,1] Base weight for the choice kernel (mapped to [-5, 5]).
    - ck_weight_oci: [0,1] OCI modulation of the kernel weight (mapped to [-5, 5]).
    """
    learning_rate, beta, w, decay, ck_weight_base, ck_weight_oci = model_parameters
    n_trials = len(action_1)
    oci = oci[0]
    
    # Map [0,1] parameters to effective weights [-5, 5]
    w_base = (ck_weight_base - 0.5) * 10.0
    w_oci = (ck_weight_oci - 0.5) * 10.0
    kernel_weight = w_base + w_oci * oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice kernel: tracks frequency/recency of choosing action 0 vs 1
    choice_kernel = np.zeros(2)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Choice Kernel bonus/penalty
        logits = beta * q_net_1 + kernel_weight * choice_kernel
        
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update Choice Kernel
        choice_kernel = decay * choice_kernel
        choice_kernel[a1] += 1.0

        # --- Stage 2 Decision ---
        q_vals_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * (q_vals_2 - np.max(q_vals_2)))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Subjective Failure Utility
This model hypothesizes that individuals with varying OCI scores perceive the "lack of reward" (0 coins) differently. Instead of treating 0 coins as a neutral outcome ($R=0$), the model treats it as a subjective value $\theta$. If $\theta < 0$, the participant experiences "no reward" as a punishment (loss aversion), driving faster switching. OCI modulates this $\theta$ value.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Subjective Failure Utility.
    
    Replaces the neutral '0' reward with a subjective utility parameter 'theta'.
    OCI modulates 'theta', allowing 'no reward' to be perceived as a punishment (theta < 0)
    or a small gain (theta > 0).
    
    Parameters:
    - learning_rate: [0,1]
    - beta: [0,10]
    - w: [0,1]
    - theta_base: [0,1] Base subjective utility for 0 coins (mapped to [-2, 1]).
    - theta_oci: [0,1] OCI modulation of theta (mapped to [-1, 1]).
    """
    learning_rate, beta, w, theta_base, theta_oci = model_parameters
    n_trials = len(action_1)
    oci = oci[0]
    
    # Map theta parameters
    # theta_base mapped to [-2.0, 1.0]
    base_val = theta_base * 3.0 - 2.0
    # theta_oci mapped to [-1.0, 1.0] scale
    oci_effect = (theta_oci - 0.5) * 2.0
    
    theta = base_val + oci_effect * oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) 

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r_raw = reward[trial]
        
        # Determine effective reward
        if r_raw == 1:
            r_eff = 1.0
        else:
            r_eff = theta

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net_1
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision ---
        q_vals_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * (q_vals_2 - np.max(q_vals_2)))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 using effective reward (subjective utility)
        delta_stage2 = r_eff - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Valence Bias
This model proposes that OCI affects the symmetry of learning from positive outcomes (wins) versus negative outcomes (losses/no-wins). It defines separate learning rates for Rewarded trials (`lr_pos`) and Unrewarded trials (`lr_neg`). OCI modulates both the overall learning speed and the **bias** between positive and negative learning.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Valence Bias in Learning.
    
    Uses separate learning rates for positive (Reward=1) and negative (Reward=0) outcomes.
    OCI modulates the overall learning rate magnitude and the bias (asymmetry) between them.
    
    Parameters:
    - beta: [0,10]
    - w: [0,1]
    - lr_base_param: [0,1] Baseline parameter for overall learning rate.
    - lr_oci_param: [0,1] OCI effect on overall learning rate.
    - bias_base_param: [0,1] Baseline bias towards positive (>0.5) or negative (<0.5) learning.
    - bias_oci_param: [0,1] OCI effect on bias.
    """
    beta, w, lr_base_param, lr_oci_param, bias_base_param, bias_oci_param = model_parameters
    n_trials = len(action_1)
    oci = oci[0]
    
    # Calculate effective Learning Rate Magnitude (pre-sigmoid logic)
    # Map params to a logit-like scale then sigmoid to [0,1]
    lr_logit = (lr_base_param - 0.5) * 4.0 + (lr_oci_param - 0.5) * 4.0 * oci
    lr_avg = 1.0 / (1.0 + np.exp(-lr_logit))
    
    # Calculate effective Bias
    # bias_factor -> 1 means only learn from Pos, 0 means only learn from Neg
    bias_logit = (bias_base_param - 0.5) * 4.0 + (bias_oci_param - 0.5) * 4.0 * oci
    bias_factor = 1.0 / (1.0 + np.exp(-bias_logit))
    
    # Derive lr_pos and lr_neg
    # We scale them so that their average approximates lr_avg, but clipped to [0,1]
    # Factor 2.0 ensures that if bias=0.5, lr_pos = lr_neg = lr_avg
    lr_pos = np.clip(lr_avg * 2.0 * bias_factor, 0.0, 1.0)
    lr_neg = np.clip(lr_avg * 2.0 * (1.0 - bias_factor), 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net_1
        exp_q1 = np.exp(logits - np.max(logits))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Decision ---
        q_vals_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * (q_vals_2 - np.max(q_vals_2)))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Select Learning Rate based on Outcome Valence
        current_lr = lr_pos if r == 1 else lr_neg

        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += current_lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
