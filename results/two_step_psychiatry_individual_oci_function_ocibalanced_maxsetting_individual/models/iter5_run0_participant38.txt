Here are the three proposed cognitive models.

### Model 1: Hybrid MB/MF with OCI-modulated Stickiness
This model extends the previous best Model-Free stickiness model by integrating a Model-Based (planning) component. It tests the hypothesis that while the participant may have high stickiness driven by OCI, they still utilize a mixture of goal-directed (MB) and habitual (MF) strategies. The OCI score scales the magnitude of the choice perseveration.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free learner with OCI-modulated choice stickiness.
    
    This model posits that the participant uses a combination of Model-Based (planning)
    and Model-Free (habitual) strategies. The OCI score modulates the degree of 
    choice perseveration (stickiness), hypothesizing that higher OCI leads to 
    more repetitive behavior regardless of the strategy used.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature (exploitation vs exploration).
    w: [0,1] - Weight of Model-Based control (0=Pure MF, 1=Pure MB).
    stickiness_base: [0,5] - Base stickiness factor, scaled by (1+OCI).
    """
    learning_rate, beta, w, stickiness_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Scale stickiness by OCI: Higher OCI -> Higher tendency to repeat last choice
    stickiness = stickiness_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        # Model-Based Value Calculation (Planning using transition matrix)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value (Weighted mix of MB and MF)
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to the Q-value of the previously chosen action
        q_net = np.copy(q_hybrid)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        last_action_1 = a1

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 Update (Model-Free)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (Model-Free component)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Hybrid MB/MF with OCI-modulated Loss Aversion
This model hypothesizes that high OCI is associated with an altered perception of outcomes, specifically "loss aversion" or "perceived failure." In this task, receiving 0 coins is usually neutral, but this model proposes that high-OCI participants perceive 0 coins as a negative value (punishment).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF learner with OCI-modulated Loss Aversion.
    
    This model hypothesizes that high OCI participants perceive the absence of reward (0 coins)
    as an active punishment (loss) rather than a neutral zero. The magnitude of this 
    'subjective loss' is scaled by their OCI score.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Weight of Model-Based control.
    loss_sensitivity: [0,5] - Magnitude of negative utility for 0-reward, scaled by OCI.
    """
    learning_rate, beta, w, loss_sensitivity = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective punishment for 0 reward based on OCI
    # If OCI is high, 0 coins feels like a negative value.
    punishment_value = -1.0 * loss_sensitivity * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r_raw = reward[trial]
        
        # Transform reward: 1 remains 1, but 0 becomes a negative value
        if r_raw == 1.0:
            r_eff = 1.0
        else:
            r_eff = punishment_value

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf
            
        exp_q1 = np.exp(beta * q_hybrid)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Update using effective reward (r_eff) instead of raw reward
        delta_stage2 = r_eff - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Hybrid MB/MF with OCI-modulated Stage 1 Rigidity
This model proposes that high OCI leads to increased rigidity (higher inverse temperature `beta`) specifically in the first-stage planning/choice. This reflects the idea that anxiety or compulsivity might cause more deterministic or rigid behavior in the high-level decision (spaceship choice) while the immediate reaction (alien choice) remains at a baseline level of exploration.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF learner with OCI-modulated Stage 1 Rigidity.
    
    This model proposes that high OCI leads to increased rigidity (higher inverse temperature beta)
    specifically in the first-stage planning/choice, while the second-stage reaction 
    remains at a baseline level. This differentiates 'planning rigidity' from general noise.
    
    Parameters:
    learning_rate: [0,1] - Update rate.
    beta_base: [0,10] - Baseline inverse temperature (used for Stage 2).
    w: [0,1] - Weight of Model-Based control.
    rigidity_factor: [0,5] - Factor increasing Beta 1 based on OCI.
    """
    learning_rate, beta_base, w, rigidity_factor = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate Stage 1 Beta: scales up with OCI (more rigid/deterministic)
    beta_1 = beta_base * (1.0 + rigidity_factor * oci_score)
    # Stage 2 Beta remains at baseline
    beta_2 = beta_base

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf
            
        # Use beta_1 for the first choice
        exp_q1 = np.exp(beta_1 * q_hybrid)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        # Use beta_2 for the second choice
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
