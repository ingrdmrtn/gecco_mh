Here are three new cognitive models exploring different mechanisms for how high OCI scores might influence decision-making in this task.

### Model 1: Hybrid Learner with OCI-Modulated Model-Based Weight
This model tests the hypothesis that high OCI scores correlate with a rigid reliance on Model-Based (MB) planning or, conversely, a degradation of MB control in favor of Model-Free (MF) habits. Here, the mixing weight `w` between MB and MF values is a function of the OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the balance (w) is modulated by OCI.
    
    Hypothesis: The degree of model-based planning (w) is influenced by obsessive-compulsive traits.
    High OCI might lead to over-thinking (high MB) or rigid habits (low MB).
    w = w_base + w_oci_factor * oci
    
    Bounds:
    learning_rate: [0,1]
    beta: [0,10]
    w_base: [0,1]
    w_oci_factor: [-1, 1]  (Can increase or decrease MB reliance)
    """
    learning_rate, beta, w_base, w_oci_factor = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Cast inputs to integers
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    reward = reward.astype(int)

    # Initialize Q-values
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF Q-values (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Fixed transition matrix for Model-Based calculation
    # Row: chosen spaceship, Col: resulting planet
    # Spaceship A (0) -> Planet X (0) (0.7), Planet Y (1) (0.3)
    # Spaceship U (1) -> Planet X (0) (0.3), Planet Y (1) (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Calculate the mixing weight based on OCI
    # Clip w to be between 0 (pure MF) and 1 (pure MB)
    w = w_base + w_oci_factor * oci_score
    w = np.clip(w, 0.0, 1.0)

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values for Stage 1
        # V_MB(s1) = sum P(s2|s1) * max Q(s2, a')
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined Q-value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice
        logits_1 = beta * q_net
        logits_1 = logits_1 - np.max(logits_1) # stability
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        p_choice_1[t] = probs_1[action_1[t]]
        
        # --- Stage 2 Decision ---
        s_idx = state[t]
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        p_choice_2[t] = probs_2[action_2[t]]
        
        # --- Learning ---
        a1 = action_1[t]
        a2 = action_2[t]
        r = reward[t]
        
        # Update Stage 2 Q-values (Standard TD)
        pe_2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * pe_2
        
        # Update Stage 1 MF Q-values (TD-lambda style, but simplified to direct update from reward here as is common)
        # Often in hybrid models, Stage 1 MF is updated via SARSA(lambda) or direct reinforcement
        # Here we use the standard update toward the Stage 2 value (TD) or reward?
        # Standard Daw et al. 2011 uses SARSA(1) equivalent: update Q1 based on Q2
        # But simple version: Q1 updates towards Q2 value (TD error at stage 1)
        # delta_1 = Q2[s_idx, a2] - Q1[a1]
        
        # However, the prompt template suggests updating Q1 based on delta_stage1 and delta_stage2
        # We will use the direct reward update for simplicity consistent with basic Q-learning unless TD(1) is needed.
        # Let's use a standard TD(0) update for Q1 based on the state reached, which is the definition of MF here.
        # Actually, standard MF in 2-step is usually TD(1) (update from reward) or TD(0) (update from state value).
        # Let's stick to update from Reward to capture "habitual" response to outcome.
        
        pe_1 = r - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free Learner with OCI-Modulated Learning Rate Asymmetry
This model investigates if high OCI is associated with a bias in how positive versus negative prediction errors are processed. For instance, obsessive-compulsive traits might involve hyper-sensitivity to negative outcomes (punishment) or "missed" rewards.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with asymmetric learning rates modulated by OCI.
    
    Hypothesis: OCI affects the balance between learning from positive (alpha_pos) 
    and negative (alpha_neg) prediction errors.
    alpha_neg = alpha_pos * (1 + asymmetry_oci * oci)
    
    Bounds:
    alpha_pos: [0,1]
    beta: [0,10]
    asymmetry_oci: [-1, 5] (Allows OCI to scale negative learning rate up or down relative to positive)
    eligibility_trace: [0,1] (lambda parameter for stage 1 update)
    """
    alpha_pos, beta, asymmetry_oci, eligibility_trace = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    reward = reward.astype(int)

    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Define negative learning rate relative to positive, scaled by OCI
    # If asymmetry_oci > 0, high OCI learns more from failure.
    alpha_neg = alpha_pos * (1.0 + asymmetry_oci * oci_score)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)

    for t in range(n_trials):
        # --- Stage 1 Choice ---
        logits_1 = beta * q_stage1
        logits_1 = logits_1 - np.max(logits_1)
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        p_choice_1[t] = probs_1[action_1[t]]
        
        # --- Stage 2 Choice ---
        s_idx = state[t]
        logits_2 = beta * q_stage2[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        p_choice_2[t] = probs_2[action_2[t]]
        
        # --- Learning ---
        a1 = action_1[t]
        a2 = action_2[t]
        r = reward[t]
        
        # Stage 2 Prediction Error
        pe_2 = r - q_stage2[s_idx, a2]
        
        # Update Stage 2
        lr_2 = alpha_pos if pe_2 >= 0 else alpha_neg
        q_stage2[s_idx, a2] += lr_2 * pe_2
        
        # Stage 1 Prediction Error (TD(lambda))
        # The immediate reward at stage 1 is 0.
        # The value of the state arrived at is q_stage2[s_idx, a2] (SARSA style)
        # PE1 = (Reward_1 + Q2) - Q1
        pe_1 = (0 + q_stage2[s_idx, a2]) - q_stage1[a1]
        
        # Combined PE for Stage 1 update: PE1 + lambda * PE2
        # This is effectively updating Q1 based on the final R, scaled by lambda
        total_pe_1 = pe_1 + eligibility_trace * pe_2
        
        lr_1 = alpha_pos if total_pe_1 >= 0 else alpha_neg
        q_stage1[a1] += lr_1 * total_pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Based Learner with OCI-Driven Inverse Temperature (Noise)
This model posits that OCI affects the exploration-exploitation trade-off. High OCI might correlate with "noisier" or more uncertain decision-making (lower beta), or conversely, very rigid, deterministic selection (high beta).

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Based learner where the inverse temperature (beta) is a function of OCI.
    
    Hypothesis: OCI impacts decision noise. High OCI might lead to rigid (high beta) 
    or erratic (low beta) behavior.
    beta = beta_base + beta_oci_slope * oci
    
    Bounds:
    learning_rate: [0,1]
    beta_base: [0,10]
    beta_oci_slope: [-5, 5]
    """
    learning_rate, beta_base, beta_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    action_1 = action_1.astype(int)
    state = state.astype(int)
    action_2 = action_2.astype(int)
    reward = reward.astype(int)

    # Transition matrix
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Only tracking values at the second stage (the aliens)
    # MB calculates Stage 1 values on the fly.
    q_stage2 = np.zeros((2, 2)) # 2 planets, 2 aliens each
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Calculate effective beta
    beta = beta_base + beta_oci_slope * oci_score
    # Ensure beta stays within reasonable positive bounds (or 0 for random)
    beta = np.clip(beta, 0.0, 20.0)

    for t in range(n_trials):
        # --- Stage 1 Decision (Model-Based) ---
        # Value of a spaceship = P(Planet X)*Max(Q_X) + P(Planet Y)*Max(Q_Y)
        max_q_stage2 = np.max(q_stage2, axis=1) # [MaxVal_PlanetX, MaxVal_PlanetY]
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        logits_1 = beta * q_stage1_mb
        logits_1 = logits_1 - np.max(logits_1)
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        p_choice_1[t] = probs_1[action_1[t]]
        
        # --- Stage 2 Decision ---
        s_idx = state[t]
        logits_2 = beta * q_stage2[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        p_choice_2[t] = probs_2[action_2[t]]
        
        # --- Learning ---
        a2 = action_2[t]
        r = reward[t]
        
        # Update the value of the specific alien chosen
        pe = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * pe

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
