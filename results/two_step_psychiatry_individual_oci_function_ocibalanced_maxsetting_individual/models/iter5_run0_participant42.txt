Here are the 3 proposed cognitive models.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Subjective Punishment Model.
    
    Hypothesis: High OCI participants perceive the lack of reward (0 coins) 
    not just as a neutral outcome, but as a negative outcome (punishment), 
    driven by perfectionism or fear of failure.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta: [0,10] Inverse temperature (decision noise).
    - w: [0,1] Model-based weight.
    - punishment_factor: [0,10] Scaling factor for subjective punishment.
                         Effective reward for 0 coins = -1 * punishment_factor * oci.
    """
    learning_rate, beta, w, punishment_factor = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Subjective punishment value for a loss (0 coins)
    punishment_val = -1.0 * punishment_factor * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Reward Processing
        r = reward[trial]
        # If reward is 0, substitute with subjective punishment
        effective_reward = r if r > 0 else punishment_val
        
        # Updates
        # TD(0) update for Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF with effective reward
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace update for Stage 1 (propagate Stage 2 RPE)
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Decaying Choice Trace Model.
    
    Hypothesis: High OCI is associated with stronger habit formation. 
    This model implements a 'choice trace' (exponentially decaying history of choices)
    that biases Stage 1 decisions. The strength of this bias is modulated by OCI.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Model-based weight.
    - trace_decay: [0,1] Decay rate of the choice trace (0=instant forget, 1=perfect memory).
    - trace_weight_oci: [0,10] Scaling of the trace influence by OCI.
                        Bonus = trace_weight_oci * oci * trace.
    """
    learning_rate, beta, w, trace_decay, trace_weight_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for Stage 1 options (A vs U)
    choice_trace = np.zeros(2)
    
    for trial in range(n_trials):
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add OCI-modulated habit bias
        habit_bonus = trace_weight_oci * oci_score * choice_trace
        
        exp_q1 = np.exp(beta * q_net + habit_bonus)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        # Update Choice Trace
        choice_trace *= trace_decay
        choice_trace[a1] += 1.0
        
        # Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stage 2 Stickiness Model.
    
    Hypothesis: OCI-related compulsivity manifests as "stickiness" (perseveration)
    specifically at the second stage (alien choice), repeating the last choice 
    made in that specific state regardless of reward.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Model-based weight.
    - stick_s2_oci: [0,10] Strength of Stage 2 stickiness modulated by OCI.
    """
    learning_rate, beta, w, stick_s2_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track last choice for each state (Planet X and Planet Y)
    # Initialize with -1 (no previous choice)
    last_choice_s2 = np.array([-1, -1])
    
    stick_weight = stick_s2_oci * oci_score
    
    for trial in range(n_trials):
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        # Stage 2 Choice with Stickiness
        # Create a stickiness vector for the two options in this state
        stick_vec = np.zeros(2)
        if last_choice_s2[state_idx] != -1:
            stick_vec[last_choice_s2[state_idx]] = stick_weight
            
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx] + stick_vec)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Update last choice
        last_choice_s2[state_idx] = a2
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
