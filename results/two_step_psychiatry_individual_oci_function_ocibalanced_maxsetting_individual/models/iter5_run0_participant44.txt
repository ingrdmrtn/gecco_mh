Here are three new cognitive models designed to explain the participant's behavior, specifically incorporating their OCI score to model mechanisms of pessimism, compulsive persistence, and counterfactual monitoring.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pessimistic Model-Based Valuation Model.

    Hypothesis: High OCI participants exhibit 'catastrophizing' or pessimism regarding 
    uncertain future outcomes. When performing Model-Based planning, instead of solely 
    maximizing expected value, they incorporate a weight for the worst-case scenario 
    (minimum Q-value of the next stage), modulated by their OCI score.

    Parameters:
    - learning_rate: MF learning rate [0, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight (MB vs MF) [0, 1]
    - lambd: Eligibility trace [0, 1]
    - pessimism: Weight of the minimum value in MB calculation, scaled by OCI [0, 1]
    """
    learning_rate, beta, w, lambd, pessimism = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action

    # Effective pessimism weight scales with OCI
    w_pess = pessimism * oci_score

    for trial in range(n_trials):
        # Stage 1 Policy
        # MB Calculation with Pessimism: Blend max (optimality) and min (pessimism)
        max_q2 = np.max(q_stage2_mf, axis=1)
        min_q2 = np.min(q_stage2_mf, axis=1)
        val_stage2 = (1 - w_pess) * max_q2 + w_pess * min_q2
        
        q_stage1_mb = transition_matrix @ val_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Learning
        # Stage 1 MF update (TD0)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace update for Stage 1
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Compulsive Choice Trace Model.

    Hypothesis: High OCI is associated with compulsivity, modeled here as 
    a choice trace (stickiness) that decays more slowly for high OCI participants.
    This creates a "buildup" of compulsion to repeat choices over long periods,
    creating a more rigid habit than standard one-step stickiness.

    Parameters:
    - learning_rate: MF learning rate [0, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight [0, 1]
    - lambd: Eligibility trace [0, 1]
    - stick_weight: Magnitude of the stickiness bias [0, 5]
    - persistence: Sensitivity of trace decay to OCI (higher OCI -> slower decay) [0, 1]
    """
    learning_rate, beta, w, lambd, stick_weight, persistence = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for Stage 1 options
    choice_trace = np.zeros(2)
    
    # Decay rate depends on OCI. 
    # If persistence=0, decay=0 (trace resets every trial -> simple 1-step stickiness).
    # If persistence=1, decay=oci (trace persists across trials based on symptom severity).
    decay_rate = persistence * oci_score

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf + stick_weight * choice_trace
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Learning
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2
        
        # Update Choice Trace
        # Decay existing trace
        choice_trace *= decay_rate
        # Add current choice
        choice_trace[a1] += 1.0

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Counterfactual Monitoring Model.

    Hypothesis: High OCI participants engage in active monitoring of the 
    unchosen option ("what if" thinking). They update the Model-Free value 
    of the unchosen spaceship towards its Model-Based estimate, weighted by OCI.
    This reduces the divergence between MF and MB values for unchosen options,
    representing a compulsive "checking" of the alternative path.

    Parameters:
    - learning_rate: MF learning rate [0, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight [0, 1]
    - lambd: Eligibility trace [0, 1]
    - cf_gain: Gain for counterfactual updating, scaled by OCI [0, 1]
    """
    learning_rate, beta, w, lambd, cf_gain = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Rate of counterfactual updating scales with OCI
    cf_rate = learning_rate * cf_gain * oci_score

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Learning
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2
        
        # Counterfactual Update for the unchosen option
        unchosen_a1 = 1 - a1
        # Target is the MB estimate for the unchosen option
        target_val = q_stage1_mb[unchosen_a1]
        # Update unchosen MF value towards MB estimate
        q_stage1_mf[unchosen_a1] += cf_rate * (target_val - q_stage1_mf[unchosen_a1])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
