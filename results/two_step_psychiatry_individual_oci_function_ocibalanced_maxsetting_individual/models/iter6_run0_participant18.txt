Here are the 3 new cognitive models based on the participant's data and OCI score.

### Model 1: OCI-Modulated Rigidity
This model hypothesizes that higher OCI scores correlate with behavioral rigidity. This is implemented by scaling the inverse temperature parameter (`beta`) by the OCI score. A higher `beta` leads to more deterministic choices (exploiting the highest value option) and less exploration, reflecting the rigid behavioral patterns often seen in compulsive phenotypes.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Rigidity (Inverse Temperature).
    Hypothesis: Higher OCI scores are associated with increased behavioral rigidity,
    modeled as an increase in the inverse temperature (beta) parameter.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta_base: Baseline inverse temperature [0,10]
    - w: Model-based weight [0,1]
    - lambda_val: Eligibility trace [0,1]
    - stickiness: Choice perseveration for stage 1 [0,5]
    - oci_rigidity: Scaling factor for OCI effect on beta [0,5]
    """
    lr, beta_base, w, lambda_val, stickiness, oci_rigidity = model_parameters
    oci_score = oci[0]
    n_trials = len(action_1)
    
    # Effective Beta: Increases with OCI, leading to "stiffer" or more rigid choices
    beta = beta_base * (1.0 + oci_rigidity * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        q_stage1_mf[a1] += lr * lambda_val * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Punishment Sensitivity
This model tests the hypothesis that OCI is linked to a hypersensitivity to negative outcomes (failure/missing coins). It modifies the learning rate specifically when a reward of 0 is received. A higher OCI score boosts the learning rate on loss trials, causing the participant to react more strongly to failure than to success.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Punishment Sensitivity.
    Hypothesis: OCI is linked to hypersensitivity to negative outcomes (punishment).
    This is modeled by increasing the learning rate specifically when the reward is 0.
    
    Parameters:
    - lr: Base learning rate [0,1]
    - beta: Inverse temperature [0,10]
    - w: Model-based weight [0,1]
    - lambda_val: Eligibility trace [0,1]
    - stickiness: Choice perseveration [0,5]
    - oci_loss_sens: Proportional boost to LR on loss [0,5]
    """
    lr, beta, w, lambda_val, stickiness, oci_loss_sens = model_parameters
    oci_score = oci[0]
    n_trials = len(action_1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Calculate Effective LR: Boost if reward is 0 (punishment)
        if r == 0:
            eff_lr = lr * (1.0 + oci_loss_sens * oci_score)
            eff_lr = min(eff_lr, 1.0) # Clip at 1.0
        else:
            eff_lr = lr

        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += eff_lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += eff_lr * delta_stage2
        
        q_stage1_mf[a1] += eff_lr * lambda_val * delta_stage2
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Stage 2 Stickiness
This model proposes that compulsivity (OCI) manifests as repetitive behavior ("stickiness") not just in the initial spaceship choice, but also in the second stage (alien choice). It introduces a specific stickiness parameter for Stage 2 that scales with the OCI score, capturing the tendency to repeatedly check the same terminal option.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Stage 2 Stickiness.
    Hypothesis: Compulsivity (OCI) leads to repetitive behaviors not just at the spaceship choice (Stage 1)
    but also at the alien choice (Stage 2), regardless of reward history.
    
    Parameters:
    - lr: Learning rate [0,1]
    - beta: Inverse temperature [0,10]
    - w: Model-based weight [0,1]
    - lambda_val: Eligibility trace [0,1]
    - stick_s1: Stage 1 stickiness [0,5]
    - stick_s2_slope: Sensitivity of Stage 2 stickiness to OCI [0,5]
    """
    lr, beta, w, lambda_val, stick_s1, stick_s2_slope = model_parameters
    oci_score = oci[0]
    n_trials = len(action_1)
    
    # Stage 2 stickiness depends on OCI
    stick_s2 = stick_s2_slope * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_a2 = [-1, -1] # Track previous choice for each state (planet)
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if prev_a1 != -1:
            logits_1[prev_a1] += stick_s1
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s_idx]
        
        # Apply Stage 2 Stickiness if we have visited this state before
        if prev_a2[s_idx] != -1:
            logits_2[prev_a2[s_idx]] += stick_s2
            
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        q_stage1_mf[a1] += lr * lambda_val * delta_stage2
        
        prev_a1 = a1
        prev_a2[s_idx] = a2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
