Here are the three cognitive models as Python functions.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Hyperplastic Transition Learning.
    
    Hypothesis: High OCI individuals have an unstable internal model of the environment's 
    transition structure. They update transition probabilities too rapidly based on 
    recent events (e.g., a Rare transition makes them think the structure has changed), 
    leading to MF-like behavior (Win-Stay on Rare) via a distorted Model-Based mechanism.
    
    Parameters:
    - learning_rate: [0, 1] Value learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - trans_lr_scale: [0, 5] Scaling factor for transition learning rate based on OCI.
      alpha_trans = learning_rate * trans_lr_scale * OCI.
    """
    learning_rate, beta, w, trans_lr_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Transition learning rate modulated by OCI
    # Bounded to [0,1] for safety
    alpha_trans = learning_rate * trans_lr_scale * oci_score
    alpha_trans = min(max(alpha_trans, 0.0), 1.0)

    # Initialize transition matrix (rows=actions, cols=states)
    # A(0) -> X(0) (0.7), U(1) -> Y(1) (0.7)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    for trial in range(n_trials):
        # Stage 1 Policy
        # MB Values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        
        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # MF Updates
        # Stage 1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # MB Transition Update (Hyperplasticity)
        # Target vector: 1 for the state visited, 0 otherwise
        target_trans = np.zeros(2)
        target_trans[s_idx] = 1.0
        
        # Update the row corresponding to action taken
        transition_matrix[a1, :] += alpha_trans * (target_trans - transition_matrix[a1, :])
        
        # Ensure normalization
        transition_matrix[a1, :] /= np.sum(transition_matrix[a1, :])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Counterfactual Regret.
    
    Hypothesis: Compulsivity is associated with rumination on forgone outcomes. 
    This model updates the unchosen Stage 1 action based on a "fictitious" outcome 
    (assuming zero-sum: if I won, other lost; if I lost, other won). 
    The strength of this counterfactual update is scaled by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Direct learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - regret_oci: [0, 5] Scaling factor for counterfactual update rate.
      alpha_cf = learning_rate * regret_oci * OCI.
    """
    learning_rate, beta, w, regret_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    alpha_cf = learning_rate * regret_oci * oci_score
    alpha_cf = min(max(alpha_cf, 0.0), 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        unchosen_a1 = 1 - a1
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Direct Update (Chosen)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Counterfactual Update (Unchosen)
        # Assume zero-sum: If R=1, R_cf=0. If R=0, R_cf=1.
        r_cf = 1.0 - r
        # We update unchosen MF value towards the counterfactual reward
        q_stage1_mf[unchosen_a1] += alpha_cf * (r_cf - q_stage1_mf[unchosen_a1])
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Frequency-Based Habit.
    
    Hypothesis: OCI correlates with the strength of long-term habits, defined as 
    an exponential moving average of past choices. This habit strength acts as a 
    bias on the decision, independent of current reward value, leading to perseveration.
    
    Parameters:
    - learning_rate: [0, 1] Value learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - habit_oci: [0, 5] Weight of the habit bias scaled by OCI.
    """
    learning_rate, beta, w, habit_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    habit_weight = habit_oci * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Habit trace for actions 0 and 1
    habit_trace = np.zeros(2) 
    
    for trial in range(n_trials):
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Habit Bias
        q_net_biased = q_net + habit_weight * habit_trace
        
        exp_q1 = np.exp(beta * q_net_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Update Values
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Habit Trace
        # Using learning_rate for decay to be parsimonious
        target_habit = np.zeros(2)
        target_habit[a1] = 1.0
        habit_trace += learning_rate * (target_habit - habit_trace)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
