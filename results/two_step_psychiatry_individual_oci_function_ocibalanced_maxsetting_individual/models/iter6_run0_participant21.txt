Here are the 3 cognitive models based on the participant's data and OCI score.

### Model 1: Stage-Specific Beta Model
This model hypothesizes that OCI symptoms specifically impact the precision (or randomness) of goal-directed decisions at Stage 2 (choosing an alien), distinct from the mixed strategic decisions at Stage 1 (choosing a spaceship). High OCI might lead to either rigid, deterministic choices (high beta) or anxious, noisy choices (low beta) specifically when faced with the immediate reward source.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage-Specific Beta Model.
    Separates exploration/exploitation (Beta) for Stage 1 (Spaceship) and Stage 2 (Alien).
    OCI modulates the Stage 2 Beta.
    
    Hypothesis: OCI affects the precision of value-based choices at the second stage
    (Alien choice) differently than the first stage (Spaceship choice).
    
    Parameters:
    lr: [0, 1] Learning rate for value updates.
    beta1: [0, 10] Inverse temperature for Stage 1 (Spaceship choice).
    beta2_base: [0, 10] Base inverse temperature for Stage 2 (Alien choice).
    beta2_oci: [-5, 5] OCI slope for Stage 2 Beta.
    w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    pers: [-2, 2] Perseveration bonus for Stage 1.
    """
    lr, beta1, beta2_base, beta2_oci, w, pers = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Stage 2 Beta based on OCI
    beta2 = beta2_base + beta2_oci * current_oci
    # Ensure beta2 stays non-negative (though softmax handles negatives, beta implies precision)
    beta2 = np.maximum(0.0, beta2)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_mf = np.zeros(2)       # Stage 1 MF values
    q2 = np.zeros((2, 2))    # Stage 2 values (State x Action)
    
    prev_action = -1
    log_lik = 0.0

    for t in range(n_trials):
        a1 = int(action_1[t])
        s = int(state[t])
        a2 = int(action_2[t])
        r = reward[t]

        # --- Stage 1 Policy ---
        # Model-Based value: expected max value of next state
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        
        # Net value
        q_net = w * q_mb + (1 - w) * q_mf
        
        # Perseveration
        if prev_action != -1:
            q_net[prev_action] += pers
            
        # Softmax Stage 1 (using beta1)
        exp_q1 = np.exp(beta1 * q_net)
        probs1 = exp_q1 / np.sum(exp_q1)
        log_lik += np.log(probs1[a1] + 1e-10)
        
        # --- Stage 2 Policy ---
        # Softmax Stage 2 (using OCI-modulated beta2)
        exp_q2 = np.exp(beta2 * q2[s])
        probs2 = exp_q2 / np.sum(exp_q2)
        log_lik += np.log(probs2[a2] + 1e-10)
        
        # --- Updates ---
        # SARSA(lambda) / TD(1) style update for Stage 1 MF
        # Use Stage 2 value to update Stage 1 MF
        delta1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += lr * delta1
        
        # Update Stage 2 value with Reward
        delta2 = r - q2[s, a2]
        q2[s, a2] += lr * delta2
        
        prev_action = a1

    return -log_lik
```

### Model 2: Compulsive Trace Model
This model captures the "streaks" observed in the participant's data (e.g., repeating the same spaceship for 15+ trials) using an Exponential Moving Average (EMA) of past choices. Unlike simple 1-back perseveration, this "trace" accumulates history. OCI modulates the weight of this trace, testing the hypothesis that high OCI leads to a "compulsive momentum" where it becomes progressively harder to switch strategies the longer one persists.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Compulsive Trace Model.
    Uses an Exponential Moving Average (EMA) of past choices to drive stickiness.
    OCI modulates the strength of this trace-based perseveration.
    
    Hypothesis: High OCI leads to stronger 'momentum' or compulsion to repeat choices,
    captured by a slowly decaying memory of past actions rather than just 1-back.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    decay: [0, 1] Decay rate for the choice trace (gamma). 
           (1 = no decay/infinite memory, 0 = instant forgetting).
    trace_w_base: [-2, 2] Base weight for choice trace.
    trace_w_oci: [-2, 2] OCI slope for trace weight.
    """
    lr, beta, w, decay, trace_w_base, trace_w_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    trace_w = trace_w_base + trace_w_oci * current_oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf = np.zeros(2)
    q2 = np.zeros((2, 2))
    
    # Choice trace for Stage 1 actions (A and U)
    # Initialize at 0
    choice_trace = np.zeros(2)
    
    log_lik = 0.0

    for t in range(n_trials):
        a1 = int(action_1[t])
        s = int(state[t])
        a2 = int(action_2[t])
        r = reward[t]

        # --- Stage 1 Policy ---
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        
        q_net = w * q_mb + (1 - w) * q_mf
        
        # Add Compulsive Trace Bonus
        q_net += trace_w * choice_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs1 = exp_q1 / np.sum(exp_q1)
        log_lik += np.log(probs1[a1] + 1e-10)
        
        # Update Choice Trace (EMA)
        # trace[a] = decay * trace[a] + (1-decay) * indicator
        # Update chosen action
        choice_trace[a1] = decay * choice_trace[a1] + (1 - decay) * 1.0
        # Update unchosen action
        choice_trace[1-a1] = decay * choice_trace[1-a1] + (1 - decay) * 0.0
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q2[s])
        probs2 = exp_q2 / np.sum(exp_q2)
        log_lik += np.log(probs2[a2] + 1e-10)
        
        # --- Updates ---
        delta1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += lr * delta1
        
        delta2 = r - q2[s, a2]
        q2[s, a2] += lr * delta2

    return -log_lik
```

### Model 3: Planet Stickiness Model
This model introduces a stickiness to the *outcome* (Planet) rather than the action. If a participant visited Planet X on the previous trial, they may have a bias to choose the spaceship that leads to Planet X again, regardless of whether that requires repeating the same action or switching (due to rare transitions). OCI modulates this "contextual rigidity."

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Planet Stickiness Model.
    Introduces a bias to choose the spaceship that leads to the *previously visited planet*,
    distinct from repeating the previous action. OCI modulates this 'outcome stickiness'.
    
    Hypothesis: OCI participants may be stuck on the *context* (Planet) they just visited,
    driving behavior to return there regardless of whether they switched actions to get there.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    pers: [-2, 2] Standard action perseveration (stickiness to spaceship).
    planet_stick_base: [-2, 2] Base bias to return to previous planet.
    planet_stick_oci: [-2, 2] OCI slope for planet stickiness.
    """
    lr, beta, w, pers, planet_stick_base, planet_stick_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    planet_stick = planet_stick_base + planet_stick_oci * current_oci
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_mf = np.zeros(2)
    q2 = np.zeros((2, 2))
    
    prev_action = -1
    prev_state = -1
    
    log_lik = 0.0

    for t in range(n_trials):
        a1 = int(action_1[t])
        s = int(state[t])
        a2 = int(action_2[t])
        r = reward[t]

        # --- Stage 1 Policy ---
        max_q2 = np.max(q2, axis=1)
        q_mb = transition_matrix @ max_q2
        
        q_net = w * q_mb + (1 - w) * q_mf
        
        # Standard Action Perseveration
        if prev_action != -1:
            q_net[prev_action] += pers
            
        # Planet Stickiness
        # If prev_state was 0 (Planet X), add bonus to Action 0 (A) which leads to X (0.7)
        # If prev_state was 1 (Planet Y), add bonus to Action 1 (U) which leads to Y (0.7)
        if prev_state != -1:
            # Action 0 commonly goes to State 0
            # Action 1 commonly goes to State 1
            # So the 'Return to Planet' action is simply the index of the planet
            target_action = prev_state 
            q_net[target_action] += planet_stick
        
        exp_q1 = np.exp(beta * q_net)
        probs1 = exp_q1 / np.sum(exp_q1)
        log_lik += np.log(probs1[a1] + 1e-10)
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q2[s])
        probs2 = exp_q2 / np.sum(exp_q2)
        log_lik += np.log(probs2[a2] + 1e-10)
        
        # --- Updates ---
        delta1 = q2[s, a2] - q_mf[a1]
        q_mf[a1] += lr * delta1
        
        delta2 = r - q2[s, a2]
        q2[s, a2] += lr * delta2
        
        prev_action = a1
        prev_state = s

    return -log_lik
```
