Here are three cognitive models expressed as Python functions. They incorporate the OCI-R score to explain individual variability in the two-step task.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-modulated Outcome Stickiness (Planet Stickiness).
    
    Hypothesis: High OCI individuals exhibit 'Outcome Stickiness'â€”a desire to return 
    to the previously visited state (Planet), distinct from simply repeating the previous 
    motor action. This biases the Stage 1 choice towards the spaceship most likely to 
    lead to the previous planet, regardless of whether the previous transition was rare or common.
    
    Parameters:
    - learning_rate: [0,1] Value updating rate.
    - beta: [0,10] Inverse temperature (softness of choice).
    - w: [0,1] Weighting parameter (0=Model-Free, 1=Model-Based).
    - stick_action: [0,5] Standard bonus for repeating the previous motor action (spaceship).
    - stick_outcome_scale: [0,5] Scaling factor for Outcome Stickiness based on OCI.
    """
    learning_rate, beta, w, stick_action, stick_outcome_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Outcome stickiness magnitude depends on OCI
    stick_outcome = stick_outcome_scale * oci_score

    # Transition matrix: Rows=Actions, Cols=States
    # T[0,:] = [0.7, 0.3] (Action 0 -> State 0 is common)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    prev_action_1 = -1
    prev_state = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Action Stickiness (Motor perseveration)
        if prev_action_1 != -1:
            q_net[int(prev_action_1)] += stick_action
            
        # Apply Outcome Stickiness (State perseveration)
        if prev_state != -1:
            # Add bonus to actions proportional to their probability of reaching prev_state
            # transition_matrix[:, prev_state] extracts P(prev_state | action) for all actions
            outcome_bonus = stick_outcome * transition_matrix[:, int(prev_state)]
            q_net += outcome_bonus

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            p_choice_1[trial] = probs_1[int(action_1[trial])]
            curr_a1 = int(action_1[trial])
        else:
            p_choice_1[trial] = 0.5
            curr_a1 = 0

        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if action_2[trial] != -1:
            p_choice_2[trial] = probs_2[int(action_2[trial])]
            curr_a2 = int(action_2[trial])
        else:
            p_choice_2[trial] = 0.5
            curr_a2 = 0

        # --- Updates ---
        # Stage 1 Update (SARSA-like)
        delta_stage1 = q_stage2_mf[state_idx, curr_a2] - q_stage1_mf[curr_a1]
        q_stage1_mf[curr_a1] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, curr_a2]
        q_stage2_mf[state_idx, curr_a2] += learning_rate * delta_stage2
        
        prev_action_1 = curr_a1
        prev_state = state_idx

    eps = 1e-10
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-modulated Stage 2 Precision (Beta 2).
    
    Hypothesis: OCI modulates the rigidity (inverse temperature) of the decision 
    process specifically in the second stage (Alien choice). High OCI leads to 
    more deterministic (higher beta) selection of aliens, reflecting a compulsion 
    to stick rigidly to the perceived 'best' alien.
    
    Parameters:
    - learning_rate: [0,1] Value updating rate.
    - beta_1: [0,10] Inverse temperature for Stage 1 (Spaceship).
    - beta_2_base: [0,10] Base inverse temperature for Stage 2 (Alien).
    - beta_2_oci_slope: [0,5] Increase in Stage 2 beta per unit of OCI.
    - w: [0,1] Model-Based weight.
    - stickiness: [0,5] Choice stickiness for Stage 1.
    """
    learning_rate, beta_1, beta_2_base, beta_2_oci_slope, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta for stage 2 based on OCI
    beta_2 = beta_2_base + beta_2_oci_slope * oci_score
    # Clamp to prevent numerical issues
    if beta_2 > 20: beta_2 = 20

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[int(prev_action_1)] += stickiness
            
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            p_choice_1[trial] = probs_1[int(action_1[trial])]
            curr_a1 = int(action_1[trial])
        else:
            p_choice_1[trial] = 0.5
            curr_a1 = 0

        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        # Use the OCI-modulated beta_2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if action_2[trial] != -1:
            p_choice_2[trial] = probs_2[int(action_2[trial])]
            curr_a2 = int(action_2[trial])
        else:
            p_choice_2[trial] = 0.5
            curr_a2 = 0

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, curr_a2] - q_stage1_mf[curr_a1]
        q_stage1_mf[curr_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, curr_a2]
        q_stage2_mf[state_idx, curr_a2] += learning_rate * delta_stage2
        
        prev_action_1 = curr_a1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-modulated Loss Reset.
    
    Hypothesis: High OCI individuals are intolerant of errors or punishments. 
    When a negative reward (loss) is encountered, they 'reset' (dampen) their 
    accumulated Model-Free values, reflecting a loss of confidence in their 
    current habits and potentially triggering re-exploration or reliance on MB.
    
    Parameters:
    - learning_rate: [0,1] Value updating rate.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Model-Based weight.
    - stickiness: [0,5] Choice stickiness.
    - loss_reset_scale: [0,1] Strength of value decay after a loss, scaled by OCI.
    """
    learning_rate, beta, w, stickiness, loss_reset_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate reset factor. If OCI is high, the reset (decay) is stronger.
    reset_factor = loss_reset_scale * oci_score
    if reset_factor > 1.0: reset_factor = 1.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_action_1 != -1:
            q_net[int(prev_action_1)] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        if action_1[trial] != -1:
            p_choice_1[trial] = probs_1[int(action_1[trial])]
            curr_a1 = int(action_1[trial])
        else:
            p_choice_1[trial] = 0.5
            curr_a1 = 0

        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        if action_2[trial] != -1:
            p_choice_2[trial] = probs_2[int(action_2[trial])]
            curr_a2 = int(action_2[trial])
        else:
            p_choice_2[trial] = 0.5
            curr_a2 = 0

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, curr_a2] - q_stage1_mf[curr_a1]
        q_stage1_mf[curr_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, curr_a2]
        q_stage2_mf[state_idx, curr_a2] += learning_rate * delta_stage2
        
        # --- OCI-Modulated Loss Reset ---
        # If punishment (-1) occurred, decay MF values
        if reward[trial] < 0:
            decay = 1.0 - reset_factor
            q_stage1_mf *= decay
            q_stage2_mf *= decay
            
        prev_action_1 = curr_a1

    eps = 1e-10
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials] + eps)) + np.sum(np.log(p_choice_2[valid_trials] + eps)))
    return log_loss
```
