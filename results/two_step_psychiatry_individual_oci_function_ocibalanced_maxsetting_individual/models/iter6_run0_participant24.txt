Here are three new cognitive models for the two-step task, incorporating the OCI score to explain individual variability in decision-making.

### Model 1: Epsilon-Greedy Exploration Modulated by OCI
This model hypothesizes that OCI symptoms relate to "checking" behaviors or uncertainty, modeled as a higher rate of random exploration (epsilon) that overrides value-based choices.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: MB/MF Hybrid with OCI-modulated Epsilon-Greedy Exploration (Lapse Rate).
    
    Hypothesis: Participants with higher OCI scores exhibit higher 'checking' behavior,
    manifesting as a higher rate of random choices (epsilon) regardless of learned values.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature for softmax (value sensitivity).
    - w: [0, 1] Weight for Model-Based control (1=MB, 0=MF).
    - eps_base: [0, 1] Baseline epsilon (randomness) for low OCI.
    - eps_oci_slope: [0, 1] Increase in epsilon per unit of OCI.
    """
    learning_rate, beta, w, eps_base, eps_oci_slope = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate epsilon based on OCI
    epsilon = eps_base + (eps_oci_slope * participant_oci)
    epsilon = np.clip(epsilon, 0.0, 1.0) 
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # Handle missing or invalid data
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]) or \
           action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax probabilities
        exp_q1 = np.exp(beta * q_net_s1)
        softmax_1 = exp_q1 / np.sum(exp_q1)
        
        # Mixture with Epsilon-Greedy (Random)
        # Prob = (1-eps)*Softmax + eps*0.5 (since there are 2 options)
        probs_1 = (1 - epsilon) * softmax_1 + epsilon * 0.5
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        # Pure MF for stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        softmax_2 = exp_q2 / np.sum(exp_q2)
        
        probs_2 = (1 - epsilon) * softmax_2 + epsilon * 0.5
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # TD(0) update for Stage 1 MF
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    # Filter 0 probabilities to avoid log(0)
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    
    valid_trials = (action_1 != -1) & (action_2 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials])) + np.sum(np.log(p_choice_2[valid_trials])))
    return log_loss
```

### Model 2: Choice Kernel (Habit) Modulated by OCI
This model distinguishes between simple one-step "stickiness" and a longer-term "habit" (Choice Kernel). It hypothesizes that OCI modulates the strength of this habit formation, where choices are repeated based on their frequency history, independent of reward.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: MB/MF Hybrid with OCI-modulated Choice Kernel (Habit Strength).
    
    Hypothesis: High OCI is associated with stronger habit formation (compulsivity).
    This is modeled as a 'Choice Kernel' that biases choices towards frequently chosen options
    (exponential moving average), independent of reward value. 
    
    Parameters:
    - learning_rate: [0, 1] Value learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - ck_lr: [0, 1] Learning rate for the choice kernel (how fast habits form/decay).
    - ck_w_base: [0, 5] Baseline weight of the choice kernel.
    - ck_w_oci_slope: [0, 5] Increase in kernel weight per unit of OCI.
    """
    learning_rate, beta, w, ck_lr, ck_w_base, ck_w_oci_slope = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate habit weight
    ck_weight = ck_w_base + (ck_w_oci_slope * participant_oci)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    choice_kernel = np.zeros(2) # Tracks choice history for Stage 1
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]) or \
           action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Habit bias (Choice Kernel) to the logits
        logits_1 = beta * q_net_s1 + ck_weight * choice_kernel
        
        # Softmax with stability fix
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) 
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Choice Kernel: Increase for chosen, decay for unchosen
        choice_kernel[a1] = choice_kernel[a1] + ck_lr * (1 - choice_kernel[a1])
        choice_kernel[1-a1] = choice_kernel[1-a1] + ck_lr * (0 - choice_kernel[1-a1])

    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    valid_trials = (action_1 != -1) & (action_2 != -1)
    
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials])) + np.sum(np.log(p_choice_2[valid_trials])))
    return log_loss
```

### Model 3: Counterfactual Updating Modulated by OCI
This model tests if OCI is linked to "counterfactual" thinkingâ€”updating the value of the option *not* chosen. It assumes participants might infer an anti-correlated reward structure (if Alien A gave 0, Alien B would have given 1).

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: MB/MF Hybrid with OCI-modulated Counterfactual Updating.
    
    Hypothesis: High OCI participants engage in more counterfactual thinking ('checking' the alternative).
    They update the value of the unchosen alien in Stage 2 assuming an anti-correlated reward structure
    (e.g., if chosen reward was 0, they learn that unchosen would have yielded 1).
    
    Parameters:
    - learning_rate: [0, 1] Direct learning rate for chosen option.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - cf_lr_base: [0, 1] Baseline counterfactual learning rate.
    - cf_lr_oci_slope: [0, 1] Increase in CF learning rate per unit OCI.
    """
    learning_rate, beta, w, cf_lr_base, cf_lr_oci_slope = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate counterfactual learning rate
    cf_lr = cf_lr_base + (cf_lr_oci_slope * participant_oci)
    cf_lr = np.clip(cf_lr, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        if np.isnan(action_1[trial]) or np.isnan(action_2[trial]) or \
           action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Direct update for chosen alien
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Counterfactual update for unchosen alien
        # Assume anti-correlated: if r=1, unchosen=0; if r=0, unchosen=1.
        unchosen_a2 = 1 - a2
        cf_target = 1.0 - r
        delta_cf = cf_target - q_stage2_mf[s_idx, unchosen_a2]
        q_stage2_mf[s_idx, unchosen_a2] += cf_lr * delta_cf
        
    eps = 1e-10
    p_choice_1 = np.maximum(p_choice_1, eps)
    p_choice_2 = np.maximum(p_choice_2, eps)
    valid_trials = (action_1 != -1) & (action_2 != -1)
    
    log_loss = -(np.sum(np.log(p_choice_1[valid_trials])) + np.sum(np.log(p_choice_2[valid_trials])))
    return log_loss
```
