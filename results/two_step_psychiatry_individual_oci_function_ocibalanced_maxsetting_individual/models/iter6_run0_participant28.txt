Here are the three cognitive models as Python functions.

import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid (MB/MF) learner with OCI-modulated asymmetric learning rates.
    
    This model tests if OCI symptoms bias the Model-Free component of the hybrid 
    learner to learn more strongly from negative (or positive) prediction errors.
    
    Parameters:
    lr_base: [0, 1] Base learning rate for positive prediction errors.
    beta: [0, 10] Inverse temperature (softmax randomness).
    w: [0, 1] Mixing weight between MB (1) and MF (0) systems.
    oci_neg_bias: [0, 1] Modulates learning rate for negative RPEs based on OCI.
                       Higher value -> OCI increases learning rate for negative outcomes.
    """
    lr_base, beta, w, oci_neg_bias = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate asymmetric learning rates
    # lr_pos is fixed at base
    # lr_neg is modulated by OCI. 
    # We define a ratio such that higher OCI amplifies the learning from negative errors.
    ratio = 1.0 + oci_neg_bias * (oci_score * 3.0) 
    lr_pos = lr_base
    lr_neg = np.clip(lr_base * ratio, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mixed Value (Hybrid)
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Updates ---
        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        cur_lr = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2_mf[s_idx, a2] += cur_lr * delta_stage2
        
        # Stage 1 RPE (TD(0) update for MF component)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        cur_lr_1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1_mf[a1] += cur_lr_1 * delta_stage1
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid (MB/MF) learner with OCI-modulated learning rate decay.
    
    This model tests the hypothesis that high OCI scores correlate with 
    cognitive rigidity or habituation, modeled as a learning rate that 
    decays (anneals) faster over trials for individuals with higher OCI.
    
    Parameters:
    lr_init: [0, 1] Initial learning rate at trial 0.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (0=MF, 1=MB).
    decay_sens: [0, 1] Sensitivity of decay to OCI. 
                       Higher value -> OCI causes faster reduction of learning rate over time.
    """
    lr_init, beta, w, decay_sens = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # Time-dependent learning rate
        # Annealing schedule: lr = lr0 / (1 + k * t)
        # The decay factor 'k' is modulated by OCI
        k = decay_sens * oci_score * 0.1 # Scale factor to keep decay reasonable
        current_lr = lr_init / (1.0 + k * trial)
        
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # --- Stage 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Updates ---
        # Standard TD updates with decaying learning rate
        q_stage2_mf[s_idx, a2] += current_lr * (r - q_stage2_mf[s_idx, a2])
        q_stage1_mf[a1] += current_lr * (q_stage2_mf[s_idx, a2] - q_stage1_mf[a1])
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid (MB/MF) learner with independent beta parameters for MB and MF systems,
    where OCI modulates the Model-Based beta specifically.
    
    Unlike a mixing weight 'w' which trades off one system for another, this model
    allows the Model-Based system to become more (or less) deterministic/precise 
    independently, based on OCI symptoms.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta_mf: [0, 10] Inverse temperature for Model-Free values.
    beta_mb_base: [0, 10] Base Inverse temperature for Model-Based values.
    oci_mb_mod: [0, 1] Modulates beta_mb based on OCI. 
                       Higher value -> OCI increases MB precision (rigidity).
    """
    lr, beta_mf, beta_mb_base, oci_mb_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate OCI-modulated MB beta
    # We assume OCI scales the base MB beta.
    # oci_score is ~0 to 1. 
    beta_mb = beta_mb_base * (1.0 + oci_mb_mod * (oci_score * 2.0))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    for trial in range(n_trials):
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Independent logits summation
        # This replaces the 'w' parameterization.
        logits_1 = beta_mf * q_stage1_mf + beta_mb * q_stage1_mb
        
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # --- Stage 2 ---
        # Stage 2 is purely Model-Free in this task structure
        # We use the MF beta here for consistency
        exp_q2 = np.exp(beta_mf * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Updates ---
        q_stage2_mf[s_idx, a2] += lr * (r - q_stage2_mf[s_idx, a2])
        q_stage1_mf[a1] += lr * (q_stage2_mf[s_idx, a2] - q_stage1_mf[a1])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
