Here are three cognitive models represented as Python functions.

### Model 1: Hybrid MB/MF with OCI-Modulated Win-Stay Lose-Shift (WSLS)
This model hypothesizes that high OCI scores correlate with a strong "corrective" or "perfectionistic" drive, manifesting as a Win-Stay Lose-Shift heuristic that overrides or biases the standard value-based learning. The strength of this heuristic is scaled by the participant's OCI score.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Modulated Win-Stay Lose-Shift (WSLS).
    
    Hypothesis: High OCI individuals exhibit strong outcome-dependent choice persistence.
    They receive a choice bonus for repeating the previous Stage 1 action if it was rewarded,
    and a penalty (encouraging switching) if it was unrewarded. This heuristic bias
    is added to the net Q-values and is proportional to OCI severity.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - wsls_scale: [0, 5] Scaling factor for the WSLS bias. 
      Bonus = wsls_scale * OCI * (1 if Reward else -1).
    """
    learning_rate, beta, w, wsls_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Transition matrix for MB (assumed fixed and known)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10
    
    prev_a1 = -1
    prev_reward = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            prev_a1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply OCI-modulated WSLS bias
        if prev_a1 != -1:
            # If previous reward was 1, bias towards staying (add to prev_a1)
            # If previous reward was 0, bias towards switching (subtract from prev_a1)
            direction = 1.0 if prev_reward == 1.0 else -1.0
            bias = wsls_scale * oci_score * direction
            q_net[prev_a1] += bias

        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # Update Stage 2 MF
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF (TD(0))
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store history
        prev_a1 = a1
        prev_reward = r

    return log_loss
```

### Model 2: Hybrid MB/MF with OCI-Modulated Stage 2 Learning Rate
This model hypothesizes that OCI severity correlates with hyper-sensitivity to immediate outcomes. While the abstract spaceship values (Stage 1) are updated normally, the concrete alien probabilities (Stage 2) are updated more aggressively by high-OCI participants, leading to volatility in the underlying model of the environment.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Modulated Stage 2 Learning Rate.
    
    Hypothesis: High OCI leads to obsessive updating of immediate reward probabilities 
    (Stage 2), making the participant more reactive to the most recent alien outcome, 
    while Stage 1 updating remains stable.
    
    Parameters:
    - lr_base: [0, 1] Base learning rate for Stage 1 updates.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - lr2_multiplier: [0, 5] Multiplier for Stage 2 learning rate based on OCI.
      LR_Stage2 = lr_base * (1 + lr2_multiplier * OCI). Clipped at 1.0.
    """
    lr_base, beta, w, lr2_multiplier = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate OCI-dependent Stage 2 learning rate
    lr_stage2 = lr_base * (1.0 + lr2_multiplier * oci_score)
    if lr_stage2 > 1.0:
        lr_stage2 = 1.0
        
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        # Update Stage 2 with boosted learning rate
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_stage2 * delta_stage2
        
        # Update Stage 1 with base learning rate
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_base * delta_stage1

    return log_loss
```

### Model 3: Hybrid MB/MF with OCI-Modulated Transition Belief
This model hypothesizes that high OCI is associated with rigidity and a need for certainty. It proposes that high-OCI participants perceive the probabilistic spaceship transitions (70/30) as more deterministic than they are. The Model-Based system uses a distorted transition matrix where the "common" probability is pushed towards 100% proportional to the OCI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Modulated Transition Belief.
    
    Hypothesis: High OCI individuals perceive the environment as more deterministic 
    (rigid) than it is. The Model-Based system uses a subjective transition matrix 
    where the probability of the common transition is increased by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Update rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - belief_scale: [0, 1] Distortion factor. 
      Subjective_P_common = 0.7 + (0.3 * belief_scale * OCI).
      If scale is 0, belief is correct (0.7). If scale is 1 and OCI is 1, belief is 1.0.
    """
    learning_rate, beta, w, belief_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate subjective transition probability
    # Base is 0.7. Max distortion pushes it to 1.0.
    p_common_subjective = 0.7 + (0.3 * belief_scale * oci_score)
    if p_common_subjective > 1.0:
        p_common_subjective = 1.0
    p_rare_subjective = 1.0 - p_common_subjective
    
    # Subjective Transition Matrix
    # A -> X (common), U -> Y (common)
    # Row 0 (A): [P(X|A), P(Y|A)] = [p_common, p_rare]
    # Row 1 (U): [P(X|U), P(Y|U)] = [p_rare, p_common]
    transition_matrix = np.array([
        [p_common_subjective, p_rare_subjective],
        [p_rare_subjective, p_common_subjective]
    ])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        # MB calculation uses the DISTORTED transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)
        
        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)
        
        # --- Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    return log_loss
```
