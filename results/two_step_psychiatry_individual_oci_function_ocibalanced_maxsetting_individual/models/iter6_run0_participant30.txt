Here are 3 new cognitive models that explore different mechanisms for how OCI symptoms might influence decision-making in this two-step task, specifically focusing on perseveration, model-based/model-free arbitration, and learning rates.

### Model 1: OCI-Modulated Perseveration (Stickiness)
This model hypothesizes that high OCI scores lead to increased "stickiness" or perseverationâ€”a tendency to repeat the previous choice regardless of reward. This reflects the repetitive nature of compulsive behaviors. The `stickiness` parameter is added to the Q-values of the previously chosen action, and this effect is modulated by the OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Perseveration (Stickiness).
    
    Hypothesis: High OCI scores are associated with increased choice perseveration 
    (stickiness), meaning the participant is more likely to repeat their previous 
    choice regardless of the outcome.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] - Weighting parameter (0 = MF, 1 = MB)
    stickiness_base: [0, 5] - Base tendency to repeat choices
    stickiness_oci_slope: [-2, 2] - How OCI modulates stickiness
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness based on OCI
    stickiness = stickiness_base + (stickiness_oci_slope * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    prev_choice_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MF and MB values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the previously chosen action
        if prev_choice_1 != -1:
            q_net[prev_choice_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        if r != -1:
            # Stage 2 Update (TD)
            delta_stage2 = r - q_stage2_mf[s, a2]
            q_stage2_mf[s, a2] += learning_rate * delta_stage2
            
            # Stage 1 Update (TD)
            delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
            q_stage1_mf[a1] += learning_rate * delta_stage1
            
            # Update previous choice
            prev_choice_1 = a1
        else:
            # If trial is missing/invalid, reset stickiness for safety or keep it? 
            # Usually we just skip updates.
            prev_choice_1 = -1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Balanced Max Setting
This model investigates if OCI affects how participants arbitrate between Model-Based (MB) and Model-Free (MF) control specifically through the `w` parameter. Instead of a linear modulation, it uses a sigmoid-like balance where `w` is determined by a base level and an OCI-dependent shift. This captures the hypothesis that high OCI might rigidly lock a participant into a specific strategy (e.g., purely habitual/MF or over-deliberative/MB).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Balanced Max Setting.
    
    Hypothesis: OCI determines the balance (w) between Model-Based and Model-Free 
    systems. The parameter w is derived directly from OCI using a scaled logistic 
    function to ensure it stays within [0, 1], allowing the model to find if 
    high OCI correlates with extreme MB or MF usage.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_bias: [-5, 5] - Bias term for the sigmoid
    w_slope: [-5, 5] - Slope of OCI effect on w
    lambd: [0, 1] - Eligibility trace parameter (connecting stage 2 to stage 1)
    """
    learning_rate, beta, w_bias, w_slope, lambd = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Sigmoid function for w to keep it in [0, 1]
    # w = 1 / (1 + exp(-(bias + slope * oci)))
    w_logit = w_bias + (w_slope * oci_score)
    w = 1.0 / (1.0 + np.exp(-w_logit))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        if r != -1:
            # Stage 2 Update
            delta_stage2 = r - q_stage2_mf[s, a2]
            q_stage2_mf[s, a2] += learning_rate * delta_stage2
            
            # Stage 1 Update using eligibility trace (lambda)
            # This allows the Stage 2 RPE to directly affect Stage 1 values
            q_stage1_mf[a1] += learning_rate * (lambda_param * delta_stage2 + (1 - lambda_param) * (q_stage2_mf[s, a2] - q_stage1_mf[a1]))
            # Note: simplified update logic to fit within param constraints and standard SARSA(lambda) approx
            # Actually, let's implement the standard TD(lambda) update for stage 1:
            # delta_1 = Q2(s, a2) - Q1(a1)
            # Q1(a1) += lr * delta_1
            # Q1(a1) += lr * lambd * delta_2
            
            delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
            q_stage1_mf[a1] += learning_rate * delta_stage1
            q_stage1_mf[a1] += learning_rate * lambd * delta_stage2 # Eligibility trace update

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Driven Asymmetric Learning Rates
This model tests the hypothesis that OCI affects how participants learn from positive versus negative prediction errors. Individuals with high OCI might be hyper-sensitive to errors (negative feedback) or, conversely, overly driven by rewards. The model splits the learning rate into `alpha_pos` and `alpha_neg`, where the balance between them is modulated by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Driven Asymmetric Learning Rates.
    
    Hypothesis: OCI modulates the ratio between learning from positive prediction 
    errors (better than expected) and negative prediction errors (worse than expected).
    High OCI might lead to a "negativity bias" or specific rigidity in updating values.
    
    Bounds:
    lr_base: [0, 1] - Base learning rate magnitude
    lr_asym_oci: [-1, 1] - How OCI skews learning towards positive (+1) or negative (-1)
    beta: [0, 10]
    w: [0, 1]
    """
    lr_base, lr_asym_oci, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate asymmetric learning rates
    # bias factor ranges from e.g. 0.5 to 1.5 based on OCI
    bias = lr_asym_oci * oci_score 
    
    # Ensure rates stay in [0, 1]
    lr_pos = np.clip(lr_base * (1 + bias), 0, 1)
    lr_neg = np.clip(lr_base * (1 - bias), 0, 1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        if r != -1:
            # Stage 2 Update
            delta_stage2 = r - q_stage2_mf[s, a2]
            lr_2 = lr_pos if delta_stage2 > 0 else lr_neg
            q_stage2_mf[s, a2] += lr_2 * delta_stage2
            
            # Stage 1 Update
            delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
            lr_1 = lr_pos if delta_stage1 > 0 else lr_neg
            q_stage1_mf[a1] += lr_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
