Here are three new cognitive models that incorporate the OCI-R score into the decision-making process for the two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Balance
This model hypothesizes that high OCI scores are associated with a rigid reliance on habitual (Model-Free) control, making it harder to engage in flexible, goal-directed (Model-Based) planning. The parameter `w` (mixing weight) is dynamically adjusted based on the OCI score. Specifically, higher OCI reduces the contribution of the model-based system.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated MB/MF Balance.
    Hypothesis: High OCI participants exhibit reduced model-based control (goal-directed behavior)
    and increased reliance on model-free (habitual) control. The mixing weight 'w' is 
    down-regulated by the OCI score.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_max: [0, 1] (Maximum possible model-based weight)
    oci_suppression: [0, 1] (Strength of OCI's negative effect on w)
    """
    learning_rate, beta, w_max, oci_suppression = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Calculate effective w based on OCI
    # Higher OCI reduces w, pushing behavior towards Model-Free
    w = w_max * (1.0 - (oci_score * oci_suppression))
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: a1 = 0
        s_idx = int(state[trial])
        if s_idx == -1: s_idx = 0
        a2 = int(action_2[trial])
        if a2 == -1: a2 = 0
        r = reward[trial]
        if r == -1: r = 0

        # Policy for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted mixture of MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Policy for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Value Updates
        # Stage 2 update (Standard Q-learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # Stage 1 update (TD-learning using Stage 2 value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Enhanced Perseveration (Stickiness)
This model posits that OCI is linked to behavioral rigidity or "stickiness"â€”a tendency to repeat the previous choice regardless of reward outcome. Here, the OCI score scales a stickiness parameter that biases the Stage 1 choice towards the previously chosen spaceship.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Enhanced Perseveration.
    Hypothesis: Higher OCI scores lead to increased choice perseveration (stickiness).
    The model adds a 'stickiness bonus' to the Q-value of the previously chosen action,
    scaled by the OCI score.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    stick_base: [0, 5] (Base stickiness magnitude)
    """
    learning_rate, beta, w, stick_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Stickiness magnitude is modulated by OCI
    stickiness_bonus = stick_base * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: a1 = 0
        s_idx = int(state[trial])
        if s_idx == -1: s_idx = 0
        a2 = int(action_2[trial])
        if a2 == -1: a2 = 0
        r = reward[trial]
        if r == -1: r = 0

        # Policy for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the previously chosen action
        q_net_stick = q_net.copy()
        if last_action_1 != -1:
            q_net_stick[last_action_1] += stickiness_bonus
        
        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Policy for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Value Updates
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Dependent Learning Rate Asymmetry
This model suggests that individuals with high OCI might learn differently from positive versus negative prediction errors. Specifically, high OCI might lead to "hyper-learning" from negative outcomes (punishment sensitivity) or rigid value updating. Here, we split the learning rate into positive (`alpha_pos`) and negative (`alpha_neg`) components, where the negative learning rate is amplified by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Dependent Learning Rate Asymmetry.
    Hypothesis: High OCI is associated with altered sensitivity to prediction errors.
    This model implements dual learning rates, where the learning rate for negative
    prediction errors (alpha_neg) is scaled up by the OCI score, reflecting increased
    sensitivity to worse-than-expected outcomes.
    
    Bounds:
    alpha_pos: [0, 1] (Learning rate for positive RPEs)
    alpha_neg_base: [0, 1] (Base learning rate for negative RPEs)
    beta: [0, 10]
    w: [0, 1]
    """
    alpha_pos, alpha_neg_base, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # OCI amplifies the learning rate for negative prediction errors
    # We clamp it to max 1.0 to ensure stability
    alpha_neg = alpha_neg_base * (1.0 + oci_score)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        if a1 == -1: a1 = 0
        s_idx = int(state[trial])
        if s_idx == -1: s_idx = 0
        a2 = int(action_2[trial])
        if a2 == -1: a2 = 0
        r = reward[trial]
        if r == -1: r = 0

        # Policy Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # Policy Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # Update Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        lr_s2 = alpha_pos if delta_stage2 >= 0 else alpha_neg
        q_stage2_mf[s_idx, a2] += lr_s2 * delta_stage2

        # Update Stage 1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        lr_s1 = alpha_pos if delta_stage1 >= 0 else alpha_neg
        q_stage1_mf[a1] += lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
