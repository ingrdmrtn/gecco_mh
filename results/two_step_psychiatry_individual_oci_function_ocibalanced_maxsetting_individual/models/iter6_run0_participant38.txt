Here are three new cognitive models for the two-step task, incorporating OCI scores to explain individual variability in decision-making.

### Model 1: Accumulating Habit Trace Model
This model hypothesizes that high OCI scores correlate with the formation of strong, persistent habits. Unlike simple "stickiness" (which usually repeats the *last* choice), this model maintains an exponentially decaying "habit trace" that integrates the history of past choices. High OCI participants are modeled as having a stronger influence of this historical trace on their current decision, leading to more rigid behavior patterns that are harder to break.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Accumulating Habit Trace Model.
    
    Hypothesis: OCI is associated with the formation of strong, persistent habits.
    Unlike simple 1-step stickiness, this model tracks an exponentially decaying 
    habit trace for each spaceship action. The influence of this habit trace on 
    choice is scaled by the OCI score. High OCI -> Stronger influence of past history.

    Parameters:
    learning_rate: [0, 1] - Learning rate for Q-values.
    beta: [0, 10] - Inverse temperature (softmax sensitivity).
    w: [0, 1] - Weight of Model-Based system (0=MF, 1=MB).
    lambda_decay: [0, 1] - Eligibility trace decay parameter.
    habit_decay: [0, 1] - Decay rate of the habit trace (persistence).
    habit_weight_oci: [0, 5] - Weight of the habit trace in decision, scaled by OCI.
    """
    learning_rate, beta, w, lambda_decay, habit_decay, habit_weight_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    habit_weight = habit_weight_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Habit trace for the two stage 1 actions
    habit_trace = np.zeros(2)

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # 1. Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB, MF, and accumulated Habit Trace
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + habit_weight * habit_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update Habit Trace
        # Decay existing trace (persistence)
        habit_trace *= habit_decay
        # Strengthen chosen action (reinforcement of habit)
        habit_trace[a1] += 1.0

        # 2. Stage 2 Policy
        qs_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # 3. Learning
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_decay * delta_stage2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    mask = action_1 != -1
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 2: Punishment Sensitivity Model
This model tests the hypothesis that high OCI participants perceive the absence of reward (0 outcomes) not as a neutral event, but as an active punishment (fear of failure/error). The model transforms 0-rewards into negative values, with the magnitude of this negativity scaled by the OCI score. This causes the Q-values for unrewarded options to drift negative, promoting active avoidance or switching, rather than just gradual extinction.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Punishment Sensitivity Model.
    
    Hypothesis: High OCI participants perceive the absence of reward (0 outcomes) 
    not as neutral, but as aversive (punishment). This model transforms 0 rewards 
    into negative values scaled by OCI. This drives Q-values negative and promotes 
    avoidance or switching via the Q-learning mechanism, rather than just extinction.

    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Weight of Model-Based system.
    lambda_decay: [0, 1] - Eligibility trace decay.
    punishment_scale_oci: [0, 5] - Magnitude of negative reward for 0 outcomes, scaled by OCI.
    """
    learning_rate, beta, w, lambda_decay, punishment_scale_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Transform 0 reward into a negative value based on OCI
    punishment_val = -1.0 * punishment_scale_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r_raw = reward[trial]
        
        # Apply transformation: if reward is 0, it becomes punishment_val
        r_eff = r_raw if r_raw > 0 else punishment_val

        # 1. Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # 2. Stage 2 Policy
        qs_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # 3. Learning
        # Use r_eff for updates. This propagates negative values to MF and MB systems.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r_eff - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_decay * delta_stage2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    mask = action_1 != -1
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Model 3: State-Directed Stickiness Model (Planet Compulsion)
This model proposes that the "compulsive" aspect of OCI is directed at the *state* (Planet) rather than the motor action (Spaceship). It hypothesizes that high OCI participants have a compulsion to revisit the location they just checked (or perhaps avoid switching locations). This is modeled by adding a stickiness bonus to the *value of the previously visited planet*. The Model-Based system then propagates this preference back to the Stage 1 choice based on the transition probabilities.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    State-Directed Stickiness Model (Planet Compulsion).
    
    Hypothesis: OCI-related stickiness is not just about repeating motor actions (spaceships),
    but about compulsively checking specific states (planets). This model adds a "stickiness" 
    bonus to the value of the previously visited planet. This bonus is integrated into the 
    Stage 1 decision via the Model-Based system (transition matrix).
    
    Parameters:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    lambda_decay: [0, 1]
    planet_stick_oci: [0, 5] - Bonus added to the previously visited planet, scaled by OCI.
    """
    learning_rate, beta, w, lambda_decay, planet_stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    planet_stick_mag = planet_stick_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_planet = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # 1. Stage 1 Policy
        # Add stickiness to the value of the planets (states) for the MB calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        state_utilities = max_q_stage2.copy()
        if prev_planet != -1:
            state_utilities[prev_planet] += planet_stick_mag
            
        # MB value is transition * state_utilities (including stickiness)
        q_stage1_mb = transition_matrix @ state_utilities
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        prev_planet = s_idx

        # 2. Stage 2 Policy
        qs_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # 3. Learning
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_decay * delta_stage2
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    mask = action_1 != -1
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
