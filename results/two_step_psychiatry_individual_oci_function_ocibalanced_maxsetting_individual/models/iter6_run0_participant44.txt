Here are the three cognitive models implemented as Python functions.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Risk-Sensitive Model-Based Planning.

    Hypothesis: High OCI participants exhibit anxiety-driven risk aversion. 
    In the Model-Based calculation, instead of assuming optimal performance at the second stage (max Q-value),
    they incorporate a 'worry' component (min Q-value) scaled by their OCI score.
    This reflects a fear of worst-case outcomes or distrust in their ability to choose the best alien,
    shifting the planning value towards the minimum potential reward of a state.

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature (softness of choice).
    - w: [0, 1] Weighting between Model-Based and Model-Free systems.
    - lambd: [0, 1] Eligibility trace for stage 1 update.
    - risk_sens: [0, 1] Sensitivity to risk/worst-case scenarios (scales with OCI).
    """
    learning_rate, beta, w, lambd, risk_sens = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Risk parameter: 0 = Optimistic (Max), 1 = Pessimistic (Min)
    # The degree of pessimism is determined by the OCI score and the risk_sens parameter.
    rho = risk_sens * oci_score

    for trial in range(n_trials):
        # Stage 1 Policy
        # MB Valuation: Weighted average of best-case (max) and worst-case (min) at Stage 2
        max_q2 = np.max(q_stage2_mf, axis=1)
        min_q2 = np.min(q_stage2_mf, axis=1)
        
        # Anxious planning mixes the max and min Q-values
        val_stage2 = (1.0 - rho) * max_q2 + rho * min_q2
        
        q_stage1_mb = transition_matrix @ val_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Compulsive Perseveration Model (OCI-Modulated Trace Decay).

    Hypothesis: High OCI leads to "stickier" habits due to slower decay of choice traces.
    A choice trace accumulates when an action is chosen and decays over time.
    The decay rate is modulated by OCI: higher OCI results in slower decay (higher retention of the trace),
    leading to compulsive repetition of actions regardless of reward history.

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - lambd: [0, 1] Eligibility trace.
    - stick_weight: [0, 5] Weight of the choice trace in the decision (habit strength).
    - decay_base: [0, 1] Base decay rate for low OCI.
    """
    learning_rate, beta, w, lambd, stick_weight, decay_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    choice_trace = np.zeros(2)
    
    # Effective decay: Higher OCI -> Decay closer to 1 (slower forgetting of the habit)
    # If OCI is 1, decay approaches 1.0 (permanent trace). If OCI is 0, decay is decay_base.
    decay_eff = decay_base + (1.0 - decay_base) * oci_score

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add stickiness bonus based on choice trace
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + stick_weight * choice_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        
        # Update Trace: Decay existing trace, then increment for chosen action
        choice_trace *= decay_eff
        choice_trace[a1] += 1.0

        # Stage 2 Policy
        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Surprise Obsession Model (OCI-Modulated Rare Transition Learning).

    Hypothesis: High OCI participants over-react to "surprising" (rare) transitions,
    possibly perceiving them as structural changes or significant errors.
    This is modeled as a boost to the learning rate specifically on trials where a rare transition occurs,
    scaled by the OCI score. This leads to rapid switching or updating after rare events.

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - lambd: [0, 1] Eligibility trace.
    - surprise_boost: [0, 5] Multiplier for learning rate on rare transitions (scaled by OCI).
    """
    learning_rate, beta, w, lambd, surprise_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        qs2 = q_stage2_mf[state_idx]
        exp_q2 = np.exp(beta * qs2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Determine if transition was rare
        # Common: 0->0, 1->1. Rare: 0->1, 1->0.
        is_rare = False
        if (a1 == 1 and state_idx == 0) or (a1 == 0 and state_idx == 1):
            is_rare = True
            
        # Modulate Learning Rate
        current_lr = learning_rate
        if is_rare:
            # Boost LR based on OCI, capping at 1.0 for stability
            boost_factor = 1.0 + oci_score * surprise_boost
            current_lr = min(1.0, learning_rate * boost_factor)

        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += current_lr * delta_stage2
        q_stage1_mf[a1] += current_lr * lambd * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
