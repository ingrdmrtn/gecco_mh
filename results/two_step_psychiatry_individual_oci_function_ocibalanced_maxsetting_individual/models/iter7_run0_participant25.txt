Here are three new cognitive models that explore different mechanisms for how OCI scores might modulate decision-making in this two-step task.

### Model 1: Hybrid Learner with OCI-Modulated Model-Based Weight (`w`)
This model hypothesizes that individuals with different OCI scores rely differently on Model-Based (planning) versus Model-Free (habitual) systems. Specifically, it tests if higher OCI scores lead to a shift in the balance parameter `w`. The model calculates a weighted average of Model-Based and Model-Free values for the first-stage choice.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the weighting parameter 'w'
    is modulated by the participant's OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature (softmax sensitivity).
    - w_base: [0, 1] Base weight for Model-Based system (0=Pure MF, 1=Pure MB).
    - oci_w_mod: [-1, 1] Modulation of 'w' by OCI score.
    """
    learning_rate, beta, w_base, oci_w_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective w, bounded between 0 and 1
    w = w_base + (oci_score * oci_w_mod)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-Free Stage 1 values
    q_stage2_mf = np.zeros((2, 2)) # Model-Free Stage 2 values

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based valuation: Q_MB(s1, a1) = Sum(P(s2|s1,a1) * max(Q_MF(s2, a2)))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid valuation: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Updates ---
        # SARSA-like update for Stage 1 MF
        # Note: Standard two-step usually updates Q_MF(s1) using Q_MF(s2) (TD(0)) or Reward (TD(1))
        # Here we use a simple TD(1) style update directly from reward for simplicity in the hybrid frame
        # or the standard TD(0) chain. Let's use TD(1) equivalent logic often used in simplified hybrid models:
        # q_stage1_mf[a1] += lr * (r - q_stage1_mf[a1])  <-- simple direct reinforcement
        # But to fit the template structure asking for delta_stage1:
        
        # Update Stage 1 MF (using the stage 2 value as prediction, TD(0) logic)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Additional eligibility trace update for Stage 1 based on Stage 2 RPE (TD(1) aspect)
        # This connects the final reward back to the first choice in the MF system
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free Learner with OCI-Modulated Choice Stickiness
This model posits that OCI symptoms manifest as a tendency to repeat previous choices (compulsivity/stickiness), regardless of reward outcomes. Unlike the previous best model, this model separates the stickiness into a base parameter and an OCI-scaling factor, and applies it purely within a Model-Free framework without a Model-Based component, to isolate this perseverative effect.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with choice perseveration (stickiness) on the first stage,
    where the degree of stickiness is modulated by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - stick_base: [-5, 5] Base stickiness parameter.
    - oci_stick_scale: [-10, 10] Scaling of stickiness by OCI score.
    """
    learning_rate, beta, stick_base, oci_stick_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness
    stickiness = stick_base + (oci_score * oci_stick_scale)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        # Add stickiness bonus to the previously chosen action
        q_net = q_stage1_mf.copy()
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # --- Updates ---
        # TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # TD(1) contribution: propagate reward error back to stage 1
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Asymmetric Learning Rates Modulated by OCI
This model investigates if OCI scores relate to an asymmetry in learning from positive versus negative prediction errors (e.g., being hyper-sensitive to missed rewards or failures). It defines separate learning rates for positive and negative updates, with the negative learning rate specifically modulated by the OCI score.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with asymmetric learning rates for positive and negative
    prediction errors. The negative learning rate is modulated by OCI.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for positive prediction errors.
    - alpha_neg_base: [0, 1] Base learning rate for negative prediction errors.
    - oci_alpha_neg_mod: [-1, 1] Modulation of negative learning rate by OCI.
    - beta: [0, 10] Inverse temperature.
    """
    alpha_pos, alpha_neg_base, oci_alpha_neg_mod, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate, bounded [0, 1]
    alpha_neg = alpha_neg_base + (oci_score * oci_alpha_neg_mod)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]
        
        # --- Updates ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        lr_2 = alpha_pos if delta_stage2 > 0 else alpha_neg
        q_stage2_mf[state_idx, a2] += lr_2 * delta_stage2
        
        # Stage 1 Update
        # We use a direct reward-based update (TD(1)-like) to capture the full outcome 
        # sensitivity, as is common in simple asymmetric models.
        delta_total = r - q_stage1_mf[a1]
        lr_1 = alpha_pos if delta_total > 0 else alpha_neg
        q_stage1_mf[a1] += lr_1 * delta_total

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
