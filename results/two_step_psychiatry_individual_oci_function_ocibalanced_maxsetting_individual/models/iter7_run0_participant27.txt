import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with Differential Stickiness for Win/Loss, modulated by OCI.
    
    Hypothesis: The participant exhibits 'compulsive' stickiness, persisting in choices 
    even after a lack of reward (loss). This model splits stickiness into 'Win-Stay' 
    and 'Lose-Stay' components, where OCI specifically exacerbates the tendency to 
    stick with a choice after a loss (or zero reward).
    
    Parameters:
    lr: [0,1] - Learning rate for both stages.
    beta: [0,10] - Inverse temperature (exploration/exploitation).
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    lam: [0,1] - Eligibility trace (credit assignment to Stage 1).
    stick_win: [0,5] - Stickiness added if the previous trial was rewarded.
    stick_loss_base: [0,5] - Base stickiness added if the previous trial was unrewarded, scaled by OCI.
    """
    lr, beta, w, lam, stick_win, stick_loss_base = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI increases the tendency to stick after a loss/no-reward
    stick_loss = stick_loss_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)      
    q_stage2_mf = np.zeros((2, 2)) 

    last_action_1 = -1
    last_reward = 0

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Handle missing/failed trials
        if a1 == -1 or s_idx == -1 or a2 == -1 or r == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1 # Reset stickiness continuity
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply differential stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                q_net_1[last_action_1] += stick_win
            else:
                q_net_1[last_action_1] += stick_loss

        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Stage 1 Prediction Error (using best Stage 2 value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Eligibility Trace: Stage 2 outcome reinforces Stage 1 choice
        q_stage1_mf[a1] += lr * lam * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dual-Beta Model: Separate exploration parameters for Model-Based and Model-Free systems.
    
    Hypothesis: The participant operates with two distinct systems that have different 
    levels of noise. OCI specifically increases the rigidity (inverse temperature) of the 
    Model-Free (habitual) system, making habits harder to break compared to goal-directed plans.
    
    Parameters:
    lr: [0,1] - Learning rate.
    beta_mb: [0,10] - Inverse temperature for the Model-Based component.
    beta_mf_base: [0,10] - Base inverse temperature for the Model-Free component, scaled by OCI.
    lam: [0,1] - Eligibility trace.
    stickiness: [0,5] - General choice perseveration.
    """
    lr, beta_mb, beta_mf_base, lam, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI modulates MF rigidity
    beta_mf = beta_mf_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1 or s_idx == -1 or a2 == -1 or r == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values weighted by their specific betas (no global w)
        # We assume the policy is Softmax(beta_mb * Q_mb + beta_mf * Q_mf + stickiness)
        # This is mathematically equivalent to weighted sum inputs with beta=1
        
        logits_1 = beta_mb * q_stage1_mb + beta_mf * q_stage1_mf
        
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness

        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Using a composite beta or just one of them? 
        # Standard assumption is MF values drive Stage 2. We use beta_mf here.
        exp_q2 = np.exp(beta_mf * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        q_stage1_mf[a1] += lr * lam * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage-Specific Learning Rates with OCI-Damped Stage 2 Learning.
    
    Hypothesis: High OCI is associated with rigid beliefs about terminal outcomes (Aliens),
    represented by a reduced learning rate at Stage 2. The participant struggles to update
    values for specific aliens even when they provide no reward, leading to repetition.
    
    Parameters:
    lr_s1: [0,1] - Learning rate for Stage 1 (Spaceships).
    lr_s2_base: [0,1] - Base learning rate for Stage 2 (Aliens), damped by OCI.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF mixing weight.
    stickiness: [0,5] - Choice perseveration.
    """
    lr_s1, lr_s2_base, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # OCI damps the learning rate for the second stage (Aliens)
    lr_s2 = lr_s2_base / (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    last_action_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1 or s_idx == -1 or a2 == -1 or r == -1:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            last_action_1 = -1
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 update uses OCI-damped learning rate
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_s2 * delta_stage2
        
        # Stage 1 update uses standard learning rate
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_s1 * delta_stage1
        
        # TD(1) / Lambda update for Stage 1 using Stage 2 RPE
        # Note: We assume lambda=1 implicitly here or 0? 
        # To distinguish from Model 1/2, we use pure TD(0) for Stage 1's second update step
        # or we can omit the lambda step entirely to make it a pure TD(0) model.
        # Let's omit the lambda step to keep it distinct and parsimonious (5 params).
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
