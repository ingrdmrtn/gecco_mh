Here are three cognitive models that explain the participant's behavior in the two-step task, incorporating the OCI-R score to account for individual variability.

### Model 1: OCI-Modulated "Lose-Stay" Bias
This model hypothesizes that high OCI scores are associated with a specific form of compulsivity: the difficulty to shift away from a choice even after receiving a negative outcome (loss). While standard reinforcement learning would lower the value of an unrewarded action, this model adds a "stickiness" bonus to the previous action specifically when the previous trial was unrewarded. This bias counteracts the learning signal, explaining the observed perseveration on unrewarded spaceships.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Modulated 'Lose-Stay' Bias.
    
    Hypothesis: High OCI individuals exhibit compulsive persistence specifically after failure.
    This model adds a choice bonus to the previous Stage 1 action ONLY if the previous
    outcome was a loss (reward = 0). The magnitude of this 'Lose-Stay' bias is scaled by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - loss_stick_oci: [0, 10] Scaling factor for stickiness after a loss.
      Bonus = loss_stick_oci * OCI (applied only if prev_reward == 0).
    """
    learning_rate, beta, w, loss_stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values (0.5 represents neutral expectation)
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10
    
    prev_a1 = -1
    prev_reward = -1

    for trial in range(n_trials):
        # Handle missing data
        if action_1[trial] == -1:
            prev_a1 = -1
            prev_reward = -1
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Hybrid Value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # Apply Lose-Stay Bias
        if prev_a1 != -1 and prev_reward == 0:
            bias = loss_stick_oci * oci_score
            q_net[prev_a1] += bias

        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # --- Learning Updates ---
        # Stage 2 Update (TD)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # Stage 1 Update (TD)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        prev_a1 = a1
        prev_reward = r

    return log_loss
```

### Model 2: Baseline plus OCI-Dependent Stickiness
This model separates the "stickiness" (tendency to repeat choices regardless of reward) into two components: a baseline stickiness shared by all subjects, and an additional stickiness component proportional to the OCI score. This allows the model to distinguish between general inertia and symptom-specific rigidity.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with Dual-Component Stickiness (Base + OCI).
    
    Hypothesis: Choice perseveration consists of a general tendency plus an
    OCI-specific compulsive component. This model uses a linear combination
    to determine the total stickiness bonus.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - stick_base: [0, 5] Baseline stickiness parameter (constant).
    - stick_oci_slope: [0, 5] Additional stickiness scaled by OCI score.
      Total Stickiness = stick_base + (stick_oci_slope * OCI).
    """
    learning_rate, beta, w, stick_base, stick_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    log_loss = 0.0
    eps = 1e-10
    
    prev_a1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            prev_a1 = -1
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # Apply Dual-Component Stickiness
        if prev_a1 != -1:
            total_stickiness = stick_base + (stick_oci_slope * oci_score)
            q_net[prev_a1] += total_stickiness

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # --- Learning Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        prev_a1 = a1

    return log_loss
```

### Model 3: OCI-Modulated Stage 1 Learning Impairment
This model posits that high OCI scores correlate with a specific deficit in updating high-level (Stage 1) values, while low-level (Stage 2) learning remains intact. The "impairment" parameter scales down the learning rate for Stage 1 based on the OCI score. A high impairment would mean the participant fails to update their spaceship preferences based on prediction errors, leading to the observed behavior of sticking to a spaceship despite negative outcomes.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF Model with OCI-Modulated Stage 1 Learning Impairment.
    
    Hypothesis: High OCI is associated with a specific deficit in updating 
    high-level (Stage 1) action values, resulting in rigidity. The Stage 1 
    learning rate is attenuated by the OCI score.
    
    Parameters:
    - lr_base: [0, 1] Base learning rate (used for Stage 2).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - s1_impairment: [0, 1] Scaling factor for Stage 1 learning reduction.
      Stage 1 LR = lr_base * (1 - s1_impairment * OCI).
    """
    lr_base, beta, w, s1_impairment = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    # Calculate effective learning rate for Stage 1
    # If impairment is high and OCI is high, lr_stage1 approaches 0
    lr_stage1 = lr_base * (1.0 - (s1_impairment * oci_score))
    lr_stage1 = np.maximum(0.0, lr_stage1) # Ensure non-negative
    
    lr_stage2 = lr_base
    
    log_loss = 0.0
    eps = 1e-10
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        log_loss -= np.log(probs_1[a1] + eps)

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        log_loss -= np.log(probs_2[a2] + eps)

        # --- Learning Updates ---
        # Update Stage 2 with base learning rate
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_stage2 * delta_stage2

        # Update Stage 1 with OCI-impaired learning rate
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1

    return log_loss
```
