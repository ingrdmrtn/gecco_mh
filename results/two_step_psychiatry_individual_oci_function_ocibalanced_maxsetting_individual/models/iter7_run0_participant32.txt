Here are three cognitive models implemented as Python functions, designed to explain the participant's behavior in the two-step task, incorporating their OCI score to modulate specific cognitive processes.

### Cognitive Model 1: OCI-Modulated Win-Stay Stickiness
This model hypothesizes that high OCI scores correlate with "reinforcement stickiness"â€”a strong compulsion to repeat actions that were immediately successful. Unlike general perseveration, this stickiness is only triggered by positive outcomes (wins), reflecting a specific sensitivity to success or safety signals in obsessive-compulsive traits.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Win-Stay Stickiness Model.
    
    Hypothesis: High OCI participants exhibit enhanced stickiness specifically after receiving a reward 
    (reinforcement amplification), leading to repetitive choices driven by success. Losses do not 
    trigger this specific sticky bonus, distinguishing it from general perseveration.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weight of Model-Based system (0=MF, 1=MB).
    - win_stick_oci: [0, 5] Stickiness bonus applied only after a win, scaled by OCI.
    """
    learning_rate, beta, w, win_stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 0
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate stickiness: Applied ONLY if the previous trial resulted in a win (reward=1)
        # The magnitude is proportional to the participant's OCI score.
        stickiness = np.zeros(2)
        if last_action_1 != -1 and last_reward == 1.0:
            stickiness[int(last_action_1)] = win_stick_oci * oci_score
            
        exp_q1 = np.exp(beta * (q_net + stickiness))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Model-Free Update Stage 1
        delta_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
        
        # Model-Free Update Stage 2
        delta_2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Modulated Risk-Averse Planning
This model suggests that high OCI is associated with pessimism or risk aversion. When the Model-Based system plans ahead (calculating the value of reaching Planet X or Y), it does not simply assume the best possible outcome (max Q). Instead, it mixes in the worst possible outcome (min Q) proportional to the OCI score, effectively devaluing uncertain states.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Risk-Averse Model-Based Planning.
    
    Hypothesis: High OCI participants are risk-averse or pessimistic. When the Model-Based system 
    evaluates future states (Planets), it mixes the maximum Q-value (optimistic) with the 
    minimum Q-value (pessimistic) proportional to the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - risk_aversion_oci: [0, 1] Scaling factor for pessimism. 0 = Rational Max, 1 = Full OCI-driven pessimism.
    """
    learning_rate, beta, w, risk_aversion_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate mixing factor for risk aversion
    # If oci is high, weight shifts towards min_vals (pessimism)
    pessim_weight = risk_aversion_oci * oci_score
    if pessim_weight > 1.0: pessim_weight = 1.0
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # MB Valuation: Mix of Max (Optimistic/Standard) and Min (Pessimistic)
        max_vals = np.max(q_stage2_mf, axis=1)
        min_vals = np.min(q_stage2_mf, axis=1)
        
        # Effective estimated value of next states (planets)
        v_stage2 = (1 - pessim_weight) * max_vals + pessim_weight * min_vals
        
        q_stage1_mb = transition_matrix @ v_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        delta_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
        
        delta_2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Tunnel Vision (Unchosen Decay)
This model posits that high OCI participants suffer from "tunnel vision" or hyper-focus. While they update the value of the chosen action normally, the values of *unchosen* actions decay (return to zero) more rapidly than in low OCI participants. This represents a difficulty in maintaining working memory for alternative paths.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Tunnel Vision (Unchosen Decay) Model.
    
    Hypothesis: High OCI participants focus narrowly on their chosen path. The values of unchosen actions
    (both spaceships and aliens) decay (are forgotten) faster than for low OCI participants.
    This creates a "tunnel vision" effect where alternative options become less relevant over time.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for chosen options.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - decay_oci: [0, 1] Decay rate for unchosen options, scaled by OCI.
    """
    learning_rate, beta, w, decay_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate decay factor
    # If OCI is high, decay is larger (more forgetting of unchosen options)
    decay_rate = decay_oci * oci_score
    if decay_rate > 1.0: decay_rate = 1.0
    
    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Update chosen Stage 1
        delta_1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
        
        # Decay unchosen Stage 1
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)
        
        # Update chosen Stage 2
        delta_2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_2
        
        # Decay unchosen Stage 2 (for the current state only)
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
