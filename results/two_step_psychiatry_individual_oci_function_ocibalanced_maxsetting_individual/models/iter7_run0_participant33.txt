Here are three cognitive models expressed as Python functions.

### Model 1: OCI-Modulated Inverse Temperature (Rigidity)
This model hypothesizes that high OCI scores lead to "behavioral rigidity" by increasing the inverse temperature ($\beta$) parameter. A higher $\beta$ makes choices more deterministic (greedy) and less exploratory, explaining why the participant might stick to one spaceship for a very long time (e.g., Spaceship 0 for ~150 trials) before switching and sticking to the other.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Inverse Temperature Model.
    
    Hypothesis: High OCI scores increase the inverse temperature (beta), leading 
    to more deterministic and rigid choice patterns (less exploration).
    
    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    w: [0, 1] - Mixing weight (0=MF, 1=MB).
    beta_base: [0, 10] - Baseline inverse temperature.
    beta_oci_slope: [0, 10] - Sensitivity of beta to OCI score.
    """
    learning_rate, w, beta_base, beta_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta based on OCI
    # High OCI -> Higher Beta -> More rigid/deterministic choices
    beta_effective = beta_base + (beta_oci_slope * oci_score * 10.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value (Hybrid MB/MF)
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax with OCI-modulated Beta
        exp_q1 = np.exp(beta_effective * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Store probability of observed action
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta_effective * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 Update (Standard Q-learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD-learning)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Dampened Learning Rate (Insensitivity)
This model hypothesizes that high OCI scores are associated with a reduced sensitivity to prediction errors (feedback). The effective learning rate is scaled down by the OCI score. This explains the participant's behavior of sticking to Spaceship 0 despite receiving many 0-coin outcomes (failures), as the negative prediction errors update the Q-values too slowly to induce a switch.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Dampened Learning Rate Model.
    
    Hypothesis: High OCI scores suppress the learning rate, making the participant
    slow to update value estimates in response to feedback (insensitivity/inertia).
    
    Parameters:
    lr_base: [0, 1] - Baseline learning rate.
    lr_oci_suppress: [0, 5] - Factor by which OCI reduces the learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight (0=MF, 1=MB).
    """
    lr_base, lr_oci_suppress, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective learning rate
    # High OCI -> Lower Learning Rate
    learning_rate = lr_base / (1.0 + (lr_oci_suppress * oci_score))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Enhanced Eligibility Trace (Direct Habit)
This model hypothesizes that high OCI increases the "eligibility trace" ($\lambda$), causing the final reward to directly reinforce the Stage 1 choice, bypassing the two-step structure. A high $\lambda$ collapses the two stages into a single habit-learning step. This reflects the compulsive nature of habits where the action is reinforced by the outcome regardless of the intermediate state structure.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Enhanced Eligibility Trace Model.
    
    Hypothesis: High OCI scores increase the eligibility trace (lambda), causing
    Stage 2 outcomes to directly reinforce Stage 1 choices (smearing credit assignment),
    characteristic of strong habit formation.
    
    Parameters:
    learning_rate: [0, 1] - Update rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight.
    lambda_base: [0, 1] - Baseline eligibility trace.
    lambda_oci_boost: [0, 1] - Increase in lambda per unit of OCI.
    """
    learning_rate, beta, w, lambda_base, lambda_oci_boost = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective eligibility trace lambda
    # High OCI -> Higher Lambda (Direct reinforcement of Stage 1 by Reward)
    lambda_eff = lambda_base + (lambda_oci_boost * oci_score)
    # Clip to valid range [0, 1]
    if lambda_eff > 1.0: lambda_eff = 1.0
    if lambda_eff < 0.0: lambda_eff = 0.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Standard TD Error for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update with Eligibility Trace
        # Standard part: Update based on Stage 2 Q-value
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Eligibility Trace part: Update Stage 1 based on Stage 2 Reward Error
        # If lambda is high, Stage 1 value is pulled directly towards the Reward
        q_stage1_mf[a1] += learning_rate * lambda_eff * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
