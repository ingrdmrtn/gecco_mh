Here are three new cognitive models based on the participant's data and OCI score.

### Cognitive Model 1: Dynamic Control Arbitration (Accelerated Habitization)
This model hypothesizes that the balance between Model-Based (goal-directed) and Model-Free (habitual) control is not static. Instead, high OCI scores accelerate the transition from Model-Based to Model-Free control over the course of the task. This reflects the theory that compulsivity involves a faster "slip" into habitual behaviors.

The weighting parameter $w$ decays exponentially over trials, with the decay rate scaled by the OCI score.
$$ w_{trial} = w_{init} \cdot \exp(-\text{decay\_scale} \cdot \text{oci} \cdot \text{trial}) $$

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Dynamic Control Arbitration Model (Accelerated Habitization).
    
    Hypothesis: High OCI participants shift from Model-Based to Model-Free control 
    more rapidly than low OCI participants. The mixing weight 'w' decays over trials,
    with the decay rate determined by the OCI score.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_init: [0, 1] (Initial Model-Based weight)
    decay_scale: [0, 0.1] (Scaling factor for OCI-dependent decay)
    """
    learning_rate, beta, w_init, decay_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf1 = np.zeros(2)      # Model-free values for stage 1
    q_mf2 = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)
    
    # Pre-calculate decay factor constant
    decay_factor = decay_scale * oci_score

    for trial in range(n_trials):
        # Calculate dynamic weight for this trial
        w = w_init * np.exp(-decay_factor * trial)

        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q_MF2)
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        # Net Value
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 RPE
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2

        # Stage 1 RPE (SARSA-like for MF)
        delta_1 = q_mf2[s2, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * delta_1
        
        # Eligibility trace (lambda=1)
        q_mf1[a1] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Stage-Specific Precision (Beta Modulation)
This model posits that OCI specifically impacts the precision (inverse temperature) of the second-stage choices (choosing the alien) differently than the first-stage planning. High OCI might lead to rigid, deterministic behavior in the immediate, concrete task (Stage 2) while leaving abstract planning (Stage 1) unaffected, or vice versa.

The model uses a base `beta_1` for the spaceship choice and a modified `beta_2` for the alien choice.
$$ \beta_2 = \beta_1 \cdot (1 + \text{beta\_mod} \cdot \text{oci}) $$

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Stage-Specific Precision Model.
    
    Hypothesis: OCI modulates the deterministic nature (inverse temperature) of the 
    second-stage choice (concrete outcome) relative to the first-stage choice.
    High OCI may lead to "locking in" on specific aliens (high beta_2).
    
    Bounds:
    learning_rate: [0, 1]
    beta_1: [0, 10] (Base inverse temperature for Stage 1)
    w: [0, 1]
    beta_2_mod: [-1, 5] (Modifier for Stage 2 beta based on OCI)
    """
    learning_rate, beta_1, w, beta_2_mod = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf1 = np.zeros(2)
    q_mf2 = np.zeros((2, 2))
    
    # Calculate Stage 2 Beta
    # Clip to ensure non-negative beta, though bounds should handle reasonable values
    beta_2 = max(0.0, beta_1 * (1.0 + beta_2_mod * oci_score))

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy (uses beta_1) ---
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy (uses beta_2) ---
        exp_q2 = np.exp(beta_2 * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2

        delta_1 = q_mf2[s2, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * delta_1
        q_mf1[a1] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Counterfactual Rumination
This model introduces an OCI-modulated mechanism where the participant updates the *unchosen* spaceship's Model-Free value based on Model-Based information. This represents a form of "rumination" or obsessive alignment, where the participant compulsively updates their habitual values to match their world model, even for actions they didn't take.

$$ Q_{MF1}(\text{unchosen}) \leftarrow Q_{MF1}(\text{unchosen}) + \alpha_{cf} \cdot (Q_{MB}(\text{unchosen}) - Q_{MF1}(\text{unchosen})) $$
Where $\alpha_{cf}$ is proportional to OCI.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Counterfactual Rumination Model.
    
    Hypothesis: High OCI participants engage in 'rumination', updating the Model-Free 
    values of the UNCHOSEN option in Stage 1 towards the Model-Based estimate.
    This reduces the discrepancy between habits and knowledge for unvisited states.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    rumination_rate: [0, 1] (Scales the learning rate for unchosen options via OCI)
    """
    learning_rate, beta, w, rumination_rate = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_mf1 = np.zeros(2)
    q_mf2 = np.zeros((2, 2))
    
    # Calculate counterfactual learning rate
    alpha_cf = learning_rate * rumination_rate * oci_score

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Calculate MB values for both options
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2

        # --- Stage 1 Policy ---
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # 1. Standard updates for chosen path
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2

        delta_1 = q_mf2[s2, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * delta_1
        q_mf1[a1] += learning_rate * delta_2
        
        # 2. Counterfactual update for the UNCHOSEN path
        # The unchosen option is pulled towards its current MB estimate
        unchosen_a1 = 1 - a1
        cf_delta = q_mb1[unchosen_a1] - q_mf1[unchosen_a1]
        q_mf1[unchosen_a1] += alpha_cf * cf_delta

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
