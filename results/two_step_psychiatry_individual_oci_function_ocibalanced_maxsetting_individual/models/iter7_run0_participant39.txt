Here are the three proposed cognitive models.

### Model 1: OCI-Modulated Choice Trace Decay
This model hypothesizes that "stickiness" is not just a repetition of the immediately previous choice, but a decaying memory trace of past choices. OCI modulates the **decay rate** of this trace. High OCI scores may lead to a slower decay (higher persistence/habit strength), explaining the long streaks of behavior observed in the participant data.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Choice Trace Decay.
    
    Hypothesis:
    Stickiness is modeled as a 'Choice Trace' that accumulates and decays over time.
    OCI modulates the decay rate of this trace.
    High OCI -> Slower decay (Longer-lasting habits/perseveration).
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for MF values.
    beta: [0, 10] - Inverse temperature (shared).
    w: [0, 1] - MB/MF mixing weight.
    stickiness_mag: [-5, 5] - Magnitude of the influence of the choice trace.
    trace_decay_base: [0, 1] - Base decay rate of the choice trace.
    trace_decay_oci: [-1, 1] - OCI modulation of the decay rate.
    """
    learning_rate, beta, w, stickiness_mag, trace_decay_base, trace_decay_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate decay rate modulated by OCI
    decay = trace_decay_base + trace_decay_oci * oci_score
    decay = np.clip(decay, 0.0, 0.95) # Clip to ensure stability
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2)) 
    
    # Choice trace for Stage 1 actions (0 and 1)
    choice_trace = np.zeros(2) 
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Handle missing/invalid trials
        if a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Choice Trace (Stickiness)
        q_net_sticky = q_net + stickiness_mag * choice_trace
        
        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update Choice Trace
        # Trace accumulates for chosen, decays for both
        mask = np.zeros(2)
        mask[a1] = 1.0
        choice_trace = decay * choice_trace + (1.0 - decay) * mask
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Loss Sensitivity
This model tests the hypothesis that high OCI is associated with an altered perception of negative outcomes. Specifically, it posits that participants may treat the absence of a coin (0 reward) not as a neutral event, but as a loss. The magnitude of this subjective loss is modulated by OCI. This affects the asymptotic Q-values (converging to a negative value instead of 0) and thus avoidance behavior.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Loss Sensitivity.
    
    Hypothesis:
    Participants with high OCI may perceive the lack of reward (0) as a loss/punishment.
    This model introduces a 'loss_sensitivity' parameter. If reward is 0, the effective reward is -loss_sensitivity.
    OCI modulates this sensitivity.
    
    Parameters:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    stickiness: [-5, 5]
    loss_sens_base: [0, 1] - Base magnitude of negative utility for 0 outcome.
    loss_sens_oci: [-1, 1] - OCI modulation of loss sensitivity.
    """
    learning_rate, beta, w, stickiness, loss_sens_base, loss_sens_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective loss sensitivity
    loss_sens = loss_sens_base + loss_sens_oci * oci_score
    loss_sens = np.clip(loss_sens, 0.0, 5.0) # Bound to reasonable range
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # Calculate subjective reward
        # If r=1, val=1. If r=0, val=-loss_sens
        effective_r = r if r > 0.5 else -loss_sens

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Use effective_r instead of r
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = effective_r - q_stage2_mf[s_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Counterfactual Regret
This model explores the idea that participants perform counterfactual updating ("What if I had chosen the other spaceship?"). When an outcome is received, the model updates the value of the *unchosen* spaceship in Stage 1 towards the opposite outcome (assuming that if the chosen one won, the unchosen would have lost, and vice versa). OCI modulates the rate of this regret-based update, potentially capturing the "doubting" nature of OCD.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Counterfactual Regret.
    
    Hypothesis:
    Participants update not only the chosen option but also the unchosen option in Stage 1,
    simulating "what if" scenarios. OCI modulates the rate of this counterfactual updating (regret).
    If I chose A and got Reward R, I update U assuming I would have got (1-R).
    
    Parameters:
    learning_rate: [0, 1] - Learning rate for chosen option.
    beta: [0, 10]
    w: [0, 1]
    stickiness: [-5, 5]
    regret_alpha_base: [0, 1] - Learning rate for unchosen option (regret).
    regret_alpha_oci: [-1, 1] - OCI modulation of regret rate.
    """
    learning_rate, beta, w, stickiness, regret_alpha_base, regret_alpha_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    regret_alpha = regret_alpha_base + regret_alpha_oci * oci_score
    regret_alpha = np.clip(regret_alpha, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Update Chosen (Standard)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Unchosen (Counterfactual)
        # We assume the unchosen path yields the inverse reward: (1 - r)
        # This is a heuristic for "The other one would have been better/worse"
        unchosen_a1 = 1 - a1
        counterfactual_r = 1.0 - r
        
        # Simple counterfactual update towards the inverse outcome
        q_stage1_mf[unchosen_a1] += regret_alpha * (counterfactual_r - q_stage1_mf[unchosen_a1])
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
