Here are three new cognitive models exploring different mechanisms for how Obsessive-Compulsive symptoms (OCI) might influence decision-making in this two-step task, specifically focusing on perseveration, model-based control, and learning rates.

### Model 1: OCI-Modulated Perseveration (Stickiness)
This model hypothesizes that high OCI scores lead to increased "stickiness" or perseverationâ€”a tendency to repeat the previous choice regardless of reward. This reflects the compulsive aspect of OCD, where behaviors are repeated rigidly. The model incorporates a stickiness parameter that is boosted by the participant's OCI score.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated choice stickiness (perseveration).
    
    Hypothesis: Participants with higher OCI scores exhibit higher 'stickiness' 
    (tendency to repeat the previous Stage 1 choice), reflecting compulsive repetition.
    
    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) systems.
    - stick_base: [0, 5] Baseline tendency to repeat the previous choice.
    - stick_oci_sens: [0, 5] Additional stickiness per unit of OCI.
    """
    learning_rate, beta, w, stick_base, stick_oci_sens = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate effective stickiness based on OCI
    stickiness = stick_base + (participant_oci * stick_oci_sens)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_action_1 = -1 # No previous action for the first trial

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the Q-value of the previous action
        q_net_stick = q_net.copy()
        if prev_action_1 != -1:
            q_net_stick[int(prev_action_1)] += stickiness

        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        s_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # TD Update Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # TD Update Stage 1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        prev_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Model-Based Weight (Decreasing)
This model posits that high OCI scores are associated with a *deficit* in goal-directed (model-based) control, leading to a greater reliance on habitual (model-free) systems. Instead of a fixed weight `w`, the weight is dynamically determined by a base parameter and an OCI-dependent penalty.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI reduces the weight of Model-Based control.
    
    Hypothesis: High OCI is associated with reduced goal-directed (Model-Based) 
    planning, favoring habitual (Model-Free) control.
    
    Parameters:
    - learning_rate: [0, 1] Value update rate.
    - beta: [0, 10] Inverse temperature.
    - w_base: [0, 1] Baseline weight for Model-Based control (for OCI=0).
    - w_oci_penalty: [0, 1] Reduction in MB weight per unit of OCI.
    """
    learning_rate, beta, w_base, w_oci_penalty = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]

    # Calculate w, bounded between 0 and 1
    w = w_base - (participant_oci * w_oci_penalty)
    w = np.clip(w, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]

        s_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Asymmetric Learning Rates (Punishment Sensitivity)
This model suggests that OCI relates to altered sensitivity to negative outcomes (or lack of reward). It uses separate learning rates for positive (reward=1) and negative (reward=0) outcomes. The hypothesis is that high OCI might be associated with "hyper-learning" from failure (or safety checking), so the negative learning rate is modulated by OCI.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Q-learning with separate learning rates for positive/negative outcomes,
    where OCI modulates the learning rate for negative outcomes (losses).
    
    Hypothesis: High OCI participants are hypersensitive to errors/losses, 
    learning more rapidly from non-rewarded trials.
    
    Parameters:
    - alpha_pos: [0, 1] Learning rate for rewarded trials.
    - alpha_neg_base: [0, 1] Base learning rate for unrewarded trials.
    - alpha_neg_oci_sens: [0, 1] Increase in negative learning rate per unit of OCI.
    - beta: [0, 10] Inverse temperature.
    - lambda_eligibility: [0, 1] Eligibility trace decay (connects stage 2 to stage 1).
    """
    alpha_pos, alpha_neg_base, alpha_neg_oci_sens, beta, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate negative learning rate, bounded [0, 1]
    alpha_neg = alpha_neg_base + (participant_oci * alpha_neg_oci_sens)
    alpha_neg = np.clip(alpha_neg, 0.0, 1.0)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        s_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Determine learning rate based on reward
        if r > 0:
            lr = alpha_pos
        else:
            lr = alpha_neg
            
        # TD Error Stage 2
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += lr * delta_stage2
        
        # TD Error Stage 1 (using Stage 2 value)
        delta_stage1 = q_stage2[s_idx, a2] - q_stage1[a1]
        
        # Update Stage 1 with eligibility trace (lambda)
        q_stage1[a1] += lr * lambda_eligibility * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
