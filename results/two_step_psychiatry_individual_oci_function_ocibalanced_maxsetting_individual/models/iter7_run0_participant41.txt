def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI distorts the subject's belief about the transition matrix (Fatalistic Beliefs).
    High OCI leads to a belief that transitions are more random (closer to 50/50),
    reducing the effectiveness of Model-Based control even if the weight 'w' is high.
    
    Parameters:
    lr: [0, 1] Learning rate for Stage 2 values (and Stage 1 MF).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based values (0=MF, 1=MB).
    distort: [0, 1] Factor scaling how much OCI flattens the transition matrix.
                    0 = no distortion (0.7/0.3), 1 = max distortion (approaching 0.5/0.5).
    """
    lr, beta, w, distort = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    # Distort transition probabilities based on OCI
    # Standard is 0.7. Max distortion reduces it towards 0.5.
    # We allow up to 0.2 reduction (0.7 -> 0.5).
    p_common = 0.7 - (0.2 * distort * oci_score)
    transition_matrix = np.array([[p_common, 1-p_common], [1-p_common, p_common]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax
        logits = beta * q_net
        logits -= np.max(logits)
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # Policy for the second choice
        q_s2 = q_stage2_mf[state_idx]
        logits_2 = beta * q_s2
        logits_2 -= np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Action value updating
        r = reward[trial]
        
        # Stage 2 Update (TD)
        pe2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * pe2
        
        # Stage 1 MF Update (TD from Stage 2 value)
        pe1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * pe1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model (MB+MF) with Choice Perseveration (Stickiness) scaled by OCI.
    High OCI leads to compulsive repetition of the previous Stage 1 choice,
    independent of reward history or model-based planning. This tests if 
    compulsivity co-exists with model-based planning.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based values.
    stick_scale: [0, 5] Scaling factor for stickiness based on OCI. 
                        Stickiness = stick_scale * OCI.
    """
    lr, beta, w, stick_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    stickiness = stick_scale * oci_score
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        
        # Add stickiness
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        logits -= np.max(logits)
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1
        
        state_idx = int(state[trial])

        # Policy for the second choice
        q_s2 = q_stage2_mf[state_idx]
        logits_2 = beta * q_s2
        logits_2 -= np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Action value updating
        r = reward[trial]
        
        pe2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * pe2
        
        pe1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * pe1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model with OCI-induced Anxiety (Noise) at Stage 1.
    High OCI participants exhibit reduced inverse temperature (beta) specifically
    at the first stage (decision conflict/planning), while Stage 2 (simple outcome) 
    remains standard.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Base inverse temperature (applied to Stage 2).
    w: [0, 1] Weight for Model-Based values.
    anxiety_damp: [0, 5] Dampening factor for Stage 1 beta based on OCI.
                         beta_1 = beta / (1 + anxiety_damp * OCI).
    """
    lr, beta, w, anxiety_damp = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate stage-specific betas
    beta_stage1 = beta / (1.0 + anxiety_damp * oci_score)
    beta_stage2 = beta
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta_stage1 * q_net
        logits -= np.max(logits)
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # Policy for the second choice
        q_s2 = q_stage2_mf[state_idx]
        logits_2 = beta_stage2 * q_s2
        logits_2 -= np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Action value updating
        r = reward[trial]
        
        pe2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * pe2
        
        pe1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * pe1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
