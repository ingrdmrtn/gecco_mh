Here are the three cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Subjective Reward Valuation Model (OCI-modulated Loss Aversion).
    
    Hypothesis: High OCI participants may perceive the absence of reward (0 coins) 
    not as a neutral event, but as a failure or penalty. This model introduces a 
    subjective negative reward for 0 outcomes, scaled by OCI score. This drives 
    Q-values lower and increases avoidance of non-rewarding options.

    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature (softmax sensitivity).
    w: [0, 1] Model-Based weight.
    stickiness: [0, 5] Choice stickiness for Stage 1.
    oci_zero_penalty: [0, 5] Magnitude of the subjective penalty for 0 reward, scaled by OCI.
    """
    lr, beta, w, stickiness, oci_zero_penalty = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Effective penalty for 0 reward
    zero_penalty = participant_oci * oci_zero_penalty

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        s_curr = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r_raw = reward[trial]
        # Subjective reward transformation: 1 stays 1, 0 becomes negative penalty
        r_eff = r_raw if r_raw == 1 else -zero_penalty
        
        # Stage 2 Update
        pe_2 = r_eff - q_stage2_mf[s_curr, a2]
        q_stage2_mf[s_curr, a2] += lr * pe_2
        
        # Stage 1 Update (MF)
        # Using direct reinforcement from effective reward (surrogate for TD(1))
        q_stage1_mf[a1] += lr * (r_eff - q_stage1_mf[a1])
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Unchosen Value Forgetting Model (OCI-modulated Decay).
    
    Hypothesis: High OCI is associated with checking behaviors, potentially driven by 
    a lack of confidence in memory or the status of unchosen options. This model 
    posits that OCI accelerates the decay (forgetting) of Q-values for unchosen 
    spaceships, pushing them towards neutrality (0) and potentially causing perseverance 
    on the chosen option if the unchosen one is forgotten and assumed low value.

    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-Based weight.
    stickiness: [0, 5] Choice stickiness.
    oci_forget_rate: [0, 1] Decay rate for unchosen action values, scaled by OCI.
    """
    lr, beta, w, stickiness, oci_forget_rate = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    decay = participant_oci * oci_forget_rate

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        s_curr = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 Update
        pe_2 = r - q_stage2_mf[s_curr, a2]
        q_stage2_mf[s_curr, a2] += lr * pe_2
        
        # Stage 1 Update (MF)
        q_stage1_mf[a1] += lr * (r - q_stage1_mf[a1])
        
        # Forgetting for unchosen Stage 1 action
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay)
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Structural Doubt Model (OCI-modulated Transition Uncertainty).
    
    Hypothesis: High OCI participants may doubt the stability of the environmental 
    structure. While the true transition probability is 0.7, high OCI leads to a 
    "flattening" of the internal transition matrix towards 0.5 (maximum uncertainty), 
    degrading the effectiveness of Model-Based control.

    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-Based weight.
    stickiness: [0, 5] Choice stickiness.
    oci_trans_doubt: [0, 1] Factor reducing the perceived transition probability from 0.7 towards 0.5.
    """
    lr, beta, w, stickiness, oci_trans_doubt = model_parameters
    n_trials = len(action_1)
    participant_oci = oci[0]
    
    # Calculate subjective transition probability
    # Base is 0.7. Max doubt (oci=1, param=1) reduces 0.7 by 0.2 to 0.5.
    p_common = 0.7 - (participant_oci * oci_trans_doubt * 0.2)
    # Ensure bounds just in case
    p_common = np.maximum(p_common, 0.5)
    
    # Construct subjective transition matrix
    subjective_T = np.array([[p_common, 1-p_common], [1-p_common, p_common]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    prev_a1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Use subjective matrix for MB calculation
        q_stage1_mb = subjective_T @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        s_curr = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 Update
        pe_2 = r - q_stage2_mf[s_curr, a2]
        q_stage2_mf[s_curr, a2] += lr * pe_2
        
        # Stage 1 Update (MF)
        q_stage1_mf[a1] += lr * (r - q_stage1_mf[a1])
        
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
