def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Stress-Induced Habit (Dynamic w).
    
    Hypothesis: High OCI individuals shift towards Model-Free (habitual) control 
    specifically after experiencing a negative outcome (stress/failure) on the previous trial.
    
    Mechanism:
    w (mixing weight) determines the balance between MB and MF control.
    If the previous trial was unrewarded (reward == 0), effective w is reduced:
    w_eff = w_base * (1.0 - oci * stress_scale)
    Otherwise, w_eff = w_base.
    
    Bounds:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w_base: [0, 1] Baseline Model-Based weight.
    stress_scale: [0, 1] Scaling factor for OCI-induced reduction in w after loss.
    """
    lr, beta, w_base, stress_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    prev_reward = 1.0 # Initialize as if previous was success to start baseline

    for trial in range(n_trials):
        # Calculate effective w based on previous outcome
        if prev_reward == 0.0:
            # Apply stress reduction
            w_eff = w_base * (1.0 - (oci_score * stress_scale))
            # Clip to ensure it stays non-negative
            w_eff = max(0.0, w_eff)
        else:
            w_eff = w_base

        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w_eff * q_stage1_mb + (1 - w_eff) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        prev_reward = r # Update for next trial

        # Updates
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        q_stage1_mf[a1] += lr * delta_stage1
        q_stage2_mf[state_idx, a2] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Accelerated Forgetting (Tunnel Vision).
    
    Hypothesis: High OCI individuals focus intensely on the chosen path and 
    rapidly forget or discount the value of unchosen options (tunnel vision).
    
    Mechanism:
    Q-values for unchosen aliens (Stage 2) and unchosen spaceships (Stage 1) 
    decay towards 0 on each trial. The decay rate is boosted by the OCI score.
    decay_rate = decay_base + (oci * oci_decay_amp)
    
    Bounds:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF weight.
    decay_base: [0, 0.5] Base decay rate for unchosen options.
    oci_decay_amp: [0, 0.5] OCI amplification of decay.
    """
    lr, beta, w, decay_base, oci_decay_amp = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective decay rate
    decay_rate = decay_base + (oci_score * oci_decay_amp)
    # Ensure decay doesn't exceed 1 or go below 0
    decay_rate = min(max(decay_rate, 0.0), 1.0)

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Updates
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        q_stage1_mf[a1] += lr * delta_stage1
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Decay unchosen values
        # Decay unchosen stage 2 action in the visited state
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        
        # Decay both actions in the unvisited state
        unvisited_state = 1 - state_idx
        q_stage2_mf[unvisited_state, :] *= (1.0 - decay_rate)
        
        # Decay unchosen stage 1 action
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Specific Stage 2 Determinism.
    
    Hypothesis: High OCI individuals exhibit higher rigidity (determinism) 
    specifically at the second stage (choosing the alien), regardless of 
    their first-stage exploration.
    
    Mechanism:
    We use two inverse temperature parameters.
    beta_1 applies to Stage 1.
    beta_2 applies to Stage 2, and is boosted by OCI.
    beta_2 = beta_1 + (oci * oci_beta_inc)
    
    Bounds:
    lr: [0, 1]
    beta_1: [0, 10] Base inverse temperature (Stage 1).
    w: [0, 1]
    oci_beta_inc: [0, 10] Increase in beta for Stage 2 per unit OCI.
    """
    lr, beta_1, w, oci_beta_inc = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate Stage 2 beta
    beta_2 = beta_1 + (oci_score * oci_beta_inc)

    for trial in range(n_trials):
        # Stage 1 Policy (uses beta_1)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # Stage 2 Policy (uses beta_2)
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Updates
        delta_stage2 = r - q_stage2_mf[state_idx, a2]

        q_stage1_mf[a1] += lr * delta_stage1
        q_stage2_mf[state_idx, a2] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
