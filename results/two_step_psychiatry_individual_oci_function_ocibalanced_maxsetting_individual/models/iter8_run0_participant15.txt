def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Post-Rare-Transition Confusion.
    
    This model hypothesizes that rare transitions (unexpected state changes) cause 
    cognitive disruption ("confusion") in participants with high OCI symptoms.
    When a rare transition occurs, the inverse temperature (beta) for the *next* 
    trial's first-stage choice is reduced, leading to more random/exploratory behavior.
    
    Parameters:
    lr: [0, 1] Learning rate for value updates.
    beta: [0, 10] Baseline inverse temperature (decision consistency).
    w: [0, 1] Weight for model-based control (0=MF, 1=MB).
    stick: [0, 5] Choice stickiness/perseveration.
    rare_confusion_oci: [0, 10] OCI-dependent reduction in beta after a rare transition.
                        Beta_eff = Beta / (1 + rare_confusion_oci * OCI) if prev was rare.
    """
    lr, beta, w, stick, rare_confusion_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Transition probabilities: 0->0 (0.7), 0->1 (0.3), 1->0 (0.3), 1->1 (0.7)
    # Common transitions: action==state. Rare: action!=state.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Q-values for stage 2 (2 states, 2 aliens)
    
    prev_action_1 = -1
    prev_was_rare = False

    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted integrated value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        if prev_action_1 != -1:
            q_net[prev_action_1] += stick
        
        # Apply OCI-Modulated Confusion (Beta scaling)
        current_beta = beta
        if prev_was_rare:
            # Reduce beta (increase noise) if previous transition was rare, scaled by OCI
            current_beta = beta / (1.0 + rare_confusion_oci * oci_score)

        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx]) # Use baseline beta for stage 2
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 1 MF Update (TD(0))
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # --- History Tracking ---
        prev_action_1 = a1
        
        # Determine if this transition was rare
        # Common: 0->0, 1->1. Rare: 0->1, 1->0.
        # This corresponds to a1 == s_idx being common.
        is_common = (a1 == s_idx)
        prev_was_rare = not is_common

    eps = 1e-10
    # Filter out invalid trials (if any)
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss


def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Habit Formation Rate.
    
    Instead of simple choice stickiness (repeating the last action), this model
    tracks a continuous 'Habit Strength' that accumulates over time. 
    OCI modulates the *learning rate* of this habit strength. High OCI leads to 
    faster habit formation (the habit trace updates more aggressively towards the 
    current choice), making recent history dominate the compulsion to repeat.
    
    Parameters:
    lr: [0, 1] Value learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weight.
    stick_weight: [0, 5] The weight of the habit strength in the decision variable.
    habit_lr_oci: [0, 1] OCI-dependent learning rate for the habit trace.
                  Habit_LR = habit_lr_oci * OCI. (Bounded [0,1])
    """
    lr, beta, w, stick_weight, habit_lr_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Habit strength vector for the two spaceships
    habit_strength = np.zeros(2) 
    
    # Calculate the effective habit learning rate based on OCI
    # We clip it to [0, 1] to ensure stability
    alpha_habit = np.clip(habit_lr_oci * oci_score, 0.0, 1.0)
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Habit Strength (weighted by stick_weight)
        # Note: habit_strength is a continuous trace, not just binary last choice
        q_net += stick_weight * habit_strength
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Value Learning ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # --- Habit Learning ---
        # Update habit strength: Chosen action moves towards 1, Unchosen towards 0
        # H(a) = H(a) + alpha * (1 - H(a))  [for chosen]
        # H(a) = H(a) + alpha * (0 - H(a))  [for unchosen]
        habit_strength[a1] += alpha_habit * (1.0 - habit_strength[a1])
        habit_strength[1-a1] += alpha_habit * (0.0 - habit_strength[1-a1])

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss


def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Punishment Sensitivity (Loss Learning Rate).
    
    This model hypothesizes that OCI symptoms correlate with a hypersensitivity 
    to negative prediction errors (disappointment/loss of expected reward).
    When the outcome is worse than expected (negative RPE), the learning rate 
    is boosted by the OCI score, leading to over-correction of values.
    
    Parameters:
    lr: [0, 1] Base learning rate (used for positive RPEs).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weight.
    stick: [0, 5] Choice stickiness.
    loss_lr_boost_oci: [0, 10] Multiplier for the learning rate when RPE is negative.
                       LR_neg = LR_base * (1 + loss_lr_boost_oci * OCI).
    """
    lr, beta, w, stick, loss_lr_boost_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1 or action_2[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        if last_action_1 != -1:
            q_net[last_action_1] += stick
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Update Stage 1 ---
        # Note: Stage 1 update uses Q-values, delta is (Q2 - Q1).
        # We apply the boost if the delta is negative (outcome < expectation).
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        lr_eff_1 = lr
        if delta_stage1 < 0:
            lr_eff_1 = lr * (1.0 + loss_lr_boost_oci * oci_score)
            
        q_stage1_mf[a1] += lr_eff_1 * delta_stage1
        
        # --- Update Stage 2 ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        lr_eff_2 = lr
        if delta_stage2 < 0:
            lr_eff_2 = lr * (1.0 + loss_lr_boost_oci * oci_score)
            
        q_stage2_mf[s_idx, a2] += lr_eff_2 * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    valid_mask = (p_choice_1 > 0) & (p_choice_2 > 0)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
