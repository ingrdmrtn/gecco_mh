Here are the three cognitive models as Python functions.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with OCI-Modulated Learning Rate Decay.
    
    Hypothesis: High OCI participants may maintain a higher learning rate for longer 
    (hyper-vigilance), resisting the natural decay of plasticity over the task.
    Low OCI participants might settle into a strategy and reduce learning rate (decay) faster.
    
    Parameters:
    lr_init: [0, 1] - Initial learning rate.
    lr_decay_base: [0, 10] - Base decay rate of learning rate over time.
    lr_decay_oci: [-5, 5] - Modulation of the decay rate by OCI.
    beta: [0, 10] - Inverse temperature (exploration/exploitation balance).
    w: [0, 1] - Weight of Model-Based system (0=MF, 1=MB).
    stickiness: [-5, 5] - Choice perseveration bonus.
    """
    lr_init, lr_decay_base, lr_decay_oci, beta, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate decay rate specific to this participant
    # Decay rate must be non-negative
    decay_rate = np.maximum(0.0, lr_decay_base + (lr_decay_oci * oci_val))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize values
    q_stage1_mf = np.zeros(2) # MF values for stage 1 actions (0, 1)
    q_stage2_mf = np.zeros((2, 2)) # MF values for stage 2: State (0,1) x Action (0,1)
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # Dynamic Learning Rate Schedule
        # lr_t = lr_0 / (1 + decay * t_normalized * scaling)
        t_norm = trial / n_trials
        current_lr = lr_init / (1.0 + decay_rate * t_norm * 10.0)
        
        # --- Stage 1 Decision ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value (Hybrid MB/MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits_1 = beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Softmax Probability
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Value Updates ---
        r = reward[trial]
        
        # Prediction Errors
        # Stage 1 PE (SARSA-style): Value of state reached (given action chosen) - Value of stage 1 choice
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Stage 2 PE: Reward - Value of stage 2 choice
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        # Update Stage 1 MF (using eligibility trace logic, effectively adding delta_stage2)
        q_stage1_mf[a1] += current_lr * (delta_stage1 + delta_stage2)
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, a2] += current_lr * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with OCI-Modulated Loss Sensitivity (Beta).
    
    Hypothesis: OCI scores correlate with how participants react to lack of reward (loss).
    High OCI might lead to either rigid (high beta) or erratic (low beta) behavior 
    immediately following a 0-reward trial, compared to a winning trial.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    w: [0, 1] - Model-Based weight.
    stickiness: [-5, 5] - Choice perseveration bonus.
    beta_win: [0, 10] - Inverse temperature after a reward (1).
    beta_loss_base: [0, 10] - Base inverse temperature after a loss (0).
    beta_loss_oci: [-5, 5] - Modulation of loss-beta by OCI.
    """
    learning_rate, w, stickiness, beta_win, beta_loss_base, beta_loss_oci = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate beta for loss trials
    beta_loss = beta_loss_base + (beta_loss_oci * oci_val)
    beta_loss = np.maximum(0.0, beta_loss) # Ensure non-negative beta
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = 1.0 # Default to optimistic/neutral start
    
    for trial in range(n_trials):
        # Determine current beta based on previous outcome
        current_beta = beta_win if last_reward == 1.0 else beta_loss
        
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = current_beta * q_net
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        
        logits_2 = current_beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_1 = a1
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with OCI-Modulated Stickiness Retention after Rare Transitions.
    
    Hypothesis: A rare transition is a surprising event that usually disrupts habits (stickiness).
    High OCI participants might be more rigid, retaining their stickiness/habit even 
    after a rare transition (ignoring the structural surprise), or conversely, be hyper-sensitive.
    
    Parameters:
    learning_rate: [0, 1] - Learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Model-Based weight.
    stickiness_mag: [0, 5] - Base magnitude of stickiness.
    rare_keep_base: [-5, 5] - Base retention rate of stickiness after rare transition (logit scale).
    rare_keep_oci: [-5, 5] - Modulation of retention rate by OCI.
    """
    learning_rate, beta, w, stickiness_mag, rare_keep_base, rare_keep_oci = model_parameters
    n_trials = len(action_1)
    oci_val = oci[0]
    
    # Calculate retention rate (0 to 1) using sigmoid
    retention_logit = rare_keep_base + (rare_keep_oci * oci_val)
    retention_rate = 1.0 / (1.0 + np.exp(-retention_logit))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    was_rare_transition = False
    
    for trial in range(n_trials):
        
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        if last_action_1 != -1:
            # Apply stickiness
            # If previous transition was rare, modulate stickiness by retention_rate
            current_stickiness = stickiness_mag
            if was_rare_transition:
                current_stickiness *= retention_rate
                
            logits_1[last_action_1] += current_stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Decision ---
        state_idx = int(state[trial])
        
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Determine if this transition was rare for next trial's stickiness logic
        # Common: (0->0) or (1->1). Rare: (0->1) or (1->0).
        # abs(a1 - state_idx) == 0 implies Common, == 1 implies Rare
        was_rare_transition = (abs(a1 - state_idx) == 1)
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
