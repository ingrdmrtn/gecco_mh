Here are the 3 proposed cognitive models.

### Cognitive Model 1: Independent Model-Based Temperature Modulation
```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI modulates the 'Inverse Temperature' (Beta) of the Model-Based system 
    independently of the Model-Free system.
    
    Hypothesis: OCI affects the strength/gain of goal-directed (Model-Based) control specifically, 
    rather than trading it off against Model-Free control (as 'w' does). This allows for states 
    where both systems are strong or weak, or one is dominant.
    
    Parameters:
    lr: Learning rate [0, 1]
    stick: General stickiness/perseveration bonus [0, 5]
    beta_mf: Inverse temperature for Model-Free values [0, 10]
    beta_mb_base: Baseline inverse temperature for Model-Based values [0, 10]
    beta_mb_oci_param: Effect of OCI on Model-Based temperature [0, 10] (can be negative in effect if scaled)
    """
    lr, stick, beta_mf, beta_mb_base, beta_mb_oci_param = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate MB beta based on OCI
    # We allow the parameter to scale the beta up or down. 
    # Assuming input is positive, we interpret it as a magnitude of change.
    # Here we define it as an additive effect, clipped to valid range.
    beta_mb = beta_mb_base + beta_mb_oci_param * current_oci
    beta_mb = np.clip(beta_mb, 0, 10)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Independent betas for MB and MF components
        logits = (beta_mb * q_stage1_mb) + (beta_mf * q_stage1_mf)
        
        # Add stickiness to the previous choice
        if trial > 0:
            prev_a1 = int(action_1[trial-1])
            logits[prev_a1] += stick
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        # Standard softmax on stage 2 MF values
        # We reuse beta_mf for stage 2 as it is pure reinforcement learning
        exp_q2 = np.exp(beta_mf * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 Update (TD-0)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update (TD-0)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Lose-Stay Stickiness Modulation
```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI modulates 'Lose-Stay' behavior (compulsive persistence).
    Stickiness is applied specifically when the previous trial was unrewarded (Loss).
    
    Hypothesis: High OCI participants may exhibit 'compulsivity', repeating actions 
    even after negative outcomes (losses), contrary to standard RL which would devalue the action.
    
    Parameters:
    lr: Learning rate [0, 1]
    beta: Inverse temperature [0, 10]
    w: Model-based mixing weight [0, 1]
    stick_loss_base: Baseline stickiness after a loss [0, 5]
    stick_loss_oci_param: Effect of OCI on stickiness after a loss [0, 5]
    """
    lr, beta, w, stick_loss_base, stick_loss_oci_param = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Lose-Stickiness based on OCI
    stick_loss = stick_loss_base + stick_loss_oci_param * current_oci
    # We allow stickiness to be high, bound reasonably
    stick_loss = np.clip(stick_loss, 0, 10)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        logits = beta * q_net
        
        # Apply stickiness ONLY if previous trial was a LOSS (reward == 0)
        if trial > 0:
            prev_r = reward[trial-1]
            if prev_r == 0:
                prev_a1 = int(action_1[trial-1])
                logits[prev_a1] += stick_loss
        
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Rare-Transition Model-Free Discounting
```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid model where OCI modulates the Model-Free learning update specifically after Rare transitions.
    
    Hypothesis: OCI affects how participants credit-assign after a model violation (rare transition). 
    They might ignore the outcome (discounting it) or over-weight it. 
    This model scales the MF prediction error by a factor 'rare_disc' when the transition was rare.
    
    Parameters:
    lr: Learning rate [0, 1]
    beta: Inverse temperature [0, 10]
    w: Model-based mixing weight [0, 1]
    rare_disc_base: Baseline discounting factor for rare transitions [0, 2] (1 = normal, 0 = ignore)
    rare_disc_oci_param: Effect of OCI on rare transition discounting [0, 2]
    """
    lr, beta, w, rare_disc_base, rare_disc_oci_param = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate discounting factor
    rare_disc = rare_disc_base + rare_disc_oci_param * current_oci
    rare_disc = np.clip(rare_disc, 0, 5) # Allow it to be > 1 (over-reaction) or < 1 (dampening)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Check if transition was rare
        # Common: 0->0, 1->1. Rare: 0->1, 1->0.
        is_rare = (a1 != s_idx)
        
        if is_rare:
            q_stage1_mf[a1] += lr * rare_disc * delta_stage1
        else:
            q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
