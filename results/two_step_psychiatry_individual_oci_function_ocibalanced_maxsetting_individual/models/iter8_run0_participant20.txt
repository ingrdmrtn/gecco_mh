Here are three new cognitive models for the two-step decision task, incorporating the OCI score to explain individual variability in decision-making processes.

### Cognitive Model 1: OCI-Modulated Risk Aversion (Minimax Planning)
**Hypothesis:** High OCI participants may exhibit "catastrophizing" or intolerance of uncertainty. In the Model-Based (MB) planning phase, rather than assuming they will make the optimal choice at the second stage (maximizing Q-value), they might weigh the "worst-case scenario" (minimum Q-value) more heavily. This model interpolates between maximizing and minimizing the expected second-stage value based on OCI.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Risk Aversion in Model-Based Planning (Minimax).
    
    Hypothesis: High OCI participants weigh the worst-case scenario (minimum Q-value) 
    at the second stage more heavily than standard agents who assume optimal choice (max Q-value)
    during model-based planning.
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - stick: [0, 5] Choice stickiness.
    - risk_oci: [0, 1] Sensitivity of risk aversion to OCI. 
      Determines the weight 'rho' on the min_Q value: rho = risk_oci * oci.
    
    Returns:
        float: Negative log-likelihood.
    """
    lr, beta, w, stick, risk_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_choice_1 = -1
    
    # Calculate risk weight rho, clipped to [0, 1]
    rho = risk_oci * oci_score
    if rho > 1.0: rho = 1.0
    if rho < 0.0: rho = 0.0
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        # Stage 1 Policy
        # MB Value Calculation with Risk Aversion
        # Standard: V(s') = max(Q(s', :))
        # Minimax: V(s') = (1-rho)*max(Q) + rho*min(Q)
        max_q2 = np.max(q_stage2_mf, axis=1)
        min_q2 = np.min(q_stage2_mf, axis=1)
        v_stage2 = (1 - rho) * max_q2 + rho * min_q2
        
        q_stage1_mb = transition_matrix @ v_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        logits_1 = beta * q_net
        
        if last_choice_1 != -1:
            logits_1[last_choice_1] += stick
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        # Stage 1 MF Update (using actual stage 2 value estimate)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2
        
        last_choice_1 = a1

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Modulated Win-Stay Stickiness
**Hypothesis:** OCI is associated with compulsive repetition and reassurance seeking. High OCI participants may exhibit excessive "Win-Stay" behavior—sticking to a choice that just provided a reward—more strongly than low OCI participants. This model separates base stickiness from an OCI-enhanced stickiness that triggers only after a win.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Win-Stay Stickiness (Reassurance Seeking).
    
    Hypothesis: High OCI participants may exhibit excessive "Win-Stay" behavior, 
    seeking reassurance from previously rewarded choices more than low OCI participants.
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - stick_base: [0, 5] Base stickiness (applied on loss or general).
    - stick_win_oci: [0, 5] Additional stickiness applied only after a Win, scaled by OCI.
    
    Returns:
        float: Negative log-likelihood.
    """
    lr, beta, w, stick_base, stick_win_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_choice_1 = -1
    last_reward = 0
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        
        if last_choice_1 != -1:
            # Base stickiness
            stick_val = stick_base
            # Add OCI-modulated stickiness if last trial was a win
            if last_reward == 1:
                stick_val += stick_win_oci * oci_score
            
            logits_1[last_choice_1] += stick_val
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2
        
        last_choice_1 = a1
        last_reward = r

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Counterfactual Updating
**Hypothesis:** High OCI participants may engage in more counterfactual thinking ("What if I had chosen the other spaceship?"). This model proposes that OCI modulates a "latent learning" mechanism where the agent updates the value of the *unchosen* spaceship based on the state they actually arrived at, weighted by the transition probability of the unchosen spaceship leading to that same state.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Counterfactual Updating (Latent Learning).
    
    Hypothesis: High OCI participants may engage in more counterfactual thinking 
    ("What if I had chosen the other spaceship?"), updating the unchosen option 
    based on the state they actually arrived at.
    
    Parameters:
    - lr: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - stick: [0, 5] Stickiness.
    - cf_oci: [0, 1] Counterfactual update rate scaling with OCI.
      Update unchosen action with weight: lr * cf_oci * oci * P(State|Unchosen).
    
    Returns:
        float: Negative log-likelihood.
    """
    lr, beta, w, stick, cf_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    last_choice_1 = -1
    
    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue
            
        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = int(reward[trial])
        
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        if last_choice_1 != -1:
            logits_1[last_choice_1] += stick
            
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        # 1. Standard MF Update for chosen action
        v_s2_est = q_stage2_mf[s2, a2]
        delta_stage1 = v_s2_est - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # 2. Counterfactual Update for unchosen action
        # If I had chosen 'other_a', I might have reached s2 with prob P(s2|other_a).
        # So I learn from v_s2_est weighted by that probability.
        other_a = 1 - a1
        prob_transition = transition_matrix[other_a, s2]
        
        # Effective learning rate for CF update
        lr_cf = lr * cf_oci * oci_score * prob_transition
        
        delta_cf = v_s2_est - q_stage1_mf[other_a]
        q_stage1_mf[other_a] += lr_cf * delta_cf
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2
        
        last_choice_1 = a1

    mask = (action_1 != -1)
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1[mask] + eps)) + np.sum(np.log(p_choice_2[mask] + eps)))
    return log_loss
```
