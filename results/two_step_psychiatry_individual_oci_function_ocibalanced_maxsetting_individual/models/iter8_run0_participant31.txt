### Cognitive Model 1: OCI-Modulated Goal Stickiness
This model hypothesizes that participants with higher OCI scores exhibit a "Goal Stickiness" or "Planet Perseveration" bias. Unlike standard stickiness which repeats the *motor action* (Spaceship), this mechanism biases the Model-Based system to prefer the *outcome state* (Planet) visited in the previous trial, regardless of whether it was rewarding. This reflects a compulsive urge to return to the same location.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Goal Stickiness Model.
    
    Hypothesis: High OCI participants exhibit a compulsion to return to the 
    previously visited state (Planet), biasing the Model-Based planning.
    
    Parameters:
    - learning_rate: [0,1] Update rate for MF values.
    - beta: [0,10] Inverse temperature for choice stochasticity.
    - w: [0,1] Weighting between MB (1) and MF (0).
    - goal_stick_base: [0,1] Baseline bias to return to previous planet (scaled to [0, 5]).
    - goal_stick_oci: [0,1] Modulation of goal stickiness by OCI (scaled to [-5, 5]).
    """
    learning_rate, beta, w, goal_stick_base, goal_stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Map parameters
    goal_stickiness = (goal_stick_base * 5.0) + ((goal_stick_oci - 0.5) * 10.0 * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    prev_planet = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Value of each planet
        
        # Apply Goal Stickiness to the value of the previous planet (if exists)
        v_planning = max_q_stage2.copy()
        if prev_planet != -1:
            v_planning[prev_planet] += goal_stickiness
            
        q_stage1_mb = transition_matrix @ v_planning
        
        # Net Q-value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Stage 1
        logits_1 = beta * q_net
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        q_vals_2 = q_stage2_mf[s_idx]
        logits_2 = beta * q_vals_2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 1 MF Update (SARSA-style using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Store state for next trial's goal stickiness
        prev_planet = s_idx

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Modulated Stickiness Decay
This model hypothesizes that OCI relates to the *persistence* of compulsive habits. Instead of simple 1-back stickiness, the model maintains a "Choice Trace" that decays over time. OCI modulates the decay rate ($\rho$), such that high OCI participants have slower-decaying (longer-lasting) urges to repeat actions.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stickiness Decay Model.
    
    Hypothesis: High OCI participants have a more persistent 'choice trace', 
    leading to stickiness that decays slower over time (compulsive habit).
    
    Parameters:
    - learning_rate: [0,1] Update rate.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Weighting MB/MF.
    - stick_weight: [0,1] Magnitude of the stickiness bias (scaled to [0, 5]).
    - rho_base: [0,1] Baseline decay rate of the choice trace.
    - rho_oci: [0,1] Modulation of decay rate by OCI (scaled to [-1, 1]).
    """
    learning_rate, beta, w, stick_weight_param, rho_base, rho_oci_param = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    stick_weight = stick_weight_param * 5.0
    rho_mod = (rho_oci_param - 0.5) * 2.0 # Map to [-1, 1]
    rho = np.clip(rho_base + rho_mod * oci_score, 0.0, 0.99)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for Stage 1 actions (0 and 1)
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Stickiness based on decayed trace
        logits_1 = beta * q_net + stick_weight * choice_trace
        
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Update Trace
        choice_trace *= rho # Decay
        choice_trace[a1] += 1.0 # Increment chosen
        
        # --- Stage 2 Policy ---
        q_vals_2 = q_stage2_mf[s_idx]
        logits_2 = beta * q_vals_2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Modulated Independent Betas
This model decouples the Model-Based and Model-Free systems, assigning them independent "strengths" (inverse temperatures) instead of a trade-off parameter $w$. OCI specifically modulates the strength of the Model-Free system ($\beta_{MF}$). This tests the hypothesis that high OCI is driven by an overactive habit system (high $\beta_{MF}$) rather than a deficit in the MB system.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Independent Betas Model.
    
    Hypothesis: MB and MF systems operate with independent strengths (betas).
    OCI specifically modulates the gain (strength) of the MF system.
    
    Parameters:
    - learning_rate: [0,1] Update rate.
    - beta_mb: [0,10] Inverse temperature for MB system.
    - beta_mf_base: [0,10] Baseline inverse temperature for MF system.
    - beta_mf_oci: [0,1] Modulation of MF beta by OCI (scaled to [-5, 5]).
    - stickiness: [0,1] Choice stickiness (scaled to [0, 5]).
    """
    learning_rate, beta_mb, beta_mf_base, beta_mf_oci_param, stick_param = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Map parameters
    beta_mf_mod = (beta_mf_oci_param - 0.5) * 10.0
    beta_mf = np.maximum(0.0, beta_mf_base + beta_mf_mod * oci_score)
    stickiness = stick_param * 5.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_a1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Independent contributions to logits
        # Note: beta_mb applies to MB Q-values, beta_mf applies to MF Q-values
        logits_1 = (beta_mb * q_stage1_mb) + (beta_mf * q_stage1_mf)
        
        if last_a1 != -1:
            logits_1[last_a1] += stickiness
            
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        last_a1 = a1
        
        # --- Stage 2 Policy ---
        # Using beta_mb for stage 2 as it's the 'directed' part, or average?
        # Standard models usually share beta. Here we use beta_mb as the 'rational' beta.
        q_vals_2 = q_stage2_mf[s_idx]
        logits_2 = beta_mb * q_vals_2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
