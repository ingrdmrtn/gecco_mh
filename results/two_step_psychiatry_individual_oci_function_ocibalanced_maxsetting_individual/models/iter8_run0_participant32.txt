def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Cumulative Habit Model.
    
    Hypothesis: High OCI participants form stronger habits that persist over time. 
    Unlike simple win-stay stickiness which relies on the immediate previous outcome, 
    this model uses a cumulative choice trace (habit) that decays slowly. 
    The influence of this habit trace on Stage 1 choices is scaled by the OCI score,
    reflecting compulsive repetition independent of reward.

    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - trace_decay: [0, 1] Decay rate of the choice trace (0 = 1-step stickiness, 1 = perfect memory).
    - habit_oci: [0, 5] Strength of the habit influence scaled by OCI.
    """
    learning_rate, beta, w, trace_decay, habit_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Choice trace for Stage 1 actions (0 or 1)
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add cumulative habit bonus
        habit_bonus = habit_oci * oci_score * choice_trace
        
        exp_q1 = np.exp(beta * (q_net + habit_bonus))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update choice trace
        a1 = int(action_1[trial])
        choice_trace *= trace_decay
        choice_trace[a1] += 1.0

        # Updates
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # SARSA update for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Q-learning/Prediction Error update for Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Post-Error Rigidity Model.
    
    Hypothesis: High OCI participants exhibit increased rigidity (decreased exploration) 
    immediately after a loss (0 coins), driven by anxiety or intolerance of failure. 
    This is modeled by increasing the inverse temperature (beta) for the trial 
    following a loss, scaled by the OCI score.

    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_base: [0, 10] Baseline inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - loss_rigidity_oci: [0, 5] Proportional increase in beta after a loss, scaled by OCI.
    """
    learning_rate, beta_base, w, loss_rigidity_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_reward = 1.0 # Initialize as if win to not trigger rigidity on first trial
    
    for trial in range(n_trials):
        # Determine current beta
        if last_reward == 0.0:
            current_beta = beta_base * (1.0 + loss_rigidity_oci * oci_score)
        else:
            current_beta = beta_base

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stage 2 Stickiness Model.
    
    Hypothesis: Compulsive behavior in high OCI participants manifests as repetitive checking
    behavior at the second stage (choosing the same alien), distinct from the planning 
    decisions at Stage 1. This model adds a stickiness bonus to the previously chosen 
    Stage 2 action, scaled by OCI.

    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight of Model-Based system.
    - stick_s2_oci: [0, 5] Stickiness bonus for the repeated Stage 2 choice, scaled by OCI.
    """
    learning_rate, beta, w, stick_s2_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_2 = -1
    
    for trial in range(n_trials):
        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # policy for the second choice
        # Calculate stickiness vector for Stage 2
        s2_stickiness = np.zeros(2)
        if last_action_2 != -1:
            s2_stickiness[last_action_2] = stick_s2_oci * oci_score
            
        exp_q2 = np.exp(beta * (q_stage2_mf[state_idx] + s2_stickiness))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        last_action_2 = a2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
