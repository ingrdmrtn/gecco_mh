Here are three new cognitive models that introduce different mechanisms for how OCI symptoms might influence decision-making in the two-step task.

### Model 1: OCI-Modulated Choice Stickiness (Perseveration)
This model hypothesizes that high OCI scores lead to increased "stickiness" or perseveration in choices. Instead of just affecting the balance between model-based and model-free control, OCI here drives a tendency to repeat the previous action regardless of reward, reflecting the repetitive nature of compulsive symptoms.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated choice stickiness (perseveration).
    
    Hypothesis: High OCI participants exhibit greater behavioral rigidity (stickiness),
    repeating their previous stage 1 choice regardless of outcomes.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1] (Mixing weight: 0=MF, 1=MB)
    stickiness_base: [0, 5] (Base tendency to repeat choice)
    stickiness_oci_gain: [0, 5] (Additional stickiness per unit of OCI)
    """
    learning_rate, beta, w, stickiness_base, stickiness_oci_gain = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective stickiness parameter
    # Higher OCI leads to higher tendency to repeat the previous action
    stickiness = stickiness_base + (stickiness_oci_gain * oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)      
    q_mf2 = np.zeros((2, 2)) 
    
    # Track the previous choice for stickiness
    prev_a1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        # Add stickiness bonus to the previously chosen action
        # We create a temporary Q-value vector for decision making
        q_decision = q_net.copy()
        if prev_a1 != -1:
            q_decision[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # TD Error Stage 2
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2
        
        # TD Error Stage 1
        delta_1 = q_mf2[s2, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * delta_1
        
        # Eligibility trace (lambda=1) update for Stage 1 based on Stage 2 outcome
        q_mf1[a1] += learning_rate * delta_2
        
        # Update previous action
        prev_a1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Second-Stage Learning Rate
This model proposes that OCI affects how strongly participants update their values based on the final reward. Specifically, high OCI might be associated with "over-learning" from specific outcomes or a hypersensitivity to feedback at the second stage (the direct reward receipt), potentially making behavior more volatile or reactive to recent wins/losses.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where the second-stage learning rate is modulated by OCI.
    
    Hypothesis: OCI affects sensitivity to direct reward feedback. High OCI participants
    might update values more aggressively (higher learning rate) or more slowly (rigidity)
    at the stage where reward is received.
    
    Bounds:
    lr_base: [0, 1] (Base learning rate)
    lr_oci_mod: [-1, 1] (Modulation of learning rate by OCI)
    beta: [0, 10]
    w: [0, 1]
    """
    lr_base, lr_oci_mod, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Sigmoid transform to keep learning rate within [0, 1]
    # The effective LR is base + modulation * OCI
    # We use a simple clamp here to ensure bounds
    raw_lr = lr_base + (lr_oci_mod * oci_score)
    learning_rate = np.clip(raw_lr, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)      
    q_mf2 = np.zeros((2, 2)) 

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Stage 2 update uses the OCI-modulated learning rate
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2
        
        # Stage 1 update also uses this rate for consistency in this model variant
        delta_1 = q_mf2[s2, a2] - q_mf1[a1]
        q_mf1[a1] += learning_rate * delta_1
        
        q_mf1[a1] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Inverse Temperature (Exploration/Exploitation)
This model posits that OCI impacts the exploration-exploitation trade-off. High OCI (compulsivity) might manifest as extremely deterministic behavior (high beta), reducing exploration even when uncertainty is high.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where the inverse temperature (beta) is modulated by OCI.
    
    Hypothesis: High OCI leads to less stochasticity/exploration (higher beta).
    The participant becomes more deterministic in selecting the option with the
    highest current value.
    
    Bounds:
    learning_rate: [0, 1]
    beta_base: [0, 10]
    beta_oci_slope: [0, 10] (Increase in beta per unit of OCI)
    w: [0, 1]
    """
    learning_rate, beta_base, beta_oci_slope, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective beta
    # Higher OCI -> Higher beta (sharper selection, less randomness)
    beta = beta_base + (beta_oci_slope * oci_score)
    # Ensure beta doesn't explode or go negative
    beta = np.clip(beta, 0.0, 20.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_mf1 = np.zeros(2)      
    q_mf2 = np.zeros((2, 2)) 

    for trial in range(n_trials):
        if action_1[trial] == -1:
            continue

        a1 = int(action_1[trial])
        s2 = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)

        # --- Updates ---
        delta_2 = r - q_mf2[s2, a2]

    return log_loss
```
