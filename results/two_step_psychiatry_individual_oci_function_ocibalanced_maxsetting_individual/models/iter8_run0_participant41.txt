Here are three cognitive models that incorporate the OCI score to explain the participant's decision-making behavior, particularly focusing on the "sticky" or perseverative behavior observed in the data.

### Model 1: MB/MF Hybrid with OCI-Modulated Choice Stickiness
This model posits that the OCI score modulates the degree of simple choice perseveration (stickiness). High OCI participants are hypothesized to have a higher tendency to repeat their previous First-Stage choice, regardless of the outcome. This is implemented as a bonus added to the net Q-value of the previously chosen spaceship.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    MB/MF Hybrid with OCI-Modulated Choice Stickiness.
    
    High OCI scores may lead to increased perseveration (stickiness) to the previously 
    chosen action. This model modulates the stickiness parameter based on the OCI score.
    A weighted mixture of Model-Based and Model-Free values determines the base policy,
    biased by the stickiness bonus.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature for softmax.
    w: [0,1] - Weighting between Model-Based (1) and Model-Free (0) control.
    stick_low: [0,5] - Stickiness bonus for low OCI (0.0).
    stick_high: [0,5] - Stickiness bonus for high OCI (1.0).
    """
    learning_rate, beta, w, stick_low, stick_high = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate OCI-specific stickiness
    stickiness = stick_low * (1 - oci_score) + stick_high * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    last_action_1 = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Stickiness to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax policy
        exp_q1 = np.exp(beta * q_net)
        if np.any(np.isinf(exp_q1)):
             probs_1 = np.zeros_like(exp_q1)
             probs_1[np.argmax(q_net)] = 1.0
        else:
             probs_1 = exp_q1 / np.sum(exp_q1)
             
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        s = int(state[trial])
        
        # Standard Q-learning for stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        if np.any(np.isinf(exp_q2)):
            probs_2 = np.zeros_like(exp_q2)
            probs_2[np.argmax(q_stage2_mf[s])] = 1.0
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = int(action_2[trial])
        if a2 != -1: # Handle valid trials
            p_choice_2[trial] = probs_2[a2]
            
            # --- Updates ---
            r = reward[trial]
            a1 = int(action_1[trial])
            
            # TD(0) Update for Stage 1 MF (using Stage 2 value)
            delta_1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
            q_stage1_mf[a1] += learning_rate * delta_1
            
            # TD(0) Update for Stage 2 MF (using Reward)
            delta_2 = r - q_stage2_mf[s, a2]
            q_stage2_mf[s, a2] += learning_rate * delta_2
            
            # Propagate Stage 2 RPE to Stage 1 (Eligibility Trace / TD(1)-like)
            q_stage1_mf[a1] += learning_rate * delta_2
            
            last_action_1 = a1
        else:
            # If trial was missed, we don't update last_action_1 or values
            pass

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: MB/MF Hybrid with OCI-Modulated Dynamic Habit Strength
This model introduces a dynamic "Habit Strength" variable that accumulates when an action is chosen and decays when it is not. This captures the "building up" of compulsive streaks better than simple one-trial stickiness. The weight of this habit strength in the final decision is modulated by OCI.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    MB/MF Hybrid with OCI-Modulated Dynamic Habit Strength.
    
    Instead of simple 1-back stickiness, this model maintains a 'habit strength' 
    trace that accumulates when actions are repeated and decays when not. 
    The influence of this habit strength on choice is modulated by OCI.
    
    Parameters:
    learning_rate: [0,1] - Update rate for Q-values.
    beta: [0,10] - Inverse temperature.
    w_mb: [0,1] - Weight of Model-Based control.
    habit_lr: [0,1] - Learning rate for habit accumulation/decay.
    w_habit_low: [0,10] - Weight of habit strength for low OCI.
    w_habit_high: [0,10] - Weight of habit strength for high OCI.
    """
    learning_rate, beta, w_mb, habit_lr, w_habit_low, w_habit_high = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate OCI-specific habit weight
    w_habit = w_habit_low * (1 - oci_score) + w_habit_high * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    habit_values = np.zeros(2) # Habit strength for spaceship 0 and 1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB, MF, and accumulated Habit
        q_net = w_mb * q_stage1_mb + (1 - w_mb) * q_stage1_mf + w_habit * habit_values
        
        exp_q1 = np.exp(beta * q_net)
        if np.any(np.isinf(exp_q1)):
             probs_1 = np.zeros_like(exp_q1)
             probs_1[np.argmax(q_net)] = 1.0
        else:
             probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update Habit Strength: Chosen strengthens, Unchosen decays
        habit_values[a1] += habit_lr * (1 - habit_values[a1])
        habit_values[1-a1] += habit_lr * (0 - habit_values[1-a1])
        
        # --- Stage 2 Choice ---
        s = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        if np.any(np.isinf(exp_q2)):
            probs_2 = np.zeros_like(exp_q2)
            probs_2[np.argmax(q_stage2_mf[s])] = 1.0
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = int(action_2[trial])
        if a2 != -1:
            p_choice_2[trial] = probs_2[a2]
            
            r = reward[trial]
            
            # Updates
            delta_1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
            q_stage1_mf[a1] += learning_rate * delta_1
            
            delta_2 = r - q_stage2_mf[s, a2]
            q_stage2_mf[s, a2] += learning_rate * delta_2
            
            q_stage1_mf[a1] += learning_rate * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: MB/MF Hybrid with OCI-Modulated Transition Belief
This model hypothesizes that OCI affects the cognitive map (transition matrix) used by the Model-Based system. High OCI participants might perceive the world as more deterministic (rigid belief that Spacehip A -> Planet X) or more chaotic, distorting the standard 0.7/0.3 probabilities. This distorted belief affects how values are propagated from Stage 2 to Stage 1.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    MB/MF Hybrid with OCI-Modulated Transition Belief.
    
    This model hypothesizes that OCI affects the internal model of the environment's 
    transition probabilities used by the Model-Based system. High OCI might lead 
    to a distorted belief about transitions (e.g., believing they are deterministic).
    
    Parameters:
    learning_rate: [0,1] - Update rate.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - MB/MF weight.
    stickiness: [0,5] - General choice stickiness.
    p_belief_low: [0.5, 1.0] - Perceived common transition prob for low OCI.
    p_belief_high: [0.5, 1.0] - Perceived common transition prob for high OCI.
    """
    learning_rate, beta, w, stickiness, p_belief_low, p_belief_high = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate perceived transition probability based on OCI
    p = p_belief_low * (1 - oci_score) + p_belief_high * oci_score
    
    # Construct distorted transition matrix used for MB planning
    # [P(S2=0|S1=0), P(S2=1|S1=0)]
    # [P(S2=0|S1=1), P(S2=1|S1=1)]
    transition_matrix = np.array([[p, 1-p], [1-p, p]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)

             probs_1[np.argmax(q_net)] = 1.0
        else:
             probs_1 = exp_q1 / np.sum(exp_q1)
             
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        s = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        if np.any(np.isinf(exp_q2)):
            probs_2 = np.zeros_like(exp_q2)
            probs_2[np.argmax(q_stage2_mf[s])] = 1.0
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = int(action_2[trial])
        if a2 != -1:
            p_choice_2[trial] = probs_2[a2]
            
            r = reward[trial]
            
            # Updates
            delta_1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
            q_stage1_mf[a1] += learning_rate * delta_1
            
            delta_2 = r - q_stage2_mf[s, a2]
            q_stage2_mf[s, a2] += learning_rate * delta_2

            # Logic here implies we don't update stickiness target if trial is invalid.
            pass
```
