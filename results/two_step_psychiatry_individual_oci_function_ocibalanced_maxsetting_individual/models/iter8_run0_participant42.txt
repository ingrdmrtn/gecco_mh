```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Post-Win Beta Rigidity Model.
    
    Hypothesis: High OCI participants exhibit excessive rigidity (high inverse temperature)
    specifically after receiving a reward, leading to 'locking in' a behavior (compulsivity).
    After a loss, rigidity relaxes to baseline, allowing for switching.
    
    Parameters:
    - learning_rate: [0,1] Value update rate.
    - beta_base: [0,10] Baseline inverse temperature for Stage 1.
    - beta_win_oci: [0,10] Additional rigidity after a win, scaled by OCI.
                      beta_1 = beta_base + beta_win_oci * oci * prev_reward
    - w: [0,1] Model-based weight.
    - stickiness: [0,5] Choice stickiness (perseveration).
    """
    learning_rate, beta_base, beta_win_oci, w, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    prev_reward = 0.0 
    last_action_1 = -1

    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        # Dynamic Beta: Increases if previous trial was rewarded, scaled by OCI
        current_beta = beta_base + beta_win_oci * oci_score * prev_reward
        
        exp_q1 = np.exp(current_beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Policy (Use base beta for stability/parsimony)
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta_base * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Propagate stage 2 RPE to stage 1 MF values
        q_stage1_mf[a1] += learning_rate * delta_stage2 
        
        last_action_1 = a1
        prev_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Transition Uncertainty Model.
    
    Hypothesis: High OCI participants doubt the transition structure, effectively 
    degrading the 'common' probability in the Model-Based system towards chance (0.5).
    This reduces the directional guidance of the MB system, causing reliance on MF/stickiness.
    
    Parameters:
    - learning_rate: [0,1]
    - beta: [0,10]
    - w: [0,1]
    - distortion: [0,1] Degree to which OCI degrades p_common towards 0.5.
                     p_perceived = 0.7 - (0.2 * distortion * oci)
    - stickiness: [0,5]
    """
    learning_rate, beta, w, distortion, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Distort transition matrix towards 0.5 based on OCI
    # If distortion=1 and oci=1, p_common becomes 0.5 (random).
    p_common = 0.7 - (0.2 * distortion * oci_score)
    p_common = np.clip(p_common, 0.5, 0.7) # Ensure it stays valid
    
    transition_matrix = np.array([[p_common, 1-p_common], [1-p_common, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    last_action_1 = -1

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]

        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Rare Transition Over-Learning Model.
    
    Hypothesis: High OCI participants over-learn from outcomes following rare transitions,
    possibly due to heightened salience of anomalies or superstitious reinforcement.
    This leads to Model-Free 'Stay' behavior even after rare transitions.
    
    Parameters:
    - learning_rate: [0,1] Base learning rate.
    - beta: [0,10]
    - w: [0,1]
    - rare_boost: [0,10] Multiplier for learning rate on rare trials, scaled by OCI.
                     lr_trial = lr * (1 + rare_boost * oci) [if rare]
    - stickiness: [0,5]
    """
    learning_rate, beta, w, rare_boost, stickiness = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    last_action_1 = -1

    for trial in range(n_trials):
        # Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Check if transition was rare (Action 0 -> State 1, or Action 1 -> State 0)
        is_rare = (a1 != state_idx)
        
        # Modulate Learning Rate
        current_lr = learning_rate
        if is_rare:
            current_lr = learning_rate * (1.0 + rare_boost * oci_score)
            current_lr = min(current_lr, 1.0) # Bound at 1.0
            
        # Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += current_lr * delta_stage2
        
        # Propagate to Stage 1
        q_stage1_mf[a1] += current_lr * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
