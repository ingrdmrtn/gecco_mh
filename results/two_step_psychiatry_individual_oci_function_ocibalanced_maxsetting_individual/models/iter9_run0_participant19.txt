Here are three new cognitive models for the two-step task, incorporating the participant's OCI score to modulate specific learning and decision-making mechanisms.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Rare Transition Learning Suppression.
    
    Hypothesis: High OCI participants may exhibit distrust or rigidity when faced 
    with unexpected outcomes (rare transitions). This model posits that they 
    suppress Model-Free learning updates specifically on trials where a rare 
    transition occurred, treating them as anomalies rather than informative signals.
    
    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based values (0=MF, 1=MB).
    - rare_suppress: [0, 5] Strength of learning suppression on rare trials, scaled by OCI.
      Effective LR = LR / (1 + rare_suppress * OCI).
    """
    learning_rate, beta, w, rare_suppress = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate suppression factor based on OCI
    suppression_denom = 1.0 + rare_suppress * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) 
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # MB Value: Expected value based on transition matrix and max stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Determine if transition was rare
        # Common: (A0->S0), (A1->S1). Rare: (A0->S1), (A1->S0)
        is_rare = (a1 == 0 and s_idx == 1) or (a1 == 1 and s_idx == 0)
        
        # Apply suppression to Stage 1 MF update if rare
        current_lr_s1 = learning_rate
        if is_rare:
            current_lr_s1 = learning_rate / suppression_denom
            
        # Stage 1 MF Update (SARSA-like)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr_s1 * delta_stage1
        
        # Stage 2 MF Update (Standard Reward Prediction Error)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Asymmetric Learning (Loss Hypersensitivity).
    
    Hypothesis: High OCI participants may be hypersensitive to negative outcomes 
    (failure/loss of reward). This model implements asymmetric learning rates, 
    where the learning rate for negative prediction errors is amplified by the OCI score.
    
    Parameters:
    - learning_rate: [0, 1] Base learning rate for positive prediction errors.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based values.
    - oci_loss_bias: [0, 5] Multiplier for learning rate on negative errors. 
      LR_neg = LR_pos * (1 + oci_loss_bias * OCI).
    """
    learning_rate, beta, w, oci_loss_bias = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    lr_pos = learning_rate
    lr_neg = learning_rate * (1.0 + oci_loss_bias * oci_score)
    # Clamp lr_neg to 1.0 to prevent instability
    if lr_neg > 1.0: lr_neg = 1.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        lr_1 = lr_neg if delta_stage1 < 0 else lr_pos
        q_stage1_mf[a1] += lr_1 * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        lr_2 = lr_neg if delta_stage2 < 0 else lr_pos
        q_stage2_mf[s_idx, a2] += lr_2 * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Outcome-State Stickiness.
    
    Hypothesis: High OCI participants may exhibit a compulsion to return to the 
    previously visited environmental state (Planet), distinct from repeating the 
    previous motor action. This model adds a bonus to the action that commonly 
    leads to the previously visited planet, regardless of reward.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for Model-Based values.
    - planet_stick: [0, 5] Bonus added to the action leading to the previous planet, 
      scaled by OCI.
    """
    learning_rate, beta, w, planet_stick = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    stick_bonus = planet_stick * oci_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_state = -1
    
    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Planet Stickiness
        # If last_state was 0 (Planet X), Action 0 (A) is the common path.
        # If last_state was 1 (Planet Y), Action 1 (U) is the common path.
        if last_state == 0:
            q_net[0] += stick_bonus
        elif last_state == 1:
            q_net[1] += stick_bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Choice ---
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx, :])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_state = s_idx

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
