Here are 3 new cognitive models as Python functions, incorporating the OCI score into different cognitive mechanisms.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with a dynamic Choice Kernel (Habit) where OCI modulates the Habit Learning Rate.
    Hypothesis: OCI affects how quickly participants form or dissolve motor habits (stickiness),
    independent of value-based learning. High OCI might lead to faster habit formation or 
    persistence (slower decay, modeled here as updating speed).
    
    Parameters:
    lr: Value learning rate [0, 1]
    beta: Inverse temperature for value [0, 10]
    w: Model-based / Model-free mixing weight [0, 1]
    lr_habit_base: Baseline learning rate for the habit trace [0, 1]
    lr_habit_oci: OCI modulation of habit learning rate [0, 1]
    """
    lr, beta, w, lr_habit_base, lr_habit_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate habit learning rate constrained to [0,1]
    alpha_habit = lr_habit_base + lr_habit_oci * current_oci
    alpha_habit = np.clip(alpha_habit, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Value functions
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    # Choice Kernel (Habit trace), initialized to 0
    choice_kernel = np.zeros(2)
    
    # Fixed weight for habit to ensure identifiability of alpha_habit
    # We set it to 1.0 so the trace magnitude (0-1) is directly added to Q-values.
    habit_weight = 1.0 

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Combine Value and Habit
        logits_1 = beta * q_net + habit_weight * choice_kernel
        
        # Softmax
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # numeric stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Choice ---
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        
        # Stage 2 Update (MF)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update (MF) - TD(0)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Habit Update: Choice Kernel
        # Update chosen option towards 1, unchosen towards 0
        choice_kernel[a1] += alpha_habit * (1 - choice_kernel[a1])
        choice_kernel[1-a1] += alpha_habit * (0 - choice_kernel[1-a1])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with separate Inverse Temperatures (Beta) for Stage 1 and Stage 2,
    where OCI modulates Stage 1 Beta (Decision Noise).
    Hypothesis: OCI affects decision consistency specifically in the higher-uncertainty
    planning stage (Stage 1), while Stage 2 (simple bandit) remains stable.
    
    Parameters:
    lr: Learning rate [0, 1]
    w: Model-based / Model-free mixing weight [0, 1]
    beta_2: Inverse temperature for Stage 2 [0, 10]
    beta_1_base: Baseline inverse temperature for Stage 1 [0, 10]
    beta_1_oci: OCI modulation of Stage 1 beta [0, 10] (can be negative)
    """
    lr, w, beta_2, beta_1_base, beta_1_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate Stage 1 Beta
    beta_1 = beta_1_base + beta_1_oci * current_oci
    beta_1 = np.clip(beta_1, 0.0, 10.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use Beta 1 for Stage 1
        logits_1 = beta_1 * q_net
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 ---
        # Use Beta 2 for Stage 2
        logits_2 = beta_2 * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model with 'Counterfactual Repulsion' where OCI modulates the update of the unchosen Stage 1 option.
    Hypothesis: OCI is associated with rigid thinking. If one option is chosen and updated (positively or negatively),
    the unchosen option is pushed in the opposite direction (repulsion), creating a stronger contrast between options.
    
    Parameters:
    lr: Learning rate [0, 1]
    beta: Inverse temperature [0, 10]
    w: Mixing weight [0, 1]
    repulse_base: Baseline repulsion rate [0, 1]
    repulse_oci: OCI modulation of repulsion [0, 1]
    """
    lr, beta, w, repulse_base, repulse_oci = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    repulse_param = repulse_base + repulse_oci * current_oci
    # Repulsion is clipped to [0, 1] to keep dynamics stable
    repulse_param = np.clip(repulse_param, 0.0, 1.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits_1 = beta * q_net
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 ---
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Counterfactual Repulsion for the unchosen option
        # We use the SAME delta, but subtract it (scaled) from the unchosen option.
        # If delta is positive (good outcome), unchosen option goes DOWN.
        # If delta is negative (bad outcome), unchosen option goes UP.
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] -= lr * repulse_param * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
