Here are three new cognitive models based on the two-step task and the participant's OCI score.

### Cognitive Model 1: OCI-Modulated Model-Based/Model-Free Trade-off
This model hypothesizes that OCI symptoms specifically impair the Model-Based (planning) system while leaving the Model-Free (habitual) system intact or relatively stronger. High OCI scores act as a "load" or interference, reducing the weight ($w$) placed on model-based value estimation.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI directly reduces Model-Based weighting.
    
    Hypothesis: Higher OCI scores correlate with cognitive rigidity or reduced 
    cognitive flexibility, leading to a lower reliance on the computationally 
    expensive model-based system (w) and higher reliance on model-free habits.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1] (Base mixing weight for MB)
    w_oci_damp: [0, 1] (Strength of OCI-driven reduction of MB weight)
    """
    learning_rate, beta, w_base, w_oci_damp = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective w: Higher OCI reduces w
    # We clip to ensure w stays within [0, 1]
    w = w_base * (1.0 - (w_oci_damp * current_oci))
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])

        # Handle missing/invalid data
        if a1 < 0 or s < 0 or a2 < 0:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        # SARSA / Q-learning at second stage (terminal state, so same)
        delta_2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_2

        # TD(0) update for Stage 1 Model-Free
        delta_1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
            
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: OCI-Enhanced Punishment Sensitivity
This model posits that individuals with higher OCI scores are hypersensitive to negative outcomes (missing a reward). Instead of a single learning rate, this model separates learning from positive rewards vs. zero rewards (perceived as punishment/failure), with the "punishment" learning rate scaled by OCI.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Pure Model-Free model with OCI-modulated Punishment Sensitivity.
    
    Hypothesis: OCI is associated with fear of failure or error. 
    When a reward is 0 (failure), the learning rate is boosted by the OCI score,
    causing the participant to abandon unrewarded options faster than they 
    adopt rewarded ones.
    
    Bounds:
    lr_pos: [0, 1] (Learning rate for reward=1)
    lr_neg_base: [0, 1] (Base learning rate for reward=0)
    beta: [0, 10]
    oci_punish_mult: [0, 5] (Multiplier for OCI effect on negative learning rate)
    """
    lr_pos, lr_neg_base, beta, oci_punish_mult = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Calculate effective negative learning rate
    # Higher OCI -> Higher learning rate when reward is 0
    lr_neg = lr_neg_base * (1.0 + oci_punish_mult * current_oci)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure MF setup (no MB component for simplicity to isolate learning rate effect)
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])

        if a1 < 0 or s < 0 or a2 < 0:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]
        
        # Select learning rate based on outcome
        current_lr = lr_pos if r > 0 else lr_neg

        # Stage 2 Update
        delta_2 = r - q_stage2[s, a2]
        q_stage2[s, a2] += current_lr * delta_2

        # Stage 1 Update (using Q-value of chosen stage 2 state-action)
        delta_1 = q_stage2[s, a2] - q_stage1[a1]
        q_stage1[a1] += current_lr * delta_1
            
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Driven State Transition Uncertainty
This model introduces a parameter representing distrust in the transition structure. While standard models assume the participant knows the 70/30 transition matrix perfectly, this model suggests high OCI participants might perceive the environment as more volatile or uncertain, effectively "flattening" the transition matrix used for Model-Based planning.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model with OCI-modulated Transition Uncertainty.
    
    Hypothesis: High OCI participants may doubt the stability of the transition 
    structure (70/30). This model distorts the internal transition matrix 
    towards 50/50 as OCI increases, making Model-Based planning less distinct 
    from random noise.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    uncertainty_scale: [0, 1] (How much OCI flattens the transition matrix)
    """
    learning_rate, beta, w, uncertainty_scale = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]

    # Standard transition matrix
    true_prob = 0.7
    
    # Distort probability based on OCI
    # If uncertainty_scale * oci is high, prob moves from 0.7 towards 0.5
    distortion = uncertainty_scale * current_oci
    # Ensure we don't flip the relationship, just flatten it
    distortion = np.clip(distortion, 0.0, 0.2) 
    
    perceived_prob = true_prob - distortion
    complement_prob = 1.0 - perceived_prob
    
    # Internal model of transitions
    transition_matrix = np.array([
        [perceived_prob, complement_prob], 
        [complement_prob, perceived_prob]
    ])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])

        if a1 < 0 or s < 0 or a2 < 0:
            p_choice_1[trial] = 1.0
            p_choice_2[trial] = 1.0
            continue

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Use the OCI-distorted matrix for planning
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]

        delta_2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_2

        delta_1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
            
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
