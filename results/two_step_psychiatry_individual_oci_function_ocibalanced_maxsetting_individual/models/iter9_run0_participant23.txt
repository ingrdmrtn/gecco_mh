Here are three new cognitive models for the two-step decision task, incorporating the OCI score to explain individual variability.

### Model 1: OCI-Modulated Transition Learning Rate
This model hypothesizes that participants do not rely on a fixed transition matrix (e.g., 70/30) but actively learn the transition probabilities between spaceships and planets. The **OCI score modulates the learning rate** of these structural beliefs. High OCI (associated with intolerance of uncertainty) may lead to hyper-sensitivity to rare transitions (over-updating the transition matrix) or, conversely, rigidity.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Transition Learning Rate.
    
    The agent learns the transition matrix (Spaceship -> Planet) online.
    OCI modulates the learning rate of these structural beliefs (lr_trans).
    
    Parameters:
    - learning_rate: [0, 1] MF value learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - stickiness: [0, 5] Choice perseveration bonus.
    - lr_trans_base: [0, 1] Baseline learning rate for transition matrix.
    - lr_trans_oci: [-1, 1] Modulation of transition learning rate by OCI.
    """
    learning_rate, beta, w, stickiness, lr_trans_base, lr_trans_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate transition learning rate
    lr_trans = lr_trans_base + lr_trans_oci * oci_score
    lr_trans = np.clip(lr_trans, 0.0, 1.0)

    # Initialize transition matrix with the instructed prior
    # Rows: Action (0/1), Cols: State (0/1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1: continue
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # --- Stage 1 Policy ---
        # Model-Based: T * max(Q_stage2) using dynamic T
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Stickiness
        stick_bonus = np.zeros(2)
        if prev_action_1 != -1:
            stick_bonus[prev_action_1] = stickiness
            
        logits_1 = beta * q_net + stick_bonus
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # MF Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Transition Matrix Update
        # Update the row corresponding to the chosen spaceship (a1)
        # to increase probability of the observed planet (s_idx)
        transition_matrix[a1, s_idx] += lr_trans * (1.0 - transition_matrix[a1, s_idx])
        # Ensure row sums to 1
        transition_matrix[a1, 1 - s_idx] = 1.0 - transition_matrix[a1, s_idx]
        
        prev_action_1 = a1

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Zero-Reward Penalty
This model hypothesizes that for individuals with high OCI, the absence of a reward (0 coins) is not merely neutral but perceived as a **punishment**. The model introduces a penalty parameter $P$ modulated by OCI. When the reward is 0, the effective reward used for updating Q-values is $-P$. This drives stronger avoidance of non-rewarding options than standard extinction ($R=0$).

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Zero-Reward Penalty.
    
    Hypothesis: OCI modulates the subjective utility of receiving 0 coins.
    High OCI acts as a penalty, treating 0 reward as negative utility.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - stickiness: [0, 5]
    - penalty_base: [0, 1] Baseline penalty magnitude for 0 reward.
    - penalty_oci: [-1, 1] Modulation of penalty by OCI.
    """
    learning_rate, beta, w, stickiness, penalty_base, penalty_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    penalty = penalty_base + penalty_oci * oci_score
    # Penalty effectively lowers the asymptote of Q-values below 0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1: continue
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r_raw = reward[trial]
        
        # Apply penalty if reward is 0
        if r_raw == 0:
            r_eff = -penalty
        else:
            r_eff = r_raw
        
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        stick_bonus = np.zeros(2)
        if prev_action_1 != -1:
            stick_bonus[prev_action_1] = stickiness
            
        logits_1 = beta * q_net + stick_bonus
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Choice
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r_eff - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        prev_action_1 = a1

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Stage 1 Value Decay
This model posits that OCI affects the **retention of Model-Free values for the unchosen spaceship**. While the previous best model applied decay to Stage 2 outcomes, this model focuses on the first step. High OCI might lead to rapid devaluation or "forgetting" of the option not currently being pursued, reflecting a focus on the current path or an inability to maintain alternative hypotheses.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Stage 1 Value Decay.
    
    Hypothesis: OCI modulates the decay rate of the unchosen Stage 1 option's value.
    This differs from Stage 2 decay by affecting the high-level choice between spaceships.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - stickiness: [0, 5]
    - decay_s1_base: [0, 1] Baseline decay for unchosen Stage 1 option.
    - decay_s1_oci: [-1, 1] Modulation of Stage 1 decay by OCI.
    """
    learning_rate, beta, w, stickiness, decay_s1_base, decay_s1_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    decay = decay_s1_base + decay_s1_oci * oci_score
    decay = np.clip(decay, 0.0, 1.0)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1

    for trial in range(n_trials):
        if action_1[trial] == -1: continue
        
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        stick_bonus = np.zeros(2)
        if prev_action_1 != -1:
            stick_bonus[prev_action_1] = stickiness
            
        logits_1 = beta * q_net + stick_bonus
        logits_1 = logits_1 - np.max(logits_1)
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Choice
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Decay unchosen Stage 1 MF value
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay)
        
        prev_action_1 = a1

    eps = 1e-10
    valid_mask = (action_1 != -1)
    log_loss = -(np.sum(np.log(p_choice_1[valid_mask] + eps)) + 
                 np.sum(np.log(p_choice_2[valid_mask] + eps)))
    return log_loss
```
