Here are three new cognitive models inspired by the participant's data and OCI score, exploring different mechanisms for how obsessive-compulsive traits might influence reinforcement learning in this two-step task.

### Model 1: OCI-Modulated Model-Based/Model-Free Hybrid with "Checking" Behavior
This model hypothesizes that high OCI scores lead to "checking" behavior, operationalized here as an increased reliance on Model-Free (habitual) values even when the Model-Based system (goal-directed) suggests otherwise, particularly after negative outcomes. The weighting parameter `w` shifts based on OCI, but specifically, the OCI score modulates how much recent negative outcomes push the agent towards safe, habitual repetition (Model-Free) vs. planning (Model-Based).

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid MB/MF model where OCI modulates the mixing weight 'w'.
    
    Hypothesis: Higher OCI leads to reduced reliance on the model-based system (lower w),
    representing a tendency towards habitual or rigid responding rather than flexible
    model-based planning.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature for softmax [0, 10]
    - w_base: Baseline mixing weight (0=MF, 1=MB) [0, 1]
    - w_oci_damp: How strongly OCI reduces the model-based weight [0, 1]
    """
    learning_rate, beta, w_base, w_oci_damp = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate effective w based on OCI. 
    # Higher OCI subtracts from the base weight, pushing towards MF (0).
    w = w_base - (w_oci_damp * current_oci)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Handle missing data
        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- STAGE 1 POLICY ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 POLICY ---
        # Standard Model-Free choice at Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- UPDATES ---
        # Stage 1 MF Update (TD(0))
        # Note: In standard hybrid models, Stage 1 MF is often updated via SARSA or Q-learning 
        # using the stage 2 value. Here we use the value of the state reached.
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Dependent Choice Perseveration (Stickiness)
This model posits that the primary effect of OCI in this task is not on the learning of values themselves, but on the *repetition* of choices (perseveration). High OCI is associated with repetitive behaviors. This model introduces a "stickiness" parameter to the softmax function that is directly scaled by the OCI score.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Based/Model-Free Hybrid with OCI-modulated Perseveration.
    
    Hypothesis: OCI drives 'stickiness' or choice perseveration. Higher OCI leads to
    a stronger tendency to repeat the previous Stage 1 choice, regardless of reward.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature [0, 10]
    - w: Mixing weight (0=MF, 1=MB) [0, 1]
    - stick_oci_factor: How much OCI amplifies the tendency to repeat the last choice [0, 5]
    """
    learning_rate, beta, w, stick_oci_factor = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate stickiness magnitude based on OCI
    stickiness = stick_oci_factor * current_oci

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_choice_1 = -1

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            last_choice_1 = -1
            continue

        # --- STAGE 1 POLICY ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the previously chosen action
        q_net_stick = q_net.copy()
        if last_choice_1 != -1:
            q_net_stick[last_choice_1] += stickiness

        exp_q1 = np.exp(beta * q_net_stick)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]
        
        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- UPDATES ---
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        last_choice_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Dependent Eligibility Trace (Lambda)
This model explores the idea that OCI affects the credit assignment process. Specifically, it modulates the eligibility trace parameter ($\lambda$) in a TD($\lambda$) framework. A higher OCI might lead to "over-attribution" or stronger linking of distal outcomes (Stage 2 reward) to proximal causes (Stage 1 choice), effectively making the agent behave more like a pure Model-Free agent that ignores the transition structure but strongly reinforces the Stage 1 choice based on the final outcome.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free TD(lambda) where OCI modulates the eligibility trace (lambda).
    
    Hypothesis: OCI affects how strongly the Stage 1 choice is reinforced by the Stage 2 reward
    directly. High OCI increases lambda, making the agent more sensitive to the final outcome 
    for the first choice, potentially ignoring the transition structure (common/rare).
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature [0, 10]
    - lambda_base: Base eligibility trace [0, 1]
    - lambda_oci_slope: How OCI modifies lambda [0, 1]
    - w: Mixing weight for completeness (though this model emphasizes MF nuance) [0, 1]
    """
    learning_rate, beta, lambda_base, lambda_oci_slope, w = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0]
    
    # Calculate lambda based on OCI
    eligibility_lambda = lambda_base + (lambda_oci_slope * current_oci)
    eligibility_lambda = np.clip(eligibility_lambda, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s_idx = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a1 == -1 or s_idx == -1 or a2 == -1:
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- STAGE 1 POLICY ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- STAGE 2 POLICY ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- UPDATES ---
        # Stage 2 Prediction Error
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 2 value
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 value using eligibility trace mechanism
        # The Stage 1 choice is updated by its own immediate PE (which is 0 usually in this task structure 
        # as there is no reward at stage 1, but technically it's the value of stage 2 state)
        # PLUS the lambda-weighted Stage 2 PE.
        
        # Standard TD(0) part for Stage 1:
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1] # PE at transition
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Eligibility trace part: Stage 1 gets a piece of Stage 2's surprise
        q_stage1_mf[a1] += learning_rate * eligibility_lambda * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
