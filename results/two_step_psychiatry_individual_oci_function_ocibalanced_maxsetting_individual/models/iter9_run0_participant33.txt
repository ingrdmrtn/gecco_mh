Here are three new cognitive models that explore different mechanisms for how Obsessive-Compulsive symptoms (OCI) might influence decision-making in this two-step task, specifically focusing on how high OCI scores (like this participant's 0.71) might alter learning rates, model-based control, or sensitivity to negative outcomes.

### Cognitive Model 1: OCI-Modulated Model-Based/Model-Free Hybrid
This model tests the hypothesis that high OCI scores relate to a deficit in goal-directed (model-based) control, leading to a greater reliance on habit (model-free) systems. It uses a mixing parameter `w` that shifts from model-based to model-free as OCI increases.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based / Model-Free RL with OCI-dependent Mixing.
    
    Hypothesis: High OCI scores are associated with reduced reliance on the 
    computationally expensive Model-Based system (goal-directed) and increased 
    reliance on the Model-Free system (habitual).
    
    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta: [0, 10] - Inverse temperature for softmax.
    w_base: [0, 1] - Baseline weight for Model-Based control (0=Pure MF, 1=Pure MB).
    w_oci_dec: [0, 1] - How much OCI reduces the Model-Based weight.
    """
    learning_rate, beta, w_base, w_oci_dec = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate the effective mixing weight w.
    # Higher OCI reduces w, making the agent more Model-Free.
    # w is clamped between 0 and 1.
    w = w_base - (w_oci_dec * oci_score)
    if w < 0: w = 0.0
    if w > 1: w = 1.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free Q-values for stage 2

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a1 == -1: # Handle missing data if any
            p_choice_1[trial] = 0.5
            p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Policy ---
        # Model-Based Valuation: V_MB(s1) = T(s1, s2) * max(Q_MF(s2, a2))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Valuation: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Update Stage 2 Q-values (Model-Free)
        delta_stage2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (Model-Free via TD-0)
        # Note: In pure MB/MF models, MF update usually uses Q(s2,a2) as target
        delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Asymmetric Learning Rates based on OCI
This model posits that OCI affects how individuals process positive versus negative prediction errors. High OCI might lead to hyper-sensitivity to "failure" (negative prediction errors) or "success," causing them to update their beliefs differently.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Asymmetric Learning Rates (Positive/Negative) Modulated by OCI.
    
    Hypothesis: High OCI individuals may learn differently from positive vs negative
    outcomes. This model scales the learning rate for negative prediction errors
    based on the OCI score.
    
    Parameters:
    lr_pos: [0, 1] - Learning rate for positive prediction errors (RPE > 0).
    lr_neg_base: [0, 1] - Base learning rate for negative prediction errors (RPE < 0).
    oci_neg_scale: [0, 5] - Scaling factor for OCI on negative learning rate.
    beta: [0, 10] - Inverse temperature.
    """
    lr_pos, lr_neg_base, oci_neg_scale, beta = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective negative learning rate
    # High OCI amplifies the learning from negative outcomes
    lr_neg = lr_neg_base * (1.0 + oci_neg_scale * oci_score)
    if lr_neg > 1.0: lr_neg = 1.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2) 
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        if a1 == -1: 
            p_choice_1[trial] = 0.5; p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Choice (Model-Free only for simplicity in this variant) ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2[s, a2]
        current_lr2 = lr_pos if delta_stage2 >= 0 else lr_neg
        q_stage2[s, a2] += current_lr2 * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2[s, a2] - q_stage1[a1]
        current_lr1 = lr_pos if delta_stage1 >= 0 else lr_neg
        q_stage1[a1] += current_lr1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: OCI-Driven Inverse Temperature (Exploration/Exploitation)
This model suggests that high OCI scores lead to more rigid, deterministic behavior (high beta), reducing exploration. The participant's high OCI score might correlate with a very steep softmax function, making them stick strongly to the perceived best option.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Inverse Temperature (Exploration/Exploitation Balance).
    
    Hypothesis: High OCI is associated with rigidity and reduced exploration.
    This model increases the inverse temperature (beta) as a function of OCI,
    making choices more deterministic (greedy) for higher OCI scores.
    
    Parameters:
    learning_rate: [0, 1] - Update rate for Q-values.
    beta_base: [0, 10] - Baseline inverse temperature.
    beta_oci_slope: [0, 10] - Increase in beta per unit of OCI.
    w: [0, 1] - Weight for Model-Based control (fixed).
    """
    learning_rate, beta_base, beta_oci_slope, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Effective Beta: Higher OCI -> Higher Beta -> Less Randomness/Exploration
    beta = beta_base + (beta_oci_slope * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        a1 = int(action_1[trial])
        s = int(state[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        if a1 == -1:
            p_choice_1[trial] = 0.5; p_choice_2[trial] = 0.5
            continue

        # --- Stage 1 Policy (Hybrid) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        delta_stage2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
