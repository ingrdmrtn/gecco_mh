Here are 3 new cognitive models that incorporate OCI scores, exploring mechanisms of habit accumulation, memory persistence, and subjective loss sensitivity.

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Cumulative Habit Weight.
    
    Hypothesis: High OCI individuals may rely more heavily on established habits (cumulative choice traces)
    rather than just immediate repetition (1-back stickiness). This model accumulates a 'choice trace' 
    for Stage 1 actions and modulates the weight of this trace based on OCI.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-values.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Weight for Model-Based (1) vs Model-Free (0).
    - stick_decay: [0, 1] Decay rate of the choice trace (how fast the habit fades).
    - stick_weight_base: [0, 5] Baseline weight of the choice trace influence.
    - stick_weight_oci: [-5, 5] Modulation of habit weight by OCI.
    """
    learning_rate, beta, w, stick_decay, stick_weight_base, stick_weight_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate OCI-modulated weight for the cumulative habit
    stick_weight = stick_weight_base + (stick_weight_oci * oci_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Cumulative choice trace for Stage 1 (initialized to 0)
    choice_trace = np.zeros(2)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value combines MB and MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add cumulative habit bonus (trace * weight)
        q_net += stick_weight * choice_trace
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Update choice trace: Decay existing trace, then increment chosen action
        a1 = int(action_1[trial])
        choice_trace *= stick_decay
        choice_trace[a1] += 1.0
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning ---
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Update Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF values (using TD(0))
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Q-Value Forgetting.
    
    Hypothesis: OCI scores may relate to the persistence of value representations. 
    High OCI might involve rigid maintenance (low forgetting) or anxiety-driven decay (high forgetting)
    of unchosen options. This model decays the Q-values of unchosen actions towards 0 on every trial.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - stickiness: [0, 5] Standard 1-back stickiness bonus.
    - forget_base: [0, 1] Baseline forgetting rate for unchosen options.
    - forget_oci: [-1, 1] Modulation of forgetting rate by OCI.
    """
    learning_rate, beta, w, stickiness, forget_base, forget_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    forget_rate = forget_base + (forget_oci * oci_score)
    if forget_rate < 0: forget_rate = 0.0
    if forget_rate > 1: forget_rate = 1.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Updates ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Stage 2 update (Chosen)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 2 Forgetting (Unchosen)
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1 - forget_rate)
        
        # Stage 1 update (Chosen)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 1 Forgetting (Unchosen)
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1 - forget_rate)
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Subjective Loss Sensitivity.
    
    Hypothesis: High OCI individuals may perceive the absence of reward (0 coins) as a 
    significant loss (punishment), rather than a neutral outcome. This model transforms 
    the reward signal, treating 0 as a negative value modulated by OCI.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weight.
    - stickiness: [0, 5] Stickiness bonus.
    - loss_sens_base: [0, 5] Baseline sensitivity to non-reward (magnitude of negative reward).
    - loss_sens_oci: [-5, 5] Modulation of loss sensitivity by OCI.
    """
    learning_rate, beta, w, stickiness, loss_sens_base, loss_sens_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate subjective magnitude of loss for 0 reward
    loss_magnitude = loss_sens_base + (loss_sens_oci * oci_score)
    if loss_magnitude < 0: loss_magnitude = 0.0 
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # --- Learning ---
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r_raw = reward[trial]
        
        # Transform reward: 1 -> 1, 0 -> -loss_magnitude
        # This creates a contrast where 0 is treated as a punishment.
        if r_raw == 1:
            r_eff = 1.0
        else:
            r_eff = -loss_magnitude
            
        # Update Stage 2
        delta_stage2 = r_eff - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
