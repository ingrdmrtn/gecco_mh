Here are the three cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Subjective Disutility of Zero Reward.
    
    Hypothesis: High OCI participants perceive a 'zero coin' outcome not as a neutral event,
    but as a negative outcome (a loss), driven by perfectionism or fear of failure.
    This expands the value contrast between wins and losses, driving stronger avoidance 
    of non-rewarding options compared to low OCI participants.
    
    Parameters:
    - learning_rate: [0,1] Rate of value updating.
    - beta: [0,10] Inverse temperature (exploration/exploitation balance).
    - w: [0,1] Weight of Model-Based control (0=MF, 1=MB).
    - stickiness: [0,5] Tendency to repeat the previous choice.
    - zero_disutility_oci: [0,5] Scaling factor for the negative value of zero rewards.
      Effective Reward (if 0) = -1.0 * zero_disutility_oci * oci.
    """
    learning_rate, beta, w, stickiness, zero_disutility_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        # Softmax
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Calculate Effective Reward based on OCI
        r_raw = reward[trial]
        if r_raw == 0:
            r_effective = -1.0 * zero_disutility_oci * oci_score
        else:
            r_effective = 1.0
            
        # Stage 1 MF Update (TD(1)-like)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r_effective - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Additional Stage 1 update from Stage 2 RPE (Eligibility Trace assumption)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Structure Bonus (Intrinsic Reward).
    
    Hypothesis: High OCI participants derive intrinsic value from the environment behaving 
    predictably (Common transitions). They receive a 'structure bonus' added to the 
    Model-Free update when a Common transition occurs, reinforcing 'correct' spaceship choices
    independent of the coin reward outcome. This leads to rigidity in choice.
    
    Parameters:
    - learning_rate: [0,1] Rate of value updating.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Model-based weight.
    - stickiness: [0,5] Choice perseveration.
    - structure_bonus_oci: [0,2] Magnitude of intrinsic reward for common transitions, scaled by OCI.
      Bonus = structure_bonus_oci * oci (if Common).
    """
    learning_rate, beta, w, stickiness, structure_bonus_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Determine if transition was Common or Rare
        # Common: 0->0 or 1->1. Rare: 0->1 or 1->0.
        is_common = (a1 == state_idx)
        
        bonus = 0.0
        if is_common:
            bonus = structure_bonus_oci * oci_score
            
        # Stage 1 MF Update
        # The bonus is added to the value of the state reached, effectively boosting the chosen action
        target_val = q_stage2_mf[state_idx, a2] + bonus
        delta_stage1 = target_val - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update (Standard)
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility trace for Stage 1 (propagate reward error)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    OCI-Modulated Rare Transition Learning Suppression.
    
    Hypothesis: High OCI participants display rigidity and confirmation bias. When a 
    Rare transition occurs (unexpected outcome), they suppress the learning update (prediction error),
    treating it as noise rather than a signal to update values. This preserves their 
    existing habits/estimates against 'bad luck'.
    
    Parameters:
    - learning_rate: [0,1] Base learning rate.
    - beta: [0,10] Inverse temperature.
    - w: [0,1] Model-based weight.
    - stickiness: [0,5] Choice perseveration.
    - rare_suppression_oci: [0,1] Factor reducing learning rate on Rare trials.
      LR_rare = LR_base * (1 - rare_suppression_oci * oci).
    """
    learning_rate, beta, w, stickiness, rare_suppression_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        state_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        # Determine effective learning rate for this trial
        is_common = (a1 == state_idx)
        
        if is_common:
            current_lr = learning_rate
        else:
            # Rare transition: apply suppression based on OCI
            suppression = rare_suppression_oci * oci_score
            # Clip suppression to max 1.0 to prevent negative LR
            if suppression > 1.0: suppression = 1.0
            current_lr = learning_rate * (1.0 - suppression)
            
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        # Stage 2 MF Update
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += current_lr * delta_stage2
        
        # Eligibility trace update
        q_stage1_mf[a1] += current_lr * delta_stage2
        
        last_action_1 = a1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
