import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None
):
    """
    Simulates choices and rewards using the OCI-Modulated Asymmetric Learning Rate Model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_pos, lr_neg_base, lr_neg_oci, beta, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr_pos, lr_neg_base, lr_neg_oci, beta, w, stickiness = parameters

    # Handle OCI input
    if isinstance(oci, (list, np.ndarray)):
        oci_val = oci[0]
    else:
        oci_val = oci if oci is not None else 0.0

    # Calculate asymmetric learning rate for negative prediction errors
    lr_neg = lr_neg_base + lr_neg_oci * oci_val
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    # Initialize variables
    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    # Stage 1 MF values (actions 0, 1)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF values (states 0, 1 x actions 0, 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # History for stickiness
    prev_a1 = -1

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # --- Stage 1 ---
        
        # Model-Based Value Calculation
        # Max Q-value for each second-stage state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted Net Q-values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus
        stick_bonus = np.zeros(2)
        if prev_a1 != -1:
            stick_bonus[prev_a1] = stickiness
            
        # Softmax choice probability for Stage 1
        logits_1 = beta * (q_net + stick_bonus)
        # Shift for numerical stability
        logits_1 = logits_1 - np.max(logits_1) 
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State index 0 or 1
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 ---
        
        # Softmax choice probability for Stage 2
        logits_2 = beta * q_stage2_mf[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Get reward probability from drifts
        if s_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # --- Learning Updates ---
        
        # 1. Update Stage 1 MF based on Stage 2 Q-values (TD Error 1)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # Apply asymmetric learning rate
        alpha_1 = lr_pos if delta_stage1 > 0 else lr_neg
        q_stage1_mf[a1] += alpha_1 * delta_stage1
        
        # 2. Update Stage 2 MF based on Reward (TD Error 2)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Apply asymmetric learning rate
        alpha_2 = lr_pos if delta_stage2 > 0 else lr_neg
        q_stage2_mf[s_idx, a2] += alpha_2 * delta_stage2
        
        # 3. Update Stage 1 MF based on Stage 2 error (Eligibility Trace)
        # Note: The fitting code applies the update directly using alpha_2 * delta_stage2
        q_stage1_mf[a1] += alpha_2 * delta_stage2
        
        # Update history
        prev_a1 = a1
        
        # Record data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
