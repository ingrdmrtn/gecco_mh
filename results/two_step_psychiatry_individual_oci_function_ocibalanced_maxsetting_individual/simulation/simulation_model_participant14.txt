import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Based learner with choice perseveration 
    (stickiness) modulated by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, stickiness_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr, beta, stickiness_base = parameters
    
    # Handle OCI input format (scalar, list, or None) to match fit code logic
    if oci is None:
        oci_score = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate effective stickiness based on OCI
    eff_stickiness = stickiness_base * (1.0 + oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values for Stage 2 (2 states x 2 actions)
    # Row 0: State 0, Row 1: State 1
    q_stage2_mf = np.zeros((2, 2)) 

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_a1 = -1 

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Apply stickiness to the previous choice
        q_stage1_augmented = q_stage1_mb.copy()
        if last_a1 != -1:
            q_stage1_augmented[last_a1] += eff_stickiness
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_augmented)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 using same beta
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        # Get reward probability from drifts
        # State 0 uses drift1 (act 0) and drift2 (act 1)
        # State 1 uses drift3 (act 0) and drift4 (act 1)
        if s_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < p_reward else 0

        # --- Learning ---
        # Update Stage 2 Model-Free values (Standard Q-Learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2

        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
