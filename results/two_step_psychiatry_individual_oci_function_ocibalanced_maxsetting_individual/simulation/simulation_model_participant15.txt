import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-Modulated Win-Stay Stickiness.
    
    This model hypothesizes that OCI symptoms increase the tendency to compulsively 
    repeat a successful action ("Win-Stay"). Stickiness is not constant but is 
    boosted by OCI specifically when the previous trial was rewarded.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, stick_base, stick_win_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list/array): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, w, stick_base, stick_win_oci = parameters
    
    # Handle OCI input (can be float or list based on context)
    if isinstance(oci, (list, np.ndarray, tuple)):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0.0

    # Fixed transition matrix as per fitting code
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF Q-values (2 states * 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Tracking variables for stickiness
    last_action_1 = -1
    last_reward = 0.0
    
    # Random number generator
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the two second-stage states
        # State 0 (index 0) corresponds to drift1/drift2
        # State 1 (index 1) corresponds to drift3/drift4
        current_drifts = [[drift1[t], drift2[t]], 
                          [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based Q-values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-Modulated Win-Stay Stickiness
        if last_action_1 != -1:
            # Stickiness is base + (boost * oci * previous_reward)
            stick_val = stick_base + (stick_win_oci * oci_score * last_reward)
            q_net[last_action_1] += stick_val
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 (pure Model-Free)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        
        # Determine reward based on drift probabilities
        r = 1 if rng.random() < current_drifts[s_idx][a2] else 0
        
        # --- Learning Updates ---
        
        # Update Stage 1 MF Q-values (TD(0) using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Update Stage 2 MF Q-values (TD(0) using Reward)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2
        
        # Update history
        last_action_1 = a1
        last_reward = float(r)
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
