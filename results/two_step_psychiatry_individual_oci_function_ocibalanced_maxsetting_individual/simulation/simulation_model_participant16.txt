import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid learner with separate inverse temperatures
    for Stage 1 and Stage 2, where OCI modulates the beta for Stage 2.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, w, beta_s1, beta_s2_base, beta_s2_oci_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, w, beta_s1, beta_s2_base, beta_s2_oci_mod = parameters
    
    # Handle OCI input (assuming single scalar or list with one element as in fitting code)
    if oci is None:
        current_oci = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        current_oci = oci[0]
    else:
        current_oci = oci

    # Calculate effective beta for stage 2 based on OCI
    eff_beta_s2 = beta_s2_base + (beta_s2_oci_mod * current_oci)
    # Apply bounds as defined in the fitting model
    if eff_beta_s2 > 10.0: eff_beta_s2 = 10.0
    if eff_beta_s2 < 0.0: eff_beta_s2 = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (model-free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (model-free)
    # Dimensions: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities for the two states (0 and 1)
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta_s1 * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State index 0 or 1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 using effective beta
        exp_q2 = np.exp(eff_beta_s2 * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning ---
        # Update Stage 2 Q-values (SARSA / Q-learning logic)
        pe_2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * pe_2

        # Update Stage 1 Q-values (TD(1) style update based on Stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for Stage 1
        pe_1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
