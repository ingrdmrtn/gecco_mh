import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None
):
    """
    Simulates the Confidence-Modulated Beta Model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta_base, w, stickiness, conf_sens_low, conf_sens_high]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta_base, w, stickiness, conf_sens_low, conf_sens_high = parameters
    
    # Handle OCI input (can be list/array or scalar)
    if oci is not None:
        if hasattr(oci, '__len__'):
            current_oci = oci[0]
        else:
            current_oci = oci
    else:
        current_oci = 0.0

    # Calculate effective confidence sensitivity based on OCI
    conf_sens = conf_sens_low * (1 - current_oci) + conf_sens_high * current_oci

    # Initialization of Q-values
    # Stage 1 MF: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF: 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    last_reward = 0.5 # Initialize neutral confidence
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Effective Beta (Confidence Modulation)
        # Beta is modulated by recent success (last_reward)
        beta_eff = beta_base * (1 + conf_sens * (last_reward - 0.5))
        beta_eff = max(0.0, beta_eff) # Ensure beta is non-negative

        # 2. Stage 1 Choice (Hybrid MB/MF)
        # Model-Based: Expected value of next stage states
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Stickiness (Perseveration)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta_eff * q_net)
        # Handle potential overflow/underflow
        if np.sum(exp_q1) == 0 or np.isinf(np.sum(exp_q1)):
            probs_1 = np.ones(2) / 2.0
        else:
            probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # 3. Transition to Stage 2
        # Transition probabilities depend on action 1
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # 4. Stage 2 Choice
        # Softmax for Stage 2 using MF values of the current state
        exp_q2 = np.exp(beta_eff * q_stage2_mf[s_idx])
        if np.sum(exp_q2) == 0 or np.isinf(np.sum(exp_q2)):
            probs_2 = np.ones(2) / 2.0
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = rng.choice([0, 1], p=probs_2)

        # 5. Reward Outcome
        # Determine probability based on current state and action
        if s_idx == 0:
            prob_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_r = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < prob_r)

        # 6. Learning Updates
        # Update Stage 2 Q-value (TD Error)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr * delta_stage2

        # Update Stage 1 Q-value (TD Error passed back from Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # 7. Update History
        last_action_1 = a1
        last_reward = r
        
        # Store Data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
