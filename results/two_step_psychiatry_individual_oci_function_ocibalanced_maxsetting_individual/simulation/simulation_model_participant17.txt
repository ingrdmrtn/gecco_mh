import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates the Model with Dynamic Model-Based Weight Decay Modulated by OCI.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_init, w_decay_base, w_decay_oci, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w_init, w_decay_base, w_decay_oci, stickiness = parameters
    
    # Handle OCI input
    oci_val = oci if oci is not None else 0.0
    
    # Initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Data storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    rng = np.random.default_rng()
    
    # Pre-calculate decay rate constant part
    decay_rate = w_decay_base + (w_decay_oci * oci_val)

    for trial in range(n_trials):
        # 1. Calculate dynamic MB weight (w)
        t_norm = trial / n_trials
        w_curr = w_init * np.exp(-decay_rate * t_norm * 5.0)
        w_curr = np.clip(w_curr, 0.0, 1.0)
        
        # 2. Stage 1 Choice
        # Calculate MB values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values
        q_net = w_curr * q_stage1_mb + (1 - w_curr) * q_stage1_mf
        
        # Calculate Logits
        logits_1 = beta * q_net
        
        # Apply Stickiness
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Softmax Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Outcome
        # Map state/action to specific drift array
        # State 0: drift1 (action 0), drift2 (action 1)
        # State 1: drift3 (action 0), drift4 (action 1)
        if s2 == 0:
            prob_r = drift1[trial] if a2 == 0 else drift2[trial]
        else:
            prob_r = drift3[trial] if a2 == 0 else drift4[trial]
            
        r = 1 if rng.random() < prob_r else 0
        
        # 6. Learning Updates
        # TD Errors
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        # Update Q-values
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # 7. Store Data
        stage1_choice[trial] = a1
        state2[trial] = s2
        stage2_choice[trial] = a2
        reward[trial] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
