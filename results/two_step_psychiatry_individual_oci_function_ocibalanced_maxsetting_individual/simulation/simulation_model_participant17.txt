import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free RL model with OCI-modulated 
    choice perseveration (stickiness).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_factor, w]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list): Obsessive-Compulsive Inventoryâ€“Revised score. If list/array, takes first element.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_factor, w = parameters
    
    # Handle OCI input (can be scalar or list/array based on fitting code usage)
    if hasattr(oci, '__len__'):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf corresponds to Q(s1, a)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q(s2, a), shape (2, 2) for two states
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities for second stage states
        # state 0 (s1 in fitting code) -> drift1, drift2
        # state 1 (s2 in fitting code) -> drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MF and MB
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-modulated stickiness
        aug_q_net = q_net.copy()
        if last_action_1 != -1:
            aug_q_net[last_action_1] += (oci_score * stickiness_factor)
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * aug_q_net)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State index 0 or 1
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (pure MF)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning Updates ---
        
        # Update Stage 1 MF Q-value (TD(0))
        # Note: The fitting code uses q_stage2_mf[s1, a2] as the target, not max Q
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] = q_stage1_mf[a1] + learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] = q_stage2_mf[s2_idx, a2] + learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update history
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
