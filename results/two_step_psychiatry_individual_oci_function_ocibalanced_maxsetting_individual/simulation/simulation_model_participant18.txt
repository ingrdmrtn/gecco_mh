import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using Model 2: OCI-influenced Perseveration (Stickiness).
    
    This model simulates a pure Model-Free learner where Stage 1 and Stage 2 values 
    are updated independently based on the final reward. It includes a stickiness 
    parameter that biases the repetition of the previous Stage 1 choice, scaled by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.
        oci (float or list): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, stickiness_oci = parameters
    
    # Handle OCI input (can be scalar or single-element array/list)
    if np.ndim(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    eff_stickiness = stickiness_base + (stickiness_oci * oci_score)

    # Initialize data storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values (matching fitting code initialization)
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    last_a1 = -1 

    # Standard 2-step task transition matrix
    # Action 0 -> 0.7 prob State 0, 0.3 prob State 1
    # Action 1 -> 0.3 prob State 0, 0.7 prob State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        q_stage1_modified = q_stage1.copy()
        
        # Apply perseveration bonus if a previous choice exists
        if last_a1 != -1:
            q_stage1_modified[last_a1] += eff_stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modified)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Select Action 1
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine Stage 2 state based on transition probabilities
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (using same beta as fitting code)
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Select Action 2
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward Generation ---
        # Map state/action to specific drift parameter
        # State 0: drift1 (a2=0), drift2 (a2=1)
        # State 1: drift3 (a2=0), drift4 (a2=1)
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.rand() < p_reward else 0
        
        # --- Value Updates ---
        # Update Stage 1 Q-value (Direct RL from outcome, ignoring transition structure)
        delta_stage1 = r - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2
        
        # Update history
        last_a1 = a1
        
        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
