import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 3: OCI-Modulated Gating of Learning from Rare Transitions.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, stick, rare_gating_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, w, stick, rare_gating_oci = parameters
    
    # Ensure OCI is a scalar
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = float(oci)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows), 2 actions (cols)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix: rows=action1, cols=state2
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_choice_1 = -1
    
    for t in range(n_trials):
        # 1. Stage 1 Choice
        
        # Calculate Model-Based values
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits_1 = beta * q_net
        
        # Apply stickiness (perseveration)
        if last_choice_1 != -1:
            logits_1[last_choice_1] += stick
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # 2. Transition
        # Determine second stage state
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        logits_2 = beta * q_stage2_mf[s2]
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get reward probability for the specific state/action pair
        # Map drifts to states: State 0 -> (drift1, drift2), State 1 -> (drift3, drift4)
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(np.random.random() < prob_reward)
        
        # 5. Learning
        is_rare = (a1 != s2)
        
        # Calculate effective learning rate based on OCI and rarity
        lr_eff = lr
        if is_rare:
            factor = 1.0 - (rare_gating_oci * oci_score)
            if factor < 0:
                factor = 0
            lr_eff = lr * factor
            
        # Update Stage 1 Q-values (TD(0) using Stage 2 Q-value as target)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_eff * delta_stage1
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr_eff * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_choice_1 = a1

    return stage1_choice, state2, stage2_choice, reward
