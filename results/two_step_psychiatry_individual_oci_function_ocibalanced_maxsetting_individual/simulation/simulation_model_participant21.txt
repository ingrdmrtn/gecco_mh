import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free Learner with OCI-Dependent Choice Perseveration.
    
    The 'stickiness' parameter adds a bonus to the previously chosen action, 
    making repetition more likely. This bonus is scaled by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, stickiness_base, stickiness_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, stickiness_base, stickiness_oci = parameters
    
    # Handle OCI input (default to 0 if None, though usually provided)
    current_oci = oci if oci is not None else 0.0

    # Calculate the effective stickiness (phi)
    phi = stickiness_base + stickiness_oci * current_oci

    # Transition matrix (standard for 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: Q-values for the first stage actions (0, 1)
    q_stage1 = np.zeros(2)
    # q_stage2: Q-values for the second stage states (0, 1) and actions (0, 1)
    q_stage2 = np.zeros((2, 2))

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1 # No previous action for first trial

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # state 0 corresponds to drift1/drift2, state 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_stage1_modified = q_stage1.copy()
        
        # Apply stickiness bonus if there was a previous action
        if prev_action_1 != -1:
            q_stage1_modified[prev_action_1] += phi
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_modified)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # Determine transition to Stage 2
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Note: In the fitting code, state was indexed directly. Here, s2 is 0 or 1.
        exp_q2 = np.exp(beta * q_stage2[s2])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Model-Free) ---
        
        # Second stage update (TD Error 2)
        rpe_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += lr * rpe_2
        
        # First stage update (TD Error 1)
        # Using max_q2_next as the estimate of value of the next state (Q-learning style update from fitting code)
        max_q2_next = np.max(q_stage2[s2])
        rpe_1 = max_q2_next - q_stage1[a1]
        q_stage1[a1] += lr * rpe_1

        # Update tracking variables
        prev_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
