import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with OCI-modulated Perseveration Decay.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, pers_mag, retention_oci_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, w, pers_mag, retention_oci_sens = parameters
    
    # Handle OCI input: if it's a list/array (like in fitting code), take first element, else use as is.
    if isinstance(oci, (list, np.ndarray)):
        current_oci = oci[0]
    else:
        current_oci = oci

    # Calculate retention based on OCI
    retention = retention_oci_sens * current_oci
    retention = np.clip(retention, 0.0, 0.95)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values and traces
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    pers_trace = np.zeros(2)

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Decision
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + pers_trace
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # 2. Update Perseveration Trace
        # Mirroring fitting code: decay first, then add magnitude to chosen action
        pers_trace *= retention
        pers_trace[a1] += pers_mag

        # 3. Transition to Stage 2
        # Transition probabilities depend on action a1
        s_probs = transition_matrix[a1]
        s = rng.choice([0, 1], p=s_probs)

        # 4. Stage 2 Decision
        # Softmax for Stage 2 using MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 5. Reward Generation
        # Map state/action to specific drift array
        if s == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
        
        r = int(rng.random() < prob_reward)

        # 6. Learning Updates
        # Update Stage 2 Q-value
        delta_2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += learning_rate * delta_2

        # Update Stage 1 Q-value (using updated Stage 2 Q-value as target)
        delta_1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
