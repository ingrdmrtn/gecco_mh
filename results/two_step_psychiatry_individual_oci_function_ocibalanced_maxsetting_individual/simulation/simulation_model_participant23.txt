import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Stage-Specific Beta Asymmetry model.
    
    Hypothesis: OCI modulates the exploration-exploitation balance differently 
    across hierarchical levels. High OCI participants are hypothesized to be 
    more rigid/deterministic (higher inverse temperature beta) at the first 
    stage (Spaceship selection) relative to the second stage (Alien selection).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta_stage2, w, stickiness, beta_s1_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta_stage2, w, stickiness, beta_s1_slope = parameters
    
    # Handle OCI input (default to 0 if not provided or if it's an array)
    if oci is None:
        oci_score = 0.0
    elif isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    # Calculate beta for stage 1 based on the hypothesis
    beta_stage1 = beta_stage2 * (1.0 + beta_s1_slope * oci_score)

    # Initialize data structures
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (0, 1) initialized to 0.5
    q_stage1_mf = np.ones(2) * 0.5
    # q_stage2_mf: Model-free values for stage 2 states (0, 1) and actions (0, 1) initialized to 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize stickiness tracking
    last_action_1 = -1
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        # Softmax choice for Stage 1 using beta_stage1
        exp_q1 = np.exp(beta_stage1 * q_net_1)
        if np.sum(exp_q1) == 0: 
            exp_q1 = np.ones(2)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 using beta_stage2
        exp_q2 = np.exp(beta_stage2 * q_stage2_mf[s])
        if np.sum(exp_q2) == 0: 
            exp_q2 = np.ones(2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        r = int(rng.random() < reward_probs[s][a2])
        
        # --- Learning ---
        
        # Update Stage 2 Q-values (SARSA / TD(0))
        # Note: The fitting code uses: q_stage2_mf[s, a2] += lr * (r - q_stage2_mf[s, a2])
        q_stage2_mf[s, a2] += lr * (r - q_stage2_mf[s, a2])
        
        # Update Stage 1 Q-values (TD(1) logic used in fitting code)
        # Note: The fitting code uses: q_stage1_mf[a1] += lr * (q_stage2_mf[s, a2] - q_stage1_mf[a1])
        # It updates Stage 1 based on the *updated* Stage 2 value.
        q_stage1_mf[a1] += lr * (q_stage2_mf[s, a2] - q_stage1_mf[a1])
        
        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
