import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free RL model with OCI-modulated Perseveration.

    The model assumes the participant is primarily model-free but has a tendency 
    to repeat choices (perseveration). The strength of this perseveration is 
    a function of the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, perseveration_base, oci_scaling]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, perseveration_base, oci_scaling = parameters
    
    # Handle OCI input (default to 0 if None, though fitting code implies it's required)
    oci_score = oci if oci is not None else 0.0
    
    # Calculate the effective perseveration weight
    pers_weight = perseveration_base + (oci_scaling * oci_score)

    # Initialize Q-values
    # q_stage1 corresponds to Q(s1, a)
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to Q(s2, a) where s2 is index 0 or 1
    q_stage2 = np.zeros((2, 2))
    
    # Transition structure (Standard 2-step task transition probabilities)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward_out = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()
    
    last_action_1 = -1 # Initialize previous action for perseveration

    for t in range(n_trials):
        # 1. Stage 1 Choice
        # Add perseveration bonus to the previously chosen action's Q-value
        q_stage1_eff = q_stage1.copy()
        if last_action_1 != -1:
            q_stage1_eff[last_action_1] += pers_weight
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_stage1_eff)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Update last action
        last_action_1 = a1

        # 2. Transition to Stage 2
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Outcome
        # Get reward probability for the specific state/action pair from drifts
        # Mapping: s_idx 0 -> (drift1, drift2), s_idx 1 -> (drift3, drift4)
        if s_idx == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)

        # 5. Learning Updates (Model-Free)
        # Update Stage 2 Q-values
        delta2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta2

        # Update Stage 1 Q-values (Direct reinforcement from reward)
        # Note: The fitting code uses 'r - q_stage1[a1]', implying simple TD(1) or direct reinforcement
        # rather than TD(0) (r + Q2 - Q1).
        delta1 = r - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward_out[t] = r

    return stage1_choice, state2, stage2_choice, reward_out
