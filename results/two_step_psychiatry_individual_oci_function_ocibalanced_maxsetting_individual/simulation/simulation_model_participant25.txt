import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free learner with choice perseveration (stickiness) 
    where the stickiness parameter is modulated by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stick_base, oci_stick_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stick_base, oci_stick_mod = parameters
    
    # Calculate effective stickiness based on OCI
    # Note: OCI is passed as a float/int or list. Handling both cases safely.
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci
        
    stickiness = stick_base + (oci_score * oci_stick_mod)

    # Initialize Q-values
    # q_stage1 corresponds to Q(s1, a)
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to Q(s2, a) for the two second-stage states
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed 70/30 structure common in 2-step tasks)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = -1

    for t in range(n_trials):
        # 1. Stage 1 Choice
        q_stage1_effective = q_stage1.copy()
        
        # Apply stickiness to the previous action if it exists
        if prev_a1 != -1:
            q_stage1_effective[prev_a1] += stickiness
            
        exp_q1 = np.exp(beta * q_stage1_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Determine the second state (s_idx) based on transition probabilities
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Get current trial reward probabilities
        current_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_probs[s_idx][a2])
        
        # 5. Learning (Update Q-values)
        # Update Stage 2 Q-values (Standard TD error)
        delta_2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * delta_2

        # Update Stage 1 Q-values (Model-Free update using the final reward)
        # Note: The fitting code uses 'r' directly to update Stage 1, implying a simple 
        # TD(1) or direct reinforcement of Stage 1 choice by the final outcome.
        delta_1 = r - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1
        
        # Store history
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
