import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4, oci):
    """
    Simulates Model 1: OCI-Modulated Inverse Temperature Asymmetry.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, w, beta_win, beta_loss_base, beta_loss_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, w, beta_win, beta_loss_base, beta_loss_oci = parameters
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Calculate effective beta for losses based on OCI
    # beta_loss = beta_loss_base + beta_loss_oci * oci
    beta_loss = beta_loss_base + (beta_loss_oci * oci)
    beta_loss = np.maximum(0.0, beta_loss)
    
    # Initialization
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # state x action
    
    prev_reward = 1 # Initialize as if previous was a win (start confident)
    
    for t in range(n_trials):
        # 1. Select Beta based on previous outcome
        if prev_reward == 1:
            current_beta = beta_win
        else:
            current_beta = beta_loss
        
        # 2. Stage 1 Choice
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net values (Weighted average of MB and MF)
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax for Stage 1
        logits_1 = current_beta * q_net_1
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # Determine state based on transition matrix probabilities for a1
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        # Softmax for Stage 2 (using MF values directly)
        logits_2 = current_beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # 5. Outcome
        # Get reward probability for current state/action from drifts
        if s_idx == 0:
            prob_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_r = drift3[t] if a2 == 0 else drift4[t]
        
        r = int(np.random.random() < prob_r)
        
        # 6. Learning Updates
        # TD Error Stage 1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Error Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Eligibility Trace / Update Stage 1 with Stage 2 RPE
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        # 7. Update History and Storage
        prev_reward = r
        
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
