import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards based on the provided fitting code (cognitive_model3).
    
    Model Description:
    Hybrid Model-Based/Model-Free model where the exploration-exploitation trade-off (beta)
    is modulated by the OCI score.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, w]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_base, w = parameters
    
    # Handle OCI modulation
    # In the fitting code, oci is accessed as oci[0]. 
    # We handle cases where oci might be a scalar, list, or None (defaulting to 0).
    oci_val = 0.0
    if oci is not None:
        if np.ndim(oci) > 0:
            oci_val = oci[0]
        else:
            oci_val = oci

    effective_beta = beta_base * (1.0 + oci_val)
    
    # Initialize structures
    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Expected value of next states based on transition probabilities
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integration of MB and MF values
        q_integrated = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax selection
        exp_q1 = np.exp(effective_beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        # transition_matrix[a1] gives [prob_state_0, prob_state_1]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        exp_q2 = np.exp(effective_beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Get reward probability for the current state/action pair
        # State 0 corresponds to drift1/drift2, State 1 to drift3/drift4
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # --- Updates ---
        
        # Update Stage 2 Q-values (TD Error)
        delta2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta2
        
        # Update Stage 1 Q-values
        # The fitting code uses the updated q_stage2_mf value for the TD target
        delta1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta1
        
        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
