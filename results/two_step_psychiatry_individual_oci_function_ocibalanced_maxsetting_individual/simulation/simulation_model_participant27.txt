import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_base, beta, asymmetry_param, decay]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).

    else:
        current_oci = oci

    # Calculate modulated learning rates based on OCI
    mod = 1.0 + (asymmetry_param * current_oci)
    mod = np.clip(mod, 0.1, 2.0) 
    
    lr_pos = np.clip(lr_base * mod, 0, 1)
    lr_neg = np.clip(lr_base * (1/mod), 0, 1)

    # Initialize Q-values
    # q_stage1 corresponds to the two actions in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two states (rows) and two actions (cols) in the second stage
    q_stage2 = np.zeros((2, 2))

    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Choice
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # 2. Transition to Stage 2

        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 4. Outcome

        # Organize drifts: state 0 uses drift1/drift2, state 1 uses drift3/drift4
        if s_idx == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)

        # 5. Learning Update (matching the fitting code logic)
        
        # Update Stage 2 Q-values
        pe2 = r - q_stage2[s_idx, a2]
        effective_lr2 = lr_pos if pe2 >= 0 else lr_neg
        q_stage2[s_idx, a2] += effective_lr2 * pe2

        # Decay unchosen Stage 2 option
        unchosen_a2 = 1 - a2
        q_stage2[s_idx, unchosen_a2] *= (1 - decay)

        # Update Stage 1 Q-values

        # Decay unchosen Stage 1 option

        q_stage1[unchosen_a1] *= (1 - decay)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
