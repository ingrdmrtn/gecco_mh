import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4, oci):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with separate learning rates
    and OCI-modulated stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_s1, lr_s2, beta, w, stickiness_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (list/array): Obsessive-Compulsive Inventory score [score].

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    # Unpack parameters
    lr_s1, lr_s2, beta, w, stickiness_base = parameters
    
    # Handle OCI input (fitting code accesses index 0)
    try:
        oci_score = oci[0]
    except (TypeError, IndexError):
        oci_score = oci if oci is not None else 0.0

    # Calculate modulated stickiness
    stickiness = stickiness_base * (1.0 + oci_score)

    rng = np.random.default_rng()
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State (0,1) x Action (0,1)
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # Organize reward probabilities for current trial: State 0 -> [d1, d2], State 1 -> [d3, d4]
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness to the previously chosen action
        if last_action_1 != -1:
            q_net_1[last_action_1] += stickiness
            
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        p_reward = current_drifts[s_idx][a2]
        r = 1 if rng.random() < p_reward else 0

        # --- Learning Updates ---
        # Update Stage 1 MF value based on prediction of Stage 2 value
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_s1 * delta_stage1

        # Update Stage 2 MF value based on reward
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_s2 * delta_stage2

        # Update Stage 1 MF value based on Stage 2 RPE (Eligibility trace logic)
        q_stage1_mf[a1] += lr_s1 * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
