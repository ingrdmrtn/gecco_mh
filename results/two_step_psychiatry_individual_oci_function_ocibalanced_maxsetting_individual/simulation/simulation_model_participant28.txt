import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Two-step model with Reinforced Stickiness 
    modulated by OCI.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, p_gen, p_rew_base, p_rew_oci_mod, lambda_val]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, p_gen, p_rew_base, p_rew_oci_mod, lambda_val = parameters
    
    # Initialize Random Number Generator
    rng = np.random.default_rng()

    # Handle OCI modulation logic
    # In fitting code, oci is a list/array, here it is a float.
    oci_score = oci if oci is not None else 0.0
    
    # Calculate modulated reinforced stickiness
    p_rew = p_rew_base + (p_rew_oci_mod - 0.5) * oci_score
    p_rew = np.clip(p_rew, 0.0, 1.0)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Model-free values for Stage 1 actions (0, 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for Stage 2. Rows: State (0, 1), Cols: Action (0, 1)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    last_reward = 0

    for t in range(n_trials):
        # --- Stage 1 ---
        
        # Model-Based Calculation: Transition * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits_1 = beta * q_net
        
        # Apply Stickiness (Perseveration)
        if last_action_1 != -1:
            # General stickiness
            stickiness_bonus = p_gen * 5.0
            
            # Reinforced stickiness (modulated by OCI)
            if last_reward == 1.0:
                stickiness_bonus += p_rew * 5.0
            
            logits_1[last_action_1] += stickiness_bonus

        # Action Selection Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine next state based on transition matrix
        # State 0 corresponds to drifts 1&2, State 1 corresponds to drifts 3&4
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 ---
        
        # Action Selection Stage 2
        logits_2 = beta * q_stage2_mf[s_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        if s_idx == 0:
            prob_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_r = drift3[t] if a2 == 0 else drift4[t]
        
        r = 1 if rng.random() < prob_r else 0

        # --- Value Updates (Learning) ---
        
        # TD Errors
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 1 MF values (with eligibility trace lambda)
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_val * delta_stage2
        
        # Update Stage 2 MF values
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
