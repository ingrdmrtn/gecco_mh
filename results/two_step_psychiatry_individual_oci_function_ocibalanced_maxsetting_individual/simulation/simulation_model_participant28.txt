def simulate_model(
    n_trials,
    parameters,

    Parameters:

        parameters (list): [lr_base, beta, lambda_param, oci_asym_bias]

        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).

        reward (np.ndarray): Rewards (0 or 1).

    lr_pos = np.clip(lr_base * (1.0 - bias_factor), 0.01, 1.0)
    lr_neg = np.clip(lr_base * (1.0 + bias_factor), 0.01, 1.0)

    # Initialize Q-values
    # q_stage1 corresponds to the two choices at the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two choices (cols) at each of the two states (rows)
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed for the 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition to Stage 2 ---
        # Determine second stage state based on transition matrix probabilities
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        r = int(rng.random() < reward_probs[s_idx][a2])

        # --- Learning ---
        # Calculate RPE at Stage 2
        delta2 = r - q_stage2[s_idx, a2]

        # Select learning rate based on sign of prediction error
        current_lr = lr_pos if delta2 >= 0 else lr_neg

        # Update Stage 2 Q-values
        q_stage2[s_idx, a2] += current_lr * delta2

        # Update Stage 1 Q-values (TD(0))
        delta1 = q_stage2[s_idx, a2] - q_stage1[a1]
        q_stage1[a1] += current_lr * delta1

        # Eligibility trace update (TD(1) / lambda)
        q_stage1[a1] += current_lr * lambda_param * delta2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
