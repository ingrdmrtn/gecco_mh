import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates the Hybrid MB/MF Model with OCI-Modulated Win-Stay Lose-Shift (WSLS).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, wsls_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    rng = np.random.default_rng()
    
    # Unpack parameters
    learning_rate, beta, w, wsls_scale = parameters
    
    # Handle OCI input: fitting code expects oci[0], simulation might pass scalar or list
    if hasattr(oci, '__getitem__') and len(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values to 0.5 (matching fitting code)
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    # Output storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # History tracking
    prev_a1 = -1
    prev_reward = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # Apply OCI-Modulated WSLS Bias
        if prev_a1 != -1:
            direction = 1.0 if prev_reward == 1.0 else -1.0
            bias = wsls_scale * oci_score * direction
            q_net[prev_a1] += bias

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        # Map state/action to specific drift parameter
        if s_idx == 0:
            prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob = drift3[t] if a2 == 0 else drift4[t]
        
        r = 1 if rng.random() < prob else 0

        # --- Learning Updates ---
        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-value (using the updated Stage 2 value as the target)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # --- Update History ---
        prev_a1 = a1
        prev_reward = r

        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
