import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates Model 3: OCI-Modulated Choice Stickiness.
    
    This model adds a 'stickiness' bonus to the Q-values of the previously chosen action.
    The magnitude of this stickiness is determined by the OCI score.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stick_oci_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, assumes 0 or handles appropriately.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_base, stick_oci_slope = parameters
    
    # Handle OCI input (if provided as a list/array in fitting, it might come as a single float here)
    # The source code used `current_oci = oci[0]`, implying oci was an array. 
    # Here we handle both scalar and list/array inputs.
    if oci is None:
        current_oci = 0.0
    elif np.ndim(oci) > 0:
        current_oci = oci[0]
    else:
        current_oci = oci

    # Calculate the fixed stickiness value based on OCI
    stickiness = stick_base + (stick_oci_slope * current_oci)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for stage 2 states and actions (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action = -1
    
    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for trial in range(n_trials):
        # Current reward probabilities for the two states (0 and 1) and their actions
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[trial], drift2[trial]], 
                        [drift3[trial], drift4[trial]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value calculation (Hybrid MB/MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-modulated stickiness
        q_decision = q_net.copy()
        if last_action != -1:
            q_decision[last_action] += stickiness
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_decision)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State is 0 or 1. In fitting code, state was passed in. Here we simulate it.
        # transition_matrix[a1] gives probs of transitioning to state 0 or 1 given action a1.
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 (purely MF based on Q_stage2)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[state_idx][a2])
        
        # --- Learning Updates ---
        # SARSA / TD(1) style updates as per the fitting code
        # Note: The fitting code uses a specific update rule:
        # delta_stage2 = r - Q2(s', a')
        # delta_stage1 = Q2(s', a') - Q1(a)
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action = a1
        
        # Store data
        stage1_choice[trial] = a1
        state2[trial] = state_idx
        stage2_choice[trial] = a2
        reward[trial] = r

    return stage1_choice, state2, stage2_choice, reward
