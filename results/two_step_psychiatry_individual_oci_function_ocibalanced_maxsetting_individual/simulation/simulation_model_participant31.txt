import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-Modulated Stickiness Model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stick_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventory score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_base, stick_oci_param = parameters
    
    # Handle OCI input (default to 0 if None, though model logic depends on it)
    oci_val = oci if oci is not None else 0.0

    # Calculate stickiness magnitude based on fitting code logic
    stickiness = (stick_base * 5.0) + ((2.0 * stick_oci_param - 1.0) * 5.0 * oci_val)

    # Transition matrix: Row=Action, Col=Next State Probability
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # Stage 1: Model-Free Q-values for 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: Model-Free Q-values for 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_a1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        # Max Q-value for each state in stage 2
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Map back to Stage 1 actions via transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values (Weighted sum of MB and MF)
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate Logits
        logits_1 = beta * q_net_1
        
        # Apply Stickiness to the previously chosen action
        if last_a1 != -1:
            logits_1[last_a1] += stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine next state (0 or 1) based on transition matrix probabilities for action a1
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        q_vals_2 = q_stage2_mf[s_idx]
        
        # Softmax for Stage 2
        logits_2 = beta * q_vals_2
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward probability based on current state and action
        # State 0 uses drift1 (action 0) and drift2 (action 1)
        # State 1 uses drift3 (action 0) and drift4 (action 1)
        if s_idx == 0:
            prob_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_r = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < prob_r else 0
        
        # --- Learning Updates ---
        # Note: Updates follow the specific order in the provided fitting code
        
        # Update Stage 1 MF Q-value
        # Based on the value of the state reached (before reward update)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value
        # Based on the received reward
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
