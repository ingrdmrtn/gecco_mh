import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free RL model where the learning rate 
    depends on transition type (Common/Rare) and OCI modulates the Rare learning rate.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr_common, lr_rare_base, lr_rare_oci_mod, beta]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr_common, lr_rare_base, lr_rare_oci_mod, beta = parameters
    
    # Handle OCI input (scalar or list/array)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0.0

    # Calculate modulated learning rate for rare transitions
    lr_rare = lr_rare_base + (lr_rare_oci_mod * oci_score)
    lr_rare = np.clip(lr_rare, 0.001, 0.999)

    # Initialize Q-values
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2)) # Shape: (State, Action)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    rng = np.random.default_rng()

    # Transition matrix: 70% common, 30% rare
    # Action 0 -> State 0 (0.7), State 1 (0.3)
    # Action 1 -> State 1 (0.7), State 0 (0.3)
    transition_probs = np.array([[0.7, 0.3], 
                                 [0.3, 0.7]])

    for t in range(n_trials):
        # --- Stage 1 ---
        # Softmax selection
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # Transition to Stage 2
        s_idx = rng.choice([0, 1], p=transition_probs[a1])
        
        # Determine if transition was common or rare
        is_common = (a1 == s_idx)
        current_lr = lr_common if is_common else lr_rare

        # --- Stage 2 ---
        # Softmax selection
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # Reward Generation
        # Map state/action to specific drift parameter
        if s_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s_idx == 1
            p_reward = drift3[t] if a2 == 0 else drift4[t]
        
        r = 1 if rng.random() < p_reward else 0

        # --- Learning Updates ---
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += current_lr * delta_stage2

        # Update Stage 1 Q-values
        # Note: Uses the updated Stage 2 values to calculate max
        val_stage2 = np.max(q_stage2[s_idx])
        delta_stage1 = val_stage2 - q_stage1[a1]
        q_stage1[a1] += current_lr * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
