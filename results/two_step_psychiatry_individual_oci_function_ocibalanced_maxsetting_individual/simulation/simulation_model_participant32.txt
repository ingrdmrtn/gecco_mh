import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-Modulated Win-Stay Stickiness Model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, win_stick_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float, list, or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, win_stick_oci = parameters
    
    # Handle OCI input (fitting code expects subscriptable, simulation might get float)
    try:
        oci_score = oci[0]
    except (TypeError, IndexError):
        oci_score = oci if oci is not None else 0.0

    rng = np.random.default_rng()
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states (rows), 2 actions (cols)
    q_stage2_mf = np.zeros((2, 2))
    
    # History storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1
    last_reward = 0
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value calculation: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value weighting MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Stickiness Logic: Win-Stay OCI
        # High OCI participants exhibit stickiness specifically after receiving a reward.
        stickiness = np.zeros(2)
        if last_action_1 != -1 and last_reward == 1:
            stickiness[int(last_action_1)] = win_stick_oci * oci_score
            
        # Softmax Stage 1
        logits_1 = beta * (q_net + stickiness)
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        logits_2 = beta * q_stage2_mf[s2_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Map state/action to drift probability
        # State 0 uses drift1 (a=0) and drift2 (a=1)
        # State 1 uses drift3 (a=0) and drift4 (a=1)
        if s2_idx == 0:
            prob_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_r = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < prob_r else 0
        
        # --- Updates ---
        
        # Update Stage 1 MF (TD(0) style using Stage 2 Q-value)
        delta_1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_1
        
        # Update Stage 2 MF (based on Reward)
        delta_2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_2
        
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        last_reward = r
        
    return stage1_choice, state2, stage2_choice, reward
