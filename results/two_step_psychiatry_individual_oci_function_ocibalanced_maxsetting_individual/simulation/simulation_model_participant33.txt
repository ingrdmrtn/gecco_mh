import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Reinforcement Learning with OCI-modulated Choice Perseveration.
    
    Hypothesis: High OCI scores increase the 'stickiness' parameter, causing the 
    agent to repeat the previous Stage 1 action regardless of reward history.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, pers_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, pers_base = parameters
    
    # Handle OCI input (default to 0 if None, though the prompt implies it will be provided)
    oci_score = oci if oci is not None else 0.0
    
    # Calculate the effective perseveration weight based on OCI
    w_pers = pers_base * (1.0 + oci_score)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1 corresponds to the two actions at stage 1
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to the two states (rows) and two actions (cols) at stage 2
    q_stage2 = np.zeros((2, 2))
    
    # Transition matrix (fixed 70/30 structure standard for this task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])
    
    last_action_1 = -1 # Initialize variable to track previous choice
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        q_effective = q_stage1.copy()
        
        # Apply perseveration bonus if there was a previous action
        if last_action_1 != -1:
            q_effective[last_action_1] += w_pers
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_effective)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine Stage 2 state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 based on current state s2
        exp_q2 = np.exp(beta * q_stage2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward outcome based on drift probabilities
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        # Update Stage 2 Q-values: Standard TD error
        delta_stage2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values: TD error driven by Stage 2 Q-value
        # Note: The fitting code uses q_stage2[s, a2] as the target, not the reward directly
        delta_stage1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_stage1
        
        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
