import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Model-Free RL with OCI-Driven Win-Stay Amplification.
    
    Logic:
    - Pure Model-Free RL (no Model-Based component, no eligibility traces).
    - Stage 1 choice is influenced by Q-values and a stickiness bonus.
    - Stickiness has a base component and an additional boost if the previous trial was rewarded.
    - The magnitude of the 'win-stay' boost is scaled by the participant's OCI score.
    - Stage 2 choice is standard softmax on Q-values.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_base, oci_winstay_boost]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score (scalar).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, stickiness_base, oci_winstay_boost = parameters
    
    # Handle OCI input (scalar or array)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0

    # Task structure
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf corresponds to Q(s1, a1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q(s2, a2). Rows are states (0 or 1), cols are actions.
    q_stage2_mf = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # History tracking for stickiness
    prev_a1 = -1
    prev_reward = 0.0

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        stick_bonus = np.zeros(2)
        if prev_a1 != -1:
            total_stick = stickiness_base
            
            # Add OCI-scaled boost if previous trial was a win
            if prev_reward > 0:
                total_stick += (oci_score * oci_winstay_boost)
            
            stick_bonus[prev_a1] = total_stick

        q_net_1 = q_stage1_mf + stick_bonus
        
        # Softmax Stage 1
        exp_q1 = np.exp(beta * q_net_1)
        # Safety check for overflow/underflow is handled by numpy usually, 
        # but in simulation rng.choice handles the normalization implicitly if p sums to 1.
        p_stage1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=p_stage1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Standard softmax on Stage 2 Q-values
        q2_vals = q_stage2_mf[s2_idx]
        exp_q2 = np.exp(beta * q2_vals)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=p_stage2)

        # --- Outcome ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning (Update Q-values) ---
        # Update Stage 2 Q-value based on reward prediction error
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-value based on Stage 2 Q-value (TD learning)
        # Note: The fitting code uses q_stage2_mf[s_idx, a2] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # --- Update History ---
        prev_a1 = a1
        prev_reward = r

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
