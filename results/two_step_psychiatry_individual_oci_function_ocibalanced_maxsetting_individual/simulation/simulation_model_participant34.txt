import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None
):
    """
    Simulates choices and rewards using the OCI-Modulated Ritual Bias model.
    
    Logic matches cognitive_model3:
    - Hybrid MB/MF control.
    - Ritual bias accumulates on rewarded Stage 1 actions (scaled by OCI).
    - Ritual bias decays every trial.
    - Ritual bias is added to the net Stage 1 Q-values.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, lambda_elig, ritual_decay, oci_ritual_amp]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventory score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, w, lambda_elig, ritual_decay, oci_ritual_amp = parameters
    
    # Handle OCI input
    current_oci = oci if oci is not None else 0.0
    
    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values and traces
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    ritual_trace = np.zeros(2)
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # Current reward probabilities
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB, MF, and Ritual Bias
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf + ritual_trace
        
        exp_q1 = np.exp(beta * q_net_stage1)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=p_stage1)
        
        # --- Transition ---
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # The fitting model uses the same beta for both stages
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        p_stage2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=p_stage2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Value Updates ---
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s2, a2]
        
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + lambda_elig * delta_stage2)
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # --- Ritual Trace Update ---
        # Decays every trial
        ritual_trace *= ritual_decay
        
        # Boosts on reward, scaled by OCI
        if r == 1:
            ritual_trace[a1] += oci_ritual_amp * current_oci
            
        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
