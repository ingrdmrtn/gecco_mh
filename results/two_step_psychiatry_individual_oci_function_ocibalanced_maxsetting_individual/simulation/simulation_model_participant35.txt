import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid Model-Based/Model-Free RL model
    where the mixing weight is modulated by the OCI score.

    Hypothesis: High OCI participants rely more on Model-Free (habitual) control.
    The mixing weight 'w' (0=MF, 1=MB) is scaled down by the OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.
                             If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    rng = np.random.default_rng()
    
    # Unpack parameters
    learning_rate, beta, w_base = parameters
    
    # Handle OCI input (default to 0 if not provided, though the prompt implies it's passed)
    oci_score = oci if oci is not None else 0.0
    
    # Calculate modulated mixing weight
    # Logic from fitting code: w = w_base * (1.0 - oci_score)
    w = w_base * (1.0 - oci_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize value tracking arrays
    # q_mf1: Model-free values for stage 1 (2 actions)
    q_mf1 = np.zeros(2)      
    # q_mf2: Model-free values for stage 2 (2 states, 2 actions)
    q_mf2 = np.zeros((2, 2)) 

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for trial in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 (index 0) has options with probs drift1, drift2
        # State 1 (index 1) has options with probs drift3, drift4
        current_reward_probs = [
            [drift1[trial], drift2[trial]],
            [drift3[trial], drift4[trial]]
        ]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_mf2 = np.max(q_mf2, axis=1)
        q_mb1 = transition_matrix @ max_q_mf2
        
        # Calculate Net values (Hybrid)
        q_net = w * q_mb1 + (1 - w) * q_mf1
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (purely model-free based on Q_mf2)
        exp_q2 = np.exp(beta * q_mf2[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning / Update ---
        
        # Update Stage 2 Q-values (TD Error 2)
        delta_2 = r - q_mf2[s2, a2]
        q_mf2[s2, a2] += learning_rate * delta_2
        
        # Update Stage 1 Q-values (TD Error 1)
        delta_1 = q_mf2[s2, a2] - q_mf1[a1] # Note: using updated q_mf2 value
        q_mf1[a1] += learning_rate * delta_1
        
        # Eligibility trace update (lambda = 1 assumption from fitting code)
        # q_mf1[a1] += learning_rate * delta_2
        q_mf1[a1] += learning_rate * delta_2

        # Store data
        stage1_choice[trial] = a1
        state2[trial] = s2
        stage2_choice[trial] = a2
        reward[trial] = r

    return stage1_choice, state2, stage2_choice, reward
