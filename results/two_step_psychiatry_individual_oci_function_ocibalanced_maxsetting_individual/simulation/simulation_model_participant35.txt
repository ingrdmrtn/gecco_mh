import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-Modulated Exploration After Non-Reward model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_win, beta_loss_base, beta_loss_oci, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta_win, beta_loss_base, beta_loss_oci, w, stickiness = parameters
    
    # Handle OCI input
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci if oci is not None else 0.0

    # Calculate beta_loss based on OCI and clamp it
    beta_loss = beta_loss_base + (beta_loss_oci * oci_score)
    if beta_loss < 0: beta_loss = 0.0
    if beta_loss > 10: beta_loss = 10.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Tracking variables
    last_action_1 = -1
    last_reward = 1.0 # Initialize assuming win or neutral state
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Determine beta based on previous reward
        if last_reward == 1.0:
            current_beta = beta_win
        else:
            current_beta = beta_loss

        # --- Stage 1 ---
        
        # Model-Based value calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value calculation (Hybrid MB/MF)
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply stickiness (Perseveration bonus)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 ---
        
        # Softmax choice for Stage 2
        exp_q2 = np.exp(current_beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Get drift probability for the specific state and action
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < prob_reward else 0
        
        # --- Updates ---
        
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values
        # Uses the updated Stage 2 value as the target
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1
        last_reward = r
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
