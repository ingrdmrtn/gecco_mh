import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: Pure Model-Free with Outcome-Dependent Stickiness.
    
    This model includes:
    - Standard Model-Free Q-learning (TD-0).
    - Outcome-dependent perseveration: 'Win-Stickiness' vs 'Lose-Stickiness'.
    - OCI modulation of the 'Lose-Stickiness' parameter.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, stick_win, stick_lose_base, oci_stick_lose_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, stick_win, stick_lose_base, oci_stick_lose_mod = parameters
    
    # Handle OCI input safely
    oci_score = oci if oci is not None else 0.0
    
    # Calculate the effective lose-stickiness based on OCI
    stick_lose = np.clip(stick_lose_base + (oci_score * oci_stick_lose_mod), 0.0, 5.0)

    # Action 0 -> 70% State 0 (Common), 30% State 1 (Rare)
    # Action 1 -> 30% State 0 (Rare), 70% State 1 (Common)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2)
    # Stage 2: 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2))

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Tracking variables for stickiness
    last_action_1 = -1
    last_reward = 0 # Assume neutral/loss start as per fitting code logic

    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Determine Stage 1 Choice
        q_net = q_stage1_mf.copy()

        # Apply Outcome-Dependent Stickiness
        if last_action_1 != -1:
            if last_reward == 1.0:
                q_net[last_action_1] += stick_win
            else:
                q_net[last_action_1] += stick_lose
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # 2. Transition to Stage 2
        # Determine state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Determine Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # 4. Outcome
        # Get reward probability for the current state (s2) and action (a2)
        # Mapping: s2=0 uses drift1/drift2, s2=1 uses drift3/drift4
        current_drift = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r_prob = current_drift[s2][a2]
        r = int(rng.random() < r_prob)

        # 5. Learning (Update Q-values)
        # Update Stage 2 Q-value: TD error based on reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2

        # Update Stage 1 Q-value: TD error based on Stage 2 Q-value

        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

        # 6. Update History
        last_action_1 = a1
        last_reward = r
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
