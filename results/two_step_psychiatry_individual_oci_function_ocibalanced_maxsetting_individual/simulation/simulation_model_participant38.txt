import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF learner with OCI-modulated Stage 1 Rigidity.
    
    The key feature of this model is that the inverse temperature (beta) for the first-stage choice
    is scaled by the OCI score, representing 'planning rigidity', while the second-stage beta
    remains at a baseline level.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, w, rigidity_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_base, w, rigidity_factor = parameters
    
    # Handle OCI input (default to 0 if not provided, though model relies on it)
    oci_score = oci if oci is not None else 0.0
    
    # Calculate betas based on the model logic
    # beta_1 is modulated by OCI
    beta_1 = beta_base * (1.0 + rigidity_factor * oci_score)
    # beta_2 is the baseline
    beta_2 = beta_base

    # Transition matrix (fixed as per standard 2-step task)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF Q-values (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities for the 4 terminal states
        # drift1/2 correspond to State 2 (index 0), drift3/4 correspond to State 3 (index 1)
        # We map them to our internal state indices 0 and 1.
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Hybrid values
        q_hybrid = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax selection for Stage 1 (using beta_1)
        exp_q1 = np.exp(beta_1 * q_hybrid)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 (using beta_2)
        # Note: In fitting code, q_stage2_mf[s_idx] is used directly
        exp_q2 = np.exp(beta_2 * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning Updates ---
        
        # Update Stage 2 Q-values (TD(0))
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF Q-values (TD(1) / SARSA-like update driven by Stage 2 Q-value)
        # The fitting code uses: delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        # Note: It uses the *updated* q_stage2_mf value.
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx # The simulation usually expects 0/1 for state
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
