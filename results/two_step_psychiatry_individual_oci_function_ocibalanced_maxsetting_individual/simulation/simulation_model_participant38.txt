import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4, oci):
    """
    Simulates the Outcome-Dependent Stickiness Model (Win-Stay, Lose-Stay OCI).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, lambda_decay, stick_win, stick_loss_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float/list): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, w, lambda_decay, stick_win, stick_loss_oci = parameters
    
    # Handle OCI input (scalar or list/array)
    if hasattr(oci, '__getitem__') and not isinstance(oci, (str, bytes)):
        oci_score = oci[0]
    else:
        oci_score = oci

    stick_loss = stick_loss_oci * oci_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_a1 = -1
    prev_reward = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value: Expected max value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Stickiness Logic
        stick_vec = np.zeros(2)
        if prev_a1 != -1:
            if prev_reward == 1:
                stick_vec[prev_a1] = stick_win
            else:
                stick_vec[prev_a1] = stick_loss
        
        # Net Q-value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + stick_vec
        
        # Softmax Choice Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # State 0 or 1
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        qs_2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Select drift probability based on state and action
        if s_idx == 0:
            prob_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_r = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < prob_r else 0
        
        # --- Learning Updates ---
        # SARSA-like update for Stage 1
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        # RPE for Stage 2
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Stage 1 MF Q-value (with eligibility trace of stage 2 error)
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * lambda_decay * delta_stage2
        
        # Update Stage 2 MF Q-value
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update history for next trial stickiness
        prev_a1 = a1
        prev_reward = r
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
