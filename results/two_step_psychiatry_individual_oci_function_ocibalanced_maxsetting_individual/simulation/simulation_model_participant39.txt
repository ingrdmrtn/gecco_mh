import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-Modulated Model-Based Beta model.
    
    In this model, Model-Based and Model-Free systems have independent strengths (betas)
    in Stage 1. OCI specifically modulates the strength of the Model-Based system.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, stickiness, beta_stage2, beta_mf, beta_mb_base, beta_mb_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, stickiness, beta_stage2, beta_mf, beta_mb_base, beta_mb_oci = parameters
    
    # Handle OCI input (scalar or array-like)
    if oci is None:
        oci_score = 0.0
    elif hasattr(oci, '__len__'):
        oci_score = oci[0]
    else:
        oci_score = oci
        
    # Calculate OCI-modulated Model-Based beta
    beta_mb = beta_mb_base + beta_mb_oci * oci_score
    beta_mb = np.clip(beta_mb, 0.0, 30.0)
    
    # Initialize structures
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Rows: State (0 or 1), Cols: Action (0 or 1)
    
    last_action_1 = -1
    
    # Data storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Logits (Independent Betas)
        logits = beta_mb * q_stage1_mb + beta_mf * q_stage1_mf
        
        # Apply Stickiness
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        # Softmax for Stage 1
        logits = logits - np.max(logits) # Numerical stability
        exp_logits = np.exp(logits)
        probs_1 = exp_logits / np.sum(exp_logits)
        
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        s_idx = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2 (based on MF Q-values)
        logits_2 = beta_stage2 * q_stage2_mf[s_idx, :]
        logits_2 = logits_2 - np.max(logits_2)
        exp_logits_2 = np.exp(logits_2)
        probs_2 = exp_logits_2 / np.sum(exp_logits_2)
        
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward probability based on current state and drifts
        if s_idx == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < prob_reward else 0
        
        # --- Learning ---
        # Calculate Prediction Errors
        # Note: The fitting code adds full delta_stage2 to stage 1 (implying lambda=1)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Update Q-values
        q_stage1_mf[a1] += learning_rate * (delta_stage1 + delta_stage2)
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # --- Record Data ---
        last_action_1 = a1
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
    return stage1_choice, state2, stage2_choice, reward
