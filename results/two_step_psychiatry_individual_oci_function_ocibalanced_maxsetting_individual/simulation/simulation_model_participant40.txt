import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free learner with OCI-modulated choice perseveration.
    
    Hypothesis: High OCI leads to higher choice repetition (stickiness) 
    in the first stage, regardless of reward history.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, stickiness_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    learning_rate, beta, stickiness_factor = parameters
    
    # Handle OCI input: if it's a list/array, take the first element, otherwise use as is.
    # Default to 0 if None provided (though the prompt implies it will be provided).
    if oci is None:
        oci_score = 0
    elif hasattr(oci, '__len__') and len(oci) > 0:
        oci_score = oci[0]
    else:
        oci_score = oci

    # Transition matrix (standard for 2-step task: 70% common, 30% rare)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1 corresponds to the two choices in the first stage
    q_stage1 = np.zeros(2)
    # q_stage2 corresponds to two states, each having two choices
    q_stage2 = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Calculate the perseveration weight based on OCI
    perseveration_weight = stickiness_factor * oci_score
    
    last_action_1 = -1 # Indicator for no previous action

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # state 0: options 0 (drift1) and 1 (drift2)
        # state 1: options 0 (drift3) and 1 (drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        logits_1 = beta * q_stage1.copy()

        # Apply OCI-modulated stickiness to the previously chosen action
        if last_action_1 != -1:
            logits_1[last_action_1] += perseveration_weight
            
        # Softmax for Stage 1
        logits_1 = logits_1 - np.max(logits_1)
        probs_1 = np.exp(logits_1) / np.sum(np.exp(logits_1))
        
        a1 = rng.choice([0, 1], p=probs_1)
        last_action_1 = a1

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        logits_2 = beta * q_stage2[s_idx]
        logits_2 = logits_2 - np.max(logits_2)
        probs_2 = np.exp(logits_2) / np.sum(np.exp(logits_2))
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s_idx][a2])

        # --- Learning (Model-Free) ---
        # Update Stage 2 Q-values
        pe_2 = r - q_stage2[s_idx, a2]
        q_stage2[s_idx, a2] += learning_rate * pe_2

        # Update Stage 1 Q-values (TD(0) / Direct Reinforcement logic from fitting code)
        # Note: The fitting code uses 'r' to update q_stage1 directly, not the value of the next state.
        pe_1 = r - q_stage1[a1]
        q_stage1[a1] += learning_rate * pe_1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
