import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4, oci):
    """
    Simulates choices and rewards using the Habitual Rigidity Model (Dynamic Beta).
    
    Hypothesis: OCI modulates the relationship between habit strength and decision noise.
    As a participant repeats an action, the inverse temperature (beta) increases,
    making the choice more deterministic (rigid).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta_base, beta_stiff_oci, w, stickiness, trace_decay, lambda_eligibility]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters matching the fitting function
    lr, beta_base, beta_stiff_oci, w, stickiness, trace_decay, lambda_eligibility = parameters
    
    # Handle OCI input (robustness for float or list input)
    if isinstance(oci, (list, np.ndarray, tuple)):
        current_oci = oci[0]
    else:
        current_oci = oci

    rng = np.random.default_rng()

    # Initialization
    # Initialize Q-values to 0.5 as in the fitting code
    q_stage1_mf = np.zeros(2) + 0.5
    q_stage2_mf = np.zeros((2, 2)) + 0.5
    
    habit_trace = np.zeros(2)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value Calculation
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Dynamic Beta Calculation (Rigidity)
        current_rigidity = np.max(habit_trace)
        beta_eff = beta_base + beta_stiff_oci * current_oci * current_rigidity
        
        logits_1 = beta_eff * q_net

        # Stickiness Bonus
        if t > 0:
            logits_1[prev_a1] += stickiness
            
        # Softmax Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1))
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # Habit Trace Update
        habit_trace *= trace_decay
        habit_trace[a1] += 1.0

        # --- Transition ---
        # Determine next state based on transition matrix and action 1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Uses baseline beta, not the rigid beta
        logits_2 = beta_base * q_stage2_mf[s2, :]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Map drift probabilities to current state and action
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < prob_reward)

        # --- Learning (Q-Value Updates) ---
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s2, a2]

        q_stage2_mf[s2, a2] += lr * delta_stage2
        q_stage1_mf[a1] += lr * delta_stage1 + lambda_eligibility * lr * delta_stage2

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
