import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates the Mixture of Model-Free RL and OCI-Modulated Habitual Perseveration model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_habit_low_oci, w_habit_high_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal state-actions.
        oci (float, list, or None): Obsessive-Compulsive Inventory score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w_habit_low_oci, w_habit_high_oci = parameters
    
    # Initialize Random Number Generator
    rng = np.random.default_rng()

    # Handle OCI input (scalar or array-like)
    try:
        oci_score = oci[0]
    except (TypeError, IndexError, KeyError):
        oci_score = float(oci) if oci is not None else 0.0

    # Calculate Habit Weight
    # Linearly interpolate between low and high OCI weights based on score
    w_habit = w_habit_low_oci * (1 - oci_score) + w_habit_high_oci * oci_score
    w_habit = np.clip(w_habit, 0.0, 1.0)

    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1 = np.zeros(2)
    # Stage 2: 2 states x 2 actions
    q_stage2 = np.zeros((2, 2))
    
    # Variables for history
    last_action_1 = -1
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Transition matrix: [Action][Next State]
    # Action 0 -> 70% State 0, 30% State 1
    # Action 1 -> 30% State 0, 70% State 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    for t in range(n_trials):
        # --- Stage 1 Choice ---
        
        # 1. Model-Free RL Policy (Softmax)
        exp_q1 = np.exp(beta * q_stage1)
        if np.any(np.isinf(exp_q1)):
            probs_rl = np.zeros_like(exp_q1)
            probs_rl[np.argmax(q_stage1)] = 1.0
        else:
            probs_rl = exp_q1 / np.sum(exp_q1)

        # 2. Habitual Policy (Perseveration)
        probs_habit = np.zeros(2)
        if last_action_1 != -1:
            probs_habit[last_action_1] = 1.0
        else:
            probs_habit[:] = 0.5 # Equal probability on first trial

        # 3. Mixture
        probs_1_mixed = (1 - w_habit) * probs_rl + w_habit * probs_habit
        # Ensure normalization
        probs_1_mixed = probs_1_mixed / np.sum(probs_1_mixed)
        
        a1 = rng.choice([0, 1], p=probs_1_mixed)

        # --- Transition to Stage 2 ---
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Choice ---
        
        # RL Policy only for Stage 2
        exp_q2 = np.exp(beta * q_stage2[s2])
        if np.any(np.isinf(exp_q2)):
            probs_2 = np.zeros_like(exp_q2)
            probs_2[np.argmax(q_stage2[s2])] = 1.0
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
            
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        # Map current state/action to the correct drift parameter
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s2 == 1
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)

        # --- Value Updates ---
        
        # TD Error 1: Difference between Stage 2 Q-value and Stage 1 Q-value
        delta_1 = q_stage2[s2, a2] - q_stage1[a1]
        q_stage1[a1] += learning_rate * delta_1
        
        # TD Error 2: Difference between Reward and Stage 2 Q-value
        delta_2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += learning_rate * delta_2
        
        # Eligibility Trace: Propagate TD Error 2 back to Stage 1 Q-value
        q_stage1[a1] += learning_rate * delta_2

        # --- Record Data ---
        last_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
