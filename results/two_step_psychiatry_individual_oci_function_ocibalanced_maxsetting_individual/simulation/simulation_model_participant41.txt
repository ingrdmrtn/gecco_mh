import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using a Model-Free learner with OCI-scaled choice perseveration.
    
    The stickiness (perseveration) magnitude is dynamically scaled by the participant's OCI score.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, persev_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, persev_base = parameters
    
    # Handle OCI input (default to 0 if not provided, though the prompt implies it will be)
    oci_score = oci if oci is not None else 0.0

    # Calculate stickiness based on OCI
    stickiness = persev_base * (1.0 + oci_score)

    # Transition matrix (standard 70/30 structure for 2-step task)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1: values for the two actions in stage 1
    q_stage1 = np.zeros(2)
    # q_stage2: values for the two actions in the two possible stage 2 states
    # Shape: (2 states, 2 actions)
    q_stage2 = np.zeros((2, 2))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 # Initialize with no previous action

    for t in range(n_trials):
        # Organize drift probabilities for easier access: [state][action]
        # State 0 (drift1, drift2), State 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        logits_1 = beta * q_stage1.copy()
        
        # Apply OCI-scaled stickiness to the previously chosen action
        if last_action_1 != -1:
            logits_1[last_action_1] += stickiness
            
        # Softmax calculation
        logits_1 = logits_1 - np.max(logits_1) # Numerical stability
        exp_q1 = np.exp(logits_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Update last action
        last_action_1 = a1

        # --- Transition ---
        # Determine state 2 based on transition matrix probabilities
        s_curr = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Standard softmax on Stage 2 Q-values
        logits_2 = beta * q_stage2[s_curr]
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[s_curr][a2])

        # --- Learning (TD Updates) ---
        
        # Update Stage 2 Q-value based on reward
        # pe_2 = r - q_stage2[s_curr, a2]
        pe_2 = r - q_stage2[s_curr, a2]
        q_stage2[s_curr, a2] += lr * pe_2

        # Update Stage 1 Q-value based on Stage 2 value (TD(0))
        # Note: The fitting code uses q_stage2[s_curr, a2] as the target for stage 1
        pe_1 = q_stage2[s_curr, a2] - q_stage1[a1]
        q_stage1[a1] += lr * pe_1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_curr
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
