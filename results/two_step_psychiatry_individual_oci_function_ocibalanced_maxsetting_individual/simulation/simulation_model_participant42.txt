import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-Modulated Post-Win Beta Rigidity Model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_base, beta_win_oci, w, stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize Random Number Generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta_base, beta_win_oci, w, stickiness = parameters
    
    # Handle OCI input (ensure it's a float)
    oci_score = oci if oci is not None else 0.0

    # Initialize Q-values and structures
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF Q-values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # State variables for the loop
    prev_reward = 0.0
    last_action_1 = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness
        logits = q_net.copy()
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        # Calculate Dynamic Beta (Rigidity)
        # Rigid if high OCI AND previous trial was a win
        current_beta = beta_base + beta_win_oci * oci_score * prev_reward
        
        # Softmax Stage 1
        exp_q1 = np.exp(current_beta * logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state (0 or 1) based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Note: Fitting code uses beta_base for Stage 2, not current_beta
        exp_q2 = np.exp(beta_base * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Generation ---
        # Select correct drift rate based on state (s2) and action (a2)
        # Mapping: State 0 -> (drift1, drift2), State 1 -> (drift3, drift4)
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < p_reward else 0
        
        # --- Learning Updates ---
        
        # 1. Stage 1 TD Update (using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 2. Stage 2 TD Update (using Reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # 3. Stage 1 Second Update (Eligibility Trace / Direct Reward propagation)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        # --- Update History ---
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1
        prev_reward = float(r)

    return stage1_choice, state2, stage2_choice, reward
