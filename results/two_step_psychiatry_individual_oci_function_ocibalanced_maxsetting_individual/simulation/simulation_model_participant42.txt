def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 3: OCI-Modulated Win-Stay Heuristic Bias.
    
    This model simulates a standard hybrid MB/MF agent but adds a specific bias term
    to the first-stage decision values. If the previous trial was rewarded, the value 
    of the previously chosen action is boosted proportional to the participant's OCI score.

    Parameters:
        n_trials (int): Number of trials to simulate.

        oci (float or None): Obsessive-Compulsive Inventoryâ€“Revised score. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1, corresponding to indices 0 and 1 in transition matrix).

        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, w, ws_strength = parameters

    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])

    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the second stage values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Variables for the Win-Stay heuristic logic
    prev_act1 = -1
    prev_reward = 0.0
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the 4 arms (State 0: [drift1, drift2], State 1: [drift3, drift4])
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---

        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply OCI-Modulated Win-Stay Bias
        # If previous trial was rewarded, boost the value of the previous action
        if t > 0 and prev_reward == 1.0 and prev_act1 != -1:
            bias = np.zeros(2)
            bias[prev_act1] = ws_strength * oci_score
            q_net += bias
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard Softmax on Stage 2 MF values
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # --- Outcome ---
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning ---
        # Update Stage 2 Q-values (TD learning)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store history for output and next trial logic
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_act1 = a1
        prev_reward = float(r)

    return stage1_choice, state2, stage2_choice, reward
