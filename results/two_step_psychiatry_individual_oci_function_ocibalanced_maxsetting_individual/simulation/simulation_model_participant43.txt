import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the OCI-Modulated Counterfactual Regret Model.
    
    Hypothesis: High OCI participants engage in counterfactual updating. 
    When an outcome is observed, they also update the unchosen option in Stage 2. 
    Specifically, if Reward=0, they assume the unchosen option would have had Reward=1, 
    driven by a 'regret' parameter scaled by OCI.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, stickiness, oci_regret]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list or np.ndarray): Obsessive-Compulsive Inventoryâ€“Revised score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    # Note: In the fitting code, 'oci' was passed as a list/array where oci[0] was used.
    # We handle scalar or list input for OCI here.
    lr, beta, w, stickiness, oci_regret = parameters
    
    if np.ndim(oci) > 0:
        participant_oci = oci[0]
    else:
        participant_oci = oci if oci is not None else 0.0

    regret_weight = participant_oci * oci_regret
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for Stage 1 (Model-Free)
    # q_stage2_mf: Q-values for Stage 2 (State x Action)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) 
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_a1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities (environment)
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based value for Stage 1
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value mixing MB and MF
        q_net_1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness (perseveration) to the previous choice
        if prev_a1 != -1:
            q_net_1[prev_a1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine Stage 2 state based on transition matrix
        s_curr = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s_curr])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Generate reward based on drift probabilities
        r = int(rng.random() < reward_probs[s_curr][a2])
        
        # --- Learning / Updating ---
        
        # 1. Update Stage 2 Chosen Option (Standard Q-learning)
        delta_s2 = r - q_stage2_mf[s_curr, a2]
        q_stage2_mf[s_curr, a2] += lr * delta_s2
        
        # 2. Counterfactual Updating (The specific mechanism of this model)
        unchosen_a2 = 1 - a2
        counterfactual_reward = 1.0 - r
        delta_cf = counterfactual_reward - q_stage2_mf[s_curr, unchosen_a2]
        q_stage2_mf[s_curr, unchosen_a2] += lr * regret_weight * delta_cf
        
        # 3. Update Stage 1 (Direct Reinforcement / TD(1)-like for Stage 1 MF)
        # Note: The fitting code uses `q_stage1_mf[a1] += lr * (r - q_stage1_mf[a1])`
        # This treats the final reward as the direct target for Stage 1, ignoring Stage 2 Q-value.
        q_stage1_mf[a1] += lr * (r - q_stage1_mf[a1])
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s_curr
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
