import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4, oci):
    """
    Simulates the Hybrid Learner with OCI-Modulated Perseveration.
    
    Based on Model 1: Hybrid Learner with OCI-Modulated Perseveration.
    Q_net = w * Q_MB + (1 - w) * Q_MF
    Logits = beta * Q_net + (persev_strength * OCI * I(action == last_action))
    """
    
    # Unpack parameters based on the fitting function signature
    learning_rate, beta, w, persev_strength = parameters
    
    # Handle OCI input (fitting code assumes it is subscriptable, simulation might pass scalar)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    rng = np.random.default_rng()
    
    # Transition probabilities
    # Action 0 -> 0.7 to State 0, 0.3 to State 1
    # Action 1 -> 0.3 to State 0, 0.7 to State 1
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action
    
    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action = -1
    
    for t in range(n_trials):
        # --- Stage 1: Choice ---
        
        # 1. Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # 2. Calculate Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # 3. Calculate Logits with Perseveration
        logits_1 = beta * q_net
        if last_action != -1:
            logits_1[last_action] += persev_strength * oci_score

        # 4. Softmax Probability
        logits_1 = logits_1 - np.max(logits_1) # Numerical stability
        exp_q1 = np.exp(logits_1)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        
        # 5. Make Choice
        a1 = rng.choice([0, 1], p=p_choice_1)

        # --- Transition ---
        # Determine State 2 based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2: Choice ---
        
        # 1. Calculate Logits (Pure MF)
        logits_2 = beta * q_stage2_mf[s2_idx]
        
        # 2. Softmax Probability
        logits_2 = logits_2 - np.max(logits_2)
        exp_q2 = np.exp(logits_2)
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        
        # 3. Make Choice
        a2 = rng.choice([0, 1], p=p_choice_2)

        # --- Reward Outcome ---
        # Map state/action to specific drift parameter
        # State 0 (index 0) uses drift1 (opt 0) and drift2 (opt 1)
        # State 1 (index 1) uses drift3 (opt 0) and drift4 (opt 1)
        if s2_idx == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < prob_reward)

        # --- Learning Updates ---
        
        # Update Stage 1 MF value (TD to Stage 2 value)
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 MF value (TD to Reward)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # --- Store Data ---
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action = a1

    return stage1_choice, state2, stage2_choice, reward
