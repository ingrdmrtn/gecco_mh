import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci):
    """
    Simulates choices and rewards using the Compulsive Perseveration Model (OCI-Modulated Trace Decay).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, lambd, stick_weight, decay_base]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list): Obsessive-Compulsive Inventory score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, lambd, stick_weight, decay_base = parameters
    
    # Handle OCI input (can be scalar or single-element container)
    if hasattr(oci, '__len__') and not isinstance(oci, str):
        oci_score = oci[0]
    else:
        oci_score = oci

    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values and traces
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    choice_trace = np.zeros(2)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Calculate effective decay rate based on OCI
    decay_eff = decay_base + (1.0 - decay_base) * oci_score

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-value combines MB, MF, and the perseveration trace
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf + stick_weight * choice_trace
        
        # Softmax choice 1
        exp_q1 = np.exp(beta * q_net)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_choice_1)
        
        # Update choice trace (Decay then increment)
        choice_trace *= decay_eff
        choice_trace[a1] += 1.0

        # --- Transition ---
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        qs2 = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * qs2)
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_choice_2)
        
        # --- Reward ---
        # Map state/action to specific drift parameter array
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < prob_reward)

        # --- Learning Updates ---
        # Stage 1 TD error
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 TD error
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Eligibility trace update for Stage 1 Q-value
        q_stage1_mf[a1] += learning_rate * lambd * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
