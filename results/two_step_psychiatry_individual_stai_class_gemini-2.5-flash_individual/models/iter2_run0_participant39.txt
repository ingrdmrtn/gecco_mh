class ParticipantModel1(CognitiveModelBase):
    """
    HYPOTHESIS: This model proposes a hybrid model-free (MF) and model-based (MB) control system.
    Participants learn values via MF TD-learning. For stage 1 decisions, these MF values are
    blended with MB values, which are computed by 'looking ahead' one step using learned transition
    probabilities and the current MF stage 2 values. High anxiety (STAI score) is hypothesized
    to reduce the contribution of the model-based system, making choices more reliant on simpler,
    model-free (habitual) value estimates, possibly to reduce cognitive load or due to impaired
    planning under stress.

    Parameter Bounds:
    -----------------
    alpha: [0, 1] - Learning rate for Q-values.
    beta: [0, 10] - Softmax inverse temperature.
    w_model_based_base: [0, 1] - Base weight for the model-based component (0=pure MF, 1=pure MB).
    stai_mb_reduction_factor: [0, 1] - Factor by which STAI score reduces model-based weight.
    """

    def unpack_parameters(self, model_parameters: tuple) -> None:
        self.alpha, self.beta, self.w_model_based_base, self.stai_mb_reduction_factor = model_parameters

    def init_model(self) -> None:
        # The base class already initializes self.trans_counts and self.T
        # We will dynamically update self.trans_counts in value_update
        pass

    def value_update(self, action_1: int, state: int, action_2: int, reward: float) -> None:
        # Update stage 2 Q-value (model-free)
        delta_2 = reward - self.q_stage2[state, action_2]
        self.q_stage2[state, action_2] += self.alpha * delta_2
        
        # Update stage 1 Q-value (model-free) based on updated stage 2 value
        delta_1 = self.q_stage2[state, action_2] - self.q_stage1[action_1]
        self.q_stage1[action_1] += self.alpha * delta_1

        # Update transition counts for the model-based component
        # Add a small pseudo-count (1) to the observed transition for robustness
        self.trans_counts[action_1, state] += 1
        # Recalculate transition probabilities
        self.T = self.trans_counts / self.trans_counts.sum(axis=1, keepdims=True)

    def policy_stage1(self) -> np.ndarray:
        # Calculate effective model-based weight, reduced by anxiety
        # The weight is capped at 0 to prevent negative contributions
        effective_w_mb = max(0, self.w_model_based_base - self.stai * self.stai_mb_reduction_factor)

        # 1. Calculate Model-Based Q-values for Stage 1
        q_mb_stage1 = np.zeros(self.n_choices)
        for a1 in range(self.n_choices):
            # Q_MB(a1) = sum_s2 P(s2|a1) * max_a2 Q_MF(s2, a2)
            # max_a2 Q_MF(s2, a2) represents the expected best reward from state s2
            expected_future_value = np.sum(self.T[a1, s2] * np.max(self.q_stage2[s2]) for s2 in range(self.n_states))
            q_mb_stage1[a1] = expected_future_value
        
        # 2. Blend Model-Free and Model-Based Q-values
        # The self.q_stage1 already holds the Model-Free Q-values
        q_blended_stage1 = effective_w_mb * q_mb_stage1 + (1 - effective_w_mb) * self.q_stage1

        # 3. Apply softmax to the blended values
        return self.softmax(q_blended_stage1, self.beta)

class ParticipantModel2(CognitiveModelBase):
    """
    HYPOTHESIS: This model incorporates separate learning rates for positive and negative reward prediction errors.
    Crucially, it posits that high anxiety (STAI score) makes participants averse to rare (unexpected) transitions.
    If a spaceship leads to an unexpected planet, its value for future choices is penalized,
    and this penalty is amplified by the participant's anxiety level. This reflects a preference for
    predictability and an aversion to uncertainty under high anxiety.

    Parameter Bounds:
    -----------------
    alpha_pos: [0, 1] - Learning rate for positive rewards.
    alpha_neg: [0, 1] - Learning rate for negative rewards.
    beta: [0, 10] - Softmax inverse temperature.
    rare_aversion_base: [0, 5] - Base penalty applied to spaceship after a rare transition.
    stai_rare_aversion_factor: [0, 5] - Factor by which STAI score amplifies the rare transition aversion.
    """

    def unpack_parameters(self, model_parameters: tuple) -> None:
        self.alpha_pos, self.alpha_neg, self.beta, self.rare_aversion_base, self.stai_rare_aversion_factor = model_parameters

    def init_model(self) -> None:
        # Initialize a penalty array for stage 1 actions, applied based on previous trial
        self.rare_transition_penalty = np.zeros(self.n_choices)
        # Define common transition patterns: spaceship 0 to planet 0, spaceship 1 to planet 1
        self.common_transitions = {0: 0, 1: 1}

    def value_update(self, action_1: int, state: int, action_2: int, reward: float) -> None:
        # Determine learning rate based on reward sign for stage 2 update
        alpha_stage2_actual = self.alpha_pos if reward >= 0 else self.alpha_neg

        # Stage 2 update
        delta_2 = reward - self.q_stage2[state, action_2]
        self.q_stage2[state, action_2] += alpha_stage2_actual * delta_2
        
        # Stage 1 update. Uses alpha_pos as it learns from the updated q_stage2 value,
        # which is usually taken as a positive target value (or at least its sign is not directly the reward sign).
        delta_1 = self.q_stage2[state, action_2] - self.q_stage1[action_1]
        self.q_stage1[action_1] += self.alpha_pos * delta_1

    def pre_trial(self) -> None:
        # Reset the penalty for the current trial. It's computed in post_trial based on the *previous* trial.
        self.rare_transition_penalty = np.zeros(self.n_choices)

    def post_trial(self, action_1: int, state: int, action_2: int, reward: float) -> None:
        # Check if the last transition was rare (i.e., not a common transition)
        is_rare_transition = (self.common_transitions.get(action_1) != state)
        
        if is_rare_transition:
            # Calculate the effective aversion, amplified by STAI
            effective_aversion = self.rare_aversion_base + self.stai * self.stai_rare_aversion_factor
            # Apply penalty to the spaceship that led to the rare transition
            # This penalty will be subtracted from Q-values in the next policy_stage1 call
            self.rare_transition_penalty[action_1] = -effective_aversion
        
        # Call base class post_trial to store last actions/state/reward
        super().post_trial(action_1, state, action_2, reward)

    def policy_stage1(self) -> np.ndarray:
        # Apply the rare transition penalty to the Q-values before softmax selection for stage 1
        q_penalized_stage1 = self.q_stage1 + self.rare_transition_penalty
        return self.softmax(q_penalized_stage1, self.beta)

class ParticipantModel3(CognitiveModelBase):
    """
    HYPOTHESIS: This model proposes that participants learn from reward prediction errors (RPEs)
    with a general learning rate, but have distinct sensitivities to positive and negative RPEs.
    Specifically, high anxiety (STAI score) is hypothesized to amplify the sensitivity to negative
    outcomes, meaning negative RPEs have a stronger impact on value updates. This reflects an
    attentional bias towards potential losses or errors under anxiety.

    Parameter Bounds:
    -----------------
    alpha: [0, 1] - General learning rate.
    beta: [0, 10] - Softmax inverse temperature.
    positive_rpe_sensitivity: [0, 2] - Multiplier for positive RPEs.
    negative_rpe_sensitivity_base: [0, 2] - Base multiplier for negative RPEs.
    stai_neg_rpe_factor: [0, 2] - Factor by which STAI score amplifies negative RPE sensitivity.
    """

    def unpack_parameters(self, model_parameters: tuple) -> None:
        self.alpha, self.beta, self.positive_rpe_sensitivity, self.negative_rpe_sensitivity_base, self.stai_neg_rpe_factor = model_parameters

    def value_update(self, action_1: int, state: int, action_2: int, reward: float) -> None:
        # Calculate effective negative RPE sensitivity, amplified by STAI
        effective_neg_rpe_sensitivity = self.negative_rpe_sensitivity_base + self.stai * self.stai_neg_rpe_factor
        
        # Stage 2 update
        delta_2 = reward - self.q_stage2[state, action_2]
        if delta_2 >= 0:
            self.q_stage2[state, action_2] += self.alpha * self.positive_rpe_sensitivity * delta_2
        else: # delta_2 < 0
            self.q_stage2[state, action_2] += self.alpha * effective_neg_rpe_sensitivity * delta_2
        
        # Stage 1 update
        # The target for stage 1 is the updated Q-value of the chosen action in stage 2
        delta_1 = self.q_stage2[state, action_2] - self.q_stage1[action_1]
        if delta_1 >= 0:
            self.q_stage1[action_1] += self.alpha * self.positive_rpe_sensitivity * delta_1
        else: # delta_1 < 0
            self.q_stage1[action_1] += self.alpha * effective_neg_rpe_sensitivity * delta_1

cognitive_model3 = make_cognitive_model(ParticipantModel3)
