An excellent cognitive scientist is not only adept at formulating theories but also at testing them through simulation. By simulating an agent's behavior according to a model's principles, we can generate synthetic data to understand the model's predictions and verify its implementation.

Here is the Python code to simulate trial-by-trial decisions from a cognitive model, such as the `ParticipantModel2` you've provided. This function is designed to be general, accepting any model class that adheres to the `CognitiveModelBase` structure.

```python
import numpy as np
from typing import Type

# Assume CognitiveModelBase and its subclasses (like ParticipantModel2) are defined elsewhere.
# from your_model_definitions import CognitiveModelBase 

def simulate_model(
    ModelClass: Type, # Use Type from typing to hint a class
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1: np.ndarray,
    drift2: np.ndarray,
    drift3: np.ndarray,
    drift4: np.ndarray,
    seed: int | None = None,
) -> tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]:
    """
    Simulate a two-step task trajectory from a cognitive model class.

    This function instantiates a given model and steps it through a simulated
    two-step task, making choices and learning from outcomes on each trial.

    Task structure:
      - Stage 1: choose action_1 in {0,1}
      - Transition: state in {0,1} sampled from model.T[action_1]
      - Stage 2: choose action_2 in {0,1} conditioned on state
      - Reward: sampled from drifting reward probabilities:
          state 0: [drift1[t], drift2[t]] for actions 0/1
          state 1: [drift3[t], drift4[t]] for actions 0/1

    Parameters
    ----------
    ModelClass : type
        A subclass of CognitiveModelBase (e.g., ParticipantModel2).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score, passed to the model constructor.
    model_parameters : tuple
        Parameters expected by ModelClass.unpack_parameters(...).
    drift1, drift2, drift3, drift4 : np.ndarray
        Trial-wise reward probabilities for the 4 (state, action) pairs.
    seed : int or None, optional
        RNG seed for reproducibility.

    Returns
    -------
    stage1_choice : np.ndarray, shape (n_trials,)
        The sequence of first-stage actions taken by the agent.
    state2 : np.ndarray, shape (n_trials,)
        The sequence of second-stage states reached.
    stage2_choice : np.ndarray, shape (n_trials,)
        The sequence of second-stage actions taken.
    reward : np.ndarray, shape (n_trials,)
        The sequence of rewards received.
    """
    rng = np.random.default_rng(seed)

    # Instantiate the cognitive model
    model = ModelClass(
        n_trials=n_trials, stai=float(stai), model_parameters=model_parameters
    )

    # Pre-allocate arrays to store the simulation trajectory
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Loop through each trial to simulate the agent's behavior
    for t in range(n_trials):
        # The model's internal state must be aware of the current trial
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        p1 = model.policy_stage1()
        # Safety normalization to handle potential floating-point inaccuracies
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 /= np.sum(p1)
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        p_transition = model.T[a1]
        p_transition = np.clip(p_transition, 1e-12, 1.0)
        p_transition /= np.sum(p_transition)
        s = int(rng.choice([0, 1], p=p_transition))

        # --- Stage 2 Choice ---
        p2 = model.policy_stage2(s)
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 /= np.sum(p2)
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Determine and Sample Reward ---
        if s == 0:
            reward_prob = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:  # s == 1
            reward_prob = float(drift3[t]) if a2 == 0 else float(drift4[t])
        
        r = int(rng.random() < reward_prob)

        # --- Update Model and Store Results ---
        # The model learns from the outcome of the trial
        model.value_update(a1, s, a2, float(r))
        model.post_trial(a1, s, a2, float(r))

        # Record the events of the trial
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
