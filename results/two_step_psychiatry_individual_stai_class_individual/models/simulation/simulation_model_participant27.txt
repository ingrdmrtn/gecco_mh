import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using ParticipantModel3 logic.
    
    The key feature of ParticipantModel3 is that anxiety (stai) scales a 
    perseveration bonus added to the Q-value of the previously chosen action 
    at Stage 1.

    Parameters
    ----------
    ModelClass : type
        The class definition (ParticipantModel3).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score.
    model_parameters : tuple
        (alpha, beta, pers_k) required by ParticipantModel3.
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 terminal states over time.
    seed : int or None
        Random number generator seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model
    # This initializes Q-values, transition matrix T, and unpacks parameters
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Update internal trial counter and run pre-trial hooks
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # ParticipantModel3.policy_stage1() adds the anxiety-weighted 
        # perseveration bonus to the Q-values before softmax.
        p1 = model.policy_stage1()
        
        # Numerical safety for probabilities
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        # Sample action 1
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Standard probabilistic transition based on model.T
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # Standard softmax on Q_stage2[state]
        p2 = model.policy_stage2(s)
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action 2
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine probability based on current state and action
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < pr)

        # --- Model Update ---
        # 1. Update Q-values (TD learning)
        model.value_update(a1, s, a2, float(r))
        
        # 2. Post-trial bookkeeping (crucial for ParticipantModel3 to know 'last_action1')
        model.post_trial(a1, s, a2, float(r))

        # Store history
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
