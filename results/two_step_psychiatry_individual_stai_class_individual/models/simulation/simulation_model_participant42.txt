To simulate the `ParticipantModel2` (Anxiety-Modulated Memory Decay), we need to instantiate the class and step through the trials, ensuring that the specific `post_trial` logic (where the decay happens) is executed after every reward update.

Here is the simulation code:

```python
import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the Anxiety-Modulated Memory Decay model.

    Parameters
    ----------
    ModelClass : type
        The class ParticipantModel2 (or compatible subclass).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score.
    model_parameters : tuple
        (alpha, beta, decay_base, decay_stai).
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 (state, action) pairs over time.
    seed : int or None
        RNG seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model
    # This sets up q-values, unpacks parameters (calculating decay_rate), etc.
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        p1 = model.policy_stage1()
        # Safety normalization
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Transition probabilities based on stage 1 choice
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        p2 = model.policy_stage2(s)
        # Safety normalization
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Outcome ---
        # Determine probability based on current state and action
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])
        
        r = int(rng.random() < pr)

        # --- Model Update ---
        # 1. Standard Q-learning update
        model.value_update(a1, s, a2, float(r))
        
        # 2. Post-trial bookkeeping AND Memory Decay logic
        # This calls ParticipantModel2.post_trial, which applies the decay
        # to unchosen options based on the calculated decay_rate.
        model.post_trial(a1, s, a2, float(r))

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
