import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the provided CognitiveModel class.

    Parameters
    ----------
    ModelClass : type
        A subclass of CognitiveModelBase (e.g., ParticipantModel1).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score.
    model_parameters : tuple
        Parameters expected by ModelClass.unpack_parameters(...).
    drift1, drift2, drift3, drift4 : array-like
        Trial-wise reward probabilities for the 4 (state, action) pairs.
    seed : int or None
        RNG seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model
    # This sets up Q-values, transition matrix T, and unpacks parameters
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Update internal trial counter and run pre-trial hooks
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # Get action probabilities from the model's policy
        p1 = model.policy_stage1()
        
        # Numerical safety: clip and re-normalize to ensure sum is exactly 1.0
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        # Sample action 1
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to State 2 ---
        # Get transition probabilities based on action 1 (usually 0.7/0.3)
        p_trans = model.T[a1]
        p_trans = np.clip(p_trans, 1e-12, 1.0)
        p_trans = p_trans / np.sum(p_trans)
        
        # Sample state (0 or 1)
        s = int(rng.choice([0, 1], p=p_trans))

        # --- Stage 2 Choice ---
        # Get action probabilities for the specific state reached
        p2 = model.policy_stage2(s)
        
        # Numerical safety
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action 2
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation (Environment) ---
        # Determine the probability of reward based on the current drift arrays
        # Mapping:
        # State 0, Action 0 -> drift1
        # State 0, Action 1 -> drift2
        # State 1, Action 0 -> drift3
        # State 1, Action 1 -> drift4
        if s == 0:
            prob_reward = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            prob_reward = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < prob_reward)

        # --- Model Update ---
        # The agent observes the outcome and updates Q-values/history
        model.value_update(a1, s, a2, float(r))
        model.post_trial(a1, s, a2, float(r))

        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
