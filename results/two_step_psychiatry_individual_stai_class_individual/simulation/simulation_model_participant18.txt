To simulate the behavior of `ParticipantModel2`, we need to instantiate the class, iterate through trials, and use the model's internal policy methods (`policy_stage1`, `policy_stage2`) to make decisions based on the current internal state (Q-values, previous actions). We then update the model using its `value_update` method just as it would happen during fitting.

Here is the simulation code:

```python
import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the logic of ParticipantModel2.

    Parameters
    ----------
    ModelClass : type
        The class definition (e.g., ParticipantModel2).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score (modulates stickiness in ParticipantModel2).
    model_parameters : tuple
        (alpha, beta, phi) required by ParticipantModel2.
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 (state, action) pairs over time.
    seed : int or None
        Random seed for reproducibility.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model
    # This initializes Q-values, calculates phi_eff based on STAI, etc.
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # Get probabilities from the model (includes stickiness logic via phi_eff)
        p1 = model.policy_stage1()
        
        # Numerical safety clipping and renormalization
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        # Sample action 1
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Transition matrix T is defined in the base class
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        
        # Sample state
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # Get probabilities for the specific state
        p2 = model.policy_stage2(s)
        
        # Numerical safety
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action 2
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine probability of reward based on current drift values
        # Mapping: State 0 -> (drift1, drift2), State 1 -> (drift3, drift4)
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < pr)

        # --- Model Update ---
        # Crucial: Update the model's internal Q-values and history so the next 
        # trial's stickiness and values are correct.
        model.value_update(a1, s, a2, float(r))
        model.post_trial(a1, s, a2, float(r))

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
