To simulate the behavior of `ParticipantModel3`, we need a function that instantiates the model class and steps through the trials, generating choices based on the model's probabilities and updating the model's internal state (Q-values, history) exactly as the fitting code does.

Here is the simulation code for the anxiety-based stickiness model.

```python
import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using ParticipantModel3 (Anxiety Stickiness).

    Task structure:
      - Stage 1: choose action_1 in {0,1} based on Q-values + stickiness bonus.
      - Transition: state in {0,1} sampled from model.T[action_1].
      - Stage 2: choose action_2 in {0,1} conditioned on state + stickiness bonus.
      - Reward: sampled from drifting reward probabilities.

    Parameters
    ----------
    ModelClass : type
        The class ParticipantModel3.
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score (STAI).
    model_parameters : tuple
        (alpha, beta, stickiness_base, anxiety_stick).
    drift1, drift2, drift3, drift4 : array-like (length n_trials)
        Trial-wise reward probabilities for the 4 (state, action) pairs.
    seed : int or None
        RNG seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the specific model (ParticipantModel3)
    # This initializes Q-values, transition matrix T, and unpacks parameters
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # The model calculates probabilities including the stickiness bonus
        # based on model.last_action1 and model.stai
        p1 = model.policy_stage1()
        
        # Numerical safety clipping and renormalization
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        # Sample action 1
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Transition probabilities based on action 1 (common/rare)
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        
        # Sample state
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # The model calculates probabilities for the specific state,
        # including stickiness if the previous trial was in the same state
        p2 = model.policy_stage2(s)
        
        # Numerical safety
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action 2
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine probability of reward based on current state, action, and drift
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < pr)

        # --- Model Update ---
        # Crucial: Update Q-values (TD learning) and history (last_action, last_state)
        # so the stickiness logic works correctly on the next iteration.
        model.value_update(a1, s, a2, float(r))
        model.post_trial(a1, s, a2, float(r))

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
