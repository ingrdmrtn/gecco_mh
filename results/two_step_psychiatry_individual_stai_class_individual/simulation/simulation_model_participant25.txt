import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the provided Cognitive Model class.
    
    This function instantiates the model and steps through trials, generating
    choices based on the model's policy (including specific logic like anxiety-driven 
    stickiness in ParticipantModel2) and updating internal values based on outcomes.

    Parameters
    ----------
    ModelClass : class
        The class of the model to simulate (e.g., ParticipantModel2).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score.
    model_parameters : tuple
        Parameters required by the model (e.g., (alpha, beta, k_stick)).
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 state-action pairs over time.
    seed : int, optional
        Random seed for reproducibility.

    Returns
    -------
    stage1_choice : np.ndarray
    state2 : np.ndarray
    stage2_choice : np.ndarray
    reward : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model
    # This sets up Q-values, transition matrix T, and unpacks parameters (alpha, beta, k_stick, etc.)
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Sync trial counter and run pre-trial hooks
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # The model calculates probabilities based on Q-values and (for Model2) stickiness
        p1 = model.policy_stage1()
        
        # Numerical safety for sampling
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to State 2 ---
        # Transition probabilities depend on action 1 (defined in model.T)
        p_trans = model.T[a1]
        p_trans = np.clip(p_trans, 1e-12, 1.0)
        p_trans = p_trans / np.sum(p_trans)
        
        s = int(rng.choice([0, 1], p=p_trans))

        # --- Stage 2 Choice ---
        # Choice depends on the specific state arrived at
        p2 = model.policy_stage2(s)
        
        # Numerical safety
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine reward probability based on current state and action
        if s == 0:
            prob_reward = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            prob_reward = float(drift3[t]) if a2 == 0 else float(drift4[t])

        r = int(rng.random() < prob_reward)

        # --- Model Update ---
        # Update Q-values based on the experience
        model.value_update(a1, s, a2, float(r))
        
        # Post-trial bookkeeping (CRITICAL for Model2: this updates last_action1 for stickiness)
        model.post_trial(a1, s, a2, float(r))

        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
