import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the provided ModelClass (e.g., ParticipantModel2).

    This function handles the specific logic of ParticipantModel2, where stage 1 choices
    are influenced by an anxiety-driven stickiness bonus added to the Q-values.

    Parameters
    ----------
    ModelClass : type
        The class definition (e.g., ParticipantModel2).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score (STAI).
    model_parameters : tuple
        Parameters required by the model (alpha, beta, stickiness_factor).
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 (state, action) pairs over time.
    seed : int or None
        Random number generator seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model
    # This initializes Q-values, transition matrix T, and unpacks parameters
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):

        model.trial = t

        # --- Stage 1 Choice ---

        p1 = model.policy_stage1()

        p1 = p1 / np.sum(p1)
        
        # Sample action 1
        a1 = int(rng.choice([0, 1], p=p1))

        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        
        # Sample state
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # Standard softmax on Q-values for the specific state
        p2 = model.policy_stage2(s)
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action 2
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine probability based on current state and action choice
        # State 0 corresponds to drift1 (action 0) and drift2 (action 1)
        # State 1 corresponds to drift3 (action 0) and drift4 (action 1)
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < pr)

        # --- Model Update ---
        # Update Q-values based on the outcome (TD learning)
        model.value_update(a1, s, a2, float(r))
        
        # Update history (last_action1, last_state, etc.) for the next trial's stickiness calculation
        model.post_trial(a1, s, a2, float(r))

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
