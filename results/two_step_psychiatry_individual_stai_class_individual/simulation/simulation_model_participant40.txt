To simulate the behavior of `ParticipantModel3` (Anxiety-Modulated Perseveration), we need to instantiate the class and step through the trials, generating choices based on the model's policy methods (`policy_stage1`, `policy_stage2`) and updating its internal state using `value_update`.

Here is the simulation code:

```python
import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the logic of ParticipantModel3
    (Anxiety-Modulated Perseveration).

    Parameters
    ----------
    ModelClass : type
        The class definition (ParticipantModel3).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score.
    model_parameters : tuple
        (alpha, beta, p_scale).
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 (state, action) pairs over time.
    seed : int or None
        RNG seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model. This sets up q-values, transition matrix T, etc.
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # ParticipantModel3 overrides policy_stage1 to add the anxiety bonus
        p1 = model.policy_stage1()
        
        # Numerical safety clipping and renormalization
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Standard transition matrix from base class
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # Standard softmax on Q-values (ParticipantModel3 uses base implementation)
        p2 = model.policy_stage2(s)
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Outcome ---
        # Determine probability based on current state and action
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        r = int(rng.random() < pr)

        # --- Model Update ---
        # Update Q-values based on prediction errors
        model.value_update(a1, s, a2, float(r))
        
        # Update history (crucial for ParticipantModel3's perseveration logic next trial)
        model.post_trial(a1, s, a2, float(r))

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
