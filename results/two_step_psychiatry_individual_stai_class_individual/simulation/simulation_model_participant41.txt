import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the logic of ParticipantModel2.
    
    This model hypothesizes that anxiety (STAI) increases choice perseveration (stickiness)
    in Stage 1. The simulation ensures that the model's internal state (q-values, 
    last_action tracking) is updated trial-by-trial so that the stickiness bonus 
    is applied correctly.

    Parameters
    ----------
    ModelClass : type
        The class ParticipantModel2.
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score.
    model_parameters : tuple
        (alpha, beta, stick_factor).
    drift1, drift2, drift3, drift4 : array-like
        Reward probabilities for the 4 terminal states over time.
    seed : int or None
        RNG seed.

    Returns
    -------
    stage1_choice : np.ndarray
    state2        : np.ndarray
    stage2_choice : np.ndarray
    reward        : np.ndarray
    """
    rng = np.random.default_rng(seed)

    # Initialize the model instance
    # This sets up q_stage1, q_stage2, and unpacks parameters (alpha, beta, stick_factor)
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Update internal trial counter and run any pre-trial logic
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # The model calculates probabilities based on Q-values + (STAI * stickiness)
        p1 = model.policy_stage1()
        
        # Safety normalization
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        # Sample action 1
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Transition probabilities based on the fixed transition matrix T
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        
        # Sample state
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # Standard softmax on Q-values for the specific state
        p2 = model.policy_stage2(s)
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action 2
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine probability based on current state and action using the drift arrays
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < pr)

        # --- Model Update ---
        # Crucial step: Update Q-values via TD learning and store history 
        # so `last_action1` is available for the next trial's stickiness calculation.
        model.value_update(a1, s, a2, float(r))
        model.post_trial(a1, s, a2, float(r))

        # Store data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
