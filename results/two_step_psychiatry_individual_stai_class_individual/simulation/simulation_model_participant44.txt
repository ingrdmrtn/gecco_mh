import numpy as np

def simulate_model(
    ModelClass,
    n_trials: int,
    stai: float,
    model_parameters: tuple,
    drift1, drift2, drift3, drift4,
    seed: int | None = None,
):
    """
    Simulate a two-step task trajectory using the provided ModelClass (e.g., ParticipantModel1).

    Task structure:
      - Stage 1: choose action_1 in {0,1} based on model.policy_stage1()
      - Transition: state in {0,1} sampled from model.T[action_1]
      - Stage 2: choose action_2 in {0,1} conditioned on state based on model.policy_stage2()
      - Reward: sampled from drifting reward probabilities provided in drift arguments.

    Parameters
    ----------
    ModelClass : type
        A subclass of CognitiveModelBase (e.g., ParticipantModel1).
    n_trials : int
        Number of trials to simulate.
    stai : float
        Participant anxiety score; passed into the model constructor.
    model_parameters : tuple
        Parameters expected by ModelClass.unpack_parameters(...).
    drift1, drift2, drift3, drift4 : array-like (length n_trials)
        Trial-wise reward probabilities for the 4 (state, action) pairs.
        State 0: Action 0 -> drift1, Action 1 -> drift2
        State 1: Action 0 -> drift3, Action 1 -> drift4
    seed : int or None
        RNG seed.

    Returns
    -------
    stage1_choice : np.ndarray, shape (n_trials,)
    state2        : np.ndarray, shape (n_trials,)
    stage2_choice : np.ndarray, shape (n_trials,)
    reward        : np.ndarray, shape (n_trials,)
    """
    rng = np.random.default_rng(seed)

    # Instantiate the model. This sets up internal state (q-values, habit traces, etc.)
    # and unpacks the parameters.
    model = ModelClass(n_trials=n_trials, stai=float(stai), model_parameters=model_parameters)

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Update model's internal trial counter and run pre-trial logic
        model.trial = t
        model.pre_trial()

        # --- Stage 1 Choice ---
        # Get probabilities from the model's policy
        p1 = model.policy_stage1()
        
        # Safety normalization to handle potential numerical instability
        p1 = np.clip(p1, 1e-12, 1.0)
        p1 = p1 / np.sum(p1)
        
        # Sample action
        a1 = int(rng.choice([0, 1], p=p1))

        # --- Transition to Stage 2 ---
        # Get transition probabilities based on stage 1 action
        pT = model.T[a1]
        pT = np.clip(pT, 1e-12, 1.0)
        pT = pT / np.sum(pT)
        
        # Sample state
        s = int(rng.choice([0, 1], p=pT))

        # --- Stage 2 Choice ---
        # Get probabilities conditioned on the observed state
        p2 = model.policy_stage2(s)
        p2 = np.clip(p2, 1e-12, 1.0)
        p2 = p2 / np.sum(p2)
        
        # Sample action
        a2 = int(rng.choice([0, 1], p=p2))

        # --- Reward Generation ---
        # Determine probability of reward based on current state, action, and drift arrays
        if s == 0:
            pr = float(drift1[t]) if a2 == 0 else float(drift2[t])
        else:
            pr = float(drift3[t]) if a2 == 0 else float(drift4[t])

        # Sample reward (Bernoulli trial)
        r = int(rng.random() < pr)

        # --- Model Update ---
        # Store probabilities for logging (optional, but consistent with model structure)
        model.p_choice_1[t] = p1[a1]
        model.p_choice_2[t] = p2[a2]

        # Update values (Q-learning, Habit learning, etc.)
        model.value_update(a1, s, a2, float(r))
        
        # Run post-trial logic (bookkeeping)
        model.post_trial(a1, s, a2, float(r))

        # Store simulation data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
