class ParticipantModel1(CognitiveModelBase):
    """
    HYPOTHESIS: Anxiety modulates memory decay for unchosen options. 
    High anxiety consumes cognitive resources (working memory load), leading to faster 
    forgetting (decay) of the value of the option not currently selected. 
    The unchosen option's value decays towards 0.
    
    Parameter Bounds:
    -----------------
    alpha: [0, 1]
    beta: [0, 10]
    decay_base: [0, 1]
    decay_stai: [-1, 1]
    """
    def unpack_parameters(self, model_parameters: tuple) -> None:
        self.alpha, self.beta, self.decay_base, self.decay_stai = model_parameters

    def value_update(self, action_1: int, state: int, action_2: int, reward: float) -> None:
        # Standard TD update for the chosen path
        super().value_update(action_1, state, action_2, reward)
        
        # Calculate anxiety-modulated decay rate
        decay_rate = self.decay_base + (self.decay_stai * self.stai)
        decay_rate = np.clip(decay_rate, 0.0, 1.0)
        
        # Apply decay to the unchosen Stage 1 action
        unchosen_a1 = 1 - action_1
        self.q_stage1[unchosen_a1] *= (1.0 - decay_rate)

cognitive_model1 = make_cognitive_model(ParticipantModel1)


class ParticipantModel2(CognitiveModelBase):
    """
    HYPOTHESIS: Anxiety differentially affects decision noise at Stage 1 (planning) 
    versus Stage 2 (reaction). This model proposes that Stage 1 exploration/exploitation 
    (beta) is modulated by STAI, while Stage 2 beta remains stable. 
    Anxiety might specifically impair the higher-level choice between spaceships.
    
    Parameter Bounds:
    -----------------
    alpha: [0, 1]
    beta_stage2: [0, 10]
    beta1_base: [0, 10]
    beta1_stai: [-5, 5]
    """
    def unpack_parameters(self, model_parameters: tuple) -> None:
        self.alpha, self.beta_stage2, self.beta1_base, self.beta1_stai = model_parameters

    def policy_stage1(self) -> np.ndarray:
        # Modulate Stage 1 beta with anxiety
        beta_1 = self.beta1_base + (self.beta1_stai * self.stai)
        beta_1 = np.maximum(0.0, beta_1) # Ensure non-negative
        return self.softmax(self.q_stage1, beta_1)

    def policy_stage2(self, state: int) -> np.ndarray:
        # Use fixed beta for Stage 2
        return self.softmax(self.q_stage2[state], self.beta_stage2)

cognitive_model2 = make_cognitive_model(ParticipantModel2)


class ParticipantModel3(CognitiveModelBase):
    """
    HYPOTHESIS: Anxiety modulates learning specifically from 'surprising' (rare transition) outcomes.
    Participants may discount (ignore) or over-weight outcomes that occur after a rare transition 
    depending on their anxiety level. This model splits the learning rate alpha into 
    'common' and 'rare' components, with the rare component modulated by STAI.
    
    Parameter Bounds:
    -----------------
    alpha_common: [0, 1]
    beta: [0, 10]
    alpha_rare_base: [0, 1]
    alpha_rare_stai: [-1, 1]
    """
    def unpack_parameters(self, model_parameters: tuple) -> None:
        self.alpha_common, self.beta, self.alpha_rare_base, self.alpha_rare_stai = model_parameters

    def value_update(self, action_1: int, state: int, action_2: int, reward: float) -> None:
        # Determine if transition was common or rare
        # Structure: A(0)->X(0) and U(1)->Y(1) are Common. 0->1 and 1->0 are Rare.
        is_common = (action_1 == state)
        
        if is_common:
            alpha_eff = self.alpha_common
        else:
            # Modulate learning rate for rare transitions based on anxiety
            alpha_rare = self.alpha_rare_base + (self.alpha_rare_stai * self.stai)
            alpha_eff = np.clip(alpha_rare, 0.0, 1.0)
            
        # Manual update with the specific effective alpha
        delta_2 = reward - self.q_stage2[state, action_2]
        self.q_stage2[state, action_2] += alpha_eff * delta_2
        
        delta_1 = self.q_stage2[state, action_2] - self.q_stage1[action_1]
        self.q_stage1[action_1] += alpha_eff * delta_1

cognitive_model3 = make_cognitive_model(ParticipantModel3)
