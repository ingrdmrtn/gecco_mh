def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Modulated Win-Stay Bias.
    
    This model suggests that anxiety increases the propensity to repeat a choice 
    specifically if it was previously rewarded ("Win-Stay"), independent of the 
    learned value of that choice. This is distinct from simple perseveration; it 
    reflects a "don't fix what isn't broken" safety heuristic driven by anxiety.
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weight.
    ws_base: [0, 5] Baseline bonus added to logits for repeating a rewarded action.
    ws_stai_sens: [0, 5] Increase in Win-Stay bonus per unit of STAI.
    """
    learning_rate, beta, w, ws_base, ws_stai_sens = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    ws_bonus = ws_base + (ws_stai_sens * participant_stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1
    prev_reward = 0

    for trial in range(n_trials):

        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net

        if prev_a1 != -1 and prev_reward == 1:
            logits[int(prev_a1)] += ws_bonus
        
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]
        prev_a1 = chosen_a1

        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        r = reward[trial]
        prev_reward = r

        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
