Here are three cognitive models designed to explain the participant's behavior in the two-step task, incorporating their high anxiety (STAI) score into the learning and decision-making mechanisms.

### Model 1: Anxiety-Modulated Exploration (MB/MF Hybrid)
This model hypothesizes that high anxiety (high STAI) reduces the willingness to explore or take risks. It implements a standard hybrid Model-Based (MB) and Model-Free (MF) reinforcement learning architecture, but the inverse temperature parameter `beta` (which controls the explore/exploit trade-off) is modulated by the STAI score. Higher anxiety leads to a higher effective beta, resulting in more deterministic (exploitative) choices and less random exploration.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where anxiety modulates exploration.
    
    Hypothesis: Higher anxiety (STAI) leads to reduced exploration (higher beta/exploitation).
    The effective beta is calculated as: beta_base * (1 + w_stai * stai).
    
    Parameters:
    - learning_rate: [0, 1] Rate of updating Q-values.
    - beta_base: [0, 10] Base inverse temperature for softmax.
    - w_stai: [0, 5] Weight of STAI score on increasing beta (reducing exploration).
    - mixing_weight: [0, 1] Balance between MB (1) and MF (0) control.
    """
    learning_rate, beta_base, w_stai, mixing_weight = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective beta based on anxiety
    beta = beta_base * (1 + w_stai * stai_score)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based value calculation: T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value: w * MB + (1-w) * MF
        q_net = mixing_weight * q_stage1_mb + (1 - mixing_weight) * q_stage1_mf
        
        # Softmax choice
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Transition to state
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        # Standard model-free choice at stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # --- Updates ---
        # TD Error Stage 1 (SARSA-style update using the value of the state reached)
        # Note: In standard 2-step, Stage 1 MF is often updated by Q(s2, a2) or just V(s2)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        # TD Error Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Learning Rate Asymmetry (Pessimism Bias)
This model posits that anxiety affects *how* the participant learns from feedback. Specifically, it tests the hypothesis that high-anxiety individuals are more sensitive to negative outcomes (lack of reward) or less sensitive to positive outcomes. The STAI score modulates the learning rate specifically for reward prediction errors, creating a "pessimism bias" where bad news updates beliefs faster than good news.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Free learner with anxiety-modulated learning rates.
    
    Hypothesis: Anxiety induces a pessimism bias. 
    The learning rate for positive prediction errors is scaled down by STAI, 
    making the agent learn less from success and retain fear of failure.
    lr_pos = alpha * (1 - w_stai * stai)
    lr_neg = alpha
    
    Parameters:
    - alpha: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w_stai: [0, 1] Weight of STAI reducing positive learning rate.
    - eligibility: [0, 1] Eligibility trace (how much stage 2 reward updates stage 1).
    """
    alpha, beta, w_stai, eligibility = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate asymmetric learning rates
    # High anxiety -> lower learning rate for positive outcomes (dampened reward sensitivity)
    lr_pos = alpha * (1 - w_stai * stai_score)
    lr_neg = alpha 
    
    # Ensure bounds
    if lr_pos < 0: lr_pos = 0

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # --- Updates ---
        # Calculate Stage 2 Prediction Error
        delta_stage2 = reward[trial] - q_stage2[state_idx, int(action_2[trial])]
        
        # Select learning rate based on sign of error
        current_lr = lr_pos if delta_stage2 > 0 else lr_neg
        
        # Update Stage 2
        q_stage2[state_idx, int(action_2[trial])] += current_lr * delta_stage2
        
        # Update Stage 1 (TD(1) / Eligibility trace approach)
        # Stage 1 is updated by the reward received at Stage 2, scaled by eligibility
        q_stage1[int(action_1[trial])] += current_lr * eligibility * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Model-Based Deficit
This model investigates the trade-off between Model-Based (planning) and Model-Free (habitual) control. The literature suggests stress and anxiety can impair prefrontal cortex function, leading to a reliance on habitual (Model-Free) control. Here, the `mixing_weight` (w) which determines the degree of Model-Based control is directly reduced by the STAI score.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF learner where anxiety impairs Model-Based control.
    
    Hypothesis: High anxiety consumes cognitive resources, reducing the ability
    to use the transition matrix (Model-Based planning).
    The mixing weight 'w' (1=Full MB, 0=Full MF) is reduced by STAI.
    w_effective = w_base * (1 - gamma_stai * stai)
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w_base: [0, 1] Base Model-Based weighting parameter.
    - gamma_stai: [0, 1] How strongly STAI reduces the Model-Based weight.
    """
    learning_rate, beta, w_base, gamma_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective mixing weight
    # If anxiety is high, w_effective drops, making the agent more Model-Free
    w_effective = w_base * (1.0 - gamma_stai * stai_score)
    if w_effective < 0: w_effective = 0

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        # 1. Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Integrate MB and MF values
        q_integrated = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # --- Updates ---
        # Standard TD updates for MF values
        
        # Stage 1 MF update (using Q-value of chosen stage 2 action)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        # Stage 2 MF update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
