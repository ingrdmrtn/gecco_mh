Here are three cognitive models designed to explain the participant's behavior, incorporating their high anxiety (STAI) score into the decision-making mechanics.

### Model 1: Anxiety-Modulated Mixing Weight (Hybrid Learner)
This model posits that anxiety influences the balance between Model-Based (planning) and Model-Free (habitual) control. High anxiety often leads to a reliance on habitual, repetitive behaviors (Model-Free) rather than flexible, goal-directed planning (Model-Based) due to cognitive load or stress. Here, the mixing weight `w` is a function of the STAI score.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the mixing weight (w) is modulated by anxiety.
    High anxiety (STAI) reduces Model-Based control (lower w), increasing reliance on habits.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta: Inverse temperature for softmax choice [0, 10]
    - w_base: Baseline mixing weight for a person with 0 anxiety [0, 1]
    - w_stai_slope: How strongly anxiety reduces the mixing weight [0, 1]
    """
    learning_rate, beta, w_base, w_stai_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate mixing weight w based on STAI. 
    # Formula: w = w_base - (slope * stai). Bounded between 0 and 1.
    # Higher anxiety -> lower w -> more Model-Free behavior.
    w = w_base - (w_stai_slope * stai_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free stage 2 (also used for MB calculation)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: Transition * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: w * MB + (1-w) * MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Transition ---
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Choice ---
        # Standard Model-Free choice based on Q-values of aliens at the current planet
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 Update (TD Error)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update (TD lambda=1 logic effectively, connecting Stage 1 to Stage 2 outcome)
        # Using the value of the state actually visited (q_stage2_mf[state_idx, a2]) as the target
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Learning Rate Asymmetry
This model suggests that high anxiety alters how individuals process positive versus negative feedback. Anxious individuals are often hypersensitive to negative outcomes (punishment/lack of reward). Here, the learning rate is split: a base learning rate is modified by STAI specifically for negative prediction errors (disappointments), making the agent learn faster from failure than success when anxiety is high.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Free learner where anxiety increases the learning rate for negative prediction errors.
    Anxious individuals may over-update their values in response to missing a reward.
    
    Parameters:
    - lr_pos: Learning rate for positive prediction errors [0, 1]
    - lr_neg_base: Base learning rate for negative prediction errors [0, 1]
    - lr_stai_boost: Additional learning rate magnitude added by STAI for negative errors [0, 1]
    - beta: Inverse temperature [0, 10]
    """
    lr_pos, lr_neg_base, lr_stai_boost, beta = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective negative learning rate
    # High anxiety increases sensitivity to negative outcomes
    lr_neg = lr_neg_base + (lr_stai_boost * stai_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Free Q-values for simplicity to isolate learning rate effect
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Learning ---
        r = reward[trial]
        
        # Determine Learning Rate based on Prediction Error sign for Stage 2
        # Note: We calculate PE first to decide which LR to use
        pe_stage2 = r - q_stage2[state_idx, a2]
        current_lr2 = lr_pos if pe_stage2 >= 0 else lr_neg
        
        q_stage2[state_idx, a2] += current_lr2 * pe_stage2
        
        # Stage 1 Update (SARSA-like update using stage 2 value)
        # We use the updated Stage 2 value as the target
        target_stage1 = q_stage2[state_idx, a2]
        pe_stage1 = target_stage1 - q_stage1[a1]
        
        # Apply asymmetric learning rate to Stage 1 as well
        current_lr1 = lr_pos if pe_stage1 >= 0 else lr_neg
        q_stage1[a1] += current_lr1 * pe_stage1
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Exploration Noise (Inverse Temperature Modulation)
This model posits that anxiety disrupts consistent decision-making, effectively acting as "neural noise" or increased randomness in choice. Instead of altering learning or planning depth, STAI modulates the `beta` (inverse temperature) parameter. High anxiety lowers `beta`, leading to more random (stochastic) behavior and less exploitation of learned values.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Based learner where anxiety modulates the inverse temperature (beta).
    High anxiety increases decision noise (lowers beta), making choices more random.
    
    Parameters:
    - learning_rate: Update rate for Q-values [0, 1]
    - beta_base: Baseline inverse temperature [0, 10]
    - beta_stai_penalty: Reduction in beta proportional to STAI score [0, 10]
    """
    learning_rate, beta_base, beta_stai_penalty = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Calculate effective beta
    # High anxiety subtracts from the base consistency/exploitation parameter
    beta_eff = beta_base - (beta_stai_penalty * stai_score)
    beta_eff = np.maximum(beta_eff, 0.0) # Ensure beta doesn't go negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Using a simplified MB structure (Stage 1 is purely MB, Stage 2 is MF)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Choice (Model-Based) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Use beta_eff modulated by anxiety
        exp_q1 = np.exp(beta_eff * q_stage1_mb)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Choice (Model-Free) ---
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Learning ---
        r = reward[trial]
        
        # Standard Q-learning update for Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # No Stage 1 MF update needed as this is a pure MB controller for Stage 1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
