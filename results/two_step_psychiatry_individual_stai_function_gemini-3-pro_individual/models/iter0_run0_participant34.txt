Here are three cognitive models designed to explain the participant's behavior, incorporating their high STAI (anxiety) score of 0.9625.

### Model 1: Anxiety-Modulated Model-Based/Model-Free Hybrid
This model hypothesizes that high anxiety (STAI) shifts the balance between Model-Based (planning) and Model-Free (habitual) control. High anxiety is often associated with reduced cognitive flexibility and increased reliance on habits (Model-Free) or simpler strategies. Here, the `stai` score acts as a mixing weight parameter `w` that determines the relative contribution of the Model-Based system.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free Reinforcement Learning.
    
    Hypothesis: High anxiety (stai) reduces Model-Based control (planning)
    and increases reliance on Model-Free control (habit). The mixing weight 'w'
    is modulated by the STAI score.

    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice rule.
    - w_base: [0, 1] Baseline mixing weight (0 = pure MF, 1 = pure MB).
    """
    learning_rate, beta, w_base = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective mixing weight w based on STAI.
    # Higher anxiety reduces w (less MB, more MF).
    # We clip to ensure it stays within [0, 1].
    w = w_base * (1.0 - stai_score)
    w = np.clip(w, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value Calculation (Bellman equation)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Transition to State
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        # Standard Model-Free Q-learning for the second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]

        # --- Learning / Updating ---
        # Prediction Error Stage 1 (SARSA-like logic for MF)
        chosen_a1 = int(action_1[trial])
        chosen_a2 = int(action_2[trial])
        
        # Update Stage 1 MF value using the value of the state reached
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF value based on reward
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Learning Rate Asymmetry
This model posits that anxiety fundamentally alters how individuals learn from positive versus negative outcomes. Highly anxious individuals often exhibit a "negativity bias" or heightened sensitivity to punishment/lack of reward. In this model, the learning rate is split into positive (`alpha_pos`) and negative (`alpha_neg`) components, and the STAI score amplifies the learning rate for negative prediction errors (or suppresses learning from positive ones).

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Asymmetric Learning Rate Model.
    
    Hypothesis: High anxiety (stai) leads to different learning rates for
    positive vs negative prediction errors. Specifically, anxiety amplifies
    sensitivity to negative outcomes (or lack of reward).

    Parameters:
    - alpha_base: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - anxiety_sens: [0, 1] How much anxiety amplifies negative learning.
    """
    alpha_base, beta, anxiety_sens = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate asymmetric learning rates
    # Positive learning is standard
    alpha_pos = alpha_base
    # Negative learning is boosted by anxiety score and sensitivity parameter
    # If anxiety is high, alpha_neg becomes larger than alpha_pos
    alpha_neg = alpha_base + (anxiety_sens * stai_score * (1.0 - alpha_base))

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Pure Model-Free Q-values for simplicity to isolate learning rate effects
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        chosen_a2 = int(action_2[trial])

        # --- Learning ---
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2[state_idx, chosen_a2]
        eff_alpha2 = alpha_pos if delta_stage2 >= 0 else alpha_neg
        q_stage2[state_idx, chosen_a2] += eff_alpha2 * delta_stage2
        
        # Stage 1 Update (TD-learning)
        delta_stage1 = q_stage2[state_idx, chosen_a2] - q_stage1[chosen_a1]
        eff_alpha1 = alpha_pos if delta_stage1 >= 0 else alpha_neg
        q_stage1[chosen_a1] += eff_alpha1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Decision Noise (Inverse Temperature Modulation)
This model suggests that anxiety acts as a disruptor to decision consistency. Rather than changing *how* values are learned (like Model 2) or *which* system is used (like Model 1), this model proposes that high anxiety increases decision noise (exploratory behavior or confusion), effectively lowering the inverse temperature `beta`.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Softmax Temperature (Noise) Model.
    
    Hypothesis: High anxiety increases decision noise (randomness).
    The effective beta (inverse temperature) is reduced as STAI increases.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_max: [0, 10] Maximum inverse temperature (consistency) for calm subjects.
    - noise_factor: [0, 10] Scaling factor for how much anxiety reduces beta.
    - lambda_eligibility: [0, 1] Eligibility trace parameter (TD-lambda).
    """
    learning_rate, beta_max, noise_factor, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective beta
    # High STAI subtracts from beta_max, making choices more random (lower beta).
    # We ensure beta doesn't drop below 0.
    beta_eff = beta_max - (noise_factor * stai_score)
    beta_eff = np.maximum(beta_eff, 0.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta_eff * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta_eff * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        chosen_a2 = int(action_2[trial])

        # --- Learning with Eligibility Traces (TD-Lambda style) ---
        # This allows the Stage 2 reward to directly update Stage 1 values
        
        # 1. Prediction error at Stage 2
        delta_2 = reward[trial] - q_stage2[state_idx, chosen_a2]
        
        # 2. Update Stage 2
        q_stage2[state_idx, chosen_a2] += learning_rate * delta_2
        
        # 3. Update Stage 1
        # Direct update from Stage 2 value (TD(0) part)
        delta_1 = q_stage2[state_idx, chosen_a2] - q_stage1[chosen_a1]
        q_stage1[chosen_a1] += learning_rate * delta_1
        
        # Eligibility trace update: Stage 1 also learns from Stage 2's RPE
        q_stage1[chosen_a1] += learning_rate * lambda_eligibility * delta_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
