Here are three cognitive models designed to capture decision-making in the two-step task, specifically incorporating the participant's high anxiety (STAI) score.

### Model 1: Anxiety-Modulated Exploration (Model-Based/Model-Free Hybrid)
This model posits that high anxiety (STAI) specifically interferes with the balance between Model-Based (planning) and Model-Free (habitual) control. High anxiety is hypothesized to reduce the weight of model-based planning (`w`) because cognitive resources are consumed by worry, pushing the agent towards habitual responding. Additionally, anxiety increases the inverse temperature (`beta`), making choices more deterministic/rigid (less exploration) as anxiety rises.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model where anxiety (STAI) modulates the mixing weight 'w' and
    exploration 'beta'.
    
    Hypothesis: Higher anxiety reduces model-based control (lower w) and 
    increases rigidity (higher beta).

    Parameters:
    learning_rate: [0, 1] - Rate of updating Q-values.
    base_beta: [0, 10] - Baseline inverse temperature (exploitation vs exploration).
    w_base: [0, 1] - Baseline weight for model-based control.
    stai_beta_sensitivity: [0, 5] - How much STAI increases beta (rigidity).
    """
    learning_rate, base_beta, w_base, stai_beta_sensitivity = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Modulate parameters based on STAI
    # High anxiety -> lower w (less model-based)
    # We clip w to be between 0 and 1
    w = w_base * (1.0 - participant_stai) 
    if w < 0: w = 0
    
    # High anxiety -> higher beta (more rigid/exploitative)
    beta = base_beta + (participant_stai * stai_beta_sensitivity)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2) # Model-free Q-values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free Q-values for stage 2 (2 states, 2 actions)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value Calculation: V_MB = T * max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Q_net = w * Q_MB + (1-w) * Q_MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Transition to state
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updating ---
        # Stage 2 Update (standard Q-learning)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 MF Update (TD-learning using Stage 2 value)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Loss Aversion (Pure Model-Free)
This model assumes the participant relies primarily on Model-Free learning but is highly sensitive to negative outcomes due to anxiety. Instead of a single learning rate, the model uses separate learning rates for positive prediction errors (`alpha_pos`) and negative prediction errors (`alpha_neg`). The STAI score is used to amplify the learning rate for negative prediction errors, reflecting a "negativity bias" common in high anxiety.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Free Q-learning with STAI-modulated asymmetric learning rates.
    
    Hypothesis: High anxiety leads to hypersensitivity to negative prediction errors
    (worse than expected outcomes). STAI increases the effective learning rate 
    for negative deltas.

    Parameters:
    alpha_pos: [0, 1] - Learning rate for positive prediction errors.
    base_alpha_neg: [0, 1] - Baseline learning rate for negative prediction errors.
    beta: [0, 10] - Inverse temperature.
    stai_neg_amp: [0, 5] - Amplification factor for negative learning rate by STAI.
    """
    alpha_pos, base_alpha_neg, beta, stai_neg_amp = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate effective negative learning rate based on anxiety
    # Higher STAI -> Higher alpha_neg (stronger reaction to disappointment)
    alpha_neg = base_alpha_neg + (participant_stai * stai_neg_amp)
    # Clip to ensure valid range [0, 1]
    if alpha_neg > 1.0: alpha_neg = 1.0

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updating ---
        
        # Stage 2 PE
        delta2 = r - q_stage2[state_idx, a2]
        lr2 = alpha_pos if delta2 > 0 else alpha_neg
        q_stage2[state_idx, a2] += lr2 * delta2
        
        # Stage 1 PE (SARSA-style update using Q-value of chosen stage 2 action)
        delta1 = q_stage2[state_idx, a2] - q_stage1[a1]
        lr1 = alpha_pos if delta1 > 0 else alpha_neg
        q_stage1[a1] += lr1 * delta1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Perseveration (Stickiness)
This model adds a "stickiness" or perseveration parameter that biases the agent to repeat the previous choice. High anxiety is hypothesized to increase this perseveration (safety behavior/avoidance of switching). The STAI score scales the magnitude of this choice autocorrelation. The underlying learning mechanism is a simple Model-Based learner.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Based learner with anxiety-modulated choice perseveration (stickiness).
    
    Hypothesis: High anxiety increases the tendency to repeat the previous choice
    regardless of reward (safety behavior). STAI scales the 'stickiness' parameter.

    Parameters:
    learning_rate: [0, 1] - Rate of updating values.
    beta: [0, 10] - Inverse temperature.
    base_stickiness: [0, 5] - Baseline tendency to repeat choice.
    stai_stick_sensitivity: [0, 5] - How much STAI amplifies stickiness.
    """
    learning_rate, beta, base_stickiness, stai_stick_sensitivity = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate effective stickiness
    stickiness = base_stickiness + (participant_stai * stai_stick_sensitivity)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage2_mf = np.zeros((2, 2))
    
    # Keep track of last choice (initialize to -1 so first trial has no stickiness)
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Pure Model-Based Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Add stickiness bonus to the Q-values before softmax
        logits = beta * q_stage1_mb
        if last_action_1 != -1:
            logits[last_action_1] += stickiness
            
        exp_q1 = np.exp(logits) # Softmax on logits (beta*Q + stickiness)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Record this action for next trial
        last_action_1 = int(action_1[trial])
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        # Standard softmax for second stage
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updating ---
        # Update Stage 2 values based on reward
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # (No Stage 1 Q-value update needed as this is pure MB using Stage 2 values)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
