Here are three cognitive models exploring different mechanisms by which anxiety (STAI) might influence decision-making in the two-step task.

### Model 1: Anxiety Modulates Model-Based vs. Model-Free Balance
This model hypothesizes that high anxiety shifts reliance away from computationally expensive Model-Based (planning) strategies toward habitual Model-Free strategies. The mixing weight `w` is modulated by STAI.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where anxiety modulates the mixing weight.
    
    Hypothesis: Higher anxiety (STAI) reduces cognitive resources for planning,
    shifting control towards Model-Free (habitual) systems.
    Effective mixing weight w = w_base * (1 - w_stai * stai).
    
    Parameters:
    - learning_rate: [0, 1] Rate of updating Q-values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w_base: [0, 1] Base weight for Model-Based control (when anxiety is 0).
    - w_stai: [0, 1] Sensitivity of MB-weight to anxiety. 
                 Higher values mean anxiety reduces MB control more strongly.
    """
    learning_rate, beta, w_base, w_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective mixing weight. Capped at 0 to prevent negative weights.
    # If stai is high and w_stai is high, w_eff approaches 0 (pure Model-Free).
    w_eff = w_base * (1.0 - w_stai * stai_score)
    w_eff = max(0.0, min(1.0, w_eff))
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * max(Q2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w_eff * q_stage1_mb + (1 - w_eff) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Updates ---
        # SARSA-style update for Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Standard Q-learning update for Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety Modulates Learning Rate (Hyper-Learning)
This model posits that anxious individuals may have an altered sensitivity to prediction errors. Specifically, high anxiety might lead to "over-updating" or a higher learning rate, making behavior more volatile and reactive to recent outcomes.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Based/Model-Free Hybrid where anxiety modulates the Learning Rate.
    
    Hypothesis: Higher anxiety leads to higher learning rates (volatile updating).
    Effective learning rate = lr_base + (lr_stai * stai).
    
    Parameters:
    - lr_base: [0, 1] Base learning rate.
    - lr_stai: [0, 1] Strength of anxiety increasing the learning rate.
    - beta: [0, 10] Inverse temperature.
    - mixing_weight: [0, 1] Fixed balance between MB and MF.
    """
    lr_base, lr_stai, beta, mixing_weight = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Calculate effective learning rate, capped at 1.0
    lr_eff = lr_base + (lr_stai * stai_score)
    lr_eff = min(1.0, max(0.0, lr_eff))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = mixing_weight * q_stage1_mb + (1 - mixing_weight) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
  
        # --- Updates ---
        # Note: We use lr_eff here
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_eff * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_eff * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety Modulates Choice Stickiness (Perseveration)
This model tests the idea that anxiety leads to rigid or perseverative behavior. Instead of altering value learning or planning, anxiety adds a "stickiness" bonus to the previously chosen action, making the participant more likely to repeat their last Stage 1 choice regardless of reward.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model with Anxiety-Modulated Choice Stickiness (Perseveration).
    
    Hypothesis: Higher anxiety increases the tendency to repeat the previous 
    Stage 1 choice (perseveration), independent of reward history.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - mixing_weight: [0, 1] MB vs MF balance.
    - stickiness_stai: [0, 5] How much anxiety adds to the choice stickiness bonus.
    """
    learning_rate, beta, mixing_weight, stickiness_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    last_choice_1 = -1 

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = mixing_weight * q_stage1_mb + (1 - mixing_weight) * q_stage1_mf
        
        # Add anxiety-driven stickiness bonus to Q-values before softmax
        # The bonus is added to the action taken on the PREVIOUS trial
        q_net_sticky = q_net.copy()
        if last_choice_1 != -1:
            stickiness_bonus = stickiness_stai * stai_score
            q_net_sticky[last_choice_1] += stickiness_bonus

        exp_q1 = np.exp(beta * q_net_sticky)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        last_choice_1 = a1 # Update for next trial

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
