Here are three new cognitive models that incorporate the STAI anxiety score into the decision-making process using different cognitive mechanisms than the previous model.

### Model 1: Anxiety-Modulated Decision Noise (Beta)
This model hypothesizes that anxiety functions as "neural noise" or interferes with the ability to consistently select the optimal option. Instead of affecting the *strategy* (MB vs MF), anxiety affects the *consistency* of the choice (Inverse Temperature/Beta). Higher anxiety leads to a lower effective Beta (more random/noisy choices), regardless of the calculated values.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Decision Noise.
    
    Hypothesis: Higher anxiety (STAI) acts as a distractor, reducing the signal-to-noise ratio 
    in decision making. This is modeled by scaling down the inverse temperature (beta) 
    as anxiety increases.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_base: [0, 10] The maximum inverse temperature (consistency) for a 0-anxiety participant.
    - w: [0, 1] Mixing weight (0=MF, 1=MB). Assumed constant for this model.
    - anxiety_noise: [0, 5] Scaling factor. Higher values mean STAI reduces beta more aggressively.
    """
    learning_rate, beta_base, w, anxiety_noise = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Anxiety reduces the effective beta (increases noise/randomness)
    # Denominator formulation ensures beta remains positive and decays smoothly
    beta = beta_base / (1.0 + (anxiety_noise * stai_score))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update Stage 1 MF values (TD0)
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1

        # Update Stage 2 MF values
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Volatility (Learning Rate)
This model hypothesizes that high anxiety correlates with a perception of the world as volatile and unstable. Consequently, highly anxious participants will have a higher learning rate, updating their beliefs more rapidly in response to recent outcomes and discarding older information faster.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Driven Volatility (Learning Rate Modulation).
    
    Hypothesis: Anxious individuals perceive the environment as more volatile. 
    Therefore, higher STAI scores lead to a higher learning rate, causing 
    values to fluctuate more rapidly based on recent feedback.
    
    Parameters:
    - lr_base: [0, 1] Baseline learning rate for low anxiety.
    - lr_sensitivity: [0, 1] How much STAI increases the learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight between MB and MF.
    """
    lr_base, lr_sensitivity, beta, w = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Calculate effective learning rate based on anxiety
    # Bounded at 1.0
    learning_rate = lr_base + (lr_sensitivity * stai_score)
    learning_rate = np.clip(learning_rate, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

    return log_loss
```

### Model 3: Anxiety-Induced Perseveration (Safety Behavior)
This model suggests that anxiety drives "safety behaviors" or repetition. Anxious individuals may be more likely to repeat their previous Stage 1 choice (perseveration) regardless of the reward outcome, simply to avoid the uncertainty of switching. The STAI score scales a "stickiness" parameter added to the Q-values of the previously chosen action.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Induced Perseveration.
    
    Hypothesis: Anxiety increases 'stickiness' or perseveration. High STAI participants 
    are more likely to repeat the previous Stage 1 choice to minimize decision conflict, 
    independent of the value of that choice.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (MB vs MF).
    - perseveration_factor: [0, 5] Base magnitude of stickiness.
    - anxiety_pers_scale: [0, 1] How much STAI amplifies the perseveration factor.
    """
    learning_rate, beta, w, perseveration_factor, anxiety_pers_scale = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Calculate effective perseveration bonus
    # This bonus is added to the logit of the previously chosen action
    pers_bonus = perseveration_factor * (1.0 + (anxiety_pers_scale * stai_score))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize as -1 (no previous choice)

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Calculate logits (pre-softmax values)
        logits = beta * q_net
        
        # Add perseveration bonus if a previous choice exists
        if last_action_1 != -1:
            logits[last_action_1] += pers_bonus

        exp_q1 = np.exp(logits) # Apply exponential to modified logits
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        current_a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[current_a1]
        
        # Store for next trial
        last_action_1 = current_a1
        
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[current_a1]
        q_stage1_mf[current_a1] += learning_rate * delta_stage1

        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
