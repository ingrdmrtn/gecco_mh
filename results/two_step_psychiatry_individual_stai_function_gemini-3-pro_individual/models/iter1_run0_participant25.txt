Here are three new cognitive models that incorporate the participant's STAI (anxiety) score into the decision-making process.

### Model 1: Anxiety-Modulated Learning Rate
This model posits that anxiety levels influence the speed at which a participant updates their value estimates (learning rate). High anxiety might lead to "hyper-learning" or over-reaction to prediction errors.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model where STAI modulates the Learning Rate.
    
    Hypothesis: Higher anxiety leads to a higher learning rate (alpha), causing 
    more rapid updating of value expectations in response to rewards/omissions.
    
    Parameters:
    alpha_base: [0,1] Baseline learning rate.
    alpha_mod:  [0,1] Strength of STAI modulation on learning rate.
    beta:       [0,10] Inverse temperature for softmax.
    w:          [0,1] Weighting between Model-Based (1) and Model-Free (0).
    """
    alpha_base, alpha_mod, beta, w = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate effective learning rate based on anxiety
    # We allow anxiety to increase the learning rate from the baseline
    learning_rate = alpha_base + (alpha_mod * participant_stai)
    learning_rate = np.clip(learning_rate, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updating ---
        # Update Stage 1 (TD-like using Stage 2 value)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 (Reward prediction error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Exploration (Beta)
This model suggests that anxiety affects the exploration-exploitation trade-off. Specifically, it tests if higher anxiety leads to more rigid, deterministic choices (higher beta) to avoid the uncertainty of exploration.

### Model 3: Anxiety-Modulated Stickiness (Perseveration)
This model introduces a "stickiness" or perseveration parameter that is modulated by anxiety. It tests the hypothesis that anxious individuals are more likely to repetitively choose the same option (safety behavior) regardless of value updates.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model where STAI modulates Choice Stickiness (Perseveration).
    
    Hypothesis: Higher anxiety leads to higher 'stickiness', causing the participant
    to repeat the previous Stage 1 choice regardless of reward outcomes.
    
    Parameters:
    learning_rate: [0,1] Learning rate for value updates.
    beta:          [0,10] Inverse temperature.
    w:             [0,1] Weighting between Model-Based (1) and Model-Free (0).
    stick_mod:     [0,1] Strength of STAI-driven choice perseveration.
    """
    learning_rate, beta, w, stick_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate stickiness magnitude
    # Scaled by 5 to provide a meaningful bonus to the Q-value
    stickiness = stick_mod * participant_stai * 5.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness bonus to the previously chosen action
        if trial > 0:
            prev_action = int(action_1[trial - 1])
            q_net[prev_action] += stickiness

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updating ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
