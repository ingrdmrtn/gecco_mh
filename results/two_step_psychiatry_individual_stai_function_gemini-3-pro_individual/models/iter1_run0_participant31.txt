Here are three new cognitive models that incorporate the STAI anxiety score into the decision-making process using different mechanisms than the previous feedback.

### Model 1: Anxiety-Impaired Model-Based Control
This model hypothesizes that high anxiety consumes cognitive resources (working memory), thereby reducing the participant's ability to engage in Model-Based (planning) control. Instead of modulating randomness (beta), here STAI modulates the weighting parameter $w$, shifting the balance toward Model-Free (habitual) control as anxiety increases.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Impaired Model-Based Control.
    
    Hypothesis: High anxiety reduces cognitive resources required for planning 
    (Model-Based control). Therefore, higher STAI scores reduce the weight (w) 
    assigned to the Model-Based system, leading to more Model-Free behavior.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature for softmax.
    - w_max: [0, 1] The maximum possible Model-Based weight (at 0 anxiety).
    - anxiety_decay: [0, 1] How strongly STAI reduces w. 
      w_effective = w_max * (1 - anxiety_decay * stai)
    """
    learning_rate, beta, w_max, anxiety_decay = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    # Calculate effective weight based on anxiety
    # If anxiety_decay is high, high STAI pushes w towards 0 (pure Model-Free)
    w_effective = w_max * (1.0 - (anxiety_decay * stai_val))
    w_effective = np.clip(w_effective, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based valuation (planning using transition matrix)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid valuation: w is modulated by STAI
        q_net = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # TD Error Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Error Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Hyper-Reactivity (Learning Rate Modulation)
This model hypothesizes that anxiety is a state of high arousal and volatility. Anxious participants may over-react to prediction errors, treating the environment as highly volatile. Here, STAI scales the `learning_rate`, causing the participant to update values more aggressively and forget past history faster.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Driven Hyper-Reactivity.
    
    Hypothesis: Anxiety correlates with hyper-arousal and a belief in environmental 
    volatility. Higher STAI scores increase the learning rate, making the 
    participant react strongly to the most recent outcomes (ignoring long-term history).
    
    Parameters:
    - lr_base: [0, 1] Baseline learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight between MB and MF.
    - anxiety_sens: [0, 1] How much STAI boosts the learning rate.
      lr_effective = lr_base + (anxiety_sens * stai)
    """
    lr_base, beta, w, anxiety_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    # Calculate effective learning rate
    # High anxiety increases LR, potentially making behavior erratic/reactive
    lr_effective = lr_base + (anxiety_sens * stai_val)
    lr_effective = np.clip(lr_effective, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates using Anxiety-Modulated Learning Rate ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_effective * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_effective * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Perseveration (Stickiness)
This model posits that anxiety leads to "safety behaviors" or rigid repetition of choices (perseveration) to avoid the uncertainty of switching. This is modeled by adding a "stickiness" bonus to the previously chosen action, where the magnitude of this bonus is directly scaled by the STAI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Induced Perseveration.
    
    Hypothesis: Anxious individuals exhibit "safety behaviors" manifesting as 
    repetitive choices (perseveration/stickiness) regardless of reward history.
    The STAI score scales a 'stickiness' bonus added to the Q-value of the 
    previously chosen action in Stage 1.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight between MB and MF.
    - stickiness_factor: [0, 5] Base magnitude of perseveration.
      Bonus = stickiness_factor * stai * (1 if action == prev_action else 0)
    """
    learning_rate, beta, w, stickiness_factor = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous action for stickiness
    prev_action = -1 

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Anxiety-Driven Stickiness Bonus
        if prev_action != -1:
            # We add a bonus to the Q-value of the action taken on the previous trial
            # The magnitude is determined by the stickiness param * STAI
            stickiness_bonus = np.zeros(2)
            stickiness_bonus[prev_action] = stickiness_factor * stai_val
            q_net = q_net + stickiness_bonus

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        prev_action = a1 # Update previous action

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
