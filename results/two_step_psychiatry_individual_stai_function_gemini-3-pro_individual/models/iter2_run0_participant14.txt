Here are 3 new cognitive models that incorporate the STAI anxiety score into the decision-making process in different ways.

### Model 1: Anxiety-Modulated Plasticity (Learning Rate)
This model hypotheses that anxiety affects **how quickly** a participant updates their beliefs (learning rate), rather than how they trade off between systems. High anxiety might lead to "over-updating" or volatility in value estimation (high learning rate), or conversely, "freezing" (low learning rate).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid model where STAI modulates the Learning Rate.
    
    Hypothesis: Anxiety affects neural plasticity and error-correction rates.
    High anxiety leads to higher learning rates (instability/over-reaction to recent outcomes).
    
    Parameters:
    - lr_base: [0, 1] Baseline learning rate.
    - lr_stai_sens: [0, 1] Sensitivity of learning rate to anxiety. 
                    Effective LR increases with STAI.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    """
    lr_base, lr_stai_sens, beta, w = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Modulate learning rate: Higher anxiety -> higher learning rate (more volatile)
    # Bounded to max 1.0
    learning_rate = lr_base + (lr_stai_sens * stai_score * (1.0 - lr_base))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value integration
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        
        # Action value updating for choice 1
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        
        # Action value updating for choice 2
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Decision Noise (Beta Modulation)
This model hypotheses that anxiety acts as a source of **cognitive noise**. According to theories like the Yerkes-Dodson law or attentional control theory, high anxiety might reduce the signal-to-noise ratio in decision-making, effectively lowering the inverse temperature ($\beta$) and making choices more random regardless of the learned values.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid model where STAI modulates the Inverse Temperature (Beta).
    
    Hypothesis: High anxiety increases decision noise (randomness), 
    effectively lowering the beta parameter.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_base: [0, 10] Baseline inverse temperature (for low anxiety).
    - beta_stai_damp: [0, 1] Dampening factor. Higher values mean anxiety 
                      reduces beta more aggressively.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    """
    learning_rate, beta_base, beta_stai_damp, w = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Modulate Beta: Higher anxiety -> Lower Beta (More random/noisy)
    beta = beta_base * (1.0 - (beta_stai_damp * stai_score))
    # Ensure beta doesn't go below 0
    beta = max(0.0, beta)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        
        # Action value updating for choice 1
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        
        # Action value updating for choice 2
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Distorted World Model (Subjective Transitions)
This model hypotheses that anxiety distorts the **internal model of the world**. While the standard Model-Based system uses the true transition probabilities (0.7/0.3), an anxious individual might perceive the environment as more unpredictable (higher entropy). They may underestimate the reliability of the "common" transition, effectively flattening the transition matrix used for planning.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid model where STAI distorts the Transition Matrix used for MB planning.
    
    Hypothesis: Anxious individuals perceive the world as more volatile or 
    unpredictable. They flatten the transition probabilities (closer to 0.5/0.5),
    making MB planning less effective even if the mixing weight 'w' is high.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - distortion: [0, 1] How much anxiety flattens the transition belief.
                  0 = Accurate model (0.7), 1 = Max distortion (closer to 0.5).
    """
    learning_rate, beta, w, distortion = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    # Calculate subjective transition probability
    # True common prob is 0.7. Anxious people might see it as lower (less reliable).
    # If distortion * stai is high, subjective_common approaches 0.5.
    true_common = 0.7
    subjective_common = true_common - (0.2 * distortion * stai_score) 
    subjective_rare = 1.0 - subjective_common
    
    # The agent uses this distorted matrix for planning
    subjective_matrix = np.array([
        [subjective_common, subjective_rare], 
        [subjective_rare, subjective_common]
    ])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # CRITICAL CHANGE: Use subjective_matrix instead of objective one
        q_stage1_mb = subjective_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        
        # Action value updating for choice 1
        q_stage1_mf[int(action_1[trial])] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        
        # Action value updating for choice 2
        q_stage2_mf[state_idx, int(action_2[trial])] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
