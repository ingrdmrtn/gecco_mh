Here are three new cognitive models that incorporate STAI scores into the decision-making process using different mechanisms than the previous best model.

### Model 1: Anxiety-Modulated Learning Rate (Hyper-vigilance)
This model hypothesizes that anxiety is a state of hyper-arousal or hyper-vigilance. Instead of affecting the balance between strategies (MB vs. MF), anxiety here affects the **speed of learning**. Participants with higher anxiety are modeled as having a higher learning rate ($\alpha$), updating their beliefs more drastically in response to every outcome, potentially making their behavior more volatile.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model with Anxiety-Modulated Learning Rate.
    
    Hypothesis: Higher anxiety leads to hyper-vigilance, resulting in a higher 
    learning rate. Anxious individuals update their value estimates more 
    rapidly in response to prediction errors.
    
    Formula:
    lr_effective = lr_base + (stai_lr_sensitivity * normalized_stai)
    
    Parameters:
    lr_base: [0, 1] Baseline learning rate for low anxiety.
    beta: [0, 10] Inverse temperature (exploration/exploitation).
    w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    stai_lr_sensitivity: [0, 1] How much STAI increases the learning rate.
    """
    lr_base, beta, w, stai_lr_sensitivity = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    # Normalize STAI (approx range 20-80) to 0-1
    norm_stai = (stai_val - 20.0) / 60.0
    norm_stai = np.clip(norm_stai, 0.0, 1.0)

    # Calculate effective learning rate
    learning_rate = lr_base + (stai_lr_sensitivity * norm_stai)
    learning_rate = np.clip(learning_rate, 0.0, 1.0) # Ensure valid bound

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted mixture of MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 MF values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updating ---
        # SARSA-style update for Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Standard Q-learning update for Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Inverse Temperature (Risk Aversion)
This model hypothesizes that anxiety drives **risk aversion** or a desire to reduce uncertainty. In a softmax decision rule, a higher inverse temperature ($\beta$) makes choices more deterministic (always picking the option with the slightly higher value) and reduces exploration. This model posits that higher STAI scores lead to a higher $\beta$, representing a "stiffening" of behavior or intolerance for noisy exploration.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model with Anxiety-Modulated Inverse Temperature (Beta).
    
    Hypothesis: Higher anxiety leads to reduced exploration and higher 
    exploitation (risk aversion). Anxious individuals are less likely to 
    explore options with lower estimated values.
    
    Formula:
    beta_effective = beta_base + (stai_beta_slope * normalized_stai)
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta_base: [0, 10] Baseline inverse temperature for low anxiety.
    w: [0, 1] Mixing weight (0=Pure MF, 1=Pure MB).
    stai_beta_slope: [0, 5] Additional beta added due to anxiety.
    """
    learning_rate, beta_base, w, stai_beta_slope = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Normalize STAI
    norm_stai = (stai_val - 20.0) / 60.0
    norm_stai = np.clip(norm_stai, 0.0, 1.0)

    # Calculate effective beta
    beta = beta_base + (stai_beta_slope * norm_stai)
    # No upper clip needed strictly, but good practice to keep reasonable
    beta = np.clip(beta, 0.0, 20.0) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updating ---
        a1 = int(action_1[trial])
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Modulated Punishment Sensitivity
This model explores the link between anxiety and **negative valence bias**. In this task, receiving 0 coins is the worst outcome. This model hypothesizes that for anxious individuals, a "0 coin" outcome is not just a lack of reward, but is perceived as an active punishment. The STAI score modulates the subjective value of the 0-reward outcome, making it negative rather than neutral.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model with Anxiety-Modulated Punishment Sensitivity.
    
    Hypothesis: Anxious individuals exhibit a negative bias. They perceive 
    reward omission (0 coins) as a punishment (negative value) rather than 
    neutral. 
    
    Formula:
    If reward == 0: effective_reward = -1 * (stai_punish_sens * norm_stai)
    Else: effective_reward = reward
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    stai_punish_sens: [0, 1] Magnitude of negative valence for 0-reward outcomes.
    """
    learning_rate, beta, w, stai_punish_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    norm_stai = (stai_val - 20.0) / 60.0
    norm_stai = np.clip(norm_stai, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updating ---
        a1 = int(action_1[trial])
        r = reward[trial]
        
        # Modulate reward perception based on anxiety
        if r == 0:
            effective_reward = -1.0 * (stai_punish_sens * norm_stai)
        else:
            effective_reward = r
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update using effective_reward instead of raw reward
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
