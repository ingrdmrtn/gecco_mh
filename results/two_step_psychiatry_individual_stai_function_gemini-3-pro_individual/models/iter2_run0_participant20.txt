Here are three new cognitive models that incorporate the STAI score into the decision-making process using different psychological mechanisms: Learning Rate Modulation (volatility), Choice Stickiness (perseveration), and Reward Valuation (pessimism).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Learning Rate Model.
    
    Hypothesis: Anxiety affects the volatility of belief updating. High anxiety (STAI)
    increases the learning rate, causing the participant to over-react to the most 
    recent outcomes (hyper-learning) rather than integrating over a longer history.
    
    The effective learning rate is: alpha = base_lr + (stai * lr_sensitivity)
    
    Bounds:
    base_lr: [0, 1]
    beta: [0, 10]
    w: [0, 1] (Mixing weight: 1=Model-Based, 0=Model-Free)
    lr_sensitivity: [0, 1] (How much STAI increases the learning rate)
    """
    base_lr, beta, w, lr_sensitivity = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    # Calculate effective learning rate based on anxiety, clipped to [0, 1]
    learning_rate = base_lr + (stai_val * lr_sensitivity)
    if learning_rate > 1.0: learning_rate = 1.0
    if learning_rate < 0.0: learning_rate = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Initialize Q-values at 0.5 (neutral expectation between 0 and 1)
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 RPE
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 Update: Includes both Stage 1 RPE (TD0) and Stage 2 RPE (TD1/Eligibility Trace)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1 
        q_stage1_mf[a1] += learning_rate * delta_stage2 

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Choice Stickiness (Perseveration) Model.
    
    Hypothesis: High anxiety increases the tendency to repeat the previous choice 
    (perseveration) to reduce decision-making load or avoid regret.
    A 'stickiness' bonus is added to the Q-value of the previously chosen spaceship.
    
    Stickiness magnitude = base_stickiness * stai
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    base_stickiness: [0, 5] (Magnitude of the bonus for repetition)
    """
    learning_rate, beta, w, base_stickiness = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    prev_a1 = -1 # No previous choice on first trial

    stickiness_bonus = base_stickiness * stai_val

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the previously chosen action
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness_bonus

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        prev_a1 = a1 # Update previous action for next trial

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Reward Valuation (Pessimism) Model.
    
    Hypothesis: Anxious individuals may undervalue positive rewards (anhedonia or pessimism).
    Instead of perceiving the gold coin as a value of 1.0, they perceive it as 
    less valuable as anxiety increases.
    
    Effective Reward = Actual Reward * (1 - (pessimism_factor * stai))
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    pessimism_factor: [0, 1] (Reduces the subjective value of the reward)
    """
    learning_rate, beta, w, pessimism_factor = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        r_raw = reward[trial]
        
        # Apply pessimism: Subjective reward is dampened by anxiety
        # If r_raw is 0, it stays 0. If 1, it becomes 1 - (pessimism * stai)
        r_effective = r_raw * (1.0 - (pessimism_factor * stai_val))
        
        delta_stage2 = r_effective - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        q_stage1_mf[a1] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
