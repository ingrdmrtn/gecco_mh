### Model 1: Anxiety-Driven Learning Volatility
This model hypothesizes that anxiety affects the **learning rate** (plasticity). High anxiety is modeled as a state of high physiological arousal, leading to faster, more volatile updates in response to prediction errors (over-reacting to the most recent outcome).

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Learning Volatility Model.
    
    Hypothesis: Anxiety (STAI) increases the learning rate, making the participant 
    more reactive to recent outcomes (higher volatility) and less capable of 
    stable integration of value over time.
    
    Mechanism:
    effective_lr = lr_base + (stai_sensitivity * stai_score)
    
    Bounds:
    lr_base: [0, 1] (Baseline learning rate)
    beta: [0, 10] (Inverse temperature)
    w: [0, 1] (MB/MF mixing weight)
    stai_sensitivity: [0, 1] (How strongly anxiety increases LR)
    """
    lr_base, beta, w, stai_sensitivity = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Modulate learning rate by STAI. 
    # High anxiety -> Higher LR -> More volatile updating.
    learning_rate = lr_base + (stai_sensitivity * stai_score)
    learning_rate = np.clip(learning_rate, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # Model-Based Value
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_current_state = q_stage2_mf[state_idx, :]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Updates ---
        # Update Stage 1 MF value using the modulated learning_rate
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF value using the modulated learning_rate
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Decision Noise

### Model 3: Anxiety-Dependent Perseveration
This model hypothesizes that anxiety increases "stickiness" or perseveration. Anxious individuals may fall back on repetitive safety behaviors or habits to avoid decision conflict. This is modeled as a value bonus added to the previously chosen action, with the magnitude of the bonus scaled by the STAI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Dependent Perseveration Model.
    
    Hypothesis: Anxiety increases the tendency to repeat the previous choice 
    (perseveration) regardless of reward history. A 'stickiness' bonus is added 
    to the Q-value of the previously chosen spaceship.
    
    Mechanism:
    Q_net[prev_action] += (stickiness_factor * stai_score)
    
    Bounds:
    lr: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    stickiness_factor: [0, 5] (Magnitude of the repetition bias per unit of STAI)
    """
    lr, beta, w, stickiness_factor = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add anxiety-modulated stickiness bonus
        if last_action_1 != -1:
            # We copy q_net to avoid modifying the underlying value permanently
            q_net_biased = q_net.copy()
            q_net_biased[last_action_1] += (stickiness_factor * stai_score)
        else:
            q_net_biased = q_net

        exp_q1 = np.exp(beta * q_net_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        
        # Update tracker for next trial
        last_action_1 = act1 
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        qs_current_state = q_stage2_mf[state_idx, :]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
