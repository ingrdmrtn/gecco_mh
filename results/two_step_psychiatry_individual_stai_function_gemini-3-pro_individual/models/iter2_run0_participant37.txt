Here are 3 new cognitive models based on the participant's STAI score.

### Model 1: Anxiety-Driven Hyper-vigilance (Global Learning Rate Boost)
This model hypothesizes that high anxiety leads to a state of "hyper-vigilance," where the participant perceives the environment as highly volatile. Consequently, they update their beliefs more rapidly in response to *all* prediction errors (both positive and negative), resulting in a globally higher learning rate.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hyper-vigilance Model.
    
    Hypothesis: Anxiety increases the global learning rate. 
    High STAI participants perceive the world as volatile and update value estimates 
    rapidly for all outcomes, not just negative ones.
    
    Parameters:
    lr_base: [0, 1] Baseline learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (1=MB, 0=MF).
    stai_lr_scale: [0, 1] Scaling factor for STAI to boost learning rate.
    """
    lr_base, beta, w, stai_lr_scale = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Mechanism: Effective LR increases with STAI
    # We clip to ensure it stays within valid bounds [0, 1]
    lr_effective = np.clip(lr_base * (1.0 + (stai_score * stai_lr_scale)), 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Update Stage 1 MF values using the effective (boosted) learning rate
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_effective * delta_stage1
        
        # Update Stage 2 MF values using the effective (boosted) learning rate
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_effective * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Rigidity (Beta Modulation)
This model hypothesizes that anxiety manifests as behavioral rigidity or a strong desire to exploit currently perceived best options (safety behaviors) rather than exploring. High STAI scores increase the inverse temperature parameter (`beta`), making the softmax choice function sharper and more deterministic.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxious Rigidity Model.
    
    Hypothesis: Anxiety leads to risk-averse, rigid decision making.
    High STAI increases the inverse temperature (beta), causing the participant
    to choose the option with the highest value more deterministically 
    (reduced exploration).
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta_base: [0, 10] Baseline inverse temperature.
    w: [0, 1] Mixing weight.
    stai_rigidity: [0, 5] Factor by which STAI increases beta.
    """
    lr, beta_base, w, stai_rigidity = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Mechanism: Beta increases with Anxiety (Choice becomes less noisy/more greedy)
    beta_effective = beta_base + (stai_score * stai_rigidity)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use beta_effective
        exp_q1 = np.exp(beta_effective * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        # Use beta_effective

        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---

        q_stage1_mf[a1] += lr * delta_stage1

    return log_loss
```

### Model 3: Anxiety-Driven Perseveration (Choice Stickiness)
This model hypothesizes that anxiety leads to repetitive behaviors (perseveration) regardless of reward outcomes. It adds a "stickiness" bonus to the choice logits of the previously chosen spaceship. The magnitude of this stickiness is modulated by the STAI score.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxious Perseveration (Stickiness) Model.
    
    Hypothesis: Anxiety increases the tendency to repeat the previous Stage 1 choice
    (perseveration), regardless of whether it was rewarded. This is modeled as a 
    'stickiness' bonus added to the Q-value of the previously chosen action.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    stai_stickiness: [0, 5] Strength of choice repetition induced by STAI.
    """
    lr, beta, w, stai_stickiness = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Mechanism: Calculate stickiness bonus based on STAI
    stickiness_bonus = stai_score * stai_stickiness

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize variable to track previous choice

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits (unnormalized log probabilities)
        logits = beta * q_net
        
        # Add stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            logits[int(last_action_1)] += stickiness_bonus
            
        # Softmax on modified logits
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Update tracker
        last_action_1 = action_1[trial]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
