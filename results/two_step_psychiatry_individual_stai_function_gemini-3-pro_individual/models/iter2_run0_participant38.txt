Here are 3 new cognitive models that incorporate the STAI anxiety score into the decision-making process using different mechanisms than previous iterations.

### Model 1: Anxiety-Modulated Eligibility Trace (Lambda)
This model hypothesizes that anxiety affects how much credit is assigned to the first-stage choice based on the second-stage outcome. High anxiety might lead to "rumination" or over-connection of distal outcomes to initial choices (higher lambda), or conversely, cognitive load might reduce the trace.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF with Anxiety-Modulated Eligibility Trace (Lambda).
    
    The eligibility trace (lambda) determines how much the Stage 2 reward prediction error 
    updates the Stage 1 choice value. This model posits that STAI modulates this trace.
    
    Parameters:
    - learning_rate: [0, 1] Alpha for Q-learning.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - lambda_base: [0, 1] Baseline eligibility trace.
    - lambda_mod: [-1, 1] Modulation of lambda by STAI.
    """
    learning_rate, beta, w, lambda_base, lambda_mod = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0] 

    # Calculate lambda based on STAI
    lambda_param = lambda_base + (lambda_mod * stai_val)
    lambda_param = np.clip(lambda_param, 0.0, 1.0)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q-values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]
  
        # Prediction Errors
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        
        # ACTION VALUE UPDATING FOR CHOICE 1 (MF)
        # Standard TD(1) / Eligibility trace update
        # Update 1: based on transition to stage 2
        q_stage1_mf[act1] += learning_rate * delta_stage1
        # Update 2: based on stage 2 outcome (mediated by lambda)
        q_stage1_mf[act1] += learning_rate * lambda_param * delta_stage2

        # ACTION VALUE UPDATING FOR CHOICE 2
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Choice Stickiness
This model investigates if anxiety leads to behavioral rigidity or "stickiness" (perseveration). Instead of modulating learning or mixing weights, STAI introduces a bias to repeat the previous Stage 1 choice, regardless of reward history.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF with Anxiety-Modulated Choice Stickiness.
    
    Adds a 'stickiness' bonus to the previously chosen action at Stage 1.
    The magnitude of this stickiness is determined by the STAI score.
    
    Parameters:
    - learning_rate: [0, 1] Alpha.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - stick_base: [0, 5] Baseline stickiness bonus.
    - stick_mod: [-5, 5] Modulation of stickiness by STAI.
    """
    learning_rate, beta, w, stick_base, stick_mod = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    # Calculate stickiness bonus
    stickiness = stick_base + (stick_mod * stai_val)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness to the Q-value of the previous action (if it exists)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        state_idx = int(state[trial])
        
        # Update tracker for next trial
        last_action_1 = act1

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]
  
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        
        # ACTION VALUE UPDATING FOR CHOICE 1
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        
        # ACTION VALUE UPDATING FOR CHOICE 2
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Asymmetric Learning from Negative Prediction Errors
This model proposes that anxiety creates a bias in how prediction errors are processed. Specifically, it tests if higher STAI scores amplify the learning rate specifically when the outcome is worse than expected (negative prediction error), reflecting a "negativity bias" common in anxiety.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF with Anxiety-Driven Asymmetric Learning (Negativity Bias).
    
    The learning rate is split into positive and negative domains.
    STAI modulates the 'negative bias', amplifying learning from disappointment
    (negative prediction errors).
    
    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF mixing weight.
    - neg_bias: [0, 5] Multiplier for learning rate when PE < 0, scaled by STAI.
    """
    learning_rate, beta, w, neg_bias = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[act1]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[act2]
  
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        
        # ACTION VALUE UPDATING FOR CHOICE 1
        # Determine effective learning rate based on sign of error
        if delta_stage1 < 0:
            # Amplify learning from negative surprise based on anxiety
            eff_lr_1 = learning_rate * (1.0 + (neg_bias * stai_val))
        else:
            eff_lr_1 = learning_rate
        
        eff_lr_1 = np.clip(eff_lr_1, 0.0, 1.0)
        q_stage1_mf[act1] += eff_lr_1 * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, act2]
        
        # ACTION VALUE UPDATING FOR CHOICE 2
        # Apply same asymmetry logic to second stage
        if delta_stage2 < 0:
            eff_lr_2 = learning_rate * (1.0 + (neg_bias * stai_val))
        else:
            eff_lr_2 = learning_rate
            
        eff_lr_2 = np.clip(eff_lr_2, 0.0, 1.0)
        q_stage2_mf[state_idx, act2] += eff_lr_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
