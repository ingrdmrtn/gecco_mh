Here are three new cognitive models that incorporate the STAI anxiety score into the decision-making process using mechanisms distinct from the previous mixing-weight modulation.

### Model 1: Anxiety-Modulated Decision Noise (Inverse Temperature)
This model hypothesizes that anxiety acts as a "distractor," consuming cognitive resources and effectively increasing decision noise. Instead of modulating *how* values are learned (Model-Based vs. Model-Free), anxiety modulates the precision of the choice (softmax temperature). High anxiety leads to a lower $\beta$ (higher temperature), resulting in more random exploration and less exploitation of learned values.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model where anxiety increases decision noise (lowers beta).
    
    Hypothesis: High anxiety acts as cognitive load, reducing the precision 
    of action selection. 
    Effective Beta = beta_base / (1 + beta_stai_sens * stai).
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_base: [0, 10] Baseline inverse temperature (precision) at 0 anxiety.
    - w: [0, 1] Fixed mixing weight (0=MF, 1=MB).
    - beta_stai_sens: [0, 10] Sensitivity of noise to anxiety. 
                      Higher values mean anxiety causes more randomness.
    """
    learning_rate, beta_base, w, beta_stai_sens = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective beta based on anxiety
    # Using division ensures beta decreases as anxiety increases (more noise)
    beta_eff = beta_base / (1.0 + beta_stai_sens * stai_score)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Fixed mixing weight w, but beta is modulated by STAI
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Update values
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
  
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Choice Perseveration
This model hypothesizes that anxiety leads to rigid, repetitive behavior (often called "safety behaviors" in clinical literature). Rather than affecting learning or randomness, anxiety adds a "stickiness" bonus to the previously chosen spaceship. The higher the anxiety, the more likely the participant is to repeat their Stage 1 choice, regardless of the reward outcome.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model with Anxiety-Modulated Choice Perseveration.
    
    Hypothesis: Anxiety increases 'stickiness' or rigidity. High anxiety 
    participants receive a bonus added to the Q-value of the previously 
    chosen action, making them more likely to repeat it.
    
    Parameters:
    - learning_rate: [0, 1] Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - stick_stai: [0, 5] Perseveration bonus scaling factor. 
                  Bonus = stick_stai * STAI.
    """
    learning_rate, beta, w, stick_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate the stickiness bonus derived from anxiety
    perseveration_bonus = stick_stai * stai_score
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus if there was a previous trial
        if trial > 0:
            prev_choice = int(action_1[trial - 1])
            q_net[prev_choice] += perseveration_bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # Update values
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
  
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Modulated Asymmetric Learning (Negative Bias)
This model hypothesizes that anxiety creates a "negativity bias" or hypersensitivity to failure. The learning rate is split: there is a base learning rate for positive outcomes, but the learning rate for "disappointment" (negative prediction errors) is amplified by the STAI score. This reflects the tendency of anxious individuals to over-update their beliefs when things go wrong.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model with Anxiety-Modulated Asymmetric Learning Rates.
    
    Hypothesis: Anxiety creates a 'negativity bias'. The learning rate 
    is boosted specifically when the prediction error is negative 
    (outcome was worse than expected).
    LR_pos = lr_base
    LR_neg = lr_base * (1 + neg_bias_stai * STAI)
    
    Parameters:
    - lr_base: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - neg_bias_stai: [0, 10] Multiplier for the learning rate on negative PEs.
    """
    lr_base, beta, w, neg_bias_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Determine LR based on sign of delta (Asymmetric updating)
        if delta_stage1 < 0:
             lr_eff_1 = lr_base * (1.0 + neg_bias_stai * stai_score)
             lr_eff_1 = min(1.0, lr_eff_1) # Cap at 1.0
        else:
             lr_eff_1 = lr_base
             
        q_stage1_mf[a1] += lr_eff_1 * delta_stage1
        
        # Update Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        if delta_stage2 < 0:
             lr_eff_2 = lr_base * (1.0 + neg_bias_stai * stai_score)
             lr_eff_2 = min(1.0, lr_eff_2)
        else:
             lr_eff_2 = lr_base
             
        q_stage2_mf[state_idx, a2] += lr_eff_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
