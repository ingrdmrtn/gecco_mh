Here are three new cognitive models that explore different mechanisms for how anxiety (STAI) might influence decision-making, specifically focusing on Reward Sensitivity, Exploration (Temperature), and Memory Decay.

### Model 1: Anxiety-Blunted Reward Sensitivity
This model hypothesizes that anxiety reduces the subjective valuation of positive outcomes (anhedonia-like effect). Instead of affecting the learning *rate* (how fast they update), it affects the *magnitude* of the reward signal itself.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Blunted Reward Sensitivity Model.
    
    Hypothesis: High anxiety dampens the subjective value of rewards (R).
    This leads to smaller prediction errors even when learning rates are normal.
    
    Parameters:
    learning_rate: [0, 1] - Standard Q-learning rate.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting between Model-Based (1) and Model-Free (0).
    reward_sens_base: [0, 1] - Baseline sensitivity to reward.
    stai_dampening: [0, 1] - How much STAI reduces reward sensitivity.
    """
    learning_rate, beta, w, reward_sens_base, stai_dampening = model_parameters
    n_trials = len(action_1)
    stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective reward sensitivity based on anxiety
    # Higher anxiety -> lower effective reward value
    r_sensitivity = np.clip(reward_sens_base - (stai * stai_dampening), 0.0, 1.0)

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Scale the observed reward by the anxiety-modulated sensitivity
        effective_reward = reward[trial] * r_sensitivity
        
        # Stage 1 Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Decision Noise
This model hypothesizes that anxiety interferes with the ability to consistently select the highest-value option. It posits that anxiety acts as a "noise generator," effectively lowering the inverse temperature ($\beta$) and making choices more random (higher entropy) regardless of value differences.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Decision Noise Model.
    
    Hypothesis: Anxiety increases decision noise (lowers beta). 
    Higher STAI leads to more random exploration/errors, reducing the 
    influence of learned Q-values on choice.
    
    Parameters:
    learning_rate: [0, 1] - Q-learning rate.
    beta_base: [0, 10] - Baseline inverse temperature (low anxiety).
    w: [0, 1] - MB/MF weight.
    stai_noise_scale: [0, 1] - Factor by which STAI reduces beta.
    """
    learning_rate, beta_base, w, stai_noise_scale = model_parameters
    n_trials = len(action_1)
    stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective beta
    # As STAI increases, beta decreases (more noise/randomness)
    # We use a divisor to ensure beta doesn't go negative
    beta_effective = beta_base / (1.0 + (stai * stai_noise_scale * 5.0))

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_effective * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta_effective * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Accelerated Forgetting
This model introduces a memory decay mechanism. It hypothesizes that anxiety consumes cognitive resources (working memory load), causing participants to "forget" the values of unchosen options faster than low-anxiety participants.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Accelerated Forgetting Model.
    
    Hypothesis: Anxiety accelerates the decay of Q-values for unchosen options.
    While chosen options are updated via prediction error, unchosen options 
    decay back to 0. High anxiety increases this decay rate.
    
    Parameters:
    learning_rate: [0, 1] - Update rate for chosen options.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weight.
    decay_base: [0, 1] - Baseline forgetting rate.
    stai_decay_boost: [0, 1] - How much STAI increases forgetting.
    """
    learning_rate, beta, w, decay_base, stai_decay_boost = model_parameters
    n_trials = len(action_1)
    stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate decay rate modulated by anxiety
    decay_rate = np.clip(decay_base + (stai * stai_decay_boost), 0.0, 1.0)

    for trial in range(n_trials):

        # --- Policy for Choice 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net_stage1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # --- Policy for Choice 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # --- Forgetting Mechanism ---
        # Decay the unchosen option in Stage 1
        q_stage1_mf[1 - a1] *= (1 - decay_rate)
        
        # Decay the unchosen option in Stage 2 (for the current state)
        q_stage2_mf[state_idx, 1 - a2] *= (1 - decay_rate)
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
