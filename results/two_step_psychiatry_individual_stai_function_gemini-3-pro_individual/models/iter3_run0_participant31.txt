Here are three new cognitive models exploring different mechanisms by which anxiety (STAI) might influence decision-making in the two-step task.

### Model 1: Anxiety-Driven Learning Rate Modulation
This model hypothesizes that anxiety affects how quickly participants update their value estimates. High anxiety might lead to "over-learning" from negative outcomes (hyper-plasticity) or general volatility in belief updating. Here, we model anxiety as modulating the learning rate directly.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Learning Rate.
    
    Hypothesis: Anxiety influences the speed of learning (learning rate). 
    Higher anxiety might lead to faster updating (higher learning rate), 
    making the subject more sensitive to recent outcomes and less stable.
    
    Parameters:
    - lr_base: [0, 1] Baseline learning rate at zero anxiety.
    - lr_anxiety_mod: [0, 1] How strongly STAI increases the learning rate.
      effective_lr = lr_base + (lr_anxiety_mod * stai) (clipped at 1.0).
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting parameter between Model-Based (1) and Model-Free (0).
    """
    lr_base, lr_anxiety_mod, beta, w = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Calculate effective learning rate based on anxiety
    learning_rate = lr_base + (lr_anxiety_mod * stai_val)
    learning_rate = np.clip(learning_rate, 0.0, 1.0)
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted combination of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update values
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        # TD(0) updates using the anxiety-modulated learning rate
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Exploratory Noise (Inverse Temperature)
This model tests the idea that anxiety increases decision noise (or conversely, rigidity). Instead of affecting *how* they learn, anxiety affects *how* they choose. Specifically, we model anxiety as reducing the inverse temperature parameter ($\beta$), leading to more random (exploratory) choices, reflecting difficulty in committing to the optimal option under stress.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Induced Decision Noise.
    
    Hypothesis: High anxiety increases decision noise (randomness), effectively 
    lowering the inverse temperature (beta). The subject becomes less deterministic 
    in exploiting known high-value options.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_max: [0, 10] Maximum inverse temperature (at 0 anxiety).
    - beta_anxiety_decay: [0, 10] How much STAI reduces beta.
      beta_effective = beta_max - (beta_anxiety_decay * stai) (clipped at 0).
    - w: [0, 1] Weighting parameter between Model-Based (1) and Model-Free (0).
    """
    learning_rate, beta_max, beta_anxiety_decay, w = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    # Calculate effective beta based on anxiety
    beta = beta_max - (beta_anxiety_decay * stai_val)
    beta = np.maximum(beta, 0.0) # Ensure beta doesn't go negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update values
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Specific Perseveration
This model posits that anxiety leads to "stickiness" or perseverationâ€”a tendency to repeat the previous choice regardless of the outcome, perhaps as a safety behavior or avoidance of decision effort. We introduce a perseveration parameter modulated by the STAI score.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Modulated Perseveration.
    
    Hypothesis: Anxiety increases 'stickiness' or perseveration. Anxious individuals 
    may be more likely to repeat their previous choice (Choice 1) regardless of reward, 
    representing a safety behavior or cognitive rigidity.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF weighting.
    - pers_anxiety: [0, 5] The magnitude of perseveration bonus added to the 
      previously chosen action, scaled by STAI.
      Bonus = pers_anxiety * stai.
    """
    learning_rate, beta, w, pers_anxiety = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for perseveration
    last_action_1 = -1

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration bonus
        if last_action_1 != -1:
            q_net[last_action_1] += (pers_anxiety * stai_val)

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update values
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        last_action_1 = a1 # Store for next trial
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
