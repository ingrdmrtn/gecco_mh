Here are three new cognitive models exploring different mechanisms by which anxiety (STAI) might influence decision-making in this task.

### Model 1: Anxiety-Modulated Learning Rate
This model hypothesizes that anxiety affects learning speed (plasticity). High anxiety might lead to "over-learning" from negative prediction errors (or general volatility), causing the learning rate to increase. This reflects a hyper-vigilant state where the participant updates their beliefs too rapidly in response to outcomes.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Learning Rate.
    
    Hypothesis: Higher anxiety leads to a higher learning rate (hyper-plasticity),
    causing the participant to over-adjust to recent outcomes.
    
    Parameters:
    base_alpha: [0, 1] Baseline learning rate for zero anxiety.
    stai_alpha_boost: [0, 1] Scaling factor for how much STAI increases the learning rate.
    beta: [0, 10] Inverse temperature (choice consistency).
    w: [0, 1] Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    """
    base_alpha, stai_alpha_boost, beta, w = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]
    
    # Calculate effective learning rate based on anxiety
    # We clip to ensure it stays within valid bounds [0, 1]
    alpha = base_alpha + (stai_alpha_boost * current_stai)
    alpha = np.clip(alpha, 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Updating ---
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += alpha * delta_stage2
        
        # Update Stage 1 Q-values (Model-Free)
        # Note: We use the updated stage 2 value for the TD error
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Choice Stochasticity (Noise)
This model posits that anxiety acts as a disruptor to decision consistency. Instead of altering how values are learned or weighted, high anxiety decreases the inverse temperature (`beta`), making choices more random (noisier). This reflects the "choking under pressure" or cognitive interference aspect of anxiety.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Induced Choice Stochasticity.
    
    Hypothesis: Higher anxiety introduces noise into the decision process, 
    effectively lowering the inverse temperature (beta). High anxiety participants
    are less consistent in exploiting their knowledge.
    
    Parameters:
    alpha: [0, 1] Learning rate.
    base_beta: [0, 10] Baseline inverse temperature for low anxiety.
    stai_beta_decay: [0, 10] How much STAI reduces beta (increases noise).
    w: [0, 1] Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    """
    alpha, base_beta, stai_beta_decay, w = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]
    
    # Calculate effective beta based on anxiety
    # Higher anxiety reduces beta, making choices more random
    beta = base_beta - (stai_beta_decay * current_stai)
    beta = np.maximum(beta, 0.0) # Ensure beta doesn't go negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        s_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Updating ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += alpha * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Stickiness (Perseveration)
This model suggests that anxiety leads to rigid, repetitive behavior ("stickiness"). High anxiety participants might be more likely to repeat their previous Stage 1 choice regardless of the outcome, as a safety behavior or due to cognitive inflexibility.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Driven Stickiness (Perseveration).
    
    Hypothesis: Anxiety increases choice perseveration (stickiness). Anxious individuals
    are more likely to repeat the previous Stage 1 action, regardless of value.
    
    Parameters:
    alpha: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight (0 = pure Model-Free, 1 = pure Model-Based).
    stai_stickiness: [0, 5] Bonus added to the Q-value of the previously chosen action based on STAI.
    """
    alpha, beta, w, stai_stickiness = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]
    
    # Calculate stickiness magnitude
    stickiness_mag = stai_stickiness * current_stai

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1 # Initialize previous action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus to q_net for the previously chosen action
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness_mag
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Store action for next trial's stickiness
        prev_a1 = a1
        
        s_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        r = reward[trial]

        # --- Updating ---
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += alpha * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
