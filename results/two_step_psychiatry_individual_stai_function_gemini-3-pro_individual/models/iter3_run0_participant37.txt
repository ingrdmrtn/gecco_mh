Here are 3 new cognitive models that incorporate the STAI score into the decision-making process using different mechanisms than the previous learning-rate modulation.

### Model 1: Anxiety-Modulated Model-Based Control
**Hypothesis:** High anxiety consumes cognitive resources (working memory), reducing the capacity for complex Model-Based (planning) strategies. This model posits that the mixing weight $w$ (which balances Model-Based and Model-Free control) is negatively correlated with the STAI score. Higher anxiety leads to more Model-Free (habitual) behavior.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model where Anxiety reduces Model-Based control.
    
    Hypothesis: High anxiety acts as a cognitive load, reducing the mixing weight (w).
    Participants with high STAI rely more on Model-Free (habitual) systems.
    
    Parameters:
    lr: [0, 1] Learning rate for TD updates.
    beta: [0, 10] Inverse temperature.
    w_max: [0, 1] Maximum possible Model-Based weight (at 0 anxiety).
    stai_penalty: [0, 1] Reduction in Model-Based weight per unit of STAI.
    """
    lr, beta, w_max, stai_penalty = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate w based on STAI. 
    # High anxiety subtracts from the max potential model-basedness.
    w = np.clip(w_max - (stai_penalty * stai_score), 0.0, 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combine MB and MF values using the anxiety-modulated w
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update Stage 1 (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1
        
        # Update Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Decision Noise
**Hypothesis:** Anxiety acts as a disruptive "noise" parameter in the decision process. Instead of affecting how values are *learned*, anxiety affects how consistently values are *translated into action*. Higher STAI scores lower the inverse temperature ($\beta$), leading to more random/exploratory behavior (or confusion) regardless of the calculated values.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model where Anxiety increases Decision Noise.
    
    Hypothesis: Anxiety reduces the signal-to-noise ratio in decision making.
    Higher STAI scales down the Beta parameter, making choices more random.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta_base: [0, 10] Base inverse temperature (for low anxiety).
    w: [0, 1] Mixing weight (fixed).
    stai_dampener: [0, 1] Proportion by which STAI reduces beta.
    """
    lr, beta_base, w, stai_dampener = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective Beta. 
    # If stai_dampener is high, high anxiety significantly flattens the softmax (more random).
    beta_eff = beta_base * (1.0 - (stai_dampener * stai_score))
    beta_eff = np.maximum(beta_eff, 0.0) # Ensure non-negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use effective beta
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        # Use effective beta here as well
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1
        
        # Update Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Perseveration (Stickiness)
**Hypothesis:** Anxious individuals are risk-averse and prefer familiarity. This model hypothesizes that STAI drives "choice stickiness" (perseveration). High anxiety results in a stronger bonus added to the previously chosen option, making the participant resistant to switching spaceships even if value estimates suggest they should.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model with Anxiety-Driven Choice Stickiness.
    
    Hypothesis: Anxiety leads to perseveration (repeating previous choices) 
    to avoid the uncertainty of switching. STAI scales a 'stickiness' bonus.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Mixing weight.
    stai_stickiness: [0, 5] Magnitude of stickiness bonus per unit of STAI.
    """
    lr, beta, w, stai_stickiness = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate the stickiness bonus specific to this participant
    stickiness_bonus = stai_stickiness * stai_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply anxiety-driven stickiness
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness_bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Record action for next trial's stickiness
        last_action_1 = int(action_1[trial])
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, int(action_2[trial])] - q_stage1_mf[int(action_1[trial])]
        q_stage1_mf[int(action_1[trial])] += lr * delta_stage1
        
        # Update Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, int(action_2[trial])]
        q_stage2_mf[state_idx, int(action_2[trial])] += lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
