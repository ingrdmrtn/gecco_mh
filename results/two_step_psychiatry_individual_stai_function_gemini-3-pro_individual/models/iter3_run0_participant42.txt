Here are three cognitive models represented as Python functions.

### Model 1: Anxiety-Modulated Choice Perseveration
This model tests the hypothesis that anxiety increases "stickiness" or perseveration. High anxiety participants may be more prone to repeating their previous Stage 1 choice (a form of safety behavior or rigidity), regardless of the reward outcome.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Choice Perseveration (Stickiness).
    
    Hypothesis: Anxiety increases the tendency to repeat the previous Stage 1 choice 
    (perseveration), independent of reward history. This is modeled as a 'stickiness' 
    bonus added to the Q-value of the previously chosen action.
    
    Parameters:
    - learning_rate: [0, 1] Rate of updating Q-values.
    - beta: [0, 10] Inverse temperature for softmax (choice consistency).
    - w: [0, 1] Mixing weight for Model-Based (1) vs Model-Free (0).
    - stick_base: [0, 5] Baseline tendency to repeat choices.
    - anxiety_stick_mod: [0, 5] How much STAI increases the stickiness bonus.
    """
    learning_rate, beta, w, stick_base, anxiety_stick_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective stickiness based on anxiety
    # Higher anxiety -> higher bonus for the previous action
    stick_effective = stick_base + (participant_stai * anxiety_stick_mod)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix Model-Based and Model-Free values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus if not the first trial
        if trial > 0:
            prev_a = int(action_1[trial-1])
            q_net[prev_a] += stick_effective
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Update Stage 1 MF Q-values (SARSA-like)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 MF Q-values (Prediction Error from Reward)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Decision Noise
This model tests the hypothesis that anxiety acts as a distractor that reduces the signal-to-noise ratio in decision-making. High anxiety lowers the inverse temperature parameter (`beta`), making choices more random (stochastic) and less driven by learned values.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Induced Decision Noise (Beta Modulation).
    
    Hypothesis: High anxiety increases decision noise (lowers the inverse temperature beta),
    making behavior more random and less sensitive to value differences.
    
    Parameters:
    - learning_rate: [0, 1] Rate of updating Q-values.
    - beta_base: [0, 10] Baseline inverse temperature (for low anxiety).
    - w: [0, 1] Mixing weight for Model-Based (1) vs Model-Free (0).
    - anxiety_noise_scale: [0, 5] Scaling factor for how anxiety reduces beta.
    """
    learning_rate, beta_base, w, anxiety_noise_scale = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective beta
    # As anxiety increases, the denominator grows, reducing beta_effective.
    # This models "choking under pressure" or confusion.
    beta_effective = beta_base / (1.0 + (participant_stai * anxiety_noise_scale))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta_effective * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_effective * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Blunted Reward Sensitivity
This model tests the hypothesis that anxious individuals experience "reward dampening" or anhedonia-like effects. They may perceive the subjective value of a coin (reward=1) as lower than non-anxious participants, leading to smaller prediction errors and slower learning from positive outcomes.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Blunted Reward Sensitivity.
    
    Hypothesis: Anxiety dampens the subjective valuation of rewards.
    Instead of a reward of 1.0, anxious participants perceive a lower value,
    scaling down the prediction error in the second stage.
    
    Parameters:
    - learning_rate: [0, 1] Rate of updating Q-values.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Mixing weight for Model-Based (1) vs Model-Free (0).
    - anxiety_dampening: [0, 1] Factor by which anxiety reduces reward value.
    """
    learning_rate, beta, w, anxiety_dampening = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate subjective reward scalar
    # If anxiety is 0, scalar is 1.0. If anxiety is high, scalar < 1.0.
    reward_scalar = 1.0 - (participant_stai * anxiety_dampening)
    reward_scalar = np.clip(reward_scalar, 0.0, 1.0) # Ensure non-negative

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Apply subjective scaling to the received reward
        subjective_reward = reward[trial] * reward_scalar
        
        delta_stage2 = subjective_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
