Here are three new cognitive models for the two-step task, incorporating the STAI anxiety score into different mechanisms of the reinforcement learning process.

### Model 1: STAI-Modulated Choice Stickiness
This model hypothesizes that anxiety increases "perseverance" or "stickiness"â€”the tendency to repeat a previous choice regardless of whether it was rewarded, often interpreted as a safety-seeking behavior or resistance to switching.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with STAI-modulated Choice Perseverance (Stickiness).
    
    Hypothesis: Higher anxiety increases the tendency to repeat the previous Stage 1 choice 
    (perseverance), regardless of the outcome, as a safety/rigidity mechanism.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weight (mixing parameter).
    stick_base: [0, 5] Baseline stickiness (perseverance bonus).
    stick_sens: [0, 5] Sensitivity of stickiness to STAI score.
    """
    lr, beta, w, stick_base, stick_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Calculate effective stickiness bonus
    # Anxiety increases the bonus added to the previously chosen action
    stickiness_bonus = stick_base + (stick_sens * stai_val)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add stickiness bonus to the action chosen in the previous trial
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness_bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        prev_a1 = a1
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Stage 2 Update (TD)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update (TD)
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: STAI-Modulated Loss Sensitivity
This model hypothesizes that anxious individuals exhibit "subjective loss aversion" even in a gain-only task. They may perceive the absence of a reward (0 coins) not as neutral, but as a painful loss, driving stronger avoidance learning.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with STAI-modulated Loss Sensitivity.
    
    Hypothesis: Higher anxiety leads to subjective loss aversion. The absence of a reward (0)
    is perceived as a negative value (loss) proportional to the anxiety score, rather than neutral.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weight.
    loss_sens: [0, 5] Magnitude of subjective loss for 0-reward outcomes, scaled by STAI.
    """
    lr, beta, w, loss_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        # Apply Subjective Loss Aversion
        # If reward is 0 (approx 0), effective reward is negative based on STAI
        if r < 0.01:
            eff_reward = -1.0 * (loss_sens * stai_val)
        else:
            eff_reward = r
        
        # Stage 2 Update
        delta_stage2 = eff_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: STAI-Modulated Inverse Temperature
This model hypothesizes that anxiety fundamentally alters the exploration-exploitation trade-off (decision noise). It tests if anxiety leads to more rigid/deterministic choices (higher beta) across both stages.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with STAI-modulated Inverse Temperature (Beta).
    
    Hypothesis: Anxiety influences the exploration-exploitation trade-off. 
    Higher anxiety increases 'beta', leading to reduced exploration and higher rigidity.
    
    Parameters:
    lr: [0, 1] Learning rate.
    beta_base: [0, 10] Baseline inverse temperature.
    w: [0, 1] Model-based weight.
    stai_beta_sens: [0, 10] Sensitivity of beta to STAI. Positive = Anxiety increases rigidity.
    """
    lr, beta_base, w, stai_beta_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Calculate effective beta
    # We clip to ensure it stays within reasonable numerical bounds
    beta_eff = beta_base + (stai_beta_sens * stai_val)
    beta_eff = np.clip(beta_eff, 0.0, 20.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Updates ---
        r = reward[trial]
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
