Here are three new cognitive models that incorporate the participant's STAI score into the decision-making process.

### Cognitive Model 1: Anxiety-Induced Decision Noise
This model hypothesizes that high anxiety acts as a distractor, increasing neural noise and reducing the precision of value-based choices (lowering the inverse temperature, $\beta$). This reflects the "choking under pressure" phenomenon where high arousal leads to more erratic exploration.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Decision Noise.

    Hypothesis: High anxiety increases decision noise, effectively lowering
    the inverse temperature (beta). This leads to more random behavior 
    regardless of learned values.

    Parameters:
    - learning_rate: [0, 1] Standard learning rate for value updates.
    - beta_base: [0, 10] Baseline inverse temperature (precision) for low anxiety.
    - w: [0, 1] Mixing weight for Model-Based vs Model-Free.
    - anxiety_noise: [0, 5] Scaling factor. Higher values mean anxiety reduces beta more strongly.
    """
    learning_rate, beta_base, w, anxiety_noise = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Anxiety acts as a divisor to precision (beta), increasing randomness (noise)
    # If anxiety_noise is 0, beta is constant. If high, beta decreases as STAI increases.
    beta_eff = beta_base / (1.0 + anxiety_noise * stai_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax with anxiety-modulated beta
        exp_q1 = np.exp(beta_eff * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        # Standard MF choice at stage 2, also subject to anxiety noise
        exp_q2 = np.exp(beta_eff * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        chosen_a2 = int(action_2[trial])

        # --- Updates ---
        # TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        # TD(0) update for Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Anxiety-Driven Choice Stickiness
This model hypothesizes that anxiety promotes "safety behaviors" or perseveration. Anxious individuals may be more likely to repeat their previous Stage 1 choice regardless of the outcome, seeking familiarity or avoiding the cognitive load of switching.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Choice Stickiness.

    Hypothesis: Anxiety increases 'stickiness' or perseveration. The participant
    receives an artificial 'bonus' to the value of the action they chose 
    on the previous trial, proportional to their anxiety.

    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight for Model-Based vs Model-Free.
    - stickiness_amp: [0, 5] Amplitude of the anxiety-driven perseveration bonus.
    """
    learning_rate, beta, w, stickiness_amp = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits (pre-softmax values)
        logits = beta * q_net
        
        # Add anxiety-driven stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            logits[last_action_1] += stickiness_amp * stai_score
            
        exp_q1 = np.exp(logits - np.max(logits)) # subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        last_action_1 = int(action_1[trial])
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        chosen_a2 = int(action_2[trial])

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Anxiety-Modulated Reward Dampening
This model hypothesizes that high anxiety is associated with anhedonia or a blunted sensitivity to positive outcomes. The subjective utility of the reward is dampened by the STAI score, leading to weaker reinforcement signals (smaller prediction errors) when gold is found.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Reward Dampening.

    Hypothesis: Anxiety dampens the subjective valuation of rewards (anhedonia).
    The effective reward received is scaled down by the anxiety score, 
    resulting in smaller updates to value estimates.

    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight for Model-Based vs Model-Free.
    - reward_dampening: [0, 5] Scaling factor. Higher values mean anxiety reduces reward value more.
    """
    learning_rate, beta, w, reward_dampening = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        chosen_a2 = int(action_2[trial])

        # --- Updates ---
        
        # Calculate Effective Reward
        # Reward is dampened by anxiety. If dampening is 0, effective reward is normal.
        effective_reward = reward[trial] / (1.0 + reward_dampening * stai_score)

        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

        # Use effective_reward for the Stage 2 update
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
