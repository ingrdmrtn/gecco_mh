Here are 3 new cognitive models exploring how anxiety (STAI) impacts decision-making mechanisms in the two-step task. These models introduce variations in how anxiety affects reward sensitivity, learning rates for negative outcomes, and perseverance.

### Model 1: Anxiety-Modulated Reward Sensitivity
This model hypothesizes that anxiety alters the subjective valuation of rewards. High anxiety might blunt the impact of positive rewards (anhedonia-like) or heighten sensitivity to them (if seeking safety). Here, STAI scales the effective reward magnitude before it enters the prediction error calculation.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Reward Sensitivity.
    
    This model posits that anxiety affects how strongly a participant perceives the reward.
    Instead of modifying the learning rate or the MB/MF balance directly, STAI scales 
    the reward signal itself.
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based control (static).
    reward_sens_base: [0, 2] Baseline sensitivity to reward.
    stai_sens_mod: [0, 2] How much STAI scales reward sensitivity.
                          (effective_reward = reward * (reward_sens_base + stai * stai_sens_mod))
    """
    learning_rate, beta, w, reward_sens_base, stai_sens_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate effective reward scalar based on anxiety
    # If stai_sens_mod is positive, anxiety increases sensitivity. If negative logic is needed, bounds handle it.
    # We use a multiplicative factor. 
    eff_sensitivity = reward_sens_base + (participant_stai * stai_sens_mod)
    # Clamp to ensure stability, though bounds usually prevent extreme values
    if eff_sensitivity < 0: eff_sensitivity = 0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        
        # --- Update ---
        # Scale the physical reward by the anxiety-modulated sensitivity
        perceived_reward = reward[trial] * eff_sensitivity
        
        delta_stage2 = perceived_reward - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # TD-learning for Stage 1 (using the updated Stage 2 Q-value as proxy)
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Asymmetric Learning Rates
This model suggests that anxiety creates a bias in how people learn from "good" news (positive prediction errors) versus "bad" news (negative prediction errors). High anxiety might result in a higher learning rate for negative outcomes (hyper-vigilance to failure).

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Driven Asymmetric Learning Rates.
    
    This model splits the learning rate into positive (alpha_pos) and negative (alpha_neg)
    components. The negative learning rate is specifically modulated by STAI, testing
    the hypothesis that anxious individuals over-learn from disappointment or lack of reward.
    
    Parameters:
    alpha_pos: [0, 1] Learning rate for positive prediction errors (RPE > 0).
    alpha_neg_base: [0, 1] Baseline learning rate for negative RPEs.
    stai_neg_mod: [0, 1] Additional boost to negative learning rate based on STAI.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF mixing weight.
    """
    alpha_pos, alpha_neg_base, stai_neg_mod, beta, w = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate effective negative learning rate
    alpha_neg_eff = alpha_neg_base + (participant_stai * stai_neg_mod)
    if alpha_neg_eff > 1: alpha_neg_eff = 1
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        r = reward[trial]

        # --- Update Stage 2 ---
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        
        # Apply asymmetric learning rates
        lr_2 = alpha_pos if delta_stage2 >= 0 else alpha_neg_eff
        q_stage2_mf[state_idx, chosen_a2] += lr_2 * delta_stage2

        # --- Update Stage 1 ---
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        
        # Apply asymmetric learning rates
        lr_1 = alpha_pos if delta_stage1 >= 0 else alpha_neg_eff
        q_stage1_mf[chosen_a1] += lr_1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Choice Stickiness
This model investigates if anxiety leads to "stickiness" or choice perseveration (repeating the previous choice regardless of reward), which is a common coping mechanism to reduce cognitive load or uncertainty. The degree of stickiness is a function of the STAI score.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Induced Choice Stickiness.
    
    This model adds a 'stickiness' parameter to the softmax decision rule.
    Stickiness biases the agent to repeat the last taken action.
    Here, the magnitude of this stickiness bias is determined by the STAI score.
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF mixing weight.
    stickiness_base: [0, 5] Baseline tendency to repeat choices.
    stai_sticky_mod: [0, 5] How much STAI increases the stickiness bias.
    """
    learning_rate, beta, w, stickiness_base, stai_sticky_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]

    # Calculate effective stickiness
    eff_stickiness = stickiness_base + (participant_stai * stai_sticky_mod)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Track previous choice (initialize to -1 or handle first trial separately)
    last_choice_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus to Q-values before softmax
        # We create a temporary Q vector for decision making
        decision_q = q_net.copy()
        if last_choice_1 != -1:
            decision_q[last_choice_1] += eff_stickiness
        
        exp_q1 = np.exp(beta * decision_q)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]
        last_choice_1 = chosen_a1 # Update for next trial

        # --- Stage 2 Policy ---
        # Note: Stickiness is usually modeled on the first stage choice in this task,
        # but could apply to second stage. Here we apply it only to stage 1 for simplicity.
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        r = reward[trial]

        # --- Update ---
        delta_stage2 = r - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
