Here are three new cognitive models exploring different mechanisms by which anxiety (STAI) might influence decision-making in the two-step task.

### Model 1: Anxiety-Modulated Learning Rate
This model tests the hypothesis that anxiety alters how quickly participants update their value estimates. High anxiety (high STAI) is modeled as increasing the learning rate (`alpha`), representing a "hyper-vigilant" state where participants over-react to the most recent feedback rather than integrating over a longer history.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Free learner where anxiety (STAI) modulates the learning rate.
    
    Hypothesis: Higher anxiety leads to 'hyper-vigilance' or over-reaction to recent
    outcomes, modeled as an increased learning rate.
    
    Parameters:
    base_alpha: [0, 1] Baseline learning rate for zero anxiety.
    stai_alpha_mod: [0, 1] How much STAI increases the learning rate.
    beta: [0, 10] Inverse temperature (choice consistency).
    w: [0, 1] Weighting between Model-Based (1) and Model-Free (0) systems.
    """
    base_alpha, stai_alpha_mod, beta, w = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]

    # Effective alpha increases with STAI
    # We clip to ensure it stays valid
    alpha = base_alpha + (stai_alpha_mod * current_stai)
    alpha = np.clip(alpha, 0, 1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Combined value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        r = reward[trial]

        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += alpha * delta_stage2
        
        # Update Stage 1 Q-values (Model-Free)
        # Using the actual Stage 2 value for the update target (SARSA-like logic here)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Choice Noise (Inverse Temperature)
This model proposes that anxiety degrades the consistency of choices. Instead of affecting learning or planning directly, anxiety reduces the `beta` (inverse temperature) parameter. A lower `beta` means choices become more random/exploratory (or erratic) regardless of the calculated values, reflecting the "noise" or distraction anxiety introduces into the decision process.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid learner where anxiety (STAI) increases decision noise (lowers Beta).
    
    Hypothesis: Anxiety acts as a distractor, making choices less deterministic 
    with respect to value estimates. High STAI reduces beta.
    
    Parameters:
    alpha: [0, 1] Learning rate.
    base_beta: [0, 10] Baseline inverse temperature (max consistency).
    stai_noise: [0, 10] How much STAI subtracts from beta (adds noise).
    w: [0, 1] Weighting between Model-Based and Model-Free systems.
    """
    alpha, base_beta, stai_noise, w = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]

    # Effective beta decreases as STAI increases
    # We clip to ensure it stays non-negative
    beta = base_beta - (stai_noise * current_stai)
    beta = np.clip(beta, 0.0, 10.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        r = reward[trial]

        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += alpha * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Stickiness (Perseveration)
This model hypothesizes that anxiety leads to rigid, repetitive behavior ("perseveration"). It introduces a "stickiness" parameter that biases the agent to repeat the previous choice. The magnitude of this stickiness bonus is scaled by the participant's STAI score.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid learner with anxiety-modulated choice perseverance (stickiness).
    
    Hypothesis: Anxiety increases the tendency to repeat the previous Stage 1 choice
    regardless of reward (safety behavior/rigidity).
    
    Parameters:
    alpha: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-Based weight.
    stai_stickiness: [0, 5] Bonus added to the Q-value of the previously chosen action, scaled by STAI.
    """
    alpha, beta, w, stai_stickiness = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]

    # Calculate the stickiness bonus based on anxiety
    stick_bonus = stai_stickiness * current_stai

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    last_action_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stick_bonus
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_action_1 = a1 # Update last action
        
        # --- Stage 2 Policy ---
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        r = reward[trial]

        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += alpha * delta_stage2
        
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
