Here are 3 new cognitive models that explore different mechanisms for how anxiety (STAI) might influence decision-making in the two-step task.

### Model 1: Anxiety Modulates Reward Sensitivity
This model hypothesizes that anxiety alters the subjective valuation of rewards. High anxiety might lead to a "blunting" effect (lower sensitivity to gains) or hypersensitivity. Here, STAI scales the effective reward received, which impacts the prediction error `delta_stage2`.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Reward Sensitivity Modulation Model.
    
    Hypothesis: Anxiety (STAI) modulates the subjective value of the reward. 
    High anxiety might dampen the perceived value of positive outcomes (anhedonia-like) 
    or heighten sensitivity to feedback.
    
    effective_reward = reward * (reward_sens_base + reward_sens_stai * stai)
    
    Parameters:
    - learning_rate: [0, 1] Alpha for Q-learning updates.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Mixing weight (0=MF, 1=MB).
    - reward_sens_base: [0, 1] Baseline sensitivity to reward.
    - reward_sens_stai: [-1, 1] How STAI changes reward sensitivity.
    """
    learning_rate, beta, w, reward_sens_base, reward_sens_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective reward sensitivity
    # We clip the multiplier to ensure it stays within a reasonable positive range or allows 0
    # Note: reward_sens_stai can be negative, implying anxiety reduces reward sensitivity.
    r_sens = reward_sens_base + (reward_sens_stai * stai_score)
    r_sens = max(0.0, min(2.0, r_sens)) # Allow up to double sensitivity, but not negative reward

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted combination of MF and MB values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        
        # MODULATION: Subjective reward value
        r_effective = reward[trial] * r_sens
        
        # Standard TD updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r_effective - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety Modulates Second-Stage Learning Rate
This model proposes that anxiety specifically affects how much participants learn from the immediate outcome at the second stage (the aliens). High anxiety might lead to "over-updating" or volatility in value estimation at the terminal state, distinct from the first-stage transition learning.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Stage-Specific Learning Rate Modulation Model.
    
    Hypothesis: Anxiety affects the learning rate specifically for the second stage 
    (alien choice), reflecting altered plasticity to immediate rewards, while the 
    first stage learning rate (transition structure/value) remains constant.
    
    lr_stage2 = lr_base + lr_stai * stai
    
    Parameters:
    - lr_stage1: [0, 1] Learning rate for Stage 1 updates.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Mixing weight.
    - lr_base_s2: [0, 1] Baseline learning rate for Stage 2.
    - lr_stai_s2: [-1, 1] Sensitivity of Stage 2 learning rate to STAI.
    """
    lr_stage1, beta, w, lr_base_s2, lr_stai_s2 = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate stage 2 specific learning rate
    lr_stage2 = lr_base_s2 + (lr_stai_s2 * stai_score)
    lr_stage2 = max(0.0, min(1.0, lr_stage2))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Update Stage 1 using fixed learning rate
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Update Stage 2 using anxiety-modulated learning rate
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety Modulates Choice Perseveration
This model introduces a perseveration parameter `p` that biases the agent to repeat the previous choice. It hypothesizes that anxiety drives a "safety behavior" or rigidity, increasing the tendency to stick with the same choice regardless of value (habitual rigidity).

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Perseveration Model.
    
    Hypothesis: Anxiety increases behavioral rigidity or perseveration. 
    A "stickiness" bonus is added to the Q-value of the previously chosen action 
    at Stage 1.
    
    perseveration_bonus = persev_base + persev_stai * stai
    
    Parameters:
    - learning_rate: [0, 1] Alpha.
    - beta: [0, 10] Beta.
    - w: [0, 1] Mixing weight.
    - persev_base: [-1, 1] Baseline tendency to repeat choice (positive) or switch (negative).
    - persev_stai: [-1, 1] How STAI modulates this stickiness.
    """
    learning_rate, beta, w, persev_base, persev_stai = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate perseveration bonus
    persev = persev_base + (persev_stai * stai_score)
    # No strict clipping needed for bonus values, but keeping it reasonable helps optimization
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1 # Indicator for previous action

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseveration bonus to the net Q-values before softmax
        q_net_biased = q_net.copy()
        if last_action_1 != -1:
            q_net_biased[last_action_1] += persev
        
        exp_q1 = np.exp(beta * q_net_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        last_action_1 = a1 # Update for next trial

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
