### Model 1: Anxiety-Dependent Working Memory Load (Dual Beta)
This model hypothesizes that anxiety acts as a cognitive load that disproportionately affects the more cognitively demanding "planning" stage (Stage 1) compared to the simpler "habitual" stage (Stage 2). Instead of a single noise parameter ($\beta$), this model allows the precision of the Stage 1 choice to be dampened by the STAI score relative to Stage 2.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Dependent Working Memory Load (Dual Beta).
    
    Hypothesis: High anxiety consumes working memory resources required for the 
    model-based planning in Stage 1, making Stage 1 choices noisier (lower beta) 
    than Stage 2 choices.
    
    Parameters:
    learning_rate: [0, 1] Standard Q-learning rate.
    w: [0, 1] Mixing weight (1=Model-Based, 0=Model-Free).
    beta_base: [0, 10] Baseline inverse temperature (precision) for Stage 2.
    stai_s1_dampening: [0, 1] Proportion by which STAI reduces Beta for Stage 1.
    """
    learning_rate, w, beta_base, stai_s1_dampening = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    # Normalize STAI (approx range 20-80) to 0-1
    norm_stai = (stai_val - 20.0) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)

    # Beta for Stage 2 is the baseline precision
    beta_2 = beta_base
    
    # Beta for Stage 1 is dampened by anxiety
    # If stai is high and dampening is high, beta_1 becomes smaller (noisier/random)
    beta_1 = beta_base * (1.0 - (stai_s1_dampening * norm_stai))
    beta_1 = max(0.0, beta_1) # Ensure non-negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Mix MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Use beta_1 (dampened by anxiety) for Stage 1
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        state_idx = int(state[trial])

        # policy for the second choice
        # Use beta_2 (baseline) for Stage 2
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # Update Q-values
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Standard TD(0) update for Stage 1 MF
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Memory Decay
This model introduces a "forgetting" or decay mechanism. While standard RL maintains Q-values indefinitely until updated, this model assumes that memory traces decay over time. The rate of this decay is modulated by STAI, hypothesizing that high anxiety (intrusive thoughts/worry) accelerates the degradation of stored value associations for unchosen options.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Memory Decay.
    
    Hypothesis: Anxiety interferes with the maintenance of value representations 
    over time. Unchosen options decay towards zero (forgetting) faster 
    in individuals with high STAI scores.
    
    Parameters:
    learning_rate: [0, 1] Update rate for chosen options.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] MB/MF mixing weight.
    decay_base: [0, 1] Baseline decay rate for unchosen options.
    stai_decay_sens: [0, 1] Sensitivity of decay rate to STAI score.
    """
    learning_rate, beta, w, decay_base, stai_decay_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    norm_stai = (stai_val - 20.0) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)

    # Calculate specific decay rate for this participant
    current_decay = decay_base + (stai_decay_sens * norm_stai)
    current_decay = np.clip(current_decay, 0, 1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # Update Chosen Q-values
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Apply Decay to Unchosen Options (Forgetting)
        # 1. Decay Stage 1 unchosen
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - current_decay)
        
        # 2. Decay Stage 2 unchosen (in the current state)
        unchosen_a2 = 1 - a2
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - current_decay)
        
        # (Optional: One could also decay the unvisited state, but we keep it simple here)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Loss Shifting (Win-Stay/Lose-Shift)
This model implements a specific heuristic often observed in anxiety: "Loss Shifting." While general stickiness implies repeating an action regardless of outcome, Loss Shifting implies a specific urge to switch away from an option that just resulted in failure (0 coins). This model tests if STAI modulates the strength of this "switch" signal specifically after a loss.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Loss Shifting.
    
    Hypothesis: Anxiety increases sensitivity to negative outcomes. Specifically, 
    after receiving 0 reward, anxious participants are more likely to 'shift' 
    (switch) their Stage 1 choice on the next trial, modeled as a negative 
    stickiness (repulsion) applied only after a loss.
    
    Parameters:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    shift_base: [0, 5] Baseline tendency to switch after a loss.
    stai_shift_sens: [0, 5] STAI modulation of loss-shifting.
    """
    learning_rate, beta, w, shift_base, stai_shift_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]

    norm_stai = (stai_val - 20.0) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)

    # Calculate repulsion strength after a loss
    loss_shift_magnitude = shift_base + (stai_shift_sens * norm_stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    prev_action_1 = -1
    prev_reward = -1

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Loss-Shift Heuristic
        # If previous trial resulted in 0 reward, penalize the previous action
        if prev_action_1 != -1 and prev_reward == 0:
            q_net[prev_action_1] -= loss_shift_magnitude

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # Update
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store for next trial logic
        prev_action_1 = a1
        prev_reward = r

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
