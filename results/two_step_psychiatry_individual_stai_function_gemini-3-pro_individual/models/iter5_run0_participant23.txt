Here are three new cognitive models that incorporate STAI scores into the two-step task, exploring mechanisms of memory decay, reward sensitivity, and stage-specific decision rigidity.

### Model 1: Anxiety-Induced Working Memory Decay
**Hypothesis:** High anxiety creates cognitive load that interferes with working memory maintenance. In this model, STAI modulates a **decay parameter**. Participants with higher anxiety "forget" learned Q-values faster than those with low anxiety, representing the interference of anxious rumination on value retention.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    RL Model with STAI-modulated Q-value Decay.
    
    Hypothesis: Anxiety consumes working memory resources, leading to faster 
    passive forgetting (decay) of learned Q-values.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - MB/MF weighting (0=MF, 1=MB).
    decay_base: [0, 1] - Baseline forgetting rate per trial.
    stai_decay_slope: [0, 1] - How much STAI increases the decay rate.
    """
    learning_rate, beta, w, decay_base, stai_decay_slope = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]
    
    # Calculate effective decay rate based on STAI
    # We clip to ensure it stays within [0, 1]
    decay_rate = decay_base + (stai_decay_slope * current_stai)
    if decay_rate > 1.0: decay_rate = 1.0
    if decay_rate < 0.0: decay_rate = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        
        # Apply decay to Q-values at the start of the trial
        q_stage1_mf *= (1.0 - decay_rate)
        q_stage2_mf *= (1.0 - decay_rate)

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted mixture of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Standard Q-learning updates
        act1 = int(action_1[trial])
        act2 = int(action_2[trial])
        r = reward[trial]

        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Blunted Reward Sensitivity
**Hypothesis:** Anxiety is often associated with dampening of positive affect (anhedonia) or pessimism. In this model, STAI scales the **perceived magnitude of the reward**. High anxiety participants perceive the "gold coin" as having less value, leading to smaller prediction errors and slower value accumulation.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    RL Model with STAI-modulated Reward Sensitivity.
    
    Hypothesis: High anxiety blunts the subjective value of rewards (anhedonia/pessimism).
    The effective reward received is scaled down as STAI increases.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - MB/MF weighting.
    lambda_elig: [0, 1] - Eligibility trace (Stage 2 -> Stage 1 update connection).
    stai_reward_damp: [0, 1] - Factor by which STAI reduces reward sensitivity.
    """
    learning_rate, beta, w, lambda_elig, stai_reward_damp = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        act1 = int(action_1[trial])
        act2 = int(action_2[trial])
        
        # Calculate effective reward based on STAI
        # If stai_reward_damp is high, effective reward decreases
        effective_reward = reward[trial] * (1.0 - (stai_reward_damp * current_stai))
        # Ensure reward doesn't invert to negative
        if effective_reward < 0: effective_reward = 0.0

        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        
        # Update Stage 1 MF (using eligibility trace lambda)
        # Note: We use the *Stage 2* PE to update Stage 1 via lambda
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, act2]
        
        q_stage1_mf[act1] += learning_rate * delta_stage1 + learning_rate * lambda_elig * delta_stage2
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Stage-Specific Rigidity (Proximal Threat)
**Hypothesis:** Anxiety affects decision-making differently depending on the proximity of the outcome. While the first stage (Spaceship) is abstract, the second stage (Alien) is concrete and immediate. This model proposes that anxiety causes **rigidity (higher beta)** specifically at **Stage 2**. High anxiety participants become more deterministic/greedy when facing the final choice, while remaining more exploratory at the abstract planning stage.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    RL Model with STAI-modulated Stage 2 Beta (Rigidity).
    
    Hypothesis: Anxiety causes rigid, risk-averse behavior specifically when the 
    outcome is immediate (Stage 2). STAI increases Beta for Stage 2 only, 
    making the agent more deterministic/greedy at the final step.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta_stage1: [0, 10] - Exploration/exploitation for the Spaceship choice.
    w: [0, 1] - MB/MF weighting.
    beta_stage2_base: [0, 10] - Baseline Beta for the Alien choice.
    stai_beta2_slope: [0, 5] - How much STAI increases Beta at Stage 2.
    """
    learning_rate, beta_stage1, w, beta_stage2_base, stai_beta2_slope = model_parameters
    n_trials = len(action_1)
    current_stai = stai[0]
    
    # Calculate Stage 2 specific beta
    beta_stage2 = beta_stage2_base + (stai_beta2_slope * current_stai)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice (Uses beta_stage1)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_stage1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice (Uses STAI-modulated beta_stage2)
        exp_q2 = np.exp(beta_stage2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        act1 = int(action_1[trial])
        act2 = int(action_2[trial])
        r = reward[trial]

        # Standard updating
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
