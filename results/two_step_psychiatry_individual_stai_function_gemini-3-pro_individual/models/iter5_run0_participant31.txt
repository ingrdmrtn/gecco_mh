Here are 3 new cognitive models that incorporate individual STAI scores into the reinforcement learning process, focusing on mechanisms distinct from the previous attempts (which focused heavily on Model-Based weight $w$ and simple learning rate modulation).

### Model 1: Anxiety-Modulated Exploration (Temperature)
This model hypothesizes that anxiety affects the "noise" or exploration in decision-making. Specifically, high anxiety might lead to more erratic behavior (panic-like/choking) or conversely, rigid behavior. Here, we model it as anxiety dampening the inverse temperature ($\beta$), making choices more random (noisier) as anxiety increases, reflecting a difficulty in utilizing learned values to drive precise actions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Decision Noise.
    
    Hypothesis: High anxiety interferes with the ability to strictly exploit learned 
    values, effectively increasing decision noise (lowering the inverse temperature beta).
    Higher STAI scores result in a lower effective beta.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_max: [0, 10] The inverse temperature at 0 anxiety (maximum precision).
    - w: [0, 1] Weight for Model-Based control (fixed, not modulated by anxiety here).
    - anxiety_noise: [0, 1] Scaling factor for how much STAI reduces beta.
      beta_effective = beta_max * (1 - anxiety_noise * stai)
    """
    learning_rate, beta_max, w, anxiety_noise = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    # Anxiety reduces the precision (beta) of choices
    beta_effective = beta_max * (1.0 - (anxiety_noise * stai_val))
    beta_effective = np.maximum(beta_effective, 0.0) # Ensure non-negative

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q-value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_effective * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta_effective * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Update logic
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Punishment Sensitivity
This model hypothesizes that anxious individuals are hypersensitive to negative outcomes (omission of reward). Instead of a global learning rate change, this model implements separate learning rates for positive and negative prediction errors, where the learning rate for negative errors is boosted by the STAI score.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Induced Punishment Sensitivity.
    
    Hypothesis: Anxiety leads to over-learning from negative outcomes (loss/omission).
    The learning rate for negative prediction errors is amplified by the STAI score.
    
    Parameters:
    - lr_base: [0, 1] Base learning rate for positive prediction errors.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - loss_amp: [0, 1] How much STAI amplifies learning from negative errors.
      lr_neg = lr_base + (loss_amp * stai)
    """
    lr_base, beta, w, loss_amp = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Calculate deltas
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        # Dynamic Learning Rate for Stage 1
        lr_1 = lr_base
        if delta_stage1 < 0:
            lr_1 = lr_base + (loss_amp * stai_val)
            lr_1 = np.clip(lr_1, 0.0, 1.0)
        q_stage1_mf[a1] += lr_1 * delta_stage1

        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        
        # Dynamic Learning Rate for Stage 2
        lr_2 = lr_base
        if delta_stage2 < 0:
            lr_2 = lr_base + (loss_amp * stai_val)
            lr_2 = np.clip(lr_2, 0.0, 1.0)
        q_stage2_mf[state_idx, a2] += lr_2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Perseveration (Stickiness)
This model hypothesizes that anxiety drives "safety behaviors" or repetition. Anxious participants may be more likely to repeat their previous Stage 1 choice regardless of the reward outcome, simply to avoid the uncertainty of switching. The STAI score scales a "stickiness" parameter that adds a bonus to the previously chosen action.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Driven Perseveration (Stickiness).
    
    Hypothesis: High anxiety promotes repetitive 'safety' behaviors. The STAI score
    scales a choice autocorrelation (stickiness) bonus added to the previously 
    chosen action in Stage 1, independent of reward history.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-Based weight.
    - stick_anxiety: [0, 5] Magnitude of stickiness induced by anxiety.
      bonus = stick_anxiety * stai
    """
    learning_rate, beta, w, stick_anxiety = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous action for stickiness
    last_action = -1 

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits (values before softmax)
        logits = beta * q_net
        
        # Add anxiety-driven stickiness bonus to the previous action
        if last_action != -1:
            logits[last_action] += (stick_anxiety * stai_val)

        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        # Update last action
        last_action = int(action_1[trial])
        state_idx = int(state[trial])

        # policy for the second choice (no stickiness assumed here)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
