Here are the three proposed cognitive models.

### Model 1: Anxiety-Modulated Perseveration
This model tests the hypothesis that anxiety drives repetitive "safety behaviors" (perseveration). Instead of affecting how values are learned, STAI modulates a "stickiness" parameter that biases the participant to repeat their previous Stage 1 choice, regardless of the reward outcome.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Perseveration (Stickiness).
    
    This model posits that anxiety increases the tendency to repeat the previous choice (perseveration),
    regardless of the model-based or model-free value estimates. This reflects a "safety behavior" 
    or rigidity often seen in high anxiety.
    
    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Weight for Model-Based control (0=MF, 1=MB).
    pers_base: [0, 5] Baseline perseveration bonus added to the logits of the repeated action.
    pers_stai_sens: [0, 5] Increase in perseveration per unit of STAI.
    """
    learning_rate, beta, w, pers_base, pers_stai_sens = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate effective perseveration based on anxiety
    pers_eff = pers_base + (pers_stai_sens * participant_stai)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1 # Initialize with no previous action

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value mixing
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits = beta * q_net
        
        # Add perseveration bonus to the previously chosen action
        if prev_action_1 != -1:
            logits[int(prev_action_1)] += pers_eff
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]
        prev_action_1 = chosen_a1 # Update for next trial

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        
        # Update values
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Transition Uncertainty
This model assumes that anxiety degrades confidence in the environmental structure. While low-anxiety participants might trust the 70/30 transition probabilities, high-anxiety participants may perceive the world as more volatile or uncertain, effectively flattening their internal transition matrix towards 50/50 and reducing the effectiveness of model-based planning.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Modulated Transition Uncertainty.
    
    This model assumes that high anxiety degrades the participant's confidence in the 
    transition structure of the task. While low-anxiety participants use the correct 
    [0.7, 0.3] probabilities, high-anxiety participants perceive the transitions as 
    noisier (closer to [0.5, 0.5]), reducing the efficacy of the Model-Based component.

    Parameters:
    learning_rate: [0, 1] Update rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based control.
    uncertainty_stai: [0, 1] How much STAI flattens the transition matrix (0=No effect).
    """
    learning_rate, beta, w, uncertainty_stai = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    # Modulate transition belief based on STAI
    # We subtract a portion from the dominant probability (0.7) based on STAI.
    # The factor 0.2 ensures that at max anxiety (stai=1, param=1), prob drops from 0.7 to 0.5.
    prob_common = 0.7 - (0.2 * uncertainty_stai * participant_stai)
    
    # Constraint: prob shouldn't drop below random chance (0.5)
    if prob_common < 0.5: prob_common = 0.5
    
    # Construct the anxiety-distorted transition matrix
    transition_matrix = np.array([[prob_common, 1-prob_common], [1-prob_common, prob_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
  
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Negative Learning Bias
This model hypothesizes that anxiety creates a hypersensitivity to failure (negative prediction errors). It implements an asymmetric learning rate where the rate of updating is boosted specifically when the participant receives 0 coins, proportional to their anxiety level.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Driven Negative Learning Bias.
    
    This model hypothesizes that anxiety creates a hypersensitivity to negative outcomes (0 coins).
    The learning rate is dynamically boosted when the participant fails to receive a reward,
    reflecting an anxious preoccupation with failure/loss.

    Parameters:
    lr_base: [0, 1] Baseline learning rate for rewards (wins).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Weight for Model-Based control.
    loss_sens_stai: [0, 5] Multiplier for the learning rate on loss trials, scaled by STAI.
    """
    lr_base, beta, w, loss_sens_stai = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        
        # Determine learning rate for this trial
        # If reward is 0 (loss), boost LR based on STAI
        current_lr = lr_base
        if reward[trial] == 0:
            boost = 1 + (loss_sens_stai * participant_stai)
            current_lr = lr_base * boost
            # Cap LR at 1.0 to prevent instability
            if current_lr > 1.0: current_lr = 1.0
  
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += current_lr * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += current_lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
