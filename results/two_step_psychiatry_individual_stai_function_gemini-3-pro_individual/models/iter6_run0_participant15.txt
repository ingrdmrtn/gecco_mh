Here are three new cognitive models that incorporate STAI anxiety scores into the decision-making process.

### Cognitive Model 1: Anxiety-Modulated Subjective Loss Valuation
This model hypothesizes that anxiety alters how "empty" outcomes (0 coins) are perceived. Rather than treating a lack of reward as a neutral zero, anxious individuals may perceive it as a negative value (loss). The magnitude of this negative valuation scales with their STAI score.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Subjective Loss Valuation.
    
    This model posits that anxiety alters the subjective perception of non-reward (0 coins).
    Instead of treating 0 coins as a neutral outcome (value 0), anxious individuals 
    may perceive it as a loss (negative value).
    
    Effective Reward (R_eff):
    If Reward = 1: R_eff = 1
    If Reward = 0: R_eff = -1 * (loss_base + stai_loss_sens * normalized_stai)
    
    Parameters:
    learning_rate: [0, 1] Learning rate for Q-values.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based (1) vs Model-free (0) weighting.
    loss_base: [0, 5] Baseline magnitude of negative value for 0 reward.
    stai_loss_sens: [0, 5] Increase in negative valuation due to anxiety.
    """
    learning_rate, beta, w, loss_base, stai_loss_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Normalize STAI to 0-1 range (approx 20-80 range)
    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    
    loss_magnitude = loss_base + (stai_loss_sens * norm_stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        # Calculate Effective Reward
        r_observed = reward[trial]
        if r_observed == 1:
            r_eff = 1.0
        else:
            r_eff = -1.0 * loss_magnitude

        # Update Q-values with effective reward
        # Stage 1 update (using value of state reached)
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 update (using effective reward)
        delta_stage2 = r_eff - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 2: Anxiety-Modulated Memory Decay
This model suggests that anxiety impairs the maintenance of value representations. Specifically, it implements a decay parameter for *unchosen* options. Higher anxiety leads to faster forgetting (decay towards 0) of the values associated with actions not currently taken.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Memory Decay.
    
    This model incorporates a decay mechanism for unchosen options. 
    Anxiety is hypothesized to increase cognitive load or worry, leading to 
    faster forgetting (decay) of the values of options that were not currently experienced.
    
    decay_rate = decay_base + (stai_decay_sens * normalized_stai)
    
    Parameters:
    learning_rate: [0, 1] Learning rate for chosen options.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weighting.
    decay_base: [0, 1] Baseline decay rate for unchosen options.
    stai_decay_sens: [0, 1] Sensitivity of decay rate to anxiety.
    """
    learning_rate, beta, w, decay_base, stai_decay_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    
    decay_rate = decay_base + (stai_decay_sens * norm_stai)
    decay_rate = np.clip(decay_rate, 0, 1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Update Chosen Stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Decay Unchosen Stage 1
        q_stage1_mf[1-a1] *= (1.0 - decay_rate)

        # Update Chosen Stage 2
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Decay Unchosen Stage 2 (in the current state)
        q_stage2_mf[state_idx, 1-a2] *= (1.0 - decay_rate)
        
        # Note: We do not decay the values of the unvisited state.
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Cognitive Model 3: Anxiety-Modulated 'Freezing' (Loss-Induced Stickiness)
This model implements a specific perseveration bias that is triggered only by negative outcomes. It hypothesizes that high anxiety leads to a "freezing" response where participants repeat their previous choice if it resulted in a loss, overriding the typical tendency to switch after failure (Win-Stay, Lose-Shift).

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated 'Freezing' (Loss-Induced Stickiness).
    
    This model implements a specific type of stickiness that occurs only after losses (0 reward).
    High anxiety is hypothesized to increase a 'freezing' response, leading the participant 
    to repeat the previous Stage 1 choice if it resulted in a loss, contrary to a 
    standard Win-Stay Lose-Shift strategy.
    
    stickiness = stick_loss_base + (stai_freeze_sens * normalized_stai)
    (Applied only if previous_reward == 0)
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-based weighting.
    stick_loss_base: [0, 5] Baseline stickiness after a loss.
    stai_freeze_sens: [0, 5] Increase in loss-stickiness due to anxiety.
    """
    learning_rate, beta, w, stick_loss_base, stai_freeze_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    
    loss_stickiness = stick_loss_base + (stai_freeze_sens * norm_stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    prev_reward = -1

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply stickiness ONLY if previous trial was a loss (reward == 0)
        if prev_action_1 != -1 and prev_reward == 0:
            q_net[prev_action_1] += loss_stickiness

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        prev_action_1 = a1
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
  
        r = reward[trial]
        prev_reward = r

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
