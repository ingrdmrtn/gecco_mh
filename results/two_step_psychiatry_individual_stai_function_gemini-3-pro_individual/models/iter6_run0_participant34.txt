Here are 3 new cognitive models that incorporate STAI scores into the decision-making process in distinct ways, focusing on mechanisms like reward sensitivity modulation, learning rate asymmetry, and modulation of model-based control.

### Model 1: Anxiety-Modulated Reward Sensitivity
This model hypothesizes that anxiety alters the subjective valuation of rewards. High anxiety might blunt the impact of positive outcomes (anhedonia-like effect) or, conversely, heighten sensitivity to feedback. Here, the STAI score scales the effective reward magnitude before it enters the prediction error calculation.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Reward Sensitivity.
    
    Hypothesis: Anxiety influences how intensely rewards are perceived. 
    The STAI score modulates the effective reward value used in prediction errors.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    - reward_sens_base: [0, 2] Baseline sensitivity to reward.
    - anxiety_sens_mod: [-1, 1] How STAI modulates reward sensitivity.
      (Positive = anxiety increases sensitivity; Negative = blunts it).
    """
    learning_rate, beta, w, reward_sens_base, anxiety_sens_mod = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective reward sensitivity
    # We clip to ensure effective reward doesn't become illogical (e.g., negative for a positive coin)
    # though mathematically a negative sensitivity implies avoidance of 'reward'.
    effective_sensitivity = reward_sens_base + (anxiety_sens_mod * stai_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        chosen_a2 = int(action_2[trial])
  
        # Update stage 1
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        # Update stage 2 with modulated reward
        subjective_reward = reward[trial] * effective_sensitivity
        delta_stage2 = subjective_reward - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Learning Rate Asymmetry
This model proposes that anxiety causes a bias in how people learn from positive versus negative prediction errors (punishment/omission vs. reward). High anxiety might lead to faster learning from negative outcomes (omission of gold) due to hypervigilance for threat/failure.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Learning Rate Asymmetry.
    
    Hypothesis: Anxious individuals may learn differently from positive outcomes
    vs. negative outcomes (omission of reward).
    STAI determines the balance between a base learning rate and a specific
    'loss' (no reward) learning rate.
    
    Parameters:
    - lr_base: [0, 1] Base learning rate for positive outcomes.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    - anxiety_neg_bias: [0, 1] How much STAI adds to the learning rate for negative prediction errors.
    """
    lr_base, beta, w, anxiety_neg_bias = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        chosen_a2 = int(action_2[trial])
  
        # Stage 1 update
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        # Apply asymmetry to stage 1 update based on sign of PE? 
        # Usually asymmetry is applied at outcome receipt (stage 2), but let's apply consistent LR here for simplicity
        q_stage1_mf[chosen_a1] += lr_base * delta_stage1
        
        # Stage 2 update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        
        # Determine effective learning rate based on outcome valence
        if delta_stage2 < 0:
            # Negative PE (disappointment)
            effective_lr = lr_base + (anxiety_neg_bias * stai_score)
            # Clip to [0, 1]
            if effective_lr > 1.0: effective_lr = 1.0
        else:
            # Positive PE
            effective_lr = lr_base
            
        q_stage2_mf[state_idx, chosen_a2] += effective_lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Induced Model-Based Suppression
This model tests the "stress-induced habit" hypothesis. It posits that higher anxiety (STAI) consumes cognitive resources (working memory), thereby reducing the ability to perform Model-Based (planning) calculations. High anxiety participants will rely more on Model-Free (habitual) control.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Model-Based Suppression.
    
    Hypothesis: High anxiety consumes working memory resources required for 
    model-based planning. Therefore, STAI score reduces the mixing weight 'w'.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta: [0, 10] Inverse temperature.
    - w_intercept: [0, 1] The theoretical 'w' (MB-weight) at 0 anxiety.
    - anxiety_slope: [0, 1] The rate at which anxiety reduces 'w'.
    """
    learning_rate, beta, w_intercept, anxiety_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective w
    # w = w_intercept - (slope * anxiety)
    # We clip w to be between 0 and 1
    w_effective = w_intercept - (anxiety_slope * stai_score)
    if w_effective < 0: w_effective = 0.0
    if w_effective > 1: w_effective = 1.0
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Use the anxiety-adjusted w
        q_net = w_effective * q_stage1_mb + (1 - w_effective) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])
        chosen_a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        chosen_a2 = int(action_2[trial])
  
        # Update stage 1
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        # Update stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
