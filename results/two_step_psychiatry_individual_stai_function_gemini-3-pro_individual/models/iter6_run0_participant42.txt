Here are three new cognitive models implemented as Python functions, focusing on how anxiety (STAI) might modulate reward perception, memory decay, and stage-specific learning rates.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Subjective Reward Perception (Loss Aversion).
    
    Hypothesis: Highly anxious individuals may perceive the absence of a reward (0 outcomes) 
    not as a neutral event, but as a negative outcome (loss). This model transforms the 
    reward signal such that 0 becomes a negative value proportional to the STAI score.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature for softmax.
    - w: [0, 1] Weighting between Model-Based (1) and Model-Free (0).
    - loss_sens_param: [0, 5] Scaling factor for how anxiety turns 0-rewards into negative values.
    """
    learning_rate, beta, w, loss_sens_param = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate the subjective negative utility for a 0 reward based on anxiety
    # If STAI is high, 0 becomes more negative (e.g., -0.5). If STAI is 0, 0 remains 0.
    subjective_loss_value = -1.0 * (participant_stai * loss_sens_param)

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Stage 1 Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update with Subjective Reward
        # Determine effective reward
        r_actual = reward[trial]
        if r_actual < 0.001: # It is 0
            r_effective = subjective_loss_value
        else:
            r_effective = r_actual # It is 1
            
        delta_stage2 = r_effective - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Accelerated Memory Decay.
    
    Hypothesis: Anxiety consumes cognitive resources (working memory), leading to faster 
    forgetting of value expectations for options that were NOT chosen. 
    The Q-values of unchosen aliens decay toward 0 at a rate modulated by STAI.
    
    Parameters:
    - learning_rate: [0, 1] Update rate for chosen options.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF balance.
    - decay_base: [0, 1] Baseline decay rate for unchosen options.
    - anxiety_decay_mod: [0, 1] Additional decay added proportional to STAI.
    """
    learning_rate, beta, w, decay_base, anxiety_decay_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective decay rate, bounded at 1.0
    decay_rate = decay_base + (participant_stai * anxiety_decay_mod)
    if decay_rate > 1.0: decay_rate = 1.0

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # --- Decay Mechanism for Unchosen Option in Stage 2 ---
        # Identify the unchosen alien on the current planet
        unchosen_a2 = 1 - a2
        # Decay the unchosen value towards 0
        q_stage2_mf[state_idx, unchosen_a2] *= (1.0 - decay_rate)
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Modulated Stage-Specific Learning Rates.
    
    Hypothesis: Anxiety differentially impacts the updating of distal (Stage 1) versus 
    proximal (Stage 2) values. This model allows the Stage 2 learning rate (reaction to 
    immediate reward) to vary with anxiety, distinct from the Stage 1 learning rate.
    
    Parameters:
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] MB/MF balance.
    - lr_stage1: [0, 1] Fixed learning rate for the first stage choices.
    - lr_stage2_base: [0, 1] Baseline learning rate for second stage.
    - anxiety_lr2_mod: [-1, 1] Modulation of Stage 2 learning rate by STAI.
    """
    beta, w, lr_stage1, lr_stage2_base, anxiety_lr2_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    # Calculate effective learning rate for stage 2
    # We clip it to ensure it stays within [0, 1] valid range
    lr_stage2 = lr_stage2_base + (participant_stai * anxiety_lr2_mod)
    if lr_stage2 < 0: lr_stage2 = 0.0
    if lr_stage2 > 1: lr_stage2 = 1.0

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
  
        # --- Updates ---
        # Update Stage 1 using its specific learning rate
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Update Stage 2 using the anxiety-modulated learning rate
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_stage2 * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
