### Model 1: Anxiety-Modulated Subjective Transition Belief
This model posits that anxiety alters the **Model-Based** component by distorting the internal model of the world. Specifically, high anxiety may lead participants to over-simplify the transition structure, perceiving the "common" transition (usually 0.7) as more deterministic (closer to 1.0) to reduce perceived uncertainty.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 1: Anxiety-Modulated Subjective Transition Belief.
    
    Hypothesis: Anxiety distorts the Model-Based transition matrix. High anxiety leads 
    to an overestimation of the 'common' transition probability (ignoring rare transitions),
    effectively simplifying the world model to be more deterministic.
    
    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Weight for Model-Based control (0=MF, 1=MB).
    distortion_stai: [0, 1] How much STAI increases the perceived common transition probability.
    """
    learning_rate, beta, w, distortion_stai = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate subjective transition probability
    # Base is 0.7. STAI adds to this, capped at 1.0.
    # If distortion_stai is high, high anxiety people treat transitions as nearly deterministic.
    subjective_p = 0.7 + (distortion_stai * participant_stai * 0.3) 
    if subjective_p > 0.99: subjective_p = 0.99
    
    # Define the subjective transition matrix used for MB planning
    transition_matrix = np.array([[subjective_p, 1 - subjective_p], 
                                  [1 - subjective_p, subjective_p]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # MB value calculation uses the ANXIETY-MODIFIED transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        
        # --- Updates ---
        # Update Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Memory Decay
This model posits that anxiety interferes with working memory maintenance. It introduces a decay mechanism where Q-values for the second stage (aliens) decay toward a neutral value (0.5) over time. The rate of this decay is modulated by the participant's STAI score.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 2: Anxiety-Modulated Memory Decay.
    
    Hypothesis: Anxiety consumes cognitive resources, leading to faster forgetting 
    of learned values. Unvisited Stage 2 Q-values decay towards a neutral prior (0.5).
    The rate of decay is a function of STAI.
    
    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Weight for Model-Based control.
    decay_base: [0, 1] Baseline decay rate.
    decay_stai_mod: [0, 1] Additional decay scaling with STAI.
    """
    learning_rate, beta, w, decay_base, decay_stai_mod = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate decay rate bounded between 0 and 1
    decay_rate = decay_base + (decay_stai_mod * participant_stai)
    if decay_rate > 1.0: decay_rate = 1.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    # Initialize Stage 2 values at neutral 0.5 (since rewards are 0 or 1)
    q_stage2_mf = np.ones((2, 2)) * 0.5

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2
        
        # --- Anxiety-Modulated Decay ---
        # All Stage 2 values decay towards 0.5
        q_stage2_mf = q_stage2_mf * (1 - decay_rate) + 0.5 * decay_rate

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Modulated Eligibility Trace (Lambda)
This model posits that anxiety affects the credit assignment process via an eligibility trace ($\lambda$). High anxiety might lead to a reliance on the final outcome (high $\lambda$), reinforcing the Stage 1 choice directly based on the Stage 2 reward, effectively bypassing the two-step structure and ignoring the state transition. This is modeled using the TD($\lambda$) update rule where $\lambda$ is modulated by STAI.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model 3: Anxiety-Modulated Eligibility Trace (Lambda).
    
    Hypothesis: Anxiety modulates the eligibility trace (lambda). High anxiety 
    increases lambda, causing the agent to reinforce the first-stage choice 
    directly from the second-stage reward (ignoring the transition structure),
    making behavior more "Model-Free" in a direct sense.
    
    Parameters:
    learning_rate: [0, 1] Update rate for Q-values.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Weight for Model-Based control.
    lambda_base: [0, 1] Baseline eligibility trace.
    lambda_stai_sens: [0, 1] Sensitivity of lambda to STAI.
    """
    learning_rate, beta, w, lambda_base, lambda_stai_sens = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate Lambda
    lam = lambda_base + (lambda_stai_sens * participant_stai)
    if lam > 1.0: lam = 1.0
    if lam < 0.0: lam = 0.0
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        chosen_a2 = action_2[trial]
        
        # --- Updates ---
        # Calculate Prediction Errors
        delta_stage1 = q_stage2_mf[state_idx, chosen_a2] - q_stage1_mf[chosen_a1]
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, chosen_a2]
        
        # Update Stage 1 MF with Eligibility Trace
        # Q1 += alpha * (delta1 + lambda * delta2)
        q_stage1_mf[chosen_a1] += learning_rate * (delta_stage1 + lam * delta_stage2)
        
        # Update Stage 2 MF
        q_stage2_mf[state_idx, chosen_a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
