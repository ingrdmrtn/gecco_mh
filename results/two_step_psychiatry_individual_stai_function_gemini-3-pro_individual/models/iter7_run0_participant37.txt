Here are three new cognitive models for the two-step task, incorporating the STAI anxiety score into the learning and decision-making mechanisms.

### Model 1: Anxiety-Driven Memory Decay
This model hypothesizes that anxiety impairs the maintenance of learned values in working memory. The `forgetting_rate` parameter, scaled by the STAI score, determines how much the action values (Q-values) decay toward zero on each trial. High anxiety leads to faster "forgetting" of past associations, forcing the participant to rely more on recent outcomes.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model with Anxiety-Driven Memory Decay.
    
    Hypothesis: High anxiety impairs the maintenance of learned values over time.
    Action values (Q-values) decay toward 0 at a rate proportional to the
    participant's STAI score.
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Mixing weight (1=Pure MB, 0=Pure MF).
    stai_forgetting: [0, 1] Decay rate scaling factor.
                        (Effective decay = stai_forgetting * stai).
    """
    learning_rate, beta, w, stai_forgetting = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Effective decay rate bounded [0, 1]
    decay = min(max(stai_forgetting * stai_score, 0.0), 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        
        # Decay step: Anxiety causes values to rot before choice
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)

        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # Policy for the second choice
        a2 = int(action_2[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Dampened Reward Sensitivity
This model hypothesizes that anxiety is associated with anhedonia or a blunted response to positive outcomes. The `stai_dampening` parameter reduces the subjective value of the reward received. This reduced "effective reward" shrinks the prediction error (`delta`), causing the participant to learn more slowly from successes while still reacting normally to the lack of reward.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model with Anxiety-Dampened Reward Sensitivity.
    
    Hypothesis: Anxiety reduces the subjective valuation of rewards (blunted
    reward sensitivity). The effective reward used for learning is scaled down
    by the STAI score.
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Mixing weight (1=Pure MB, 0=Pure MF).
    stai_dampening: [0, 1] Factor reducing reward value.
                       (Effective Reward = Reward * (1 - dampening * stai)).
    """
    learning_rate, beta, w, stai_dampening = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate dampening factor, ensuring it doesn't flip reward sign
    damp_factor = min(max(stai_dampening * stai_score, 0.0), 0.99)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # Policy for the second choice
        a2 = int(action_2[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Modify reward perception based on anxiety
        effective_reward = reward[trial] * (1.0 - damp_factor)
        
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Modulated Eligibility (Outcome Focus)
This model tests if anxiety shifts Model-Free learning from state-value bootstrapping (TD-0) to direct outcome reinforcement (TD-1). The `stai_lambda` parameter acts as a mixing weight for the Stage 1 update target. High anxiety increases `lambda`, causing the participant to update their first-stage choice based on the final coin received (outcome focus) rather than the value of the planet reached (state-structure focus).

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid MB/MF model with Anxiety-Modulated Eligibility (Outcome Focus).
    
    Hypothesis: Anxiety shifts the Model-Free learning strategy from bootstrapping
    (TD-0, relying on the value of the 2nd stage state) to direct reinforcement
    from the final outcome (TD-1).
    
    Parameters:
    learning_rate: [0, 1] Standard learning rate.
    beta: [0, 10] Inverse temperature for softmax.
    w: [0, 1] Mixing weight (1=Pure MB, 0=Pure MF).
    stai_lambda: [0, 1] Eligibility trace scaler.
                    Lambda = stai_lambda * stai.
                    Lambda=0 -> Pure TD(0) (Standard template).
                    Lambda=1 -> Pure TD(1) (Direct reward update).
    """
    learning_rate, beta, w, stai_lambda_scale = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Lambda determines the mix between Q-value bootstrapping and direct reward
    lambda_val = min(max(stai_lambda_scale * stai_score, 0.0), 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # Policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # Policy for the second choice
        a2 = int(action_2[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]
        
        # Stage 1 Update: Mixed TD(0) and TD(1)
        # Target is weighted average of Next-State Value (Q2) and Final Reward (R)
        target_val = (lambda_val * reward[trial]) + \
                     ((1 - lambda_val) * q_stage2_mf[state_idx, a2])
                     
        delta_stage1 = target_val - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
