Here are 3 new cognitive models based on the two-step task and anxiety (STAI) literature.

### Model 1: Anxiety-Modulated Transition Belief (Fuzzy Map)
**Hypothesis:** High anxiety degrades the precision of the "cognitive map" used for Model-Based planning. While low-anxiety participants use the correct transition probabilities (70/30), high-anxiety participants perceive the transitions as more random (closer to 50/50), reducing the effectiveness of their model-based updates.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Transition Belief Model.
    
    Hypothesis: Anxiety acts as a 'blur' on the transition matrix used for 
    Model-Based planning. Higher STAI scores mix the true transition matrix 
    (70/30) with a uniform matrix (50/50), representing uncertainty about 
    the environment's structure.
    
    Parameters:
    learning_rate: [0, 1] - Rate of Q-value updating.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting between MB and MF values.
    map_blur_slope: [0, 2] - How much STAI degrades the transition map precision.
    """
    learning_rate, beta, w, map_blur_slope = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate the subjective transition matrix based on anxiety
    # True matrix
    true_trans = np.array([[0.7, 0.3], [0.3, 0.7]])
    # Uniform matrix (maximum uncertainty)
    flat_trans = np.array([[0.5, 0.5], [0.5, 0.5]])
    
    # Calculate blur factor (bounded between 0 and 1)
    blur = participant_stai * map_blur_slope
    if blur > 1.0: blur = 1.0
    if blur < 0.0: blur = 0.0
    
    # The subject's internal model of transitions
    subjective_trans = (1 - blur) * true_trans + (blur * flat_trans)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # USE SUBJECTIVE MATRIX HERE
        q_stage1_mb = subjective_trans @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Update Q-values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Volatility (Global Learning Rate)
**Hypothesis:** Anxious individuals perceive the world as more volatile and changeable. Therefore, higher STAI scores lead to a higher global learning rate, causing participants to over-update their values based on the most recent outcomes and fail to integrate long-term history effectively.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Volatility Model.
    
    Hypothesis: Anxiety correlates with a belief in environmental volatility.
    Higher STAI increases the global learning rate, making the participant 
    more reactive to recent outcomes (recency bias) across both stages.
    
    Parameters:
    lr_base: [0, 1] - Baseline learning rate.
    lr_stai_slope: [0, 1] - Increase in learning rate per unit of STAI.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Weighting between MB and MF values.
    """
    lr_base, lr_stai_slope, beta, w = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate effective learning rate based on anxiety
    learning_rate = lr_base + (participant_stai * lr_stai_slope)
    # Ensure bounds [0, 1]
    if learning_rate > 1.0: learning_rate = 1.0
    if learning_rate < 0.0: learning_rate = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Update Q-values using the anxiety-modulated learning rate
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]

    return log_loss
```

### Model 3: Anxiety-Induced Stage 2 Noise
**Hypothesis:** Anxiety specifically impacts the "moment of truth" (Stage 2) differently than the planning phase (Stage 1). High anxiety leads to "choking" or increased exploration/randomness specifically when choosing the alien, while the abstract spaceship choice remains more rational. This is modeled by a separate, anxiety-modulated `beta` for the second stage.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Stage 2 Noise Model.
    
    Hypothesis: Anxiety increases decision noise (lowers beta) specifically 
    at the second stage (choosing aliens), potentially due to immediate 
    reward pressure, while Stage 1 (planning) uses a baseline beta.
    
    Parameters:
    learning_rate: [0, 1] - Rate of Q-value updating.
    beta_1: [0, 10] - Inverse temperature for Stage 1 (Spaceships).
    beta_2_base: [0, 10] - Baseline inverse temperature for Stage 2.
    beta_2_stai_penalty: [0, 10] - Reduction in Stage 2 beta per unit of STAI.
    w: [0, 1] - Weighting between MB and MF values.
    """
    learning_rate, beta_1, beta_2_base, beta_2_stai_penalty, w = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
    
    # Calculate Stage 2 beta: Higher anxiety -> Lower beta (more random)
    beta_2 = beta_2_base - (participant_stai * beta_2_stai_penalty)
    if beta_2 < 0.0: beta_2 = 0.0 # Prevent negative beta
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # policy for the first choice (Uses beta_1)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        
        state_idx = int(state[trial])
        a1 = int(action_1[trial])

        # policy for the second choice (Uses anxiety-modulated beta_2)
        exp_q2 = np.exp(beta_2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a2 = int(action_2[trial])
        r = reward[trial]

        # Update Q-values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
