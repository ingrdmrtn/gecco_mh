Here are 3 new cognitive models that incorporate the STAI anxiety score into the decision-making process, distinct from the previously explored combinations.

### Model 1: Anxiety-Driven Reward Devaluation
**Hypothesis:** High anxiety (STAI) leads to a subjective devaluation of positive outcomes (anhedonia or pessimism). While the objective reward is 1 coin, highly anxious participants may perceive this as less valuable, effectively lowering the asymptote of their Q-values. This results in weaker reinforcement for good choices and a "flatter" value landscape, making behavior more prone to noise.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Reward Devaluation Model.
    
    Hypothesis: Anxiety dampens the subjective value of rewards.
    Effective Reward = Reward * (1 - STAI * devaluation_factor).
    
    Parameters:
    learning_rate: [0, 1] - Standard learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weighting.
    devaluation_scale: [0, 1] - Scaling factor for how much STAI reduces reward value.
    """
    learning_rate, beta, w, devaluation_scale = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
  
        # Calculate Effective Reward based on Anxiety
        # We clip to ensure it doesn't invert the reward (become negative)
        devaluation = participant_stai * devaluation_scale
        effective_reward = reward[trial] * (1.0 - min(devaluation, 0.99))

        # UPDATING
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])

        # Update Stage 1 (TD(0) style as implied by template structure)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 using Devalued Reward
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Plan Updating (Hyper-Correction)
**Hypothesis:** Anxiety specifically affects how aggressively participants update their *first-stage* plans (Model-Free Stage 1 values) based on outcomes, distinct from their second-stage learning. High anxiety leads to "Hyper-Correction" or over-updating of the initial choice value ($Q_{MF1}$), reflecting a tendency to over-attribute the final outcome to the initial decision.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Plan Updating (Hyper-Correction).
    
    Hypothesis: STAI increases the learning rate specifically for the 
    first-stage (planning) step, reflecting over-attribution of outcomes 
    to initial choices.
    
    Parameters:
    lr_base: [0, 1] - Base learning rate (used for Stage 2).
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weighting.
    lr_s1_stai_boost: [0, 1] - Additional learning rate added to Stage 1 updates based on STAI.
    """
    lr_base, beta, w, lr_s1_stai_boost = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate specific learning rate for Stage 1
    lr_stage1 = lr_base + (participant_stai * lr_s1_stai_boost)
    # Clamp to max 1.0
    if lr_stage1 > 1.0: lr_stage1 = 1.0

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # UPDATING
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
  
        # Update Stage 1 using STAI-boosted learning rate
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_stage1 * delta_stage1
        
        # Update Stage 2 using base learning rate
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr_base * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Distorted Transition Beliefs
**Hypothesis:** High anxiety degrades the accuracy of the internal model of the environment. Instead of believing the transition probability is 0.7 (as instructed/observed), anxious participants perceive the world as more volatile or uncertain. Their internal transition matrix shifts towards 0.5 (randomness) as STAI increases, degrading the quality of Model-Based control without explicitly changing the weighting parameter $w$.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Distorted Transition Beliefs.
    
    Hypothesis: Anxiety reduces the perceived reliability of the transition structure.
    The probability of the common transition (0.7) is reduced by STAI.
    P_common = 0.7 - (STAI * distortion).
    
    Parameters:
    learning_rate: [0, 1] - Standard learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - MB/MF weighting.
    trans_distortion: [0, 0.2] - How much STAI reduces the common transition probability.
    """
    learning_rate, beta, w, trans_distortion = model_parameters
    n_trials = len(action_1)
    participant_stai = stai[0]
  
    # Distort the transition matrix based on anxiety
    # Base is 0.7. distortion reduces this towards 0.5
    # Bound distortion to ensure prob >= 0.5
    distortion = participant_stai * trans_distortion
    p_common = 0.7 - distortion
    if p_common < 0.5: p_common = 0.5
    p_rare = 1.0 - p_common
    
    transition_matrix = np.array([[p_common, p_rare], [p_rare, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # MB calculation uses the DISTORTED matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        # UPDATING
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
  
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
