import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Win-Stay Bias (Reward-Dependent Stickiness).
    
    This model hypothesizes that anxiety increases reliance on a simple 
    "Win-Stay" heuristic. If the previous choice resulted in a reward, 
    anxiety adds a bonus to the value of repeating that choice, 
    bypassing complex Model-Based or Model-Free value calculations.
    
    Q_net(prev_action) += (base_wsls + sensitivity * stai) * prev_reward
    
    Parameters:
    learning_rate: [0, 1] Rate of Q-value updating.
    beta: [0, 10] Inverse temperature (softness of choice).
    w: [0, 1] Weighting of Model-Based (1) vs Model-Free (0).
    wsls_base: [0, 5] Baseline Win-Stay bonus.
    stai_wsls_sens: [0, 5] How much STAI amplifies the Win-Stay bonus.
    """
    learning_rate, beta, w, wsls_base, stai_wsls_sens = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Normalize STAI to roughly [0, 1] range (20-80 scale)
    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    
    wsls_bonus = wsls_base + (stai_wsls_sens * norm_stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1
    prev_reward = 0
    
    for trial in range(n_trials):
        # Stage 1 Policy
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Win-Stay bonus if previous trial was rewarded
        if prev_action_1 != -1 and prev_reward == 1.0:
            q_net[prev_action_1] += wsls_bonus
            
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Store for next trial
        prev_action_1 = a1
        
        # Stage 2 Policy
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        r = reward[trial]
        prev_reward = r # Store reward for next trial's heuristic
        
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # TD(0) update for Stage 1
        value_state_reached = q_stage2_mf[s_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Differential Learning Rates (Focus on Immediate Threat).
    
    This model hypothesizes that anxiety creates a disparity between how 
    participants learn about immediate outcomes (Stage 2 aliens) versus 
    antecedent choices (Stage 1 spaceships). High anxiety may increase 
    learning/volatility for the direct reward stage relative to the planning stage.
    
    lr_stage2 = lr_base * (1 + stai_s2_boost * normalized_stai)
    
    Parameters:
    lr_base: [0, 1] Base learning rate (applied to Stage 1).
    beta: [0, 10] Inverse temperature.
    w: [0, 1] Model-Based weight.
    stai_s2_boost: [0, 5] Multiplier for Stage 2 learning rate based on STAI.
    """
    lr_base, beta, w, stai_s2_boost = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    
    # Stage 1 LR is base; Stage 2 LR is boosted by anxiety
    lr_s1 = lr_base
    lr_s2 = lr_base * (1.0 + stai_s2_boost * norm_stai)
    lr_s2 = np.clip(lr_s2, 0, 1) # Ensure valid bound

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    for trial in range(n_trials):
        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Choice
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Updates
        r = reward[trial]
        
        # Update Stage 2 with boosted rate
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += lr_s2 * delta_stage2
        
        # Update Stage 1 with base rate
        value_state_reached = q_stage2_mf[s_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Rare Transition Reactivity (MB Destabilization).
    
    This model proposes that anxiety causes a temporary loss of confidence 
    in the Model-Based map following a "Rare" transition. If the previous 
    trial involved a rare transition, the weight 'w' is dampened for the 
    current trial, proportional to anxiety level.
    
    If prev_rare: w_eff = w_base * (1 - dampening * normalized_stai)
    
    Parameters:
    learning_rate: [0, 1] Learning rate.
    beta: [0, 10] Inverse temperature.
    w_base: [0, 1] Baseline Model-Based weight.
    stai_rare_dampening: [0, 1] How much STAI reduces 'w' after a rare transition.
    """
    learning_rate, beta, w_base, stai_rare_dampening = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_transition_rare = False
    
    for trial in range(n_trials):
        # Determine effective w based on previous transition type
        w_eff = w_base
        if prev_transition_rare:
            damp_factor = stai_rare_dampening * norm_stai
            w_eff = w_base * (1.0 - damp_factor)
            w_eff = max(0.0, w_eff) # Ensure non-negative

        # Stage 1 Choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = w_eff * q_stage1_mb + (1 - w_eff) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Stage 2 Choice
        s_idx = int(state[trial])
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # Determine if this transition was rare
        # Common transitions: (Action 0 -> State 0) or (Action 1 -> State 1)
        # Rare transitions: (Action 0 -> State 1) or (Action 1 -> State 0)
        is_rare = (a1 != s_idx)
        prev_transition_rare = is_rare

        # Updates
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        value_state_reached = q_stage2_mf[s_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
