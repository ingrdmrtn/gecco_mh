### Model 1: STAI-modulated Passive Decay
This model hypothesizes that anxiety consumes cognitive resources (e.g., working memory), leading to faster "forgetting" of learned values for options that are not currently being attended to.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    STAI-modulated Passive Decay Model.
    
    Hypothesis: Anxiety (STAI) accelerates the passive decay (forgetting) of 
    Q-values for unchosen options. High anxiety individuals may struggle to 
    maintain value representations over time due to cognitive interference.
    
    Parameters:
    learning_rate: [0,1] - Update rate for chosen options.
    beta: [0,10] - Inverse temperature.
    w: [0,1] - Mixing weight (0=MF, 1=MB).
    decay_slope: [0,1] - Sensitivity of decay rate to STAI.
                         decay_rate = decay_slope * STAI.
                         Unchosen Q-values are multiplied by (1 - decay_rate).
    """
    learning_rate, beta, w, decay_slope = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    # Calculate decay rate based on anxiety, capped at 1.0
    decay_rate = decay_slope * stai_val
    if decay_rate > 1.0: decay_rate = 1.0
    if decay_rate < 0.0: decay_rate = 0.0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Action value updating
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # Passive decay for unchosen options
        # Decay unchosen stage 1 option
        q_stage1_mf[1 - a1] *= (1.0 - decay_rate)
        
        # Decay unchosen stage 2 option in the visited state
        q_stage2_mf[state_idx, 1 - a2] *= (1.0 - decay_rate)
        
        # Optionally decay options in the unvisited state as well (common in some decay models)
        q_stage2_mf[1 - state_idx, :] *= (1.0 - decay_rate)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: STAI-modulated Pessimism Bias
This model hypothesizes that anxiety adds a negative bias to the perception of outcomes, effectively lowering the subjective value of all rewards.

```python
import numpy as np

    STAI-modulated Pessimism Bias Model.
    
    Hypothesis: Anxiety (STAI) creates a generalized pessimistic bias. 
    Subjective rewards are calculated as the objective reward minus a bias term 
    proportional to STAI. This shifts the baseline of evaluation downwards.
    
    Parameters:
    learning_rate: [0,1]
    beta: [0,10]
    w: [0,1]
    pessimism_slope: [0,1] - Magnitude of the negative bias scaling with STAI.

    p_choice_2 = np.zeros(n_trials)

        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]
        
        # Apply Pessimism Bias to the reward
        # Even a neutral outcome (0) becomes negative for high anxiety
        r_effective = r - pessimism_bias

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Use effective reward for stage 2 update
        delta_stage2 = r_effective - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: STAI-modulated Surprise Learning
This model hypothesizes that anxiety involves hyper-vigilance to unexpected events. When a "rare" transition occurs, anxious participants increase their learning rate, potentially over-reacting to what might be noise.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    STAI-modulated Surprise Learning Model.
    
    Hypothesis: Anxiety (STAI) modulates the learning rate specifically on trials 
    where a rare transition occurs. High anxiety leads to hyper-vigilance, 
    increasing the learning rate (alpha) when the transition was unexpected.
    
    Parameters:
    learning_rate: [0,1] - Base learning rate for common transitions.
    beta: [0,10]
    w: [0,1]
    rare_lr_boost: [0,1] - Additional learning rate added on rare trials, scaled by STAI.
                           alpha_rare = learning_rate + (rare_lr_boost * STAI).
    """
    learning_rate, beta, w, rare_lr_boost = model_parameters
    n_trials = len(action_1)
    stai_val = stai[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[int(action_1[trial])]
        state_idx = int(state[trial])

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[int(action_2[trial])]
        
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # Determine if transition was rare
        # Common: 0->0, 1->1. Rare: 0->1, 1->0.
        is_rare = (a1 != state_idx)
        
        # Calculate dynamic learning rate
        if is_rare:
            current_lr = learning_rate + (rare_lr_boost * stai_val)
            # Clip to valid range [0, 1]
            if current_lr > 1.0: current_lr = 1.0
        else:
            current_lr = learning_rate

        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += current_lr * delta_stage1
        
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += current_lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
