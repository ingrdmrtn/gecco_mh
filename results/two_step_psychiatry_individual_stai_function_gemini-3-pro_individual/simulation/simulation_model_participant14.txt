import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid model where STAI modulates choice perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, p_base, p_stai]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the two states (4 arms total).
        stai (float or None): STAI score for the participant. Defaults to 0 if None.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, p_base, p_stai = parameters
    
    # Handle STAI input safely
    stai_score = stai if stai is not None else 0.0
    
    # Calculate effective perseveration based on STAI
    p_eff = p_base + (p_stai * stai_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    # q_stage1_mf: Q-values for first stage (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for second stage (2 states x 2 actions)
    # Note: In the fitting code, states were indexed 0 and 1.
    q_stage2_mf = np.zeros((2, 2))

    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # Current reward probabilities for the 4 arms
        # State 0 corresponds to drift1, drift2
        # State 1 corresponds to drift3, drift4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values (Weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add Perseveration Bonus
        q_net_pers = q_net.copy()
        if prev_action_1 != -1:
            q_net_pers[int(prev_action_1)] += p_eff
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_pers)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 (pure MF)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < current_drifts[s2][a2])
        
        # --- Learning Updates ---
        
        # Stage 1 TD error (using Q-value of chosen stage 2 state/action as target)
        # Note: The fitting code uses SARSA-like update for stage 1: Q(s1, a1) <- Q(s1, a1) + alpha * (Q(s2, a2) - Q(s1, a1))
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 TD error
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update history
        prev_action_1 = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
