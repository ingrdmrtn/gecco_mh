import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid Model with Anxiety-Modulated Choice Stickiness.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stai_stick_sensitivity]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, stick_base, stai_stick_sensitivity = parameters

    # Handle STAI calculation
    # Assuming stai is passed as a float based on the function signature, 
    # but handling the case where it might be wrapped in a list like the fitting code
    if isinstance(stai, (list, np.ndarray)):
        stai_val = stai[0]
    else:
        stai_val = stai if stai is not None else 20.0 # Default to min if None

    norm_stai = (stai_val - 20) / 60.0
    norm_stai = np.clip(norm_stai, 0, 1)
    stickiness = stick_base + (stai_stick_sensitivity * norm_stai)

    # Initialize structures
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State 0 (row 0) and State 1 (row 1)

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_action_1 = -1

    for t in range(n_trials):
        # 1. Calculate Stage 1 Q-values
        # Model-Based Component
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Stickiness
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness
            
        # 2. Stage 1 Choice
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # State 0 or 1 based on transition probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Choice
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 5. Reward Generation
        # Map drift inputs to state/action pairs
        # State 0: drift1 (action 0), drift2 (action 1)
        # State 1: drift3 (action 0), drift4 (action 1)
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < current_drifts[s2][a2])
        
        # 6. Learning Updates
        # Note: The fitting code updates Stage 2 first, then uses the updated value for Stage 1
        
        # Update Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1
        # Uses the updated value of the specific state-action pair reached
        value_state_reached = q_stage2_mf[s2, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 7. Recording and Bookkeeping
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
