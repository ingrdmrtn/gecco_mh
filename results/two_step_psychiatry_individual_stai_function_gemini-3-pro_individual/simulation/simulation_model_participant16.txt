import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates Model 1: Anxiety-Modulated Perseveration.
    
    Hypothesis: Higher anxiety leads to higher choice perseveration (stickiness) 
    at the first stage.
    
    Parameters:
    - parameters: [learning_rate, beta, w, persev_base, anxiety_persev]
    """
    learning_rate, beta, w, persev_base, anxiety_persev = parameters
    
    # Handle STAI input: if it's an array/list, take the first element, otherwise use as is.
    # Default to 0 if None provided (though the prompt implies it will be provided).
    if stai is None:
        stai_score = 0
    elif hasattr(stai, '__len__'):
        stai_score = stai[0]
    else:
        stai_score = stai

    # Calculate the composite perseveration strength
    perseveration_strength = persev_base + (anxiety_persev * stai_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (0 and 1)
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf: Model-free values for stage 2 states (0 and 1) and actions (0 and 1)
    q_stage2_mf = np.zeros((2, 2)) 

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values (hybrid model)
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)

        # Apply perseveration bonus
        q_net_persev = q_net.copy()
        if prev_action_1 != -1:
            q_net_persev[int(prev_action_1)] += perseveration_strength
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_persev)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        prev_action_1 = a1 # Update for next trial

        # --- Transition ---
        # Determine second stage state based on transition matrix probabilities
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        state_idx = s2 # 0 or 1

        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 (pure model-free)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward Generation ---
        # Map state index to the correct drift arrays
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        if state_idx == 0:
            current_reward_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_reward_prob = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_reward_prob)

        # --- Learning Updates ---
        
        # Stage 2 Update (TD Error)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # Stage 1 Update (TD Error using updated Stage 2 value as target)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
