import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates Model 1: Anxiety-Modulated Choice Stickiness.
    
    Logic:
    - Anxiety (stai) drives perseveration (stickiness).
    - The participant receives an additional 'bonus' to the action value of the choice 
      made in the previous trial, scaled by their anxiety level (stai * stickiness_sensitivity).
    - Uses a standard hybrid Model-Based/Model-Free architecture.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_sensitivity]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, w, stickiness_sensitivity = parameters
    
    # Handle STAI input: if it's a list/array, take the first element, otherwise use as is.
    # Default to 0 if None provided (though model logic implies it's required).
    if stai is None:
        stai_score = 0
    elif hasattr(stai, '__len__'):
        stai_score = stai[0]
    else:
        stai_score = stai

    stickiness_bonus = stickiness_sensitivity * stai_score
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    # q_stage1_mf: shape (2,) for stage 1 actions
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: shape (2, 2) for stage 2 states and actions
    q_stage2_mf = np.zeros((2, 2))
    
    prev_action_1 = -1 # No previous action on trial 0
    
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 (index 0) has options drift1, drift2
        # State 1 (index 1) has options drift3, drift4
        # Note: drift arrays are usually 0-indexed corresponding to trial t
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        logits = beta * q_net
        
        # Apply anxiety-modulated stickiness
        if prev_action_1 != -1:
            logits[prev_action_1] += stickiness_bonus
            
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State is 0 or 1 (mapped to indices 0 and 1 for Q-table)
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard softmax on MF values for the specific state
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning ---
        # Update Stage 2 Q-values (SARSA / TD(0))
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (TD(1) logic used in fitting code: Q1 updates towards Q2)
        # In fitting: delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        # Note: Using the *updated* Q2 value as the target, consistent with the fitting code provided.
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store history
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
