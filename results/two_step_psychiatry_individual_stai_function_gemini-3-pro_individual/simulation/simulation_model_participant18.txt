import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates the Hybrid MB/MF model with STAI-modulated Choice Stickiness.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stick_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant. If None, assumes 0 or handles accordingly.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w, stick_base, stick_mod = parameters
    
    # Handle STAI input (if provided as a list/array in fitting, usually a scalar here)
    stai_score = stai if stai is not None else 0.0
    
    # Calculate effective stickiness based on STAI
    stickiness = stick_base + (stick_mod * stai_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    # q_stage1_mf: Model-free values for Stage 1 actions (0 and 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for Stage 2 states (0 and 1) and actions (0 and 1)
    q_stage2_mf = np.zeros((2, 2))

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_action_1 = -1 # Initialize as -1 (no previous choice)

    for t in range(n_trials):
        # Current reward probabilities for the four terminal states
        # State 0 corresponds to drifts 1 & 2, State 1 corresponds to drifts 3 & 4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate MB values: Transition matrix * max Q-value of next stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness (perseveration) to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness

        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine Stage 2 state based on transition matrix
        # Note: In fitting code provided, state is just passed in. Here we simulate it.
        # transition_matrix[a1] gives [p(s2=0|a1), p(s2=1|a1)]
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax selection for Stage 2 (pure MF based on fitting code logic)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        
        # TD(0) update for Stage 1 (SARSA logic based on fitting code)
        # Note: Fitting code uses: delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
