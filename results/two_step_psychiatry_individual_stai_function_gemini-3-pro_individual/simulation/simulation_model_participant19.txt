import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with STAI-modulated choice perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stick_stai]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 terminal states.
        stai (float or list): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_base, stick_stai = parameters
    
    # Handle STAI input (robust to scalar or list/array input)
    if np.ndim(stai) > 0:
        stai_val = stai[0]
    else:
        stai_val = stai

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize model variables
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # shape: (state, action)
    
    stickiness_bonus = stick_base + (stick_stai * stai_val)
    last_action = -1

    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply Stickiness (Perseveration)
        if last_action != -1:
            q_net[last_action] += stickiness_bonus
            
        # Softmax selection Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # 3. Stage 2 Decision
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # 4. Reward Generation
        # Map drifts to states/actions:
        # State 0: Action 0 -> drift1, Action 1 -> drift2
        # State 1: Action 0 -> drift3, Action 1 -> drift4
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < p_reward else 0
        
        # 5. Learning Updates
        
        # Stage 1 MF Update (using Q-value of chosen stage 2 action, pure TD chain)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # 6. Update history and storage
        last_action = a1
        
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
