import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates the Anxiety-Induced Choice Stickiness (Perseveration) Model.
    
    Logic:
    - Hybrid Model-Based/Model-Free reinforcement learning.
    - Anxiety (STAI) modulates a stickiness bonus applied to the previously chosen first-stage action.
    - Stickiness bonus = base_stickiness * stai.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, base_stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.
        stai (float or None): STAI score for the participant. Defaults to 0 if None.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Extract parameters
    learning_rate, beta, w, base_stickiness = parameters
    
    # Handle STAI input safely
    stai_val = stai if stai is not None else 0.0

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Model initialization
    # Transition matrix: P(State 1 | Action 0) = 0.7, P(State 2 | Action 0) = 0.3, etc.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization (0.5 is standard prior for prob 0-1)
    q_stage1_mf = np.ones(2) * 0.5
    q_stage2_mf = np.ones((2, 2)) * 0.5
    
    prev_a1 = -1 # No previous choice on first trial
    stickiness_bonus = base_stickiness * stai_val
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for current trial: [[s1_a0, s1_a1], [s2_a0, s2_a1]]
        # Note: drift1/2 correspond to state 0 (often called State 2 in literature), 
        # drift3/4 correspond to state 1 (often called State 3).
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value calculation (Hybrid MB/MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Anxiety-Induced Stickiness
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness_bonus

        # Softmax selection
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Transition to Stage 2
        # transition_matrix[a1] gives [prob_state0, prob_state1]
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_drifts[state_idx][a2])
        
        # --- Learning (Update Q-values) ---
        
        # Second stage update (TD error)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
        
        # First stage update
        # 1. Update based on prediction of second stage value
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # 2. Update based on second stage outcome (eligibility trace equivalent)
        q_stage1_mf[a1] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous action for stickiness in next trial
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
