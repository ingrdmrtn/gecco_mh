import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using Model 1: Anxiety-Driven Perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, perseveration_base, stai_perseveration_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    lr, beta, w, perseveration_base, stai_perseveration_mod = parameters
    
    # Handle STAI input
    stai_val = stai if stai is not None else 0.0
    
    # Calculate the fixed perseveration bonus based on anxiety
    perseveration_bonus = perseveration_base + (stai_perseveration_mod * stai_val)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # Stage 1: 2 actions
    q_stage1_mf = np.zeros(2) 
    # Stage 2: 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2)) 

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_choice_1 = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Value Calculation (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Perseveration (stickiness)
        if prev_choice_1 != -1:
            q_net[prev_choice_1] += perseveration_bonus

        # Softmax Choice Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition ---
        
        # Determine next state based on transition matrix probabilities for chosen action
        # a1=0 -> [0.7, 0.3], a1=1 -> [0.3, 0.7]
        s2 = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax Choice Stage 2 (using same beta as fitting code)
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward ---
        
        # Get reward probability for the specific terminal state/action
        # State 0 uses drift1/drift2, State 1 uses drift3/drift4
        current_drifts = [drift1[t], drift2[t]] if s2 == 0 else [drift3[t], drift4[t]]
        prob_reward = current_drifts[a2]
        
        r = 1 if np.random.random() < prob_reward else 0

        # --- Learning Updates ---
        
        # Update Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += lr * delta_stage2
        
        # Update Stage 1 MF values
        # Note: Uses the updated stage 2 value, consistent with fitting code logic
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1

        # Update history
        prev_choice_1 = a1

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
