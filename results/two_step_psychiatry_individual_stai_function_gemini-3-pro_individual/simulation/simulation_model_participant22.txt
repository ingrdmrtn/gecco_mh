import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates Model 1: Anxiety-Modulated Transition Uncertainty.
    
    This simulation reflects the logic where STAI score degrades the perceived 
    transition probability used in the Model-Based calculation, effectively 
    making the MB system less certain about state transitions.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness, trans_deg_stai]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness, trans_deg_stai = parameters
    
    # Handle STAI input
    stai_score = stai if stai is not None else 0.0

    # Calculate Perceived Transition Matrix based on STAI
    # The objective probability is 0.7, but anxiety reduces this towards 0.5
    perceived_common_p = 0.7 - (trans_deg_stai * stai_score)
    if perceived_common_p < 0.5:
        perceived_common_p = 0.5
        
    perceived_transition_matrix = np.array([[perceived_common_p, 1-perceived_common_p], 
                                            [1-perceived_common_p, perceived_common_p]])
    
    # True objective transition matrix for environment simulation
    true_transition_matrix = np.array([[0.7, 0.3], 
                                       [0.3, 0.7]])

    # Initialize Q-values
    # q_stage1_mf: Q-values for Stage 1 actions (Model-Free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for Stage 2 states (0, 1) and actions (0, 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward_history = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for trial in range(n_trials):
        # 1. Stage 1 Decision
        
        # Model-Based Calculation
        # MB values are derived from max values of Stage 2 MF Q-values, 
        # weighted by the *perceived* transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = perceived_transition_matrix @ max_q_stage2
        
        # Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness (Choice Perseveration)
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Action Selection (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1
        
        # 2. Transition to Stage 2
        # Use TRUE transition matrix for actual environment dynamics
        s2_idx = rng.choice([0, 1], p=true_transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Standard Model-Free Q-learning for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Reward Outcome
        # Get current trial reward probabilities
        current_reward_probs = [
            [drift1[trial], drift2[trial]],  # State 0 (index 0)
            [drift3[trial], drift4[trial]]   # State 1 (index 1)
        ]
        
        r = int(rng.random() < current_reward_probs[s2_idx][a2])
        
        # 5. Learning Updates (SARSA logic as implied by fitting code)
        
        # Stage 1 Update (TD(0))
        # Update Stage 1 MF Q-value using Stage 2 MF Q-value as the target
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Stage 2 Update
        # Update Stage 2 MF Q-value using received reward
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[trial] = a1
        state2[trial] = s2_idx
        stage2_choice[trial] = a2
        reward_history[trial] = r

    return stage1_choice, state2, stage2_choice, reward_history
