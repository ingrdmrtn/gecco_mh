import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Model-Based/Model-Free Hybrid 
    with STAI-modulated Transition Learning.

    Hypothesis: High anxiety (STAI) leads to instability in the internal model of 
    the task structure by increasing the transition matrix learning rate.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, lr_trans_base, stai_trans_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, lr_trans_base, stai_trans_slope = parameters
    
    # Handle STAI input
    current_stai = stai if stai is not None else 0.0

    # Calculate transition learning rate based on STAI
    lr_trans = min(1.0, lr_trans_base + (stai_trans_slope * current_stai))

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize model variables
    # The fitting code initializes trans_probs as 0.5s (flat prior)
    trans_probs = np.ones((2, 2)) * 0.5 
    
    # Q-values: Stage 1 has 2 actions, Stage 2 has 2 states * 2 actions
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # True transition matrix for generating the environment states (standard 70/30 split)
    # Note: While the agent *learns* the transition matrix, the *environment* usually 
    # follows a fixed stochastic rule (e.g., common vs rare transitions).
    true_transition_matrix = np.array([[0.7, 0.3],
                                       [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Stage 1 Decision
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = trans_probs @ max_q_stage2

        # Net Q-values (hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 2. Transition to Stage 2
        # Use the true environment dynamics to determine the next state
        s2 = rng.choice([0, 1], p=true_transition_matrix[a1])
        
        # 3. Stage 2 Decision
        # Softmax selection for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 4. Outcome
        # Get reward probability for the current state/action pair from drifts
        # Mapping: s2=0 -> (drift1, drift2), s2=1 -> (drift3, drift4)
        if s2 == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < p_reward)
        
        # 5. Learning / Updates
        
        # Update internal transition model (agent's belief)
        state_onehot = np.zeros(2)
        state_onehot[s2] = 1.0
        trans_probs[a1] += lr_trans * (state_onehot - trans_probs[a1])
        # Normalize (though the math above keeps it normalized if started normalized)
        trans_probs[a1] /= np.sum(trans_probs[a1])

        # Update Q-values (TD(1) style as per fitting code)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        delta_stage2 = r - q_stage2_mf[s2, a2]

        # Note: The fitting code updates stage 1 with both prediction errors
        q_stage1_mf[a1] += learning_rate * delta_stage1 + learning_rate * delta_stage2
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
