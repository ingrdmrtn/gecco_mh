import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Anxiety-Modulated Learning Rate Asymmetry Model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg_base, beta, w, stai_neg_sensitivity]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    alpha_pos, alpha_neg_base, beta, w, stai_neg_sensitivity = parameters
    
    # Handle STAI input (default to 0 if None)
    stai_val = stai if stai is not None else 0.0
    
    # Calculate anxiety-modulated negative learning rate
    alpha_neg = np.clip(alpha_neg_base + (stai_neg_sensitivity * stai_val), 0.0, 1.0)

    # Initialize Q-values and structures
    # Stage 1 MF values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Value (Hybrid)
        q_net_stage1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Softmax Choice Stage 1
        exp_q1 = np.exp(beta * q_net_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix probabilities for chosen action a1
        state_probs = transition_matrix[a1]
        state_idx = rng.choice([0, 1], p=state_probs)
        
        # --- Stage 2 Decision ---
        
        # Softmax Choice Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Generation ---
        
        # Map state/action to specific drift probability
        if state_idx == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # state_idx == 1
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if rng.random() < prob_reward else 0
        
        # --- Learning Updates ---
        
        # 1. Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        
        if delta_stage2 >= 0:
            q_stage2_mf[state_idx, a2] += alpha_pos * delta_stage2
        else:
            q_stage2_mf[state_idx, a2] += alpha_neg * delta_stage2

        # 2. Update Stage 1 Q-values
        # Note: Fitting code uses the *updated* q_stage2_mf value for the stage 1 prediction error
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        
        if delta_stage1 >= 0:
            q_stage1_mf[a1] += alpha_pos * delta_stage1
        else:
            q_stage1_mf[a1] += alpha_neg * delta_stage1
            
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
