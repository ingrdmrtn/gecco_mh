import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the STAI-Modulated Choice Stickiness model.
    
    Anxiety (STAI) modulates the tendency to repeat the previous Stage 1 choice 
    regardless of the outcome.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stai_stick_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, w, stick_base, stai_stick_mod = parameters
    
    # Handle STAI input safely
    participant_stai = stai if stai is not None else 0.0
    
    # Calculate the fixed stickiness value based on STAI
    stickiness = stick_base + (stai_stick_mod * participant_stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free Q-values for stage 1 actions (2 actions)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free Q-values for stage 2 states and actions (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_action_1 = -1

    for t in range(n_trials):
        # Organize drift probabilities for easy indexing: reward_probs[state_idx][action]
        # Note: In the fitting code, state is 0 or 1. Here we map drift inputs to those states.
        # usually drift1/2 are for state 0, drift3/4 for state 1.
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Add stickiness bonus if there was a previous action
        if prev_action_1 != -1:
            q_net[prev_action_1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine Stage 2 state based on transition matrix
        s2_state = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 (purely model-free based on Q_stage2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward Outcome ---
        r = int(rng.random() < reward_probs[s2_state][a2])
        
        # --- Learning (Update Q-values) ---
        
        # TD Update for Stage 1 Q-values (SARSA-like logic using Q_stage2 of chosen action)
        delta_stage1 = q_stage2_mf[s2_state, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD Update for Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[s2_state, a2]
        q_stage2_mf[s2_state, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_state
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
