import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid Model-Based/Model-Free model
    where the weighting parameter (omega) is modulated by STAI (anxiety).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant. If None, defaults to 0.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, w_scale = parameters
    
    # Handle STAI input: if it's an array/list, take first element, otherwise use float value
    if stai is None:
        stai_val = 0.0
    elif hasattr(stai, '__len__') and not isinstance(stai, str):
        stai_val = stai[0]
    else:
        stai_val = float(stai)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    q_stage1_mf = np.zeros(2)       # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2))  # Model-free values for stage 2 (2 states, 2 actions)

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Calculate omega based on STAI scaling
    # Logic from fitting model: omega = w_scale * (1.0 - stai_val)
    omega = w_scale * (1.0 - stai_val) 

    for t in range(n_trials):
        # Drift reward probabilities for the current trial
        # State 0 (drift1, drift2), State 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Integrated Q-value (Hybrid MB/MF)
        q_integrated = (omega * q_stage1_mb) + ((1 - omega) * q_stage1_mf)

        exp_q1 = np.exp(beta * q_integrated)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Note: In the fitting code, `state` is 0 or 1.
        # q_stage2_mf is shape (2, 2) -> [state_idx, action_idx]
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])

        # --- Learning Updates ---
        # 1. Update Stage 2 Q-values based on reward prediction error
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # 2. Update Stage 1 Q-values (TD(0))
        # Note: The fitting code uses q_stage2_mf[state_idx, action_2] as the target for stage 1 update
        # This is strictly TD(0) rather than SARSA(lambda) or eligibility traces
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
