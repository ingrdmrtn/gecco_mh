import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid Model-Based/Model-Free RL model
    with STAI-modulated Choice Perseverance (Stickiness).
    
    Hypothesis: Higher anxiety increases the tendency to repeat the previous Stage 1 choice 
    (perseverance), regardless of the outcome, as a safety/rigidity mechanism.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [lr, beta, w, stick_base, stick_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    lr, beta, w, stick_base, stick_sens = parameters
    
    # Handle STAI input safely
    stai_val = stai if stai is not None else 0.0
    
    # Calculate the stickiness bonus based on STAI
    stickiness_bonus = stick_base + (stick_sens * stai_val)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (model-free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (model-free)
    # Rows: State (0 or 1), Cols: Action (0 or 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Track previous choice for stickiness
    prev_a1 = -1
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        
        # --- Stage 1 Decision ---
        
        # Model-Based calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add perseverance bonus if applicable
        if prev_a1 != -1:
            q_net[prev_a1] += stickiness_bonus
            
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State index is 0 or 1 (unlike the example which used 1-based indexing for q_mf)
        state_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Simple Model-Free decision at Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        r = int(rng.random() < reward_probs[state_idx][a2])
        
        # --- Learning ---
        
        # Stage 2 Update (TD Error)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += lr * delta_stage2
        
        # Stage 1 Update (TD Error based on value of state reached)
        # Note: The fitting code uses 'value_state_reached' (Q(s',a')) 
        # rather than just the reward or max Q(s').
        value_state_reached = q_stage2_mf[state_idx, a2]
        delta_stage1 = value_state_reached - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr * delta_stage1
        
        # Update history
        prev_a1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = state_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
