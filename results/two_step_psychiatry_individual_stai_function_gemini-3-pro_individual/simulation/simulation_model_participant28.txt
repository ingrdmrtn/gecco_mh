import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with Anxiety-Driven Memory Decay.
    
    Logic:
    - Standard Hybrid MB/MF architecture.
    - Anxiety (STAI) causes a passive decay of learned Q-values towards zero at the end of each trial.
    - The decay rate is proportional to the STAI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, decay_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, w, decay_scale = parameters
    
    # Handle STAI input. The fitting code assumes stai is a list/array, but simulation usually passes a float.
    # We treat it as a scalar here. If None, default to 0 (no decay effect).
    stai_val = stai if stai is not None else 0.0

    # Calculate decay rate based on STAI
    decay_rate = np.clip(decay_scale * stai_val, 0.0, 1.0)
    retention = 1.0 - decay_rate

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (0, 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for stage 2 states (0, 1) and actions (0, 1)
    q_stage2_mf = np.zeros((2, 2))

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values: weighted sum of MB and MF
        q_net_s1 = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice
        exp_q1 = np.exp(beta * q_net_s1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        qs_stage2 = q_stage2_mf[s_idx]
        exp_q2 = np.exp(beta * qs_stage2)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Outcome ---
        r = int(rng.random() < reward_probs[s_idx][a2])

        # --- Learning ---
        # Update Stage 2 Q-values (TD Error)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-values (TD Error using updated Stage 2 value as target)
        # Note: The fitting code uses q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # --- Memory Decay ---
        # Apply retention factor to all Q-values
        q_stage1_mf *= retention
        q_stage2_mf *= retention

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
