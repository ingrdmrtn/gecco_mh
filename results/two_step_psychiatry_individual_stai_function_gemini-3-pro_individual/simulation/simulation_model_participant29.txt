import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4, stai):
    """
    Simulates the Hybrid MB/MF model with Anxiety-modulated Perseveration (Stickiness).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stai_sticky_scale]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or list): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stai_sticky_scale = parameters
    
    # Handle STAI input (if it's a list/array, take the first element, otherwise treat as float)
    if isinstance(stai, (list, np.ndarray)):
        stai_score = stai[0]
    else:
        stai_score = stai

    # Calculate the anxiety-modulated stickiness bonus
    stickiness_bonus = stai_score * stai_sticky_scale

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the first stage actions (MF)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the second stage states (rows) and actions (cols)
    q_stage2_mf = np.zeros((2, 2))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action_1 = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Model-Based Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Stickiness Bonus
        q_decision = q_net.copy()
        if last_action_1 != -1:
            q_decision[last_action_1] += stickiness_bonus

        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_decision)
        p_choice_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = rng.choice([0, 1], p=p_choice_1)

        # --- Transition ---
        # Determine state 2 based on transition matrix probabilities
        s_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # --- Stage 2 Decision ---
        # Softmax for Stage 2 (purely MF based on stage 2 Q-values)
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        p_choice_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = rng.choice([0, 1], p=p_choice_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s_idx][a2])

        # --- Learning / Updating ---
        # SARSA / TD(1) style updates as per the fitting code logic
        
        # 1. Update Stage 1 MF Q-value based on Stage 2 Q-value (TD error 1)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # 2. Update Stage 2 MF Q-value based on Reward (TD error 2)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2

        # 3. Update Stage 1 MF Q-value again based on Stage 2 prediction error (Eligibility trace logic)
        q_stage1_mf[a1] += learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = a1
        state2[t] = s_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
