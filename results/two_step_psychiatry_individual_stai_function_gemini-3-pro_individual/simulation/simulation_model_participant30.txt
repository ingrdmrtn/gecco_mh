import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates the Hybrid MB/MF model with STAI-modulated choice perseverance.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_factor]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float): STAI score for the participant (required for this model).

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stickiness_factor = parameters
    
    # Handle STAI input (default to 0 if None, though model logic implies it's needed)
    stai_score = stai if stai is not None else 0.0
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to Q(s1, a1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q(s2, a2) where s2 is index 0 or 1
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Calculate effective stickiness based on STAI
    effective_stickiness = stickiness_factor * stai_score
    
    prev_a1 = -1 # Initialize previous action placeholder

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 (index 0) has options with prob drift1, drift2
        # State 1 (index 1) has options with prob drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Model-Based Value Calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value mixing MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add STAI-modulated stickiness to the previous action
        if prev_a1 != -1:
            q_net[prev_a1] += effective_stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        prev_a1 = a1
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State index 0 or 1
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        qs_current_state = q_stage2_mf[s2_idx, :]
        exp_q2 = np.exp(beta * qs_current_state)
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning Updates ---
        
        # TD(0) update for Stage 1 (SARSA logic used in fitting code)
        # Note: The fitting code uses Q(s2, a2) - Q(s1, a1) as the prediction error
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
