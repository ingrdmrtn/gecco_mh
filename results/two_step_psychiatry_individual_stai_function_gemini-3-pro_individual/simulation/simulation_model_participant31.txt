import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using Model 2: Anxiety-Induced Choice Perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_sensitivity]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or list): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, w, stickiness_sensitivity = parameters
    
    # Handle STAI input (can be scalar or list/array based on fitting code usage)
    if isinstance(stai, (list, np.ndarray, tuple)):
        stai_val = float(stai[0])
    else:
        stai_val = float(stai) if stai is not None else 0.0

    stickiness_bonus = stickiness_sensitivity * stai_val

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf corresponds to the two Stage 1 actions
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the two Stage 2 states, each with 2 actions
    q_stage2_mf = np.zeros((2, 2))

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize current trial reward probabilities (drifts)
        # State 0 (index 0): actions 0/1 -> drift1, drift2
        # State 1 (index 1): actions 0/1 -> drift3, drift4
        current_drifts = [[drift1[t], drift2[t]], 
                          [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Anxiety-Induced Stickiness
        if t > 0:
            prev_a1 = stage1_choice[t-1]
            q_net[prev_a1] += stickiness_bonus
        
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = np.random.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        state_probs = transition_matrix[a1]
        s2 = np.random.choice([0, 1], p=state_probs)

        # --- Stage 2 Decision ---
        q2_vals = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * q2_vals)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        r_prob = current_drifts[s2][a2]
        r = 1 if np.random.random() < r_prob else 0

        # --- Learning Updates ---
        # Update Stage 1 MF Q-value using the value of the state reached (TD-0 style)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF Q-value using the actual reward
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
