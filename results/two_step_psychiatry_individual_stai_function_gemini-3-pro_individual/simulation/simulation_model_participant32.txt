import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Anxiety-Modulated Perseveration model.
    
    Logic:
    - Calculates a stickiness parameter based on STAI.
    - Adds stickiness to the Q-value of the previously chosen action at Stage 1.
    - Uses a hybrid Model-Based/Model-Free valuation for Stage 1 choice.
    - Uses Model-Free valuation for Stage 2 choice.
    - Updates Q-values based on prediction errors.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stick_anxiety_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    # Unpack parameters
    learning_rate, beta, w, stick_base, stick_anxiety_slope = parameters
    
    # Handle STAI input (if it's a list/array, take first element, otherwise treat as float)
    stai_val = stai[0] if isinstance(stai, (list, np.ndarray)) else stai
    if stai_val is None:
        stai_val = 0 # Default if not provided, though logic implies it should be present

    # Calculate anxiety-modulated stickiness
    stickiness = stick_base + (stick_anxiety_slope * stai_val)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: 2 actions
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: 2 states x 2 actions
    q_stage2_mf = np.zeros((2, 2)) 

    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1 

    for t in range(n_trials):
        # Organize reward probabilities for the current trial
        # State 0 (index 0) corresponds to drift1/drift2
        # State 1 (index 1) corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness bonus to the previously chosen action
        if last_action_1 != -1:
            q_net[last_action_1] += stickiness
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # Transition to Stage 2
        # State is 0 or 1. If a1=0, usually state 0 (70%). If a1=1, usually state 1 (70%).
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard Model-Free Q-learning for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # Reward outcome
        r = int(rng.random() < current_reward_probs[s2_idx][a2])

        # --- Updates ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Stage 1 Update (TD-learning using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
