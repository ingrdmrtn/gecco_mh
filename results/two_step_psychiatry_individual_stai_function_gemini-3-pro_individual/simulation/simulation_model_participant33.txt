import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with Anxiety-Modulated Perseveration.
    
    The model features:
    - Weighted combination of Model-Based (MB) and Model-Free (MF) values.
    - A perseveration bonus applied to the previously chosen stage 1 action.
    - The magnitude of perseveration is modulated by the participant's STAI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, persev_base, persev_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the four terminal states.
        stai (float or list): STAI score for the participant. If a list/array is passed, the first element is used.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, persev_base, persev_slope = parameters
    
    # Handle STAI input (if it's a list/array, take the first element, otherwise treat as float)
    if isinstance(stai, (list, np.ndarray)):
        stai_score = stai[0]
    else:
        stai_score = stai if stai is not None else 0.0

    # Calculate the fixed perseveration bonus based on anxiety
    perseveration_bonus = persev_base + (persev_slope * stai_score)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (MF)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (MF)
    # Rows: State 0 (Common for A1=0), State 1 (Common for A1=1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Track last action for perseveration (initially -1 means no previous action)
    last_action_1 = -1
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Calculate Model-Based values for Stage 1
        # MB value of a stage 1 action is the weighted sum of max Q-values of potential stage 2 states
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Calculate Net Q-values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # 3. Apply Perseveration Bonus
        if last_action_1 != -1:
            q_net[last_action_1] += perseveration_bonus
            
        # 4. Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 5. Transition to Stage 2
        # State logic: State 0 corresponds to drifts 1&2, State 1 to drifts 3&4
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 6. Stage 2 Choice (Softmax based on MF values only)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # 7. Reward Generation
        # Map state/action to specific drift probability
        # State 0, Action 0 -> drift1
        # State 0, Action 1 -> drift2
        # State 1, Action 0 -> drift3
        # State 1, Action 1 -> drift4
        if s2_idx == 0:
            current_drift = drift1[t] if a2 == 0 else drift2[t]
        else:
            current_drift = drift3[t] if a2 == 0 else drift4[t]
            
        r = int(rng.random() < current_drift)
        
        # 8. Learning Updates (TD Learning)
        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 MF Q-value (TD(0) style, based on stage 2 value, not reward directly)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update tracker for next trial
        last_action_1 = a1

    return stage1_choice, state2, stage2_choice, reward
