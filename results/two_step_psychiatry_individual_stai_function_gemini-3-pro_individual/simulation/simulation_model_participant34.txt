import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates the Anxiety-Driven Perseveration model.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, pers_base, anxiety_stickiness]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    learning_rate, beta, w, pers_base, anxiety_stickiness = parameters
    
    # Handle STAI input: if None, assume 0 or handle appropriately (here we assume scalar input)
    stai_score = stai if stai is not None else 0.0

    # Calculate the total perseveration bonus based on anxiety
    pers_bonus = pers_base + (anxiety_stickiness * stai_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf corresponds to Q_MF(s1, a)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q_MF(s2, a). Rows are states (0 or 1), cols are actions.
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for current trial
        # State 0 (index 0) has options drift1, drift2
        # State 1 (index 1) has options drift3, drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply perseveration bonus to the previously chosen action
        if last_action_1 != -1:
            q_net[int(last_action_1)] += pers_bonus
        
        # Softmax selection for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax selection for Stage 2 (purely model-free based on fitting code logic)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning Updates ---
        # Note: The fitting code uses a SARSA-like update for stage 1 (using Q(s2, a2))
        # and a standard TD update for stage 2.
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Update history
        last_action_1 = a1

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
