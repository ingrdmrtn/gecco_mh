```python
import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using Model 3: Anxiety-Modulated Win-Stay Bias.
    
    This model includes a bias to repeat the previous Stage 1 choice if and only if
    it resulted in a reward on the previous trial. The strength of this bias is 
    modulated by the participant's STAI (anxiety) score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, ws_base, ws_stai_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant. Defaults to 0 if None.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    # Unpack parameters
    learning_rate, beta, w, ws_base, ws_stai_sens = parameters
    
    # Handle STAI input
    participant_stai = stai if stai is not None else 0.0
    
    # Calculate the Win-Stay bonus based on anxiety
    ws_bonus = ws_base + (ws_stai_sens * participant_stai)

    # Transition matrix for the 2-step task
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf: Model-free values for stage 1 actions (0 and 1)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Model-free values for stage 2 states (0 and 1) and actions (0 and 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Arrays to store simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Variables to track previous trial history for the Win-Stay heuristic
    prev_a1 = -1
    prev_reward = 0

    for t in range(n_trials):
        # Current trial reward probabilities (mapped to state indices 0 and 1)
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        current_reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (hybrid MB/MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits = beta * q_net

        # Apply Anxiety-Modulated Win-Stay Bias
        # If previous action exists and resulted in a reward, add bonus to that action's logit
        if prev_a1 != -1 and prev_reward == 1:
            logits[int(prev_a1)] += ws_bonus
        
        # Softmax choice probability
        exp_q1 = np.exp(logits)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 Choice
        a1 = rng.choice([0, 1], p=p_stage1)
        
        # --- Transition ---
        
        # Determine Stage 2 State based on transition matrix
        # transition_matrix[a1] gives probs of transitioning to state 0 or 1 given action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Stage 2 is purely model-free in this paradigm
        q2 = q_stage2_mf[s2]
        exp_q2 = np.exp(beta * q2)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 Choice
        a2 = rng.choice([0, 1], p=p_stage2)
        
        # --- Reward ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        
        # Update Stage 1 MF value (SARSA-like update using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Update Stage 2 MF value (Prediction Error based on reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # --- Update History ---
        prev_a1 = a1
        prev_reward = r
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
