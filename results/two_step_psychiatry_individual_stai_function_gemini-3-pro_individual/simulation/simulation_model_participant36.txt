import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the hybrid model with STAI-modulated Choice Stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, w, stick_base, stick_stai_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    alpha, beta, w, stick_base, stick_stai_sens = parameters
    
    # Handle STAI input (assuming stai is a scalar or list with at least one element)
    current_stai = stai if isinstance(stai, (float, int)) else stai[0]

    # Calculate the fixed stickiness value based on STAI
    stickiness_val = stick_base + (stick_stai_sens * current_stai)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (model-free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (model-free)
    # Shape is (2, 2) corresponding to [State 0/1, Action 0/1]
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    last_action = -1
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probabilities for the 4 terminal states
        # drift1/2 correspond to State 0 (Actions 0, 1)
        # drift3/4 correspond to State 1 (Actions 0, 1)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values mixing MB and MF
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Calculate logits
        logits = beta * q_net
        
        # Apply STAI-modulated stickiness if not the first trial
        if last_action != -1:
            logits[int(last_action)] += stickiness_val
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 Choice
        a1 = rng.choice([0, 1], p=probs_1)
        last_action = a1
        
        # --- Transition ---
        # Determine Stage 2 State based on transition matrix probabilities for chosen action a1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard Softmax on Stage 2 MF Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 Choice
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning (Update Q-values) ---
        # TD(0) update for Stage 1
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta_stage1
        
        # TD(0) update for Stage 2
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += alpha * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
