import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with Anxiety-Driven Memory Decay.
    
    Hypothesis: High anxiety impairs the maintenance of learned values over time.
    Action values (Q-values) decay toward 0 at a rate proportional to the
    participant's STAI score.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stai_forgetting]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    learning_rate, beta, w, stai_forgetting = parameters
    
    # Handle STAI input: if it's an array/list, take the first element, otherwise use as float
    if stai is not None:
        if isinstance(stai, (list, np.ndarray)):
            stai_score = stai[0]
        else:
            stai_score = stai
    else:
        # Default fallback if STAI is missing (though logic implies it's required)
        stai_score = 0.0

    # Calculate decay based on STAI
    decay = min(max(stai_forgetting * stai_score, 0.0), 1.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf corresponds to Q(s1, a)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q(s2, a) where row 0 is state 2A, row 1 is state 2B
    q_stage2_mf = np.zeros((2, 2))

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Drift reward probs organized by state (0 or 1)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Anxiety-Driven Decay ---
        # Both stage 1 and stage 2 MF values decay before choice
        q_stage1_mf *= (1 - decay)
        q_stage2_mf *= (1 - decay)

        # --- Stage 1 Decision ---
        # MB calculation
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Softmax choice 2 (pure MF at stage 2)
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])

        # --- Learning ---
        # Update Stage 1 MF (TD(0))
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target for stage 1 update,
        # which is standard SARSA-like logic often used in these 2-step tasks.
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2

        # Record data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
