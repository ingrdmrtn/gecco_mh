import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with Anxiety-Modulated Perseveration.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, p_base, p_mod]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, p_base, p_mod = parameters
    
    # Handle STAI normalization as done in the fitting code
    # If stai is None, default to 0 (though the prompt implies it will be provided)
    stai_val = (stai / 80.0) if stai is not None else 0.0
    
    # Calculate the fixed perseveration value based on anxiety
    perseveration_value = p_base + (p_mod * stai_val)

    # Initialize Q-values
    # q_stage1_mf corresponds to the Q-values for the two actions at stage 1 (model-free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the two actions in the two stage 2 states
    # Shape (2, 2): rows are states (0 or 1), columns are actions (0 or 1)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix (fixed as per fitting code)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_action_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        # State 0 corresponds to drifts 1 & 2, State 1 corresponds to drifts 3 & 4
        current_drifts = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based Q-values for Stage 1
        # Max Q from stage 2 states
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values (weighted sum of MB and MF)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Perseveration Bonus
        q_net_biased = q_net.copy()
        if prev_action_1 != -1:
            q_net_biased[int(prev_action_1)] += perseveration_value

        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net_biased)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine second stage state based on transition matrix
        # transition_matrix[act1] gives [p(s2=0|a1), p(s2=1|a1)]
        # Note: In the fitting code, `state` is 0 or 1.
        state_idx = rng.choice([0, 1], p=transition_matrix[act1])
        
        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 (purely MF based on fitting code)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = rng.choice([0, 1], p=probs_2)
        
        # --- Outcome ---
        
        # Determine reward based on drift probabilities
        r = int(rng.random() < current_drifts[state_idx][act2])
        
        # --- Learning ---
        
        # Update Stage 2 Q-values (SARSA / TD(0))
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD(1) - like update using Stage 2 Q-value)
        # Note: The fitting code uses q_stage2_mf[state_idx, act2] as the target for stage 1
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1
        
        # Update history
        prev_action_1 = act1
        
        # Store data
        stage1_choice[t] = act1
        state2[t] = state_idx
        stage2_choice[t] = act2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
