import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Anxiety-Modulated Perseveration (Stickiness) Model.
    
    Hypothesis: Higher anxiety (STAI) leads to increased perseveration (stickiness) 
    at the first stage choice.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_base, stick_stai_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant. Defaults to 0 if None.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_base, stick_stai_slope = parameters
    
    # Handle STAI input
    participant_stai = stai if stai is not None else 0.0
    
    # Calculate effective stickiness based on STAI
    stickiness = stick_base + (participant_stai * stick_stai_slope)
    
    # Initialize arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Q-values for stage 1 actions (Model-Free)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf: Q-values for stage 2 states and actions (Model-Free) [state, action]
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current trial reward probabilities for the 4 terminal states
        # State 0 (index 0) has options with drift1 and drift2
        # State 1 (index 1) has options with drift3 and drift4
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-value (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Calculate logits
        logits_1 = beta * q_net
        
        # Apply stickiness if not the first trial
        if t > 0:
            prev_a = stage1_choice[t-1]
            logits_1[prev_a] += stickiness
            
        # Softmax for Stage 1
        exp_q1 = np.exp(logits_1 - np.max(logits_1)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make choice 1
        a1 = rng.choice([0, 1], p=probs_1)
        stage1_choice[t] = a1
        
        # --- Transition ---
        # Determine second stage state
        # If a1=0: 70% chance of state 0, 30% chance of state 1
        # If a1=1: 30% chance of state 0, 70% chance of state 1
        s2_probs = transition_matrix[a1]
        state_idx = rng.choice([0, 1], p=s2_probs)
        state2[t] = state_idx
        
        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2 (purely Model-Free based on state)
        logits_2 = beta * q_stage2_mf[state_idx]
        exp_q2 = np.exp(logits_2 - np.max(logits_2))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make choice 2
        a2 = rng.choice([0, 1], p=probs_2)
        stage2_choice[t] = a2
        
        # --- Outcome ---
        # Determine reward based on drift probabilities
        prob_reward = current_reward_probs[state_idx][a2]
        r = int(rng.random() < prob_reward)
        reward[t] = r
        
        # --- Learning ---
        
        # Update Stage 2 Q-values (SARSA / TD)
        delta_stage2 = r - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD using Stage 2 Q-value as target)
        # Note: The fitting code uses q_stage2_mf[state_idx, a2] as the target, not r directly
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    return stage1_choice, state2, stage2_choice, reward
