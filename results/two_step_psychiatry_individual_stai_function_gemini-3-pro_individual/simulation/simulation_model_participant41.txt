import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates Model 1: Anxiety-Modulated Perseveration (Stickiness).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, perseveration_base, stai_p_sens]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, w, perseveration_base, stai_p_sens = parameters
    
    # Handle STAI input
    participant_stai = stai if stai is not None else 0.0

    # Calculate the anxiety-modulated perseveration bonus
    perseveration_bonus = perseveration_base + (stai_p_sens * participant_stai)
    
    # Initialize Q-values
    # Stage 1 MF values (2 actions)
    q_stage1_mf = np.zeros(2) 
    # Stage 2 MF values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Transition matrix: Rows are Stage 1 actions (0, 1), Cols are Stage 2 states (0, 1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1

    for t in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-values (Hybrid)
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Calculate logits
        logits = beta * q_net
        
        # Apply anxiety-modulated perseveration bonus to the previously chosen action
        if last_action_1 != -1:
            logits[last_action_1] += perseveration_bonus

        # Softmax for Stage 1
        exp_q1 = np.exp(logits - np.max(logits)) # Subtract max for numerical stability
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Stage 1 Choice
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition to Stage 2 ---
        # Determine state 2 based on action 1 and transition matrix
        current_transition_probs = transition_matrix[a1]
        s2 = np.random.choice([0, 1], p=current_transition_probs)
        
        # --- Stage 2 Decision ---
        qs_current_state = q_stage2_mf[s2]
        
        # Softmax for Stage 2
        exp_q2 = np.exp(beta * qs_current_state - np.max(beta * qs_current_state))
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Stage 2 Choice
        a2 = np.random.choice([0, 1], p=probs_2)

        # --- Reward Outcome ---
        # Get reward probability from drifts based on state and action
        if s2 == 0:
            prob_reward = drift1[t] if a2 == 0 else drift2[t]
        else: # s2 == 1
            prob_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < prob_reward else 0
        
        # --- Value Updating ---
        
        # Update Stage 2 Q-value
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # Update Stage 1 Q-value (TD learning using the updated Q-value of the chosen Stage 2 state/action)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update history
        last_action_1 = a1
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
