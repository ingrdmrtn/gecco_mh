import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates Model 1: Anxiety-Modulated Choice Perseveration (Stickiness).
    
    Parameters:
    - learning_rate: [0, 1] Rate of updating Q-values.
    - beta: [0, 10] Inverse temperature for softmax (choice consistency).
    - w: [0, 1] Mixing weight for Model-Based (1) vs Model-Free (0).
    - stick_base: [0, 5] Baseline tendency to repeat choices.
    - anxiety_stick_mod: [0, 5] How much STAI increases the stickiness bonus.
    """
    
    # Unpack parameters
    learning_rate, beta, w, stick_base, anxiety_stick_mod = parameters
    
    # Handle STAI input (default to 0 if None, though usually provided)
    participant_stai = stai if stai is not None else 0
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf corresponds to Q(s1, a)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q(s2, a) where s2 is state 0 or 1
    q_stage2_mf = np.zeros((2, 2))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    rng = np.random.default_rng()
    
    # Calculate effective stickiness based on anxiety
    stick_effective = stick_base + (participant_stai * anxiety_stick_mod)
    
    prev_a = None

    for t in range(n_trials):
        # Current reward probabilities for the second stage states
        # State 0 (index 0) has options with drift1 and drift2
        # State 1 (index 1) has options with drift3 and drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply Anxiety-Modulated Stickiness
        if prev_a is not None:
            q_net[prev_a] += stick_effective
            
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        # Standard softmax on Q_stage2_mf
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning Updates ---
        
        # TD(0) update for Stage 1 (using Stage 2 Q-value as estimation)
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # TD(0) update for Stage 2 (using actual reward)
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        # Update previous action for next trial's stickiness
        prev_a = a1

    return stage1_choice, state2, stage2_choice, reward
