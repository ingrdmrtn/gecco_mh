import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using Model 1: Anxiety-modulated Model-Based/Model-Free trade-off.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w_baseline, w_stai_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Initialize random number generator
    rng = np.random.default_rng()

    # Unpack parameters
    learning_rate, beta, w_baseline, w_stai_slope = parameters
    
    # Handle STAI input: if it's a list/array, take the first element, otherwise use as is.
    # Default to 0.4 (approx mean) if None to prevent crash, though input is expected.
    if stai is None:
        stai_score = 0.4
    elif hasattr(stai, '__len__'):
        stai_score = stai[0]
    else:
        stai_score = stai

    # Calculate the weighting parameter w based on STAI
    # Note: In the fitting code, logit_w = w_baseline + w_stai_slope * (stai_score - 0.4)
    # However, w_baseline is likely already on the logit scale or unbounded in fitting, 
    # but the docstring says [0, 1]. 
    # Looking closely at the fitting code: `logit_w = w_baseline + ...` implies w_baseline is a logit intercept.
    # But usually, parameters passed to simulation are raw values. 
    # If the fitting optimizer passes unbounded params, w_baseline is the intercept.
    # Let's follow the exact logic of the fitting function provided.
    
    logit_w = w_baseline + w_stai_slope * (stai_score - 0.4)
    w = 1 / (1 + np.exp(-logit_w)) 

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2) # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (2 states, 2 actions)
    
    # Arrays to store simulation data
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Organize drift probabilities for the current trial
        # State 0 corresponds to drift1/drift2, State 1 corresponds to drift3/drift4
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB and MF values
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        act1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        # State 0 (Blue planet) or State 1 (Purple planet)
        state_idx = rng.choice([0, 1], p=transition_matrix[act1])
        
        # --- Stage 2 Decision ---
        # Softmax choice for Stage 2 based on MF values only
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        act2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        # Determine reward based on drift probabilities
        r = int(rng.random() < reward_probs[state_idx][act2])

        # --- Learning Updates ---
        # Update Stage 2 Q-values
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2

        # Update Stage 1 Q-values (TD-learning)
        # Note: The fitting code updates stage 1 twice.
        # 1. Prediction error from Stage 1 to Stage 2 value
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1] # Note: using updated Q2 value as per fitting code logic order? 
        # Actually, standard SARSA/Q-learning usually uses old Q2, but the fitting code writes:
        # q_stage2_mf[...] += ...
        # delta_stage1 = q_stage2_mf[...] - q_stage1_mf[...]
        # So it uses the *updated* Q2 value for the TD error. We replicate that here.
        
        q_stage1_mf[act1] += learning_rate * delta_stage1 

        # 2. Eligibility trace / Direct reinforcement from reward to Stage 1
        # The fitting code adds `learning_rate * delta_stage2` to stage 1.
        # This acts effectively like lambda=1 eligibility trace without a separate lambda parameter.
        q_stage1_mf[act1] += learning_rate * delta_stage2

        # Store data
        stage1_choice[t] = act1
        state2[t] = state_idx
        stage2_choice[t] = act2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
