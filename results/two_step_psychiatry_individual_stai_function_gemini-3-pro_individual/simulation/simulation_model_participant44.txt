import numpy as np

rng = np.random.default_rng()

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai=None):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with Anxiety-modulated Choice Perseveration.
    
    Hypothesis: High anxiety (STAI) increases the tendency to repeat the 
    previous Stage 1 choice (perseveration), modeled as a value bonus 
    added to the previously chosen action.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_stai]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai (float or None): STAI score for the participant.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    learning_rate, beta, w, stick_stai = parameters
    
    # Handle STAI input safely
    if stai is None:
        stai_score = 0.0
    elif isinstance(stai, (list, np.ndarray)):
        stai_score = stai[0]
    else:
        stai_score = stai

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Q-values initialization
    # q_stage1_mf corresponds to the Q-values for the two actions in stage 1
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to the Q-values for the two actions in each of the two stage 2 states
    # Shape: (2 states, 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current trial reward probabilities for state 0 (drift1, drift2) and state 1 (drift3, drift4)
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Net Q-values combining MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply Anxiety-modulated Perseveration Bonus
        if last_action_1 != -1:
            q_net[last_action_1] += stick_stai * stai_score

        # Softmax choice for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        act1 = rng.choice([0, 1], p=probs_1)

        # --- Transition ---
        
        # Determine Stage 2 state based on transition matrix
        state_idx = rng.choice([0, 1], p=transition_matrix[act1])

        # --- Stage 2 Decision ---
        
        # Softmax choice for Stage 2 (purely MF)
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        act2 = rng.choice([0, 1], p=probs_2)

        # --- Reward ---
        r = int(rng.random() < reward_probs[state_idx][act2])

        # --- Learning Updates ---

        # Stage 1 TD error (SARSA-like update using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[state_idx, act2] - q_stage1_mf[act1]
        q_stage1_mf[act1] += learning_rate * delta_stage1

        # Stage 2 TD error (Reward prediction error)
        delta_stage2 = r - q_stage2_mf[state_idx, act2]
        q_stage2_mf[state_idx, act2] += learning_rate * delta_stage2

        # Update tracking variables
        last_action_1 = act1
        
        # Store data
        stage1_choice[t] = act1
        state2[t] = state_idx
        stage2_choice[t] = act2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
