Here are three new cognitive models based on the two-step task structure, incorporating the OCI score in novel ways to explain decision-making behavior.

### Model 1: Hybrid Learner with OCI-modulated Model-Based Weight (`w`)
This model assumes agents use a mixture of Model-Based (MB) and Model-Free (MF) strategies. The core hypothesis is that high OCI scores might correlate with a rigidity or over-reliance on habitual (MF) control, or conversely, an obsessive tracking of transition structures (MB). Here, we model the mixing weight `w` as a function of OCI.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where the mixing weight (w) is modulated by OCI.
    
    Hypothesis: The balance between goal-directed (MB) and habitual (MF) control is 
    influenced by OCI traits. A base weight is adjusted linearly by OCI.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1] - Base weighting for Model-Based control (0=MF, 1=MB).
    oci_w_mod: [-1, 1] - How much OCI shifts the balance. Positive = more MB, Negative = more MF.
    """
    learning_rate, beta, w_base, oci_w_mod = model_parameters
    n_trials = len(action_1)
    # Normalize OCI roughly to 0-1 range if it comes in as 0-100, otherwise assume it's scaled
    current_oci = oci[0] / 100.0 if oci[0] > 1 else oci[0]

    # Calculate effective w, clamped between 0 and 1
    w = w_base + (current_oci * oci_w_mod)
    w = np.clip(w, 0.0, 1.0)
    
    # Transition matrix (fixed for this task structure)
    # A->X (0->0) is 0.7, A->Y (0->1) is 0.3
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State (Planet) x Action (Alien)

    for trial in range(n_trials):
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        # Model-Based Value: Transition * Max(Q_stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value: w * MB + (1-w) * MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        # Stage 2 Update (standard Q-learning)
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        q_stage2_mf[s_idx, a2] += learning_rate * delta_stage2
        
        # Stage 1 MF Update (TD(1) logic: updates based on Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free Learner with OCI-modulated Learning Rate Asymmetry
This model investigates if OCI affects how people learn from positive versus negative prediction errors. It posits that high OCI might lead to "hyper-learning" from negative outcomes (avoidance) or positive outcomes, creating an asymmetry in the learning rate.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free Learner with Asymmetric Learning Rates modulated by OCI.
    
    Hypothesis: OCI affects sensitivity to prediction errors. Specifically, it acts as a
    multiplier on the learning rate for NEGATIVE prediction errors (disappointments),
    potentially reflecting perfectionism or error-sensitivity.
    
    Bounds:
    lr_pos: [0, 1] - Learning rate for positive prediction errors.
    beta: [0, 10]
    neg_bias_base: [0, 2] - Base multiplier for negative prediction errors.
    oci_neg_sens: [0, 5] - Additional sensitivity to negative errors per unit of OCI.
    """
    lr_pos, beta, neg_bias_base, oci_neg_sens = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] / 100.0 if oci[0] > 1 else oci[0]

    # Calculate multiplier for negative learning rate
    # If neg_bias_multiplier > 1, agent learns faster from disappointment than reward
    neg_bias_multiplier = neg_bias_base + (current_oci * oci_neg_sens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating ---
        # Stage 2 Update
        delta_stage2 = r - q_stage2_mf[s_idx, a2]
        
        # Determine effective learning rate based on sign of PE
        if delta_stage2 >= 0:
            eff_lr_s2 = lr_pos
        else:
            eff_lr_s2 = lr_pos * neg_bias_multiplier
            # Clip to ensure stability (optional but good practice)
            eff_lr_s2 = min(eff_lr_s2, 1.0)

        q_stage2_mf[s_idx, a2] += eff_lr_s2 * delta_stage2
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[s_idx, a2] - q_stage1_mf[a1]
        
        if delta_stage1 >= 0:
            eff_lr_s1 = lr_pos
        else:
            eff_lr_s1 = lr_pos * neg_bias_multiplier
            eff_lr_s1 = min(eff_lr_s1, 1.0)
            
        q_stage1_mf[a1] += eff_lr_s1 * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Eligibility Trace Learner (TD(lambda)) with OCI-modulated Decay
This model uses an eligibility trace parameter ($\lambda$) to bridge the gap between Stage 1 choice and Stage 2 reward. The hypothesis is that OCI affects the "credit assignment" processâ€”how much the final outcome is attributed back to the very first choice. High OCI might correlate with higher $\lambda$ (obsessing over the chain of causality) or lower $\lambda$.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    TD(lambda) Learner where the eligibility trace decay (lambda) is modulated by OCI.
    
    Hypothesis: OCI scores influence how far back in time credit is assigned for outcomes.
    Instead of just updating Stage 1 based on Stage 2 values (TD(0)), this model updates
    Stage 1 directly from the reward using an eligibility trace.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    lambda_base: [0, 1] - Base eligibility trace decay parameter.
    oci_lambda_slope: [-1, 1] - How OCI alters the decay rate.
    """
    learning_rate, beta, lambda_base, oci_lambda_slope = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] / 100.0 if oci[0] > 1 else oci[0]

    # Calculate lambda, clamped between 0 and 1
    lam = lambda_base + (current_oci * oci_lambda_slope)
    lam = np.clip(lam, 0.0, 1.0)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        s_idx = int(state[trial])
        a1 = int(action_1[trial])
        a2 = int(action_2[trial])
        r = reward[trial]

        # --- Stage 1 Policy ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[a1]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2[s_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[a2]

        # --- Updating with Eligibility Traces ---
        
        # 1. Prediction error at Stage 1 (State 1 -> State 2)
        # Value of state 2 is approximated by Q(s2, a2)
        pe_1 = q_stage2[s_idx, a2] - q_stage1[a1]
        
        # Update Stage 1 Q-value based on immediate transition
        q_stage1[a1] += learning_rate * pe_1
        
        # 2. Prediction error at Stage 2 (State 2 -> Reward)
        pe_2 = r - q_stage2[s_idx, a2]
        
        # Update Stage 2 Q-value
        q_stage2[s_idx, a2] += learning_rate * pe_2
        
        # 3. Pass Stage 2 PE back to Stage 1 via eligibility trace (lambda)
        # This allows the final reward to directly influence the first choice
        q_stage1[a1] += learning_rate * lam * pe_2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
