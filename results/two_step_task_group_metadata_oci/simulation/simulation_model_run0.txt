import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the Model-Free learner with OCI-modulated Memory Decay.
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, decay_base, oci_decay_slope]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta, decay_base, oci_decay_slope = parameters
    
    # Setup OCI modulation
    # Note: In fitting code, oci was passed as a list/array. Here it is a single float or None.
    # We handle the scaling (divide by 100 if > 1) similar to fitting code.
    if oci is None:
        current_oci = 0.0
    else:
        current_oci = oci / 100.0 if oci > 1 else oci

    # Calculate decay rate based on OCI
    decay_rate = decay_base + (current_oci * oci_decay_slope)
    decay_rate = np.clip(decay_rate, 0.0, 1.0)

    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    # Initialize Q-values
    # q_stage1_mf: Values for Spaceships A (idx 0), U (idx 1)
    q_stage1_mf = np.zeros(2) 
    # q_stage2_mf: Values for Aliens (Planet X (idx 0): W,S; Planet Y (idx 1): P,H)
    q_stage2_mf = np.zeros((2, 2)) 

    # Transition matrix (fixed as per fitting code)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # Organize drift probabilities for easy access: reward_probs[state][action]
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1_mf)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        # Determine state 2 based on transition matrix probabilities
        s2_probs = transition_matrix[a1]
        s2 = rng.choice([0, 1], p=s2_probs) # 0 for Planet X, 1 for Planet Y
        
        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[s2])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2][a2])
        
        # --- Learning / Updates ---
        
        # Update Stage 1 chosen
        delta_stage1 = r - q_stage1_mf[a1] # Note: Fitting code uses 'reward' directly for stage 1 update, not TD(1) or TD(0) difference
        q_stage1_mf[a1] += learning_rate * delta_stage1

        # Decay Stage 1 unchosen
        unchosen_a1 = 1 - a1
        q_stage1_mf[unchosen_a1] *= (1.0 - decay_rate)

        # Update Stage 2 chosen
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2

        # Decay Stage 2 unchosen
        unchosen_a2 = 1 - a2
        q_stage2_mf[s2, unchosen_a2] *= (1.0 - decay_rate)

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
