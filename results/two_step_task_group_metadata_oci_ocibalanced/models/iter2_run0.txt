Here are 3 new cognitive models that incorporate the OCI score into the decision-making process in different ways, exploring mechanisms like eligibility trace decay, inverse temperature modulation, and mixing model-based/model-free strategies.

### Model 1: Hybrid Learner with OCI-Modulated Model-Based Weight
This model assumes that individuals with higher OCI scores might rely more (or less) on rigid, habitual (model-free) learning versus flexible, planning-based (model-based) learning. Here, OCI modulates the mixing weight `w` between these two systems.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Hybrid Model-Based/Model-Free learner where OCI modulates the balance (w).
    
    Hypothesis: High OCI might be associated with a reliance on habit (MF) over 
    flexible planning (MB), or vice versa. This model tests if OCI shifts the 
    weight 'w' towards one system.
    
    Parameters:
    learning_rate: [0,1] Learning rate for value updates.
    beta: [0,10] Inverse temperature (softmax sensitivity).
    w_base: [0,1] Baseline weighting for Model-Based control (0=Pure MF, 1=Pure MB).
    oci_w_shift: [0,1] How much OCI shifts the weight w.
                 (Implemented as: w = sigmoid(logit(w_base) + oci * shift))
    """
    learning_rate, beta, w_base, oci_w_shift = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] / 20.0 # Normalize OCI roughly to 0-1 range

    # Transform w_base to logit space to apply shift safely
    # Clip to avoid log(0)
    w_base_clamped = np.clip(w_base, 0.01, 0.99)
    logit_w = np.log(w_base_clamped / (1 - w_base_clamped))
    
    # Apply OCI shift (could increase or decrease MB reliance depending on sign, 
    # but here we model it as a perturbation magnitude). 
    # We assume OCI pushes away from the baseline, modeled here as an additive effect.
    # Note: To allow bidirectional fitting, we'd usually need a range [-1, 1], 
    # but given constraints [0,1], we model it as reducing MB (increasing habit).
    logit_w_mod = logit_w - (current_oci * oci_w_shift * 5.0) 
    
    w = 1.0 / (1.0 + np.exp(-logit_w_mod))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Value: T * max(Q2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        
        if a2 == -1:
            p_choice_2[trial] = 1.0
            # No update if missing data
        else:
            p_choice_2[trial] = probs_2[a2]
            r = reward[trial]

            # --- Updates ---
            # Stage 2 RPE
            delta_stage2 = r - q_stage2_mf[state_idx, a2]
            q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2
            
            # Stage 1 RPE (TD(0) update for MF)
            # Note: In a full hybrid model, MF updates via TD using the Q-value of the state reached
            delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
            q_stage1_mf[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Model-Free with OCI-Modulated Eligibility Trace
This model introduces an eligibility trace parameter ($\lambda$) that connects the stage 2 reward directly back to the stage 1 choice. The OCI score modulates this decay parameter.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner with Eligibility Traces (TD(lambda)), where OCI modulates lambda.
    
    Hypothesis: OCI is related to the credit assignment process. A higher lambda 
    means the reward at stage 2 is more strongly attributed to stage 1 choice directly.
    High OCI might involve 'over-attribution' or rigid causal chains (high lambda).
    
    Parameters:
    learning_rate: [0,1] Learning rate.
    beta: [0,10] Inverse temperature.
    lambda_base: [0,1] Baseline eligibility trace decay.
    oci_lambda_boost: [0,1] Effect of OCI on increasing lambda.
    """
    learning_rate, beta, lambda_base, oci_lambda_boost = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] / 20.0

    # Calculate lambda based on OCI
    # We clip the result to ensure it stays within [0, 1]
    eligibility_lambda = np.clip(lambda_base + (current_oci * oci_lambda_boost), 0.0, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        
        if a2 == -1:
            p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = probs_2[a2]
            r = reward[trial]

            # --- Updates ---
            # 1. Stage 2 Prediction Error
            delta_stage2 = r - q_stage2[state_idx, a2]
            
            # 2. Stage 1 Prediction Error (TD error at transition)
            delta_stage1 = q_stage2[state_idx, a2] - q_stage1[a1]
            
            # Update Stage 2 Q-value
            q_stage2[state_idx, a2] += learning_rate * delta_stage2
            
            # Update Stage 1 Q-value
            # In TD(lambda), the stage 1 update includes its own error plus 
            # the discounted stage 2 error carried back by lambda.
            # Here we simplify: Update = alpha * (delta1 + lambda * delta2)
            q_stage1[a1] += learning_rate * (delta_stage1 + eligibility_lambda * delta_stage2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Model-Free with OCI-Modulated Choice Determinism (Beta)
This model posits that OCI affects the exploration-exploitation trade-off. Specifically, high OCI might lead to more rigid, deterministic choices (higher beta), reducing random exploration.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model-Free learner where OCI modulates the inverse temperature (beta).
    
    Hypothesis: High OCI is associated with rigidity and reduced exploration.
    This is modeled by OCI increasing the beta parameter, making choice probabilities
    more extreme (closer to 0 or 1) for a given value difference.
    
    Parameters:
    learning_rate: [0,1] Learning rate.
    beta_base: [0,10] Baseline inverse temperature.
    oci_beta_scale: [0,1] Scaling factor for OCI's effect on beta. 
                    (beta_final = beta_base * (1 + oci * scale * 5))
    stickiness: [0,1] Choice stickiness (repetition bias).
    """
    learning_rate, beta_base, oci_beta_scale, stickiness = model_parameters
    n_trials = len(action_1)
    current_oci = oci[0] / 20.0

    # OCI amplifies beta (rigidity)
    # If oci_beta_scale is 0, beta is constant. If high, beta increases with OCI.
    # We multiply by 5 to give the [0,1] parameter significant range impact.
    beta_eff = beta_base * (1.0 + current_oci * oci_beta_scale * 5.0)
    
    # Cap beta to prevent overflow
    beta_eff = min(beta_eff, 20.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    # Track previous choice for stickiness
    last_choice_1 = -1

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Add stickiness bonus to Q-values before softmax
        q_stage1_mod = q_stage1.copy()
        if last_choice_1 != -1:
            q_stage1_mod[last_choice_1] += stickiness

        exp_q1 = np.exp(beta_eff * q_stage1_mod)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        last_choice_1 = a1
        
        state_idx = int(state[trial])

        # --- Stage 2 Choice ---
        # No stickiness modeled for stage 2 (forced choice context usually)
        exp_q2 = np.exp(beta_eff * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        
        if a2 == -1:
            p_choice_2[trial] = 1.0
        else:
            p_choice_2[trial] = probs_2[a2]
            r = reward[trial]

            # --- Updates ---
            delta_stage2 = r - q_stage2[state_idx, a2]
            q_stage2[state_idx, a2] += learning_rate * delta_stage2
            
            # Simple TD(0) update for Stage 1
            delta_stage1 = q_stage2[state_idx, a2] - q_stage1[a1]
            q_stage1[a1] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
