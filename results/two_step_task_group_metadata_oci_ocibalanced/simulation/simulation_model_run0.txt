rng = np.random.default_rng()

def simulate_model(

    drift4,

    the inverse temperature (beta).

    This is modeled by OCI increasing the beta parameter.

    Parameters:

    transition_matrix = np.array([[0.7, 0.3],

    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_choice_1 = -1

    for t in range(n_trials):
        # Current trial reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        exp_q1 = np.exp(beta_eff * q_stage1_mod)
        probs_1 = exp_q1 / np.sum(exp_q1)

        last_choice_1 = a1
        
        # --- Transition ---
        # Determine second stage state based on transition matrix
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])

        # Softmax for Stage 2 (using same beta_eff as per fitting code logic)
        exp_q2 = np.exp(beta_eff * q_stage2[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        r = int(rng.random() < reward_probs[s2_idx][a2])
        
        # --- Learning (Model-Free SARSA-like updates) ---
        
        # Update Stage 2 Q-values based on reward
        # Prediction Error: R - Q2(s2, a2)
        delta_stage2 = r - q_stage2[s2_idx, a2]

        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
