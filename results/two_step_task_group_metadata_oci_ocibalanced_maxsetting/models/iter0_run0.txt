Here are three cognitive models based on the participant data and the two-step task structure. These models incorporate the OCI (Obsessive-Compulsive Inventory) score to explain individual differences in decision-making strategies, specifically focusing on the balance between model-based/model-free control, perseveration (stickiness), and learning rigidity.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Hybrid Model (MB/MF Balance).
    
    This model assumes that the balance between Model-Based (MB) and Model-Free (MF) 
    strategies is modulated by the participant's OCI score. Higher OCI scores 
    influence the mixing weight 'w', typically associated with a shift towards 
    habitual (MF) control or a deficit in goal-directed (MB) control.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature (exploration/exploitation).
    w_base: [0, 10] - Base logit for the mixing weight (before OCI).
    w_oci_slope: [-5, 5] - How strongly OCI affects the MB/MF balance. 
                         Negative values imply high OCI -> less MB control.
    """
    learning_rate, beta, w_base, w_oci_slope = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate mixing weight w based on OCI (Sigmoid transformation to keep in [0,1])
    # w represents the weight of the Model-Based system.
    w_logit = w_base + (w_oci_slope * oci_score)
    w = 1.0 / (1.0 + np.exp(-w_logit))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-Free values for stage 1 (A/U)
    q_stage2_mf = np.zeros((2, 2)) # Model-Free values for stage 2 (Planets X/Y -> Aliens)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        # 1. Model-Based Value Calculation
        # MB Value = Transition Matrix * Max(Stage 2 Values)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Value Calculation
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # 3. Softmax Choice Probability
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        # Standard Softmax on Stage 2 values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Prediction Error for Stage 1 (TD-error using Stage 2 value)
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Prediction Error for Stage 2 (Reward Prediction Error)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Perseveration (Stickiness).
    
    This model posits that OCI relates to "stickiness" or perseverationâ€”the 
    tendency to repeat the previous action regardless of reward history. 
    High OCI scores increase the bonus added to the previously chosen action 
    in the decision policy.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Fixed mixing weight between MB and MF.
    stick_oci: [0, 5] - Scaling factor for OCI-induced perseveration.
    """
    learning_rate, beta, w, stick_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    prev_a1 = -1 # Initialize previous action placeholder

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid value
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply Perseveration Bonus modulated by OCI
        # If OCI is high, the bonus for repeating the previous action is larger.
        if prev_a1 != -1:
            q_net[prev_a1] += (stick_oci * oci_score)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        # Update previous action for next trial
        prev_a1 = a1
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Learning Rigidity.
    
    This model tests the hypothesis that OCI is associated with cognitive rigidity,
    implemented here as a modulation of the learning rate. Higher OCI scores 
    dampen the learning rate, making the participant less sensitive to new 
    reward information (slower updating).
    
    Parameters:
    lr_base: [0, 1] - Base learning rate.
    lr_oci_decay: [0, 10] - Factor by which OCI reduces the learning rate.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Fixed mixing weight between MB and MF.
    """
    lr_base, lr_oci_decay, beta, w = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    # Calculate effective learning rate
    # High OCI -> Lower effective learning rate (Rigidity)
    learning_rate = lr_base / (1.0 + (lr_oci_decay * oci_score))
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]

        # --- Updates ---
        # Note: Both stages use the OCI-modulated learning rate
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
