Here are three cognitive models that incorporate the OCI (Obsessive-Compulsive Inventory) score to explain individual differences in decision-making strategies.

### Model 1: OCI-Modulated Coupled Updating
This model hypothesizes that high OCI scores correlate with "black-and-white" thinking. When a participant learns about the chosen option (e.g., "Spaceship A is good"), they simultaneously update the unchosen option in the opposite direction (e.g., "Spaceship B must be bad"), proportional to the OCI score. This creates a coupled belief system where options are viewed as competing zeros-sum entities.

```python
def cognitive_model1(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 1: OCI-Modulated Coupled Updating (Black-and-White Thinking).
    
    High OCI participants update the unchosen option's value in the opposite 
    direction of the chosen option's prediction error. A positive outcome for 
    one choice implies a negative outcome for the alternative.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature for softmax.
    w: [0, 1] - Mixing weight (1=Model-Based, 0=Model-Free).
    coupling_oci: [0, 5] - Strength of the counterfactual update scaling with OCI.
    """
    learning_rate, beta, w, coupling_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Alien

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        # Stage 1 MF Update
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Coupled Update: Update unchosen option in the opposite direction
        unchosen_a1 = 1 - a1
        # If delta is positive (good), unchosen decreases. If negative (bad), unchosen increases.
        # Scaling by OCI implies this rigidity increases with symptom severity.
        coupling_factor = coupling_oci * oci_score
        q_stage1_mf[unchosen_a1] -= learning_rate * delta_stage1 * coupling_factor
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: OCI-Modulated Random Lapse (Checking)
This model posits that OCI relates to uncertainty-driven "checking" behavior or attentional lapses. The decision policy is modeled as a mixture of the standard value-based Softmax policy and a uniform random policy. The weight of the random component (epsilon) scales with the OCI score, representing the frequency of disengagement from the learned values to "check" other options.

```python
def cognitive_model2(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 2: OCI-Modulated Random Lapse (Uncertainty Checking).
    
    The decision policy is a mixture of Softmax and Uniform random. 
    High OCI scores increase the 'epsilon' parameter, leading to more 
    random exploration or 'checking' regardless of learned value.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight (MB vs MF).
    lapse_scale: [0, 1] - Scaling factor converting OCI to lapse rate (epsilon).
    """
    learning_rate, beta, w, lapse_scale = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Calculate epsilon (lapse rate). Clip to max 0.5 (random guessing) or slightly less than 1.
    # We clip at 0.99 to avoid log(0) issues, though the mixture handles it.
    epsilon = lapse_scale * oci_score
    if epsilon > 0.99: epsilon = 0.99
    
    for trial in range(n_trials):
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q1 = np.exp(beta * q_net)
        softmax_probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Mixture Policy: (1 - epsilon) * Greedy + epsilon * Random
        probs_1 = (1 - epsilon) * softmax_probs_1 + epsilon * 0.5
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        softmax_probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Apply same lapse rate to Stage 2
        probs_2 = (1 - epsilon) * softmax_probs_2 + epsilon * 0.5
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: OCI-Modulated Spatial Bias
This model tests whether high OCI scores are associated with a fixed, intrinsic bias towards a specific response option (Option 0, "Spaceship A"), independent of reward history. This could reflect a "safe spot" heuristic, motor bias, or spatial neglect often associated with anxious/compulsive phenotypes.

```python
def cognitive_model3(action_1, state, action_2, reward, oci, model_parameters):
    """
    Model 3: OCI-Modulated Spatial Bias.
    
    High OCI scores induce a static bias added to the value of Option 0 (Spaceship A).
    This models a fixed preference or 'default' choice that persists regardless 
    of dynamic value updates.
    
    Parameters:
    learning_rate: [0, 1] - Rate of value updating.
    beta: [0, 10] - Inverse temperature.
    w: [0, 1] - Mixing weight (MB vs MF).
    bias_oci: [0, 5] - Magnitude of the fixed bias for Option 0, scaled by OCI.
    """
    learning_rate, beta, w, bias_oci = model_parameters
    n_trials = len(action_1)
    oci_score = oci[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Static bias term
    bias_val = bias_oci * oci_score

    for trial in range(n_trials):
        # --- Stage 1 ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply bias to Option 0
        q_net[0] += bias_val
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = int(action_1[trial])
        p_choice_1[trial] = probs_1[a1]
        
        state_idx = int(state[trial])

        # --- Stage 2 ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = int(action_2[trial])
        p_choice_2[trial] = probs_2[a2]
        
        # --- Learning ---
        delta_stage1 = q_stage2_mf[state_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, a2]
        q_stage2_mf[state_idx, a2] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
