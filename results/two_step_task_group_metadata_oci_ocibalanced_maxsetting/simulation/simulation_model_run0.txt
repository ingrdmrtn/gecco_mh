import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    where OCI modulates the Inverse Temperature (Beta) specifically after losses.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta_win, beta_loss_base, beta_loss_oci, w, stick]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters
    learning_rate, beta_win, beta_loss_base, beta_loss_oci, w, stick = parameters

    # Handle OCI input (scalar expected in simulation, but robust to list/array)
    subject_oci = 0.0
    if oci is not None:
        if hasattr(oci, '__len__') and not isinstance(oci, str):
            subject_oci = oci[0]
        else:
            subject_oci = oci

    # Calculate the specific beta used after a loss, modulated by OCI
    beta_loss = beta_loss_base + (beta_loss_oci * subject_oci)
    beta_loss = max(0.0, beta_loss)

    # Initialize model structures
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # Shape: [State, Action]

    # Initialize output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Initialize state variables
    last_action_1 = -1
    last_reward = 1.0 # Initialize assuming a 'win' or neutral state for the first trial
    
    rng = np.random.default_rng()

    for t in range(n_trials):
        # 1. Determine current Beta based on previous reward
        if last_reward > 0.5:
            current_beta = beta_win
        else:
            current_beta = beta_loss

        # 2. Stage 1 Decision Process
        # Model-Based Value: Transition Matrix * Max(Q_Stage2)
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Net Q-Value: Weighted sum of MB and MF
        q_net_s1 = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply Stickiness (Perseveration)
        if last_action_1 != -1:
            q_net_s1[last_action_1] += stick
            
        # Softmax Choice Stage 1
        exp_q1 = np.exp(current_beta * q_net_s1)
        # Handle potential overflow/underflow or zero sum
        if np.sum(exp_q1) == 0:
            probs_1 = np.array([0.5, 0.5])
        else:
            probs_1 = exp_q1 / np.sum(exp_q1)
        
        a1 = rng.choice([0, 1], p=probs_1)
        
        # 3. Transition to Stage 2
        # State 0 is common for Action 0, State 1 is common for Action 1
        s2 = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 4. Stage 2 Decision Process
        # Softmax Choice Stage 2 using the same current_beta
        exp_q2 = np.exp(current_beta * q_stage2_mf[s2])
        if np.sum(exp_q2) == 0:
            probs_2 = np.array([0.5, 0.5])
        else:
            probs_2 = exp_q2 / np.sum(exp_q2)
        
        a2 = rng.choice([0, 1], p=probs_2)

        # 5. Reward Outcome
        # Map drift parameters to states/actions
        # State 0: [drift1, drift2], State 1: [drift3, drift4]
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]
        
        prob_reward = current_reward_probs[s2][a2]
        r = 1 if rng.random() < prob_reward else 0

        # 6. Learning Updates
        # Update Stage 1 MF (TD(0)-like update using Stage 2 Q-value)
        delta_stage1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF (Standard Q-learning update using Reward)
        delta_stage2 = r - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += learning_rate * delta_stage2
        
        # 7. Store Data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r
        
        # 8. Update history for next trial
        last_action_1 = a1
        last_reward = r

    return stage1_choice, state2, stage2_choice, reward
