import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    oci=None):
    """
    Simulates choices and rewards using Model 2: OCI-Modulated Perseveration (Stickiness).
    
    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stick_oci]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        oci (float or list): Obsessive-Compulsive Inventoryâ€“Revised score.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    # Unpack parameters based on cognitive_model2 definition
    learning_rate, beta, w, stick_oci = parameters
    
    # Handle OCI input (scalar or list/array)
    if isinstance(oci, (list, np.ndarray)):
        oci_score = oci[0]
    else:
        oci_score = oci

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    # q_stage1_mf corresponds to Q_MF(s1, a)
    q_stage1_mf = np.zeros(2)
    # q_stage2_mf corresponds to Q_MF(s2, a) where s2 has 2 states
    q_stage2_mf = np.zeros((2, 2))
    
    # Initialize storage arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    prev_a1 = -1 # Initialize previous action placeholder

    for t in range(n_trials):
        
        # --- Stage 1 Decision ---
        
        # Calculate Model-Based values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Calculate Net Q-values (Weighted MB + MF)
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Add OCI-modulated stickiness
        if prev_a1 != -1:
            q_net[prev_a1] += (stick_oci * oci_score)
            
        # Softmax for Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        
        # Make Choice 1
        a1 = np.random.choice([0, 1], p=probs_1)
        
        # --- Transition ---
        
        # Determine State 2 based on Action 1 and Transition Matrix
        # transition_matrix[a1] gives [p(s2=0|a1), p(s2=1|a1)]
        s2_idx = np.random.choice([0, 1], p=transition_matrix[a1])
        
        # --- Stage 2 Decision ---
        
        # Softmax for Stage 2
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        # Make Choice 2
        a2 = np.random.choice([0, 1], p=probs_2)
        
        # --- Reward ---
        
        # Get reward probability based on current state and action
        # State 0 uses drift1 (action 0) and drift2 (action 1)
        # State 1 uses drift3 (action 0) and drift4 (action 1)
        if s2_idx == 0:
            p_reward = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_reward = drift3[t] if a2 == 0 else drift4[t]
            
        r = 1 if np.random.random() < p_reward else 0
        
        # --- Value Updates ---
        
        # Update Stage 1 MF values (TD0)
        # Note: Model 2 in fitting code does not use eligibility traces (lambda).
        # It updates S1 based on the value of the state reached in S2.
        delta_stage1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF values
        delta_stage2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * delta_stage2
        
        # --- Record Data ---
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r
        
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
