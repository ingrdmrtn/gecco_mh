Here are three cognitive models that incorporate STAI scores to explain decision-making in the two-step task.

### Model 1: Anxiety-Modulated Model-Based/Model-Free Trade-off
This model hypothesizes that anxiety (STAI) shifts the balance between Model-Based (planning) and Model-Free (habitual) control. High anxiety is often associated with reduced cognitive resources for complex planning, leading to a greater reliance on simple, habitual (Model-Free) strategies.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated MB/MF Trade-off Model.
    
    Hypothesis: Higher anxiety (STAI) reduces the weight (w) placed on Model-Based planning,
    biasing the agent towards Model-Free (habitual) control.
    
    Parameters:
    - learning_rate: [0, 1] Rate at which Q-values are updated.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w_base: [0, 1] Baseline mixing weight for Model-Based control (0=pure MF, 1=pure MB).
    - w_anxiety_slope: [0, 1] How strongly anxiety reduces the MB weight.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w_base: [0, 1]
    w_anxiety_slope: [0, 1]
    """
    learning_rate, beta, w_base, w_anxiety_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate the mixing weight 'w' based on anxiety.
    # Higher anxiety reduces w, making the agent more Model-Free.
    # We clip to ensure w stays between 0 and 1.
    w = np.clip(w_base - (w_anxiety_slope * stai_score), 0.0, 1.0)

    # Transition matrix: A->X (0.7), A->Y (0.3), U->X (0.3), U->Y (0.7)
    # Rows: Actions (A=0, U=1), Cols: States (X=0, Y=1)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)       # Model-free values for stage 1 actions
    q_stage2_mf = np.zeros((2, 2))  # Model-free values for stage 2 (State X/Y, Alien 1/2)

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based Valuation: V(state) = max(Q_stage2(state, :))
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q-value: Mix MB and MF based on anxiety-modulated weight
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax policy
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial] # 0 for X, 1 for Y
        
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Stage 2 Update (TD Error)
        pe_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_2
        
        # Stage 1 Update (TD Error using Stage 2 value)
        # Note: Standard MF update uses the value of the state actually reached
        pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Learning Rate Asymmetry
This model posits that anxiety affects how people learn from positive versus negative outcomes. High anxiety individuals might be hyper-sensitive to negative outcomes (or lack of reward) or have difficulty updating beliefs after positive outcomes. This model splits the learning rate based on the STAI score.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Learning Rate Asymmetry Model.
    
    Hypothesis: Anxiety modulates learning rates differently for positive vs. negative prediction errors.
    High anxiety might increase learning from negative outcomes (loss sensitivity) or dampen learning from positive ones.
    
    Parameters:
    - lr_base: [0, 1] Base learning rate.
    - lr_mod: [0, 1] Magnitude of anxiety's effect on learning rate asymmetry.
    - beta: [0, 10] Inverse temperature.
    - lambda_eligibility: [0, 1] Eligibility trace parameter (how much stage 2 outcome updates stage 1).
    
    Bounds:
    lr_base: [0, 1]
    lr_mod: [0, 1]
    beta: [0, 10]
    lambda_eligibility: [0, 1]
    """
    lr_base, lr_mod, beta, lambda_eligibility = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Define separate learning rates based on anxiety
    # If anxiety is high, lr_pos decreases and lr_neg increases (simulating negativity bias)
    # We ensure bounds [0, 1]
    lr_pos = np.clip(lr_base - (lr_mod * stai_score), 0.01, 1.0)
    lr_neg = np.clip(lr_base + (lr_mod * stai_score), 0.01, 1.0)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Calculate Prediction Error at Stage 2
        pe_2 = reward[trial] - q_stage2[state_idx, action_2[trial]]
        
        # Select learning rate based on sign of PE
        current_lr = lr_pos if pe_2 >= 0 else lr_neg
        
        # Update Stage 2
        q_stage2[state_idx, action_2[trial]] += current_lr * pe_2
        
        # Update Stage 1 using eligibility trace (TD(lambda) logic simplified)
        # Stage 1 PE is driven by the value of the state reached
        pe_1 = q_stage2[state_idx, action_2[trial]] - q_stage1[action_1[trial]]
        
        # We apply the same asymmetry logic to Stage 1 updates
        current_lr_1 = lr_pos if pe_1 >= 0 else lr_neg
        
        # Direct update + eligibility trace from stage 2 outcome
        q_stage1[action_1[trial]] += current_lr_1 * pe_1 + (current_lr * lambda_eligibility * pe_2)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Exploration/Exploitation (Inverse Temperature Modulation)
This model suggests that anxiety impacts the exploration-exploitation trade-off. Specifically, high anxiety might lead to more erratic behavior (random exploration) due to stress, or conversely, rigid exploitation (low temperature). Here, we model anxiety as modifying the inverse temperature parameter ($\beta$).

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Inverse Temperature Modulation.
    
    Hypothesis: Anxiety changes the randomness of choice (beta). 
    High anxiety might lower beta (more random/noise) or raise it (rigid behavior).
    This model assumes a linear relationship between STAI and beta.
    
    Parameters:
    - learning_rate: [0, 1] Standard learning rate.
    - beta_intercept: [0, 10] Beta value when STAI is 0.
    - beta_slope: [-5, 5] How STAI changes beta. (Note: bounds in template are usually positive, 
      but here we implement slope logic using a [0,1] param scaled to represent direction).
      Let's use a parameter 'beta_mod' [0, 1] where <0.5 reduces beta, >0.5 increases it.
    - mixing_weight: [0, 1] Fixed balance between MB and MF.
    
    Bounds:
    learning_rate: [0, 1]
    beta_intercept: [0, 10]
    beta_mod: [0, 1]
    mixing_weight: [0, 1]
    """
    learning_rate, beta_intercept, beta_mod, mixing_weight = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Transform beta_mod to a slope range, e.g., -5 to +5
    # If beta_mod is 0.5, slope is 0 (no effect).
    # If beta_mod is 1.0, slope is +5 (anxiety increases rigidity).
    # If beta_mod is 0.0, slope is -5 (anxiety increases noise).
    slope = (beta_mod - 0.5) * 10.0 
    
    # Calculate effective beta
    beta = np.clip(beta_intercept + (slope * stai_score), 0.0, 10.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Choice ---
        # Model-Based component
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Q
        q_net = mixing_weight * q_stage1_mb + (1 - mixing_weight) * q_stage1_mf
        
        # Softmax with anxiety-modulated beta
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Choice ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Standard Q-learning updates
        pe_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_2
        
        pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
