Here are 3 new cognitive models based on the hypothesis that anxiety affects reward valuation, internal model uncertainty, and reaction to failure.

### Model 1: Anxiety-Induced Pessimism (Reward Valuation)
This model hypothesizes that anxiety introduces a "pessimism bias" in how rewards are perceived. Rather than affecting the learning rate, STAI scales a penalty applied to the reward signal itself. A high-anxiety individual might perceive a "0" outcome as a negative punishment rather than a neutral event, or undervalue a "1" outcome.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Pessimism Model.
    
    Hypothesis: Anxiety introduces a subjective valuation bias. 
    High anxiety individuals may perceive rewards as less valuable or 
    neutral outcomes as punishments (pessimism).
    
    The effective reward is calculated as: R_eff = Reward - (STAI * pessimism_factor).
    
    Parameters:
    - learning_rate: [0, 1] Update rate for Q-values.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-based weight.
    - pessimism_factor: [0, 1] Scales how much STAI reduces the perceived reward.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    pessimism_factor: [0, 1]
    """
    learning_rate, beta, w, pessimism_factor = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Standard transition matrix
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Calculate Subjective Reward
        # If pessimism_factor is high and STAI is high, rewards are dampened 
        # and 0s become negative (punishments).
        effective_reward = reward[trial] - (stai_score * pessimism_factor)
        
        # Stage 2 Update
        pe_2 = effective_reward - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_2
        
        # Stage 1 Update
        pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Transition Uncertainty
This model hypothesizes that anxiety degrades the precision of the Model-Based system. While the `w` parameter controls the *weight* of the MB system, this model assumes anxiety affects the *quality* of the internal model itself. High anxiety "flattens" the subjective transition matrix towards 50/50 (maximum entropy), representing doubt about the spaceship-planet structure.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Transition Uncertainty Model.
    
    Hypothesis: Anxiety degrades the internal model of the environment structure.
    High anxiety causes the participant to doubt the reliability of the transitions 
    (0.7/0.3), pushing their subjective internal model closer to random (0.5/0.5).
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - uncertainty_scale: [0, 1] How strongly STAI flattens the transition matrix.
      If 0, matrix is [0.7, 0.3]. If 1 and STAI=1, matrix approaches [0.5, 0.5].
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    uncertainty_scale: [0, 1]
    """
    learning_rate, beta, w, uncertainty_scale = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate Subjective Transition Matrix
    # True probability of common transition
    p_true = 0.7
    # Distortion amount (0 to 0.2). 
    # If distortion is 0.2, p becomes 0.5 (random).
    distortion = 0.2 * (stai_score * uncertainty_scale)
    p_subjective = p_true - distortion
    
    # Subjective matrix used for MB calculation
    transition_matrix = np.array([
        [p_subjective, 1 - p_subjective], 
        [1 - p_subjective, p_subjective]
    ])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # MB calculation uses the anxiety-distorted matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        pe_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_2
        
        pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * pe_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven "Lose-Shift"
Previous models explored general stickiness (repeating choices regardless of outcome). This model separates "Win-Stay" (habit) from "Lose-Shift" (avoidance). It hypothesizes that anxiety specifically amplifies the urge to switch away from a choice that resulted in a loss (0 coins), reflecting a hypersensitivity to negative feedback.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Lose-Shift Model.
    
    Hypothesis: Anxiety specifically modulates reaction to failure.
    While 'stickiness_win' captures standard habit formation after rewards,
    'repulsion_loss_anxiety' captures the urge to switch after a loss, 
    which scales with STAI.
    
    Parameters:
    - learning_rate: [0, 1]
    - beta: [0, 10]
    - w: [0, 1]
    - stickiness_win: [0, 1] Tendency to repeat choice after Reward=1 (Habit).
    - repulsion_loss_anxiety: [0, 1] Tendency to switch after Reward=0, scaled by STAI.
    
    Bounds:
    learning_rate: [0, 1]
    beta: [0, 10]
    w: [0, 1]
    stickiness_win: [0, 1]
    repulsion_loss_anxiety: [0, 1]
    """
    learning_rate, beta, w, stickiness_win, repulsion_loss_anxiety = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    last_action_1 = -1
    last_reward = -1

    for trial in range(n_trials):
        
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply Outcome-Dependent Stickiness
        if last_action_1 != -1:
            if last_reward == 1:
                # Win-Stay: Add positive bonus to previous action
                q_net[last_action_1] += stickiness_win
            else:
                # Lose-Shift: Add negative penalty (repulsion) to previous action
                # High anxiety -> High repulsion -> More switching
                repulsion = stai_score * repulsion_loss_anxiety
                q_net[last_action_1] -= repulsion
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store history for next trial
        last_action_1 = action_1[trial]
        
        state_idx = state[trial]
        
        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        pe_2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * pe_2
        
        pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * pe_1
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
