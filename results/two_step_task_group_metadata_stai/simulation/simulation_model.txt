```python
import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4,
    stai_score=40,  # Default STAI score if not provided in a wrapper
    seed=None
):
    """
    Simulates choices and rewards using the Anxiety-Modulated Perseveration Model.
    
    Logic:
    - Hybrid Model-Based/Model-Free control.
    - Stickiness (perseveration) on the first stage is modulated by the STAI score.
    - Standard Q-learning updates.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [learning_rate, beta, w, stickiness_base, stickiness_anxiety]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities.
        stai_score (float): The anxiety score for the simulated agent.
        seed (int or None): Random seed for reproducibility.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    
    rng = np.random.default_rng(seed)
    
    # Unpack parameters
    learning_rate, beta, w, stickiness_base, stickiness_anxiety = parameters
    
    # Calculate effective stickiness based on STAI
    effective_stickiness = stickiness_base + (stai_score * stickiness_anxiety)

    # Transition matrix (fixed as per standard 2-step task)
    transition_matrix = np.array([[0.7, 0.3], 
                                  [0.3, 0.7]])

    # Initialize Q-values
    # Stage 1 MF Q-values (2 actions)
    q_stage1_mf = np.zeros(2)
    # Stage 2 MF Q-values (2 states x 2 actions)
    q_stage2_mf = np.zeros((2, 2))
    
    # Storage for simulation history
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)
    
    last_action_1 = -1  # Indicator for no previous action

    for t in range(n_trials):
        # 1. Organize current reward probabilities
        # State 0 (drift1, drift2), State 1 (drift3, drift4)
        current_reward_probs = [[drift1[t], drift2[t]], 
                                [drift3[t], drift4[t]]]

        # 2. Calculate Model-Based values for Stage 1
        # Max Q-value for each second-stage state
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        # Bellman equation using transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 3. Calculate Net Q-values for Stage 1
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf

        # Apply stickiness if there was a previous action
        if last_action_1 != -1:
            q_net[last_action_1] += effective_stickiness
            
        # 4. Make Stage 1 Choice (Softmax)
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=probs_1)
        
        last_action_1 = a1 # Update for next trial

        # 5. Determine Transition to Stage 2
        # If a1=0, probs are [0.7, 0.3]. If a1=1, probs are [0.3, 0.7].
        s2_idx = rng.choice([0, 1], p=transition_matrix[a1])
        
        # 6. Make Stage 2 Choice (Softmax)
        # Note: The fitting code uses the same beta for both stages
        exp_q2 = np.exp(beta * q_stage2_mf[s2_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=probs_2)

        # 7. Observe Reward
        r = int(rng.random() < current_reward_probs[s2_idx][a2])

        # 8. Update Q-values (TD Learning)
        # Update Stage 2 Q-value
        pe_2 = r - q_stage2_mf[s2_idx, a2]
        q_stage2_mf[s2_idx, a2] += learning_rate * pe_2

        # Update Stage 1 Q-value
        # Note: The fitting code uses q_stage2_mf[s2, a2] as the target, not r directly (SARSA-like logic for stage 1)
        # It uses the *updated* stage 2 value in the fitting code provided:
        # pe_1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        pe_1 = q_stage2_mf[s2_idx, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += learning_rate * pe_1

        # Store data
        stage1_choice[t] = a1
        state2[t] = s2_idx
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
```
