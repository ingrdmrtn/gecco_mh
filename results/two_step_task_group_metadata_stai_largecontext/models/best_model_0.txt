def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hypothesis: Anxiety Increases "Stickiness" (Perseveration).
    STAI modulates a choice autocorrelation parameter. High anxiety leads to a higher
    tendency to repeat the previous Stage 1 choice, regardless of value.
    
    Parameters:
    - learning_rate: [0, 1]
    - inverse_temperature: [0, 10]
    - w: [0, 1] - Fixed mixing weight.
    - stick_base: [-2, 2] - Baseline stickiness.
    - stick_stai_slope: [-2, 2] - How STAI changes stickiness.
    
    Stickiness bonus = stick_base + stick_stai_slope * stai
    This bonus is added to the Q-value of the previously chosen action.
    """
    learning_rate, inverse_temperature, w, stick_base, stick_stai_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    stickiness = stick_base + (stick_stai_slope * stai_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1 # No previous action on trial 0

    for trial in range(n_trials):

        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)



        q_choice = q_net.copy()
        if last_action_1 != -1:
            q_choice[last_action_1] += stickiness
        
        exp_q = np.exp(inverse_temperature * q_choice)
        probs_1 = exp_q / np.sum(exp_q)
        
        chosen_a1 = int(action_1[trial])
        log_likelihood += np.log(probs_1[chosen_a1] + eps)

        current_state = int(state[trial])
        exp_q2 = np.exp(inverse_temperature * q_stage2_mf[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_likelihood += np.log(probs_2[chosen_a2] + eps)

        delta_stage1 = q_stage2_mf[current_state, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[current_state, chosen_a2]
        q_stage2_mf[current_state, chosen_a2] += learning_rate * delta_stage2

        last_action_1 = chosen_a1

    return -log_likelihood
