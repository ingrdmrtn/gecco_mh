Here are three cognitive models that incorporate the STAI (anxiety) score into the decision-making process for the two-step task.

### Model 1: Anxiety-Modulated Model-Based/Model-Free Balance
This model hypothesizes that anxiety (STAI) shifts the balance between Model-Based (planning) and Model-Free (habitual) control. High anxiety is often associated with reduced cognitive flexibility or a reliance on habits under stress. Here, the `w` parameter (mixing weight) is modulated by STAI.

```python
def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hybrid Model-Based/Model-Free RL with Anxiety-Modulated Mixing Weight.
    
    Hypothesis: Anxiety (STAI) influences the trade-off between goal-directed (MB) 
    and habitual (MF) control. Higher anxiety might bias towards one system 
    (modeled via a linear shift on the mixing weight 'w').

    Parameters:
    - learning_rate: [0, 1] Learning rate for Q-value updates.
    - beta: [0, 10] Inverse temperature for softmax choice.
    - w_base: [0, 1] Baseline mixing weight (0=Pure MF, 1=Pure MB).
    - w_stai_mod: [-1, 1] Effect of STAI on mixing weight.
    """
    learning_rate, beta, w_base, w_stai_mod = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate the effective mixing weight based on STAI
    # We clip to ensure w remains between 0 and 1
    w = w_base + (w_stai_mod * stai_score)
    w = np.clip(w, 0.0, 1.0)

    # Fixed transition matrix (A->X ~0.7, U->Y ~0.7)
    # Rows: Actions (0=A, 1=U), Cols: States (0=X, 1=Y)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for Stage 1 actions
    q_stage2 = np.zeros((2, 2))    # Values for Stage 2 (State X/Y, Aliens)
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        # Model-Based Value: Expected value of next state's max Q-value
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Integrated Value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax Policy Stage 1
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial] # 0 for X, 1 for Y
        
        # Softmax Policy Stage 2
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        # Stage 2 Update (TD Error)
        r = reward[trial]
        # Note: Aliens return -1, 0, or 1. We treat -1 as a loss.
        delta_stage2 = r - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Stage 1 MF Update (TD(1) style - using the reward from stage 2)
        # We update Stage 1 MF value based on the reward received at the end
        delta_stage1 = r - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Modulated Learning Rate (Punishment Sensitivity)
This model hypothesizes that anxiety specifically affects how participants learn from negative outcomes. High anxiety individuals might over-weight punishments (negative coins) or lack of rewards compared to positive rewards. This is modeled by having separate learning rates for positive and non-positive outcomes, where the non-positive learning rate is scaled by STAI.

```python
def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Free RL with Anxiety-Modulated Punishment Sensitivity.
    
    Hypothesis: Anxiety increases the learning rate specifically for negative 
    or zero outcomes (punishment/omission), leading to faster avoidance learning.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive rewards (>0).
    - lr_neg_base: [0, 1] Base learning rate for non-positive rewards (<=0).
    - lr_neg_stai: [0, 1] Additional sensitivity to negative outcomes due to STAI.
    - beta: [0, 10] Inverse temperature.
    """
    lr_pos, lr_neg_base, lr_neg_stai, beta = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate effective negative learning rate
    # High anxiety increases learning from failure/loss
    lr_neg = lr_neg_base + (lr_neg_stai * stai_score)
    lr_neg = np.clip(lr_neg, 0.0, 1.0)

    # Q-values (Pure Model-Free for simplicity to isolate learning rate effects)
    q_stage1 = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        exp_q1 = np.exp(beta * q_stage1)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        r = reward[trial]
        
        # Determine which learning rate to use
        if r > 0:
            curr_lr = lr_pos
        else:
            curr_lr = lr_neg
            
        # Stage 2 Update
        delta_2 = r - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += curr_lr * delta_2
        
        # Stage 1 Update (SARSA-like TD(1))
        # Using the actual reward to update the first stage choice
        delta_1 = r - q_stage1[action_1[trial]]
        q_stage1[action_1[trial]] += curr_lr * delta_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Modulated Choice Randomness (Exploration)
This model hypothesizes that anxiety acts as a "noise" parameter or affects the exploration-exploitation trade-off. High anxiety might lead to more erratic behavior (higher temperature/lower beta) due to difficulty concentrating, or conversely, rigid behavior. Here, we model the inverse temperature `beta` as a function of STAI.

```python
def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Model-Based RL with Anxiety-Modulated Inverse Temperature (Beta).
    
    Hypothesis: Anxiety affects decision noise. High anxiety might decrease 
    the inverse temperature (beta), making choices more random (less deterministic 
    based on value), representing 'choking' or confusion under stress.
    
    Parameters:
    - learning_rate: [0, 1] Learning rate.
    - beta_base: [0, 10] Baseline inverse temperature.
    - beta_stai_slope: [-5, 5] How STAI changes beta. Negative = anxiety adds noise.
    - w: [0, 1] Fixed mixing weight for MB/MF (simplified to focus on beta).
    """
    learning_rate, beta_base, beta_stai_slope, w = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate effective beta
    # We allow slope to be negative (anxiety reduces precision) or positive
    beta = beta_base + (beta_stai_slope * stai_score)
    # Ensure beta stays within reasonable bounds [0, 20] for numerical stability
    # though the optimizer sees [0,10] for base, the effective can vary slightly
    beta = np.clip(beta, 0.0, 20.0)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    q_stage1_mf = np.zeros(2)
    q_stage2 = np.zeros((2, 2))
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Weighted combination
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Decision ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Learning ---
        r = reward[trial]
        
        # Stage 2 Update
        delta_2 = r - q_stage2[state_idx, action_2[trial]]
        q_stage2[state_idx, action_2[trial]] += learning_rate * delta_2
        
        # Stage 1 MF Update
        delta_1 = r - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_1

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
