import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hypothesis: Anxiety Impairs Model-Based Control.
    STAI modulates the mixing weight 'w' between Model-Based (MB) and Model-Free (MF) systems.
    Higher anxiety (STAI) is hypothesized to reduce 'w', leading to more habitual behavior.
    
    Parameters:
    - learning_rate: [0, 1] - Rate at which Q-values are updated.
    - inverse_temperature: [0, 10] - Softmax exploration parameter (beta).
    - w_intercept: [-5, 5] - Base logit for the mixing weight.
    - w_stai_slope: [-5, 5] - Effect of STAI on the mixing weight logit.
    
    The mixing weight w is calculated as:
    w = sigmoid(w_intercept + w_stai_slope * stai)
    """
    learning_rate, inverse_temperature, w_intercept, w_stai_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0] # STAI is constant per participant
    
    # Calculate mixing weight w based on STAI
    # Using sigmoid to ensure w stays within [0, 1]
    w_logit = w_intercept + (w_stai_slope * stai_score)
    w = 1.0 / (1.0 + np.exp(-w_logit))
    
    # Transition matrix: Row 0=Action A, Row 1=Action U. Col 0=Planet X, Col 1=Planet Y.
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    # Initialize Q-values
    q_stage1_mf = np.zeros(2)      # MF values for stage 1 choices (A vs U)
    q_stage2_mf = np.zeros((2, 2)) # MF values for stage 2 choices (Aliens at Planet X, Aliens at Planet Y)
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # 1. Model-Based Valuation
        # MB value of action 1 is the expected value of the best action at the likely states
        max_q_stage2 = np.max(q_stage2_mf, axis=1) # Best value at Planet X, Best value at Planet Y
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # 2. Hybrid Valuation
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # 3. Action Selection (Softmax)
        exp_q = np.exp(inverse_temperature * q_net)
        probs_1 = exp_q / np.sum(exp_q)
        
        # Accumulate Log Likelihood
        chosen_a1 = int(action_1[trial])
        log_likelihood += np.log(probs_1[chosen_a1] + eps)
        
        # --- Stage 2 Decision ---
        current_state = int(state[trial]) # 0 for Planet X, 1 for Planet Y
        
        # Standard MF choice at stage 2
        exp_q2 = np.exp(inverse_temperature * q_stage2_mf[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_likelihood += np.log(probs_2[chosen_a2] + eps)
        
        # --- Learning / Updates ---
        
        # Update Stage 1 MF values (TD learning using Stage 2 Q-value)
        # Note: We use the Q-value of the chosen stage 2 action (SARSA-like)
        delta_stage1 = q_stage2_mf[current_state, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        # Update Stage 2 MF values (Reward Prediction Error)
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[current_state, chosen_a2]
        q_stage2_mf[current_state, chosen_a2] += learning_rate * delta_stage2

    return -log_likelihood


def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hypothesis: Anxiety Increases "Stickiness" (Perseveration).
    STAI modulates a choice autocorrelation parameter. High anxiety leads to a higher
    tendency to repeat the previous Stage 1 choice, regardless of value.
    
    Parameters:
    - learning_rate: [0, 1]
    - inverse_temperature: [0, 10]
    - w: [0, 1] - Fixed mixing weight.
    - stick_base: [-2, 2] - Baseline stickiness.
    - stick_stai_slope: [-2, 2] - How STAI changes stickiness.
    
    Stickiness bonus = stick_base + stick_stai_slope * stai
    This bonus is added to the Q-value of the previously chosen action.
    """
    learning_rate, inverse_temperature, w, stick_base, stick_stai_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate effective stickiness
    stickiness = stick_base + (stick_stai_slope * stai_score)
    
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10
    last_action_1 = -1 # No previous action on trial 0

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        # Apply Stickiness Bonus
        # We create a copy of q_net to apply the bonus only for probability calculation
        # (we don't want to learn the stickiness into the Q-values)
        q_choice = q_net.copy()
        if last_action_1 != -1:
            q_choice[last_action_1] += stickiness
        
        exp_q = np.exp(inverse_temperature * q_choice)
        probs_1 = exp_q / np.sum(exp_q)
        
        chosen_a1 = int(action_1[trial])
        log_likelihood += np.log(probs_1[chosen_a1] + eps)
        
        # --- Stage 2 Decision ---
        current_state = int(state[trial])
        exp_q2 = np.exp(inverse_temperature * q_stage2_mf[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_likelihood += np.log(probs_2[chosen_a2] + eps)
        
        # --- Learning ---
        delta_stage1 = q_stage2_mf[current_state, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        r = reward[trial]
        delta_stage2 = r - q_stage2_mf[current_state, chosen_a2]
        q_stage2_mf[current_state, chosen_a2] += learning_rate * delta_stage2
        
        # Update history
        last_action_1 = chosen_a1

    return -log_likelihood


def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Hypothesis: Anxiety Alters Reward Sensitivity.
    STAI modulates the subjective valuation of the reward.
    Anxious individuals might perceive rewards as less valuable (blunting) or 
    be hypersensitive to outcomes.
    
    Parameters:
    - learning_rate: [0, 1]
    - inverse_temperature: [0, 10]
    - w: [0, 1]
    - sensitivity_base: [0, 2] - Baseline reward sensitivity.
    - sensitivity_stai_slope: [-2, 2] - How STAI changes sensitivity.
    
    Effective Reward = Reward * (sensitivity_base + sensitivity_stai_slope * stai)
    (Clipped to be non-negative to prevent sign flipping).
    """
    learning_rate, inverse_temperature, w, sensitivity_base, sensitivity_stai_slope = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]
    
    # Calculate subjective sensitivity
    sensitivity = sensitivity_base + (sensitivity_stai_slope * stai_score)
    # Ensure sensitivity doesn't invert the reward (e.g. making a win feel like a loss)
    if sensitivity < 0:
        sensitivity = 0.0
        
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    log_likelihood = 0.0
    eps = 1e-10

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        q_net = (w * q_stage1_mb) + ((1 - w) * q_stage1_mf)
        
        exp_q = np.exp(inverse_temperature * q_net)
        probs_1 = exp_q / np.sum(exp_q)
        
        chosen_a1 = int(action_1[trial])
        log_likelihood += np.log(probs_1[chosen_a1] + eps)
        
        # --- Stage 2 Decision ---
        current_state = int(state[trial])
        exp_q2 = np.exp(inverse_temperature * q_stage2_mf[current_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        
        chosen_a2 = int(action_2[trial])
        log_likelihood += np.log(probs_2[chosen_a2] + eps)
        
        # --- Learning ---
        # Update Stage 1
        delta_stage1 = q_stage2_mf[current_state, chosen_a2] - q_stage1_mf[chosen_a1]
        q_stage1_mf[chosen_a1] += learning_rate * delta_stage1
        
        # Update Stage 2 with SUBJECTIVE reward
        r_objective = reward[trial]
        r_subjective = r_objective * sensitivity
        
        delta_stage2 = r_subjective - q_stage2_mf[current_state, chosen_a2]
        q_stage2_mf[current_state, chosen_a2] += learning_rate * delta_stage2

    return -log_likelihood
