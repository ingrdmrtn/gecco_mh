Here are three new cognitive models that incorporate STAI anxiety scores into the decision-making process in distinct ways.

### Model 1: Anxiety-Modulated Reward Sensitivity
This model hypothesizes that anxiety blunts the perception of positive rewards (anhedonia-like effect) or amplifies the impact of lack of reward. High anxiety participants might perceive a "1" coin as less valuable, effectively lowering the learning signal for positive outcomes.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Reward Sensitivity Model.
    
    Hypothesis: Anxiety reduces the subjective valuation of rewards. High anxiety 
    participants effectively 'dampen' the reward signal, leading to slower 
    value accumulation for successful outcomes.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-based weight.
    - reward_damp: [0, 1] Factor by which anxiety reduces reward sensitivity.
      Effective reward = reward * (1 - reward_damp * stai).
    """
    learning_rate, beta, w, reward_damp = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate effective reward sensitivity based on anxiety
    # If stai is 1 and reward_damp is 1, reward becomes 0.
    # If stai is 0, reward is full.
    sensitivity = 1.0 - (reward_damp * stai_score)
    # Ensure sensitivity doesn't go below 0
    if sensitivity < 0: sensitivity = 0

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Update Stage 1
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Update Stage 2 with modulated reward
        effective_reward = reward[trial] * sensitivity
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Exploration Noise (Inverse Beta Modulation)
This model posits that anxiety acts as a distractor or a source of noise in the decision process. Instead of altering how values are learned, anxiety alters how values are converted into actions. Higher anxiety lowers the inverse temperature (`beta`), making choices more random (higher entropy) regardless of the learned values.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Exploration Noise Model.
    
    Hypothesis: Anxiety introduces decision noise (randomness). High anxiety 
    participants have a lower effective 'beta' (inverse temperature), making 
    their choices less deterministic and less driven by value differences.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_base: [0, 10] Baseline inverse temperature for low anxiety.
    - w: [0, 1] Model-based weight.
    - noise_scale: [0, 10] How much anxiety reduces beta.
      Effective beta = beta_base / (1 + noise_scale * stai).
    """
    learning_rate, beta_base, w, noise_scale = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate effective beta
    # As anxiety increases, the denominator increases, lowering beta (more random).
    effective_beta = beta_base / (1.0 + (noise_scale * stai_score))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(effective_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(effective_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Transition Learning (Dynamic Transition Matrix)
Standard models assume the transition matrix is fixed (70/30). This model hypothesizes that anxious individuals are hyper-vigilant to changes in environmental structure. They do not assume the transition probabilities are static; instead, they update their internal model of the spaceship-planet transitions trial-by-trial, with the rate of this update modulated by anxiety.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Transition Learning Model.
    
    Hypothesis: Anxious individuals do not rely on the fixed 70/30 transition 
    assumption. Instead, they actively learn the transition matrix (Model-Based 
    component) from experience. Anxiety modulates the learning rate of this 
    transition structure (hyper-vigilance to structural change).
    
    Parameters:
    - learning_rate_R: [0, 1] Learning rate for reward values (Q-values).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Model-based weight.
    - lr_transition_mod: [0, 1] How much anxiety increases the transition learning rate.
      Transition LR = 0.05 + (lr_transition_mod * stai).
    """
    learning_rate_R, beta, w, lr_transition_mod = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Base transition learning rate is small (0.05), anxiety adds to it.
    lr_transition = 0.05 + (lr_transition_mod * stai_score)
    if lr_transition > 1.0: lr_transition = 1.0

    # Initialize internal belief about transitions (start at true 0.7/0.3)
    # Row 0: Prob(State X | Action A), Prob(State Y | Action A)
    # Row 1: Prob(State X | Action U), Prob(State Y | Action U)
    # Note: The template uses a simplified matrix, here we track full state probs.
    # We map Action A (0) -> State X (0) usually. Action U (1) -> State Y (1) usually.
    
    # Internal model of transitions: T[action, next_state]
    internal_trans = np.array([[0.7, 0.3], [0.3, 0.7]])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        
        # Construct transition matrix for MB calculation based on current belief
        # The standard matrix is [[P(X|A), P(Y|A)], [P(X|U), P(Y|U)]]
        # But the template logic uses: transition_matrix @ max_q_stage2
        # where max_q_stage2 is vector [V(X), V(Y)].
        # So row 0 of matrix should be transitions for Action A.
        
        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = internal_trans @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Observe state transition
        chosen_action = action_1[trial]
        observed_state = state[trial] # 0 for X, 1 for Y
        
        # Update internal transition belief
        # We increase prob of observed transition, decrease others for that action
        # Create a 1-hot vector for the observed state
        outcome_vector = np.zeros(2)
        outcome_vector[observed_state] = 1.0
        
        # Delta rule for transition probabilities
        prediction_error_trans = outcome_vector - internal_trans[chosen_action]
        internal_trans[chosen_action] += lr_transition * prediction_error_trans
        
        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[observed_state])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[observed_state, action_2[trial]] - q_stage1_mf[chosen_action]
        q_stage1_mf[chosen_action] += learning_rate_R * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[observed_state, action_2[trial]]
        q_stage2_mf[observed_state, action_2[trial]] += learning_rate_R * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
