Here are three new cognitive models that incorporate the STAI anxiety score into the decision-making process in distinct ways.

### Model 1: Anxiety-Modulated Transition Learning
**Hypothesis:** High anxiety impairs the ability to update the internal model of the world (the transition matrix). While low-anxiety individuals update their belief about which spaceship goes to which planet based on observed transitions, high-anxiety individuals may have a "frozen" or rigid belief system, or update it much more slowly, relying more on the initial priors.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Transition Learning Model.
    
    Hypothesis: Anxiety dampens the learning rate for the state-transition matrix.
    High anxiety participants are less sensitive to changes in the transition structure
    (e.g., rare transitions becoming common), leading to rigid model-based values.
    
    Parameters:
    - learning_rate_reward: [0, 1] Learning rate for reward values (Q-values).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - lr_transition_base: [0, 1] Base learning rate for updating the transition matrix.
    - anxiety_damp: [0, 1] How much STAI reduces the transition learning rate.
    """
    learning_rate_reward, beta, w, lr_transition_base, anxiety_damp = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Effective transition learning rate is reduced by anxiety
    # If anxiety is 0, lr = base. If anxiety is 1, lr = base * (1 - damp)
    lr_transition = lr_transition_base * (1.0 - (anxiety_damp * stai_score))

    # Initialize transition matrix (0.7 common, 0.3 rare)
    # Rows: Action 1 (0=A, 1=U), Cols: State (0=X, 1=Y)
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State x Action

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Model-Based Value: Uses the dynamic transition matrix
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        
        # 1. Update Reward Q-values (Standard TD)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate_reward * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate_reward * delta_stage2
        
        # 2. Update Transition Matrix (State Prediction Error)
        # We observed action_1 leading to state_idx.
        # Create a one-hot vector for the observed state (0 or 1)
        observed_transition = np.zeros(2)
        observed_transition[state_idx] = 1.0
        
        # Update the row corresponding to the chosen action
        # T(s'|a) = T(s'|a) + lr_trans * (1_observed - T(s'|a))
        row = action_1[trial]
        prediction_error = observed_transition - transition_matrix[row]
        transition_matrix[row] += lr_transition * prediction_error
        
        # Normalize to ensure probabilities sum to 1
        transition_matrix[row] /= np.sum(transition_matrix[row])

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced "Safe Option" Bias
**Hypothesis:** Anxiety creates a specific bias against uncertainty. In this task, Spaceship A is often perceived as the "default" or "safe" option because it maps to Planet X (often listed first/left). High anxiety participants might have a baseline bias towards Action 0 (Spaceship A) regardless of value, treating it as a safety behavior to avoid the cognitive load of calculating the alternative.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced 'Safe Option' Bias Model.
    
    Hypothesis: Anxiety introduces a fixed bias towards Action 0 (Spaceship A) in Stage 1.
    This represents a 'safety behavior' or default heuristic often seen in anxious individuals
    who prefer familiar or default options over active exploration of alternatives.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - safety_bias_scale: [0, 5] Magnitude of the bias towards Action 0, scaled by STAI.
    """
    learning_rate, beta, w, safety_bias_scale = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate the bias added to Action 0
    safety_bias = safety_bias_scale * stai_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Add the safety bias specifically to Action 0 (Spaceship A)
        q_net[0] += safety_bias
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Negative Outcome Sensitivity
**Hypothesis:** Anxious individuals are hypersensitive to negative outcomes (omission of reward). Instead of a single learning rate, this model splits learning into positive (reward=1) and negative (reward=0) updates. Anxiety specifically amplifies the learning rate for negative outcomes, causing participants to abandon options quickly after a failure.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Negative Outcome Sensitivity Model.
    
    Hypothesis: Anxiety amplifies learning from negative outcomes (punishment/lack of reward).
    The model uses two learning rates: one for positive rewards (lr_pos) and one for 
    negative outcomes (lr_neg). lr_neg is boosted by the STAI score.
    
    Parameters:
    - lr_pos: [0, 1] Learning rate for positive outcomes (reward = 1).
    - lr_neg_base: [0, 1] Base learning rate for negative outcomes (reward = 0).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - anxiety_sens: [0, 5] How much STAI boosts the negative learning rate.
    """
    lr_pos, lr_neg_base, beta, w, anxiety_sens = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate effective negative learning rate
    # We clip at 1.0 to ensure stability
    lr_neg = min(1.0, lr_neg_base + (anxiety_sens * stai_score))

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        
        # Determine which learning rate to use based on the outcome of the trial
        current_lr = lr_pos if reward[trial] == 1 else lr_neg
        
        # Update Stage 1 MF
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += current_lr * delta_stage1
        
        # Update Stage 2 MF
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
