Here are three new cognitive models that incorporate STAI anxiety scores into the decision-making process in distinct ways.

### Model 1: Anxiety-Induced Model-Based Suppression
This model hypothesizes that high anxiety consumes cognitive resources (working memory), thereby suppressing the ability to use Model-Based (planning) strategies. Instead of a fixed weight `w`, the weight given to the model-based system is dynamically reduced as anxiety increases.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Model-Based Suppression.

    Hypothesis: High anxiety consumes cognitive resources (working memory), 
    making it harder to maintain and use the transition model. Therefore, 
    higher STAI scores reduce the weight (w) placed on Model-Based values, 
    defaulting more to Model-Free habits.

    Parameters:
    - learning_rate: [0, 1] Learning rate for value updates.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w_max: [0, 1] The maximum possible weight for Model-Based control (at 0 anxiety).
    - suppression_rate: [0, 5] How strongly anxiety reduces w_max.
    """
    learning_rate, beta, w_max, suppression_rate = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate effective w based on anxiety
    # As STAI increases, w decreases from w_max towards 0.
    w = w_max * np.exp(-suppression_rate * stai_score)
    
    # Ensure w stays within bounds [0, 1]
    w = np.clip(w, 0, 1)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)      # Model-free values for stage 1
    q_stage2_mf = np.zeros((2, 2)) # Model-free values for stage 2 (State x Action)

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based Calculation: Use transition matrix + max stage 2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Hybrid Value: Weighted sum of MB and MF
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Softmax choice probability
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial] # 0 for Planet X, 1 for Planet Y

        # --- Stage 2 Decision ---
        
        # Standard Softmax on Stage 2 Q-values
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Updates ---
        
        # Stage 1 MF Update (TD(0))
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 MF Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven Transition Learning Impairment
This model posits that anxiety interferes with the ability to learn the structure of the environment (the transition matrix). While the standard model assumes a fixed 70/30 transition matrix, this model assumes the agent learns the transitions, but anxiety acts as "noise" or a dampener on the learning rate for these state transitions.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Transition Learning Impairment.

    Hypothesis: Anxiety interferes with the learning of state transitions. 
    Instead of assuming a fixed transition matrix, the agent learns it. 
    High anxiety reduces the learning rate for state transitions (lr_transition),
    making the agent slower to adapt to the environmental structure or 
    less confident in the causal structure.

    Parameters:
    - learning_rate_reward: [0, 1] Learning rate for reward values (Q-values).
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - lr_transition_base: [0, 1] Base learning rate for state transitions.
    - anxiety_damp: [0, 5] How strongly anxiety dampens the transition learning rate.
    """
    learning_rate_reward, beta, w, lr_transition_base, anxiety_damp = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Modulate transition learning rate by anxiety
    # Higher anxiety -> Lower learning rate for structure
    lr_transition = lr_transition_base / (1 + anxiety_damp * stai_score)
    
    # Initialize estimated transition matrix (start uniform 0.5/0.5)
    # Rows: Action 1 (A or U), Cols: State (X or Y)
    # We track counts or probabilities. Here we update probabilities directly.
    trans_probs = np.array([[0.5, 0.5], [0.5, 0.5]]) 

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        
        # Model-Based: Use CURRENT estimated transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = trans_probs @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Observe State
        state_idx = state[trial]
        chosen_a1 = action_1[trial]

        # --- Update Transition Matrix ---
        # Move the probability of the observed transition towards 1.0
        # and the unobserved transition towards 0.0
        # Row chosen_a1 is updated.
        trans_probs[chosen_a1, state_idx] += lr_transition * (1 - trans_probs[chosen_a1, state_idx])
        trans_probs[chosen_a1, 1-state_idx] += lr_transition * (0 - trans_probs[chosen_a1, 1-state_idx])

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Reward Learning Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate_reward * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate_reward * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Modulated Loss Sensitivity
This model focuses on valence asymmetry. It hypothesizes that anxious individuals are hypersensitive to negative outcomes (losses or lack of reward). When a reward of 0 is received, the learning rate is boosted by the anxiety score, causing "bad" outcomes to have a larger impact on value updates than "good" outcomes.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Loss Sensitivity.

    Hypothesis: Anxiety increases sensitivity to negative prediction errors 
    (receiving 0 coins). When the outcome is worse than expected (negative delta), 
    the learning rate is effectively increased by a factor proportional to STAI, 
    causing the agent to unlearn 'bad' options faster than they learn 'good' ones.

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - loss_sens: [0, 5] Multiplier for how much anxiety boosts learning from negative errors.
    """
    learning_rate, beta, w, loss_sens = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Decision ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Decision ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Learning Updates ---
        
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update with Anxiety Modulation
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        
        # Determine effective learning rate for this specific update
        # If delta is negative (disappointment), boost LR based on anxiety
        if delta_stage2 < 0:
            effective_lr = learning_rate * (1 + loss_sens * stai_score)
            # Cap at 1.0 to prevent instability
            effective_lr = min(effective_lr, 1.0)
        else:
            effective_lr = learning_rate
            
        q_stage2_mf[state_idx, action_2[trial]] += effective_lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
