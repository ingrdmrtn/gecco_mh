Here are 3 new cognitive models implemented as Python functions.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Rare Transition Reward Discounting Model.

    Hypothesis: Anxious individuals may distrust outcomes that result from unexpected 
    events ("flukes"). When a reward is received following a Rare transition 
    (e.g., choosing A but ending up at Y), high anxiety leads to a discounting 
    of that reward signal during the update, effectively treating it as less informative.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - rare_discount: [0, 1] The percentage by which the learning rate is reduced 
      following a rare transition, scaled by STAI.
    """
    learning_rate, beta, w, rare_discount = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Fixed transition matrix for MB calculation: A->X (0.7), U->Y (0.7)
    # Indices: 0=A, 1=U; 0=X, 1=Y
    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    # Q-values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2)) # State (X,Y) x Action (Alien1, Alien2)

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial] # 0 for X, 1 for Y
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        # Determine if transition was Common or Rare
        # Common: (Action 0 -> State 0) or (Action 1 -> State 1)
        is_common = (action_1[trial] == state_idx)
        
        # Calculate effective learning rate for this trial
        if not is_common:
            # Rare transition: Discount the learning rate based on anxiety
            lr_eff = learning_rate * (1.0 - (rare_discount * stai_score))
            # Ensure non-negative
            lr_eff = max(0.0, lr_eff)
        else:
            lr_eff = learning_rate

        # Stage 1 Update (TD-0)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += lr_eff * delta_stage1
        
        # Stage 2 Update
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += lr_eff * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Transition Sharpening Model.

    Hypothesis: Anxious individuals are intolerant of uncertainty. To reduce cognitive 
    dissonance, they may mentally exaggerate the probability of common transitions 
    in their Model-Based planning. Instead of treating the transition as 70/30, 
    high anxiety leads them to treat it as more deterministic (e.g., 90/10), 
    distorting the MB value calculation.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - sharpening: [0, 1] The degree to which STAI pushes the perceived transition 
      probability from 0.7 toward 1.0.
    """
    learning_rate, beta, w, sharpening = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate Subjective Transition Matrix
    # Base is 0.7. Max is 1.0.
    # perceived_prob = 0.7 + (0.3 * sharpening * stai)
    base_prob = 0.7
    distortion = (1.0 - base_prob) * sharpening * stai_score
    p_common = base_prob + distortion
    p_rare = 1.0 - p_common
    
    # Subjective matrix used for planning
    subjective_matrix = np.array([[p_common, p_rare], [p_rare, p_common]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Use the distorted (sharpened) matrix for planning
        q_stage1_mb = subjective_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Post-Loss Rigidity Model.

    Hypothesis: Anxiety induces a "reactive rigidity." While standard models assume 
    a constant exploration/exploitation trade-off (beta), this model posits that 
    anxious individuals become significantly more rigid (higher beta, less exploration) 
    immediately following a loss (0 reward), attempting to regain control.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_base: [0, 10] Baseline inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - loss_rigidity: [0, 5] Increase in beta for the trial immediately following 
      a zero-reward outcome, scaled by STAI.
    """
    learning_rate, beta_base, w, loss_rigidity = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous reward to modulate current beta
    last_reward = 1.0 # Initialize as 1 (win) so no rigidity on first trial

    for trial in range(n_trials):
        
        # Calculate dynamic beta based on previous outcome
        # If last reward was 0 (loss), increase beta (rigidity)
        if last_reward == 0.0:
            current_beta = beta_base + (loss_rigidity * stai_score)
        else:
            current_beta = beta_base

        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # --- Stage 2 Policy ---
        state_idx = state[trial]
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
        
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        last_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
