Here are three cognitive models formulated as Python functions.

### Model 1: Anxiety-Driven Beta Decay (Exhaustion)
This model hypothesizes that high anxiety leads to faster cognitive fatigue. While the participant starts with a specific level of decision consistency (inverse temperature `beta`), this consistency decays over time proportional to their anxiety level, leading to more random behavior in later trials.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Beta Decay Model.

    Hypothesis: Anxiety accelerates cognitive fatigue. High anxiety participants 
    show a progressive decline in decision consistency (exploration/exploitation balance) 
    over the course of the task, becoming more random (lower beta) as trials progress.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_initial: [0, 10] Initial inverse temperature at trial 0.
    - w: [0, 1] Weight for model-based control (0=MF, 1=MB).
    - fatigue_rate: [0, 0.1] Rate at which beta decays per trial, scaled by STAI.
    """
    learning_rate, beta_initial, w, fatigue_rate = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # Calculate current beta based on fatigue and anxiety
        # Beta cannot go below 0
        current_beta = max(0.0, beta_initial * np.exp(-fatigue_rate * stai_score * trial))

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(current_beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(current_beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Driven MB-to-MF Consolidation
This model hypothesizes that anxious individuals attempt to reduce uncertainty by "caching" their model-based plans into model-free habits. High anxiety drives a transfer of value from the Model-Based system (transition-dependent) to the Model-Free system (cached value) at the end of every trial, effectively overwriting pure experience with calculated expectations.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven MB-to-MF Consolidation Model.

    Hypothesis: Anxiety drives a desire to "habitualize" complex plans to reduce 
    future cognitive load. Anxious agents actively pull their Model-Free (MF) values 
    towards their Model-Based (MB) estimates after every update, consolidating 
    the plan into a habit.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate for reward updates.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control during choice.
    - consolidation_rate: [0, 1] Rate at which MF values absorb MB values, scaled by STAI.
    """
    learning_rate, beta, w, consolidation_rate = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Standard TD updates
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        # Consolidation Step:
        # Recalculate MB values after the Stage 2 update
        new_max_q_stage2 = np.max(q_stage2_mf, axis=1)
        new_q_stage1_mb = transition_matrix @ new_max_q_stage2
        
        # MF values are nudged towards MB values based on anxiety
        # This represents the agent "memorizing" the plan
        transfer = consolidation_rate * stai_score
        q_stage1_mf = q_stage1_mf + transfer * (new_q_stage1_mb - q_stage1_mf)

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven Rare Outcome Filtering
This model suggests that anxious participants are intolerant of "outliers" or unexpected events. When a Rare transition occurs (e.g., Spaceship A goes to Planet Y), anxious agents suppress the learning rate for that trial, effectively treating the outcome as noise rather than a valid signal for updating their Stage 1 preferences.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Rare Outcome Filtering Model.

    Hypothesis: Anxious participants distrust outcomes resulting from unexpected 
    (Rare) transitions. If a transition is Rare, the learning rate for the 
    Stage 1 update is suppressed proportional to STAI. They learn normally 
    from Common transitions but ignore "outliers".

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - rare_suppression: [0, 1] How much to reduce LR on rare trials (1 = no learning).
    """
    learning_rate, beta, w, rare_suppression = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):

        # policy for the first choice
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        state_idx = state[trial]

        # policy for the second choice
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # Determine if transition was Common or Rare
        # Action 0 (A) -> State 0 (X) is Common
        # Action 1 (U) -> State 1 (Y) is Common
        is_common = (action_1[trial] == state_idx)
        
        # Calculate effective learning rate for Stage 1
        if is_common:
            eff_lr_s1 = learning_rate
        else:
            # Suppress learning if Rare and Anxiety is high
            eff_lr_s1 = learning_rate * (1.0 - (rare_suppression * stai_score))
            eff_lr_s1 = max(0.0, eff_lr_s1)

        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += eff_lr_s1 * delta_stage1
        
        # Stage 2 learning remains standard (bandit task is unambiguous)
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
