import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven Memory Reversion Model.

    Hypothesis: Anxiety consumes cognitive resources, leading to a specific type of memory decay 
    for unchosen options. Instead of decaying to 0, the values of unchosen actions revert 
    towards a neutral prior (0.5), representing a loss of specific value information and 
    a regression to the mean.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - reversion_rate: [0, 1] Rate at which unchosen Q-values drift to 0.5, scaled by STAI.
    """
    learning_rate, beta, w, reversion_rate = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate the effective reversion rate based on anxiety
    effective_reversion = reversion_rate * stai_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2) # Initialize at 0
    q_stage2_mf = np.zeros((2, 2)) # Initialize at 0

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        # 1. MF Update for Chosen Action (Stage 1)
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # 2. Reversion Update for UNCHOSEN Action (Stage 1)
        # The unchosen action drifts towards 0.5 (neutral expectation)
        unchosen_action = 1 - action_1[trial]
        q_stage1_mf[unchosen_action] += effective_reversion * (0.5 - q_stage1_mf[unchosen_action])

        # 3. MF Update for Stage 2
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Stage-Specific Anxiety Rigidity Model.

    Hypothesis: Anxiety manifests differently depending on the decision stage. 
    While Stage 1 allows for strategic exploration (controlled by base beta), 
    Stage 2 (immediate reward proximity) triggers a "panic" or "freeze" response 
    in anxious individuals, leading to significantly higher rigidity (exploitation) 
    specifically in the second stage.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta_stage1: [0, 10] Inverse temperature for the first decision.
    - w: [0, 1] Weight for model-based control.
    - rigidity_factor: [0, 5] Additional beta added to Stage 2 choices, scaled by STAI.
    """
    learning_rate, beta_stage1, w, rigidity_factor = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Stage 2 beta is boosted by anxiety
    beta_stage2 = beta_stage1 + (rigidity_factor * stai_score)

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy (Uses beta_stage1) ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta_stage1 * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy (Uses beta_stage2 - Rigidity) ---
        exp_q2 = np.exp(beta_stage2 * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Luck Discounting Model.

    Hypothesis: Anxious individuals may attribute positive outcomes from unexpected 
    (rare) events to "luck" rather than reliable structure. Consequently, they 
    discount the reward signal when it follows a rare transition, updating their 
    values less aggressively than when a reward follows a common transition.

    Parameters:
    - learning_rate: [0, 1] Base learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - luck_discount: [0, 1] Percentage reduction in learning rate for rare-transition rewards, scaled by STAI.
    """
    learning_rate, beta, w, luck_discount = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]

        # --- Determine if transition was Rare ---
        # Action 0 (A) -> State 0 (X) is Common
        # Action 1 (U) -> State 1 (Y) is Common
        # Anything else is Rare
        is_common = (action_1[trial] == state_idx)
        is_rare = not is_common

        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Calculate effective learning rate for Stage 2 update
        # If transition was rare and we got a reward, discount the learning
        current_lr = learning_rate
        if is_rare and reward[trial] == 1.0:
            discount_factor = luck_discount * stai_score
            current_lr = learning_rate * (1.0 - discount_factor)
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += current_lr * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
