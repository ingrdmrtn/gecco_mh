Here are three new cognitive models that incorporate STAI scores into the decision-making process in novel ways, focusing on mechanisms distinct from the previous iterations.

### Model 1: Anxiety-Modulated Reward Sensitivity (Anhedonia/Hypersensitivity)
This model hypothesizes that anxiety alters the subjective valuation of rewards. High anxiety might lead to blunted reward sensitivity (anhedonia-like) or heightened sensitivity to outcomes, scaling the effective reward signal before it updates values.

```python
import numpy as np

def cognitive_model1(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Modulated Reward Sensitivity Model.

    Hypothesis: Anxiety scales the subjective value of rewards. High anxiety participants 
    may perceive rewards as less valuable (blunting/anhedonia) or more valuable (hypersensitivity),
    affecting how strongly positive outcomes reinforce behavior.
    
    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature (exploration/exploitation).
    - w: [0, 1] Weight for model-based control (0=MF, 1=MB).
    - reward_scaling: [0, 2] A multiplier for the reward signal. 
      If < 1, anxiety blunts reward (anhedonia). If > 1, anxiety amplifies reward.
      The effective reward is reward * (1 + (reward_scaling - 1) * stai).
    """
    learning_rate, beta, w, reward_scaling = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate the subjective reward multiplier based on anxiety
    # If reward_scaling is 1.0, anxiety has no effect.
    # If reward_scaling is 0.5, high anxiety reduces reward value.
    # If reward_scaling is 1.5, high anxiety increases reward value.
    subjective_multiplier = 1.0 + (reward_scaling - 1.0) * stai_score

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        # Stage 1 Update
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        # Stage 2 Update with Anxiety-Modulated Reward
        effective_reward = reward[trial] * subjective_multiplier
        delta_stage2 = effective_reward - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 2: Anxiety-Induced Transition Uncertainty (Blurring)
This model posits that anxiety degrades the precision of the internal model of the environment. Instead of using the true transition probabilities (0.7/0.3), anxious individuals "blur" these probabilities towards chance (0.5/0.5), representing uncertainty or lack of confidence in the causal structure.

```python
import numpy as np

def cognitive_model2(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Induced Transition Uncertainty (Blurring) Model.

    Hypothesis: Anxiety degrades the precision of the Model-Based transition matrix.
    High anxiety participants act as if the transition probabilities are closer to 
    random (0.5/0.5) rather than the true structure (0.7/0.3), reducing the effectiveness 
    of model-based planning.

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature.
    - w: [0, 1] Weight for model-based control.
    - blur_param: [0, 1] How much anxiety blurs the transition matrix.
      0 = No blurring (uses 0.7). 1 = Full blurring (uses 0.5 for high anxiety).
    """
    learning_rate, beta, w, blur_param = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    # Calculate the subjective transition probability
    # True probability is 0.7. Random is 0.5.
    # Anxiety shifts the perceived probability from 0.7 down towards 0.5.
    # Effective P = 0.7 - (0.2 * blur_param * stai)
    # If blur_param is 0, P remains 0.7.
    # If blur_param is 1 and stai is 1, P becomes 0.5.
    
    true_prob = 0.7
    distortion = 0.2 * blur_param * stai_score
    perceived_prob = true_prob - distortion
    
    # Construct the subjective transition matrix
    # [[P, 1-P], [1-P, P]]
    transition_matrix = np.array([
        [perceived_prob, 1.0 - perceived_prob], 
        [1.0 - perceived_prob, perceived_prob]
    ])

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        
        # Use the distorted matrix for planning
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```

### Model 3: Anxiety-Driven "Win-Stay, Lose-Shift" (WSLS) Override
This model suggests that anxiety triggers a primitive, heuristic "Win-Stay, Lose-Shift" strategy that competes with the standard Reinforcement Learning mechanisms. High anxiety increases the reliance on this reactive heuristic, bypassing value calculation.

```python
import numpy as np

def cognitive_model3(action_1, state, action_2, reward, stai, model_parameters):
    """
    Anxiety-Driven WSLS Override Model.

    Hypothesis: Anxiety promotes reliance on a reactive "Win-Stay, Lose-Shift" (WSLS) 
    heuristic at Stage 1. This heuristic competes with the hybrid RL controller.
    High anxiety participants are more likely to simply repeat a choice if rewarded 
    and switch if unrewarded, ignoring the transition structure (Common/Rare).

    Parameters:
    - learning_rate: [0, 1] Standard Q-learning rate.
    - beta: [0, 10] Inverse temperature for the RL component.
    - w: [0, 1] Weight for model-based control within the RL component.
    - wsls_bias: [0, 5] The strength of the WSLS heuristic, scaled by STAI.
      This is added as a bias to the Q-values based on the previous trial's outcome.
    """
    learning_rate, beta, w, wsls_bias = model_parameters
    n_trials = len(action_1)
    stai_score = stai[0]

    transition_matrix = np.array([[0.7, 0.3], [0.3, 0.7]])
    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))
    
    # Track previous trial info for WSLS
    prev_action_1 = -1
    prev_reward = 0

    for trial in range(n_trials):
        # --- Stage 1 Policy ---
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_mb = transition_matrix @ max_q_stage2
        
        # Base RL value
        q_net = w * q_stage1_mb + (1 - w) * q_stage1_mf
        
        # Apply WSLS Bias if not the first trial
        if prev_action_1 != -1:
            heuristic_val = np.zeros(2)
            bias_magnitude = wsls_bias * stai_score
            
            if prev_reward == 1:
                # Win-Stay: Boost the previous action
                heuristic_val[prev_action_1] = bias_magnitude
            else:
                # Lose-Shift: Boost the OTHER action (or penalize previous)
                # Here we boost the alternative action
                other_action = 1 - prev_action_1
                heuristic_val[other_action] = bias_magnitude
            
            q_net += heuristic_val

        exp_q1 = np.exp(beta * q_net)
        probs_1 = exp_q1 / np.sum(exp_q1)
        p_choice_1[trial] = probs_1[action_1[trial]]
        
        # Store for next trial
        prev_action_1 = action_1[trial]
        state_idx = state[trial]

        # --- Stage 2 Policy ---
        exp_q2 = np.exp(beta * q_stage2_mf[state_idx])
        probs_2 = exp_q2 / np.sum(exp_q2)
        p_choice_2[trial] = probs_2[action_2[trial]]
  
        # --- Updates ---
        delta_stage1 = q_stage2_mf[state_idx, action_2[trial]] - q_stage1_mf[action_1[trial]]
        q_stage1_mf[action_1[trial]] += learning_rate * delta_stage1
        
        delta_stage2 = reward[trial] - q_stage2_mf[state_idx, action_2[trial]]
        q_stage2_mf[state_idx, action_2[trial]] += learning_rate * delta_stage2
        
        prev_reward = reward[trial]

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
```
