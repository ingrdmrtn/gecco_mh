def cognitive_model1(action_1, state, action_2, reward, model_parameters):
    """Adaptive hybrid (model-based/model-free) with learned transitions and uncertainty-gated MB control.
    
    This model learns second-stage values (model-free) and first-stage model-free values.
    It also learns the first-stage transition structure online, and blends MB and MF
    values at stage 1 with a trial-by-trial adaptive weight that decreases under
    value-uncertainty at stage 2.
    
    Parameters
    ----------
    action_1 : array-like of int (0 or 1)
        Chosen spaceship at stage 1 (0=A, 1=U) for each trial.
    state : array-like of int (0 or 1)
        Observed planet (0=X, 1=Y) reached after the stage-1 transition.
    action_2 : array-like of int (0 or 1)
        Chosen alien at stage 2 on the reached planet (0 or 1) for each trial.
    reward : array-like of float (typically 0 or 1)
        Reward outcome for each trial.
    model_parameters : tuple/list of floats
        (alpha, beta, omega0, kappa_unc, alpha_tr)
        - alpha in [0,1]: Learning rate for value updates (both stages).
        - beta in [0,10]: Inverse temperature for softmax at both stages.
        - omega0 in [0,1]: Baseline weight on model-based value at stage 1.
        - kappa_unc in [0,1]: Strength of uncertainty gating: higher values down-weight
                              model-based control when stage-2 values are uncertain.
        - alpha_tr in [0,1]: Learning rate for updating the transition probabilities.
    
    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    alpha, beta, omega0, kappa_unc, alpha_tr = model_parameters
    n_trials = len(action_1)

    # Initialize learned transition probabilities P(state | action1)
    # Start agnostic (0.5/0.5) and learn from experience.
    transition_matrix = np.array([[0.5, 0.5],
                                  [0.5, 0.5]], dtype=float)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Model-free value functions
    q_stage1_mf = np.zeros(2)          # for spaceship A/U
    q_stage2_mf = np.zeros((2, 2))     # for each planet (X/Y) and two aliens

    for t in range(n_trials):
        # Compute model-based stage-1 values via current learned transition model
        max_q_stage2 = np.max(q_stage2_mf, axis=1)  # best alien per planet
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Uncertainty of stage-2 values (across both planets), normalized to [0,1]
        # Entropy of softmax over the two aliens for each planet, then average.
        entropies = np.zeros(2)
        for s_idx in range(2):
            q2 = q_stage2_mf[s_idx, :]
            # Use a softmax policy at stage 2 to define a distribution, then entropy
            q2_centered = q2 - np.max(q2)
            exp_q2 = np.exp(beta * q2_centered)
            probs2 = exp_q2 / np.sum(exp_q2)
            # Entropy in nats; normalize by log(2) to be in [0,1] for 2 actions
            ent = -np.sum(probs2 * (np.log(probs2 + 1e-10)))
            entropies[s_idx] = ent / np.log(2.0)
        uncertainty = np.mean(entropies)  # in [0,1]

        # Adaptive MB weight: when uncertain, down-weight MB; when certain, up-weight
        omega_t = (1.0 - kappa_unc) * omega0 + kappa_unc * (1.0 - uncertainty)
        omega_t = min(1.0, max(0.0, omega_t))  # clamp numerically

        # Stage-1 choice probabilities
        q1 = (1.0 - omega_t) * q_stage1_mf + omega_t * q_stage1_mb
        q1c = q1 - np.max(q1)
        exp_q1 = np.exp(beta * q1c)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = action_1[t]
        p_choice_1[t] = probs_1[a1]

        # Stage-2 choice probabilities for the reached state
        s = state[t]
        q2 = q_stage2_mf[s, :]
        q2c = q2 - np.max(q2)
        exp_q2 = np.exp(beta * q2c)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = action_2[t]
        p_choice_2[t] = probs_2[a2]

        # Outcomes
        r = reward[t]

        # Learning updates
        # Stage-2 MF update
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta2

        # Stage-1 MF update bootstrapped from updated stage-2 value
        target1 = q_stage2_mf[s, a2]
        delta1 = target1 - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1

        # Update the transition model row for the chosen action a1 using delta rule
        # Move probability toward the observed state s.
        for sp in range(2):
            target = 1.0 if sp == s else 0.0
            transition_matrix[a1, sp] += alpha_tr * (target - transition_matrix[a1, sp])
        # Re-normalize row to guard against drift
        row_sum = np.sum(transition_matrix[a1, :])
        if row_sum > 0:
            transition_matrix[a1, :] /= row_sum

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model2(action_1, state, action_2, reward, model_parameters):
    """Model-free with asymmetric valence learning, stage-1 credit assignment, and a static stage-2 action bias.
    
    This model uses purely model-free control at both stages. Stage-2 learning rates
    differ for positive vs. negative reward prediction errors. A portion of the stage-2
    prediction error is credited back to the stage-1 chosen action. Additionally,
    there is a stable bias toward alien 0 (W on X, P on Y) at stage 2.
    
    Parameters
    ----------
    action_1 : array-like of int (0 or 1)
        Chosen spaceship at stage 1 (0=A, 1=U) for each trial.
    state : array-like of int (0 or 1)
        Observed planet (0=X, 1=Y) reached after the stage-1 transition.
    action_2 : array-like of int (0 or 1)
        Chosen alien at stage 2 on the reached planet (0 or 1) for each trial.
    reward : array-like of float (typically 0 or 1)
        Reward outcome for each trial.
    model_parameters : tuple/list of floats
        (alpha_pos, alpha_neg, beta, chi, bias2)
        - alpha_pos in [0,1]: Learning rate when the stage-2 prediction error is positive.
        - alpha_neg in [0,1]: Learning rate when the stage-2 prediction error is negative.
        - beta in [0,10]: Inverse temperature for softmax at both stages.
        - chi in [0,1]: Credit assignment weight from stage 2 back to the stage-1 MF value.
        - bias2 in [0,1]: Additive bias toward action 0 at stage 2 (applied to action values).
    
    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    alpha_pos, alpha_neg, beta, chi, bias2 = model_parameters
    n_trials = len(action_1)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    # Purely model-free action values
    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for t in range(n_trials):
        # Stage-1 policy from MF values only
        q1 = q_stage1_mf.copy()
        q1c = q1 - np.max(q1)
        exp_q1 = np.exp(beta * q1c)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = action_1[t]
        p_choice_1[t] = probs_1[a1]

        # Stage-2 policy with a static bias toward action 0
        s = state[t]
        q2 = q_stage2_mf[s, :].copy()
        q2[0] += bias2  # bias toward alien 0 (W on X, P on Y)
        q2c = q2 - np.max(q2)
        exp_q2 = np.exp(beta * q2c)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = action_2[t]
        p_choice_2[t] = probs_2[a2]

        # Outcome and prediction errors
        r = reward[t]
        pe2 = r - q_stage2_mf[s, a2]
        # Valence-asymmetric learning rate
        alpha2 = alpha_pos if pe2 > 0.0 else alpha_neg

        # Stage-2 update
        q_stage2_mf[s, a2] += alpha2 * pe2

        # Credit assignment from stage 2 to stage 1
        q_stage1_mf[a1] += chi * alpha2 * pe2

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss


def cognitive_model3(action_1, state, action_2, reward, model_parameters):
    """Successor-representation-inspired hybrid with decay (forgetting).
    
    Stage-1 values are a mixture of a learned successor-representation (here,
    learned action->state occupancy akin to the transition model) and model-free
    cached values. Stage-2 values are model-free. All values decay toward zero
    each trial, capturing forgetting or non-stationarity tracking costs.
    
    Parameters
    ----------
    action_1 : array-like of int (0 or 1)
        Chosen spaceship at stage 1 (0=A, 1=U) for each trial.
    state : array-like of int (0 or 1)
        Observed planet (0=X, 1=Y) reached after the stage-1 transition.
    action_2 : array-like of int (0 or 1)
        Chosen alien at stage 2 on the reached planet (0 or 1) for each trial.
    reward : array-like of float (typically 0 or 1)
        Reward outcome for each trial.
    model_parameters : tuple/list of floats
        (alpha, beta, sigma, w_sr, decay)
        - alpha in [0,1]: Learning rate for value updates (both stages).
        - beta in [0,10]: Inverse temperature for softmax at both stages.
        - sigma in [0,1]: Learning rate for the successor map (action->state occupancy).
        - w_sr in [0,1]: Weight on SR-derived value at stage 1 (vs MF cache).
        - decay in [0,1]: Per-trial forgetting toward zero for all Q-values.
    
    Returns
    -------
    float
        Negative log-likelihood of the observed choices under the model.
    """
    alpha, beta, sigma, w_sr, decay = model_parameters
    n_trials = len(action_1)

    # Successor map M[a, s]: expected immediate occupancy of state s after taking action a
    # Initialize near-agnostic and learn from transitions.
    M = np.array([[0.5, 0.5],
                  [0.5, 0.5]], dtype=float)

    p_choice_1 = np.zeros(n_trials)
    p_choice_2 = np.zeros(n_trials)

    q_stage1_mf = np.zeros(2)
    q_stage2_mf = np.zeros((2, 2))

    for t in range(n_trials):
        # Stage-1 SR-derived value: expected max second-stage value under learned M
        max_q_stage2 = np.max(q_stage2_mf, axis=1)
        q_stage1_sr = M @ max_q_stage2

        # Mixture of SR and MF at stage 1
        q1 = (1.0 - w_sr) * q_stage1_mf + w_sr * q_stage1_sr
        q1c = q1 - np.max(q1)
        exp_q1 = np.exp(beta * q1c)
        probs_1 = exp_q1 / np.sum(exp_q1)
        a1 = action_1[t]
        p_choice_1[t] = probs_1[a1]

        # Stage-2 policy for reached state
        s = state[t]
        q2 = q_stage2_mf[s, :]
        q2c = q2 - np.max(q2)
        exp_q2 = np.exp(beta * q2c)
        probs_2 = exp_q2 / np.sum(exp_q2)
        a2 = action_2[t]
        p_choice_2[t] = probs_2[a2]

        # Outcome
        r = reward[t]

        # Apply forgetting (decay) before updates
        q_stage1_mf *= (1.0 - decay)
        q_stage2_mf *= (1.0 - decay)

        # Stage-2 MF update
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta2

        # Stage-1 MF update bootstrapped from updated stage-2 value
        target1 = q_stage2_mf[s, a2]
        delta1 = target1 - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1

        # Update successor map for chosen action a1 toward the observed state s
        for sp in range(2):
            target = 1.0 if sp == s else 0.0
            M[a1, sp] += sigma * (target - M[a1, sp])
        # Re-normalize row to ensure it remains a probability distribution
        row_sum = np.sum(M[a1, :])
        if row_sum > 0:
            M[a1, :] /= row_sum

    eps = 1e-10
    log_loss = -(np.sum(np.log(p_choice_1 + eps)) + np.sum(np.log(p_choice_2 + eps)))
    return log_loss
