def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the perseverative model-free SARSA(λ) model
    (same logic as cognitive_model2), with a single inverse temperature for both stages
    and separate learning rates for stage 1 and stage 2. Second-stage states follow
    a standard two-step transition structure, and rewards are generated from provided
    drifting reward probabilities.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha1, alpha2, beta, lam, pers]
            - alpha1: learning rate for stage-1 values.
            - alpha2: learning rate for stage-2 values.
            - beta: inverse temperature for both stages.
            - lam: eligibility trace coefficient.
            - pers: perseveration strength added to previously chosen action's preference.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            second-stage actions:
              - state 0, action 0 -> drift1[t]
              - state 0, action 1 -> drift2[t]
              - state 1, action 0 -> drift3[t]
              - state 1, action 1 -> drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha1, alpha2, beta, lam, pers = parameters

    # Standard two-step transition structure (common=0.7, rare=0.3)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    # Initialize MF Q-values
    q_stage1 = np.zeros(2)        # actions at stage 1: A/U -> 0/1
    q_stage2 = np.zeros((2, 2))   # states 0/1 x actions 0/1 at stage 2

    # Perseveration trackers: previous choice at stage 1 and stage 2 (state-specific)
    prev_a1 = None
    prev_a2 = [None, None]

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Construct trial-specific reward probabilities
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Stage 1 preferences with perseveration bias
        pref1 = q_stage1.copy()
        if prev_a1 is not None:
            pref1[prev_a1] += pers

        # Softmax for stage 1
        pref1_centered = pref1 - np.max(pref1)
        exp1 = np.exp(beta * pref1_centered)
        p1 = exp1 / np.sum(exp1)

        a1 = rng.choice([0, 1], p=p1)

        # Transition to stage 2 state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 preferences with state-specific perseveration bias
        pref2 = q_stage2[s].copy()
        if prev_a2[s] is not None:
            pref2[prev_a2[s]] += pers

        # Softmax for stage 2 (same beta)
        pref2_centered = pref2 - np.max(pref2)
        exp2 = np.exp(beta * pref2_centered)
        p2 = exp2 / np.sum(exp2)

        a2 = rng.choice([0, 1], p=p2)

        # Generate reward
        r = int(rng.random() < reward_probs[s][a2])

        # SARSA(λ) MF updates
        delta2 = r - q_stage2[s, a2]
        q_stage2[s, a2] += alpha2 * delta2

        delta1_direct = q_stage2[s, a2] - q_stage1[a1]
        q_stage1[a1] += alpha1 * (delta1_direct + lam * delta2)

        # Update perseveration history
        prev_a1 = a1
        prev_a2[s] = a2

        # Record trial outcomes
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
