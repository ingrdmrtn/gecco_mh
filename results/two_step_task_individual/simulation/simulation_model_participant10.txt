def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the pure model-free 2-step model
    with asymmetric second-stage learning, PE gain to stage-1, and first-stage stay bias.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, pe_gain, beta, stay1]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage states (rows) by two actions (cols):
            [[drift1, drift2],
             [drift3, drift4]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    rng = np.random.default_rng()

    alpha_pos, alpha_neg, pe_gain, beta, stay1 = parameters

    # Task transition structure (common = 0.7, rare = 0.3)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Model-free Q-values
    q1 = np.zeros(2)          # first-stage Q-values
    q2 = np.zeros((2, 2))     # second-stage Q-values: Q2[state, action]

    # First-stage perseveration indicator (bias toward last chosen a1)
    pers_array = np.zeros(2)  # one-hot of last a1; zero on first trial

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Trial-specific reward probabilities for each state-action
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Stage 1: softmax over q1 with stay bias
        pref1 = q1 + stay1 * pers_array
        pref1 = pref1 - np.max(pref1)  # stability
        exp1 = np.exp(beta * pref1)
        p1 = exp1 / np.sum(exp1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2: softmax over q2 for visited state
        pref2 = q2[s2] - np.max(q2[s2])
        exp2 = np.exp(beta * pref2)
        p2 = exp2 / np.sum(exp2)
        a2 = rng.choice([0, 1], p=p2)

        # Reward outcome
        r = int(rng.random() < reward_probs[s2, a2])

        # Learning: asymmetric second-stage update
        td2 = r - q2[s2, a2]
        alpha2 = alpha_pos if td2 >= 0.0 else alpha_neg
        q2[s2, a2] += alpha2 * td2

        # First-stage MF update using the updated q2 value (as in fitting code)
        td1 = q2[s2, a2] - q1[a1]
        q1[a1] += alpha2 * td1

        # Propagate second-stage PE to first stage via gain
        q1[a1] += pe_gain * alpha2 * td2

        # Update perseveration (applies next trial)
        pers_array[:] = 0.0
        pers_array[a1] = 1.0

        # Log trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
