def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the pure model-free 2-step model
    with exponential decay (forgetting), risk-sensitive utility, and dual
    inverse temperatures, plus eligibility-like propagation scaled by decay.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta1, beta2, rho, d]
            - alpha: learning rate (0..1)
            - beta1: inverse temperature stage 1 (>=0)
            - beta2: inverse temperature stage 2 (>=0)
            - rho: utility curvature (0..1)
            - d: decay/forgetting rate (0..1), also scales eligibility term
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for (state 0, action 0), (state 0, action 1),
            (state 1, action 0), (state 1, action 1), respectively.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta1, beta2, rho, d = parameters

    rng = np.random.default_rng()

    # Standard two-step transition structure (common = 0.7)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize model-free Q-values
    q_stage1_mf = np.zeros(2)        # First-stage values for actions [0,1]
    q_stage2_mf = np.zeros((2, 2))   # Second-stage values: state x action

    # Allocate outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Apply exponential decay (forgetting) to all Q-values
        q_stage1_mf *= (1.0 - d)
        q_stage2_mf *= (1.0 - d)

        # Stage 1 choice via softmax
        q1 = q_stage1_mf
        q1_centered = q1 - np.max(q1)
        p_stage1 = np.exp(beta1 * q1_centered)
        p_stage1 /= np.sum(p_stage1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice via softmax in observed state
        q2 = q_stage2_mf[s2]
        q2_centered = q2 - np.max(q2)
        p_stage2 = np.exp(beta2 * q2_centered)
        p_stage2 /= np.sum(p_stage2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Trial-wise reward probabilities
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]
        r = int(rng.random() < reward_probs[s2][a2])

        # Risk-sensitive utility transform (rewards are >=0 here)
        u = r ** rho

        # Model-free updates with eligibility propagation scaled by d
        delta2 = u - q_stage2_mf[s2, a2]
        q_stage2_mf[s2, a2] += alpha * delta2

        delta1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1 + d * alpha * delta2

        # Store outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
