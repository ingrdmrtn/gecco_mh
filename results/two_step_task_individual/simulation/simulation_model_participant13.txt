import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with a single learning rate, single inverse temperature, eligibility trace, and
    first-stage perseveration (centered to [-1,1]).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w, lam, kappa]
            - alpha in [0,1]: learning rate for value updates (both stages).
            - beta in [0,10]: inverse temperature for both stages.
            - w in [0,1]: weight on model-based values at stage 1.
            - lam in [0,1]: eligibility trace mixing for bootstrapping with outcome.
            - kappa in [0,1]: first-stage perseveration strength; mapped to [-1,1] internally.
        drift1, drift2, drift3, drift4 (array-like): Trial-wise reward probabilities for
            the two aliens on each planet:
            - Planet X (state 0): aliens 0->drift1[t], 1->drift2[t]
            - Planet Y (state 1): aliens 0->drift3[t], 1->drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    alpha, beta, w, lam, kappa = parameters

    rng = np.random.default_rng()

    # Fixed transition structure (common=0.7)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize MF Q-values
    q_stage1_mf = np.zeros(2)        # actions at stage 1: A(0), U(1)
    q_stage2_mf = np.zeros((2, 2))   # states X(0), Y(1) and 2 aliens per state

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None  # for perseveration bias
    kappa_centered = 2.0 * (kappa - 0.5)  # map [0,1] -> [-1,1]

    # Convert drifts to array for convenient indexing
    drift1 = np.asarray(drift1)
    drift2 = np.asarray(drift2)
    drift3 = np.asarray(drift3)
    drift4 = np.asarray(drift4)

    for t in range(n_trials):
        # Trial-wise reward probabilities by state and action
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Compute model-based Q at stage 1 from MF stage-2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)          # length 2 (per state)
        q_stage1_mb = transition_matrix @ max_q_stage2       # length 2 (per action)

        # Hybrid valuation with perseveration bias toward previous stage-1 action
        q1 = w * q_stage1_mb + (1.0 - w) * q_stage1_mf
        if prev_a1 is not None:
            bias = np.zeros(2)
            bias[prev_a1] = kappa_centered
            q1 = q1 + bias

        # Stage 1 softmax
        q1_shift = q1 - np.max(q1)
        p1 = np.exp(beta * q1_shift)
        p1 /= np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 softmax over MF values
        q2 = q_stage2_mf[s2, :]
        q2_shift = q2 - np.max(q2)
        p2 = np.exp(beta * q2_shift)
        p2 /= np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Reward sampling from provided drifting probabilities
        r = int(rng.random() < reward_probs[s2, a2])

        # TD updates (MF) with eligibility trace
        delta2 = r - q_stage2_mf[s2, a2]
        delta1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]

        # Update second-stage MF action value
        q_stage2_mf[s2, a2] += alpha * delta2
        # Update first-stage MF action value with delta1 and eligibility from delta2
        q_stage1_mf[a1] += alpha * delta1 + alpha * lam * delta2

        # Store outputs and update perseveration
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
