def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the surprise-modulated arbitration model
    structure from cognitive_model2, with model-free learning at both stages and
    model-based planning at stage 1, plus a first-stage stay-bias.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta, omega0, gamma_surprise, stick1]
            - alpha: learning rate for MF values (both stages).
            - beta: inverse temperature for softmax (both stages).
            - omega0: baseline MB arbitration weight at stage 1.
            - gamma_surprise: gain on transition surprise (included for completeness; see note).
            - stick1: stay-bias added to the previously chosen first-stage action.
        drift1, drift2, drift3, drift4 (np.ndarray):
            Trial-wise reward probabilities for:
            - state 0 action 0, state 0 action 1,
            - state 1 action 0, state 1 action 1.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np
    rng = np.random.default_rng()

    alpha, beta, omega0, gamma_surprise, stick1 = parameters

    # Transition matrix (common = 0.7, rare = 0.3)
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]])

    # Initialize MF Q-values
    q1_mf = np.zeros(2)        # stage-1 MF values for two spaceships
    q2_mf = np.zeros((2, 2))   # stage-2 MF values for two states x two actions

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = -1  # for stay-bias

    for t in range(n_trials):
        # Compute MB action values at stage 1
        max_q2 = np.max(q2_mf, axis=1)   # best stage-2 action per state
        q1_mb = T @ max_q2               # MB value = expected value via transitions

        # Arbitration weight:
        # In the fitting code, surprise uses realized transitions to modulate w.
        # For generative simulation prior to observing the transition, we use the
        # baseline weight omega0 (surprise cannot be known yet).
        w = omega0

        # Combine MB and MF for stage-1 choice values
        q1 = (1 - w) * q1_mf + w * q1_mb

        # Apply stay-bias to previously chosen action at stage 1
        if prev_a1 in (0, 1):
            q1[prev_a1] += stick1

        # Softmax to choose stage-1 action
        q1c = q1 - np.max(q1)
        p1 = np.exp(beta * q1c)
        p1 = p1 / np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state conditional on a1
        s2 = rng.choice([0, 1], p=T[a1])

        # Stage-2 choice via softmax over MF Q-values for the visited state
        q2 = q2_mf[s2].copy()
        q2c = q2 - np.max(q2)
        p2 = np.exp(beta * q2c)
        p2 = p2 / np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Realize reward from drifting probabilities
        if s2 == 0:
            p_r = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_r = drift3[t] if a2 == 0 else drift4[t]
        r = int(rng.random() < p_r)

        # Learning updates (model-free) as in cognitive_model2
        pe2 = r - q2_mf[s2, a2]
        q2_mf[s2, a2] += alpha * pe2

        pe1 = r - q1_mf[a1]
        q1_mf[a1] += alpha * pe1

        # Log for next trial
        prev_a1 = a1

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
