def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using a model-free two-step model with:
    - Uncertainty bonus (UCB) at stage 2,
    - Reward sensitivity on outcomes,
    - Value forgetting toward zero at both stages.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, rho, c, delta]
            alpha in [0, 1]: MF learning rate.
            beta  in [0, 10]: inverse temperature (both stages).
            rho   in [0, 1]: reward sensitivity.
            c     in [0, 1]: UCB weight at stage 2.
            delta in [0, 1]: value forgetting rate toward 0 per trial.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage actions on each state:
            state 0: [drift1[t], drift2[t]]
            state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, rho, c, delta = parameters
    rng = np.random.default_rng()

    # Transition structure (common = 0.7)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize values and visit counts
    q_stage1 = np.zeros(2)          # MF Q at stage 1 (actions 0/1)
    q_stage2 = np.zeros((2, 2))     # MF Q at stage 2 (state 0/1 x action 0/1)
    visit_counts = np.zeros((2, 2)) # Visit counts for UCB at stage 2

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    eps = 1e-12

    for t in range(n_trials):
        # Apply forgetting toward zero before decisions
        q_stage1 *= (1.0 - delta)
        q_stage2 *= (1.0 - delta)

        # Stage 1 choice (purely model-free)
        q1 = q_stage1
        q1_shift = q1 - np.max(q1)
        exp_q1 = np.exp(beta * q1_shift)
        p_stage1 = exp_q1 / (np.sum(exp_q1) + eps)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice with UCB bonus
        n_sa = visit_counts[s2] + 1.0  # avoid division by zero
        bonus = c / np.sqrt(n_sa)
        q2_eff = q_stage2[s2] + bonus
        q2_shift = q2_eff - np.max(q2_eff)
        exp_q2 = np.exp(beta * q2_shift)
        p_stage2 = exp_q2 / (np.sum(exp_q2) + eps)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Generate reward from drifting probabilities
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]
        r = int(rng.random() < reward_probs[s2][a2])

        # Reward sensitivity
        u = rho * r

        # Learning updates (model-free with eligibility to stage 1 via target = q_stage2[s2,a2])
        pe2 = u - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha * pe2

        target1 = q_stage2[s2, a2]
        pe1 = target1 - q_stage1[a1]
        q_stage1[a1] += alpha * pe1

        # Update visit counts for UCB
        visit_counts[s2, a2] += 1.0

        # Record trial
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
