def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with a single learning rate (alpha), single inverse temperature (beta),
    an eligibility trace (lam), and perseveration at both stages (p_stick),
    matching cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta, w, lam, p_stick]
            - alpha: learning rate for both stages.
            - beta: inverse temperature used at both stages.
            - w: MB/MF mixture weight at stage 1 (1 = purely MB, 0 = purely MF).
            - lam: eligibility trace strength propagating stage-2 PE to stage-1.
            - p_stick: perseveration bonus added to the previously chosen action (per stage).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            state=0: actions 0,1 -> drift1, drift2
            state=1: actions 0,1 -> drift3, drift4

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w, lam, p_stick = parameters

    rng = np.random.default_rng()

    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Model-free Q-values
    q_stage1_mf = np.zeros(2)        # First-stage actions (2)
    q_stage2_mf = np.zeros((2, 2))   # Second-stage states x actions

    # Perseveration tracking: previous action at stage 1 and per state at stage 2
    prev_a1 = -1
    prev_a2 = np.array([-1, -1], dtype=int)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Trial-wise reward probabilities by state and action
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Model-based first-stage Q via planning with current MF second-stage values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)        # best option per second-stage state
        q_stage1_mb = transition_matrix @ max_q_stage2    # expected value of each first-stage action

        # Hybrid Q for stage 1
        q1 = w * q_stage1_mb + (1.0 - w) * q_stage1_mf

        # Add perseveration bias at stage 1
        if prev_a1 in (0, 1):
            q1[prev_a1] += p_stick

        # Softmax choice at stage 1
        logits1 = beta * (q1 - np.max(q1))
        p1 = np.exp(logits1)
        p1 /= p1.sum()
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 Q for the reached state, with perseveration
        q2 = q_stage2_mf[s].copy()
        if prev_a2[s] in (0, 1):
            q2[prev_a2[s]] += p_stick

        # Softmax choice at stage 2
        logits2 = beta * (q2 - np.max(q2))
        p2 = np.exp(logits2)
        p2 /= p2.sum()
        a2 = rng.choice([0, 1], p=p2)

        # Outcome
        r_prob = reward_probs[s][a2]
        r = int(rng.random() < r_prob)

        # Store choices/outcomes
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

        # Learning updates (match cognitive_model1)
        q2_old = q_stage2_mf[s, a2]
        delta2 = r - q2_old
        q_stage2_mf[s, a2] = q2_old + alpha * delta2

        # Stage-1 MF update with eligibility trace
        q_stage1_mf[a1] = q_stage1_mf[a1] + alpha * (q2_old - q_stage1_mf[a1]) + alpha * lam * (r - q2_old)

        # Update perseveration trackers
        prev_a1 = a1
        prev_a2[s] = a2

    return stage1_choice, state2, stage2_choice, reward
