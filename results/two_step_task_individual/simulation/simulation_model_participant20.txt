def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the transition-learning model-based planner
    with confidence-weighted planning and Stage-1 perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha2, beta, alpha_T, chi_conf, rho1_stay]
            - alpha2 in [0,1]: learning rate for Stage-2 Q-values.
            - beta in [0,10]: inverse temperature for softmax at both stages.
            - alpha_T in [0,1]: learning rate for the learned transition matrix T_hat.
            - chi_conf in [0,1]: confidence-sensitivity; higher values reduce planning when transitions are uncertain.
            - rho1_stay in [0,1]: strength of Stage-1 perseveration (bias to repeat last first-stage action).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state 0 action 0, state 0 action 1, state 1 action 0, state 1 action 1.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha2, beta, alpha_T, chi_conf, rho1_stay = parameters

    # True environment transition matrix (common vs rare)
    env_T = np.array([[0.7, 0.3],
                      [0.3, 0.7]], dtype=float)

    # Agent's learned transition model (starts uninformative)
    T_hat = np.array([[0.5, 0.5],
                      [0.5, 0.5]], dtype=float)

    # Stage-2 MF values initialized to 0.5
    q2 = np.zeros((2, 2), dtype=float) + 0.5

    # Perseveration tracker
    last_a1 = -1

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    rng = np.random.default_rng()
    eps = 1e-12

    for t in range(n_trials):
        # Confidence scaling per first-stage option based on entropy of learned transitions
        conf_scale = np.zeros(2, dtype=float)
        for i in range(2):
            row = T_hat[i].copy()
            row = np.clip(row, eps, 1.0)
            row /= np.sum(row)
            H = -np.sum(row * np.log(row))        # entropy in nats, in [0, ln 2]
            H_norm = H / np.log(2.0)              # normalize to [0,1]
            conf_scale[i] = 1.0 - chi_conf * H_norm

        # Model-based action values using learned transitions
        max_q2 = np.max(q2, axis=1)               # best available value per second-stage state
        q1_mb = T_hat @ max_q2                    # MB value for each first-stage action

        # Apply confidence scaling and perseveration
        logits1 = conf_scale * q1_mb
        if last_a1 >= 0:
            stick = np.zeros(2, dtype=float)
            stick[last_a1] = rho1_stay
            logits1 = logits1 + stick

        # Softmax choice at Stage 1
        l1 = beta * (logits1 - np.max(logits1))
        p1_unnorm = np.exp(l1)
        p1 = p1_unnorm / (np.sum(p1_unnorm) + eps)
        a1 = rng.choice([0, 1], p=p1)

        # State transition according to true environment
        s2 = rng.choice([0, 1], p=env_T[a1])

        # Stage-2 choice softmax
        l2 = beta * (q2[s2] - np.max(q2[s2]))
        p2_unnorm = np.exp(l2)
        p2 = p2_unnorm / (np.sum(p2_unnorm) + eps)
        a2 = rng.choice([0, 1], p=p2)

        # Generate reward from drifting probabilities
        if s2 == 0:
            p_rew = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_rew = drift3[t] if a2 == 0 else drift4[t]
        r = int(rng.random() < p_rew)

        # Learning updates
        pe2 = r - q2[s2, a2]
        q2[s2, a2] += alpha2 * pe2

        target_vec = np.array([0.0, 0.0], dtype=float)
        target_vec[s2] = 1.0
        T_hat[a1] = (1.0 - alpha_T) * T_hat[a1] + alpha_T * target_vec
        T_hat[a1] = T_hat[a1] / (np.sum(T_hat[a1]) + eps)

        # Log trial
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        last_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
