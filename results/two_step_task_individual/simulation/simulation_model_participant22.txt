def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the successor-representation augmented hybrid model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha2, eta, beta, omega, stick1]
            - alpha2: learning rate for second-stage Q-values and model-free stage-1 value.
            - eta: learning rate for the successor map (action->state occupancy probabilities).
            - beta: inverse temperature for softmax at both stages.
            - omega: weight on SR-derived model-based value in stage-1.
            - stick1: additive bias for repeating the previous first-stage action.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            - state 0, action 0: drift1
            - state 0, action 1: drift2
            - state 1, action 0: drift3
            - state 1, action 1: drift4

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha2, eta, beta, omega, stick1 = parameters

    # Environment transition structure (true, used to generate states)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    # Initialize value functions and SR map
    q2 = np.zeros((2, 2))          # second-stage Q-values: state x action
    q1_mf = np.zeros(2)            # model-free first-stage values
    M = np.full((2, 2), 0.5)       # SR-like action->state occupancy (rows: a1, cols: s2)

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None

    for t in range(n_trials):
        # Compute SR-derived model-based component for stage 1
        max_q2 = np.max(q2, axis=1)   # best action in each second-stage state
        q1_mb = M @ max_q2            # expected value of each first-stage action under learned M

        # Blend MF and SR-MB, add stickiness bias
        q1_net = (1.0 - omega) * q1_mf + omega * q1_mb
        if prev_a1 is not None:
            bias = np.zeros(2)
            bias[prev_a1] += stick1
            q1_net = q1_net + bias

        # Stage-1 choice via softmax
        q1_shift = q1_net - np.max(q1_net)
        p_stage1 = np.exp(beta * q1_shift)
        p_stage1 = p_stage1 / np.sum(p_stage1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state from environment
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 choice via softmax on q2[s2]
        q2_s = q2[s2].copy()
        q2_shift = q2_s - np.max(q2_s)
        p_stage2 = np.exp(beta * q2_shift)
        p_stage2 = p_stage2 / np.sum(p_stage2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Reward based on drifting probabilities
        if s2 == 0:
            p_rew = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_rew = drift3[t] if a2 == 0 else drift4[t]
        r = int(rng.random() < p_rew)

        # Learning updates
        # Second-stage TD update
        delta2 = r - q2[s2, a2]
        q2[s2, a2] += alpha2 * delta2

        # Model-free first-stage update toward obtained second-stage value
        target1 = q2[s2, a2]
        q1_mf[a1] += alpha2 * (target1 - q1_mf[a1])

        # Update SR-like action->state mapping for chosen first-stage action
        one_hot_s = np.array([1.0 if i == s2 else 0.0 for i in range(2)])
        M[a1, :] += eta * (one_hot_s - M[a1, :])

        # Save and proceed
        prev_a1 = a1

        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
