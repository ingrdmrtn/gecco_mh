def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards for the hybrid MB/MF model with a single beta,
    single learning rate, eligibility trace, and choice stickiness at both stages.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w, lam, kappa]
            - alpha: learning rate for MF updates at both stages.
            - beta: inverse temperature for both stages.
            - w: MB/MF weight at stage 1 (w=1 pure MB, w=0 pure MF).
            - lam: eligibility trace parameter.
            - kappa: choice stickiness strength (bias to repeat previous choice).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            second-stage states and actions arranged as:
                state 0: [drift1[t], drift2[t]]
                state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w, lam, kappa = parameters

    rng = np.random.default_rng()

    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize MF value functions
    q_stage1_mf = np.zeros(2)          # first-stage MF values for actions [0,1]
    q_stage2_mf = np.zeros((2, 2))     # second-stage MF values [state, action]

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Choice stickiness memory (previous actions)
    prev_a1 = None
    prev_a2 = None

    for t in range(n_trials):
        # Trial-wise reward probabilities for second-stage [state, action]
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Model-based value at stage 1 via transition model and max over stage-2 Q
        max_q_stage2 = np.max(q_stage2_mf, axis=1)  # best action value in each state
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Hybrid value for stage 1
        q1 = w * q_stage1_mb + (1.0 - w) * q_stage1_mf

        # Stage-1 stickiness
        stickiness1 = np.zeros(2)
        if prev_a1 is not None:
            stickiness1[prev_a1] += kappa

        # Softmax for stage 1
        logits1 = beta * q1 + stickiness1
        logits1 -= np.max(logits1)
        probs1 = np.exp(logits1)
        probs1 /= probs1.sum()

        a1 = rng.choice([0, 1], p=probs1)

        # Sample second-stage state given first-stage choice
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 stickiness (global last second-stage action, not state-specific)
        stickiness2 = np.zeros(2)
        if prev_a2 is not None:
            stickiness2[prev_a2] += kappa

        # Softmax for stage 2 using MF Q-values for the reached state
        q2 = q_stage2_mf[s]
        logits2 = beta * q2 + stickiness2
        logits2 -= np.max(logits2)
        probs2 = np.exp(logits2)
        probs2 /= probs2.sum()

        a2 = rng.choice([0, 1], p=probs2)

        # Outcome
        r = int(rng.random() < reward_probs[s, a2])

        # Learning updates
        # Second-stage TD update
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta2

        # First-stage TD update (bootstrapping off updated second-stage value)
        delta1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1

        # Eligibility trace: propagate stage-2 PE to stage 1
        q_stage1_mf[a1] += alpha * lam * delta2

        # Update stickiness memory
        prev_a1 = a1
        prev_a2 = a2

        # Record
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
