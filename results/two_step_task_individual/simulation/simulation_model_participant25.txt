import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with eligibility traces and first-stage perseveration, matching cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w, lam, kappa]
            - alpha: learning rate for MF updates at both stages.
            - beta: inverse temperature for softmax at both stages.
            - w: weight of model-based control at stage 1.
            - lam: eligibility trace parameter for propagating second-stage PE to stage 1 MF.
            - kappa: first-stage perseveration bias.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            second-stage options:
            [[drift1[t], drift2[t]],  # state X (0)
             [drift3[t], drift4[t]]]  # state Y (1)

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    alpha, beta, w, lam, kappa = parameters
    rng = np.random.default_rng()

    transition_matrix = np.array([
        [0.7, 0.3],  # from A (0) to X(0), Y(1)
        [0.3, 0.7]   # from U (1) to X(0), Y(1)
    ])

    # MF values
    q_stage1_mf = np.zeros(2)        # first-stage actions A,U
    q_stage2_mf = np.zeros((2, 2))   # second-stage states X,Y each with two actions

    # Perseveration vector (one-hot of last first-stage choice)
    s1_prev = np.zeros(2)

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Construct current trial reward probabilities for second-stage options
        reward_probs = np.array([
            [drift1[t], drift2[t]],  # state X
            [drift3[t], drift4[t]]   # state Y
        ])

        # Model-based values: expected best value on each next-state given transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)  # best alien in X and Y
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Combine MB, MF, and perseveration at stage 1
        q1_combined = w * q_stage1_mb + (1.0 - w) * q_stage1_mf + kappa * s1_prev

        # Stage-1 choice via softmax with single beta
        q1_shift = q1_combined - np.max(q1_combined)
        p1 = np.exp(beta * q1_shift)
        p1 /= p1.sum()
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 choice via softmax with same beta
        q2 = q_stage2_mf[s, :]
        q2_shift = q2 - np.max(q2)
        p2 = np.exp(beta * q2_shift)
        p2 /= p2.sum()
        a2 = rng.choice([0, 1], p=p2)

        # Generate reward from drifting probabilities
        r = int(rng.random() < reward_probs[s, a2])

        # Learning updates (match fitting code order and formulas)
        # 1) Update second-stage MF
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta2

        # 2) Update first-stage MF with bootstrapped value and eligibility trace
        bootstrapped_value = q_stage2_mf[s, a2]  # after stage-2 update
        delta1_boot = bootstrapped_value - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1_boot + alpha * lam * delta2

        # Update perseveration trace
        s1_prev.fill(0.0)
        s1_prev[a1] = 1.0

        # Record trial
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
