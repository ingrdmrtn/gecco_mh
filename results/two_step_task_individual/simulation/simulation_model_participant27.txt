import numpy as np

def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with eligibility traces and first-stage perseveration, matching cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w, lam, kappa]
            - alpha: learning rate for MF values in both stages [0,1]
            - beta: inverse temperature for softmax at both stages [0,10]
            - w: weight of model-based relative to model-free at stage 1 [0,1]
            - lam: eligibility trace parameter λ for bootstrapping MF(1) from δ2 [0,1]
            - kappa: perseveration bias (adds +kappa to last chosen action and -kappa to the other)
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the
            two second-stage actions on each state:
            - state 0: [drift1[t], drift2[t]]
            - state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    alpha, beta, w, lam, kappa = parameters

    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Model-free values
    q_stage1_mf = np.zeros(2)        # first-stage actions (2 spaceships)
    q_stage2_mf = np.zeros((2, 2))   # second-stage actions per state (2 states x 2 actions)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None

    for t in range(n_trials):
        # Trial-wise reward probabilities for second-stage choices
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]], dtype=float)

        # Model-based value: expected value of each first-stage action via transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)   # best action per state
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Blend MB and MF for stage 1
        q1 = w * q_stage1_mb + (1.0 - w) * q_stage1_mf

        # First-stage perseveration (+kappa to previous choice, -kappa to other)
        if prev_a1 is not None:
            stickiness = np.array([-kappa, -kappa])
            stickiness[prev_a1] += 2.0 * kappa
            q1 = q1 + stickiness

        # Stage 1 choice via softmax
        q1_shift = q1 - np.max(q1)
        p1 = np.exp(beta * q1_shift)
        p1 = p1 / (np.sum(p1) + 1e-12)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice via softmax over MF values in observed state
        q2 = q_stage2_mf[s].copy()
        q2_shift = q2 - np.max(q2)
        p2 = np.exp(beta * q2_shift)
        p2 = p2 / (np.sum(p2) + 1e-12)
        a2 = rng.choice([0, 1], p=p2)

        # Bernoulli reward from drift probabilities
        r = int(rng.random() < reward_probs[s, a2])

        # Learning updates (same alpha and beta at both stages)
        # Second-stage TD error and update
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta2

        # First-stage MF bootstrapping from second-stage value
        delta1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1

        # Eligibility trace: propagate delta2 back to first-stage MF
        q_stage1_mf[a1] += alpha * lam * delta2

        # Store outputs and update perseveration memory
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
