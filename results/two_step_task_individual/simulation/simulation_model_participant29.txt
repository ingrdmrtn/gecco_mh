def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards for a purely model-based planner with:
      - uncertainty bonus at stage 2,
      - stage-1 stickiness,
      - forgetting (decay) of second-stage Q-values.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, stick1, upsilon, nu]
            - alpha: learning rate for Q2 in [0,1]
            - beta: inverse temperature used at both stages
            - stick1: perseveration weight on previous stage-1 action
            - upsilon: weight of UCB-like uncertainty bonus at stage 2
            - nu: forgetting rate for Q2 in [0,1]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two actions in each second-stage state.
            Reward probability layout per trial t:
                state 0: [drift1[t], drift2[t]]
                state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, stick1, upsilon, nu = parameters

    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize value and visit structures
    q2 = np.zeros((2, 2), dtype=float)     # second-stage Q-values
    visits = np.zeros((2, 2), dtype=float) # visit counts for UCB bonus

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None

    for t in range(n_trials):
        # Construct reward probabilities for this trial
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Uncertainty bonus (UCB-like): upsilon / sqrt(visit_count + 1)
        bonus = upsilon / np.sqrt(visits + 1.0)  # shape (2,2)

        # Model-based planning at stage 1:
        # Use max over stage-2 actions of (Q2 + bonus) for each state
        max_q2_bonus = np.max(q2 + bonus, axis=1)  # shape (2,)
        q1 = transition_matrix @ max_q2_bonus      # shape (2,)

        # Add stage-1 stickiness to previously chosen action
        if prev_a1 is not None:
            stick_vec = np.zeros(2)
            stick_vec[prev_a1] = 1.0
            q1 = q1 + stick1 * stick_vec

        # Softmax to choose stage-1 action
        q1_shift = q1 - np.max(q1)
        p_stage1_unnorm = np.exp(beta * q1_shift)
        p_stage1 = p_stage1_unnorm / np.sum(p_stage1_unnorm)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 choice uses Q2[s] + bonus[s]
        q2_s = q2[s2] + bonus[s2]
        q2_s_shift = q2_s - np.max(q2_s)
        p_stage2_unnorm = np.exp(beta * q2_s_shift)
        p_stage2 = p_stage2_unnorm / np.sum(p_stage2_unnorm)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Sample reward
        r = int(rng.random() < reward_probs[s2][a2])

        # Forgetting (decay) of all second-stage Q-values
        q2 *= (1.0 - nu)

        # Update chosen second-stage Q-value with reward prediction error
        delta2 = r - q2[s2, a2]
        q2[s2, a2] += alpha * delta2

        # Update visit counts for UCB bonus
        visits[s2, a2] += 1.0

        # Store choices and outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
