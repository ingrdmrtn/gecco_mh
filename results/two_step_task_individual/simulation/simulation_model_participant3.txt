def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4
):
    """
    Simulates choices and rewards using the pure model-free RL with asymmetric learning rates,
    forgetting, and first-stage stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha_pos, alpha_neg, beta, rho, kappa]
            - alpha_pos: learning rate for nonnegative prediction errors
            - alpha_neg: learning rate for negative prediction errors
            - beta: inverse temperature used at both stages
            - rho: forgetting rate applied to all Q-values each trial
            - kappa: first-stage stickiness bias added to last chosen action
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for second-stage states 0 and 1, actions 0 and 1 respectively:
            state 0 -> actions [drift1, drift2]
            state 1 -> actions [drift3, drift4]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha_pos, alpha_neg, beta, rho, kappa = parameters
    rng = np.random.default_rng()

    # Standard two-step transition structure: common = 0.7, rare = 0.3
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    q1 = np.zeros(2)        # first-stage actions
    q2 = np.zeros((2, 2))   # second-stage (state x action)

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    last_a1 = None  # for stickiness

    for t in range(n_trials):
        # Apply uniform forgetting/decay before making choices (as in fitting code)
        q1 *= (1.0 - rho)
        q2 *= (1.0 - rho)

        # First-stage stickiness bias
        bias1 = np.zeros(2)
        if last_a1 is not None:
            bias1[last_a1] += kappa

        # Stage 1 choice
        q1_bias = q1 + bias1
        q1_centered = q1_bias - np.max(q1_bias)
        p_stage1_unnorm = np.exp(beta * q1_centered)
        p_stage1 = p_stage1_unnorm / np.sum(p_stage1_unnorm)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice
        q2_state = q2[s2, :]
        q2_centered = q2_state - np.max(q2_state)
        p_stage2_unnorm = np.exp(beta * q2_centered)
        p_stage2 = p_stage2_unnorm / np.sum(p_stage2_unnorm)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Reward sampling from provided drifting probabilities
        # state 0 -> actions [drift1, drift2]; state 1 -> actions [drift3, drift4]
        if s2 == 0:
            r_prob = drift1[t] if a2 == 0 else drift2[t]
        else:
            r_prob = drift3[t] if a2 == 0 else drift4[t]
        r = int(rng.random() < r_prob)

        # Learning updates with asymmetric learning rates
        # Second-stage update
        delta2 = r - q2[s2, a2]
        lr2 = alpha_pos if delta2 >= 0.0 else alpha_neg
        q2[s2, a2] += lr2 * delta2

        # First-stage update towards updated second-stage Q-value
        delta1 = q2[s2, a2] - q1[a1]
        lr1 = alpha_pos if delta1 >= 0.0 else alpha_neg
        q1[a1] += lr1 * delta1

        # Update stickiness tracker
        last_a1 = a1

        # Record trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
