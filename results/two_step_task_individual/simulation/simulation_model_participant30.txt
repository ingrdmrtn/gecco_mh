def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a model-based RL agent that:
    - Learns first-stage transition probabilities via a delta rule.
    - Learns second-stage Q-values model-free.
    - Exhibits choice stickiness at both stages.
    - Uses a single inverse temperature for both stages.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alphaQ, alphaT, beta, kappa1, kappa2]
            - alphaQ: learning rate for second-stage Q-values.
            - alphaT: learning rate for transition probabilities.
            - beta: inverse temperature for both stages.
            - kappa1: first-stage stickiness weight.
            - kappa2: second-stage stickiness weight (state-dependent).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the 2nd stage:
            - state 0: actions 0,1 -> drift1[t], drift2[t]
            - state 1: actions 0,1 -> drift3[t], drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alphaQ, alphaT, beta, kappa1, kappa2 = parameters

    # True environment transition matrix (used to generate states)
    true_T = np.array([[0.7, 0.3],
                       [0.3, 0.7]])

    rng = np.random.default_rng()

    # Agent's learned transition model T (initialized uniform)
    T = np.full((2, 2), 0.5)

    # Second-stage Q-values (state x action)
    Q2 = np.zeros((2, 2))

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Stickiness memory
    prev_a1 = None
    prev_a2_by_state = [None, None]

    for t in range(n_trials):
        # Compute model-based first-stage action values using learned transitions
        maxQ_by_state = np.max(Q2, axis=1)   # best action value in each state
        Q1_mb = T @ maxQ_by_state            # expected value per first-stage action

        # Add first-stage stickiness
        pref1 = Q1_mb.copy()
        if prev_a1 is not None:
            bias = np.zeros(2)
            bias[prev_a1] = 1.0
            pref1 = pref1 + kappa1 * bias

        # Softmax for stage-1 choice
        pref1c = pref1 - np.max(pref1)
        p1_unnorm = np.exp(beta * pref1c)
        p_stage1 = p1_unnorm / np.sum(p1_unnorm)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Generate second-stage state using true environment transitions
        s2 = rng.choice([0, 1], p=true_T[a1])

        # Second-stage preferences with stickiness (state-dependent)
        pref2 = Q2[s2].copy()
        prev_a2 = prev_a2_by_state[s2]
        if prev_a2 is not None:
            bias2 = np.zeros(2)
            bias2[prev_a2] = 1.0
            pref2 = pref2 + kappa2 * bias2

        # Softmax for stage-2 choice
        pref2c = pref2 - np.max(pref2)
        p2_unnorm = np.exp(beta * pref2c)
        p_stage2 = p2_unnorm / np.sum(p2_unnorm)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Sample reward from drifting reward probabilities for the current state/action
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < reward_probs[s2][a2])

        # Update second-stage Q-values (model-free)
        Q2[s2, a2] += alphaQ * (r - Q2[s2, a2])

        # Update learned transition probabilities for the chosen first-stage action
        for sp in (0, 1):
            target = 1.0 if sp == s2 else 0.0
            T[a1, sp] += alphaT * (target - T[a1, sp])

        # Renormalize the updated row to be a valid probability distribution
        row_sum = T[a1, 0] + T[a1, 1]
        if row_sum > 0:
            T[a1, :] = T[a1, :] / row_sum

        # Update stickiness memory
        prev_a1 = a1
        prev_a2_by_state[s2] = a2

        # Store outputs
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
