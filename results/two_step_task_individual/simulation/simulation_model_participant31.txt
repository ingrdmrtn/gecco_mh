def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards for the asymmetric TD learning model
    with eligibility trace and value forgetting (purely model-free).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha_pos, alpha_neg, beta, lambda_et, decay]
            - alpha_pos in [0,1]: Learning rate for non-negative TD errors.
            - alpha_neg in [0,1]: Learning rate for negative TD errors.
            - beta in [0,10]: Inverse temperature for both stages.
            - lambda_et in [0,1]: Eligibility trace from stage 2 to stage 1.
            - decay in [0,1]: Value forgetting rate (per trial).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage states and two actions:
            state 0: [drift1, drift2], state 1: [drift3, drift4].

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha_pos, alpha_neg, beta, lambda_et, decay = parameters

    # Standard two-step transition structure (common = 0.7, rare = 0.3)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    # Q-values: stage-1 (2 actions), stage-2 (2 states x 2 actions)
    q1 = np.zeros(2)
    q2 = np.zeros((2, 2))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Reward probabilities at this trial
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Stage 1: softmax over q1
        q1c = q1 - np.max(q1)
        pi1_unnorm = np.exp(beta * q1c)
        pi1 = pi1_unnorm / np.sum(pi1_unnorm)
        a1 = rng.choice([0, 1], p=pi1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2: softmax over q2 at reached state
        q2s = q2[s2, :]
        q2c = q2s - np.max(q2s)
        pi2_unnorm = np.exp(beta * q2c)
        pi2 = pi2_unnorm / np.sum(pi2_unnorm)
        a2 = rng.choice([0, 1], p=pi2)

        # Outcome
        r = int(rng.random() < reward_probs[s2, a2])

        # Learning: stage 2 TD error with asymmetric learning rate
        delta2 = r - q2[s2, a2]
        alpha2 = alpha_pos if delta2 >= 0.0 else alpha_neg
        q2[s2, a2] += alpha2 * delta2

        # Eligibility trace update to stage 1 using updated q2[s2, a2] as target
        target1 = q2[s2, a2]
        delta1 = target1 - q1[a1]
        alpha1 = alpha_pos if delta1 >= 0.0 else alpha_neg
        q1[a1] += (lambda_et * alpha1) * delta1

        # Value forgetting (decay toward zero) applied each trial
        q2 *= (1.0 - decay)
        q1 *= (1.0 - decay)

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
