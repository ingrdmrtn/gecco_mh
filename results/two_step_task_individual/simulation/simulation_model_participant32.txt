def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the model-free 2-step model with
    asymmetric learning, transition-dependent credit assignment, and forgetting.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha_pos, alpha_neg, beta, eta, f]
            - alpha_pos in [0,1]: learning rate when reward = 1 at stage 2.
            - alpha_neg in [0,1]: learning rate when reward = 0 at stage 2.
            - beta in [0,10]: inverse temperature for softmax at both stages.
            - eta in [0,1]: reduces stage-1 credit after rare transitions
                           (0 = no reduction; 1 = no learning after rare).
            - f in [0,1]: forgetting toward priors (Q1 toward 0, Q2 toward 0.5) each trial.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for each state (rows) and action (cols):
                [[drift1[t], drift2[t]],
                 [drift3[t], drift4[t]]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha_pos, alpha_neg, beta, eta, f = parameters

    # Transition structure: action matches common state (0->state 0, 1->state 1)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    # Initialize value functions with same priors as fitting code
    q1 = np.zeros(2)            # Stage-1 MF values, prior 0
    q2 = np.full((2, 2), 0.5)   # Stage-2 MF values, prior 0.5

    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Reward probabilities for this trial by state and action
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Stage 1 choice via softmax over q1
        q1_centered = q1 - np.max(q1)
        p1 = np.exp(beta * q1_centered)
        p1 /= np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state based on chosen action
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice via softmax over q2 for current state
        q2s = q2[s2]
        q2s_centered = q2s - np.max(q2s)
        p2 = np.exp(beta * q2s_centered)
        p2 /= np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Outcome
        r = int(rng.random() < reward_probs[s2, a2])

        # Learning rate at stage 2 depends on reward outcome (asymmetric)
        alpha2 = alpha_pos if r > 0.0 else alpha_neg

        # Stage-2 update
        q2[s2, a2] += alpha2 * (r - q2[s2, a2])

        # Transition-dependent credit assignment at stage 1
        is_common = (a1 == s2)
        credit_scale = 1.0 if is_common else (1.0 - eta)
        td1 = q2[s2, a2] - q1[a1]
        q1[a1] += credit_scale * alpha2 * td1

        # Forgetting toward priors
        q1 = (1.0 - f) * q1               # toward 0
        q2 = (1.0 - f) * q2 + f * 0.5     # toward 0.5

        # Log trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
