def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with eligibility traces and stickiness, matching the provided fitting model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta, w, lam, kappa]
            - alpha: learning rate for MF updates and eligibility propagation
            - beta: inverse temperature for softmax at both stages
            - w: weight of MB value in first-stage decision (0=MF only, 1=MB only)
            - lam: eligibility trace parameter propagating second-stage RPE to first-stage
            - kappa: stickiness added to the last chosen action at each stage
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for state0-action0, state0-action1, state1-action0, state1-action1 respectively.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w, lam, kappa = parameters

    rng = np.random.default_rng()

    # Known transition structure
    transition_matrix = np.array([[0.7, 0.3],  # P(state | action 0)
                                  [0.3, 0.7]]) # P(state | action 1)

    # Initialize MF values
    q1_mf = np.zeros(2)        # First-stage MF Q for actions [0,1]
    q2_mf = np.zeros((2, 2))   # Second-stage MF Q for states [0,1] x actions [0,1]

    # Stickiness memory (last choices)
    last_a1 = -1
    last_a2 = np.array([-1, -1], dtype=int)  # per state

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Trial-wise reward probabilities for [state][action]
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Compute MB first-stage Q as expected max Q at stage 2 under transitions
        max_q2 = np.max(q2_mf, axis=1)        # shape (2,)
        q1_mb = transition_matrix @ max_q2    # shape (2,)

        # Blend MB and MF for first-stage decision
        q1 = w * q1_mb + (1.0 - w) * q1_mf

        # Add first-stage stickiness (favor repeating last first-stage action)
        if last_a1 != -1:
            q1 = q1.copy()
            q1[last_a1] += kappa

        # Softmax choice at stage 1
        q1_center = q1 - np.max(q1)
        p1 = np.exp(beta * q1_center)
        p1 /= p1.sum()
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Second-stage decision values with per-state stickiness
        q2 = q2_mf[s2].copy()
        if last_a2[s2] != -1:
            q2[last_a2[s2]] += kappa

        q2_center = q2 - np.max(q2)
        p2 = np.exp(beta * q2_center)
        p2 /= p2.sum()
        a2 = rng.choice([0, 1], p=p2)

        # Generate reward
        r = int(rng.random() < reward_probs[s2][a2])

        # Store old second-stage Q for eligibility computation
        q2_sa_old = q2_mf[s2, a2]

        # MF updates
        # Stage-2 TD error and update
        delta2 = r - q2_mf[s2, a2]
        q2_mf[s2, a2] += alpha * delta2

        # Stage-1 TD error from bootstrap on second-stage Q (before its update)
        delta1 = q2_sa_old - q1_mf[a1]
        q1_mf[a1] += alpha * delta1

        # Eligibility trace propagation of stage-2 RPE to stage-1 Q
        q1_mf[a1] += alpha * lam * delta2

        # Update stickiness memory
        last_a1 = a1
        last_a2[s2] = a2

        # Log outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
