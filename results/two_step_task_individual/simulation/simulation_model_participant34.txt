def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with reward utility curvature
    and global value decay (forgetting), matching cognitive_model3.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, lambda_u, mu, theta]
            - alpha in [0,1]: learning rate for MF value updates at both stages.
            - beta in [0,10]: inverse temperature for softmax at both stages.
            - lambda_u in [0,1]: reward utility curvature; r_tilde = r ** lambda_u.
            - mu in [0,1]: global decay toward zero applied to all Q-values before each update.
            - theta in [0,1]: weight on model-based value at stage 1; (1-theta) weights MF Q1.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state 0/action 0, state 0/action 1, state 1/action 0, state 1/action 1 respectively.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, lambda_u, mu, theta = parameters

    rng = np.random.default_rng()

    # Fixed transition matrix
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]])

    # Initialize Q-values
    q1_mf = np.zeros(2)      # model-free Q at stage 1
    q2 = np.zeros((2, 2))    # stage-2 Q-values per state and action

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Trial-wise reward probabilities by state and action
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Global decay (forgetting) before decision and learning
        q1_mf *= (1.0 - mu)
        q2 *= (1.0 - mu)

        # Model-based Q at stage 1 from current stage-2 values
        max_q2 = np.max(q2, axis=1)           # best action per state at stage 2
        q1_mb = T @ max_q2                    # expected value per action at stage 1 via transitions

        # Hybrid Q at stage 1
        q1 = theta * q1_mb + (1.0 - theta) * q1_mf

        # Stage-1 choice via softmax
        logits1 = beta * q1
        logits1 -= np.max(logits1)
        p1 = np.exp(logits1)
        p1 /= np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state given action a1
        s2 = rng.choice([0, 1], p=T[a1])

        # Stage-2 choice via softmax
        logits2 = beta * q2[s2]
        logits2 -= np.max(logits2)
        p2 = np.exp(logits2)
        p2 /= np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Outcome
        r = int(rng.random() < reward_probs[s2][a2])

        # Utility transform (note: for binary rewards, this equals r unless lambda_u=0 and r=0)
        r_tilde = r ** lambda_u

        # Learning updates (MF at stage 2, then MF at stage 1)
        pe2 = r_tilde - q2[s2, a2]
        q2[s2, a2] += alpha * pe2

        pe1 = q2[s2, a2] - q1_mf[a1]
        q1_mf[a1] += alpha * pe1

        # Store trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
