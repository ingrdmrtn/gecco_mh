def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the volatility-gated MB/MF arbitration model
    with MF value decay and stage-2 stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha_q, beta, kappa2, decay, phi_vol]
            - alpha_q: learning rate for MF values and volatility trace (0-1).
            - beta: inverse temperature for softmax at both stages (0-10).
            - kappa2: stage-2 perseveration bias for repeating previous action in same state (0-1).
            - decay: MF forgetting factor applied each trial (0-1), 1 means no decay.
            - phi_vol: arbitration gain; effective MB weight is phi_vol*(1 - v_expected), where
              v_expected is the expected volatility from the second stage (0-1).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            - drift1: state 0, action 0
            - drift2: state 0, action 1
            - drift3: state 1, action 0
            - drift4: state 1, action 1

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    rng = np.random.default_rng()

    alpha_q, beta, kappa2, decay, phi_vol = parameters

    # Fixed transition structure
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]], dtype=float)

    # Initialize value functions and volatility trace
    q_s2 = np.zeros((2, 2), dtype=float)   # Q(s2, a2) model-free
    q_s1_mf = np.zeros(2, dtype=float)     # First-stage MF values
    v_trace = np.zeros((2, 2), dtype=float)  # Running avg |RPE| per state-action (volatility proxy)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Stage-2 perseveration memory per state
    prev_a2 = [None, None]

    for t in range(n_trials):
        # Compute MB action values at stage 1 from MF stage-2 values
        max_q2 = np.max(q_s2, axis=1)               # best MF value per state
        q1_mb = T @ max_q2                          # expected MB value per first-stage action

        # Compute volatility-gated arbitration weight (state-dependent)
        v_per_state = np.max(v_trace, axis=1)       # volatility of best action per state
        v_expected = T @ v_per_state                # expected volatility per first-stage action
        v_expected = np.clip(v_expected, 0.0, 1.0)
        w_mb = phi_vol * (1.0 - v_expected)         # vector of length 2

        # Blend MB and MF at stage 1
        q1_blend = w_mb * q1_mb + (1.0 - w_mb) * q_s1_mf

        # Stage-1 softmax and choice
        logits1 = beta * q1_blend
        logits1 -= np.max(logits1)  # numerical stability
        exp1 = np.exp(logits1)
        probs1 = exp1 / np.sum(exp1)
        a1 = rng.choice([0, 1], p=probs1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=T[a1])

        # Stage-2 softmax with stickiness
        bias2 = np.zeros(2)
        if prev_a2[s] is not None:
            bias2[prev_a2[s]] = 1.0
        logits2 = beta * q_s2[s, :] + kappa2 * bias2
        logits2 -= np.max(logits2)
        exp2 = np.exp(logits2)
        probs2 = exp2 / np.sum(exp2)
        a2 = rng.choice([0, 1], p=probs2)

        # Generate reward from drift probabilities
        # Map trial-wise probabilities to [state, action]
        if s == 0:
            p_rwd = drift1[t] if a2 == 0 else drift2[t]
        else:
            p_rwd = drift3[t] if a2 == 0 else drift4[t]
        r = int(rng.random() < p_rwd)

        # Apply decay (forgetting) to MF values
        q_s2 *= decay
        q_s1_mf *= decay

        # Stage-2 MF update
        pe2 = r - q_s2[s, a2]
        q_s2[s, a2] += alpha_q * pe2

        # Update volatility trace with abs RPE
        v_trace[s, a2] = (1.0 - alpha_q) * v_trace[s, a2] + alpha_q * abs(pe2)

        # Stage-1 MF update via bootstrapping from stage-2 PE
        q_s1_mf[a1] += alpha_q * pe2

        # Update stickiness memory
        prev_a2[s] = a2

        # Record
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
