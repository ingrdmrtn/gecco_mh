import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a model-based planner with Pearce–Hall associability
    and one-step perseveration (matching cognitive_model2).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [base_lr, beta, assoc_tau, stick1, stick2]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            second-stage actions:
            - State 0 (X): action 0 -> drift1[t], action 1 -> drift2[t]
            - State 1 (Y): action 0 -> drift3[t], action 1 -> drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    base_lr, beta, assoc_tau, stick1, stick2 = parameters

    rng = np.random.default_rng()
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]])

    # Second-stage Q-values and associability
    q2 = np.zeros((2, 2))
    A = np.ones((2, 2))  # start with full associability as in fitting code

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Perseveration memory
    prev_a1 = None
    prev_a2 = [None, None]  # one per state

    for t in range(n_trials):
        # Trial-wise reward probabilities
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Model-based first-stage values from current second-stage Qs
        V = np.max(q2, axis=1)       # value of states X(0) and Y(1)
        q1_mb = T @ V                # model-based action values at stage 1

        # Stage 1 perseveration bias
        bias1 = np.zeros(2)
        if prev_a1 is not None:
            bias1[prev_a1] += stick1

        # Softmax for stage 1
        logits1 = beta * q1_mb + bias1
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 = p1 / np.sum(p1)

        a1 = rng.choice([0, 1], p=p1)

        # Transition to stage-2 state
        s2 = rng.choice([0, 1], p=T[a1])

        # Stage 2 perseveration bias (state-dependent)
        bias2 = np.zeros(2)
        if prev_a2[s2] is not None:
            bias2[prev_a2[s2]] += stick2

        # Softmax for stage 2 within observed state
        logits2 = beta * q2[s2] + bias2
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 = p2 / np.sum(p2)

        a2 = rng.choice([0, 1], p=p2)

        # Outcome
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning at stage 2 with Pearce–Hall associability
        delta2 = r - q2[s2, a2]
        A[s2, a2] = (1.0 - assoc_tau) * A[s2, a2] + assoc_tau * abs(delta2)
        eff_lr = base_lr * A[s2, a2]
        q2[s2, a2] += eff_lr * delta2

        # Update perseveration memories
        prev_a1 = a1
        prev_a2[s2] = a2

        # Log trial
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
