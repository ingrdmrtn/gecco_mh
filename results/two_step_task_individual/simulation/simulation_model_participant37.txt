def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with eligibility trace and stickiness, matching cognitive_model1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w, lam, kappa]
            - alpha: learning rate for both stages [0,1]
            - beta: inverse temperature for both stages [0,10]
            - w: weight on model-based vs model-free [0,1]
            - lam: eligibility trace [0,1]
            - kappa: stickiness for repeating the last action at a stage [0,1]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage states (rows) and actions (columns):
            state 0: actions 0-> drift1, 1-> drift2
            state 1: actions 0-> drift3, 1-> drift4

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w, lam, kappa = parameters

    rng = np.random.default_rng()

    transition_matrix = np.array([[0.7, 0.3],   # from action 0 to [state 0, state 1]
                                  [0.3, 0.7]])  # from action 1 to [state 0, state 1]

    # Initialize values
    q1_mf = np.zeros(2)          # model-free first-stage Q-values
    q2 = np.zeros((2, 2))        # second-stage Q-values: [state, action]

    last_a1 = None
    last_a2 = np.array([None, None], dtype=object)  # per-state last second-stage action

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Build trial-specific reward probability matrix for stage 2
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Compute model-based first-stage values from current second-stage values
        max_q2 = np.max(q2, axis=1)          # best action value in each state
        q1_mb = transition_matrix @ max_q2   # expected value of first-stage actions

        # Combine MB and MF values and add stickiness for stage 1
        q1_comb = (1.0 - w) * q1_mf + w * q1_mb
        stick1 = np.zeros(2)
        if last_a1 is not None:
            stick1[last_a1] += kappa

        logits1 = beta * q1_comb + stick1
        logits1 -= np.max(logits1)
        probs1 = np.exp(logits1)
        probs1 /= probs1.sum()

        a1 = rng.choice([0, 1], p=probs1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 choice with per-state stickiness
        stick2 = np.zeros(2)
        if last_a2[s] is not None:
            stick2[last_a2[s]] += kappa

        logits2 = beta * q2[s].copy() + stick2
        logits2 -= np.max(logits2)
        probs2 = np.exp(logits2)
        probs2 /= probs2.sum()

        a2 = rng.choice([0, 1], p=probs2)

        # Outcome
        r = int(rng.random() < reward_probs[s, a2])

        # Learning updates (match cognitive_model1)
        delta2 = r - q2[s, a2]
        q2[s, a2] += alpha * delta2

        delta1 = q2[s, a2] - q1_mf[a1]
        q1_mf[a1] += alpha * (delta1 + lam * delta2)

        # Update stickiness memory
        last_a1 = a1
        last_a2[s] = a2

        # Save trial data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
