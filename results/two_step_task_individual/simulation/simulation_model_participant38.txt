def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the model-free 2-step model with a dynamic
    stage-1 choice kernel (no model-based planning).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha_r, alpha_ck, w_ck, beta1, beta2]
            - alpha_r: learning rate for MF values (stage 1 and 2)
            - alpha_ck: learning rate for the choice kernel
            - w_ck: weight of the choice kernel at stage 1
            - beta1: inverse temperature at stage 1
            - beta2: inverse temperature at stage 2
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage states and two actions:
            state 0: [drift1, drift2], state 1: [drift3, drift4]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha_r, alpha_ck, w_ck, beta1, beta2 = parameters

    # Standard two-step transition matrix
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Value tables
    q1_mf = np.zeros(2)          # Stage-1 MF values for actions [0,1]
    q2 = np.zeros((2, 2))        # Stage-2 MF values: state x action

    # Dynamic choice kernel for stage 1
    ck = np.zeros(2)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    rng = np.random.default_rng()

    for t in range(n_trials):
        # Current reward probability matrix for stage-2 [state][action]
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Stage 1 decision values: MF + centered choice kernel
        ck_centered = ck - np.mean(ck)
        q1 = (1.0 - w_ck) * q1_mf + w_ck * ck_centered

        # Softmax for stage 1
        q1c = q1 - np.max(q1)
        exp_q1 = np.exp(beta1 * q1c)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice via softmax on q2[s2]
        q2s = q2[s2]
        q2c = q2s - np.max(q2s)
        exp_q2 = np.exp(beta2 * q2c)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Outcome
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning updates (match fitting code logic)
        delta2 = r - q2[s2, a2]
        q2[s2, a2] += alpha_r * delta2

        target1 = q2[s2, a2]  # use updated q2 value as target
        q1_mf[a1] += alpha_r * (target1 - q1_mf[a1])

        # Update choice kernel (recency-weighted)
        ck *= (1.0 - alpha_ck)
        ck[a1] += alpha_ck

        # Record trial data
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
