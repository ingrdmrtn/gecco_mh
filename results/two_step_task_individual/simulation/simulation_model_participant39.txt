import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a pure model-based planner that learns
    second-stage values and the transition matrix online, with perseveration bias
    on stage-1 choices.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, eta, pi]
            - alpha: learning rate for second-stage values.
            - beta: inverse temperature for both stages.
            - eta: transition learning rate (row-wise toward observed state).
            - pi: perseveration strength added to previously chosen stage-1 action.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the four second-stage options:
            - State X: actions (0=W -> drift1, 1=S -> drift2)
            - State Y: actions (0=P -> drift3, 1=H -> drift4)

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    alpha, beta, eta, pi = parameters

    # True environment transition matrix (common=0.7, rare=0.3)
    true_T = np.array([[0.7, 0.3],
                       [0.3, 0.7]])

    rng = np.random.default_rng()

    # Agent's learned quantities
    T_learned = np.full((2, 2), 0.5)   # learned transition matrix; rows=actions, cols=states
    q_stage2 = np.zeros((2, 2))         # Q-values at second stage; rows=states (X=0,Y=1), cols=actions (0,1)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None

    for t in range(n_trials):
        # Trial-wise reward probabilities from drifts
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Model-based planning using learned T and current second-stage values
        max_q2 = np.max(q_stage2, axis=1)          # best action per state (2,)
        q1_mb = T_learned @ max_q2                 # value of each stage-1 action via planning (2,)

        # Perseveration bias on stage-1
        bias = np.zeros(2)
        if prev_a1 is not None:
            bias[prev_a1] += pi

        prefs1 = q1_mb + bias
        prefs1 = beta * (prefs1 - np.max(prefs1))  # stable softmax
        p_stage1 = np.exp(prefs1)
        p_stage1 /= p_stage1.sum()

        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state using true environment transitions
        s2 = rng.choice([0, 1], p=true_T[a1])

        # Second-stage choice using same beta
        q2 = q_stage2[s2]
        prefs2 = beta * (q2 - np.max(q2))
        p_stage2 = np.exp(prefs2)
        p_stage2 /= p_stage2.sum()

        a2 = rng.choice([0, 1], p=p_stage2)

        # Outcome
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning second-stage values
        delta2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha * delta2

        # Learn transitions online: update row toward observed state
        T_row = T_learned[a1].copy()
        T_learned[a1] = (1 - eta) * T_row
        T_learned[a1, s2] += eta  # keeps row normalized

        # Update perseveration memory
        prev_a1 = a1

        # Record
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
