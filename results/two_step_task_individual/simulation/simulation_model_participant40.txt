def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a pure model-free 2-step model with:
    - eligibility trace from stage 2 TD error to stage 1,
    - misattribution of credit to the unchosen first-stage option on rare transitions,
    - value decay toward zero for both Q1 and Q2 each trial.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [eta_q, beta, elig, phi_mis, nu_decay]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 2x2 second-stage actions.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    eta_q, beta, elig, phi_mis, nu_decay = parameters
    rng = np.random.default_rng()

    # Transition structure: action 0 commonly -> state 0, action 1 commonly -> state 1
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    Q1_mf = np.zeros(2)         # First-stage MF action values
    Q2 = np.zeros((2, 2))       # Second-stage state-action values

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    eps = 1e-10

    for t in range(n_trials):
        # Reward probabilities for this trial, by state and action
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Stage 1 choice via softmax over Q1_mf
        logits1 = beta * Q1_mf
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 = p1 / (np.sum(p1) + eps)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice via softmax over Q2[s]
        logits2 = beta * Q2[s]
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 = p2 / (np.sum(p2) + eps)
        a2 = rng.choice([0, 1], p=p2)

        # Reward sampling
        r = int(rng.random() < reward_probs[s][a2])

        # Value decay toward zero
        Q1_mf *= (1.0 - nu_decay)
        Q2 *= (1.0 - nu_decay)

        # Second-stage TD update
        delta2 = r - Q2[s, a2]
        Q2[s, a2] += eta_q * delta2

        # Eligibility trace to first-stage MF value
        Q1_mf[a1] += elig * delta2

        # Rare-transition misattribution: give fraction of credit to unchosen a1 if rare transition
        is_common = (a1 == s)
        if not is_common:
            unchosen = 1 - a1
            Q1_mf[unchosen] += phi_mis * elig * delta2

        # Save trial outcomes
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
