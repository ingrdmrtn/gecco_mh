def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the provided hybrid model with:
    - learned transitions,
    - uncertainty-sensitive arbitration,
    - forgetting of second-stage Q-values,
    - novelty bonus for less-visited second-stage actions.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta, omega, eta, phi]
            alpha: learning rate for Q-values and transition probabilities [0,1]
            beta: inverse temperature for both stages [0,10]
            omega: base model-based weight at stage 1 [0,1]
            eta: forgetting rate for stage-2 Q-values [0,1]
            phi: novelty bonus magnitude [0,1]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            state 0 action 0, state 0 action 1, state 1 action 0, state 1 action 1.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    rng = np.random.default_rng()

    alpha, beta, omega, eta, phi = parameters

    # Initialize learned transition matrix
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]], dtype=float)

    # Initialize value functions
    q_stage1_mf = np.zeros(2)       # model-free stage-1 Qs
    q_stage2 = np.zeros((2, 2))     # stage-2 Qs per state-action

    # Counts for novelty bonus (start at 1 to avoid division by zero)
    counts = np.ones((2, 2), dtype=float)

    # Output arrays
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Trial-specific reward probabilities
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]], dtype=float)

        # Apply forgetting to all second-stage values
        q_stage2 *= (1.0 - eta)

        # Model-based stage-1 values from current learned transitions
        max_q2 = np.max(q_stage2, axis=1)     # best action per second-stage state
        q1_mb = T @ max_q2                    # expected value of each first-stage action

        # Compute transition uncertainty (entropy) per first-stage action
        with np.errstate(divide='ignore', invalid='ignore'):
            ent = -np.sum(T * np.log(np.clip(T, 1e-12, 1.0)), axis=1) / np.log(2.0)
        ent = np.clip(ent, 0.0, 1.0)

        # Uncertainty-sensitive arbitration weight (per action)
        w_eff = omega * (1.0 - ent)

        # Combined stage-1 action values
        q1 = w_eff * q1_mb + (1.0 - w_eff) * q_stage1_mf

        # Softmax choice at stage 1
        q1_centered = beta * (q1 - np.max(q1))
        exp_q1 = np.exp(q1_centered)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Sample second-stage state from the current learned transitions
        s2 = rng.choice([0, 1], p=T[a1])

        # Stage-2 choice with novelty bonus
        bonus = phi / np.sqrt(counts[s2] + 0.0)
        q2_aug = q_stage2[s2] + bonus
        q2_centered = beta * (q2_aug - np.max(q2_aug))
        exp_q2 = np.exp(q2_centered)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Sample reward from drifting probabilities
        r = int(rng.random() < reward_probs[s2, a2])

        # Learning updates
        # Stage-2 Q update
        delta2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha * delta2
        counts[s2, a2] += 1.0

        # Stage-1 model-free Q update (bootstrapping from updated stage-2 value)
        delta1 = q_stage2[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1

        # Transition learning: move chosen action's row toward observed state
        one_hot = np.array([1.0 if ss == s2 else 0.0 for ss in range(2)], dtype=float)
        T[a1] = (1.0 - alpha) * T[a1] + alpha * one_hot
        # Normalize and clip for numerical stability
        row_sum = np.sum(T[a1])
        if row_sum > 0:
            T[a1] = T[a1] / row_sum
        T[a1] = np.clip(T[a1], 1e-6, 1.0)
        T[a1] = T[a1] / np.sum(T[a1])

        # Store outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
