def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a model-based controller with cross-state
    generalization and value decay (matching cognitive_model3).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, g, d]
            - alpha: learning rate for second-stage Q-value updates.
            - beta: inverse temperature for softmax choices at both stages.
            - g: cross-state generalization strength for the corresponding alien in the other state.
            - d: per-trial decay rate applied to all second-stage Q-values.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            state 0 action 0, state 0 action 1, state 1 action 0, state 1 action 1.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, g, d = parameters

    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Second-stage Q-values: shape (state, action)
    q_stage2 = np.zeros((2, 2))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Construct the reward probabilities for this trial
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Stage 1: MB planning over current (pre-decay) second-stage values
        max_q_stage2 = np.max(q_stage2, axis=1)              # best action values in each state
        q_stage1_mb = transition_matrix @ max_q_stage2       # expected value of each first-stage action

        # Softmax for stage-1 choice
        logits1 = q_stage1_mb - np.max(q_stage1_mb)
        p_stage1 = np.exp(beta * logits1)
        p_stage1 = p_stage1 / np.sum(p_stage1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # State transition
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2: softmax over current state's Q-values (pre-decay)
        logits2 = q_stage2[s2] - np.max(q_stage2[s2])
        p_stage2 = np.exp(beta * logits2)
        p_stage2 = p_stage2 / np.sum(p_stage2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Outcome
        r = int(rng.random() < reward_probs[s2][a2])

        # Apply global decay to second-stage Q-values
        q_stage2 *= (1.0 - d)

        # TD update at chosen state-action using decayed value as baseline
        q_old = q_stage2[s2, a2]
        delta2 = r - q_old
        q_stage2[s2, a2] += alpha * delta2

        # Cross-state generalization to the corresponding action in the other state
        other_s = 1 - s2
        q_stage2[other_s, a2] += alpha * g * delta2

        # Record
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
