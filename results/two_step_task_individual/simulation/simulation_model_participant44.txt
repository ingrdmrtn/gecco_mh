def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the model-free two-stage TD model
    with eligibility to stage 1, counterfactual learning at stage 2, and a
    baseline bias at stage 1 (matches cognitive_model2 logic).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, elig, rho_cf, b0]
            - alpha: TD learning rate for both stages.
            - beta: Inverse temperature for both stages.
            - elig: Eligibility strength from stage 2 to stage 1.
            - rho_cf: Counterfactual learning strength at stage 2 (applied to unchosen action).
            - b0: Baseline bias toward spaceship A vs U; transformed to signed bias b = (b0 - 0.5).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage states (rows) and two actions (columns), i.e.:
            [[drift1, drift2],  # state 0 (planet X)
             [drift3, drift4]]  # state 1 (planet Y)

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np
    rng = np.random.default_rng()

    alpha, beta, elig, rho_cf, b0 = parameters

    # Two-stage task transitions: common = 0.7, rare = 0.3
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    Q1 = np.zeros(2, dtype=float)          # stage-1: actions A(0)/U(1)
    Q2 = np.zeros((2, 2), dtype=float)     # stage-2: states X(0)/Y(1) x actions

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    bias = b0 - 0.5  # signed bias added to stage-1 logits

    for t in range(n_trials):
        # Trial-wise reward probabilities for each state-action
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Stage 1 choice with bias and softmax
        Q1_eff = Q1.copy()
        Q1_eff[0] += bias
        Q1_eff[1] -= bias
        Q1_eff = Q1_eff - np.max(Q1_eff)  # numerical stability
        exp_q1 = np.exp(beta * Q1_eff)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice softmax
        Q2_eff = Q2[s2, :].copy()
        Q2_eff = Q2_eff - np.max(Q2_eff)
        exp_q2 = np.exp(beta * Q2_eff)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Reward sampling
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning updates
        # Stage 2 TD error and update
        delta2 = r - Q2[s2, a2]
        Q2[s2, a2] += alpha * delta2

        # Counterfactual/opponent update for unchosen action
        alt2 = 1 - a2
        Q2[s2, alt2] -= alpha * rho_cf * delta2

        # Stage 1 eligibility-trace update bootstrapping from updated Q2
        bootstrap = Q2[s2, a2]
        delta1 = bootstrap - Q1[a1]
        Q1[a1] += alpha * elig * delta1

        # Log outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
