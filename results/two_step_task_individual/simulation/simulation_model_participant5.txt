import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using Model 3:
    Uncertainty-adaptive temperature with forgetting and model-based stage-1.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [lr2, beta_base, decay, temp_slope, stick1]
            - lr2 in [0,1]: learning rate for stage-2 Q-updates.
            - beta_base in [0,10]: base inverse temperature before uncertainty modulation.
            - decay in [0,1]: per-trial forgetting applied to all stage-2 Q-values.
            - temp_slope in [0,1]: strength of uncertainty-to-temperature modulation.
            - stick1 in [0,1]: perseveration bias added to the previously chosen stage-1 action.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state-action pairs:
              state 0: [drift1[t], drift2[t]]
              state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    rng = np.random.default_rng()

    lr2, beta_base, decay, temp_slope, stick1 = parameters

    # Fixed transition structure
    T_fixed = np.array([[0.7, 0.3],
                        [0.3, 0.7]])

    # Stage-2 Q-values (states x actions)
    q2 = np.zeros((2, 2))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None

    for t in range(n_trials):
        # Apply forgetting to all stage-2 Q-values
        q2 *= (1.0 - decay)

        # Compute uncertainty-adaptive inverse temperature
        entropies = np.zeros(2)
        for ss in range(2):
            logits_tmp = beta_base * q2[ss, :]
            logits_tmp -= np.max(logits_tmp)
            p_tmp = np.exp(logits_tmp) / np.sum(np.exp(logits_tmp))
            eps_h = 1e-12
            entropies[ss] = -np.sum(p_tmp * np.log(p_tmp + eps_h))
        H = np.mean(entropies)
        H_max = np.log(2.0)
        uncertainty_factor = 1.0 - (H / H_max)  # 0 when max uncertainty, 1 when deterministic
        beta_adapt = beta_base * (1.0 + temp_slope * uncertainty_factor)

        # Stage-1 model-based planning from current stage-2 Q-values
        V_states = np.max(q2, axis=1)  # value of best action in each state
        q1_mb = T_fixed @ V_states

        # Perseveration bias
        bias = np.zeros(2)
        if prev_a1 is not None:
            bias[prev_a1] = stick1

        # Choose stage-1 action
        logits1 = beta_adapt * q1_mb + bias
        logits1 -= np.max(logits1)
        p_stage1 = np.exp(logits1) / np.sum(np.exp(logits1))
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=T_fixed[a1])

        # Choose stage-2 action using same adaptive temperature
        logits2 = beta_adapt * q2[s2, :]
        logits2 -= np.max(logits2)
        p_stage2 = np.exp(logits2) / np.sum(np.exp(logits2))
        a2 = rng.choice([0, 1], p=p_stage2)

        # Generate reward from drifting probabilities
        reward_prob = [ [drift1[t], drift2[t]],
                        [drift3[t], drift4[t]] ][s2][a2]
        r = int(rng.random() < reward_prob)

        # Stage-2 Q-learning update
        pe2 = r - q2[s2, a2]
        q2[s2, a2] += lr2 * pe2

        # Record and update perseveration
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
