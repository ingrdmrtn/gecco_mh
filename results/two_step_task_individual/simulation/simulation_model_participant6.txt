def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a pure model-free SARSA agent with
    risk-sensitive utility, value decay, and second-stage within-planet stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta, rho, decay, sigma2]
            - alpha (0-1): learning rate for both stages.
            - beta (0-10): inverse temperature for softmax at both stages.
            - rho (0-1): risk sensitivity; u(r) = sign(r) * |r|^rho.
            - decay (0-1): forgetting rate; Q-values multiplied by (1 - decay) each trial.
            - sigma2 (0-1): second-stage stickiness (within visited planet).
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for [state 0 action 0, state 0 action 1, state 1 action 0, state 1 action 1].

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, rho, decay, sigma2 = parameters

    rng = np.random.default_rng()

    # Standard two-step transition structure (common 0.7 / rare 0.3)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    q1 = np.zeros(2)          # First-stage MF Q-values over spaceships
    q2 = np.zeros((2, 2))     # Second-stage MF Q-values over aliens by planet

    # For within-planet stickiness at second stage
    prev_a2_for_state = [None, None]

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Apply value decay
        q1 *= (1.0 - decay)
        q2 *= (1.0 - decay)

        # Stage 1 choice via softmax over q1
        logits1 = beta * (q1 - np.max(q1))
        probs_1 = np.exp(logits1)
        probs_1 /= probs_1.sum()
        a1 = rng.choice([0, 1], p=probs_1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice with within-planet stickiness
        stick2 = np.zeros(2)
        if prev_a2_for_state[s2] is not None:
            stick2[prev_a2_for_state[s2]] = sigma2

        logits2 = beta * (q2[s2, :] - np.max(q2[s2, :])) + stick2
        probs_2 = np.exp(logits2)
        probs_2 /= probs_2.sum()
        a2 = rng.choice([0, 1], p=probs_2)

        # Trial-wise reward probabilities from drifts
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]
        r = int(rng.random() < reward_probs[s2][a2])

        # Risk-sensitive utility transform
        u = np.sign(r) * (np.abs(r) ** rho)

        # SARSA updates (second stage first, then first stage)
        delta2 = u - q2[s2, a2]
        q2[s2, a2] += alpha * delta2

        td1 = q2[s2, a2] - q1[a1]
        q1[a1] += alpha * td1

        # Update stickiness memory for this planet
        prev_a2_for_state[s2] = a2

        # Log trial outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
