def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the 'Uncertainty-weighted hybrid control with learned transitions and stage-1 stickiness' model.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha_v, beta, trans_alpha, mb_gate, kappa_s1]
            - alpha_v: learning rate for Q-value updates (both stages).
            - beta: inverse temperature for both stages.
            - trans_alpha: transition learning rate for T rows.
            - mb_gate: how strongly certainty boosts model-based control at stage 1.
            - kappa_s1: stickiness toward the previous stage-1 action.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state 0 action 0, state 0 action 1, state 1 action 0, state 1 action 1, respectively.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha_v, beta, trans_alpha, mb_gate, kappa_s1 = parameters

    # Environment's (true) transition matrix used to generate states
    P_env = np.array([[0.7, 0.3],
                      [0.3, 0.7]], dtype=float)

    rng = np.random.default_rng()

    # Initialize agent's internal variables
    T = np.array([[0.5, 0.5],
                  [0.5, 0.5]], dtype=float)  # learned transition model

    q1_mf = np.zeros(2)        # model-free first-stage values
    q2 = np.zeros((2, 2))      # second-stage Q-values

    prev_a1 = -1               # for stickiness

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Compute model-based first-stage values from learned T and current q2
        v2 = np.max(q2, axis=1)           # value of each second-stage state
        q1_mb = T @ v2                    # MB value per first-stage action

        # Action-specific certainty from learned transitions: certainty = 1 - entropy_base2
        row_ent = np.zeros(2)
        for a in range(2):
            p_row = np.clip(T[a], 1e-12, 1.0)
            row_ent[a] = -(p_row[0] * np.log2(p_row[0]) + p_row[1] * np.log2(p_row[1]))
        certainty = 1.0 - row_ent  # in [0,1]

        # Arbitration weight per action
        w = mb_gate * certainty + (1.0 - mb_gate) * 0.5

        # Combined first-stage value
        q1 = w * q1_mb + (1.0 - w) * q1_mf

        # Add stickiness toward previous action
        pref1 = q1.copy()
        if prev_a1 in (0, 1):
            pref1[prev_a1] += kappa_s1

        # Softmax choice at stage 1
        pref1_sm = pref1 - np.max(pref1)
        p1 = np.exp(beta * pref1_sm)
        p1 /= np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state using environment dynamics
        s = rng.choice([0, 1], p=P_env[a1])

        # Softmax choice at stage 2 from q2[s]
        pref2 = q2[s].copy()
        pref2_sm = pref2 - np.max(pref2)
        p2 = np.exp(beta * pref2_sm)
        p2 /= np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Reward from drifting Bernoulli probabilities
        # Map (s, a2) -> corresponding drift array
        if s == 0 and a2 == 0:
            p_r = drift1[t]
        elif s == 0 and a2 == 1:
            p_r = drift2[t]
        elif s == 1 and a2 == 0:
            p_r = drift3[t]
        else:
            p_r = drift4[t]
        r = int(rng.random() < p_r)

        # Learning updates
        # Stage 2 Q-learning
        pe2 = r - q2[s, a2]
        q2[s, a2] += alpha_v * pe2

        # Stage 1 model-free update bootstrapping from Stage 2 chosen Q
        target1 = q2[s, a2]
        pe1 = target1 - q1_mf[a1]
        q1_mf[a1] += alpha_v * pe1

        # Transition model learning toward observed state
        target_row = np.array([0.0, 0.0])
        target_row[s] = 1.0
        T[a1, :] = (1.0 - trans_alpha) * T[a1, :] + trans_alpha * target_row
        T[a1, :] /= (np.sum(T[a1, :]) + 1e-12)

        # Save outputs and update stickiness
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
