def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid RL with learned transitions,
    anxiety-weighted arbitration, and anxiety-scaled forgetting.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha2, beta, alpha_T, w0, decay, stai]
            - alpha2: stage-2 MF learning rate.
            - beta: inverse temperature for softmax at both stages.
            - alpha_T: learning rate for the transition matrix.
            - w0: baseline arbitration weight for MB control at stage 1.
            - decay: base forgetting rate applied each trial.
            - stai: anxiety level in [0,1]; higher = more anxious.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two actions on each of the two planets:
            [[drift1, drift2],   # planet 0
             [drift3, drift4]]   # planet 1

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np
    rng = np.random.default_rng()

    alpha2, beta, alpha_T, w0, decay, st = parameters
    st = float(np.clip(st, 0.0, 1.0))

    # Initialize learned transition matrix (rows = stage-1 actions; cols = planets)
    T = np.full((2, 2), 0.5)

    # Model-free Q-values
    q1_mf = np.zeros(2)        # stage-1 MF values for actions A/U
    q2_mf = np.zeros((2, 2))   # stage-2 MF values for each planet (rows) and alien (cols)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Anxiety-scaled forgetting
    decay_eff = np.clip(decay * (0.5 + 0.5 * st), 0.0, 1.0)

    for t in range(n_trials):
        # Trial-wise reward probabilities (environmental drift)
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Apply forgetting to MF values
        q1_mf *= (1.0 - decay_eff)
        q2_mf *= (1.0 - decay_eff)

        # Model-based value from current learned transitions
        max_q2 = np.max(q2_mf, axis=1)   # value of each planet
        q1_mb = T @ max_q2               # MB value of each spaceship

        # Transition certainty and anxiety-weighted arbitration
        c_actions = 2.0 * np.abs(T[:, 0] - 0.5)   # certainty per first-stage action
        certainty = np.mean(c_actions)            # overall certainty in [0,1]
        w_eff = w0 * (certainty ** (1.0 - st))
        w_eff = float(np.clip(w_eff, 0.0, 1.0))

        # Stage-1 choice via softmax over weighted MB/MF blend
        q1 = w_eff * q1_mb + (1.0 - w_eff) * q1_mf
        q1_centered = q1 - np.max(q1)
        probs1_unnorm = np.exp(beta * q1_centered)
        probs1 = probs1_unnorm / np.sum(probs1_unnorm)
        a1 = int(rng.choice([0, 1], p=probs1))

        # Transition to second-stage state sampled from learned T
        s = int(rng.choice([0, 1], p=T[a1]))

        # Stage-2 choice via softmax over MF values for that state
        q2 = q2_mf[s]
        q2_centered = q2 - np.max(q2)
        probs2_unnorm = np.exp(beta * q2_centered)
        probs2 = probs2_unnorm / np.sum(probs2_unnorm)
        a2 = int(rng.choice([0, 1], p=probs2))

        # Sample reward from drifting environment
        r_prob = reward_probs[s][a2]
        r = int(rng.random() < r_prob)

        # Update learned transition model for chosen first-stage action
        for sp in range(2):
            target = 1.0 if sp == s else 0.0
            T[a1, sp] += alpha_T * (target - T[a1, sp])
        # Normalize chosen row to ensure it remains a valid probability distribution
        T[a1] = T[a1] / np.sum(T[a1])

        # MF updates
        delta2 = r - q2_mf[s, a2]
        q2_mf[s, a2] += alpha2 * delta2

        v2 = q2_mf[s, a2]
        delta1 = v2 - q1_mf[a1]
        q1_mf[a1] += alpha2 * delta1

        # Record trial data
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
