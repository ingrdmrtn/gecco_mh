def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the uncertainty-penalized hybrid model
    with anxiety-modulated arbitration and stickiness (matching cognitive_model1).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or np.ndarray): [alpha, beta, psi, gamma, kappa1, stai]
            - alpha in [0,1]: learning rate for both stages.
            - beta in [0,10]: inverse temperature (both stages).
            - psi in [0,1]: baseline MB arbitration weight.
            - gamma in [0,1]: weight on uncertainty penalty at stage 1.
            - kappa1 in [0,1]: baseline first-stage stickiness strength.
            - stai in [0,1]: anxiety scalar.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the four second-stage options arranged by state-action:
            state X: [drift1, drift2], state Y: [drift3, drift4].

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0=A, 1=U).
        state2 (np.ndarray): Second-stage states (0=X, 1=Y).
        stage2_choice (np.ndarray): Second-stage choices within state (0 or 1).
        reward (np.ndarray): Obtained rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, psi, gamma, kappa1, stai = parameters
    rng = np.random.default_rng()

    # Transition matrix and its action-wise entropy (uncertainty)
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]])
    eps_h = 1e-12
    H = -np.sum(T * np.log(T + eps_h), axis=1)  # entropy per first-stage action

    # Effective arbitration weight and stickiness modulated by anxiety
    w_eff = float(np.clip(psi * (1.0 - 0.6 * float(stai)), 0.0, 1.0))
    kappa_eff = float(kappa1 * (1.0 + float(stai)))

    # Initialize value functions
    q1_mf = np.zeros(2)       # model-free values for first-stage actions [A,U]
    q2 = np.zeros((2, 2))     # second-stage Q-values: q2[state, action]

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = 0  # initial perseveration target

    for t in range(n_trials):
        # Current reward probabilities for second-stage options
        reward_probs = [[float(drift1[t]), float(drift2[t])],
                        [float(drift3[t]), float(drift4[t])]]

        # Compute model-based stage-1 values from current stage-2 values
        max_q2 = np.max(q2, axis=1)     # value of states X(0) and Y(1)
        q1_mb = T @ max_q2              # MB values for A(0), U(1)

        # Hybrid stage-1 values
        q1_hybrid = w_eff * q1_mb + (1.0 - w_eff) * q1_mf

        # Stickiness bias (on previous first-stage choice)
        bias1 = np.zeros(2)
        bias1[prev_a1] += kappa_eff

        # Uncertainty penalty on stage-1 actions
        unc_penalty = gamma * float(stai) * H

        # Stage-1 choice via softmax over penalized, biased hybrid values
        logits1 = beta * q1_hybrid + bias1 - unc_penalty
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 = p1 / np.sum(p1)
        a1 = int(rng.choice([0, 1], p=p1))

        # Transition to second-stage state
        s2 = int(rng.choice([0, 1], p=T[a1]))

        # Stage-2 choice via softmax over q2[s2]
        logits2 = beta * q2[s2]
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 = p2 / np.sum(p2)
        a2 = int(rng.choice([0, 1], p=p2))

        # Sample reward from current drifting probabilities
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning updates (match fitting code)
        pe2 = r - q2[s2, a2]
        q2[s2, a2] += alpha * pe2

        target1 = q2[s2, a2]
        pe1 = target1 - q1_mf[a1]
        q1_mf[a1] += alpha * pe1

        # Record and prepare next trial
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
