def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using a purely model-free 2-step controller with:
      - anxiety-modulated asymmetric learning rates,
      - anxiety-modulated perseveration at stage 1,
      - a single inverse temperature for both stages,
      - an eligibility trace from stage 2 to stage 1,
      - an additional stage-1 TD alignment step.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or array-like):
            [alpha_base, beta, pi_base, stai, stai_weight, lambd]
              alpha_base in [0,1]  : base learning rate
              beta in [0,10]       : inverse temperature (both stages)
              pi_base in [0,1]     : baseline perseveration strength (stage 1)
              stai in [0,1]        : anxiety score
              stai_weight in [0,1] : sensitivity of parameters to STAI (0=no effect, 1=max effect)
              lambd in [0,1]       : eligibility trace
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for second-stage actions by state:
              state 0: [drift1, drift2]
              state 1: [drift3, drift4]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np
    rng = np.random.default_rng()

    alpha_base, beta, pi_base, stai, stai_weight, lambd = parameters

    # Standard two-step transition structure (common = 0.7, rare = 0.3)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize Q-values
    q_stage1_mf = np.zeros(2)       # first-stage MF values for actions A(0) and U(1)
    q_stage2_mf = np.zeros((2, 2))  # second-stage MF values for states 0/1 and actions 0/1

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Anxiety modulation term (same as in fitting code)
    mod = (2.0 * stai - 1.0) * (2.0 * stai_weight - 1.0)

    # Asymmetric learning rates and perseveration strength
    alpha_pos = np.clip(alpha_base * (1.0 + mod), 0.0, 1.0)
    alpha_neg = np.clip(alpha_base * (1.0 - mod), 0.0, 1.0)
    alpha1 = np.clip(alpha_base * (1.0 + 0.5 * mod), 0.0, 1.0)
    pi_eff = pi_base * (1.0 + mod)

    prev_a1 = None

    for t in range(n_trials):
        # Stage 1 choice probabilities with perseveration bias
        q1 = q_stage1_mf.copy()
        if prev_a1 is not None:
            stick = np.zeros(2)
            stick[prev_a1] = 1.0
            q1 = q1 + pi_eff * stick

        q1_centered = q1 - np.max(q1)
        exp_q1 = np.exp(beta * q1_centered)
        p_stage1 = exp_q1 / np.sum(exp_q1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice
        q2_s = q_stage2_mf[s2].copy()
        q2_centered = q2_s - np.max(q2_s)
        exp_q2 = np.exp(beta * q2_centered)
        p_stage2 = exp_q2 / np.sum(exp_q2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Reward outcome from drifting probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]
        r = int(rng.random() < reward_probs[s2][a2])

        # Stage 2 PE and update with asymmetric learning rate
        pe2 = r - q_stage2_mf[s2, a2]
        a2_lr = alpha_pos if pe2 >= 0 else alpha_neg
        q_stage2_mf[s2, a2] += a2_lr * pe2

        # Eligibility trace backup to stage 1
        q_stage1_mf[a1] += a2_lr * lambd * pe2

        # Additional stage-1 TD alignment toward current second-stage value
        td1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha1 * td1

        # Update perseveration memory
        prev_a1 = a1

        # Store
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
