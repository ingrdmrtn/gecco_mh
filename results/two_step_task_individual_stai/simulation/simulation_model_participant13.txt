import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free SARSA(Î»)
    with anxiety-modulated arbitration and perseveration (matching cognitive_model1).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or array-like): [alpha, beta, lam, w_mb, stickiness, stai]
            - alpha in [0,1]: Learning rate for both stages.
            - beta in [0,10]: Inverse temperature used at both stages.
            - lam in [0,1]: Eligibility trace strength to propagate stage-2 RPE to stage 1.
            - w_mb in [0,1]: Baseline model-based weight before anxiety modulation.
            - stickiness in [0,1]: Strength of first-stage perseveration (scaled by STAI).
            - stai in [0,1]: Participant anxiety score used to modulate arbitration and perseveration.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state 0 action 0/1 and state 1 action 0/1 respectively.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    rng = np.random.default_rng()

    alpha, beta, lam, w_mb, stickiness, stai = parameters
    stai = float(stai)

    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Q-values
    q_stage2 = np.zeros((2, 2))   # Q2[state, action]
    q_stage1_mf = np.zeros(2)     # model-free Q at stage 1

    # Anxiety-modulated arbitration weight
    eps = 1e-10
    w_mb = float(np.clip(w_mb, eps, 1 - eps))
    logit = np.log(w_mb) - np.log(1 - w_mb)
    w_eff = 1 / (1 + np.exp(-(logit + 2.0 * (0.5 - stai))))
    w_eff = float(np.clip(w_eff, 0.0, 1.0))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = -1  # for perseveration

    for t in range(n_trials):
        # Current trial reward probabilities by state/action
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Compute first-stage action values
        max_q2 = np.max(q_stage2, axis=1)               # best alien per planet
        q1_mb = transition_matrix @ max_q2              # model-based expected value per ship
        q1 = w_eff * q1_mb + (1 - w_eff) * q_stage1_mf  # arbitration

        # Perseveration bias (scaled by anxiety)
        if prev_a1 >= 0:
            kappa_eff = stickiness * (0.5 + stai)
            bias = np.zeros(2)
            bias[prev_a1] += kappa_eff
            q1 = q1 + bias

        # Stage 1 choice via softmax
        q1_centered = q1 - np.max(q1)
        p1 = np.exp(beta * q1_centered)
        p1 = p1 / (np.sum(p1) + eps)
        a1 = int(rng.choice([0, 1], p=p1))

        # State transition
        s2 = int(rng.choice([0, 1], p=transition_matrix[a1]))

        # Stage 2 choice via softmax over current state's Q
        q2_s = q_stage2[s2]
        q2_centered = q2_s - np.max(q2_s)
        p2 = np.exp(beta * q2_centered)
        p2 = p2 / (np.sum(p2) + eps)
        a2 = int(rng.choice([0, 1], p=p2))

        # Reward sampling
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning updates (match fitting code order)
        delta2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha * delta2

        td_to_stage1 = (q_stage2[s2, a2] - q_stage1_mf[a1])  # bootstrapped from updated Q2
        q_stage1_mf[a1] += alpha * td_to_stage1 + alpha * lam * delta2

        # Log trial
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
