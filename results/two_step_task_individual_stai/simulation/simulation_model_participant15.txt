def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the cognitive_model3 logic:
    - Online learning of transition probabilities with forgetting.
    - Stage-1 uses a hybrid of learned model-based (MB) values and stage-1 model-free (MF) values.
    - Anxiety-weighted safety bias penalizes high transition entropy at stage-1.
    - Stage-2 values are learned model-free.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [eta, beta, alpha_t, f_forget, omega_safe, stai]
            - eta: learning rate for Stage-2 MF and Stage-1 MF bootstrapping.
            - beta: inverse temperature for both stages.
            - alpha_t: learning rate for updating transition matrix given observed transitions.
            - f_forget: per-trial forgetting toward uniform in transition model.
            - omega_safe: strength of safety bias (scaled by stai).
            - stai: anxiety score in [0,1].
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            state0-action0, state0-action1, state1-action0, state1-action1.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    eta, beta, alpha_t, f_forget, omega_safe, stai = parameters
    stai = float(stai)
    rng = np.random.default_rng()

    # True environment transition matrix (fixed), and agent's learned model (adaptive)
    true_T = np.array([[0.6, 0.4],
                       [0.4, 0.6]], dtype=float)
    T_learn = true_T.copy()

    # Initialize value functions
    q2 = 0.5 * np.ones((2, 2), dtype=float)  # Stage-2 MF values: shape (state, action)
    q1_mf = np.zeros(2, dtype=float)         # Stage-1 MF values

    # Anxiety-dependent weights
    w_mb = 0.5 + 0.5 * (1.0 - stai)     # in [0.5, 1.0]
    safety_scale = omega_safe * stai    # scales safety bias

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Build reward probabilities for this trial
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]], dtype=float)

        # Model-based Q at stage-1 from learned transition model and current q2
        max_q2 = np.max(q2, axis=1)          # best action value in each state
        q1_mb = T_learn @ max_q2             # expected value under learned transitions

        # Safety bias: penalize high transition entropy rows (predictability preference)
        ent = np.zeros(2, dtype=float)
        for a in range(2):
            p = np.clip(T_learn[a, 0], 1e-8, 1.0 - 1e-8)
            q = 1.0 - p
            H = -(p * np.log(p) + q * np.log(q))  # nats
            ent[a] = H / np.log(2.0)              # normalize to bits in [0,1]
        bias = -safety_scale * ent

        # Net stage-1 value and softmax choice
        q1 = w_mb * q1_mb + (1.0 - w_mb) * q1_mf + bias
        z1 = beta * (q1 - np.max(q1))
        p1 = np.exp(z1)
        p1 /= p1.sum()
        a1 = rng.choice([0, 1], p=p1)

        # Environment transition using true transition matrix
        s = rng.choice([0, 1], p=true_T[a1])

        # Stage-2 softmax choice
        z2 = beta * (q2[s] - np.max(q2[s]))
        p2 = np.exp(z2)
        p2 /= p2.sum()
        a2 = rng.choice([0, 1], p=p2)

        # Reward draw
        r = int(rng.random() < reward_probs[s, a2])

        # Learning updates
        # Stage-2 MF
        pe2 = r - q2[s, a2]
        q2[s, a2] += eta * pe2

        # Stage-1 MF bootstrapping from observed stage-2 value
        target1 = q2[s, a2]
        pe1 = target1 - q1_mf[a1]
        q1_mf[a1] += eta * pe1

        # Transition model learning with forgetting toward uniform
        T_learn = (1.0 - f_forget) * T_learn + f_forget * 0.5
        # Update observed row toward realized state
        T_learn[a1, s] += alpha_t * (1.0 - T_learn[a1, s])
        other = 1 - s
        T_learn[a1, other] += alpha_t * (0.0 - T_learn[a1, other])
        # Renormalize and clip
        for a in range(2):
            row = np.clip(T_learn[a], 1e-8, 1.0)
            T_learn[a] = row / row.sum()

        # Record
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
