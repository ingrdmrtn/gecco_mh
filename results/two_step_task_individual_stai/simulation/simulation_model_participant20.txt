def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards for the two-step task using the hybrid model-based/model-free RL
    with anxiety-modulated arbitration, eligibility trace, and stickiness (as in the provided fitting code).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w0, lam, pers, stai]
            - alpha: learning rate for MF Q-updates (both stages)
            - beta: inverse temperature used at both stages
            - w0: base model-based weight
            - lam: eligibility trace parameter (stage-2 PE into stage-1 MF update)
            - pers: base perseveration (stickiness) strength for stage-1
            - stai: anxiety level in [0,1] modulating arbitration and stickiness
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for:
            - state 0: actions 0,1 -> drift1[t], drift2[t]
            - state 1: actions 0,1 -> drift3[t], drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w0, lam, pers, stai = parameters
    stai = float(stai)

    rng = np.random.default_rng()

    # Transition structure: each first-stage action predominantly leads to one second-stage state
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize value functions
    q_stage1_mf = np.zeros(2)        # model-free first-stage Q values
    q_stage2_mf = np.zeros((2, 2))   # model-free second-stage Q values per state

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_choice1 = None  # for stickiness at stage 1

    for t in range(n_trials):
        # Trial-wise reward probabilities per state-action
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]])

        # Model-based component: expected value of first-stage actions by propagating
        # max second-stage values through the transition matrix
        max_q_stage2 = np.max(q_stage2_mf, axis=1)    # best alien per state
        q_stage1_mb = transition_matrix @ max_q_stage2

        # Anxiety-modulated arbitration and stickiness
        w_eff = np.clip(w0 * (1.0 - 0.6 * stai), 0.0, 1.0)
        kappa = pers * (0.5 + stai)

        # Combine MB and MF for stage 1
        q1_combined = w_eff * q_stage1_mb + (1.0 - w_eff) * q_stage1_mf

        # Add stickiness (only to the previously chosen action)
        pref1 = beta * q1_combined
        if prev_choice1 is not None:
            stick = np.zeros(2)
            stick[prev_choice1] = 1.0
            pref1 = pref1 + kappa * stick

        # Softmax choice at stage 1
        pref1_shift = pref1 - np.max(pref1)
        p_stage1 = np.exp(pref1_shift)
        p_stage1 = p_stage1 / np.sum(p_stage1)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 choice via softmax over MF Q in reached state
        pref2 = beta * q_stage2_mf[s2]
        pref2_shift = pref2 - np.max(pref2)
        p_stage2 = np.exp(pref2_shift)
        p_stage2 = p_stage2 / np.sum(p_stage2)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Reward outcome (Bernoulli draw)
        r = int(rng.random() < reward_probs[s2, a2])

        # TD errors and updates (matching the fitting code)
        delta1 = q_stage2_mf[s2, a2] - q_stage1_mf[a1]
        delta2 = r - q_stage2_mf[s2, a2]

        # Update second-stage MF Q
        q_stage2_mf[s2, a2] += alpha * delta2

        # Update first-stage MF Q with bootstrapped term and eligibility trace
        q_stage1_mf[a1] += alpha * (delta1 + lam * delta2)

        # Log outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        # Update stickiness memory
        prev_choice1 = a1

    return stage1_choice, state2, stage2_choice, reward
