def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the hybrid MB/MF model with anxiety-modulated planning
    weight and bi-stage perseveration (matching cognitive_model1).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or array):
            [alpha, beta, omega0, anx_mod, pi] or [alpha, beta, omega0, anx_mod, pi, stai]
            - alpha (0-1): learning rate
            - beta (0-10): inverse temperature for both stages
            - omega0 (0-1): baseline model-based weight
            - anx_mod (0-1): strength of anxiety modulation on MB weight
            - pi (0-1): baseline perseveration magnitude (reduced by anxiety)
            - stai (0-1, optional): anxiety score; if omitted, defaults to 0.5
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the two
            second-stage states and their two actions each:
            reward_probs[state][action] = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    # Unpack parameters; allow optional stai in the parameter vector
    if len(parameters) == 5:
        alpha, beta, omega0, anx_mod, pi = parameters
        stai = 0.5
    else:
        alpha, beta, omega0, anx_mod, pi, stai = parameters

    rng = np.random.default_rng()

    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Value functions
    q1_mf = np.zeros(2)        # model-free values at stage 1 (two actions)
    q2 = np.zeros((2, 2))      # stage-2 Q-values: states x actions

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Perseveration bookkeeping
    prev_a1 = None
    prev_a2_by_state = [None, None]

    # Anxiety-modulated weights
    omega_eff = np.clip(omega0 + anx_mod * (0.5 - float(stai)), 0.0, 1.0)
    stick_strength = pi * (1.0 - float(stai))

    for t in range(n_trials):
        # Trial-wise reward probabilities
        reward_probs = [[drift1[t], drift2[t]], [drift3[t], drift4[t]]]

        # Compute model-based action values for stage 1
        max_q2 = np.max(q2, axis=1)             # per state, best stage-2 value
        q1_mb = transition_matrix @ max_q2      # expected value under known transitions

        # Add stage-1 perseveration bias
        bias1 = np.zeros(2)
        if prev_a1 is not None:
            bias1[prev_a1] += stick_strength

        # Hybrid value for stage 1
        q1_net = omega_eff * q1_mb + (1.0 - omega_eff) * q1_mf + bias1

        # Softmax for stage 1
        q1c = q1_net - np.max(q1_net)
        p1 = np.exp(beta * q1c)
        p1 /= p1.sum()
        a1 = rng.choice([0, 1], p=p1)

        # Transition to stage-2 state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 perseveration bias (state-dependent)
        bias2 = np.zeros(2)
        if prev_a2_by_state[s] is not None:
            bias2[prev_a2_by_state[s]] += stick_strength

        # Stage-2 choice
        q2_net = q2[s] + bias2
        q2c = q2_net - np.max(q2_net)
        p2 = np.exp(beta * q2c)
        p2 /= p2.sum()
        a2 = rng.choice([0, 1], p=p2)

        # Reward sampling
        r = int(rng.random() < reward_probs[s][a2])

        # Learning updates
        delta2 = r - q2[s, a2]
        q2[s, a2] += alpha * delta2

        delta1 = q2[s, a2] - q1_mf[a1]
        q1_mf[a1] += alpha * delta1

        # Persist choices for perseveration
        prev_a1 = a1
        prev_a2_by_state[s] = a2

        # Log outputs
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
