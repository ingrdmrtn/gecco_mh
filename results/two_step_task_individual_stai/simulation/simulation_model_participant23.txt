def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the learned-transitions, UCB-exploration,
    anxiety-modulated forgetting hybrid model (cognitive_model3).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple):
            [alpha, beta, zeta0, phi0, forget0, stai]
            - alpha in [0,1]: learning rate for Q-values at both stages.
            - beta in [0,10]: inverse temperature for both stages.
            - zeta0 in [0,1]: baseline exploration bonus scale (reduced by anxiety).
            - phi0 in [0,1]: baseline model-based weight at stage 1 (reduced by anxiety).
            - forget0 in [0,1]: baseline per-trial forgetting rate (increased by anxiety).
            - stai in [0,1]: anxiety score.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two second-stage states and two actions:
            reward_probs[state][action] at trial t equals
            [[drift1[t], drift2[t]],
             [drift3[t], drift4[t]]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, zeta0, phi0, forget0, stai = parameters
    stai = float(stai)

    # Anxiety-modulated parameters
    zeta = np.clip(zeta0 * (1.0 - stai), 0.0, 1.0)             # less exploration with anxiety
    phi = np.clip(phi0 * (1.0 - 0.5 * stai), 0.0, 1.0)         # less MB weighting with anxiety
    forget = np.clip(forget0 * (0.5 + 0.5 * stai), 0.0, 1.0)   # more forgetting with anxiety

    # Environment transition structure (fixed, "true" transitions)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    rng = np.random.default_rng()

    # Agent's internal variables
    trans_counts = np.ones((2, 2))   # Dirichlet counts over states given first-stage action
    q1_mf = np.zeros(2)              # Model-free values at stage 1
    q2_mf = np.zeros((2, 2))         # Model-free values at stage 2: [state, action]
    visit_counts = np.ones((2, 2))   # For UCB bonus at stage 2

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Trial-wise reward probabilities for current trial
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]], dtype=float)

        # Forgetting (decay toward 0) before acting
        q1_mf *= (1.0 - forget)
        q2_mf *= (1.0 - forget)

        # Learned transition probabilities from counts
        trans_probs = trans_counts / np.sum(trans_counts, axis=1, keepdims=True)

        # Model-based stage-1 values: expected best stage-2 value under learned transitions
        max_q2 = np.max(q2_mf, axis=1)         # best per state
        q1_mb = trans_probs @ max_q2

        # Hybrid stage-1 value
        q1_hybrid = phi * q1_mb + (1.0 - phi) * q1_mf

        # Stage 1 choice via softmax
        logits1 = beta * q1_hybrid
        logits1 -= np.max(logits1)
        probs1 = np.exp(logits1)
        probs1 /= np.sum(probs1)
        a1 = int(rng.choice([0, 1], p=probs1))

        # Environment transition to state 2
        s2 = int(rng.choice([0, 1], p=transition_matrix[a1]))

        # Stage 2 choice with UCB bonus
        bonus = zeta / np.sqrt(visit_counts[s2] + 1e-8)
        logits2 = beta * (q2_mf[s2] + bonus)
        logits2 -= np.max(logits2)
        probs2 = np.exp(logits2)
        probs2 /= np.sum(probs2)
        a2 = int(rng.choice([0, 1], p=probs2))

        # Reward sampling from drifting probabilities
        r = int(rng.random() < reward_probs[s2, a2])

        # Update visit counts (for UCB)
        visit_counts[s2, a2] += 1.0

        # Learning updates
        # Stage 2 MF update
        delta2 = r - q2_mf[s2, a2]
        q2_mf[s2, a2] += alpha * delta2

        # Stage 1 MF update with backup from realized stage-2 value
        backup = q2_mf[s2, a2]
        delta1 = backup - q1_mf[a1]
        q1_mf[a1] += alpha * delta1

        # Update learned transition counts
        trans_counts[a1, s2] += 1.0

        # Log trial outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
