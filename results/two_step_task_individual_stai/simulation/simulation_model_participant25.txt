def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the Hybrid MB/MF model with anxiety-modulated
    arbitration and eligibility traces, matching the provided fitting logic.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, w_mb, lam, kappa, stai]
            - alpha: learning rate for both stages (0..1)
            - beta: inverse temperature for both stages (>=0)
            - w_mb: baseline model-based weight (0..1)
            - lam: eligibility trace parameter lambda (0..1)
            - kappa: baseline perseveration weight (>=0)
            - stai: anxiety score (typically 0..1); higher reduces MB weight and increases perseveration
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two actions in each second-stage state:
              state 0: [drift1[t], drift2[t]]
              state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w_mb, lam, kappa, stai = parameters
    rng = np.random.default_rng()

    # Transition structure (common transitions = 0.7)
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Model-free Q-values
    q_stage1_mf = np.zeros(2)       # MF values for stage-1 actions
    q_stage2_mf = np.zeros((2, 2))  # MF values for stage-2 actions per state

    # Anxiety-modulated arbitration/perseveration
    w_eff = w_mb * (1.0 - float(stai))
    kappa_eff = kappa * (1.0 + float(stai))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Perseveration memory (previous choices)
    prev_a1 = -1
    prev_a2_state0 = -1
    prev_a2_state1 = -1

    for t in range(n_trials):
        # Reward probabilities for this trial by state/action
        reward_probs = [[float(drift1[t]), float(drift2[t])],
                        [float(drift3[t]), float(drift4[t])]]

        # Model-based action values at stage 1: expected max over stage-2 values via transitions
        max_q_stage2 = np.max(q_stage2_mf, axis=1)          # best value at each state
        q_stage1_mb = transition_matrix @ max_q_stage2       # MB value for each stage-1 action

        # Hybrid value
        q1_hybrid = w_eff * q_stage1_mb + (1.0 - w_eff) * q_stage1_mf

        # Perseveration bias at stage 1
        bias1 = np.zeros(2)
        if prev_a1 in (0, 1):
            bias1[prev_a1] += kappa_eff

        # Softmax at stage 1
        logits1 = beta * q1_hybrid + bias1
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 /= np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Perseveration bias at stage 2 depends on last action in this state
        bias2 = np.zeros(2)
        prev_a2 = prev_a2_state0 if s2 == 0 else prev_a2_state1
        if prev_a2 in (0, 1):
            bias2[prev_a2] += kappa_eff

        # Softmax at stage 2 using MF Q-values
        logits2 = beta * q_stage2_mf[s2] + bias2
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 /= np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Generate reward from drifting Bernoulli probs for current state/action
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning updates
        q2_old = q_stage2_mf[s2, a2]
        delta2 = r - q2_old
        q_stage2_mf[s2, a2] += alpha * delta2

        delta1 = q2_old - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * (delta1 + lam * delta2)

        # Update perseveration memory
        prev_a1 = a1
        if s2 == 0:
            prev_a2_state0 = a2
        else:
            prev_a2_state1 = a2

        # Save trial outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
