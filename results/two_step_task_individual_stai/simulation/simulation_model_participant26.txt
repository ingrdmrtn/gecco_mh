def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the transition-learning hybrid model with
    anxiety-modulated arbitration, asymmetric learning, and perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or array):
            [alpha, beta, kappa0, phi, a_neg, stai]
            - alpha: base learning rate in [0,1]
            - beta: inverse temperature for softmax (used at both stages)
            - kappa0: base learning rate for transition model updating in [0,1]
            - phi: perseveration strength in [0,1]
            - a_neg: asymmetry coefficient in [0,1]
            - stai: participant anxiety in [0,1]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            - state 0, action 0 -> drift1[t]
            - state 0, action 1 -> drift2[t]
            - state 1, action 0 -> drift3[t]
            - state 1, action 1 -> drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np
    rng = np.random.default_rng()
    eps = 1e-12

    alpha, beta, kappa0, phi, a_neg, st = parameters
    st = float(st)

    # True environment transition matrix (fixed)
    T_true = np.array([[0.7, 0.3],
                       [0.3, 0.7]], dtype=float)

    # Agent's estimated transition model (learned)
    T_est = np.array([[0.7, 0.3],
                      [0.3, 0.7]], dtype=float)

    # Action-value estimates
    q_stage1_mf = np.zeros(2)        # model-free Q at stage 1
    q_stage2 = np.zeros((2, 2))      # Q-values for each state (rows) and action (cols)

    # Arbitration weights and learning-rate modulations by anxiety
    w_mf = st
    w_mb = 1.0 - w_mf
    kappa = kappa0 * (0.25 + 0.75 * st)
    alpha_pos = np.clip(alpha * (1.0 - a_neg * st), 0.0, 1.0)
    alpha_neg = np.clip(alpha * (1.0 + a_neg * st), 0.0, 1.0)

    # Perseveration
    prev_a1 = None

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Reward probabilities for this trial by state, action
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]], dtype=float)

        # Compute model-based Q at stage 1 via planning through learned T_est
        max_q2 = np.max(q_stage2, axis=1)  # best action value on each planet
        q1_mb = T_est @ max_q2

        # Perseveration bias
        persev_bias = np.zeros(2)
        if prev_a1 is not None:
            persev_bias[prev_a1] = phi * (0.5 + 0.5 * st)

        # Combine MB and MF values at stage 1
        q1_total = w_mb * q1_mb + w_mf * q_stage1_mf + persev_bias

        # Stage 1 choice via softmax
        q1c = q1_total - np.max(q1_total)
        p_stage1 = np.exp(beta * q1c)
        p_stage1 /= (np.sum(p_stage1) + eps)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state using true environment transitions
        s = rng.choice([0, 1], p=T_true[a1])

        # Stage 2 choice via softmax
        q2_vec = q_stage2[s]
        q2c = q2_vec - np.max(q2_vec)
        p_stage2 = np.exp(beta * q2c)
        p_stage2 /= (np.sum(p_stage2) + eps)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Reward sampling from environment drifts
        r = int(rng.random() < reward_probs[s, a2])

        # Update transition model toward observed state
        target = np.array([1.0 if i == s else 0.0 for i in range(2)], dtype=float)
        T_est[a1] = (1.0 - kappa) * T_est[a1] + kappa * target
        T_est[a1] /= (np.sum(T_est[a1]) + eps)

        # Stage-2 value update with valence-asymmetric learning rate
        pe2 = r - q_stage2[s, a2]
        lr2 = alpha_pos if pe2 >= 0.0 else alpha_neg
        q_stage2[s, a2] += lr2 * pe2

        # Stage-1 MF update using same lr2 as in fitting code
        pe1 = q_stage2[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += lr2 * pe1

        # Store perseveration
        prev_a1 = a1

        # Save trial outcomes
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
