import numpy as np

def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with eligibility traces and perseveration (matching cognitive_model1 logic).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, w, lambda_, perseveration]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the 4 second-stage options:
            - state 0: actions 0->drift1[t], 1->drift2[t]
            - state 1: actions 0->drift3[t], 1->drift4[t]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """

    alpha, beta, w, lambda_, perseveration, stai = parameters

    rng = np.random.default_rng()
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Model-free Q-values
    q_stage1_mf = np.zeros(2)          # First-stage actions (A=0, U=1)
    q_stage2_mf = np.zeros((2, 2))     # Second-stage actions per state

    # Perseveration trackers
    prev_a1 = None
    prev_a2_by_state = [None, None]

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Reward probabilities for this trial
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Compute model-based values from MF second-stage
        max_q_stage2 = np.max(q_stage2_mf, axis=1)         # best action at each state
        q_stage1_mb = transition_matrix @ max_q_stage2      # expected value of first-stage actions

        # Mixture of MB and MF for first stage
        q1 = (1.0 - w) * q_stage1_mf + w * q_stage1_mb

        # Perseveration bias at stage 1
        bias1 = np.zeros(2)
        if prev_a1 is not None:
            bias1[prev_a1] += perseveration

        # Stage-1 softmax and choice
        logits1 = beta * q1 + bias1
        logits1 -= np.max(logits1)
        probs_1 = np.exp(logits1) / np.sum(np.exp(logits1))
        a1 = rng.choice([0, 1], p=probs_1)

        # Transition to second-stage state
        s = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage-2 values and perseveration bias within state
        q2 = q_stage2_mf[s].copy()
        bias2 = np.zeros(2)
        if prev_a2_by_state[s] is not None:
            bias2[prev_a2_by_state[s]] += perseveration

        # Stage-2 softmax and choice
        logits2 = beta * q2 + bias2
        logits2 -= np.max(logits2)
        probs_2 = np.exp(logits2) / np.sum(np.exp(logits2))
        a2 = rng.choice([0, 1], p=probs_2)

        # Reward outcome
        r = int(rng.random() < reward_probs[s][a2])

        # TD updates with eligibility trace
        delta1 = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        delta2 = r - q_stage2_mf[s, a2]

        q_stage2_mf[s, a2] += alpha * delta2
        q_stage1_mf[a1] += alpha * (delta1 + lambda_ * delta2)

        # Update perseveration memory
        prev_a1 = a1
        prev_a2_by_state[s] = a2

        # Log trial
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
