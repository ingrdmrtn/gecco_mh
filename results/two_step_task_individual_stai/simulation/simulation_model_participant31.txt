def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the model-free with anxiety-modulated forgetting
    and a model-based heuristic bias toward common transitions.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, decay, k_anx_decay, tr_bias, stai]
            - alpha: learning rate in [0,1]
            - beta: inverse temperature in [0,10]
            - decay: baseline forgetting rate in [0,1]
            - k_anx_decay: how much anxiety increases forgetting in [0,1]
            - tr_bias: strength of heuristic transition bias in [0,1]
            - stai: anxiety scalar in [0,1]
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            the two actions in each second-stage state:
            - state 0: [drift1[t], drift2[t]]
            - state 1: [drift3[t], drift4[t]]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, decay, k_anx_decay, tr_bias, stai = parameters
    rng = np.random.default_rng()

    # Transition matrix with common (0.7) and rare (0.3) structure
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Effective forgetting given anxiety
    decay_eff = decay + k_anx_decay * float(stai)
    decay_eff = min(1.0, max(0.0, decay_eff))
    keep = 1.0 - decay_eff

    # Initialize Q-values
    q1 = np.zeros(2)          # first-stage MF values
    q2 = np.zeros((2, 2))     # second-stage MF values: states x actions

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    for t in range(n_trials):
        # Current reward probabilities per state/action
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Heuristic bias toward the spaceship whose common destination has higher value
        v_planet = np.max(q2, axis=1)  # [v_state0, v_state1]
        dv = v_planet[0] - v_planet[1]
        bias1 = np.array([tr_bias * dv, -tr_bias * dv])

        # Stage 1 choice
        logits1 = beta * q1 + bias1
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 = p1 / (np.sum(p1) + 1e-12)
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second stage
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Stage 2 choice within state s2
        logits2 = beta * q2[s2]
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 = p2 / (np.sum(p2) + 1e-12)
        a2 = rng.choice([0, 1], p=p2)

        # Outcome
        prob_r = reward_probs[s2][a2]
        r = int(rng.random() < prob_r)

        # Apply forgetting
        q1 *= keep
        q2 *= keep

        # Learning updates (pure MF at both stages)
        delta2 = r - q2[s2, a2]
        q2[s2, a2] += alpha * delta2

        delta1 = q2[s2, a2] - q1[a1]
        q1[a1] += alpha * delta1

        # Record
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
