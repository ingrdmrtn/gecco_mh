def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the hybrid model-based/model-free 2-step model
    with anxiety-modulated arbitration, eligibility trace, and perseveration.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha, beta, w_base, lambda_e, perseveration, stai]
            - alpha in [0,1]: learning rate.
            - beta in [0,10]: inverse temperature used at both stages.
            - w_base in [0,1]: baseline model-based weight at stage 1.
            - lambda_e in [0,1]: eligibility trace for MF backprop from stage 2 to stage 1.
            - perseveration in [0,1]: perseveration strength on previous first-stage choice.
            - stai: trait anxiety scalar; higher means higher anxiety.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            second-stage actions:
            - state X (0): actions 0-> drift1[t], 1-> drift2[t]
            - state Y (1): actions 0-> drift3[t], 1-> drift4[t]

    Returns:
        stage1_choice (np.ndarray, int): First-stage choices (0 or 1).
        state2 (np.ndarray, int): Second-stage states (0 or 1).
        stage2_choice (np.ndarray, int): Second-stage choices (0 or 1).
        reward (np.ndarray, int): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, w_base, lambda_e, persev, stai = parameters
    stai = float(stai)

    rng = np.random.default_rng()

    # Fixed transition dynamics
    transition_matrix = np.array([[0.7, 0.3],
                                  [0.3, 0.7]])

    # Initialize value functions
    q_stage2 = np.zeros((2, 2))   # Q at second stage: state x action
    q_stage1_mf = np.zeros(2)     # Model-free first-stage values

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Anxiety-modulated arbitration weight and perseveration
    w = w_base * (1.0 - 0.7 * stai)
    w = min(1.0, max(0.0, w))
    persev_eff = persev * (1.0 + 0.5 * stai)

    prev_a1 = None

    for t in range(n_trials):
        # Reward probabilities for this trial
        reward_probs = [[float(drift1[t]), float(drift2[t])],
                        [float(drift3[t]), float(drift4[t])]]

        # Model-based value: expected max over stage-2 states given transitions
        mb_values = transition_matrix @ np.max(q_stage2, axis=1)  # shape (2,)

        # Perseveration bias on previous first-stage action
        bias = np.zeros(2)
        if prev_a1 is not None:
            bias[prev_a1] += persev_eff

        # Net first-stage values
        q1 = w * mb_values + (1.0 - w) * q_stage1_mf + bias

        # Softmax for stage 1
        q1_shift = q1 - np.max(q1)
        exp_q1 = np.exp(beta * q1_shift)
        p_stage1 = exp_q1 / (np.sum(exp_q1) + 1e-12)
        a1 = rng.choice([0, 1], p=p_stage1)

        # Transition to second-stage state based on chosen action
        s2 = rng.choice([0, 1], p=transition_matrix[a1])

        # Softmax for stage 2 (same beta as fitting code)
        q2 = q_stage2[s2, :].copy()
        q2_shift = q2 - np.max(q2)
        exp_q2 = np.exp(beta * q2_shift)
        p_stage2 = exp_q2 / (np.sum(exp_q2) + 1e-12)
        a2 = rng.choice([0, 1], p=p_stage2)

        # Generate reward
        r = int(rng.random() < reward_probs[s2][a2])

        # Learning updates
        # Stage-2 TD error and update
        delta2 = r - q_stage2[s2, a2]
        q_stage2[s2, a2] += alpha * delta2

        # Stage-1 MF update toward the chosen stage-2 value (SARSA-like)
        delta1 = q_stage2[s2, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * delta1

        # Eligibility trace: backpropagate stage-2 TD error to stage-1 MF
        q_stage1_mf[a1] += alpha * lambda_e * delta2

        # Bookkeeping
        prev_a1 = a1
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
