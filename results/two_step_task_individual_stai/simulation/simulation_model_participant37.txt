def simulate_model(n_trials, parameters, drift1, drift2, drift3, drift4):
    """
    Simulates choices and rewards using the transition-learning hybrid model with
    anxiety-modulated transition learning, MB weight, and stage-1 stickiness.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or tuple): [alpha, beta, alpha_T0, alpha_Tgain, kappa, stai]
            - alpha in [0,1]: reward learning rate for MF Q values (both stages).
            - beta in [0,10]: inverse temperature (both stages).
            - alpha_T0 in [0,1]: baseline transition learning rate.
            - alpha_Tgain in [0,1]: how much anxiety increases transition learning rate.
            - kappa in [0,1]: baseline perseveration (stickiness) strength at stage 1.
            - stai in [0,1]: trait anxiety score.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the two actions in each second-stage state:
            - state 0: actions (0,1) have probs (drift1[t], drift2[t])
            - state 1: actions (0,1) have probs (drift3[t], drift4[t])

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta, alpha_T0, alpha_Tgain, kappa, stai = parameters
    stai = float(stai)

    rng = np.random.default_rng()

    # True environment transition structure (fixed, used to generate states)
    true_T = np.array([[0.7, 0.3],
                       [0.3, 0.7]])

    # Agent-internal learned transition matrix (starts uncertain, learned over time)
    T = np.full((2, 2), 0.5)

    # MF Q values
    q_stage1_mf = np.zeros(2)        # stage-1 MF Q for actions (0,1)
    q_stage2_mf = np.zeros((2, 2))   # stage-2 MF Q for states (0,1) x actions (0,1)

    # Anxiety-modulated parameters
    alpha_T = alpha_T0 + alpha_Tgain * stai
    alpha_T = min(1.0, max(0.0, alpha_T))

    # Stickiness scaled by anxiety
    stick = kappa * (1.0 + 0.5 * (stai - 0.51))

    # Anxiety-determined MB weight (no extra parameter)
    weight_mb = 0.5 + 0.4 * (0.51 - stai)
    weight_mb = min(1.0, max(0.0, weight_mb))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = -1

    for t in range(n_trials):
        # Reward probabilities for current trial
        reward_probs = [[float(drift1[t]), float(drift2[t])],
                        [float(drift3[t]), float(drift4[t])]]

        # Model-based stage-1 Q from learned transitions and MF stage-2 values
        max_q_stage2 = np.max(q_stage2_mf, axis=1)  # V(state) = max_a Q(s,a)
        q_stage1_mb = T @ max_q_stage2

        # Hybrid Q at stage 1
        q1 = weight_mb * q_stage1_mb + (1.0 - weight_mb) * q_stage1_mf

        # Add stickiness bias toward previous stage-1 choice
        bias = np.zeros(2)
        if prev_a1 >= 0:
            bias[prev_a1] = stick

        # Stage 1 choice via softmax
        logits1 = beta * q1 + bias
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 = p1 / p1.sum()
        a1 = rng.choice([0, 1], p=p1)

        # Transition to second-stage state using true environment transitions
        s = rng.choice([0, 1], p=true_T[a1])

        # Stage 2 choice via softmax over MF Q(s, a)
        q2 = q_stage2_mf[s, :]
        logits2 = beta * q2
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 = p2 / p2.sum()
        a2 = rng.choice([0, 1], p=p2)

        # Reward from drifting probabilities
        r_prob = reward_probs[s][a2]
        r = int(rng.random() < r_prob)

        # Update learned transition matrix using delta rule toward observed outcome
        T[a1, s] += alpha_T * (1.0 - T[a1, s])
        T[a1, 1 - s] = 1.0 - T[a1, s]

        # MF learning at stage 2
        delta2 = r - q_stage2_mf[s, a2]
        q_stage2_mf[s, a2] += alpha * delta2

        # MF learning at stage 1 with eligibility trace lam = stai
        lam = stai
        delta1_bootstrap = q_stage2_mf[s, a2] - q_stage1_mf[a1]
        q_stage1_mf[a1] += alpha * (delta1_bootstrap + lam * delta2)

        # Log choices and outcomes
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
