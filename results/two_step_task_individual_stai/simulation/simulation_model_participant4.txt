def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a model-based planner with anxiety-inflated noise
    after rare transitions and perseveration (matches cognitive_model2 logic).

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list or array): [alpha, beta_base, eta_surp, kappa_pers, stai]
            - alpha: stage-2 learning rate (Rescorla-Wagner).
            - beta_base: baseline inverse temperature for both stages.
            - eta_surp: scales the beta reduction on rare transitions.
            - kappa_pers: additive stay bias applied to logits at each stage.
            - stai: trait anxiety in [0,1] scaling the surprise effect.
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state 0 action 0, state 0 action 1, state 1 action 0, state 1 action 1.

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha, beta_base, eta_surp, kappa_pers, stai = parameters

    rng = np.random.default_rng()
    T_known = np.array([[0.7, 0.3],
                        [0.3, 0.7]], dtype=float)

    # Stage-2 Q-values (states 0 and 1, actions 0 and 1), initialized to 0.5
    q2 = np.full((2, 2), 0.5, dtype=float)

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Perseveration trackers
    prev_a1 = None
    prev_a2 = [None, None]  # one for each state

    for t in range(n_trials):
        # Trial-wise reward probabilities for state-action pairs
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Model-based stage-1 Q from known transitions and current stage-2 values
        max_q2 = np.max(q2, axis=1)  # best action value in each second-stage state
        q1_mb = T_known @ max_q2

        # Perseveration biases (additive to logits)
        bias1 = np.zeros(2, dtype=float)
        if prev_a1 is not None:
            bias1[prev_a1] += kappa_pers

        # Stage-1 choice using baseline beta (transition not yet observed)
        logits1 = beta_base * (q1_mb - np.max(q1_mb)) + bias1
        logits1 -= np.max(logits1)  # numerical stability
        p1 = np.exp(logits1)
        p1 /= p1.sum()
        a1 = int(rng.choice([0, 1], p=p1))

        # Transition to second-stage state
        s = int(rng.choice([0, 1], p=T_known[a1]))

        # Surprise/anxiety-adjusted inverse temperature for this trial
        is_rare = (a1 == 0 and s == 1) or (a1 == 1 and s == 0)
        beta_eff = beta_base * (1.0 - eta_surp * stai * (1.0 if is_rare else 0.0))
        beta_eff = max(1e-6, float(beta_eff))

        # Stage-2 choice with perseveration bias for the reached state
        bias2 = np.zeros(2, dtype=float)
        if prev_a2[s] is not None:
            bias2[int(prev_a2[s])] += kappa_pers

        q2_s = q2[s]
        logits2 = beta_eff * (q2_s - np.max(q2_s)) + bias2
        logits2 -= np.max(logits2)
        p2 = np.exp(logits2)
        p2 /= p2.sum()
        a2 = int(rng.choice([0, 1], p=p2))

        # Generate reward
        r = int(rng.random() < reward_probs[s][a2])

        # Update stage-2 Q-values with RW rule
        pe2 = r - q2[s, a2]
        q2[s, a2] += alpha * pe2

        # Update perseveration trackers
        prev_a1 = a1
        prev_a2[s] = a2

        # Save trial outcomes
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

    return stage1_choice, state2, stage2_choice, reward
