def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using a model-based controller with learned transitions,
    stage-2 TD learning, and first-stage stickiness. Anxiety dampens transition learning.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list/tuple): [alpha2, beta, tau, kappa1, stai]
            - alpha2: learning rate for second-stage Q-values (0-1)
            - beta: inverse temperature for softmax at both stages (>=0)
            - tau: transition learning rate (0-1)
            - kappa1: first-stage stickiness (0-1)
            - stai: trait anxiety in [0,1]; reduces effective transition learning
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for the
            two aliens on each second-stage state:
            - state 0: [drift1, drift2]
            - state 1: [drift3, drift4]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    alpha2, beta, tau, kappa1, stai_val = parameters
    rng = np.random.default_rng()

    # True environment transitions (fixed; agent learns its own T)
    transition_matrix_true = np.array([[0.7, 0.3],
                                       [0.3, 0.7]])

    # Agent-internal transition beliefs (start uninformative)
    T = np.full((2, 2), 0.5)

    # Agent Q-values for stage-2 (state x action)
    Q2 = np.zeros((2, 2))

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Stickiness
    prev_a1 = None

    # Anxiety-dampened transition learning rate
    tau_eff = tau * (1.0 - float(stai_val))

    for t in range(n_trials):
        # Reward probabilities for this trial
        reward_probs = [[drift1[t], drift2[t]],
                        [drift3[t], drift4[t]]]

        # Stage 1: compute model-based Q using learned transitions
        max_Q2 = np.max(Q2, axis=1)            # best action per state
        Q1_MB = T @ max_Q2                     # expected value per first-stage action

        # Add stickiness bias to logits
        bias1 = np.zeros(2)
        if prev_a1 is not None:
            bias1[prev_a1] = kappa1

        # Softmax to choose stage-1 action
        logits1 = beta * Q1_MB + bias1
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(logits1)
        p1 = p1 / np.sum(p1)
        a1 = rng.choice([0, 1], p=p1)

        # Environment transitions to state 2 (true transitions)
        s2 = rng.choice([0, 1], p=transition_matrix_true[a1])

        # Stage 2: softmax on Q2 at the visited state
        logits2 = beta * Q2[s2]
        logits2 = logits2 - np.max(logits2)
        p2 = np.exp(logits2)
        p2 = p2 / np.sum(p2)
        a2 = rng.choice([0, 1], p=p2)

        # Reward realization (Bernoulli with provided drifting probabilities)
        r = int(rng.random() < reward_probs[s2][a2])

        # Update learned transition model with anxiety-dampened rate
        if tau_eff > 0.0:
            # Decay entire chosen row
            T[a1, :] = (1.0 - tau_eff) * T[a1, :]
            # Add mass to the observed state and renormalize
            T[a1, s2] += tau_eff
            row_sum = np.sum(T[a1, :])
            if row_sum > 0:
                T[a1, :] /= row_sum

        # Stage-2 TD update
        delta2 = r - Q2[s2, a2]
        Q2[s2, a2] += alpha2 * delta2

        # Store outcomes
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        # Update stickiness memory
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
