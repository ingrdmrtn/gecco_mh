def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the successor-like planning + model-free integration
    with anxiety-modulated arbitration (set to neutral anxiety) and memory decay.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [alpha, beta, omega, zeta, bias_prev]
            - alpha in [0,1]: learning rate for value updates (both stages).
            - beta in [0,10]: inverse temperature for both stages.
            - omega in [0,1]: base weight on transition-based planning at stage 1.
            - zeta in [0,1]: forgetting rate toward zero for unrefreshed Q-values.
            - bias_prev in [0,1]: structure-based stay/shift bias magnitude.
              (Applied after rewarded trials only in this simulation, matching fitting code.)
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities for
            state-action pairs:
              state 0: [drift1 (action 0), drift2 (action 1)]
              state 1: [drift3 (action 0), drift4 (action 1)]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np

    rng = np.random.default_rng()

    alpha, beta, omega, zeta, bias_prev, stai = parameters

    # Neutral anxiety assumption for simulation (no stai input here)
    omega_eff = np.clip(omega * (1.0 - 0.4 * stai), 0.0, 1.0)
    bias_prev_eff_base = np.clip(bias_prev * (1.0 - stai), 0.0, 1.0)

    # Transition structure (common = 0.7, rare = 0.3)
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]], dtype=float)

    # Value caches
    q1_mf = np.zeros(2)       # model-free values for stage 1 actions
    q2 = np.zeros((2, 2))     # stage 2 action values per state

    # Outputs
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    # Previous trial info for structure-based stay/shift bias
    a1_prev = None
    s_prev = None
    r_prev = None

    for t in range(n_trials):
        # Reward probabilities this trial
        reward_probs = np.array([[drift1[t], drift2[t]],
                                 [drift3[t], drift4[t]]], dtype=float)

        # Planning values from model of transitions using current q2
        max_q2_by_state = np.max(q2, axis=1)      # best attainable on each planet
        q1_plan = T @ max_q2_by_state             # expected value for each first-stage action

        # Structure-based stay/shift bias based on previous trial
        pref = np.zeros(2)
        if a1_prev is not None:
            was_common = 1 if ((a1_prev == 0 and s_prev == 0) or (a1_prev == 1 and s_prev == 1)) else 0
            # In fitting code: sign = 1 if r_prev > 0, -1 if r_prev < 0, else 0
            # Here rewards are 0/1, so bias is applied only after reward=1
            sign = 1.0 if r_prev > 0 else (0.0 if r_prev == 0 else -1.0)
            bias = bias_prev_eff_base * sign * (1.0 if was_common else -1.0)
            pref[a1_prev] += bias
            pref[1 - a1_prev] -= bias

        # Arbitration between planning and model-free with bias
        q1 = omega_eff * q1_plan + (1.0 - omega_eff) * q1_mf + pref

        # Softmax choice at stage 1
        q1_shift = q1 - np.max(q1)
        p1 = np.exp(beta * q1_shift)
        p1 /= p1.sum()
        a1 = int(rng.choice([0, 1], p=p1))

        # Transition to stage 2 state given a1
        s = int(rng.choice([0, 1], p=T[a1]))

        # Stage 2 choice softmax
        q2_s = q2[s]
        q2_shift = q2_s - np.max(q2_s)
        p2 = np.exp(beta * q2_shift)
        p2 /= p2.sum()
        a2 = int(rng.choice([0, 1], p=p2))

        # Stochastic reward from drifting probabilities
        r = int(rng.random() < reward_probs[s, a2])

        # Soft forgetting (decay) toward zero for unrefreshed Q-values
        q2 *= (1.0 - zeta)
        q1_mf *= (1.0 - zeta)

        # Update stage 2 value
        delta2 = r - q2[s, a2]
        q2[s, a2] += alpha * delta2

        # Model-free update at stage 1 toward obtained stage 2 value
        target1 = q2[s, a2]
        delta1 = target1 - q1_mf[a1]
        q1_mf[a1] += alpha * delta1

        # Save outputs
        stage1_choice[t] = a1
        state2[t] = s
        stage2_choice[t] = a2
        reward[t] = r

        # Update previous-trial variables for bias on next trial
        a1_prev, s_prev, r_prev = a1, s, r

    return stage1_choice, state2, stage2_choice, reward
