def simulate_model(
    n_trials,
    parameters,
    drift1,
    drift2,
    drift3,
    drift4):
    """
    Simulates choices and rewards using the leaky Bayesian directed-exploration model
    (cognitive_model2) with perseveration modulated by anxiety.

    Parameters:
        n_trials (int): Number of trials to simulate.
        parameters (list): [rho_decay, beta, phi0, pi0, stai]
            - rho_decay: leaky decay factor for counts and successes (0..1)
            - beta: inverse temperature for both stages
            - phi0: baseline directed exploration bonus weight (0..1)
            - pi0: baseline perseveration bias magnitude at stage 1 (0..1)
            - stai: trait anxiety scalar in [0,1]; higher increases exploration bonus
        drift1, drift2, drift3, drift4 (np.ndarray): Trial-wise reward probabilities
            for the 4 second-stage options arranged as:
                state 0: [drift1, drift2]
                state 1: [drift3, drift4]

    Returns:
        stage1_choice (np.ndarray): First-stage choices (0 or 1).
        state2 (np.ndarray): Second-stage states (0 or 1).
        stage2_choice (np.ndarray): Second-stage choices (0 or 1).
        reward (np.ndarray): Rewards (0 or 1).
    """
    import numpy as np
    rng = np.random.default_rng()

    rho_decay, beta, phi0, pi0, st = parameters

    # Known transition structure
    T = np.array([[0.7, 0.3],
                  [0.3, 0.7]], dtype=float)

    # Initialize leaky counts and successes with weak priors
    N = np.zeros((2, 2), dtype=float) + 1.0   # counts
    S = np.zeros((2, 2), dtype=float) + 0.5   # successes

    # Anxiety-modulated directed exploration and perseveration
    phi = float(phi0) * (0.5 + float(st))           # higher anxiety -> larger exploration bonus
    persev = float(pi0) * (1.0 - 0.5 * float(st))   # higher anxiety -> less perseveration

    # Storage
    stage1_choice = np.zeros(n_trials, dtype=int)
    state2 = np.zeros(n_trials, dtype=int)
    stage2_choice = np.zeros(n_trials, dtype=int)
    reward = np.zeros(n_trials, dtype=int)

    prev_a1 = None

    for t in range(n_trials):
        # Trial-wise reward probabilities per state-action
        reward_probs = [[float(drift1[t]), float(drift2[t])],
                        [float(drift3[t]), float(drift4[t])]]

        # Leaky decay
        N *= rho_decay
        S *= rho_decay

        # Posterior mean and uncertainty bonus
        m = S / (N + 1e-8)
        u = 1.0 / (N + 1.0)
        q2 = m + phi * u  # 2x2 array

        # Model-based projection to stage 1
        vmax2 = np.max(q2, axis=1)   # length-2: best second-stage option per state
        q1 = T @ vmax2               # length-2

        # Perseveration bias on stage 1
        bias = np.zeros(2, dtype=float)
        if prev_a1 is not None:
            bias[prev_a1] += persev

        # Stage 1 choice via softmax
        logits1 = q1 + bias
        logits1 = logits1 - np.max(logits1)
        p1 = np.exp(beta * logits1)
        p1 /= p1.sum()
        a1 = int(rng.choice([0, 1], p=p1))

        # Transition to second-stage state
        s2 = int(rng.choice([0, 1], p=T[a1]))

        # Stage 2 choice via softmax using state-specific q2
        logits2 = q2[s2] - np.max(q2[s2])
        p2 = np.exp(beta * logits2)
        p2 /= p2.sum()
        a2 = int(rng.choice([0, 1], p=p2))

        # Reward sampling from drifting Bernoulli
        r = int(rng.random() < reward_probs[s2][a2])

        # Update leaky counts and successes for the sampled arm
        N[s2, a2] += 1.0
        S[s2, a2] += r

        # Store
        stage1_choice[t] = a1
        state2[t] = s2
        stage2_choice[t] = a2
        reward[t] = r

        # Update perseveration memory
        prev_a1 = a1

    return stage1_choice, state2, stage2_choice, reward
